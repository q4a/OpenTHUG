104 LPDIRECT3DTEXTURE8 openPlaybackImage( uint32 width, uint32 height )
105 {
106  // Need to this prevent the single poly being culled.
107  D3DDevice_GetRenderState( D3DRS_CULLMODE, &cullMode );
108  D3DDevice_SetRenderState( D3DRS_CULLMODE, D3DCULL_NONE );
109 
110  D3DDevice_GetRenderState( D3DRS_MULTISAMPLEANTIALIAS, &multisampleAntialias );
111  D3DDevice_SetRenderState( D3DRS_MULTISAMPLEANTIALIAS, FALSE );
112 
113  D3DDevice_SetRenderState( D3DRS_LIGHTING, FALSE );
114 
115  NxXbox::set_render_state( RS_ZWRITEENABLE, 0 );
116  NxXbox::set_render_state( RS_ZTESTENABLE, 0 );
117 
118  D3DDevice_GetTextureStageState( 0, D3DTSS_MINFILTER, &minFilter );
119  D3DDevice_SetTextureStageState( 0, D3DTSS_MINFILTER, D3DTEXF_LINEAR );
120 
121  textureWidth = (float)width;
122  textureHeight = (float)height;
123 
124  // Adjust for our current screen buffer.
125  playbackWidth = (float)NxXbox::EngineGlobals.backbuffer_width;
126  playbackHeight = (float)NxXbox::EngineGlobals.backbuffer_height;
127 
128  // Create a surface for our texture with the DirectDraw handle.
129  LPDIRECT3DTEXTURE8 p_texture_surface;
130  if( SUCCEEDED( D3DDevice_CreateTexture( width, height, 1, 0, D3DFMT_LIN_X8R8G8B8, 0, &p_texture_surface )))
131  {
132  return p_texture_surface;
133  }
134 
135  return NULL;
136 }

161 // Advance a Bink file by one frame into a 3D image buffer.
162 static void decompressFrame( HBINK bink, LPDIRECT3DTEXTURE8 p_image, bool copy_all )
163 {
164  // Decompress the Bink frame.
165  BinkDoFrame( bink );
166 
167  // Lock the 3D image so that we can copy the decompressed frame into it.
168  D3DLOCKED_RECT lock_rect;
169  if( SUCCEEDED( p_image->LockRect( 0, &lock_rect, 0, 0 )))
170  {
171  void* pixels = lock_rect.pBits;
172  uint32 pixel_pitch = lock_rect.Pitch;
173 
174  // Copy the decompressed frame into the 3D image.
175  BinkCopyToBuffer( bink,
176  pixels,
177  pixel_pitch,
178  bink->Height,
179  0, // Left pixel offset.
180  0, // Top pixel offset.
181  currentPlaybackSurfaceType | (( copy_all ) ? BINKCOPYALL : 0 ));
182 
183  // Unlock the 3D image.
184  p_image->UnlockRect( 0 );
185  }
186 }

190 // Blit a 3D image onto the render target.
191 void blitImage( LPDIRECT3DTEXTURE8 p_image, float x_offset, float y_offset, float x_scale, float y_scale, float alpha_level )
192 {
193  if( p_image == NULL )
194  {
195  return;
196  }
197 
198  NxXbox::set_blend_mode( NxXbox::vBLEND_MODE_DIFFUSE );
199 
200  // Turn on clamping so that the linear textures work
201  NxXbox::set_render_state( RS_UVADDRESSMODE0, 0x00010001UL );
202 
203  // Use a default vertex and pixel shader
204  NxXbox::set_vertex_shader( D3DFVF_XYZRHW | D3DFVF_DIFFUSE | D3DFVF_TEX1 );
205  NxXbox::set_pixel_shader( PixelShader4 );
206 
207  // Select the texture.
208  NxXbox::set_texture( 0, NULL );
209  NxXbox::set_texture( 0, p_image );
210 
211  // Setup up the vertices.
212  MOVIE_VERT vertices[4];
213 
214  vertices[0].sx = x_offset;
215  vertices[0].sy = y_offset;
216  vertices[0].sz = 0.0f;
217  vertices[0].rhw = 0.0f;
218  vertices[0].color = ((int)(( alpha_level * 255.0f )) << 24 ) | 0x808080;
219  vertices[0].tu = -0.5f;
220  vertices[0].tv = -0.5f;
221 
222  vertices[1] = vertices[ 0 ];
223  vertices[1].sx = x_offset + (((float)playbackWidth ) * x_scale );
224  vertices[1].tu = ((float)textureWidth ) - 0.5f;
225 
226  vertices[2] = vertices[0];
227  vertices[2].sy = y_offset + (((float)playbackHeight ) * y_scale );
228  vertices[2].tv = ((float)textureHeight ) - 0.5f;
229 
230  vertices[3] = vertices[1];
231  vertices[3].sy = vertices[2].sy;
232  vertices[3].tv = ((float)textureHeight ) - 0.5f;
233 
234  // Draw the vertices.
235  D3DDevice_DrawVerticesUP( D3DPT_TRIANGLESTRIP, 4, vertices, sizeof( MOVIE_VERT ));
236 }

243 static void showFrame( HBINK bink, LPDIRECT3DTEXTURE8 p_image )
244 {
245  // Begin a 3D frame.
246  D3DDevice_Clear( 0, 0, D3DCLEAR_TARGET, 0, 1.0f, 0 );
247 
248  // Draw the image on the screen (centered)...
249  float x = 0.0f;
250  float y = 0.0f;
251  blitImage( p_image, x, y, playbackWidthScale, playbackHeightScale, 1.0f );
252 
253  // End a 3D frame.
254  D3DDevice_Swap( D3DSWAP_DEFAULT );
255 
256  // Keep playing the movie.
257  BinkNextFrame( bink );
258 }

267 static bool sDebounceCheck(uint16 *p_debounceFlags, uint8 *p_data, uint16 mask)
268 {
269  if (p_data==NULL)
270  {
271  return false;
272  }
273 
274  Dbg_MsgAssert(p_debounceFlags,("NULL p_debounceFlags"));
275 
276  uint16 data=(p_data[2]<<8)|p_data[3];
277  if ((data & mask)==0)
278  {
279  // Button is pressed.
280 
281  if (*p_debounceFlags & mask)
282  {
283  // OK to return true, because the button got released at some point in the past.
284  return true;
285  }
286  }
287  else
288  {
289  // Button is not pressed, so set the appropriate debounce flag so that next time it is
290  // pressed it will be detected.
291  *p_debounceFlags |= mask;
292  }
293  return false;
294 }

296 void PMovies_PlayMovie( const char* pName )
297 {
298  // Figure the volume we want to play at.
299  Spt::SingletonPtr< Sfx::CSfxManager > sfx_manager;
300  float music_vol = Pcm::GetVolume();
301  float sfx_vol = sfx_manager->GetMainVolume( );
302  float vol = music_vol > sfx_vol ? music_vol : sfx_vol;
303 
304  // Incoming movie name is in the form movies<name>, convert to d:\data\movies\bik<name>.bik.
305  char name_conversion_buffer[256] = "d:\\data\\movies\\bik\\";
306  int length = strlen( pName );
307  int backwards = length;
308  while( backwards )
309  {
310  if( pName[backwards] == '\\' )
311  {
312  ++backwards;
313  break;
314  }
315  --backwards;
316  }
317  strncpy( name_conversion_buffer + 19, pName + backwards, length - backwards );
318  length = strlen( name_conversion_buffer );
319  name_conversion_buffer[length] = '.';
320  name_conversion_buffer[length + 1] = 'b';
321  name_conversion_buffer[length + 2] = 'i';
322  name_conversion_buffer[length + 3] = 'k';
323  name_conversion_buffer[length + 4] = 0;
324 
325  // Stop music and streams.
326  Pcm::StopMusic();
327  Pcm::StopStreams();
328 
329  playbackBinkHandle = BinkOpen( name_conversion_buffer, 0 );
330  if( playbackBinkHandle == NULL )
331  {
332  // Movie not there, just quit.
333  return;
334  }
335 
336  // Switch the presentation interval to 30fps.
337  DWORD presentation_interval;
338  D3DDevice_GetRenderState( D3DRS_PRESENTATIONINTERVAL, &presentation_interval );
339  D3DDevice_SetRenderState( D3DRS_PRESENTATIONINTERVAL, D3DPRESENT_INTERVAL_ONE );
340 
341  // Open a 3D image for the Bink.
342  LPDIRECT3DTEXTURE8 p_image = openPlaybackImage( playbackBinkHandle->Width, playbackBinkHandle->Height );
343  Dbg_Assert( p_image );
344 
345  // K: Call XGetDeviceChanges once before going into the loop, otherwise the first time this function
346  // is run it will think new cards and pads have been inserted.
347 // bool running_demo_movie=Script::GetInt("RunningDemoMovie");
348  bool running_demo_movie=false;
349  if (running_demo_movie)
350  {
351  DWORD insertions=0;
352  DWORD removals=0;
353  XGetDeviceChanges(XDEVICE_TYPE_MEMORY_UNIT, &insertions, &removals);
354  XGetDeviceChanges(XDEVICE_TYPE_GAMEPAD, &insertions, &removals);
355  }
356 
357  if( p_image )
358  {
359  BinkSetSoundOnOff( playbackBinkHandle, 1 );
360 
361  // For BinkSetVolume(), 32768 is the 'nornal volume', effectively full volume on systems that don't
362  // do amplification, like Xbox.
363  BinkSetVolume( playbackBinkHandle, 0, (int)( 16384 * ( vol * 0.01f )));
364 
365  // Will need this in the loop...
366  Spt::SingletonPtr< SIO::Manager > sio_manager;
367  uint16 debounce_flags[4] = { 0,0,0,0 };
368 
369  // Start the playback loop.
370  while( true )
371  {
372  if( !BinkWait( playbackBinkHandle ))
373  {
374  decompressFrame( playbackBinkHandle, p_image, true );
375  showFrame( playbackBinkHandle, p_image );
376 
377  if( playbackBinkHandle->FrameNum >= ( playbackBinkHandle->Frames - 1 ))
378  {
379  break;
380  }
381 
382  // Check for a button being pressed to skip out of movie playback.
383  sio_manager->ProcessDevices();
384 
385  // If running a demo movie, then pad or mem card insertions must make it quit. (TRC requirement)
386  if( running_demo_movie )
387  {
388  // Use the XGetDeviceChanges function to determine if a card or pad has been inserted ...
389  DWORD insertions=0;
390  DWORD removals=0;
391  if( XGetDeviceChanges( XDEVICE_TYPE_MEMORY_UNIT, &insertions, &removals ))
392  {
393  if( insertions )
394  {
395  break;
396  }
397  }
398  if( XGetDeviceChanges( XDEVICE_TYPE_GAMEPAD, &insertions, &removals ))
399  {
400  if( insertions )
401  {
402  break;
403  }
404  }
405  }
406 
407  bool quit = false;
408  for( int d = 0; d < 4; ++d )
409  {
410  SIO::Device* p_device = sio_manager->GetDeviceByIndex( d );
411  if( p_device )
412  {
413  unsigned char* p_data = p_device->GetControlData();
414 
415  if( sDebounceCheck( &debounce_flags[d], p_data, 0x0800 ) || // Start
416  sDebounceCheck( &debounce_flags[d], p_data, 0x0040 )) // A
417  {
418  quit = true;
419  break;
420  }
421 
422  // If running a demo movie, then these buttons must also make it quit. (TRC requirement)
423  if( running_demo_movie )
424  {
425  if( sDebounceCheck(&debounce_flags[d], p_data, 0x0100) || // Back
426  sDebounceCheck(&debounce_flags[d], p_data, 0x0020) || // B
427  sDebounceCheck(&debounce_flags[d], p_data, 0x0080) || // X
428  sDebounceCheck(&debounce_flags[d], p_data, 0x0010) || // Y
429  sDebounceCheck(&debounce_flags[d], p_data, 0x0002) || // Black
430  sDebounceCheck(&debounce_flags[d], p_data, 0x0001) || // White
431  sDebounceCheck(&debounce_flags[d], p_data, 0x0200) || // Thumbstick 1
432  sDebounceCheck(&debounce_flags[d], p_data, 0x0400 )) // Thumbstick 2
433  {
434  quit = true;
435  break;
436  }
437  }
438  }
439  }
440  if( quit )
441  {
442  break;
443  }
444  }
445  }
446  }
447 
448  closePlaybackImage( p_image );
449  BinkClose( playbackBinkHandle );
450  playbackBinkHandle = NULL;
451 
452  // Switch the presentation interval bacl to what it was.
453  D3DDevice_SetRenderState( D3DRS_PRESENTATIONINTERVAL, presentation_interval );
454 }
