109 /*
110  Adds the specified CObject to the manager, registers it with the CTracker singleton.
111 
112  If the CObject has no ID, it is automatically assigned one at this point.
113 */
114 void CBaseManager::RegisterObject ( CObject& obj )
115 {
116  CTracker* p_tracker = CTracker::Instance();
117  Dbg_MsgAssert(!obj.mp_manager, ("object already registered with manager"));
118 
119  obj.mp_manager = this;
120  m_object_list.AddNode ( &obj.m_node );
121 
122  // set all list changed bits
123  m_list_changed = 0xFFFFFFFF;
124 
125  if (obj.GetID() == vNO_OBJECT_ID)
126  obj.SetID(p_tracker->GetFreshId());
127 
128  p_tracker->addObject(&obj);
129 }

134 /*
135  Removes the specified CObject to the manager, unregisters it from the CTracker singleton.
136 
137  After this, the CObject is no longer "in the system", and probably should be deleted. Its
138  ID will be available for usage by other CObjects again.
139 */
140 void CBaseManager::UnregisterObject( CObject& obj )
141 {
142  CTracker* p_tracker = CTracker::Instance();
143  Dbg_MsgAssert(obj.mp_manager == this, ("object not registered with this manager"));
144  Dbg_AssertType( &obj, CObject );
145 
146  p_tracker->removeObject(&obj, m_new_id_of_object_being_momentarily_removed, m_momentary_removal);
147 
148  if ( obj.m_node.InList() )
149  {
150  obj.m_node.Remove();
151  obj.mp_manager = NULL;
152 
153  // set all list changed bits
154  m_list_changed = 0xFFFFFFFF;
155  }
156 }

161 /*
162  Called by CObject::SetID() when a CObject's ID is changed. Shouldn't be called
163  from any other place.
164 */
165 void CBaseManager::ReregisterObject(CObject& obj, uint32 newId)
166 {
167  /*
168  In the CTracker singleton, we are concerned with keeping:
169  -Aliases. They don't use object ID, just pointers
170  -Event listeners: don't use ID, so no worries
171  -Waiting script entries: scan through and change stored ID
172  */
173 
174  m_new_id_of_object_being_momentarily_removed = newId;
175  m_momentary_removal = true;
176  UnregisterObject(obj);
177  obj.m_id = newId;
178  RegisterObject(obj);
179  m_momentary_removal = false;
180 
181 }

197 /*
198  Deletes CObjects that have been killed by CObject::mark_as_dead() (which is called
199  by the "Die" member command). These CObjects will be "out of play" by this point, and
200  only need be deleted.
201 */
202 void CGeneralManager::FlushDeadObjects()
203 {
204  Lst::Head< CObject >* head = &m_kill_list;
205  Lst::Node< CObject >* next = head->GetNext();
206 
207  while ( next )
208  {
209  Lst::Node< CObject >* kill = next;
210  next = next->GetNext();
211  kill->Remove();
212  CObject *pObject = kill->GetData( );
213  #ifdef __NOPT_ASSERT__
214  // Lock Assert is not needed for objects that have been flagged for deletion
215  pObject->SetLockAssertOff();
216  #endif
217  delete pObject;
218  }
219 }

249 /*
250  Runs the specified function on all CObjects of the given type. Deleting is allowed.
251 */
252 void CGeneralManager::ProcessAllObjectsOfType( sint type, Callback* process, void* data )
253 {
254  Lst::Search< CObject > sh;
255 
256  uint32 stamp_mask = m_stamp_bit_manager.RequestBit();
257 
258  // clear the appropriate stamp bit
259  for (CObject* pObject = sh.FirstItem(m_object_list); pObject; pObject = sh.NextItem())
260  {
261  pObject->ClearStampBit(stamp_mask);
262  }
263 
264  // traverse the object list
265  do
266  {
267  // clear the appropriate list changed bit
268  m_list_changed &= ~stamp_mask;
269 
270  CObject* pNextObject = sh.FirstItem(m_object_list);
271 
272  while (pNextObject)
273  {
274  CObject* pObject = pNextObject;
275  pNextObject = sh.NextItem();
276 
277  if (pObject->GetType() == type && !pObject->CheckStampBit(stamp_mask))
278  {
279  pObject->SetStampBit(stamp_mask);
280 
281  process(pObject, data);
282 
283  if (m_list_changed & stamp_mask) break;
284  }
285  }
286  }
287  while (m_list_changed & stamp_mask);
288 
289  m_stamp_bit_manager.ReturnBit(stamp_mask);
290 }
