383 {
384  if ( isHiRes )
385  {
386  (*pQuat)[X] = ((CHiResAnimQKey*)p_in)->qx;
387  (*pQuat)[Y] = ((CHiResAnimQKey*)p_in)->qy;
388  (*pQuat)[Z] = ((CHiResAnimQKey*)p_in)->qz;
389 // (*pQuat)[W] = ((CHiResAnimQKey*)p_in)->qw;
390  }
391  else
392  {
393  (*pQuat)[X] = quatDown( ((CStandardAnimQKey*)p_in)->qx );
394  (*pQuat)[Y] = quatDown( ((CStandardAnimQKey*)p_in)->qy );
395  (*pQuat)[Z] = quatDown( ((CStandardAnimQKey*)p_in)->qz );
396 // (*pQuat)[W] = quatDown( ((CStandardAnimQKey*)p_in)->qw );
397  }
398 
399  float qx = (*pQuat)[X];
400  float qy = (*pQuat)[Y];
401  float qz = (*pQuat)[Z];
402 
403  // Dave note: added 09/12/02 - a simple check to ensure we don't try to take the square root of a negative
404  // number, which will hose Nan-sensitive platforms later on...
405  float sum = 1.0f - qx * qx - qy * qy - qz * qz;
406  (*pQuat)[W] = sqrtf(( sum >= 0.0f ) ? sum : 0.0f );
407 
408  if ( p_in->signBit )
409  {
410  (*pQuat)[W] = -(*pQuat)[W];
411  }
412 }

468 inline void get_rotation_from_standard_key( CStandardAnimQKey* p_in, Mth::Quat* pQuat )
469 {
470  if ( p_in->qx == 0 && p_in->qy == 0 && p_in->qz == 0 )
471  {
472  // Optimization: this handles the identity rotation, which is so common that it's not
473  // worth doing a square root...
474  pQuat->SetVector( 0.0f, 0.0f, 0.0f );
475  pQuat->SetScalar( 1.0f );
476  return;
477  }
478 
479  float qx = quatDown( p_in->qx );
480  float qy = quatDown( p_in->qy );
481  float qz = quatDown( p_in->qz );
482 
483  // Dave note: added 09/12/02 - a simple check to ensure we don't try
484  // to take the square root of a negative number, which will hose
485  // Nan-sensitive platforms later on...
486  float sum = 1.0f - qx * qx - qy * qy - qz * qz;
487 
488 // This assert was firing off in one of the cutscenes...
489 // probably worth looking into later...
490 // Dbg_Assert( sum >= 0.0f );
491  if ( sum < 0.0f )
492  {
493  sum = 0.0f;
494  }
495 
496  (*pQuat)[X] = qx;
497  (*pQuat)[Y] = qy;
498  (*pQuat)[Z] = qz;
499  (*pQuat)[W] = ( p_in->signBit ) ? -sqrtf( sum ) : sqrtf( sum );
500 }
