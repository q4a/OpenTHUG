118 void sMaterial::figure_wibble_uv( void )
119 {
120  if( m_uv_wibble )
121  {
122  // Get the time.
123  float t = (float)Tmr::GetTime() * 0.001f;
124 
125  for( uint32 p = 0; p < m_passes; ++p )
126  {
127  // Figure out UV offsets for wibbling if required.
128  if( m_flags[p] & MATFLAG_UV_WIBBLE )
129  {
130  if( !( m_flags[p] & MATFLAG_EXPLICIT_UV_WIBBLE ))
131  {
132  float uoff, voff;
133 
134  uoff = ( t * mp_UVWibbleParams[p]->m_UVel ) + ( mp_UVWibbleParams[p]->m_UAmplitude * sinf( mp_UVWibbleParams[p]->m_UFrequency * t + mp_UVWibbleParams[p]->m_UPhase ));
135  voff = ( t * mp_UVWibbleParams[p]->m_VVel ) + ( mp_UVWibbleParams[p]->m_VAmplitude * sinf( mp_UVWibbleParams[p]->m_VFrequency * t + mp_UVWibbleParams[p]->m_VPhase ));
136 
137  // Reduce offset mod 16 and put it in the range -8 to +8.
138  uoff += 8.0f;
139  uoff -= (float)(( Ftoi_ASM( uoff ) >> 4 ) << 4 );
140  voff += 8.0f;
141  voff -= (float)(( Ftoi_ASM( voff ) >> 4 ) << 4 );
142 
143  mp_UVWibbleParams[p]->m_UVMatrix[2] = ( uoff < 0.0f ) ? ( uoff + 8.0f ) : ( uoff - 8.0f );
144  mp_UVWibbleParams[p]->m_UVMatrix[3] = ( voff < 0.0f ) ? ( voff + 8.0f ) : ( voff - 8.0f );
145  }
146  }
147  }
148  }

157 void sMaterial::figure_wibble_vc( void )
158 {
159  // The vertex color wibble flag is placed in pass 0.
160  if( m_flags[0] & MATFLAG_VC_WIBBLE )
161  {
162  for( uint32 i = 0; i < m_num_wibble_vc_anims; ++i )
163  {
164  struct sVCWibbleParams *p_sequence = mp_wibble_vc_params + i;
165 
166  // Get phase-shift.
167  int phase_shift = p_sequence->m_phase;
168 
169  // Time parameters.
170  int num_keys = p_sequence->m_num_keyframes;
171  int start_time = p_sequence->mp_keyframes[0].m_time;
172  int end_time = p_sequence->mp_keyframes[num_keys - 1].m_time;
173  int period = end_time - start_time;
174  int time = start_time + ( NxXbox::EngineGlobals.render_start_time + phase_shift ) % period;
175 
176  // Locate the keyframe.
177  int key;
178  for( key = num_keys - 1; key >= 0; --key )
179  {
180  if( time >= p_sequence->mp_keyframes[key].m_time )
181  {
182  break;
183  }
184  }
185 
186  Dbg_Assert( key < ( num_keys - 1 ));
187 
188  // Parameter expressing how far we are between between this keyframe and the next.
189  float t = (float)( time - p_sequence->mp_keyframes[key].m_time ) * ReciprocalEstimateNR_ASM((float)( p_sequence->mp_keyframes[key + 1].m_time - p_sequence->mp_keyframes[key].m_time ));
190 
191  // Interpolate the color.
192  uint32 red = (uint32)Ftoi_ASM((( 1.0f - t ) * p_sequence->mp_keyframes[key].m_color.r ) + ( t * p_sequence->mp_keyframes[key + 1].m_color.r ));
193  uint32 grn = (uint32)Ftoi_ASM((( 1.0f - t ) * p_sequence->mp_keyframes[key].m_color.g ) + ( t * p_sequence->mp_keyframes[key + 1].m_color.g ));
194  uint32 blu = (uint32)Ftoi_ASM((( 1.0f - t ) * p_sequence->mp_keyframes[key].m_color.b ) + ( t * p_sequence->mp_keyframes[key + 1].m_color.b ));
195  uint32 alp = (uint32)Ftoi_ASM((( 1.0f - t ) * p_sequence->mp_keyframes[key].m_color.a ) + ( t * p_sequence->mp_keyframes[key + 1].m_color.a ));
196 
197  // Switch red and blue, and store.
198  mp_wibble_vc_colors[i] = ( alp << 24 ) | ( red << 16 ) | ( grn << 8 ) | blu;
199  }
200  }
201 }

209 void sMaterial::figure_wibble_texture( void )
210 {
211  // The vertex color wibble flag is placed in pass 0.
212  if( m_texture_wibble )
213  {
214  int current_time = (int)Tmr::GetTime();
215 
216  struct sTextureWibbleParams *p_sequence = mp_wibble_texture_params;
217 
218  for( int pass = 0; pass < MAX_PASSES; ++pass )
219  {
220  if( m_flags[pass] & MATFLAG_PASS_TEXTURE_ANIMATES )
221  {
222  // Get phase-shift.
223  int phase_shift = p_sequence->m_phase[pass];
224 
225  // Time parameters.
226  int num_keys = p_sequence->m_num_keyframes[pass];
227  int start_time = p_sequence->mp_keyframes[pass][0].m_time;
228  int end_time = p_sequence->mp_keyframes[pass][num_keys - 1].m_time;
229  int period = end_time - start_time;
230  int time = start_time + (( current_time + phase_shift ) % period );
231 
232  // Keep track of the iterations, if iterations on this animation are required.
233  if( p_sequence->m_num_iterations[pass] > 0 )
234  {
235  int iteration = ( current_time - start_time ) / period;
236  if( iteration >= p_sequence->m_num_iterations[pass] )
237  {
238  // Set the time such that the animation no longer continues.
239  time = start_time + ((( period * p_sequence->m_num_iterations[pass] ) + phase_shift ) % period );
240  }
241  }
242 
243  // Locate the keyframe.
244  int key;
245  for( key = num_keys - 1; key >= 0; --key )
246  {
247  if( time >= p_sequence->mp_keyframes[pass][key].m_time )
248  {
249  break;
250  }
251  }
252 
253  // Set the texture.
254 // Dbg_Assert( p_sequence->mp_keyframes[pass][key].mp_texture );
255  mp_tex[pass] = p_sequence->mp_keyframes[pass][key].mp_texture;
256  }
257  }
258  }

554 Lst::HashTable< sMaterial > *LoadMaterialsFromMemory( void **pp_mem, Lst::HashTable< Nx::CTexture > *p_texture_table )
555 {
556  uint8 *p_data = (uint8*)( *pp_mem );
557  uint32 MMAG, MMIN, K, L, NumSeqs, seq, NumKeys;
558 
559  // Get number of materials.
560  uint32 new_materials;
561  MemoryRead( &new_materials, sizeof( uint32 ), 1, p_data );
562 
563  Lst::HashTable< sMaterial >* pMaterialTable;
564 
565  // Create table, dynamically sizing it based on the number of new materials.
566  uint32 optimal_table_size = new_materials * 2;
567  uint32 test = 4;
568  uint32 size = 2;
569 
570  for( ;; test <<= 1, ++size )
571  {
572  // Check if this iteration of table size is sufficient, or if we have hit the maximum size.
573  if(( optimal_table_size < test ) || ( size >= 12 ))
574  {
575  pMaterialTable = new Lst::HashTable< sMaterial >( size );
576  break;
577  }
578  }
579 
580  // Loop over materials.
581  for( uint32 i = 0; i < new_materials; ++i )
582  {
583  // Create new material.
584  sMaterial *pMat = new sMaterial;
585 
586  // Get material checksum.

823  {
824  pMat->m_flags[0] |= MATFLAG_NO_MAT_COL_MOD;
825  }
826

828  if( pMat->m_specular_color[3] > 0.0f )
829  {
830  pMat->m_flags[0] |= MATFLAG_SPECULAR;
831  }

836  // that one.
837  if( pMat->m_checksum == 0 )

842  }
843  }
844  else

849  }
850  else
851  {
852  pMaterialTable->PutItem( pMat->m_checksum, pMat );
853  }
854  }
855  }
856 
857  // Set the data pointer to the new position on return.
858  *pp_mem = p_data;
859 
860  return pMaterialTable;
861 }
