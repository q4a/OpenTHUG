357 bool CAsyncFileHandle::open(const char *filename, bool blocking, int priority)
358 {
359  Dbg_MsgAssert(!IsBusy(), ("Can't open file %s: asynchronous file handle already busy.", filename));
360  Dbg_MsgAssert(!mp_pre_file, ("Can't open file %s: already open as a PRE file.", filename));
361 
362  m_priority = priority;
363  m_blocking = blocking;
364 
365  m_current_function = FUNC_OPEN;
366 
367  if (PreMgr::sPreEnabled())
368  {
369  mp_pre_file = PreMgr::pre_fopen(filename, "rb", true);
370  if (mp_pre_file)
371  {
372  Dbg_Assert(PreMgr::sPreExecuteSuccess());
373 
374  m_blocking = true; // Since PRE files are in memory, they will finish "instantly"
375  m_last_result = true;
376 
377  inc_busy_count();
378  io_callback(m_current_function, m_last_result, 0); // Must call this when we are done
379  return m_last_result;
380  }
381  }
382 
383  return plat_open(filename);
384 }

617 /* */
618 /* */
619 /******************************************************************/
620 void CAsyncFileLoader::sWaitForIOEvent(bool all_io_events)
621 {
622  bool done = false;
623 
624  while (!done)
625  {
626  printf("CAsyncFileLoader waiting for io completion: busy count %d completion %d\n", s_manager_busy_count, s_new_io_completion);
627 
628  // Wait for an event
629  while (!s_new_io_completion)
630  ;
631 
632  printf("CAsyncFileLoader got completion: busy count %d completion %d\n", s_manager_busy_count, s_new_io_completion);
633 
634  // execute callbacks
635  s_execute_callback_list();
636 
637  done = true; // assume we are done for now
638  if (all_io_events)
639  {
640  for (int i = 0; i < MAX_FILE_HANDLES; i++)
641  {
642  if (s_file_handles[i]->IsBusy())
643  {
644  printf("CAsyncFileLoader still needs to wait for file handle %d\n", i);
645  Dbg_MsgAssert(s_manager_busy_count, ("CAsyncFileLoader busy count is 0 while file handle %d is busy", i));
646  done = false;
647  break;
648  }
649  }
650  }
651  }
652 }

682 CAsyncFileHandle * CAsyncFileLoader::s_get_file_handle()
683 {
684  if (s_free_handle_index < MAX_FILE_HANDLES)
685  {
686  // Find a non-busy handle and exchange with busy, if necessary
687  if (s_file_handles[s_free_handle_index]->IsBusy())
688  {
689  int i;
690  for (i = s_free_handle_index + 1; i < MAX_FILE_HANDLES; i++)
691  {
692  if (!s_file_handles[i]->IsBusy())
693  {
694  CAsyncFileHandle *p_temp = s_file_handles[i];
695  s_file_handles[i] = s_file_handles[s_free_handle_index];
696  s_file_handles[s_free_handle_index] = p_temp;
697  break;
698  }
699  }
700 
701  // If we are full, wait for this one
702  if (i == MAX_FILE_HANDLES)
703  {
704  Dbg_Message("CAsyncFileLoader::sOpen(): waiting for old handle to finish up");
705  s_file_handles[s_free_handle_index]->WaitForIO();
706  Dbg_Message("CAsyncFileLoader::sOpen(): done waiting.");
707  }
708  }
709 
710  s_file_handles[s_free_handle_index]->init(); // Clear out any old stuff first
711 
712  return s_file_handles[s_free_handle_index++];
713  }
714  else
715  {
716  return NULL;
717  }
718 }

724 
725 bool CAsyncFileLoader::s_free_file_handle(CAsyncFileHandle *p_file_handle)
726 {
727  Dbg_Assert(s_free_handle_index);
728 
729  for (int i = 0; i < MAX_FILE_HANDLES; i++)
730  {
731  if (p_file_handle == s_file_handles[i])
732  {
733  // Found it, exchange it with last open handle
734  s_file_handles[i] = s_file_handles[--s_free_handle_index];
735  s_file_handles[s_free_handle_index] = p_file_handle;
736  return true;
737  }
738  }
739 
740  return false;
741 }

770 /******************************************************************/
771 /* */
772 /* */
773 /******************************************************************/
774 
775 void CAsyncFileLoader::s_execute_callback_list()
776 {
777  s_plat_swap_callback_list();
778 
779  int list_index = s_cur_callback_list_index ^ 1; // Look at old list
780  int num_callbacks = s_num_callbacks[list_index];
781 
782  for (int i = 0; i < num_callbacks; i++)
783  {
784  volatile SCallback * p_callback_entry = &s_callback_list[list_index][i];
785 
786  CAsyncFileHandle *p_file = p_callback_entry->mp_file_handle;
787  Dbg_Assert(p_file->mp_callback);
788 
789  //Dbg_Message("Executing callback for handle %x", p_file);
790 
791  // Call the callback and the post callback
792  (*p_file->mp_callback)(p_file, p_callback_entry->m_function, p_callback_entry->m_result,
793  p_file->m_callback_arg0, p_file->m_callback_arg1);
794  p_file->post_io_callback();
795  }
796 
797  s_num_callbacks[list_index] = 0;
798 }
