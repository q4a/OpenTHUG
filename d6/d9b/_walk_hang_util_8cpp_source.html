383 bool CWalkComponent::FilterHangRail ( const Mth::Vector& pos_a, const Mth::Vector& pos_b, const Mth::Vector& preliminary_hang_point, const Mth::Vector& grab_from_point, float along_rail_factor, SHangRailData& rail_data, bool drop_to_hang )
384 {
385  // check this rail for potential hang use
386 
387  float hang_vert_origin_offset = s_get_param(CRCD(0xa8c13e74, "hang_vert_origin_offset"));
388 
389  // never snap upward to a hang rail
390  if (m_state == WALKING_AIR && preliminary_hang_point[Y] > grab_from_point[Y]) return false;
391 
392  // never snap too far vertically to a hang rail
393  if (m_state == WALKING_AIR && Mth::Abs(preliminary_hang_point[Y] - grab_from_point[Y]) > 12.0f) return false;
394 
395  // is the rail too steep
396  Mth::Vector rail = pos_b - pos_a;
397  Mth::Vector rail_direction = rail;
398  rail_direction.Normalize();
399  if (Mth::Abs(rail_direction[Y]) > sinf(Mth::DegToRad(s_get_param(CRCD(0x98724ebc, "hang_max_rail_ascent"))))) return false;
400 
401  // calculate the facing we'd have while hanging
402 
403  rail_data.hang_facing.Set(-(rail[Z]), 0.0f, rail[X]);
404  rail_data.hang_facing.Normalize();
405  if (!drop_to_hang)
406  {
407  // normally, we attempt retain our current facing
408  if (Mth::DotProduct(rail_data.hang_facing, m_facing) < 0.0f)
409  {
410  rail_data.hang_facing.Negate();
411  }
412  }
413  else
414  {
415  // unless dropping to a hang
416  if (Mth::DotProduct(rail_data.hang_facing, m_facing) > 0.0f)
417  {
418  rail_data.hang_facing.Negate();
419  }
420  }
421 
422  bool rail_points_left = rail_data.hang_facing[Z] * rail_direction[X] - rail_data.hang_facing[X] * rail_direction[Z] > 0.0f;
423  if ((rail_data.p_rail_node->GetFlag(NO_HANG_WITH_LEFT_ALONG_RAIL) && rail_points_left)
424  || (rail_data.p_rail_node->GetFlag(NO_HANG_WITH_RIGHT_ALONG_RAIL) && !rail_points_left))
425  {
426  rail_data.hang_facing.Negate();
427  rail_points_left = !rail_points_left;
428  }
429 
430  // investigate the hang point to see if its a good hang
431 
432  CFeeler feeler;
433  Nx::CCollCache* p_coll_cache = Nx::CCollCacheManager::sCreateCollCache();
434  Mth::CBBox bbox(
435  preliminary_hang_point - Mth::Vector(24.0f, 24.0f, 24.0f),
436  preliminary_hang_point + Mth::Vector(24.0f, 12.0f + hang_vert_origin_offset, 24.0f)
437  );

440

442 
443  if (!investigate_hang_rail_vicinity(feeler, preliminary_hang_point, rail_data))
444  {
445  // if we're attempting to grab a rail from the air, try both possible facings
446  if (m_state == WALKING_AIR || !drop_to_hang)
447  {
448  // see if we're allowed to try the other facing
449  if ((rail_data.p_rail_node->GetFlag(NO_HANG_WITH_LEFT_ALONG_RAIL) && !rail_points_left)
450  || (rail_data.p_rail_node->GetFlag(NO_HANG_WITH_RIGHT_ALONG_RAIL) && rail_points_left))
451  {
452  Nx::CCollCacheManager::sDestroyCollCache(p_coll_cache);
453  return false;
454  }
455

457  if (!investigate_hang_rail_vicinity(feeler, preliminary_hang_point, rail_data))
458  {
459  Nx::CCollCacheManager::sDestroyCollCache(p_coll_cache);
460  return false;
461  }
462  }

466  return false;
467  }
468  }
469

471  rail_data.horizontal_hang_offset = Mth::DotProduct(rail_data.hang_facing, rail_data.hang_point - preliminary_hang_point);
472

475  feeler.m_start[Y] += s_get_param(CRCD(0xdbb3d53e, "hang_init_anim_feeler_height"));

477  feeler.m_end += s_get_param(CRCD(0x2f83ae83, "hang_init_anim_feeler_length")) * rail_data.hang_facing;

496  rail_data.initial_animation = CRCD(0x1ee948a5, "SWING");

835 void CWalkComponent::hang_movement ( float movement )
836 {
837  // cache current critical point location and entire state
838  Mth::Vector initial_critical_point = m_pos + m_critical_point_offset;
839  Mth::Vector initial_pos = m_pos;
840  float initial_along_rail_factor = m_along_rail_factor;
841  const CRailNode* p_initial_rail_start = mp_rail_start;
842  const CRailNode* p_initial_rail_end = mp_rail_end;
843  Mth::Vector initial_critical_point_offset = m_critical_point_offset;
844  Mth::Vector initial_facing = m_facing;
845 
846  // allow left/right movement along rail
847 
848  Mth::Vector perp_facing(-m_facing[Z], 0.0f, m_facing[X]);
849 
850  // the offset from rail start to rail end
851  Mth::Vector rail = mp_rail_manager->GetPos(mp_rail_start) - mp_rail_manager->GetPos(mp_rail_end);
852 
853  // move left or right
854  bool left = movement > 0.0f;
855 
856  // is rail fractional position parameter ascends or descends with the movement
857  bool with_rail;
858  if (Mth::DotProduct(perp_facing, rail) < 0.0f)
859  {
860  with_rail = left;
861  }
862  else
863  {
864  with_rail = !left;
865  }
866

868  if (check_environment_for_hang_movement_collisions(m_facing, m_along_rail_factor, m_pos, left, perp_facing, rail, mp_rail_start, mp_rail_end))
869  {
870  m_hang_move_vel = 0.0f;
871  m_frame_event = CRCD(0x4194ecca, "Hang");
872  return;
873  }
874 
875  // the movement in terms of the fraction of the rail
876  float delta_factor = Mth::Abs(movement) / rail.Length();
877 
878  // move along the rail
879  m_along_rail_factor += with_rail ? delta_factor : -delta_factor;
880 
881  // if we're moving off the rail
882  if (m_along_rail_factor < 0.0f || m_along_rail_factor > 1.0f)
883  {
884  bool found_rail = find_next_rail(m_along_rail_factor, mp_rail_start, mp_rail_end, mp_rail_start, mp_rail_end);
885 
886  // there's no connecting rail
887  if (!found_rail)
888  {
889  // clamp the movement
890  m_hang_move_vel = 0.0f;
891  m_along_rail_factor = Mth::Clamp(m_along_rail_factor, 0.0f, 1.0f);
892  m_frame_event = CRCD(0x4194ecca, "Hang");
893  }
894  else
895  {
896  // move onto the neighboring rail
897  float remaining_movement = Mth::Abs(m_along_rail_factor - Mth::Clamp(m_along_rail_factor, 0.0f, 1.0f)) * rail.Length();
898  move_onto_neighboring_hang_rail(remaining_movement, m_along_rail_factor < 0.0f, with_rail ? rail : -rail);
899  m_frame_event = left ? CRCD(0x2d9815c3, "HangMoveLeft") : CRCD(0x279b1f0b, "HangMoveRight");
900  }
901  }
902  else
903  {
904  m_frame_event = left ? CRCD(0x2d9815c3, "HangMoveLeft") : CRCD(0x279b1f0b, "HangMoveRight");
905  }
906 
907  // check to see if the critical point has gone through geo
908 
909  Mth::Vector critical_point = m_pos + m_critical_point_offset;
910  CFeeler feeler;
911  feeler.m_start = initial_critical_point;
912  feeler.m_end = critical_point;
913  if (feeler.GetCollision())

915  // restore entire to state to initial state

919  mp_rail_start = p_initial_rail_start;
920  mp_rail_end = p_initial_rail_end;
