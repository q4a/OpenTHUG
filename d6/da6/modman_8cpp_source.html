114 void Manager::process_modules( const Tsk::Task< Manager >& task )
115 {
116 
117 
118  Module* next;
119  Module* current;
120  Lst::Search< Module > sh;
121 
122  Manager& manager = task.GetData();
123 
124  if ( !manager.control_change )
125  {
126  return; // nothing to do
127  }
128 
129  manager.control_change = false;
130 
131  next = sh.FirstItem( manager.module_list );
132 
133  while ( next )
134  {
135  current = next;
136  next = sh.NextItem(); // get next item before excuting callback
137 
138  switch ( current->state )
139  {
140  case Module::vSTOPPED :
141  {
142  if (( current->command == Module::vSTART ) ||
143  ( current->command == Module::vRESTART ))
144  {
145 #ifndef __PLAT_XBOX__
146 #ifndef __PLAT_NGC__
147 #ifdef __NOPT_DEBUG__
148  Dbg_Notify ( "Starting module %s @ %d", current->GetName(),
149  Tmr::GetTime() ); // should use Game Clock not system clock
150 #endif
151 #endif
152 #endif
153  current->v_start_cb();
154  current->state = Module::vRUNNING;
155  current->command = Module::vNONE;
156  }
157 
158  break;
159  }
160 
161  case Module::vRUNNING :
162  {
163  if ( current->command == Module::vSTOP )
164  {
165 #ifndef __PLAT_XBOX__
166 #ifndef __PLAT_NGC__
167 #ifdef __NOPT_DEBUG__
168  Dbg_Notify ( "Stopping module %s @ %d", current->GetName(),
169  Tmr::GetTime() ); // should use Game Clock not system clock
170 #endif
171 #endif
172 #endif
173  current->v_stop_cb();
174  current->state = Module::vSTOPPED;
175  current->command = Module::vNONE;
176  }
177  else if ( current->command == Module::vRESTART )
178  {
179 #ifndef __PLAT_XBOX__
180 #ifndef __PLAT_NGC__
181 #ifdef __NOPT_DEBUG__
182  Dbg_Notify ( "Restarting module %s @ %d", current->GetName(),
183  Tmr::GetTime() ); // should use Game Clock not system clock
184 #endif
185 #endif
186 #endif
187  current->v_stop_cb();
188  current->v_start_cb();
189  current->command = Module::vNONE;
190  }
191 
192  break;
193  }
194 
195  default:
196  {
197  Dbg_MsgAssert( false,( "Invalid module control state" ));
198  }
199  }
200  }
201 }
