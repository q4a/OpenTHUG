46 float CXboxViewport::plat_transform_to_screen_coord( const Mth::Vector & world_pos, float & screen_pos_x, float & screen_pos_y, ZBufferValue & screen_pos_z )
47 {
48  D3DXVECTOR3 in( world_pos[X], world_pos[Y], world_pos[Z] );
49  D3DXVECTOR4 mid;
50  D3DXVECTOR3 out;
51  D3DXVec3Transform( &mid, &in, (D3DXMATRIX*)&NxXbox::EngineGlobals.view_matrix );
52 
53  in.x = mid.x;
54  in.y = mid.y;
55  in.z = mid.z;
56  D3DXVec3TransformCoord( &out, &in, (D3DXMATRIX*)&NxXbox::EngineGlobals.projection_matrix );
57 
58  // Convert from homogenous cube (-1.0, 1.0) to pixel coordinates. Have to be careful here, the initial
59  // feeling is to use the backbuffer width and height as multipliers, but since this value is being used
60  // for text which has it's position re-converted at draw time, we actually want to use the fixed Ps2 screen
61  // values here.
62  screen_pos_x = (( out.x + 1.0f ) * 0.5f ) * 640.0f;
63  screen_pos_y = (( -out.y + 1.0f ) * 0.5f ) * 448.0f;
64 
65  float scale = -1.0f;
66 
67  if( out.z < 1.0f )
68  {
69  // Text is not clipped. Calculate scale factor.
70  float d_noscale = Nx::CViewportManager::sGet2DIn3DSpaceNoscaleDistance();
71  float s_min = Nx::CViewportManager::sGet2DIn3DSpaceMinScale();
72  float s_max = Nx::CViewportManager::sGet2DIn3DSpaceMaxScale();
73 
74  scale = d_noscale / -mid.z;
75 
76  // Clamp scale.
77  if( scale > s_max )
78  {
79  scale = s_max;
80  }
81  else if( scale < s_min )
82  {
83  scale = s_min;
84  }
85 
86  // The z value is passed in as a uint32, so set the value here based on fixed point 16:16.
87  screen_pos_z = (uint32)( out.z * 65536.0f );
88  }
89 
90  // This is the scale factor. Returning a value < 0 will cause the text to be hidden.
91  return scale;
92 }
