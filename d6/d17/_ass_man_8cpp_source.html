123 EAssetType CAssMan::FindAssetType( const char *p_assetName )
124 {
125  // given a file name, then find the type of asset that this represents
126 
127  // the asset type is determined by the file extension
128  // so first, find the last . in the assetName
129  const char *p_ext = p_assetName + strlen(p_assetName);
130  while ( p_ext > p_assetName && *p_ext != '.' )
131  {
132  p_ext--;
133  }
134 
135  if ( *p_ext != '.' )
136  {
137  return ASSET_UNKNOWN;
138  }
139 
140  p_ext++;
141  // p_ext now points to the extension for the file name
142 
143  SAssetLookup *p_lookup = &s_asset_lookup[0];
144  while ( p_lookup->p_extension != NULL )
145  {
146  // note, ignoring case
147  if ( strcmpi( p_lookup->p_extension, p_ext ) == 0 )
148  {
149  return p_lookup->type;
150  }
151  p_lookup++;
152  }
153 
154  return ASSET_UNKNOWN;
155 }

162 void* CAssMan::LoadAssetFromStream(uint32 asset_name, uint32 asset_type, uint32* p_data, int data_size, bool permanent, uint32 group)
163 {
164  // Note: There's no reason to support async loads when
165  // loading an asset from a data stream...
166 
167  // load an asset, and return a pointer to the asset data
168 
169  // given an asset name checksum, find the type of the asset, and then load it
170 
171  Dbg_MsgAssert(!AssetAllocated(asset_name),("Asset %08x already loaded",asset_name));
172 
173  CAsset* p_asset = NULL;
174 
175  // based on the asset type, create an asset of the correct type
176  switch (asset_type)
177  {
178  case ASSET_ANIM:
179  p_asset = new CAnimAsset;
180  break;
181 
182  case ASSET_SKIN:
183  p_asset = new CSkinAsset;
184  break;
185 
186  default:
187  // right now, the only file type that's supported
188  // for data streams is the "cutscene model"
189  Dbg_MsgAssert(0,("Asset %08x is of unsupported type %d",asset_name,asset_type));
190  }
191 
192  p_asset->m_permanent = permanent;
193  p_asset->m_group = group;
194  p_asset->m_dead = false; // it is not dead
195 
196  // fake an asset name for debugging
197  char pDebugAssetString[256];
198  sprintf( pDebugAssetString, "%08x from data stream", asset_name );
199  p_asset->SetText(pDebugAssetString);
200 
201  // finally, call the asset's Load member function to load it
202  int error = p_asset->Load(p_data, data_size);
203  if ( error != 0 )
204  {
205  if ( Script::GetInteger( CRCD(0x25dc7904,"AssertOnMissingAssets") ) )
206  {
207  Dbg_MsgAssert(0,("Loading asset %08x returned error %d",asset_name,error));
208  }
209  else
210  {
211  delete p_asset;
212  p_asset = NULL;
213  }
214  return NULL;
215  }
216  else
217  {
218  // only add it to the table once we're sure there's no error
219  uint32 checksum = asset_name;
220  Dbg_MsgAssert(!mp_asset_table->GetItem( checksum ),("Asset %08x already in table",asset_name));
221  mp_asset_table->PutItem( checksum, p_asset);
222  p_asset->SetID(checksum);
223 
224  // link in at the end, so things are added in order
225  p_asset->mp_next = mp_assetlist_head;
226  p_asset->mp_prev = mp_assetlist_head->mp_prev;
227  mp_assetlist_head->mp_prev->mp_next = p_asset;
228  mp_assetlist_head->mp_prev = p_asset;
229 
230  Dbg_MsgAssert(mp_asset_table->GetItem( checksum ) == p_asset,("Asset does not match entry in table"));
231  Dbg_Assert( p_asset );
232  return p_asset->GetData();
233  }
234 }

241 void* CAssMan::LoadAsset(const char *p_assetName, bool async_load, bool use_pip, bool permanent, uint32 group, void* pExtraData, Script::CStruct *pParams)
242 {
243  // load an asset, and return a pointer to the asset data
244 
245  // given an asset filename, find the type of the asset, and then load it
246 
247  Dbg_MsgAssert(!AssetAllocated(p_assetName),("Asset %s already loaded",p_assetName));
248 
249  CAsset * p_asset = NULL;
250 
251 // Find the asset type
252  EAssetType asset_type = FindAssetType(p_assetName);
253  Dbg_MsgAssert(asset_type != ASSET_UNKNOWN,("Asset %s is of unknown type",p_assetName));
254 
255 // based on that, create an asset of the correct type
256  switch (asset_type)
257  {
258  case ASSET_ANIM:
259  p_asset = new CAnimAsset;
260  break;
261 
262  case ASSET_SKELETON:
263  p_asset = new CSkeletonAsset;
264  break;
265 
266  case ASSET_SKIN:
267  p_asset = new CSkinAsset;
268  break;
269 
270  case ASSET_CUTSCENE:
271  p_asset = new CCutsceneAsset;
272  break;
273 
274  case ASSET_NODEARRAY:
275  p_asset = new CNodeArrayAsset;
276  break;
277 
278  case ASSET_SCENE:
279  case ASSET_TEXTURES:
280  case ASSET_COLLISION:
281  case ASSET_BINARY:
282  default:
283  Dbg_MsgAssert(0,("Asset %s is of unsupported type %d",p_assetName,asset_type));
284  }
285 
286  p_asset->m_permanent = permanent;
287  p_asset->m_group = group;
288  p_asset->m_dead = false; // it is not dead
289 
290  p_asset->SetText(p_assetName); // record asset name for debugging
291 
292  // Garrett: Make sure that we only set async on the things that can handle them
293  if (async_load)
294  {
295  Dbg_MsgAssert(asset_type == ASSET_ANIM, ("Can't load this asset type asynchronously: %d", asset_type));
296  }
297 
298  // finally, call the asset's Load member function to load it
299  int error = p_asset->Load(p_assetName, async_load, use_pip, pExtraData, pParams);
300  if ( error != 0 )
301  {
302  if ( Script::GetInteger( CRCD(0x25dc7904,"AssertOnMissingAssets") ) )
303  {
304  Dbg_MsgAssert(0,("Loading asset %s returned error %d",p_assetName,error));
305  }
306  else
307  {
308  delete p_asset;
309  p_asset = NULL;
310  }
311  return NULL;
312  }
313  else
314  {
315  // only add it to the table once we're sure there's no error
316  uint32 checksum = Script::GenerateCRC( p_assetName );
317  Dbg_MsgAssert(!mp_asset_table->GetItem( checksum ),("Asset %s already in table",p_assetName));
318  mp_asset_table->PutItem( checksum, p_asset);
319  p_asset->SetID(checksum);
320 
321  // link in at the end, so things are added in order
322  p_asset->mp_next = mp_assetlist_head;
323  p_asset->mp_prev = mp_assetlist_head->mp_prev;
324  mp_assetlist_head->mp_prev->mp_next = p_asset;
325  mp_assetlist_head->mp_prev = p_asset;
326 
327  Dbg_MsgAssert(mp_asset_table->GetItem( Script::GenerateCRC( p_assetName )) == p_asset,("Asset does not match entry in table"));
328  Dbg_Assert( p_asset );
329  return p_asset->GetData();
330  }
331 }

351 void* CAssMan::GetNthInGroup( uint32 group, int n )
352 {
353  // return the nth asset in the asset manager that is in this group
354  // will return NULL if there are none of the specified group
355 
356  CAsset* p_asset = mp_assetlist_head->mp_next;
357  while (p_asset != mp_assetlist_head)
358  {
359 // printf ("%d: Checkingfor %x, group = %x\n",n,group,p_asset->m_group);
360  CAsset *p_next = p_asset->mp_next;
361  if (p_asset->m_group == group)
362  {
363 // printf ("%d: GROUP MATCH %x, group = %x\n",n,group,p_asset->m_group);

365  {
366 // printf ("%d: COUNT MATCH %x, group = %x\n",n,group,p_asset->m_group);
367  return p_asset->GetData();
368  }
369  n--;
370  }
371  p_asset = p_next;
372  }
373  return NULL;

393 int CAssMan::GetIndexInGroup(uint32 checksum)
394 {
395  // find this entry, and see which number it is in the group
396  // i.e, returns the intex into the group
397 
398  CAsset* p_actual = mp_asset_table->GetItem(checksum);
399  Dbg_MsgAssert(p_actual, ("GetIndexInGroup with asset not found\n"));
400 
401  int n = 0; // index starts at 0
402  CAsset* p_asset = mp_assetlist_head->mp_next;
403  while (p_asset != mp_assetlist_head)
404  {
405  CAsset *p_next = p_asset->mp_next;
406  if (p_asset == p_actual)
407  {
408  return n; // if we've found it, then return the asset
409  }
410  if (p_asset->m_group == p_actual->m_group) // if in same group
411  {
412  n++; // increment index
413  }
414  p_asset = p_next;
415  }
416  Dbg_MsgAssert(0, ("ERROR IN GetIndexInGroup\n"));
417  return 0;
418

805 bool CAssMan::LoadAnim(const char* pFileName, uint32 animName, uint32 referenceChecksum, bool async_load, bool use_pip)
806 {
807  // If the asset was previously loaded, then we are just creating a reference to it
808  // so the m_permanent flag should be set based on m_default_permanent, and NOT INHERITED
809 
810  if (AssetAllocated(pFileName))
811  {
812  // if descChecksum was supplied, then adding a reference
813  if (animName && !AssetAllocated(animName + referenceChecksum))
814  {
815  // adding a reference
816  AddRef(pFileName, animName + referenceChecksum, referenceChecksum);
817  // the m_permanent will have been inherited from the parent asset
818  // Now get the reference asset, and set the permanent flag to m_defualt_permanet
819  CAsset *p_asset = mp_asset_table->GetItem(animName + referenceChecksum);
820  Dbg_MsgAssert(p_asset,("(Mick) Error re-getting asset ref for %s\n",pFileName));
821  //p_asset->SetPermanent(m_default_permanent); // <<<< this would also set the perm flag on the parent
822 
823  p_asset->m_permanent = m_default_permanent; // just in the reference!!!
824  return true;
825  }
826  else
827  {
828  //Dbg_MsgAssert( 0, ( "This file was already loaded %s", pFileName ) );
829  return false;
830  }
831  }
832  else
833  {
834  if (!LoadOrGetAsset(pFileName, async_load, use_pip, m_default_permanent, 0))
835  {
836  Dbg_MsgAssert(0,("Failed to load anim %s",pFileName));
837  return false;
838  }
839 
840  // Now add the reference to it, if one was requested
841  // this gets combined with the "reference checksum"
842  if (animName && !AssetAllocated(animName + referenceChecksum))
843  {
844  AddRef(pFileName, animName + referenceChecksum, referenceChecksum);
845  }
846 
847  return true;
