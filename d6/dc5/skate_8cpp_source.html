192 
193  Dbg_AssertType ( &task, Tsk::Task< Skate > );
194 
195  Skate& mdl = task.GetData();
196 
197  mdl.DoUpdate();
198 
199  if( gamenet_man->OnServer() && ( mdl.GetNumSkaters() > 0 ))
200  {
201  mdl.CheckSkaterCollisions();
202  }
203 
204  if ( mdl.GetGameMode()->IsFrontEnd() )
205  {
206  }
207  else if( ( mdl.GetGameMode()->GetNameChecksum() == CRCD(0x6ef8fda0,"netking") ) ||
208  ( mdl.GetGameMode()->GetNameChecksum() == CRCD(0x5d32129c,"king") ))
209  {
210  if( !mdl.GetGameMode()->EndConditionsMet())
211  {
212  GameNet::PlayerInfo* king;
213 
214  king = gamenet_man->GetKingOfTheHill();
215  if( king && !king->m_Skater->IsPaused())
216  {
217  Mdl::Score* score;
218  int current_score, total_score, goal_score;
219 
220  score = king->m_Skater->GetScoreObject();
221  current_score = score->GetTotalScore();
222  total_score = current_score + (int) ( ( Tmr::FrameLength() * 60.0f ) *
223  ( Tmr::vRESOLUTION / 60.0f ));
224  score->SetTotalScore( total_score );
225 
226  goal_score = mdl.GetGameMode()->GetTargetScore();
227 
228  if( mdl.GetGameMode()->IsTeamGame())
229  {
230  total_score = gamenet_man->GetTeamScore( king->m_Team );
231  }
232  else
233  {
234  total_score = score->GetTotalScore();
235  }
236 
237  // When within 10 seconds of victory, play a timer sound periodically
238  mdl.BeepTimer( (float) ( goal_score - total_score) / 1000.0f,
239  10.0f,
240  10.0f,
241  "timer_runout_beep");
242  }
243  }
244  }

347 int Skate::GetNextUnusedSkaterHeapIndex( bool for_local_skater )
348 {
349  int start_index, heap_index, num_tested;
350  bool taken;
351 
352  // Ok... kinda complicated, but it was the simplest solution I could come up with that solves the problem.
353  // The issue is that we only have vMAX_SKATERS skater heaps. And we always want to reserve skater heap 0
354  // for our local skater. But we might be observing a game that has vMAX_SKATERS, in which case you will
355  // not have a heap for the local skater at all. But if we observe a server that has ( vMAX_SKATERS - 1 )
356  // skaters and join late, we want to make sure that our skater (the last one to be loaded) gets skater
357  // heap 0
358  start_index = 1;
359  if( for_local_skater )
360  {
361  start_index = 0;
362  }
363 
364  heap_index = start_index;
365  for( num_tested = 0; num_tested < (int) GetGameMode()->GetMaximumNumberOfPlayers(); num_tested ++ )
366  {
367  taken = false;
368  for( uint i = 0; i < GetNumSkaters(); i++ )
369  {
370  if( m_skaters.GetItem(i) != NULL )
371  {
372  Obj::CSkater *pSkater = m_skaters.GetItem(i)->GetData();
373  if( pSkater->GetHeapIndex() == heap_index )
374  {
375  taken = true;
376  break;
377  }
378  }
379  }
380  if( taken == false )
381  {
382  break;
383  }
384 
385  heap_index = ( heap_index + 1 ) % GetGameMode()->GetMaximumNumberOfPlayers();
386  }
387 
388  // We assume that these skater heaps are totally empty
389  // so if something is aleft on them, then assert
390 
391 #ifdef __NOPT_ASSERT__
392  Mem::Heap *skater_heap = Mem::Manager::sHandle().SkaterHeap(heap_index);
393  Dbg_MsgAssert( skater_heap != NULL, ( "Invalid skater heap : %d\n", heap_index ));
394  if (skater_heap->mUsedBlocks.m_count)
395  {
396  printf ("Skater heap has %d used blocks still on it, it should be empty\n",skater_heap->mUsedBlocks.m_count);
397 #ifndef __PLAT_NGC__
398  MemView_AnalyzeHeap(skater_heap);
399  MemView_DumpHeap(skater_heap);
400 #endif // __PLAT_NGC__
401  Dbg_MsgAssert(0,("Skater heap has %d used blocks still on it, it should be empty\n",skater_heap->mUsedBlocks.m_count));
402  }
403 
404  Mem::Heap *geom_heap = Mem::Manager::sHandle().SkaterGeomHeap(heap_index);
405  if (geom_heap->mUsedBlocks.m_count)
406  {
407  printf ("SkaterGeom heap has %d used blocks still on it, it should be empty\n",geom_heap->mUsedBlocks.m_count);
408 #ifndef __PLAT_NGC__
409  MemView_AnalyzeHeap(geom_heap);
410  MemView_DumpHeap(geom_heap);
411 #endif // __PLAT_NGC__
412  Dbg_MsgAssert(0,("SkaterGeom heap has %d used blocks still on it, it should be empty\n",geom_heap->mUsedBlocks.m_count));
413  }
414 
415  #endif
416 
417 
418  return heap_index;
419 }

426 Obj::CSkater* Skate::add_skater ( Obj::CSkaterProfile* pSkaterProfile, bool local_client, int obj_id, int player_num )

431 
432  // GJ: copy the face texture (if it has one) from the local client's skater profile

434  // network message hasn't been implemented yet.
435  //Dbg_MsgAssert( local_client, ( "Gary's kludge for face textures only works on local client" ) );
436  if ( local_client )
437  {
438  Gfx::CModelAppearance* pModelAppearance = pSkaterProfile->GetAppearance();
439  Obj::CSkaterProfile* pOriginalProfile;
440  if (CFuncs::ScriptInSplitScreenGame( NULL, NULL ))
441  {
442  // (Mick) Kluge modified for 2 players
443  // get the correct profile
444  pOriginalProfile = mp_PlayerProfileManager->GetProfile( player_num );
445  }
446  else
447  {
448  // Otherwise, use the old kluge
449  pOriginalProfile = mp_PlayerProfileManager->GetProfile( 0 );
450  }
451 
452 
453  Gfx::CModelAppearance* pOriginalAppearance = pOriginalProfile->GetAppearance();
454 
455  *pModelAppearance = *pOriginalAppearance;
456  }
457 
458  // Change our ID to what the server says it should be to avoid ID collision amongs clients
459  if( local_client )
460  {
461  Obj::CSkater* skater;
462 
463  Script::RunScript( "show_panel_stuff" );
464  skater = GetLocalSkater();
465  if( skater )
466  {
467  // In Split-screen games, we'll have two local skaters
468  if( CFuncs::ScriptInSplitScreenGame( NULL, NULL ))
469  {
470  if( skater->GetID() == (uint32) obj_id )
471  {
472  return skater;
473  }
474 #ifdef __PLAT_NGC__
475  else
476  {
477  File::PreMgr* pre_mgr = File::PreMgr::Instance();
478  pre_mgr->LoadPre( "skaterparts.pre", false);
479  }
480 #endif
481  }
482  else
483  {
484  // In Net games, we'll only have one, but his ID can change in order to avoid
485  // ID collision over the net. So, if someone is trying to add another local skater with a different
486  // ID, assume that they just want to assign a new ID to our already-existing local skater
487  if( skater->GetID() != (uint32) obj_id )
488  {
489  Score* score;
490 
491  node = skater->GetLinkNode();
492  node->SetPri( obj_id );
493  node->Remove();
494  m_skaters.AddUniqueSequence( node );
495 
497 // Changing the id means we have to destroy and re-create the particle systems
498 // as they are referenced by the id of the skater.
499  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().SkaterHeap(skater->GetHeapIndex()));
500  Script::RunScript( "DestroySkaterParticles", NULL, skater );
501 
502  skater->SetID( obj_id );
503 
504  Script::RunScript( "InitSkaterParticles", NULL, skater );
505  Mem::Manager::sHandle().PopContext();
507 
508  score = skater->GetScoreObject();
509  score->SetSkaterId( obj_id );
510  }
511 
512  return skater;
513  }
514  }
515  else
516  {
517  Obj::CCompositeObject* cam_obj;
518 
519  // If we're about to re-create the skater, delete the skatercam as it is about to
520  // be re-created. This is to handle the case where someone goes into observer mode.
521  // In this case, we destroy the skater, but leave the camera. So now we're going back
522  // to the front end and it's re-creating the skater.
523  cam_obj = (Obj::CCompositeObject*) Obj::CCompositeObjectManager::Instance()->GetObjectByID(CRCD(0x967c138c,"skatercam0"));
524  if( cam_obj )
525  {
526  cam_obj->SetLockOff();
527  delete cam_obj;
528  }
529  }
530  }
531  else
532  {
533  // Stop vibration when loading client skaters in net games
534  if ( gamenet_man->InNetGame() )
535  {
536  Obj::CSkater* p_skater;
537 
538  p_skater = GetLocalSkater();
539  if( p_skater )
540  {
541  Obj::CVibrationComponent* p_vibration_component = GetVibrationComponentFromObject(p_skater);
542  if (p_vibration_component)
543  {
544  p_vibration_component->StopAllVibration();
545  }
546  }
547  }
548  }
549 
550  if ( gamenet_man->InNetGame() )
551  {
552  Dbg_MsgAssert( GetNumSkaters() < GetGameMode()->GetMaximumNumberOfPlayers(),( "Trying to add too many players %d : %d", GetNumSkaters(), GetGameMode()->GetMaximumNumberOfPlayers()));
553  }
554 
555  Mem::PushMemProfile("Skater"); // Could possibly replace this with the name of the skater, when we have one, so can compare in multi player
556 
557 
558  Dbg_MsgAssert(( obj_id >= 0 ) && ( obj_id < vMAX_SKATERS ),( "Invalid skater object id\n" ));
559 
560  heap_index = GetNextUnusedSkaterHeapIndex( local_client );
561 
562  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().SkaterHeap(heap_index));

573  //Pcm::PauseMusic(0);

578  {
579  Script::RunScript( "ClientSkaterInit", NULL, skater );
580  }
581

583  // so the order will be consistent across all clients
584  node = skater->GetLinkNode();
585  node->SetPri( skater->GetID() );
586  m_skaters.AddUniqueSequence( node );
587 
588  if ( GetGameMode()->IsFrontEnd() )
589  {
590  // front end doesn't have a server, so we need

597  // printf("skaterIndex = %i, player_num = %i\n", mp_controller_preferences[player_num].skaterIndex, player_num);

599

602  {

604  {
605  printf("CAT: Restoring Created Tricks\n");

607  delete spCATData;
608  spCATData=NULL;
609  }
610 
611  // restore stat goal data too!
612  if (spStatData)
613  {
614  printf("CAT: Restoring Stat Goal data\n");
615  skater->LoadStatGoalInfo(spStatData);
616  delete spStatData;
617  spStatData=NULL;
618  }
619  }
620 
621  // update trick mappings if this is the second skater in a split screen game
622  if ( skater->GetID() == 0x00000001 && CFuncs::ScriptInSplitScreenGame( NULL, NULL ) )
623  {
624  // Obj::CPlayerProfileManager* pProfileMan = GetPlayerProfileManager();
625  // Obj::CSkaterProfile* pProfile = pProfileMan->GetProfile( 1 );
626 
627  Obj::CTrickComponent* p_trick_component = GetTrickComponentFromObject(skater);
628  Dbg_Assert(p_trick_component);
629  p_trick_component->UpdateTrickMappings( pSkaterProfile );
630  }
631 
632  Mem::PopMemProfile(/*"Skater"*/);
633  Mem::Manager::sHandle().PopContext();
634 
635  return skater;
636 }

677 int Skate::find_restart_node( int index )
678 {
679  int node = -1;
680  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
681 
682  // get appropriate restart point...
683  if( IsMultiplayerGame())
684  {
685  if( GetGameMode()->IsTeamGame())
686  {
687  if( GetGameMode()->GetNameChecksum() == Script::GenerateCRC( "netlobby" ))
688  {
689  node = Obj::GetRestartNode( Script::GenerateCRC( "team" ), 0 );
690  }
691  else if( GetGameMode()->GetNameChecksum() == Script::GenerateCRC( "netctf" ))
692  {
693  GameNet::PlayerInfo* player;
694 
695  player = gamenet_man->GetLocalPlayer();
696  if( player )
697  {
698  switch( player->m_Team )
699  {
700  case GameNet::vTEAM_RED:
701  node = SkateScript::FindNamedNode( "TRG_CTF_Restart_Red" );
702  break;
703  case GameNet::vTEAM_BLUE:
704  node = SkateScript::FindNamedNode( "TRG_CTF_Restart_Blue" );
