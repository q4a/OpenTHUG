226 void CCompetitionGoal::AwardCash( Script::CStruct* p_reward_params )
227 {
228 #ifdef __NOPT_ASSERT__
229  CGoalManager* pGoalManager = GetGoalManager();
230  Dbg_Assert( pGoalManager );
231 #endif // __NOPT_ASSERT__
232 
233  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
234  Mdl::CCompetition* pComp = skate_mod->GetCompetition();
235 
236  // figure any cash reward
237  if ( pComp->PlaceIs( 1 ) || pComp->PlaceIs( 2 ) || pComp->PlaceIs( 3 ) )
238  {
239  int total_reward;
240  int current_reward = 0;
241  if ( mp_params->GetInteger( "reward_cash", &total_reward, Script::NO_ASSERT ) )
242  {
243  if ( pComp->PlaceIs( 1 ) && !m_gotGold )
244  {
245  current_reward = total_reward - m_rewardGiven;
246  m_gotGold = true;
247  m_gotSilver = true;
248  m_gotBronze = true;
249  }
250  else if ( pComp->PlaceIs( 2 ) && !m_gotSilver )
251  {
252  current_reward = total_reward * 2 / 3;
253  if ( m_gotBronze )
254  current_reward -= total_reward / 3;
255 
256  m_gotSilver = true;
257  m_gotBronze = true;
258 
259  }
260  else if ( pComp->PlaceIs( 3 ) && !m_gotBronze )
261  {
262  current_reward = total_reward / 3;
263  m_gotBronze = true;
264  }
265 
266  if ( current_reward > 0 )
267  {
268  // pGoalManager->AddCash( current_reward );
269  // Script::RunScript( "goal_got_cash" );
270  p_reward_params->AddInteger( "cash", current_reward );
271  m_rewardGiven += current_reward;
272  }
273  }
274  }
275 }
