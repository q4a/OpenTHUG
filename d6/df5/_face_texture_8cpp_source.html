44 void CFaceTexture::LoadFace( const char* pFaceName, bool fullPathIncluded )
45 {
46  // As a test, set the freak image
47  // (otherwise, we'd need to go through the face download interface)
48  char extended_filename[512];
49  int file_size;
50 
51  if ( fullPathIncluded )
52  {
53  sprintf( extended_filename, "%s.img.ps2", pFaceName );
54  }
55  else
56  {
57  sprintf( extended_filename, "images\\%s.img.ps2", pFaceName );
58  }
59 
60  char* pData = (char*)File::LoadAlloc( extended_filename, &file_size );
61  memcpy( m_rawData, pData, file_size );
62  Dbg_MsgAssert( file_size <= vRAW_TEXTURE_SIZE, ( "File too big (file is %d, buffer is %d)", file_size, vRAW_TEXTURE_SIZE ) );
63  Mem::Free( pData );
64 
65  m_isValid = true;
66 }

98 int CFaceTexture::WriteToBuffer( uint8* pBuffer, int bufferSize )
99 {
100  int totalSize = 0;
101 
102  totalSize += vRAW_TEXTURE_SIZE;
103  Dbg_MsgAssert( bufferSize >= totalSize, ( "Buffer size (%d bytes) must >= %d bytes", bufferSize, totalSize ) );
104  memcpy( pBuffer, &m_rawData[0], vRAW_TEXTURE_SIZE );
105  pBuffer += vRAW_TEXTURE_SIZE;
106 
107  totalSize += sizeof(Nx::SFacePoints);
108  Dbg_MsgAssert( bufferSize >= totalSize, ( "Buffer size (%d bytes) must >= %d bytes", bufferSize, totalSize ) );
109  memcpy( pBuffer, &m_points, sizeof(Nx::SFacePoints) );
110  pBuffer += sizeof(Nx::SFacePoints);
111 
112  totalSize += ( strlen( m_overlayTextureName ) + 1 );
113  Dbg_MsgAssert( bufferSize >= totalSize, ( "Buffer size (%d bytes) must >= %d bytes", bufferSize, totalSize ) );
114  memcpy( pBuffer, m_overlayTextureName, strlen( m_overlayTextureName ) + 1 );
115  pBuffer += ( strlen( m_overlayTextureName ) + 1 );
116 
117  return totalSize;

170 void CFaceTexture::WriteIntoStructure( Script::CStruct* pSubStruct )
171 {
172  Dbg_Assert( pSubStruct );
173  Dbg_MsgAssert( IsValid(), ( "Face texture is not valid!" ) );
174 
175  Script::CArray* pRawTextureArray = new Script::CArray;
176  Dbg_MsgAssert( ( vRAW_TEXTURE_SIZE ) % 4 == 0, ( "Was expecting raw texture size to be a multiple of 4", vRAW_TEXTURE_SIZE ) );
177  pRawTextureArray->SetSizeAndType( vRAW_TEXTURE_SIZE / 4, ESYMBOLTYPE_NAME );
178 
179  uint32* pFaceData = (uint32*)&m_rawData[0];
180 
181  for ( int i = 0; i < (vRAW_TEXTURE_SIZE / 4); i++ )
182  {
183 
184  pRawTextureArray->SetChecksum( i, *pFaceData );
185  pFaceData++;
186  }
187 
188  pSubStruct->AddArrayPointer( CRCD(0x99790d96,"rawTextureData"), pRawTextureArray );
189  pSubStruct->AddString( CRCD(0xa7e7a264,"overlayTextureName"), m_overlayTextureName );
190  Nx::SetFacePointsStruct( m_points, pSubStruct );
191 
192  // readfromstructure should also set the valid flag
193 }

200 void CFaceTexture::ReadFromStructure( Script::CStruct* pSubStruct )
201 {
202  Dbg_Assert( pSubStruct );
203 
204  Script::CArray* pRawTextureArray;
205  pSubStruct->GetArray( CRCD(0x99790d96,"rawTextureData"), &pRawTextureArray, Script::ASSERT );
206 
207  uint32* pFaceData = (uint32*)&m_rawData[0];
208  Dbg_Assert( pRawTextureArray->GetSize() * sizeof(uint32) == vRAW_TEXTURE_SIZE );
209  for ( uint32 i = 0; i < pRawTextureArray->GetSize(); i++ )
210  {
211  *pFaceData = pRawTextureArray->GetChecksum( i );
212  pFaceData++;
213  }
214 
215  const char* pOverlayTextureName;
216  pSubStruct->GetString( CRCD(0xa7e7a264,"overlayTextureName"), &pOverlayTextureName, Script::ASSERT );
217  strcpy( m_overlayTextureName, pOverlayTextureName );
218 
219  Nx::GetFacePointsStruct( m_points, pSubStruct );
220 }
