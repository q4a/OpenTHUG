206 void CWeaponComponent::ProcessStickyTarget( float heading_change, float tilt_change, float *p_extra_heading_change, float *p_extra_tilt_change )
207 {
208  // Zero the extra chnage values.
209  *p_extra_heading_change = 0.0f;
210  *p_extra_tilt_change = 0.0f;
211 
212  // If the player is trying to aim, this section of code tries to 'nudge' the aim in the right direction imperceptibly by
213  // moving the cursor just a little more or less based on the requested movement of the cursor.
214  if( mp_current_target )
215  {
216  if( heading_change != 0.0f )
217  {
218  // Figure default adjustment amount.
219  float heading_cursor_suck = Script::GetFloat( CRCD( 0xd90b2dfb, "GunslingerLookaroundHeadingCursorSuck" ), Script::ASSERT );
220  float default_adjustment = heading_change * heading_cursor_suck;
221 
222  Mth::Vector immediate_camera_pos = m_sight_pos;
223  Mth::Vector ped_pos = mp_current_target->GetPos() + Mth::Vector( 0.0f, 36.0f, 0.0f );
224  Mth::Vector ped_to_cam = ped_pos - immediate_camera_pos;
225  ped_to_cam[Y] = 0.0f;
226  ped_to_cam.Normalize();
227 
228  // Figure exactly what the angular difference is.
229  float dp = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
230  float angular_difference = acosf( dp );
231 
232  // If the angular difference is actually smaller than the default adjustment amount, make the default adjustment amount the angular difference.
233  if( Mth::Abs( angular_difference ) < Mth::Abs( default_adjustment ))
234  {
235  default_adjustment = angular_difference;
236  }
237 
238  // Now save off this matrix, and increase the rotation by some percentage of the rotation added to the lookaround heading.
239  Mth::Matrix saved_mat = m_sight_matrix;
240  m_sight_matrix.RotateYLocal( default_adjustment );
241 
242  // See if this brings us closer.
243  float dp2 = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
244  if( dp2 > dp )
245  {
246  // Yes it did, so store this change.
247  *p_extra_heading_change = default_adjustment;
248  }
249  else
250  {
251  // No it didn't so revert to the old matrix.
252  m_sight_matrix = saved_mat;
253 
254  // Maybe this change went too far? Try undoing 10% of the last change.
255  m_sight_matrix.RotateYLocal( -default_adjustment );
256 
257  // See if this brings us closer.
258  float dp3 = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
259  if( dp3 > dp )
260  {
261  // Yes it did, so store this change.
262  *p_extra_heading_change = -default_adjustment;
263  }
264  else
265  {
266  // No it didn't so revert to the old matrix.
267  m_sight_matrix = saved_mat;
268  }
269  }
270  }
271  }
272 
273  if( mp_current_target )
274  {
275  if( tilt_change != 0.0f )
276  {
277  // Figure default adjustment amount.
278  float tilt_cursor_suck = Script::GetFloat( CRCD( 0x4ece2698, "GunslingerLookaroundTiltCursorSuck" ), Script::ASSERT );
279  float default_adjustment = tilt_change * tilt_cursor_suck;
280 
281  Mth::Vector immediate_camera_pos = m_sight_pos;
282  Mth::Vector ped_pos = mp_current_target->GetPos() + Mth::Vector( 0.0f, 36.0f, 0.0f );
283  Mth::Vector ped_to_cam = ped_pos - immediate_camera_pos;
284  ped_to_cam.Normalize();
285 
286  // Figure exactly what the angular difference is.
287  float dp = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
288  float angular_difference = acosf( dp );
289 
290  // If the angular difference is actually smaller than the default adjustment amount, make the default adjustment amount the angular difference.
291  if( Mth::Abs( angular_difference ) < Mth::Abs( default_adjustment ))
292  {
293  default_adjustment = angular_difference;
294  }
295 
296  // Now save off this matrix, and increase the rotation by some percentage of the rotation added to the lookaround heading.
297  Mth::Matrix saved_mat = m_sight_matrix;
298  m_sight_matrix.RotateXLocal( default_adjustment );
299 
300  // See if this brings us closer.
301  float dp2 = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
302  if( dp2 > dp )
303  {
304  // Yes it did, so store this change.
305  *p_extra_tilt_change = default_adjustment;
306  }
307  else
308  {
309  // No it didn't so revert to the old matrix.
310  m_sight_matrix = saved_mat;
311 
312  // Maybe this change went too far? Try undoing 10% of the last change.
313  m_sight_matrix.RotateXLocal( -default_adjustment );
314 
315  // See if this brings us closer.
316  float dp3 = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
317 
318  if( dp3 > dp )
319  {
320  // Yes it did, so store this change.
321  *p_extra_tilt_change = -default_adjustment;
322  }
323  else
324  {
325  // No it didn't so revert to the old matrix.
326  m_sight_matrix = saved_mat;
327  }
328  }

378 CCompositeObject* CWeaponComponent::GetCurrentTarget( Mth::Vector& start_pos, Mth::Vector* p_reticle_max )
379 {
380  // This is the maximum distance that we will check for targets.
381  float target_distance = Script::GetFloat( CRCD( 0x89495011,"GunslingerTargetDistance" ), Script::ASSERT );
382 
383  Mth::Vector reticle_min = start_pos;
384  m_reticle_max = reticle_min - ( m_sight_matrix[Z] * target_distance );
385 
386  // Ignore faces based on face flags.
387  CFeeler feeler;
388  feeler.SetIgnore((uint16)( mFD_NON_COLLIDABLE | mFD_NON_CAMERA_COLLIDABLE ), 0 );
389  feeler.SetLine( reticle_min, m_reticle_max );
390  bool collision = feeler.GetCollision( true );
391  if( collision )
392  {
393  m_reticle_max = feeler.GetPoint();
394 
395  // Set the new target distance at the point of contact, to avoid considering targets beyond the collision point.
396  target_distance = target_distance * feeler.GetDist();
397  }
398 
399  // Only targets within the allowed angle will be considered.
400  mp_current_target = NULL;
401  float best_dp = SPIN_MODULATION_ANGLE;
402  Mth::Vector best_pos;
403 
404  // Rather than cycling through the ped logic components, eventually everything that is targettable should contain a targetting component.
405  Obj::CPedLogicComponent *p_ped_logic_component = static_cast<Obj::CPedLogicComponent*>( Obj::CCompositeObjectManager::Instance()->GetFirstComponentByType( CRC_PEDLOGIC ));
406  while( p_ped_logic_component )
407  {
408  Obj::CCompositeObject *p_ped = p_ped_logic_component->GetObject();
409 
410  // Have to hack in a height adjustment here - the default position appears to be at ground level.
411  Mth::Vector ped_pos = p_ped->GetPos() + Mth::Vector( 0.0f, 36.0f, 0.0f );
412 
413  // Want to get the position of this ped.
414  Mth::Vector ped_to_cam = ped_pos - reticle_min;
415 
416  // Check not too far away.
417  if( ped_to_cam.LengthSqr() < ( target_distance * target_distance ))
418  {
419  ped_to_cam.Normalize();
420 
421  // Calculate angle this ped subtends at the camera position.
422  float dp = Mth::DotProduct( ped_to_cam, -m_sight_matrix[Z] );
423  if( dp > best_dp )
424  {
425  best_dp = dp;
426  best_pos = ped_pos;
427  mp_current_target = p_ped;
428  }
429  }
430  p_ped_logic_component = static_cast<Obj::CPedLogicComponent*>( p_ped_logic_component->GetNextSameType());
431  }
432 
433  // Reset spin and tilt modulators.
434  m_spin_modulator = 1.0f;
435  m_tilt_modulator = 1.0f;
436 
437  if( mp_current_target )
438  {
439  // We have a potential candidate for targetting.
440  // However we still want to check that it is close enough to our line of sight.
441  float dist_to_ped = Mth::Distance( best_pos, reticle_min );
442  float dist_to_line = dist_to_ped * sqrtf( 1.0f - ( best_dp * best_dp ));
443 
444  // This value should really be per-weapon in some script array.
445  if( dist_to_line < Script::GetFloat( CRCD( 0x755235db, "GunslingerLookaroundModulationDistance" ), Script::ASSERT ))
446  {
447  // Is this the same target as last time?
448 // if( mp_best_target == p_selected_target )
449 // {
450 // // If so, increase the targetting timer.
451 // target_selection_timer += frame_length;
452 // }
453 // else
454 // {
455 // // Otherwise, reset the timer.
456 // p_selected_target = mp_best_target;
457 // target_selection_timer = frame_length;
458 // }
459 
460  // Modulator gets smaller as result tends to 1.0.
461  float max_damping = Script::GetFloat( CRCD( 0xef314a12, "GunslingerLookaroundDamping" ), Script::ASSERT );
462  m_spin_modulator = 1.0f - ( max_damping * (( best_dp - SPIN_MODULATION_ANGLE ) / ( 1.0f - SPIN_MODULATION_ANGLE )));
463 
464 // if( gun_fired )
465 // {
466 // // Make this ped fall down.
467 // mp_best_target->SelfEvent( CRCD( 0xfaeec40f, "SkaterInAvoidRadius" ));
468 // }
469  }
470  else
471  {
472  mp_current_target = NULL;
473  }
474  }
475 
476  // Currently, set tilt modulator to be the same as spin modulator.
477  m_tilt_modulator = m_spin_modulator;
478 
479  if( p_reticle_max )
480  {
481  *p_reticle_max = m_reticle_max;
482  }
483 
484  return mp_current_target;
485 }
