673 
674  if ( affect_tree && mp_parents && mp_parents->m_relative )

718 bool CGoal::Win( void )
719 {
720  mp_params->AddInteger( CRCD(0xc22a2b72,"win_record"), 0 );
721  // see if it's a leaf node
722  if ( mp_children->m_relative == 0 )
723  {
724  mp_params->AddChecksum( NONAME, CRCD( 0xc309cad1, "just_won_goal" ) );
725 
726  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
727 
728  m_numTimesWon++;
729 
730  // look for a new record if we're not in a net game
731  bool set_new_record = false;
732  uint32 record_type;
733  if ( !gamenet_man->InNetGame() && mp_params->GetChecksum( CRCD(0x96963902,"record_type"), &record_type, Script::NO_ASSERT ) )
734  {
735  // printf("record type is %s\n", Script::FindChecksumName( record_type ) );
736  if ( record_type == CRCD(0xcd66c8ae,"score") )
737  {
738  // check for custom score record
739  int custom_record;
740  if ( mp_params->GetInteger( CRCD(0xfc9630ab,"custom_score_record"), &custom_record, Script::NO_ASSERT ) )
741  {
742  if ( custom_record > m_winScoreRecord )
743  {
744  m_winScoreRecord = custom_record;
745  mp_params->AddInteger( CRCD(0xc22a2b72,"win_record"), m_winScoreRecord );
746  set_new_record = true;
747  }
748  }
749  else
750  {
751  // printf("found a score record type\n");
752  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
753  Obj::CSkater *pSkater = skate_mod->GetLocalSkater();
754  Dbg_Assert( pSkater );
755 
756  Obj::CSkaterScoreComponent* pSkaterScoreComponent = GetSkaterScoreComponentFromObject(pSkater);
757  Dbg_Assert( pSkaterScoreComponent );
