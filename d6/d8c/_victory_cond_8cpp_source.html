177 bool
178 CScoreReachedVictoryCondition::ConditionComplete()
179 {
180  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
181 
182  if( skate_mod->GetGameMode()->IsTeamGame())
183  {
184  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
185  Lst::Search< GameNet::PlayerInfo > sh;
186  GameNet::PlayerInfo* player;
187  int i;
188  int total_score[GameNet::vMAX_TEAMS] = {0};
189 
190  for( player = gamenet_man->FirstPlayerInfo( sh ); player; player = gamenet_man->NextPlayerInfo( sh ))
191  {
192  Mdl::Score* pScore = player->m_Skater->GetScoreObject();
193 
194  total_score[player->m_Team] += pScore->GetTotalScore();
195  }
196 
197  for( i = 0; i < GameNet::vMAX_TEAMS; i++ )
198  {
199  if( total_score[i] >= (int32)m_TargetScore )
200  {
201  return true;
202  }
203  }
204  }
205  else
206  {
207  for ( uint32 i = 0; i < skate_mod->GetNumSkaters(); i++)
208  {
209  // score object
210  Mdl::Score* pScore = skate_mod->GetSkater(i)->GetScoreObject();
211 
212  if ( pScore->GetTotalScore() >= (int32)m_TargetScore )
213  {
214  return true;
215  }
216  }
217  }
218 
219 
220  return false;
221 }

302 bool GoalAttackComplete()
303 {
304  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
305  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
306  Game::CGoalManager* pGoalManager;
307 
308  pGoalManager = Game::GetGoalManager();
309  if( skate_mod->GetGameMode()->IsTeamGame())
310  {
311  int i, num_selected_goals;
312 
313  num_selected_goals = pGoalManager->GetNumSelectedGoals();
314  for( i = 0; i < skate_mod->GetGameMode()->NumTeams(); i++ )
315  {
316  if( gamenet_man->GetTeamScore( i ) >= num_selected_goals )
317  {
318  return true;
319  }
320  }
321  }
322  else
323  {
324  uint32 i, j, num_goals;
325  bool completed_all;
326  Game::CGoal* pGoal;
327 
328  num_goals = pGoalManager->GetNumGoals();
329  for( i = 0; i < skate_mod->GetNumSkaters(); i++ )
330  {
331  Obj::CSkater* skater = skate_mod->GetSkater(i);
332  completed_all = true;
333  for( j = 0; j < num_goals; j++ )
334  {
335  pGoal = pGoalManager->GetGoalByIndex( j );
336 
337  if( pGoal->GetParents()->m_relative != 0 )
338  {
339  continue;
340  }
341  if( pGoal->IsSelected())
342  {
343  if( pGoal->HasWonGoal( skater->GetID()) == false )
344  {
345  completed_all = false;
346  break;
347  }
348  }
349  }
350 
351  if( completed_all )
352  {
353  return true;
354  }
355  }
356  }
357 
358  return false;
359 }

487 CVictoryCondition* CreateVictoryConditionInstance( Script::CScriptStructure* p_structure )
488 {
489 
490 
491  Dbg_Assert( p_structure );
492 
493  uint32 condition_type;
494  p_structure->GetChecksum( "type", &condition_type, true );
495 
496  CVictoryCondition* pVictoryCondition = NULL;
497 
498  switch ( condition_type )
499  {
500  case vVICTORYCOND_HIGHESTSCORE:
501  {
502  pVictoryCondition = new CHighestScoreVictoryCondition( );
503  }
504  break;
505  case vVICTORYCOND_TARGETSCORE:
506  {
507  int target_score;
508  p_structure->GetInteger( "score", &target_score, true );
509  pVictoryCondition = new CScoreReachedVictoryCondition( (uint32)target_score );
510  }
511  break;
512  case vVICTORYCOND_BESTCOMBO:
513  {
514  int num_combos;
515  p_structure->GetInteger( "num_combos", &num_combos, true );
516  pVictoryCondition = new CBestComboVictoryCondition( (uint32)num_combos );
517  }
518  break;
519  case vVICTORYCOND_COMPLETEGOALS:
520  {
521  pVictoryCondition = new CGoalsCompletedVictoryCondition;
522  }
523  break;
524  case vVICTORYCOND_LASTMANSTANDING:
525  {
526  pVictoryCondition = new CLastManStandingVictoryCondition;
527  }
528  break;
529  default:
530  {
531  Dbg_MsgAssert( 0,( "Unrecognized victory condition type %08x %s", condition_type, Script::FindChecksumName( condition_type ) ));
532  }
533  break;
534  }
535 
536  return pVictoryCondition;
537 }

553 bool CLastManStandingVictoryCondition::ConditionComplete()
554 {
555  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
556  GameNet::PlayerInfo* player;
557 
558  // This logic may seem strange. Basically, when ConditionComplete() returns true,
559  // the local skater ends his run and he can observe other players. So, it will return
560  // true when his hit points have been depleted rather than when he has won.
561 
562  player = gamenet_man->GetLocalPlayer();
563  if( player && player->m_Skater )
564  {
565  Mdl::Score* pScore = player->m_Skater->GetScoreObject();
566 
567  if( pScore->GetTotalScore() <= 0 )
568  {
569  return true;
570  }
571  }
572 
573  return false;
574 }
