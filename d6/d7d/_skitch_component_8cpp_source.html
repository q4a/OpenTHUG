93 CBaseComponent::EMemberFunctionResult CSkitchComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
94 {
95  switch ( Checksum )
96  {
97  // @script | Obj_AllowSkitching | Enable skitching for this object
98  // @flag off | disable skitching
99  case 0xc38cce3b: // Obj_AllowSkitching
100  {
101  AllowSkitch(( pParams->ContainsFlag( "off" )) ? false : true );
102  break;
103  }
104 
105  default:
106  return CBaseComponent::MF_NOT_EXECUTED;
107  }
108 
109  // the "default" case of the switch statement handles
110  // unrecognized functions; if we make it down here,
111  // that means that the component already handled it
112  // somehow
113  return CBaseComponent::MF_TRUE;
114 }

140 int CSkitchComponent::GetNearestSkitchPoint( Mth::Vector *p_point, Mth::Vector &pos )
141 {
142  int nearest_node_number = -1; // -1 means not found yet
143 
144  // Only continue if we have a parent object (we always should) and a object hook manager component.
145  CCompositeObject *p_parent_object = GetObject();
146  if( mp_object_hook_manager_component && p_parent_object )
147  {
148  Obj::CObjectHookManager *p_object_hook_man = mp_object_hook_manager_component->GetObjectHookManager();
149  if( p_object_hook_man )
150  {
151  // Form a transformation matrix.
152  Mth::Matrix total_mat = p_parent_object->GetMatrix();
153  total_mat[X][W] = 0.0f;
154  total_mat[Y][W] = 0.0f;
155  total_mat[Z][W] = 0.0f;
156  total_mat[W] = p_parent_object->GetPos();
157  total_mat[W][W] = 1.0f;
158 
159  p_object_hook_man->UpdateTransform( total_mat );
160 
161  int node_number = 0;
162  CObjectHookNode *p_node = p_object_hook_man->GetFirstNode();
163  CObjectHookNode *p_nearest = p_node;
164  float nearest_dist_squared = 100000000000000.0f;
165  Mth::Vector nearest_point(0.0f, 0.0f, 0.0f);
166 
167  while( p_node )
168  {
169  // Get the postition of the hook.
170  Mth::Vector point = p_object_hook_man->GetPos( p_node );
171 
172  // Project the point onto the base plane of the object so we can ignore the height.
173  point -= p_parent_object->GetPos();
174  point.ProjectToPlane( p_parent_object->GetMatrix()[Y] );
175  point += p_parent_object->GetPos();
176 
177  float dist_squared = ( point - pos ).LengthSqr();
178  if( dist_squared < nearest_dist_squared )
179  {
180  nearest_dist_squared = dist_squared;
181  nearest_point = point;
182  p_nearest = p_node;
183  nearest_node_number = node_number;
184  }
185  p_node = p_node->GetNext();
186  node_number++;
187  }
188  *p_point = nearest_point;
189  }
190  }
191 
192  return nearest_node_number;
193 }

197 /******************************************************************/
198 /* */
199 /* Given the index of a skitch point, then get the position of */
200 /* that point. */
201 /* Adjust the point to lay on the ground plane of the object. */
202 /* Returns the position in *p_point returns true if object still */
203 /* exists, and false otherwise. */
204 /* */
205 /******************************************************************/
206 bool CSkitchComponent::GetIndexedSkitchPoint( Mth::Vector *p_point, int index )
207 {
208  // Only continue if we have a parent object (we always should) and a object hook manager component.
209  CCompositeObject *p_parent_object = GetObject();
210  if( mp_object_hook_manager_component && p_parent_object )
211  {
212  Obj::CObjectHookManager *p_object_hook_man = mp_object_hook_manager_component->GetObjectHookManager();
213  if( p_object_hook_man )
214  {
215  // Form a transformation matrix.
216  Mth::Matrix total_mat = p_parent_object->GetMatrix();
217  total_mat[X][W] = 0.0f;
218  total_mat[Y][W] = 0.0f;
219  total_mat[Z][W] = 0.0f;
220  total_mat[W] = p_parent_object->GetPos();
221  total_mat[W][W] = 1.0f;
222 
223  p_object_hook_man->UpdateTransform( total_mat );
224 
225  int node_number = 0;
226  CObjectHookNode *p_node = p_object_hook_man->GetFirstNode();
227 
228  while( p_node )
229  {
230  if( node_number == index )
231  {
232  Mth::Vector point = p_object_hook_man->GetPos( p_node );
233 
234  // Project the point onto the base plane of the object so we can ignore the height.
235  point -= p_parent_object->GetPos();
236  point.ProjectToPlane( p_parent_object->GetMatrix()[Y] );
237  point += p_parent_object->GetPos();
238 
239  *p_point = point;
240 
241  // There was no break here in the original code in MovingObject, but I see no reason
242  // to continue the loop iteration at this point.
243  break;
244  }
245  p_node = p_node->GetNext();
246  node_number++;
247  }
248  return true;
249  }
250  }
251  return false;
252 }
