65 void CPs2AsyncFileHandle::io_callback(EAsyncFunctionType function, int result, uint32 data)
66 {
67  SFileIOResultPacket *p_packet = (SFileIOResultPacket *) data;
68  int return_data = p_packet->m_return_data;
69 
70  // Do any special function processing
71  switch (function)
72  {
73  case FUNC_OPEN:
74  m_file_handle_index = result; // Save IOP file handle index
75  if (result < 0)
76  {
77  //Dbg_MsgAssert(0, ("Error: Async open returned %d", result));
78  m_file_size = 0;
79  } else {
80  m_file_size = return_data;
81  ShowAsyncInfo("io_callback: Open returned handle index %d with filesize %d\n", result, return_data);
82  }
83  break;
84 
85  case FUNC_SEEK:
86  if (result < 0)
87  {
88  Dbg_MsgAssert(0, ("Error: Async seek returned %d", result));
89  } else {
90  m_position = result;
91  }
92  break;
93 
94  case FUNC_LOAD:
95  case FUNC_READ:
96  if (mp_temp_aligned_buffer && (result > 0))
97  {
98  ShowAsyncInfo("Copying from %x to %x of size %d", mp_temp_aligned_buffer, mp_non_aligned_buffer, result);
99  memcpy(mp_non_aligned_buffer, mp_temp_aligned_buffer, result);
100  }
101 
102  // don't break, continue on below
103 
104  case FUNC_WRITE:
105  if (result < 0)
106  {
107  Dbg_MsgAssert(0, ("Error: Async IO returned %d", result));
108  } else {
109  m_position += result;
110  if (mp_temp_aligned_buffer)
111  {
112  CPs2AsyncFileLoader::sIAddBufferToFreeList(mp_temp_aligned_buffer_base);
113  mp_non_aligned_buffer = NULL;
114  mp_temp_aligned_buffer = NULL;
115  mp_temp_aligned_buffer_base = NULL;
116  }
117  }
118  break;
119 
120  case FUNC_CLOSE:
121  if (get_busy_count() > 1)
122  {
123  scePrintf("********* ERROR: Still other %d requests open after close completed on handle %x and request id %d\n", get_busy_count() - 1, this, p_packet->m_header.opt);
124  //while (1)
125  // ;
126  }
127  break;
128 
129  case FUNC_IDLE:
130  // Don't want m_busy to change or user-defined callback to be executed
131  return;
132 
133  default:
134  break;
135  }
136 
137  // Check to make sure request buffer will be clear
138  if (get_busy_count() > 0)
139  {
140  //Dbg_MsgAssert(m_num_open_requests == 1, ("Sill other open requests"));
141  }
142 
143  // Now handle the above p-line stuff
144  CAsyncFileHandle::io_callback(function, result, return_data);
145 }

206  // And re-enable interrupts

209  //scePrintf("Adding request %d to entry #%d for %x with busy %d\n", request_id, (m_num_open_requests-1), this, get_busy_count());

462 size_t CPs2AsyncFileHandle::plat_read(void *p_buffer, size_t size, size_t count)
463 {
464  //Dbg_MsgAssert(m_file_handle_index != -1, ("Invalid IOP file handle index: %d", m_file_handle_index));
465  //Dbg_MsgAssert((((uint32) p_buffer) & 0xF) == 0, ("Buffer %x needs to be quadword aligned", p_buffer));
466 
467  // Now send the packet
468  SFileIORequestPacket read_packet __attribute__ ((aligned(16)));
469 
470  uint32 non_aligned_cache_start_bytes = 0;
471  uint32 non_aligned_cache_end_bytes = 0;
472 
473  // Read in non-aligned bytes first (this is a temp hack)
474  if (m_mem_destination == MEM_EE)
475  {
476  // only look for 16-byte alignment here
477  bool non_aligned_bytes = ((uint32) p_buffer) & 0xF;
478 
479  // Also check to see if we aren't cache aligned on either end of a small buffer, since we assume the
480  // cache alignment problem is at the end of a small buffer.
481  int non_aligned_cache_bytes = ((uint32) p_buffer) & FILEIO_CACHE_BLOCK_NONALIGN_MASK;
482  if (non_aligned_cache_bytes)
483  {
484  non_aligned_cache_bytes = FILEIO_CACHE_BLOCK_SIZE - non_aligned_cache_bytes;
485  }
486  //bool non_aligned_cache_size = ((size * count) - non_aligned_cache_bytes) & FILEIO_CACHE_BLOCK_NONALIGN_MASK;
487 
488  if (non_aligned_bytes || ((non_aligned_cache_bytes /*|| non_aligned_cache_size*/) && (size * count) < FILEIO_BUFFER_SIZE))
489  {
490  Dbg_MsgAssert(!mp_temp_aligned_buffer_base, ("Handling two small reads in succession not implemented yet"));
491 
492  int aligned_size = ((size * count) + (2 * (FILEIO_CACHE_BLOCK_SIZE - 1))) & FILEIO_CACHE_BLOCK_ALIGN_MASK;
493  mp_non_aligned_buffer = (uint8 *) p_buffer;
494  mp_temp_aligned_buffer_base = CPs2AsyncFileLoader::sGetBuffer(aligned_size);
495  mp_temp_aligned_buffer = (uint8 *) (((uint32) mp_temp_aligned_buffer_base + (FILEIO_CACHE_BLOCK_SIZE - 1)) & FILEIO_CACHE_BLOCK_ALIGN_MASK);
496  p_buffer = mp_temp_aligned_buffer;
497  }
498  else
499  {
500  // Now check for cache-alignment problems that will be handled with the messages
501  non_aligned_cache_start_bytes = (((uint32) p_buffer) & FILEIO_CACHE_BLOCK_NONALIGN_MASK);
502  if (non_aligned_cache_start_bytes)
503  {
504  non_aligned_cache_start_bytes = FILEIO_CACHE_BLOCK_SIZE - non_aligned_cache_start_bytes;
505  }
506  non_aligned_cache_end_bytes = ((size * count) - non_aligned_cache_start_bytes) & FILEIO_CACHE_BLOCK_NONALIGN_MASK;
507  }
508  }
509 
510  read_packet.m_request.m_command = FILEIO_READ;
511  read_packet.m_request.m_param.m_rw.m_file_handle = m_file_handle_index;
512  read_packet.m_request.m_param.m_rw.m_buffer = p_buffer;
513  read_packet.m_request.m_param.m_rw.m_size = size * count;
514  read_packet.m_request.m_param.m_rw.m_non_aligned_start_bytes = non_aligned_cache_start_bytes;
515  read_packet.m_request.m_param.m_rw.m_non_aligned_end_bytes = non_aligned_cache_end_bytes;
516 
517  // Mark as busy before sending command
518  inc_busy_count();
519 
520  // Garrett (4/18/03): According to the Sony Newsgroups, a FlushCache is not needed if only 1 packet is sent.
521  // Only the additional packet needs the flush. I think this was helping the issue of non-64 byte aligned
522  // read buffers. We should try taking this out after the buffers are aligned.
523  FlushCache(WRITEBACK_DCACHE); // This one is needed, but I think invalidating the cache will work, instead
524 
525  CFileIOManager::sSendCommand(&read_packet, this, true);
526 
527  Dbg_Assert(((non_aligned_cache_start_bytes == 0) || (non_aligned_cache_end_bytes == 0)) || ((size * count) >= FILEIO_BUFFER_SIZE));
528 
529  if (m_blocking)
530  {
531  return WaitForIO();
532  } else {
533  return 0;
534  }
535 }
