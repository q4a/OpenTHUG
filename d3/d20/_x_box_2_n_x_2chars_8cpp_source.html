114 void SwizzleTexture( void *dstBuffer, void *srcBuffer, int width, int height, int32 depth, int32 stride )
115 {
116  int32 tilesX, tilesY;
117  int32 tilesSizeX, tilesSizeY;
118  int32 tileSize;
119 
120  generateSwizzleTable();
121 
122  if( width > height )
123  {
124  tilesX = width / height;
125  tilesY = 1;
126 
127  tilesSizeX = width / tilesX;
128  tilesSizeY = height;
129  }
130  else
131  {
132  tilesX = 1;
133  tilesY = height / width;
134 
135  tilesSizeX = width;
136  tilesSizeY = height / tilesY;
137  }
138 
139  tileSize = tilesSizeX * tilesSizeY;
140 
141  switch (depth)
142  {
143  case 4:
144  case 8:
145  {
146  int32 j;
147 
148  for (j = 0; j < tilesY; j++)
149  {
150  int32 i;
151 
152  for (i = 0; i < tilesX; i++)
153  {
154  int32 y;
155  uint8 *base;
156 
157  base = (uint8 *)(((uint8 *)dstBuffer) +
158  ((tileSize * tilesX) * j) +
159  (tileSize * i));
160 
161  for (y = 0; y < tilesSizeY; y++)
162  {
163  uint8 *srcPixel;
164  int32 x;
165 
166  srcPixel = (uint8 *)(((uint8 *)srcBuffer) +
167  (stride * (tilesSizeY * j)) +
168  (tilesSizeX * i) +
169  (stride * y));
170 
171  for (x = 0; x < tilesSizeX; x++)
172  {
173  uint8 *dstPixel;
174  dstPixel = (uint8 *)(base + TWIDDLE(x, y));
175  *dstPixel = *srcPixel;
176 
177  srcPixel++;
178  }
179  }
180  }
181  }
182  }
183  break;
184 
185  case 16:
186  {
187  int32 j;
188 
189  for (j = 0; j < tilesY; j++)
190  {
191  int32 i;
192 
193  for (i = 0; i < tilesX; i++)
194  {
195  int32 y;
196  uint8 *base;
197 
198  base = (uint8 *)(((uint16 *)dstBuffer) +
199  ((tileSize * tilesX) * j) +
200  (tileSize * i));
201 
202  for (y = 0; y < tilesSizeY; y++)
203  {
204  uint16 *srcPixel;
205  int32 x;
206 
207  srcPixel = (uint16 *)(((uint8 *)srcBuffer) +
208  (stride * (tilesSizeY * j)) +
209  (2 * tilesSizeX * i) +
210  (stride * y));
211 
212  for (x = 0; x < tilesSizeX; x++)
213  {
214  uint16 *dstPixel;
215  dstPixel = (uint16 *)(base + (TWIDDLE(x, y) << 1));
216  *dstPixel = *srcPixel;
217 
218  srcPixel++;
219  }
220  }
221  }
222  }
223  }
224  break;
225 
226  case 24:
227  case 32:
228  {
229  int32 j;
230 
231  for (j = 0; j < tilesY; j++)
232  {
233  int32 i;
234 
235  for (i = 0; i < tilesX; i++)
236  {
237  int32 y;
238  uint8 *base;
239 
240  base = (uint8 *)(((uint32 *)dstBuffer) +
241  ((tileSize * tilesX) * j) +
242  (tileSize * i));
243 
244  for (y = 0; y < tilesSizeY; y++)
245  {
246  uint32 *srcPixel;
247  int32 x;
248 
249  srcPixel = (uint32 *)(((uint8 *)srcBuffer) +
250  (stride * (tilesSizeY * j)) +
251  (4 * tilesSizeX * i) +
252  (stride * y));
253 
254  for (x = 0; x < tilesSizeX; x++)
255  {
256  uint32 *dstPixel;
257  dstPixel = (uint32 *)(base + (TWIDDLE(x, y) << 2));
258  *dstPixel = *srcPixel;
259 
260  srcPixel++;
261  }
262  }
263  }
264  }
265  }
266  break;
267 
268  default:
269  exit( 0 );
270  break;
271  }

291 SFont* LoadFont( const char *Filename, bool memory_resident )
292 {
293  SFont* pFont;
294  SChar* pChar;
295  uint8* pData;
296  void* p_FH;
297  int i,Len,NumChars,Width,Height,Depth,NumBytes;
298 
299  // Build the full filename.
300  char filename[128];
301 
302  if( !memory_resident )
303  {
304  strcpy( filename, "fonts/" );
305  strcat( filename, Filename );
306  strcat( filename, ".fnt.xbx" );
307  }
308 
309  // Open the font file.
310  if( !memory_resident )
311  p_FH = File::Open( filename, "rb" );
312 
313  // Allocate memory for the font structure.
314  pFont = new SFont();
315 
316  // Allocate a temporary buffer.
317  uint8 FontBuf[2048];
318 
319  // Load file header.
320  if( !memory_resident )
321  {
322  Len = File::Read( FontBuf, 16, 1, p_FH );
323  Dbg_MsgAssert( Len == 16, ( "couldn't read file header from font file %s", Filename ));
324  }
325  else
326  {
327  CopyMemory( FontBuf, Filename, 16 );
328  Filename += 16;
329  }
330 
331  NumChars = ((uint32 *)FontBuf)[1];
332  pFont->DefaultHeight = ((uint32 *)FontBuf)[2];
333  pFont->DefaultBase = ((uint32 *)FontBuf)[3];
334 
335  // Clear character map to zero.
336  memset( pFont->Map, 0, 256 );
337  memset( pFont->SpecialMap, 0, 32 );
338 
339  // Allocate memory for character table.
340  pFont->pChars = new SChar[NumChars];
341 
342  // Load character map and character table.
343  if( !memory_resident )
344  {
345  Len = File::Read( FontBuf, NumChars << 2, 1, p_FH );
346  Dbg_MsgAssert( Len == ( NumChars << 2 ), ( "couldn't read character table in font file %s", Filename ));
347  }
348  else
349  {
350  CopyMemory( FontBuf, Filename, NumChars << 2 );
351  Filename += NumChars << 2;
352  }
353 
354  for( i = 0, pChar = pFont->pChars, pData = FontBuf; i < NumChars; i++,pChar++,pData += 4 )
355  {
356  pChar->Baseline = ((uint16 *)pData)[0];
357  sint16 ascii_val = ((sint16 *)pData)[1];
358  if (ascii_val >= 0)
359  pFont->Map[(uint8) ascii_val] = i;
360  else
361  {
362  Dbg_Assert(ascii_val >= -32)
363  pFont->SpecialMap[(uint8) (-ascii_val - 1)] = i;
364  }
365  }
366 
367  // If there is a null character in the font, make characters that could not be found
368  // in the font display that instead of 'A'
369  if( pFont->SpecialMap[31] != 0 )
370  {
371  for( i = 0; i < 256; ++i )
372  {
373  if( pFont->Map[i] == 0 && i != 'A' && i != 'a')
374  pFont->Map[i] = pFont->SpecialMap[31];
375 
376  if( i < 31 && pFont->SpecialMap[i] == 0 )
377  pFont->SpecialMap[i] = pFont->SpecialMap[31];
378  }
379  }
380 
381  // Load texture header.
382  if( !memory_resident )
383  {
384  Len = File::Read( FontBuf, 16, 1, p_FH );
385  Dbg_MsgAssert( Len == 16, ( "couldn't read texture header from font file %s", Filename ));
386  }
387  else
388  {
389  CopyMemory( FontBuf, Filename, 16 );
390  Filename += 16;
391  }
392 
393  Width = ((uint16 *)FontBuf)[2];
394  Height = ((uint16 *)FontBuf)[3];
395  Depth = ((uint16 *)FontBuf)[4];
396 
397  // Create texture.
398  Dbg_Assert( Depth == 8 );
399  if( D3D_OK != D3DDevice_CreateTexture( Width, Height, 1, 0, D3DFMT_P8, 0, &pFont->pD3DTexture ))
400  {
401  Dbg_Assert( 0 );
402  return NULL;
403  }
404 
405  // Read texture bitmap data (into temp buffer so we can then swizzle it).
406  NumBytes = ( Width * Height + 3 ) & 0xFFFFFFFC;
407 
408  uint8* p_temp_texel_data = new uint8[NumBytes];
409  if( !memory_resident )
410  {
411  Len = File::Read( p_temp_texel_data, NumBytes, 1, p_FH );
412  Dbg_MsgAssert( Len == NumBytes, ( "Couldn't read texture bitmap from font file %s", Filename ));
413  }
414  else
415  {
416  CopyMemory( p_temp_texel_data, Filename, NumBytes );
417  Filename += NumBytes;
418  }
419 
420  // Lock the texture so we can swizzle into it directly.
421  D3DLOCKED_RECT locked_rect;
422  if( D3D_OK != pFont->pD3DTexture->LockRect( 0, &locked_rect, NULL, 0 ))
423  {
424  Dbg_Assert( 0 );
425  return NULL;
426  }
427 
428  // Swizzle the texture data.
429  SwizzleTexture( locked_rect.pBits, p_temp_texel_data, Width, Height, 8, Width );
430 
431  // No longer need this data.
432  delete[] p_temp_texel_data;
433 
434  // Create palette.
435  if( D3D_OK != D3DDevice_CreatePalette( D3DPALETTE_256, &pFont->pD3DPalette ))
436  {
437  Dbg_Assert( 0 );
438  return NULL;
439  }
440 
441  // Read clut bitmap data.
442  D3DCOLOR *p_clut;
443  pFont->pD3DPalette->Lock( &p_clut, 0 );
444 
445  if( !memory_resident )
446  {
447  Len = File::Read( p_clut, 1024, 1, p_FH );
448  Dbg_MsgAssert( Len == 1024, ( "couldn't read clut bitmap from font file %s", Filename ));
449  }
450  else
451  {
452  CopyMemory( p_clut, Filename, 1024 );
453  Filename += 1024;
454  }
455 
456  // Switch from RGBA to BGRA format palette.
457  for( i = 0; i < 256; ++i )
458  {
459  uint32 red = p_clut[i] & 0xFF;
460  uint32 blu = ( p_clut[i] >> 16 ) & 0xFF;
461 
462  // Double the alpha in the clut (currently limited to 0x80).
463  uint32 alpha = p_clut[i] >> 24;
464  alpha = ( alpha >= 0x80 ) ? 0xFF : ( alpha * 2 );
465  p_clut[i] = ( alpha << 24 ) | ( p_clut[i] & 0x0000FF00 ) | ( red << 16 ) | ( blu );
466  }
467 
468  // Skip numsubtextures, and load subtextures.
469  if( !memory_resident )
470  {
471  Len = File::Read( FontBuf, ( NumChars << 3 ) + 4, 1, p_FH );
472  Dbg_MsgAssert( Len == ( NumChars << 3 ) + 4, ( "couldn't read subtexture table from font file %s", Filename ));
473  }
474  else
475  {
476  CopyMemory( FontBuf, Filename, ( NumChars << 3 ) + 4 );
477  Filename += ( NumChars << 3 ) + 4;
478  }
479 
480  for( i = 0, pChar = pFont->pChars, pData = FontBuf + 4; i < NumChars; i++, pChar++, pData += 8 )
481  {
482  uint16 x = ((uint16 *)pData )[0];
483  uint16 y = ((uint16 *)pData )[1];
484  uint16 w = ((uint16 *)pData )[2];
485  uint16 h = ((uint16 *)pData )[3];
486 
487  pChar->w = w;
488  pChar->h = h;
489  pChar->u0 = (float)x / (float)Width;
490  pChar->v0 = (float)y / (float)Height;
491  pChar->u1 = pChar->u0 + ((float)w / (float)Width );
492  pChar->v1 = pChar->v0 + ((float)h / (float)Height );
493  }
494 
495  // Add font to font list.
496  pFont->pNext = pFontList;
497  pFontList = pFont;
498 
499  // We're done with the font file now.
500  if( !memory_resident )
501  File::Close( p_FH );
502 
503  // this will serve as the default spacing
504  pFont->mSpaceSpacing = pFont->pChars[pFont->Map['I']].w;
505 
506  return pFont;
507 }

587

589  bool got_char_tag = false; // tag resulting in output of character

591  {
592  pLetter++;
593  if (*pLetter == '\\')
594  break;
595 
596  switch(*pLetter)
597  {
598  case '\\':

600  break;

602  case 'C':

604  break;

606  case 'S':

611  got_char_tag = true;

613  }
614  case 'b':

640  }

642  Dbg_MsgAssert(0, ("unknown tag"));
643  break;
644  }
645  } // end while
646 
647  if (*pLetter == '\0') break;
648 
649  if (*pLetter!=' ' || pChar)
650  {
651  if (!pChar)
652  pChar = p_font->pChars + p_font->Map[(uint8)*pLetter];
653  x1 = x0 + pChar->w;
654  }

658  }
659

662  }
