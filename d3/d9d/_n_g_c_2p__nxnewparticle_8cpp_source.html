525 void CNgcNewParticle::update_position( void )
526 {
527  // Convert 3-point -> PVA format
528  float t1 = m_params.m_Lifetime * m_params.m_MidpointPct * 0.01f;
529  float t2 = m_params.m_Lifetime;
530  Mth::Vector u, a_;
531 
532  Mth::Vector x0 = m_params.m_BoxPos[0];
533  x0[3] = m_params.m_Radius[0];
534  Mth::Vector x1 = m_params.m_BoxPos[1];
535  x1[3] = m_params.m_Radius[1];
536  Mth::Vector x2 = m_params.m_BoxPos[2];
537  x2[3] = m_params.m_Radius[2];
538 
539  if( m_params.m_UseMidpoint )
540  {
541  u = ( t2 * t2 * ( x1 - x0 ) - t1 * t1 * ( x2 - x0 )) / ( t1 * t2 * ( t2 - t1 ));
542  a_ = ( t1 * ( x2 - x0 ) - t2 * ( x1 - x0 )) / ( t1 * t2 * ( t2 - t1 ));
543  }
544  else
545  {
546  u = ( x2 - x0 ) / t2;
547  a_.Set( 0, 0, 0, 0 );
548  }
549 
550  m_p0 = x0 - 1.5f * m_s0;
551  m_p1 = u - 1.5f * m_s1;
552  m_p2 = a_ - 1.5f * m_s2;
553  m_p0[3] = x0[3] - 1.5f * m_s0[3];
554  m_p1[3] = u[3] - 1.5f * m_s1[3];
555  m_p2[3] = a_[3] - 1.5f * m_s2[3];
556 }
