51 void CEventLog::AddEntry(uint32 type, uint32 target, uint32 source, uint32 script, uint32 receiverID, EOccurenceType occurenceType)
52 {
53  if (!mp_event_type_hash_table)
54  {
55  mp_event_type_hash_table = new Lst::HashTable<EventType>(8);
56  Script::CArray *p_event_type_array = Script::GetArray(CRCD(0x8114f90,"event_type_array"), Script::ASSERT);
57  for (int i = 0; i < (int) p_event_type_array->GetSize(); i++)
58  {
59  strcpy(m_event_type_table[i].mName, p_event_type_array->GetString(i));
60  uint32 name_crc = Script::GenerateCRC(p_event_type_array->GetString(i));
61  mp_event_type_hash_table->PutItem(name_crc, &m_event_type_table[i]);
62  }
63  }
64 
65  int last_entry = m_next_entry - 1;
66  if (last_entry < 0)
67  last_entry = MAX_LOG_ENTRIES - 1;
68 
69  if (occurenceType == vUPDATE && m_table[last_entry].m_occurence_type == vUPDATE)
70  {
71  // no need to keep piling on update entries, just increase tick count
72  m_table[last_entry].m_tick_count++;
73  }
74  else
75  {
76  if (occurenceType == vHANDLED)
77  m_event_depth--;
78 
79  Entry *p_entry = m_table + m_next_entry;
80  if (occurenceType == vUPDATE)
81  p_entry->m_tick_count = 0;
82  else
83  p_entry->m_type = type;
84  p_entry->m_target = target;
85  p_entry->m_source = source;
86  p_entry->m_script = script;
87  p_entry->m_receiver_id = receiverID;
88  p_entry->m_occurence_type = occurenceType;
89  p_entry->m_depth = m_event_depth;
90 
91  m_next_entry++;
92  if (m_next_entry == MAX_LOG_ENTRIES)
93  {
94  m_next_entry = 0;
95  m_wrapped = true;
96  }
97 
98  if (occurenceType == vLAUNCHED)
99  m_event_depth++;
100 
101  m_num_new_entries++;
102 
103  //PrintLatestEntry();
104  }

319 void CTracker::forward_event_to_listeners(CEvent *pEvent, CObject *pObject)
320 {

324  // The ref counting stuff will fire an assert if an event listener gets deleted

328  // prevents listener from being deleted (without an assert)

343 void CTracker::remove_aliases(CObject *pObject)
344 {
345  while(1)
346  {
347  //if (mp_alias_table->GetSize() == 0)
348  // break;
349 
350  mp_alias_table->IterateStart();
351  uint32 entry_key;
352  CObject *p_entry = mp_alias_table->IterateNext(&entry_key);
353  while(p_entry)
354  {
355  if (p_entry->GetID() == pObject->GetID())
356  {
357  // Found match. Remove it and repeat outer loop
358  p_entry->RemoveReference();
359  mp_alias_table->FlushItem(entry_key);

362  p_entry = mp_alias_table->IterateNext(&entry_key);
363  }
364 
365  if (!p_entry)
366  // no more matches left, we're done
367  break;
368  }

519 bool CTracker::LaunchEvent(uint32 type, uint32 target, uint32 source, Script::CStruct *pData, bool broadcast)
520 {
521 // printf("launch event, type=%s, target=0x%x, source= 0x%x, pData = %p\n", Script::FindChecksumName(type),target,source, pData);
522 
523 // Dbg_MsgAssert(!broadcast,("Don't use the broadcast flag yet!!!"));
524 
525  Dbg_MsgAssert(!m_block_event_launching, ("event launches are blocked"));
526  Dbg_MsgAssert(m_event_recurse_depth < vMAX_EVENT_RECURSE_DEPTH, ("too many nested LaunchEvents -- check your scripts"));
527  //printf("### launch event depth %d\n", m_event_recurse_depth);
528  m_event_recurse_depth++;
529 
530  // log event
531  m_event_log.AddEntry(type, target, source, m_next_event_script, 0, CEventLog::vLAUNCHED);
532  m_next_event_script = 0;
533 
534  CEvent event;
535  event.m_type = type;
536  event.m_target_id = target;
537  event.m_source_id = source;
538  event.mp_data = pData;
539 
540 
541 
542 
543 
544  if (broadcast)
545  {
546  #ifdef __SCRIPT_EVENT_TABLE__
547 
548  // broadcast to each script that has an event handler table
549  CEventReceiverList *p_event_list = mp_event_receiver_table->GetItem(type);
550  if (p_event_list)
551  {
552  int scripts = p_event_list->CountItems();
553  Dbg_MsgAssert(scripts, ("Empty list for event %s",Script::FindChecksumName(type)));
554  if (scripts == 1)
555  {
556  // Just one object, so do a quicker sending of the event
557  Lst::Node< Script::CScript >*p_node = p_event_list->FirstItem();
558  Script::CScript *p_script = p_node->GetData();
559  p_script->PassTargetedEvent(&event, broadcast);
560  }
561  else
562  {
563  // For multiple scripts, we need to make a copy of the the list of scripts
564  // that will receive this. Otherwise chaos ensues, as scripts change their event handlers based on events.
565  // We also use a list of smart pointers to the mp_reference in a script, so if one event handler deletes
566  // another object (and hence its script)
567  // then we can safely ignore that object (since it no longer exists)
568  // as the pointer will be set to NULL.
569 
570  Script::CScript ** pp_scripts = new Script::CScript*[scripts];
571  CSmtPtr<CRefCounted> * pp_refs = new CSmtPtr<CRefCounted>[scripts];
572 
573  Lst::Node< Script::CScript >*p_node = p_event_list->FirstItem();
574  int n=0;
575  while (p_node)
576  {
577  pp_scripts[n] = p_node->GetData();
578  CRefCounted * p_ref = &(p_node->GetData()->m_reference_counter);
579  pp_refs[n] = p_ref;
580 
581  n++;
582  p_node = p_node->GetNext();
583  }
584  for (n = 0; n<scripts; n++)
585  {
586  if (pp_refs[n]) // Smart pointer to script reference counter might have been deleted by some other object
587  {
588  pp_scripts[n]->PassTargetedEvent(&event, broadcast);
589  }
590  }
591  delete [] pp_refs;
592  delete [] pp_scripts;
593 
594 
595  }
596 
597  }
598 
599 
600  #else
601  // broadcast to each object that has an event handler table
602  CEventReceiverList *p_event_list = mp_event_receiver_table->GetItem(type);
603  if (p_event_list)
604  {
605  int objects = p_event_list->CountItems();
606  Dbg_MsgAssert(objects, ("Empty list for event %s",Script::FindChecksumName(type)));
607  if (objects == 1)
608  {
609  // Just one object, so do a quicker sending of the event
610  Lst::Node< Obj::CObject >*p_node = p_event_list->FirstItem();
611  Obj::CObject *p_obj = p_node->GetData();
612  p_obj->PassTargetedEvent(&event, broadcast);
613  }
614  else
615  {
616  // For multiple objects, we need to make a copy of the the list of objects
617  // that will receive this. Otherwise chaos ensues, as objects change their event handlers based on events.
618  // We use a list of smart pointers, so if one object handler deletes another object
619  // then we can safely ignore that object (since it no longer exists)
620  // as the pointer will be set to NULL.
621  CObjectPtr * pp_objects = new CObjectPtr[objects];
622  Lst::Node< Obj::CObject >*p_node = p_event_list->FirstItem();
623  int n=0;
624  while (p_node)
625  {
626  pp_objects[n++] = p_node->GetData();
627  p_node = p_node->GetNext();
628  }
629  for (n = 0; n<objects; n++)
630  {
631  if (pp_objects[n]) // Smart pointer might have been deleted by some other object
632  {
633  pp_objects[n]->PassTargetedEvent(&event, broadcast);
634  }
635  }
636  delete [] pp_objects;
637 
638 
639  }
640 
641  }
642  #endif
643  }
644  else
645  {
646 
647 
648  // try event on all registered listeners
649  forward_event_to_listeners(&event, NULL);
650 
651  // if event has a specific target, then send to that CObject
652  if (event.m_target_id != CEvent::vSYSTEM_EVENT && !event.WasHandled())
653  {
654  // look for a object with the target id
655  CObject *p_object = GetObject(event.m_target_id);
656  if (!p_object)
657  {
658  // or look for a script with the target id
659  Script::CScript *p_script = Script::FindSpawnedScriptWithID(event.m_target_id);
660 
661  if (!p_script)
662  {
663  m_event_log.Print(256);
664  #ifdef __NOPT_ASSERT__
665  mp_hash_table->PrintContents();
666  #endif
667  }
668 
669  Dbg_MsgAssert(p_script, ("couldn't find object or script id %s", Script::FindChecksumName(event.m_target_id)));
670 
671  p_script->PassTargetedEvent(&event);
672  }
673  else
674  {
675  p_object->PassTargetedEvent(&event);
676  }
677  // p_object may have been deleted
678  }
679  }
680 
681  /*
682  See if any scripts were waiting on the event.
