31 // returns the scale that needs to be applied to (modelWorldW,modelWorldH) to make it fit (screenW,screenH)
32 static float sGetScaleFromScreenDimensions(float screenW, float screenH, float zOffset, float modelWorldW, float modelWorldH)
33 {
34  // Inverse project.
35  float desired_world_w = screenW * zOffset / (640.0f * (1.0f + screenW / 1280.0f));
36  float desired_world_h = screenH * zOffset / (448.0f * (1.0f + screenW / 896.0f));
37 
38  float scale_x = desired_world_w / modelWorldW;
39  float scale_y = desired_world_h / modelWorldH;
40  return (-scale_x < -scale_y) ? -scale_x : -scale_y;
41 }

48 // screenX, screenY indicate where the origin of the model will be located
49 static void sGetWorldMatrixFromScreenPosition(int camera_num, Mth::Matrix *p_world_matrix, float screenX, float screenY, float zOffset, Mth::Vector &model_offs)
50 {
51  Dbg_MsgAssert(p_world_matrix,("NULL p_world_matrix"));
52  Dbg_MsgAssert((camera_num >= 0) && (camera_num < Nx::CViewportManager::vMAX_NUM_ACTIVE_VIEWPORTS), ("Camera number is out of range: %d", camera_num));
53 
54  Mth::Vector camera_space_pos;
55 
56 #ifdef __PLAT_XBOX__
57  // Don't think Mick's assumption below is correct - for 3D objects going through the rendering pipeline, no screen correction is required.
58  screenX = screenX - ( 640 / 2 );
59  screenY = screenY - ( 448 / 2 );
60 
61  // Adjust X for aspect ratio that is different than 4/3 (1.3333333)
62  screenX *= (Nx::CViewportManager::sGetScreenAspect() / 1.3333333f);
63 
64  // Also, adjust for 720p (horrible hack).
65  if( NxXbox::EngineGlobals.backbuffer_width > 640 )
66  {
67  screenX *= 0.9f;
68  screenY *= 0.9f;
69  }
70 
71  camera_space_pos.Set( -screenX * zOffset / 448.0f, screenY * zOffset / 448.0f, zOffset );
72 #else
73  // Mick: Need to convert from logical to physical screen coordinates
74  screenX = SCREEN_CONV_X(screenX);
75  screenY = SCREEN_CONV_Y(screenY);
76 
77  // Make ScreenX and ScreenY relative to the centre of the screen.
78  screenX -= SCREEN_CONV_X(640)/2;
79  screenY -= SCREEN_CONV_Y(448)/2;
80 
81  // Adjust X for aspect ratio that is different than 4/3 (1.3333333)
82  screenX *= (Nx::CViewportManager::sGetScreenAspect() / 1.3333333f);
83 
84  // Inverse project.
85  camera_space_pos.Set(-screenX * zOffset / SCREEN_CONV_X(448),
86  screenY * zOffset / SCREEN_CONV_Y(448),
87  zOffset);
88 #endif // __PLAT_XBOX__
89 
90  camera_space_pos += model_offs;
91 
92  // We now have a 3d vector in camera space, so rotate by the camera matrix
93  // to get world coords.
94 
95  p_world_matrix->Ident();
96 
97  // Get the camera.
98  Gfx::Camera *p_camera = Nx::CViewportManager::sGetActiveCamera(camera_num);
99  if (p_camera)
100  {
101  Mth::Matrix cam_matrix=p_camera->GetMatrix();
102 
103  // Camera matrix might have been incorrectly set up with a translation in W
104  // so clear it out to be safe
105  // Probably should be handled at a higher level.
106  cam_matrix[W].Set();
107 
108  // Apply the camera matrix to the camera_space_pos
109  Mth::Vector world_pos = camera_space_pos * cam_matrix;
110  // and add the camera position to get the final world position.
111  Mth::Vector cam_pos = p_camera->GetPos();
112  cam_pos[W] = 0.0f;
113  world_pos += cam_pos;
114  world_pos[W] = 0.0f;
115 
116 
117  // Make the rotation part of the passed matrix be the camera matrix so that the model
118  // always faces the camera.
119  *p_world_matrix*=cam_matrix;
120  // Then add in the calculated world position.
121  p_world_matrix->Translate(world_pos);
122  }
123 }

632 void CElement3d::update()
633 {
634  if (m_waiting_to_die)
635  {
636  // Return straight away to prevent repeated spawnings of the KillElement3d script
637  return;
638  }
639 
640  if (m_num_models)
641  {
642  if (m_parent_object_name)
643  {
644  // Get a pointer to the parent object.
645  Obj::CMovingObject *p_pos_obj=NULL;
646 
647 // Skate module
648 // if (m_parent_object_name==0x5b8ab877/*Skater*/)
649 // {
650 // // This is
651 // Mdl::Skate * skate_mod = Mdl::Skate::Instance();
652 // p_pos_obj=skate_mod->GetSkater(m_skater_number);
653 // }
654 // else
655  {
656 // p_pos_obj=Obj::CMovingObject::m_hash_table.GetItem(m_parent_object_name);
657  p_pos_obj = (Obj::CMovingObject*) Obj::CCompositeObjectManager::Instance()->GetObjectByID(m_parent_object_name);
658  }
659 
660  if (p_pos_obj)
661  {
662  m_model_matrix.Ident();
663  m_model_matrix.SetPos(p_pos_obj->m_pos+m_parent_offset*m_model_matrix);
664  }
665  else
666  {
667  // The parent object has disappeared!
668  // So call the KillElement3d script to safely delete this arrow.
669 
670  // Tried various ways of killing directly from c-code, but they didn't work.
671  // Instant deletion here will mess up the update loop, and calling
672  // Die() to flag the element for deletion later also cause problems, because
673  // the parent element needs to be unlocked at the time of deletion.
674 
675  // Running a script seems to be safe ...
676  // The KillElement3d script will wait a gameframe, then call the usual Die
677  Script::CScript *p_script=Script::SpawnScript(0xa66c1c31/*KillElement3d*/);
678  #ifdef __NOPT_ASSERT__
679  p_script->SetCommentString("Created by CElement3d::update()");
680  #endif
681  p_script->mpObject=this;
682  m_waiting_to_die=true;
683  }
684  }
685  else if (m_parent_node_name)
686  {
687  // FindNamedNode will assert if it does not find the node.
688  int node_index=SkateScript::FindNamedNode(m_parent_node_name);
689  Script::CStruct *p_node=SkateScript::GetNode(node_index);
690  Dbg_MsgAssert(p_node,("NULL p_node"));
691 
692  Mth::Vector pos;
693  SkateScript::GetPosition(p_node,&pos);
694 
695  m_model_matrix.Ident();
696  m_model_matrix.SetPos( pos + m_parent_offset * m_model_matrix );
697  // m_model_matrix.SetPos(pos);
698  }
699  else
700  {
701  Mth::Vector pos_change(m_offset_of_origin_from_center[X] * m_scale3d,
702  m_offset_of_origin_from_center[Y] * m_scale3d,
703  m_offset_of_origin_from_center[Z] * m_scale3d);
704  /*
705  Mth::Vector pos_change(m_offset_of_origin_from_center[X] * m_summed_props.scalex * m_scale3d,
706  m_offset_of_origin_from_center[Y] * m_summed_props.scaley * m_scale3d,
707  m_offset_of_origin_from_center[Z] * m_summed_props.scalex * m_scale3d);
708  */
709  sGetWorldMatrixFromScreenPosition(m_active_viewport_number, &m_model_matrix,
710  m_summed_props.GetScreenUpperLeftX() + GetAbsW() / 2.0f,
711  m_summed_props.GetScreenUpperLeftY() + GetAbsH() / 2.0f,
712  m_camera_z,
713  pos_change);
714 
715  // If the model were rendered using m_model_matrix as it is then it will be unscaled
716  // and facing the camera.
717 
718  // If the object is required to point to somewhere in the world then change the rotation
719  // the be the required one.
720  Mth::Vector position_to_point_to;
721  switch (m_point_type)
722  {
723  case POINT_TYPE_NODE:
724  position_to_point_to=m_node_position_to_point_to;
725  break;
726  case POINT_TYPE_OBJECT:
727  {
728  // Get a pointer to the object.
729  // Doing a naughty cast up to a CMovingObject, because it probably will be one, and
730  // won't cause a crash if it isn't, it'll just get a bad position.
731  Obj::CMovingObject *p_pos_obj=(Obj::CMovingObject*)Obj::ResolveToObject(m_id_of_object_to_point_to);
732  if (p_pos_obj)
733  {
734  position_to_point_to=p_pos_obj->m_pos;
735  }
736  break;
737  }
738  default:
739  break;
740  }
741 
742  if (m_point_type!=POINT_TYPE_NONE)
743  {
744  // Save the position
745  Mth::Vector model_pos=m_model_matrix.GetPos();
746  // Clear out the rotation calculated by sGetWorldMatrixFromScreenPosition
747  m_model_matrix.Ident();
748 
749  Mth::Vector dir=position_to_point_to-model_pos;
750 
751  dir.Normalize();
752  m_model_matrix[Z]=dir;
753 
754  m_model_matrix[X].Set(dir.GetZ(),0.0f,-dir.GetX());
755  m_model_matrix[X].Normalize();
756 
757  m_model_matrix.OrthoNormalizeAbout(Z);
758 
759  // Tilt so that the arrow is not end on to the camera.
760  // However, make the tilt drop linearly down to zero as the
761  // arrow goes towards being vertical so that the arrow never
762  // ends up pointing backwards.
763  float r=m_tilt-(-m_model_matrix[Z].GetY()*m_tilt);
764  m_model_matrix.RotateXLocal(Mth::DegToRad(r));
765 
766  // Put the position back in.
767  m_model_matrix.Translate(model_pos);
768  }
769  }
770 
771  // Now create a rotation matrix containing any extra rotation that we want.
772  Mth::Matrix rot_matrix;
773  rot_matrix.Ident();
774 
775  rot_matrix.Translate(-m_pivot_point);
776 
777  // Update the angles using the angular velocities.
778  m_angle_x+=m_angvel_x;
779  m_angle_y+=m_angvel_y;
780  m_angle_z+=m_angvel_z;
781 
782  // Reduce modulo 2pi so that the calculations don't go weird after a few days
783  // due to the angles getting huge.
784  m_angle_x=m_angle_x-((int)(m_angle_x/(2.0f*3.141592654f)))*(2.0f*3.141592654f);
785  m_angle_y=m_angle_y-((int)(m_angle_y/(2.0f*3.141592654f)))*(2.0f*3.141592654f);
786  m_angle_z=m_angle_z-((int)(m_angle_z/(2.0f*3.141592654f)))*(2.0f*3.141592654f);
787 
788  //rot_matrix.RotateY(m_angle_y);
789  //rot_matrix.RotateX(m_angle_x);
790  //rot_matrix.RotateZ(m_angle_z);
791  s_apply_rotations(rot_matrix, m_angle_x, m_angle_y, m_angle_z, m_angle_order);
792 
793  rot_matrix.Translate(m_pivot_point);
794 
795  // Apply scaling.
796  //Ryan("oogbat %.4f, %.4f, %.4f\n", m_summed_props.scalex, m_scale3d, m_summed_props.scalex * m_scale3d);
797  Mth::Vector scale(m_summed_props.GetScaleX() * m_scale3d,
798  m_summed_props.GetScaleY() * m_scale3d,
799  m_summed_props.GetScaleX() * m_scale3d,
800  1.0f);
801 
802  // The ped arrows (or stars) need to scale according to the distance from the camera so that
803  // they are visible from a distance. (TT2006)
804  if (m_scale_with_distance)
805  {
806  Gfx::Camera *p_camera = Nx::CViewportManager::sGetActiveCamera(m_active_viewport_number);
807  if (p_camera)
808  {
809  Mth::Vector cam_offset=m_model_matrix.GetPos()-p_camera->GetPos();
810  float multiplier=cam_offset.Length()*m_scale_multiplier;
811  if (multiplier < 1.0f)
812  {
813  multiplier=1.0f;
814  }
815  if (multiplier > m_max_scale)
816  {
817  multiplier = m_max_scale;
818  }
819 
820  scale[X]*=multiplier;
821  scale[Y]*=multiplier;
822  scale[Z]*=multiplier;
823  }
824  }
825 
826  rot_matrix.Scale(scale);
827 
828  // Apply the rotation and scaling to the m_model_matrix
829  m_model_matrix=rot_matrix*m_model_matrix;
830 
831 
832  // Calculate any hover offset that may need to be applied.
833  Mth::Vector hover_offset;
834  if (m_hover_amp>0.0f)
835  {
836  int t=Tmr::ElapsedTime(0)%m_hover_period;
837  float h=m_hover_amp*sinf(t*2*3.141592653f/m_hover_period);
838  // Use m_model_matrix[Y] so that it hovers up and down in the model's
839  // coordinate system.
840  hover_offset=h*m_model_matrix[Y];
841  }
842 
843  // Render the models
844  bool show_models = (m_summed_props.alpha >= .0001f);
845 
846  // If running a replay, hide arrows that are pointing to something,
847  // otherwise they go all spazzy. (TT8083 and TT8819)
848  if ((m_point_type!=POINT_TYPE_NONE || m_parent_object_name) && Replay::RunningReplay())
849  {
850  show_models=false;
851  }
852 
853 
854  for (int i=0; i<m_num_models; ++i)
855  {
856  Dbg_MsgAssert(mp_models[i].mpModel,("NULL mp_models[i].mpModel"));
857 
858  Mth::Vector p=mp_models[i].mOffset*m_model_matrix;
859  Mth::Matrix display_matrix=m_model_matrix;
860 
861  if (m_hover_amp>0.0f)
862  {
863  p+=hover_offset;
864  }
865 
866  display_matrix.SetPos(p);
867 
868  mp_models[i].mpModel->SetActive(show_models);
869  mp_models[i].mpModel->Render(&display_matrix,false,mp_models[i].mpSkeleton);
870  }
871  }
