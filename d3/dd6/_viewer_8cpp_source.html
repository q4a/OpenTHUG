99 {
100  Dbg_AssertType ( &handler, Inp::Handler< CViewer > );
101 
102  CViewer& mdl = handler.GetData();
103 
104  mdl.m_shift_commands.ClearAll();
105 
106  if( handler.m_Input->m_Buttons & Inp::Data::mD_SELECT )
107  {
108  mdl.m_shift_commands.SetMask( handler.m_Input->m_Makes );
109  if (Script::GetInt(CRCD(0xf3e055e1,"select_shift")))
110  {
111  // only mask the buttons if we are using select as a shift button

127 
128 // printf ("s_view_mode = %d rightx = %d\n",s_view_mode,mdl.m_right[X]);
129 
130 // just return if we are not in viewer mode
131  if (!s_view_mode)
132  {
133  return;
134  }
135 
136  // GJ: no longer clearing the last frame's data
137  // at the beginning of the loop... this allows
138  // the viewer object to add additional m_commands
139  // to the viewer (useful for customizing controller
140  // configs in different viewer modes)
141 // mdl.m_commands.ClearAll();
142 
143  // translate m_Input data to module-specific commands
144  mdl.m_right[X] = handler.m_Input->m_Event[Inp::Data::vA_RIGHT_X_UNCLAMPED] - 128;
145  mdl.m_right[Y] = handler.m_Input->m_Event[Inp::Data::vA_RIGHT_Y_UNCLAMPED] - 128;
146  mdl.m_left[X] = handler.m_Input->m_Event[Inp::Data::vA_LEFT_X_UNCLAMPED] - 128;
147  mdl.m_left[Y] = handler.m_Input->m_Event[Inp::Data::vA_LEFT_Y_UNCLAMPED] - 128;

149

151  {
152  if( ( handler.m_Input->m_Buttons & Inp::Data::mD_R1 ) &&
153  ( handler.m_Input->m_Makes & Inp::Data::mD_R2 ))

155  if (!Config::CD())
156  {
157  // Mick - took out viewport changing for non-PS2, as it messes up X-Box artists, with only one shoulder button

164  {
165  if( !( handler.m_Input->m_Buttons & Inp::Data::mD_R2 ))

171  {
172  mdl.m_commands.SetMask( mSTRAFE_DOWN );
173  }
174 
175  // if (!Config::CD())

177  if( handler.m_Input->m_Makes & Inp::Data::mD_START )
178  {
179  mdl.m_commands.SetMask( mRESET );
180  }
181  if( mdl.m_rotation_mode == vROTATE_CAMERA )

279 void CViewer::s_logic_code ( const Tsk::Task< CViewer >& task )
280 {
281  CViewer& mdl = task.GetData();
282 
283  Dbg_AssertType ( &task, Tsk::Task< CViewer > );
284 
285  //static const float ldelta = 1.0f;
286  //static const float lcolordelta = .0125f;
287  static const float scale = ( 1.0f / ( 128.0f - Inp::vANALOGUE_TOL ));
288 
289 // Gfx::Camera* pCam = Nx::CViewportManager::sGetCamera( 0 );
290 // if ( !pCam )
291 // {
292 // return;
293 // }
294 
295  float elapsed_seconds;
296  Tmr::Time elapsed_time;
297  static Tmr::Time last_time = 0;
298 
299  // store the old camera pos before updating anything...
300 // pCam->StoreOldPos( );
301 
302  elapsed_time = Tmr::ElapsedTime( last_time );
303  elapsed_seconds = (int)elapsed_time / 1000.0f;
304  last_time = last_time + elapsed_time;
305 
306  if ( mdl.m_commands.TestMask( mTOGGLE_ROTATE_MODE ))
307  {
308  // Mick: Don't switch to rotating the camera around the object unless there is an object
309  if ( mdl.mp_viewerObject && mdl.m_rotation_mode == vROTATE_CAMERA )
310  {
311  mdl.m_rotation_mode = vROTATE_OBJECTS;
312  }
313  else
314  {
315  mdl.m_rotation_mode = vROTATE_CAMERA;
316  }
317  }
318 
319  mdl.m_right[Y] =
320  ( mdl.m_right[Y] > Inp::vANALOGUE_TOL ) ? ( mdl.m_right[Y] - Inp::vANALOGUE_TOL ) * scale :
321  ( mdl.m_right[Y] < -Inp::vANALOGUE_TOL ) ? ( mdl.m_right[Y] + Inp::vANALOGUE_TOL ) * scale : 0.0f;
322  mdl.m_right[X] =
323  ( mdl.m_right[X] > Inp::vANALOGUE_TOL ) ? ( mdl.m_right[X] - Inp::vANALOGUE_TOL ) * scale :
324  ( mdl.m_right[X] < -Inp::vANALOGUE_TOL ) ? ( mdl.m_right[X] + Inp::vANALOGUE_TOL ) * scale : 0.0f;
325 
326  mdl.m_left[Y] =
327  ( mdl.m_left[Y] > Inp::vANALOGUE_TOL ) ? ( mdl.m_left[Y] - Inp::vANALOGUE_TOL ) * scale :
328  ( mdl.m_left[Y] < -Inp::vANALOGUE_TOL ) ? ( mdl.m_left[Y] + Inp::vANALOGUE_TOL ) * scale : 0.0f;
329  mdl.m_left[X] =
330  ( mdl.m_left[X] > Inp::vANALOGUE_TOL ) ? ( mdl.m_left[X] - Inp::vANALOGUE_TOL ) * scale :
331  ( mdl.m_left[X] < -Inp::vANALOGUE_TOL ) ? ( mdl.m_left[X] + Inp::vANALOGUE_TOL ) * scale : 0.0f;
332 
333  Obj::CCompositeObject * p_obj = (Obj::CCompositeObject *) Obj::CCompositeObjectManager::Instance()->GetObjectByID(CRCD(0xeb17151b,"viewer_cam"));
334  if (p_obj)
335  {
336  if ( mdl.m_rotation_mode == vROTATE_CAMERA )
337  {
338  if (mdl.m_left[Y])
339  {
340  p_obj->GetMatrix().RotateLocal( Mth::Vector( 1, 0, 0 ),
341  mdl.m_left[Y] * elapsed_seconds * mdl.s_rotate_vel * Mth::PI / 180.0f );
342  }
343  if (mdl.m_left[X])
344  {
345  // this rotation is *NOT* local, as the designers seem to prefer it this way.
346  Mth::Matrix mat = p_obj->GetMatrix();
347  mat.RotateY( -mdl.m_left[X] * elapsed_seconds * mdl.s_rotate_vel * Mth::PI / 180.0f );
348  p_obj->SetMatrix( mat );
349  }
350  }
351  else if ( mdl.m_rotation_mode == vROTATE_OBJECTS )
352  {
353  Dbg_Assert( mdl.mp_viewerObject );
354  mdl.mp_viewerObject->SetRotation( mdl.m_left[X], mdl.m_left[Y] );
355  }
356 
357 
358 
359  // Move with right analog stick:
360  if (mdl.m_right[X] || mdl.m_right[Y])
361  {
362  Mth::Vector transVec;
363  transVec[X] = mdl.m_right[X] * ( elapsed_seconds * mdl.s_movement_vel );
364  transVec[Y] = 0.0f;
365  transVec[Z] = mdl.m_right[Y] * ( elapsed_seconds * mdl.s_movement_vel );
366  transVec[W] = 1.0f;
367  s_translate_local( p_obj, transVec );
368  }
369  }
370 
371  if ( mdl.m_commands.TestMask( mSTRAFE_UP ))
372  {
373  s_translate_local( p_obj, Mth::Vector( 0, elapsed_seconds * mdl.s_movement_vel, 0 ) );
374  }
375  if ( mdl.m_commands.TestMask( mSTRAFE_DOWN ))
376  {
377  s_translate_local( p_obj, Mth::Vector( 0, -elapsed_seconds * mdl.s_movement_vel, 0 ) );
378  }
379 
380  if ( !Script::GetInt( CRCD(0xd8c91578,"DisableViewerRoll"), false ) )
381  {
382  if ( mdl.m_commands.TestMask( mROLL_LEFT ))
383  {
384  p_obj->GetMatrix().RotateLocal( Mth::Vector( 0, 0, 1 ),
385  -0.5f * elapsed_seconds * mdl.s_rotate_vel * Mth::PI / 180.0f );
386  }
387  if ( mdl.m_commands.TestMask( mROLL_RIGHT ))
388  {
389  p_obj->GetMatrix().RotateLocal( Mth::Vector( 0, 0, 1 ),
390  0.5f * elapsed_seconds * mdl.s_rotate_vel * Mth::PI / 180.0f );
391  }
392  }
393 
394  p_obj->Pause(false); // dang, it's suspended due to game being paused, so need to update manually...
395  // THIS IS A PATCH - needs to be more explicity handled.....
396 
397 
398  if ( mdl.m_commands.TestMask( mCHANGE_SCREEN_MODE ))
399  {
400  Nx::CViewportManager::sSetScreenMode ( (Nx::ScreenMode) (( (int) Nx::CViewportManager::sGetScreenMode() + 1 ) % Nx::vNUM_SCREEN_MODES) );
401  }
402 
403  // GJ: clear the frame's data, now that we're done with it
404  // (this used to be done at the beginning of the loop)
405  mdl.m_commands.ClearAll();
406 }

413 void CViewer::s_translate_local( Obj::CCompositeObject* p_obj, const Mth::Vector& vec )
414 {
415  // need to build a temporary matrix,
416  // because the rotation and translation
417  // are stored separately in the camera
418  Mth::Matrix mat = p_obj->GetMatrix();
419  mat[Mth::POS] = p_obj->GetPos();
420 
421  mat.TranslateLocal( vec );
422 
423  p_obj->SetPos(mat[Mth::POS]);
424 }
