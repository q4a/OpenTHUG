135 {
136  // GJ: toggle field, but only the first time
137  // (otherwise, the double-buffer gets messed
138  // up if SetBoneTransforms() is called more
139  // than once per frame...)
140  uint8 thisFrame = ( render::Frame & 0xff );
141  if ( m_lastFrame != thisFrame )
142  {
143  m_field = !m_field;
144  m_lastFrame = thisFrame;
145  }
146 
147  int startIndex = m_field ? 0 : m_num_bones;
148 
149  Mth::Matrix* pSrc = pMat;
150  Mth::Matrix* pDst = mp_bone_transforms + startIndex;
151 
152  // copy the given matrices into the correct half of the matrix double-buffer
153  for ( int i = 0; i < m_num_bones; i++ )
154  {
