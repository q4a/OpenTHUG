135 void sVerletConstraint::Apply( sVerletParticle *p_particles )
136 {
137  sVerletParticle* p_p0 = p_particles + m_particle0;
138  sVerletParticle* p_p1 = p_particles + m_particle1;
139 
140  Mth::Vector delta = p_p1->m_pos - p_p0->m_pos;
141  float delta_length = delta.Length();
142  float diff = ( delta_length - m_constraint_dist ) / delta_length;
143 
144  if( m_particle0 == 0 )
145  {
146  // Move the points.
147  p_p0->m_pos += delta * 0.0f * diff;
148  p_p1->m_pos -= delta * 1.0f * diff;
149  }
150  else
151  {
152  // Move the points.
153  p_p0->m_pos += delta * 0.5f * diff;
154  p_p1->m_pos -= delta * 0.5f * diff;
