163 /*
164  Destroying CObject disconnects it from manager, stops all scripts running on it.
165 */
166 CObject::~CObject()
167 {
168 
169  #ifdef __NOPT_ASSERT__
170  if (GetLockAssert())
171  {
172  if (mp_script)
173  {
174  Dbg_MsgAssert(0,("\n%s\n Object %s at %p has been deleted while updating a script running on it, maybe goal manager callback? See Mick.",mp_script->GetScriptInfo(),Script::FindChecksumName(GetID()),this));
175  }
176  Dbg_MsgAssert(0,("Object %s at %p has been deleted while updating a script running on it, maybe goal manager callback? See Mick.",Script::FindChecksumName(GetID()),this));
177  }
178  #endif
179 
180  Dbg_MsgAssert(!IsLocked(), ("can't destroy CObject, is locked"));
181 
182  if (mp_tags)
183  delete mp_tags;
184 
185 #ifndef __SCRIPT_EVENT_TABLE__
186  if (mp_event_handler_table)
187  {
188  // Remove the references to the event handler from the tracker
189  mp_event_handler_table->unregister_all(this);
190 
191  if (mp_event_handler_table->m_in_immediate_use_counter)
192  {
193  // table still in use by its pass_event() function,
194  // so leave it up to that function to destroy the table
195  mp_event_handler_table->m_valid = false;
196  }
197  else
198  {
199  delete mp_event_handler_table;
200  }
201  }
202 #endif
203 
204  if (mp_manager)
205  {
206  mp_manager->UnregisterObject( *this );
207  }
208 
209  if (mp_script)
210  {
211  delete mp_script;
212  mp_script = NULL;
213  }
214 
215  // Stop any scripts that have this object as their parent.
216  // Note: This won't actually delete the scripts, to prevent dangling pointers.
217  Script::StopAllScriptsUsingThisObject(this);
218 }

559 /******************************************************************/
560 /* CObject::CallMemberFunction */
561 /* Call a member function, based on a checksum */
562 /* This is usually the checksum of the name of the function */
563 /* but can actually be any number, as it just uses a switch */
564 /* note this is a virtual function, so the same checksum */
565 /* can do differnet things for different objects */
566 /* */
567 /* */
568 /******************************************************************/
569 
570 bool CObject::CallMemberFunction( uint32 Checksum, Script::CScriptStructure *pParams, Script::CScript *pScript )
571 {
572 
573  Dbg_MsgAssert(pScript,("NULL pScript"));
574 
575  switch(Checksum)
576  {
577  case CRCC(0x3611c136, "GetTags"):
578  CopyTagsToScriptStruct(pScript->GetParams());
579  break;
580 
581  case CRCC( 0xa58079eb, "SetTags" ):
582  SetTagsFromScript( pParams );
583  break;
584 
585  case 0xc6870028:// "Die"
586  MarkAsDead( );
587  break;
588 
589  case 0xb3c262ec:// "DisassociateFromObject"
590  pScript->DisassociateWithObject(this);
591  break;
592 
593  // @script | Obj_GetId |
594  case 0x500eb224: // Obj_GetId
595  Dbg_Assert( pScript && pScript->GetParams() );
596  pScript->GetParams()->AddChecksum( "objId", m_id );
597  break;
598 
599  // @script | Obj_FlagSet | Check to see if a flag has been set. Flags can
600  // be defined anywhere, but you should keep them in your personal scripts file.
601  // It is important that you prefix all flags with your initials to ensure that
602  // there are no conflicts with the other designers.
603  // <nl>Example 1:
604  // <nl>MJD_LAMP_GOT_BROKEN = 14
605  // <nl>MJD_LAMP_GOT_HIT = 15
606  // <nl>MJD_LAMP_GOT_DISABLED = 16
607  // <nl>if Obj_FlagSet 15
608  // <nl> Obj_ClearFlag MJD_LAMP_GOT_HIT
609  // <nl> WiggleProfusely
610  // <nl>endif
611  // <nl>Example 2:
612  // <nl>if Obj_FlagSet MJD_LAMP_GOT_HIT clear
613  // <nl> WiggleProfusely
614  // <nl>endif
615  // <nl>Example 3:
616  // <nl>If Obj_FlagSet JKU_FLAG_PED_START // JKU_FLAG_PED_START would be defined in JKU_Scripts.q
617  // <nl> Printf Yes
618  // <nl>endif
619  // @flag all | Clear all flags
620  // @flag clear | Clear the flag
621  // @flag reset | I just added a 'reset' flag you can send in to clear
622  // the flag after checking it, if it's set
623  case 0x4babc987: // Obj_FlagSet
624  {
625  uint32 scriptFlags = mScriptFlags;
626  uint32 Flags = 0;
627  Flags = GetFlags( pParams, pScript );
628  if ( !Flags )
629  {
630  Dbg_MsgAssert( 0,( "\n%s\nObj_FlagSet command requires a flag to be specified.\n(Either an integer or a name defined to be an integer)",pScript->GetScriptInfo()));
631  return ( false );
632  }
633  if ( pParams->ContainsFlag( 0x1a4e0ef9 ) ) // clear
634  {
635  mScriptFlags &= ~( Flags );
636  }
637  if ( scriptFlags & Flags )
638  {
639  return true;
640  }
641  else
642  {
643  return false;
644  }
645  break;
646  }
647 
648  // @script | Obj_FlagNotSet | Check to see if a flag has not been set.
649  case 0x53ebee03: // Obj_FlagNotSet
650  {
651  uint32 Flags = 0;
652  Flags = GetFlags( pParams, pScript );
653  if ( !Flags )
654  {
655  Dbg_MsgAssert( 0,( "\n%s\nObj_FlagNotSet command requires a flag to be specified.\n(Either an integer or a name defined to be an integer)",pScript->GetScriptInfo()));
656  return ( false );
657  }
658  if ( mScriptFlags & ( Flags ) )
659  {
660  return false;
661  }
662  else
663  {
664  return true;
665  }
666  break;
667  }
668 
669  // @script | Obj_ClearFlag | Object member function which clears the specified flag or flags.
670  // <nl>There are 3 ways of using it, it can clear just one flag, or an array of flags,
671  // or all the flags.
672  // <nl>Example 1:
673  // <nl>Obj_ClearFlag JKU_FLAG_PED_START // Set the flag to 0
674  // <nl>Example 2:
675  // <nl>Obj_ClearFlag [ JKU_FLAG_PED_START JKU_FLAG_PED_STOP ] // Clears an array of flags
676  // <nl>Example 3:
677  // <nl>Obj_ClearFlag All // Clears all the flags.
678  // @flag all | Clear all the flags
679  case 0x6c2b67f9: // Obj_ClearFlag
680  {
681  if (pParams->ContainsFlag(0xc4e78e22/*All*/))
682  {
683  mScriptFlags=0;
684  return true;
685  }
686  uint32 Flags = 0;
687 
688  Flags = GetFlags( pParams, pScript );
689  if ( !Flags )
690  {
691  Dbg_MsgAssert( 0,( "\n%s\nObj_ClearFlag command requires a flag to be specified.\n(Either an integer or a name defined to be an integer)",pScript->GetScriptInfo()));
692  return ( false );
693  }
694  mScriptFlags &= ~Flags;
695  break;
696  }
697 
698  // @script | Obj_SetFlag | Sets the flag (or an array of flags) on an object.
699  // <nl>Same parameters as Obj_ClearFlag (except doesn't recognize the
700  // 'all' parameter).
701  case 0xbe563426: // Obj_SetFlag
702  {
703  uint32 Flags = 0;
704  Flags = GetFlags( pParams, pScript );
705  if ( !Flags )
706  {
707  Dbg_MsgAssert( 0,( "\n%s\nObj_SetFlag command requires a flag to be specified.\n(Either an integer or a name defined to be an integer)",pScript->GetScriptInfo()));
708  return ( false );
709  }
710  mScriptFlags |= Flags;
711  break;
712  }
713 
714  // @script | Obj_KillSpawnedScript | Kills a spawned script. Can be passeed a name or id.
715  // @parm name | name | Name of the script you want to spawn (no quotes)
716  // @parm name | id | id of the script you want to spawn (no quotes)
717  case 0xfbd89cd5: // Obj_KillSpawnedScript
718  {
719  uint32 ScriptChecksum=0;
720  if (pParams->GetChecksum(CRCD(0xa1dc81f9,"Name"),&ScriptChecksum))
721  {
722  // Got a script name, so kill all spawned scripts that ran that script.
723  // BUT NOT THE SCRIPT CALLING THIS!
724  Script::KillSpawnedScriptsWithObjectAndName( this, ScriptChecksum, pScript );
725  return true;
726  }
727 
728  uint32 Id=0;
729  if (pParams->GetChecksum(CRCD(0x40c698af,"Id"),&Id))
730  {
731  // They specified an Id, so kill all spawned scripts with this Id,
732  // BUT NOT THE SCRIPT CALLING THIS!
733  Script::KillSpawnedScriptsWithObjectAndId( this, Id, pScript );
734  return ( true );
735  }
736  Dbg_MsgAssert( 0,( "\n%s\nMust specify Name or ID on Obj_KillSpawnedScript", pScript->GetScriptInfo() ));
737  return true;
738  break;
739  }
740 
741  // @script | Obj_SpawnScript | Causes the object to run a script, in parallel
742  // to whatever script is running on the object.
743  // @flag ScriptName | Name of the script you want to spawn (no quotes)
744  // @parmopt name | Params | {} | Any parameters you want to pass to the script being
745  // spawned. Must surround params in { }
746  case 0x23a4e5c2: // Obj_SpawnScript
747  {
748  Script::CComponent* p_component = pParams->GetNextComponent();
749  if ( p_component && p_component->mType == ESYMBOLTYPE_NAME )
750  {
751  uint32 scriptChecksum = p_component->mChecksum;
752 
753  // The spawned script can optionally be given an Id, so that it can be deleted
754  // by KillSpawnedScript.
755  uint32 Id=0;
756  // keep the same ID as the parent if not specified...
757  Id = Script::FindSpawnedScriptID(pScript);
758  pParams->GetChecksum("Id",&Id);
759  Script::CScriptStructure *pScriptParams = NULL;
760  pParams->GetStructure( "Params", &pScriptParams );
761  #ifdef __NOPT_ASSERT__
762  Script::CScript *p_script=SpawnScriptPlease( scriptChecksum, pScriptParams, Id, pParams->ContainsFlag(CRCD(0x8757d0bb, "PauseWithObject")) );
763  p_script->SetCommentString("Created by Obj_SpawnScript");
764  p_script->SetOriginatingScriptInfo(pScript->GetCurrentLineNumber(),pScript->mScriptChecksum);
765  #else
766  SpawnScriptPlease( scriptChecksum, pScriptParams, Id );
767  #endif
768 
769  }
770  break;
771  }
772 
773  // @script | Obj_SwitchScript | Causes the object to replace the current script
774  // attached to it with the script specified by ScriptName. <nl>
775  // Can use the pass params just like Obj_SpawnScript.
776  // @flag ScriptName | Name of the script you want to spawn (no quotes)
