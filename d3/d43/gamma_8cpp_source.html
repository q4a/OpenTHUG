86 static void gammaInitTable( void )
87 {
88  // Create the gamma intensity lookup table.
89  // It works like this: given a colour value r [0,255] do r' = gammaTable.red[r] etc.
90  for( int i = 0; i < gammaDomain; ++i )
91  {
92  gammaTable.red[i] = (BYTE)((float)gammaRange * pow(((float)i / (float)gammaDomain ), 1.0f / gammaValueR ));
93  gammaTable.green[i] = (BYTE)((float)gammaRange * pow(((float)i / (float)gammaDomain ), 1.0f / gammaValueG ));
94  gammaTable.blue[i] = (BYTE)((float)gammaRange * pow(((float)i / (float)gammaDomain ), 1.0f / gammaValueB ));
95  }
96 }

104 void gammaSetValue( float gr, float gg, float gb )
105 {
106  gammaValueR = gr;
107  gammaValueG = gg;
108  gammaValueB = gb;

111  gammaInitTable();
112 
113  // Pass the table along to the hardware.
114  D3DDevice_SetGammaRamp( D3DSGR_NO_CALIBRATION, &gammaTable );
115 }

127 void SetGammaNormalized( float fr, float fg, float fb )
128 {
129  // Cap it....
130  fr = ( fr < 0.0f ) ? 0.0f : (( fr > 1.0f ) ? 1.0f : fr );
131  fg = ( fg < 0.0f ) ? 0.0f : (( fg > 1.0f ) ? 1.0f : fg );
132  fb = ( fb < 0.0f ) ? 0.0f : (( fb > 1.0f ) ? 1.0f : fb );
133 
134  // Scale it....
135  fr *= MAX_GAMMA;
136  fg *= MAX_GAMMA;
137  fb *= MAX_GAMMA;
138 
139  // Offset it.
140  fr += 0.5f;
141  fg += 0.5f;
142  fb += 0.5f;
143 
144  // f<col> now goes from 0.5 to 2.5.
145  gammaSetValue( fr, fg, fb );
146 }
