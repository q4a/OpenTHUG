45 static int sort_by_material_draw_order( const void *p1, const void *p2 )
46 {
47  sMesh *p_mesh1 = *((sMesh**)p1 );
48  sMesh *p_mesh2 = *((sMesh**)p2 );
49  sMaterialHeader *p_material1 = p_mesh1->mp_dl->m_material.p_header;
50  sMaterialHeader *p_material2 = p_mesh2->mp_dl->m_material.p_header;
51 
52  Dbg_Assert( p_material1 != NULL );
53  Dbg_Assert( p_material2 != NULL );
54 
55  if( p_material1->m_draw_order == p_material2->m_draw_order )
56  {
57  // Have to do some special case processing for the situation where two or more meshes have the same draw order, but only some are
58  // marked as being dynamically sorted. In such a situation the non dynamically sorted ones should come first.
59  if( ( p_material1->m_flags & (1<<0) ) == ( p_material2->m_flags & (1<<0) ) )
60  {
61  if( p_material1 == p_material2 )
62  {
63  // Same material, no further sorting required.
64  return 0;
65  }
66  else
67  {
68  // If the blend mode is the same, sort by material address.
69  if( p_material1->m_base_blend == p_material2->m_base_blend )
70  {
71  // If the pixel shaders are the same, sort by material address, otherwise sort by pixel shader value.
72  if( p_material1->m_texture_dl_id == p_material2->m_texture_dl_id )
73  {
74  return ((uint32)p_material1 > (uint32)p_material2 ) ? 1 : -1;
75  }
76  else
77  {
78  return ((uint32)p_material1->m_texture_dl_id < (uint32)p_material2->m_texture_dl_id ) ? 1 : -1;
79  }
80  }
81  else
82  {
83  return ((uint32)p_material1->m_base_blend > (uint32)p_material2->m_base_blend ) ? 1 : -1;
84  }
85  }
86  }
87  else if( p_material1->m_flags & (1<<0) )
88  {
89  return 1;
90  }
91  return -1;
92  }
93  return ( p_material1->m_draw_order > p_material2->m_draw_order ) ? 1 : -1;
94 }

112 static void hide_mesh( uint32 mask, sCASData *p_cas_data, uint32 num_entries, sMesh * p_mesh )
113 {
114  uint16 index_workbuffer[INDEX_WORKBUFFER_SIZE][MAX_INDEX_WIDTH];
115  uint32 new_indices_index = 0;
116 
117  int index_strip_len[MAX_STRIPS];
118  int num_strips = 0;
119 
120  // Clear strips.
121  for ( int lp = 0; lp < MAX_STRIPS; lp++ )
122  {
123  index_strip_len[lp] = 0;
124  }
125 
126  unsigned char * p_start = (unsigned char *)&p_mesh->mp_dl[1];
127  unsigned char * p_end = &p_start[p_mesh->mp_dl->m_size];
128  p_start = &p_start[p_mesh->mp_dl->m_index_offset]; // Skip to actual 1st GDBegin.
129  unsigned char * p8 = p_start;
130 
131  uint8 begin_token = p8[0]; // Save token for use when rebuilding mesh.
132 
133  int stride = p_mesh->mp_dl->m_index_stride;
134 
135  int w;
136 
137  while ( p8 < p_end )
138  {
139  if ( ( p8[0] & 0xf8 ) == GX_TRIANGLESTRIP )
140  {
141  // Found a triangle strip - parse it.
142  int num_verts = ( p8[1] << 8 ) | p8[2];
143  p8 += 3; // Skip GDBegin
144 
145  uint16 idx0[MAX_INDEX_WIDTH];
146  uint16 idx1[MAX_INDEX_WIDTH];
147  uint16 idx2[MAX_INDEX_WIDTH];
148 
149  for ( w = 0; w < stride; w++ ) { idx1[w] = ( p8[0] << 8 ) | p8[1]; p8 += 2; }
150  for ( w = 0; w < stride; w++ ) { idx2[w] = ( p8[0] << 8 ) | p8[1]; p8 += 2; }
151 
152  for ( int v = 2; v < num_verts; v++ )
153  {
154  // Read next index to form triangle.
155  for ( w = 0; w < stride; w++ ) idx0[w] = idx1[w];
156  for ( w = 0; w < stride; w++ ) idx1[w] = idx2[w];
157  for ( w = 0; w < stride; w++ ) { idx2[w] = ( p8[0] << 8 ) | p8[1]; p8 += 2; }
158 
159  // There shuold be no degenerate triangles.
160  // ...check against every CAS entry.
161  bool keep = true;
162  for( uint32 entry = 0; entry < num_entries; ++entry )
163  {
164  // Check this CAS entry has the correct mask...
165  if( p_cas_data[entry].mask & mask )
166  {
167  // ...and the right mesh...
168  uint32 mesh = p_cas_data[entry].data0 >> 16;
169 
170  // The CAS data is ordered first in increasing mesh size - so we can early out here if applicable.
171  if ( mesh > p_mesh->mp_dl->m_mesh_index )
172  {
173  break;
174  }
175 
176  if ( mesh == p_mesh->mp_dl->m_mesh_index )
177  {
178  // ...and the correct index.
179  uint32 i0 = ( p_cas_data[entry].data0 & 0xFFFF );
180 
181  // The CAS data is next ordered in increasing size of i0 - so we can early out here if applicable.
182  if ( i0 > idx0[0] )
183  {
184  break;
185  }
186 
187  uint32 i1 = ( p_cas_data[entry].data1 >> 16 );
188  uint32 i2 = ( p_cas_data[entry].data1 & 0xFFFF );
189  if ( ( i0 == idx0[0] ) && ( i1 == idx1[0] ) && ( i2 == idx2[0] ) )
190  {
191  keep = false;
192 
193  break;
194  }
195  }
196  }
197  }
198 
199  // Copy index.
200  if ( keep )
201  {
202  if ( index_strip_len[num_strips] )
203  {
204  // Already put 1st triangle in, just add index.
205  for ( w = 0; w < stride; w++ ) index_workbuffer[new_indices_index][w] = idx2[w];
206  new_indices_index++;
207  index_strip_len[num_strips]++;
208  }
209  else
210  {
211  // 1st triangle, so add all 3 verts.
212  for ( w = 0; w < stride; w++ ) index_workbuffer[new_indices_index][w] = idx0[w];
213  new_indices_index ++;
214  for ( w = 0; w < stride; w++ ) index_workbuffer[new_indices_index][w] = idx1[w];
215  new_indices_index ++;
216  for ( w = 0; w < stride; w++ ) index_workbuffer[new_indices_index][w] = idx2[w];
217  new_indices_index ++;
218  index_strip_len[num_strips] += 3;
219  }
220  }
221  else
222  {
223  // If we don't keep this tri & we already started a strip, we need
224  // to start a new strip.
225  if ( index_strip_len[num_strips] )
226  {
227  // Need to start a new strip.
228  num_strips++;
229  }
230  }
231  }
232  // Start a new strip if we put data in this one.
233  if ( index_strip_len[num_strips] )
234  {
235  // Need to start a new strip.
236  num_strips++;
237  }
238  }
239  else
240  {
241  break;
242  }
243  }
244  do_audio_update();
245 
246  Dbg_MsgAssert( new_indices_index <= INDEX_WORKBUFFER_SIZE, ( "Too many indices in new mesh: %d\n", new_indices_index ) );
247 
248  // See if we'll fit in the existing buffer.
249  uint32 new_size = ( 3 * num_strips ) + ( sizeof( uint16 ) * stride * new_indices_index ) + p_mesh->mp_dl->m_index_offset;
250 
251 // OSReport( "New: %d Old: %d Diff: %d\n", new_size, p_mesh->mp_dl->m_size, p_mesh->mp_dl->m_size - new_size );
252 
253 // if ( new_size > p_mesh->mp_dl->m_original_dl_size )
254 // {
255 // // Need to allocate a new buffer for this.
256 // }
257 
258  if ( new_size <= p_mesh->mp_dl->m_original_dl_size )
259  {
260  p8 = p_start;
261  int buffer_index = 0;
262 
263  for ( int strip = 0; strip < num_strips; strip++ )
264  {
265  *p8++ = begin_token;
266  *p8++ = (uint8)(index_strip_len[strip] >> 8 );
267  *p8++ = (uint8)(index_strip_len[strip] & 0xff );
268  for ( int index = 0; index < index_strip_len[strip]; index++ )
269  {
270  for ( w = 0; w < stride; w++ )
271  {
272  *p8++ = (uint8)(index_workbuffer[buffer_index][w] >> 8 );
273  *p8++ = (uint8)(index_workbuffer[buffer_index][w] & 0xff );
274  }
275  buffer_index++;
276  }
277  }
278  // Set new size of display list.
279  int new_rounded_size = ( new_size + 31 ) & ~31;
280  p_mesh->mp_dl->m_size = new_rounded_size;
281 
282  // Pad the DL with 0s to avoid spurious crap at the end & flush cache.
283  int pad_size = new_rounded_size - new_size;
284  for ( int pad = 0; pad < pad_size; pad++ ) *p8++ = 0;
285 
286  DCFlushRange( &p_mesh->mp_dl[1], new_rounded_size );
287  }
288  else
289  {
290 #ifdef __NOPT_FINAL__
291  OSReport( "Warning: DL too big after hide_mesh - New: %d Old: %d Diff: %d\n", new_size, p_mesh->mp_dl->m_size, p_mesh->mp_dl->m_size - new_size );
292 #else
293  Dbg_MsgAssert( false, ( "Error: DL too big after hide_mesh - New: %d Old: %d Diff: %d\n", new_size, p_mesh->mp_dl->m_size, p_mesh->mp_dl->m_size - new_size ) );
294 #endif // __NOPT_FINAL__
295  }
296  do_audio_update();
297 }

616 void sScene::SortMeshes( void )
617 {
618  // Sort the list of meshes.
619  qsort( mpp_mesh_list, m_num_filled_meshes, sizeof( sMesh* ), sort_by_material_draw_order );
620 
621  if ( mp_scene_data )
622  {
623  m_num_opaque_meshes = 0;
624  m_num_pre_semitrans_meshes = 0;
625  m_num_dynamic_semitrans_meshes = 0;
626  m_num_post_semitrans_meshes = 0;
627 
628  // Force opaque to be before semitrans.
629  sMesh *p_mesh[8192];
630  int mesh_entry = 0;
631  Dbg_MsgAssert( m_num_filled_meshes < 8192, ( "Too many meshes for temporary sort buffer." ) );
632 
633  // Find 1st dynamic entry.
634  int first_dynamic = -1;
635  for( int i = 0; i < m_num_filled_meshes; ++i )
636  {
637  if ( mpp_mesh_list[i] )
638  {
639  bool sorted = mpp_mesh_list[i]->mp_dl->m_material.p_header->m_flags & (1<<0) ? true : false;
640 
641  if ( ( first_dynamic == -1 ) && sorted ) first_dynamic = i;
642  }
643  }
644 
645 
646  for ( int pass = 0; pass < 4; pass++ )
647  {
648  for( int i = 0; i < m_num_filled_meshes; ++i )
649  {
650  if ( mpp_mesh_list[i] )
651  {
652  bool transparent = ( mp_material_pass[mpp_mesh_list[i]->mp_dl->m_material.p_header->m_pass_item].m_flags & (1<<3) );
653 // bool transparent = ( mpp_mesh_list[i]->mp_dl->m_material.p_header->m_flags & (1<<3) ) ? true : false;
654 
655  bool sorted = mpp_mesh_list[i]->mp_dl->m_material.p_header->m_flags & (1<<0) ? true : false;
656 
657  if ( ( first_dynamic == -1 ) && sorted ) first_dynamic = i;
658 
659  switch ( pass )
660  {
661  case 0: // Opaque
662  if ( !transparent )
663  {
664  p_mesh[mesh_entry] = mpp_mesh_list[i];
665  mpp_mesh_list[i] = NULL;
666  mesh_entry++;
667  m_num_opaque_meshes++;
668  }
669  break;
670  case 1: // Pre semi
