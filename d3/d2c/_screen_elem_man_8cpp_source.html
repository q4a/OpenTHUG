342 void CScreenElementManager::set_tree_lock_state(CScreenElement::ELockState state)
343 {
344  if (mp_root_element)
345  {
346  CScreenElement* p_stack[32];
347  p_stack[0] = mp_root_element;
348  int depth = 1;
349 
350  while(depth)
351  {
352  // pop top value off stack
353  CScreenElement* p_node = p_stack[--depth];
354 
355  #ifdef __NOPT_ASSERT__

357  #endif
358

365  // so we can't depend on it being referenced

367  p_node->RemoveReference();
368  }
369 
370  Dbg_Assert(depth <= 30);
371 
372  // put sibling and child on stack
373  if (p_node->GetNextSibling())
374  p_stack[depth++] = p_node->GetNextSibling();
375  if (p_node->GetFirstChild())
376  p_stack[depth++] = p_node->GetFirstChild();
377  }

550 /*
551  This function takes a complex ID and reduces it to a regular checksum that designates a single screen
552  element. The following are examples of complex ID's:
553 
554  id=blah1 <-- also a regular ID
555  id={blah1 child=0} <-- first child of blah1
556  id=<blah1 child={0 child=2}} <-- third child of first child of blah1
557 
558  id={blah1 child=local_blah} <-- returns child of blah1 with *LOCAL* ID local_blah
559  (can be recursed as above, or mixed and matched with indices)
560 
561  TO BE SUPPORTED IN FUTURE:
562 
563  id={blah1 parent=0} <-- parent of blah1
564  id={blah1 parent=1} <-- grandparent of blah1
565 
566  id={blah1 child=all_children} <-- all children of blah1
567 
568  Returns 0 if ID could not be resolved, or if nothing matches the name in pIdSubStructName
569 */
570 uint32 CScreenElementManager::ResolveComplexID(Script::CStruct *pStructContainingId, uint32 IdSubStructName)
571 {
572  // mp_resolve_temp will be NULL when we enter this function at the beginning of a recursion chain,
573  // non-NULL when inside the chain. In the latter case, it points to the element closest to
574  // desired one (so far)
575 
576  Script::CStruct *p_recurse_struct = NULL;
577  uint32 id = 0;
578  int index = -1;
579  if (pStructContainingId->GetStructure(IdSubStructName, &p_recurse_struct))
580  {
581  // Expecting the form something={...}
582  // May be recursively inside some other ID structure
583 
584  // grab the 'x' part of structure, where x is something={id ...}
585 
586  // grab the ID part of something={id ...}
587  // we expect either this or a child index (see below)
588  //
589  uint32 contained_id = 0;
590  p_recurse_struct->GetChecksum(NONAME, &contained_id);
591 
592  // grab the child index part of something={child_index ...}
593  int child_index = -1;
594  p_recurse_struct->GetInteger(NONAME, &child_index);
595 
596  if (contained_id != 0)
597  {
598  if (!mp_resolve_temp)
599  // we're at the top level of the recursion chain
600  mp_resolve_temp = GetElement(contained_id);
601  else
602  {
603  // we're NOT at the top level of the recursion chain, so treat ID as
604  // a LOCAL id
605  mp_resolve_temp = get_element_by_local_id(mp_resolve_temp, contained_id);
606  }
607  if (!mp_resolve_temp)
608  return 0;
609  }
610  else if (child_index != -1)
611  {
612  Dbg_MsgAssert(mp_resolve_temp, ("can't map child %d of %s to anything, no parent", child_index, Script::FindChecksumName(IdSubStructName)));
613  mp_resolve_temp = mp_resolve_temp->GetChildByIndex(child_index);
614  if (!mp_resolve_temp)
615  return 0;
616  }
617  else
618  {
619  #ifdef __NOPT_ASSERT__
620  Script::PrintContents(pStructContainingId);
621  Dbg_MsgAssert(0, ("can't resolve complex ID %s, no ID or index given. See struct above.", Script::FindChecksumName(IdSubStructName)));
622  #endif
623  }
624  // we can expect to recurse further
625  uint32 result_id = ResolveComplexID(p_recurse_struct, CRCD(0xdd4cabd6,"child"));
626  mp_resolve_temp = NULL;
627 
628  return result_id;
629  }
630  else if (pStructContainingId->GetChecksum(IdSubStructName, &id))
631  {
632  // Expecting the form something=some_checksum
633  // May be recursively inside some other ID structure
634 
635  if (mp_resolve_temp)
636  {
637  // we're NOT at the top level of the recursion chain, so treat ID as
638  // a LOCAL id
639  mp_resolve_temp = get_element_by_local_id(mp_resolve_temp, id);
640  id = mp_resolve_temp->GetID();
641  }
642  else
643  {
644  // see if ID is really an alias
645  Obj::CTracker* p_tracker = Obj::CTracker::Instance();
