30 static void s_rotate_map_vector(int *p_x, int *p_z, Mth::ERot90 rotation)
31 {
32  int in_x=*p_x;
33  int in_z=*p_z;
34  switch (rotation)
35  {
36  case Mth::ROT_0:
37  break;
38  case Mth::ROT_90:
39  *p_x=in_z;
40  *p_z=-in_x;
41  break;
42  case Mth::ROT_180:
43  *p_x=-in_x;
44  *p_z=-in_z;
45  break;
46  case Mth::ROT_270:
47  *p_x=-in_z;
48  *p_z=in_x;
49  break;
50  default:
51  Dbg_MsgAssert(0, ("Wacky rotation"));
52  break;
53  }
54 }

85 void CClipboardEntry::CalculateMapCoords(GridDims *p_mapCoords, uint8 centre_x, uint8 centre_z, CCursor *p_cursor)
86 {
87  GridDims meta_area;
88  if (mpMeta)
89  {
90  meta_area=mpMeta->GetArea();
91  }
92  else
93  {
94  // If there is no meta then the clipboard entry is defining a height for one cell, so
95  // just set the w and l to 1.
96  meta_area.SetWHL(1,1,1);
97  }
98 
99  Mth::ERot90 total_rotation=Mth::ERot90((p_cursor->GetRotation()+m_rot)&3);
100  switch ( total_rotation )
101  {
102  case Mth::ROT_0:
103  case Mth::ROT_180:
104  p_mapCoords->SetWHL(meta_area.GetW(),1,meta_area.GetL());
105  break;
106  case Mth::ROT_90:
107  case Mth::ROT_270:
108  p_mapCoords->SetWHL(meta_area.GetL(),1,meta_area.GetW());
109  break;
110  default:
111  Dbg_MsgAssert(0, ("Wacky rotation"));
112  break;
113  }
114 
115  // Get the vector from the centre of the selection area to the piece.
116  int vector_x=mX-centre_x;
117  int vector_z=mZ-centre_z;
118 
119  // Rotate the vector according to the cursor's current rotation
120  s_rotate_map_vector(&vector_x, &vector_z, p_cursor->GetRotation());
121 
122  // A frig is needed after rotation because the piece always appears to be
123  // displayed with the coords referring to the top-left corner, even when the
124  // piece is rotated.
125  switch ( p_cursor->GetRotation() )
126  {
127  case Mth::ROT_0:
128  break;
129  case Mth::ROT_90:
130  vector_z-=mWidth-1;
131  break;
132  case Mth::ROT_180:
133  vector_x-=mWidth-1;
134  vector_z-=mLength-1;
135  break;
136  case Mth::ROT_270:
137  vector_x-=mLength-1;
138  break;
139  default:
140  Dbg_MsgAssert(0, ("Wacky rotation"));
141  break;
142  }
143 
144  // Get the new map coords of the meta such that it is positioned relative to the cursor.
145  GridDims cursor_position=p_cursor->GetPosition();
146  p_mapCoords->SetXYZ(cursor_position.GetX()+vector_x, mY, cursor_position.GetZ()+vector_z);
147 }

163 void CClipboardEntry::Paste(uint8 centre_x, int raiseAmount, uint8 centre_z, CCursor *p_cursor)
164 {
165  if (mpMeta && mpMeta->IsRiser())
166  {
167  return;
168  }
169 
170  GridDims map_coords;
171  CalculateMapCoords(&map_coords,centre_x,centre_z,p_cursor);
172 
173  int new_y=mY+raiseAmount;
174  if (new_y >= CParkManager::MAX_HEIGHT)
175  {
176  new_y=CParkManager::MAX_HEIGHT-1;
177  }
178 
179  map_coords.SetXYZ(map_coords.GetX(),new_y,map_coords.GetZ());
180 
181  CParkManager *p_manager=CParkManager::sInstance();
182 
183  // Adjust the floor height to match that of the piece
184  int count=0;
185  while (true)
186  {
187  bool uniform_height=false;
188  int current_height = p_manager->GetFloorHeight(map_coords, &uniform_height);
189 
190  if (current_height > new_y)
191  {
192  break;
193  }
194 
195  if (count>40)
196  {
197  Dbg_MsgAssert(0,("Too many recursions of ChangeFloorHeight in CClipboardEntry::Paste"));
198  break;
199  }
200 
201  if (uniform_height && current_height==new_y)
202  {
203  break;
204  }
205 
206  p_manager->ChangeFloorHeight(map_coords,1); // 1 is the step amount, 1 upwards
207  ++count;
208  }
209 
210  if (mpMeta)
211  {
212  // Place the piece
213  CConcreteMetaPiece *p_concrete = p_manager->CreateConcreteMeta(mpMeta);
214  p_concrete->SetRot(Mth::ERot90((m_rot+p_cursor->GetRotation())&3));

258 // This gets called when the CCursor is in paste mode, and needs to display the clipboard contents at
259 // the cursor position.
260 // pos and rot are the world position and rotation in degrees of the cursor.
261 // They may not be cell aligned, and the rotation may not be a multiple of 90 degrees.
262 // This function will position the meta such that the centre of the clipboard selection is at pos.
263 void CClipboardEntry::ShowMeta(uint8 centre_x, int raiseAmount, uint8 centre_z, Mth::Vector pos, float rot)
264 {
265  if (!mpMeta)
266  {
267  // No meta to display, meaning the clipboard entry is defining a cell height
268  return;
269  }
270 
271  // Get the vector (in cell units) from the centre of the selection area to the piece.
272  float vector_x=mX-centre_x;
273  float vector_z=mZ-centre_z;
274 
275  // Calculate the vector, in world coords, from the centre of the selection to the centre of the piece.
276  Mth::Vector p(0.0f,0.0f,0.0f);
277  p[X]=vector_x*CParkGenerator::CELL_WIDTH+(mWidth-1)*CParkGenerator::CELL_WIDTH/2.0f;
278  p[Z]=vector_z*CParkGenerator::CELL_LENGTH+(mLength-1)*CParkGenerator::CELL_LENGTH/2.0f;
279 
280  // Rotate by rot degrees
281  float rad=rot*3.141592654f/180.0f;
282  float s=sinf(rad);
283  float c=cosf(rad);
284 
285  Mth::Vector rotated;
286  rotated[X]=c*p[X]+s*p[Z];
287  rotated[Y]=0.0f;
288  rotated[Z]=c*p[Z]-s*p[X];
289 
290  // Add the passed pos to get the final world position of the piece.
291  Mth::Vector world_pos=pos+rotated;
292 
293  // Add in the cell height
294  world_pos[Y]+=mY*CParkGenerator::CELL_HEIGHT;
295  if (mpMeta->IsRiser())
296  {
297  world_pos[Y]-=CParkGenerator::CELL_HEIGHT;
298  }
299 
300  // Create the concrete meta if it does not exist already.
301  if (!mpConcreteMeta)
302  {
303  Dbg_MsgAssert(mpMeta,("NULL mpMeta ?"));
304  mpConcreteMeta=CParkManager::sInstance()->CreateConcreteMeta(mpMeta,true);
305  mpConcreteMeta->MarkLocked();
306  }
307 
308  // Position the piece, adding in the pieces local rotation value.
309  mpConcreteMeta->SetSoftRot(world_pos,rot+90.0f*m_rot);
310 }

549 {

551  // a new height, to prevent unnecessary column slides.

558  {

560  }
561  p_entry=p_entry->mpNext;
562  }
563 
564  // If there is a height defined for x,z already then just update it's y
565  // if the passed y is bigger.
566  p_entry=mp_entries;
567  while (p_entry)
568  {
569  if (p_entry->mpMeta==NULL &&
570  p_entry->mX <= x && x < p_entry->mX+p_entry->mWidth &&
571  p_entry->mZ <= z && z < p_entry->mZ+p_entry->mLength)
572  {

575  p_entry->mY=y;
576  }
577  return true;
578  }
579  p_entry=p_entry->mpNext;
580  }
581

583  Mem::CPoolable<CClipboardEntry>::SGetTotalItems())
584  {

690  }
691  }

694  }

699  // to above the ground, in which case they won't have pieces filling the gap, which looks odd.

737 // Pastes the contents of the clipboard into the position that the cursor is at,
738 // using the cursor's rotation value.
739 bool CClipboard::Paste(CCursor *p_cursor)
740 {
741  Spt::SingletonPtr<CParkEditor> p_editor;
742  if (!p_editor->RoomToCopyOrPaste(m_max_x-m_min_x+1, m_max_z-m_min_z+1))
743  {
744  return false;
745  }
746 
747  // Get the map coords of the centre cell of the set of meta's
748  uint8 centre_x=(m_min_x+m_max_x)/2;
749  uint8 centre_z=(m_min_z+m_max_z)/2;
750 
751  // Bail out if any entry cannot be pasted
752  CClipboardEntry *p_entry=mp_entries;
753  while (p_entry)
754  {
755  if (!p_entry->CanPaste(centre_x,centre_z,p_cursor))
756  {
757  return false;
758  }
759  p_entry=p_entry->mpNext;
760  }
761 
762  // Bail out if there are not enough free rail points to create the edited rails
763  int num_rail_points_to_paste=0;
764  Obj::CEditedRail *p_rail=mp_rails;
765  while (p_rail)
766  {
767  num_rail_points_to_paste += p_rail->CountPoints();
768  p_rail=p_rail->mpNext;
769  }
770  if (num_rail_points_to_paste > Obj::GetRailEditor()->GetNumFreePoints() )
771  {
772  return false;
773  }
774 
775 
776  // OK to proceed, so paste the entries
777  p_entry=mp_entries;
778  while (p_entry)
779  {
780  p_entry->Paste(centre_x,p_cursor->GetClipboardY(),centre_z,p_cursor);
781  p_entry=p_entry->mpNext;
782  }
783 
784  // and paste the rails
785  Mth::Vector clipboard_pos=GetClipboardWorldPos();
786  Mth::Vector cursor_pos=p_editor->GetCursorPos();
787 
788  p_rail=mp_rails;
789  while (p_rail)
790  {
791  Obj::CEditedRail *p_new_rail=Obj::GetRailEditor()->NewRail();
792  p_new_rail->CopyRail(p_rail);
793  p_new_rail->RotateAndTranslate(clipboard_pos, cursor_pos, p_cursor->GetRotation()*90.0f);
794 
795  p_new_rail->UpdateRailGeometry();
796  p_new_rail->UpdatePostGeometry();
797 
798  p_rail=p_rail->mpNext;
799  }
800 
801  return true;
802 }

880 void CClipboard::GetArea(CCursor *p_cursor, GridDims *p_area)
881 {
882  // Get the map coords of the centre cell of the set of meta's
883  uint8 centre_x=(m_min_x+m_max_x)/2;
884  uint8 centre_z=(m_min_z+m_max_z)/2;
885 
886  int idx=m_min_x-centre_x;
887  int idz=m_min_z-centre_z;
888  int dx=0;
889  int dz=0;
890 
891  switch ( p_cursor->GetRotation() )
892  {
893  case Mth::ROT_0:
894  dx=idx;
895  dz=idz;
896  p_area->SetWHL(m_area.GetW(),1,m_area.GetL());
897  break;
898  case Mth::ROT_270:
899  dx=-idz;
900  dz=idx;
901  p_area->SetWHL(m_area.GetL(),1,m_area.GetW());
902  // Shift dx,dz so that they refer to the top-left point again
903  dx=dx-m_area.GetL()+1;
904  break;
905  case Mth::ROT_180:
906  dx=-idx;
907  dz=-idz;
908  p_area->SetWHL(m_area.GetW(),1,m_area.GetL());
909  // Shift dx,dz so that they refer to the top-left point again
910  dx=dx-m_area.GetW()+1;
911  dz=dz-m_area.GetL()+1;
912  break;
913  case Mth::ROT_90:
914  dx=idz;
915  dz=-idx;
916  p_area->SetWHL(m_area.GetL(),1,m_area.GetW());
917  // Shift dx,dz so that they refer to the top-left point again
918  dz=dz-m_area.GetW()+1;
919  break;
920  default:
921  break;
922  }
923 
924  GridDims cursor_position=p_cursor->GetPosition();
925  p_area->SetXYZ(cursor_position.GetX()+dx,0,cursor_position.GetZ()+dz);
926 }
