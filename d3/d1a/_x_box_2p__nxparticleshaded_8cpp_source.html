152 {
153  // Draw the particles.
154  if( m_num_particles > 0 )
155  {
156  // Used to figure the right and up vectors for creating screen-aligned particle quads.
157  D3DXMATRIX *p_matrix = (D3DXMATRIX*)&NxXbox::EngineGlobals.view_matrix;
158 
159  // Concatenate p_matrix with the emmission angle to create the direction.
160  Mth::Vector up( 0.0f, 1.0f, 0.0f );
161 
162  // Get the 'right' vector as the cross product of camera 'at and world 'up'.
163  Mth::Vector at( p_matrix->m[0][2], p_matrix->m[1][2], p_matrix->m[2][2] );
164  Mth::Vector screen_right = Mth::CrossProduct( at, up );
165  Mth::Vector screen_up = Mth::CrossProduct( screen_right, at );
166 
167  screen_right.Normalize();
168  screen_up.Normalize();
169 
170  int lp;
171  CParticleEntry *p_particle;
172  float *p_v;
173 // Mth::Vector min, max; // For dynamic bounding box calculation.
174 
175  // Obtain push buffer lock.
176  DWORD *p_push;
177  DWORD dwords_per_particle = 24;
178  DWORD dword_count = dwords_per_particle * m_num_particles;
179 
180  // Submit particle material.
181  mp_engine_particle->mp_material->Submit();
182 
183  // Set up correct vertex and pixel shader.
184  NxXbox::set_vertex_shader( D3DFVF_XYZ | D3DFVF_DIFFUSE | D3DFVF_TEX1 | D3DFVF_TEXCOORDSIZE2( 0 ));
185  NxXbox::set_pixel_shader( PixelShader0 );
186 
187  // The additional number (+5 is minimum) is to reserve enough overhead for the encoding parameters. It can safely be more, but no less.
188  p_push = D3DDevice_BeginPush( dword_count + 16 );
189 
190  // Note that p_push is returned as a pointer to write-combined memory. Writes to write-combined memory should be
191  // consecutive and in increasing order. Reads should be avoided. Additionally, any CPU reads from memory or the
192  // L2 cache can force expensive partial flushes of the 32-byte write-combine cache.
193  p_push[0] = D3DPUSH_ENCODE( D3DPUSH_SET_BEGIN_END, 1 );
194  p_push[1] = D3DPT_QUADLIST;
195  p_push += 2;
196 
197  // Set up loop variables here, since we be potentially enetering the loop more than once.
198  lp = 0;
199  p_particle = mp_particle_array;
200  p_v = mp_vertices;
201 
202  while( dword_count > 0 )
203  {
204  int dwords_written = 0;
205 
206  // NOTE: A maximum of 2047 DWORDs can be specified to D3DPUSH_ENCODE. If there is more than 2047 DWORDs of vertex
207  // data, simply split the data into multiple D3DPUSH_ENCODE( D3DPUSH_INLINE_ARRAY ) sections.
208  p_push[0] = D3DPUSH_ENCODE( D3DPUSH_NOINCREMENT_FLAG | D3DPUSH_INLINE_ARRAY, ( dword_count > 2047 ) ? ((int)( 2047 / dwords_per_particle )) * dwords_per_particle: dword_count );
209  ++p_push;
210 
211  for( ; lp < m_num_particles; lp++, p_particle++, p_v += 3 )
212  {
213  // Check to see if writing another particle will take us over the edge.
214  if(( dwords_written + dwords_per_particle ) > 2047 )
215  {
216  break;
217  }
218 
219  float terp = p_particle->m_time / p_particle->m_life;
220  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
221  float h = p_particle->m_sh + ( ( p_particle->m_eh - p_particle->m_sh ) * terp );
222 
223  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
224  Mth::Vector pos( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
225  Mth::Vector ss_right, ss_up, ss_pos;
226  Mth::Vector tmp;
227 
228  // Dynamic bounding box calculation.
229 // if( lp == 0 )
230 // {
231 // min = pos;
232 // max = pos;
233 // }
234 // else
235 // {
236 // if( pos[X] < min[X] ) min[X] = pos[X]; else if( pos[X] > max[X] ) max[X] = pos[X];
237 // if( pos[Y] < min[Y] ) min[Y] = pos[Y]; else if( pos[Y] > max[Y] ) max[Y] = pos[Y];
238 // if( pos[Z] < min[Z] ) min[Z] = pos[Z]; else if( pos[Z] > max[Z] ) max[Z] = pos[Z];
239 // }
240

242  ss_up = screen_up * h;
243 
244  Image::RGBA color[4];
245  Image::RGBA *p_col0;
246  Image::RGBA *p_col1;
247 
248  if( m_mid_time >= 0.0f )
249  {
250  if( terp < m_mid_time )
251  {
252  p_col0 = m_start_color;
253  p_col1 = m_mid_color;
254 
255  // Adjust interpolation for this half of the color blend.
256  terp = terp / m_mid_time;
257  }
258  else
259  {
260  p_col0 = m_mid_color;
261  p_col1 = m_end_color;
262 
263  // Adjust interpolation for this half of the color blend.
264  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
265  }
266  }
267  else
268  {
269  // No mid color specified.
270  p_col0 = m_start_color;
271  p_col1 = m_end_color;
272  }
273 
274  for ( int c = 0; c < 4; c++ )
275  {
276  Image::RGBA start = *p_col0++;
277  Image::RGBA end = *p_col1++;
278 
279  // Swap red and blue here.
280  color[c].b = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
281  color[c].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
282  color[c].r = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
283  color[c].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
284  }
285 
286  tmp = pos - ss_right + ss_up;
287  p_push[0] = *((DWORD*)&tmp[X] );
288  p_push[1] = *((DWORD*)&tmp[Y] );
289  p_push[2] = *((DWORD*)&tmp[Z] );
290  p_push[3] = *((DWORD*)&color[0] );
291  p_push[4] = 0x00000000UL;
292  p_push[5] = 0x00000000UL;
293  p_push += 6;
294 
295  tmp = pos + ss_right + ss_up;
296  p_push[0] = *((DWORD*)&tmp[X] );
297  p_push[1] = *((DWORD*)&tmp[Y] );
298  p_push[2] = *((DWORD*)&tmp[Z] );
299  p_push[3] = *((DWORD*)&color[1] );
300  p_push[4] = 0x3F800000UL;
301  p_push[5] = 0x00000000UL;
302  p_push += 6;
303 
304  tmp = pos + ss_right - ss_up;
305  p_push[0] = *((DWORD*)&tmp[X] );
306  p_push[1] = *((DWORD*)&tmp[Y] );
307  p_push[2] = *((DWORD*)&tmp[Z] );
308  p_push[3] = *((DWORD*)&color[2] );
309  p_push[4] = 0x3F800000UL;
310  p_push[5] = 0x3F800000UL;
311  p_push += 6;
312 
313  tmp = pos - ss_right - ss_up;
314  p_push[0] = *((DWORD*)&tmp[X] );
315  p_push[1] = *((DWORD*)&tmp[Y] );
316  p_push[2] = *((DWORD*)&tmp[Z] );
317  p_push[3] = *((DWORD*)&color[3] );
318  p_push[4] = 0x00000000UL;
319  p_push[5] = 0x3F800000UL;
320  p_push += 6;
321 
322  dwords_written += dwords_per_particle;
323  dword_count -= dwords_per_particle;
324  }
325  }

328  p_push[1] = 0;
329  p_push += 2;
