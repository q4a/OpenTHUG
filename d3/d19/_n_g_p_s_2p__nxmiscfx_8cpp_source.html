196 bool subdivide_tri_stack( float shatterArea, sNGPSVert **p_write, NxPs2::CGeomNode *p_node )
197 {
198  // If there are elements on the stack, pop off the top three vertices and subdivide if necessary.
199  if( triSubdivideStack.IsEmpty())
200  {
201  return false;
202  }
203 
204  // Three temporary buffers.
205  sNGPSVert v0;
206  sNGPSVert v1;
207  sNGPSVert v2;
208 
209  // Stack is LIFO, so Pop() off in reverse order.
210  triSubdivideStack.Pop( &v2 );
211  triSubdivideStack.Pop( &v1 );
212  triSubdivideStack.Pop( &v0 );
213 
214  // Calculate the area of this tri.
215  Mth::Vector p( v1.pos[X] - v0.pos[X], v1.pos[Y] - v0.pos[Y], v1.pos[Z] - v0.pos[Z] );
216  Mth::Vector q( v2.pos[X] - v0.pos[X], v2.pos[Y] - v0.pos[Y], v2.pos[Z] - v0.pos[Z] );
217  Mth::Vector r(( p[Y] * q[Z] ) - ( q[Y] * p[Z] ), ( p[Z] * q[X] ) - ( q[Z] * p[X] ), ( p[X] * q[Y] ) - ( q[X] * p[Y] ));
218  float area_squared = r.LengthSqr();
219 
220  if( area_squared > shatterArea )
221  {
222  // We need to subdivide this tri. Calculate the three intermediate points.
223  int block_size = triSubdivideStack.GetBlockSize();
224 
225  // Three temporary buffers.
226  sNGPSVert i01;
227  sNGPSVert i12;
228  sNGPSVert i20;
229 
230  memcpy( &i01, &v0, block_size );
231  memcpy( &i12, &v1, block_size );
232  memcpy( &i20, &v2, block_size );
233 
234  // Deal with positions (always present).
235  i01.pos[X] = v0.pos[X] + (( v1.pos[X] - v0.pos[X] ) * 0.5f );
236  i01.pos[Y] = v0.pos[Y] + (( v1.pos[Y] - v0.pos[Y] ) * 0.5f );
237  i01.pos[Z] = v0.pos[Z] + (( v1.pos[Z] - v0.pos[Z] ) * 0.5f );
238 
239  i12.pos[X] = v1.pos[X] + (( v2.pos[X] - v1.pos[X] ) * 0.5f );
240  i12.pos[Y] = v1.pos[Y] + (( v2.pos[Y] - v1.pos[Y] ) * 0.5f );
241  i12.pos[Z] = v1.pos[Z] + (( v2.pos[Z] - v1.pos[Z] ) * 0.5f );
242 
243  i20.pos[X] = v2.pos[X] + (( v0.pos[X] - v2.pos[X] ) * 0.5f );
244  i20.pos[Y] = v2.pos[Y] + (( v0.pos[Y] - v2.pos[Y] ) * 0.5f );
245  i20.pos[Z] = v2.pos[Z] + (( v0.pos[Z] - v2.pos[Z] ) * 0.5f );
246 
247  // Deal with colors (not always present).
248  //if( p_node->m_diffuse_offset > 0 )
249  {
250  uint8 *p_v0col = (uint8*)( &(v0.col));
251  uint8 *p_v1col = (uint8*)( &(v1.col));
252  uint8 *p_v2col = (uint8*)( &(v2.col));
253  uint8 *p_i01col = (uint8*)( &(i01.col));
254  uint8 *p_i12col = (uint8*)( &(i12.col));
255  uint8 *p_i20col = (uint8*)( &(i20.col));
256 
257  for( int i = 0; i < 4; ++i )
258  {
259  p_i01col[i] = p_v0col[i] + (char)(( p_v1col[i] - p_v0col[i] ) * 0.5f );
260  p_i12col[i] = p_v1col[i] + (char)(( p_v2col[i] - p_v1col[i] ) * 0.5f );
261  p_i20col[i] = p_v2col[i] + (char)(( p_v0col[i] - p_v2col[i] ) * 0.5f );
262  }
263  }
264 
265  // Deal with uv0 (not always present).
266  //if( p_node->m_uv0_offset > 0 )
267  {
268  float *p_v0uv = (float*)( &v0.u );
269  float *p_v1uv = (float*)( &v1.u );
270  float *p_v2uv = (float*)( &v2.u );
271  float *p_i01uv = (float*)( &i01.u );
272  float *p_i12uv = (float*)( &i12.u );
273  float *p_i20uv = (float*)( &i20.u );
274 
275  // We know that v is contiguous to u
276  for( int i = 0; i < 2; ++i )
277  {
278  p_i01uv[i] = p_v0uv[i] + (( p_v1uv[i] - p_v0uv[i] ) * 0.5f );
279  p_i12uv[i] = p_v1uv[i] + (( p_v2uv[i] - p_v1uv[i] ) * 0.5f );
280  p_i20uv[i] = p_v2uv[i] + (( p_v0uv[i] - p_v2uv[i] ) * 0.5f );
281  }
282  }
283 
284  // Push the four new tris onto the stack.
285  triSubdivideStack.Push( &v0 );
286  triSubdivideStack.Push( &i01 );
287  triSubdivideStack.Push( &i20 );
288 
289  triSubdivideStack.Push( &i01 );
290  triSubdivideStack.Push( &v1 );
291  triSubdivideStack.Push( &i12 );
292 
293  triSubdivideStack.Push( &i01 );
294  triSubdivideStack.Push( &i12 );
295  triSubdivideStack.Push( &i20 );
296 
297  triSubdivideStack.Push( &i20 );
298  triSubdivideStack.Push( &i12 );
299  triSubdivideStack.Push( &v2 );
300  }
301  else
302  {
303  // Don't need to subdivide this tri.
304  int block_size = triSubdivideStack.GetBlockSize();
305 
306  // Just copy the tri into the next available slot.
307  memcpy( *p_write, &v0, block_size );
308  (*p_write)++;
309  memcpy( *p_write, &v1, block_size );
310  (*p_write)++;
311  memcpy( *p_write, &v2, block_size );
312  (*p_write)++;
313  }
314  return true;
315 }

347 static inline bool line_segment_intersection( float x1, float y1, float x2, float y2, float x3, float y3, float x4, float y4 )
348 {
349  float den = (( y4 - y3 ) * ( x2 - x1 )) - (( x4 - x3 ) * ( y2 - y1 ));
350 
351  if( den == 0.0f )
352  {
353  // Parallel lines.
354  return false;
355  }
356 
357  float num_a = (( x4 - x3 ) * ( y1 - y3 )) - (( y4 - y3 ) * ( x1 - x3 ));
358  float num_b = (( x2 - x1 ) * ( y1 - y3 )) - (( y2 - y1 ) * ( x1 - x3 ));
359 
360  num_a /= den;
361  num_b /= den;
362 
363  if(( num_a <= 1.0f ) && ( num_b <= 1.0f ))
364  return true;
365 
366  return false;
367 }

373 static inline bool tri_texture_intersect( float u0, float v0, float u1, float v1, float u2, float v2 )
374 {
375  // Trivial check to see if all three points are outside range of texture.
376  if(( u0 < -1.0f ) && ( u1 < -1.0f ) && ( u2 < -1.0f ))
377  return false;
378  if(( u0 > 1.0f ) && ( u1 > 1.0f ) && ( u2 > 1.0f ))
379  return false;
380  if(( v0 < -1.0f ) && ( v1 < -1.0f ) && ( v2 < -1.0f ))
381  return false;
382  if(( v0 > 1.0f ) && ( v1 > 1.0f ) && ( v2 > 1.0f ))
383  return false;
384 
385  // Check that at least one corner of the texture falls within the tri.
386  Mth::Vector texture_square[4] = { Mth::Vector( -1.0f, -1.0f, 0.0f, 0.0f ),
387  Mth::Vector( 1.0f, -1.0f, 0.0f, 0.0f ),
388  Mth::Vector( 1.0f, 1.0f, 0.0f, 0.0f ),
389  Mth::Vector( -1.0f, 1.0f, 0.0f, 0.0f )};
390 
391  Mth::Vector a( u0, v0, 0.0f );
392  Mth::Vector b( u1, v1, 0.0f );
393  Mth::Vector c( u2, v2, 0.0f );
394 
395  for( int p = 0; p < 4; ++p )
396  {
397  if( point_in_triangle( texture_square[p], a, b, c ))
398  {
399  return true;
400  }
401  }
402 
403  // No corners of the texture fall within the tri. There are 3 possible explanations:
404  // 1) The tri intersects the texture, but does not contain a texture corner.
405  // 2) The texture lies entirely outside of the tri.
406  // 3) The tri falls entirely within the texture.
407  // Given the relatively small size of the textures, case (3) is extremely unlikely.
408 
409  // Perform a trivial check to see whether a corner of the tri lies within the texture. This will catch (3) and sometimes (1).
410  if(( u0 >= -1.0f ) && ( u0 <= 1.0f ) && ( v0 >= -1.0f ) && ( v0 <= 1.0f ))
411  return true;
412  if(( u1 >= -1.0f ) && ( u1 <= 1.0f ) && ( v1 >= -1.0f ) && ( v1 <= 1.0f ))
413  return true;
414  if(( u2 >= -1.0f ) && ( u2 <= 1.0f ) && ( v2 >= -1.0f ) && ( v2 <= 1.0f ))
415  return true;
416 
417  // Perform the complete check to see if any line segment forming the tri intersects any line segment forming the texture.
418  for( int p = 0; p < 4; ++p )
419  {
420  int q = ( p + 1 ) % 4;
421  if( line_segment_intersection( u0, v0, u1, v1, texture_square[p][X], texture_square[p][Y], texture_square[q][X], texture_square[q][Y] ))
422  return true;
423  }
424 
425  return false;
426 }

512 bool plat_texture_splat( Nx::CSector **pp_sectors, Nx::CCollStatic **pp_collision, Mth::Vector& start, Mth::Vector& end,
513  float size, float lifetime, Nx::CTexture *p_texture, Nx::sSplatTrailInstanceDetails *p_trail_details )
514 {
515  Mth::Matrix view_matrix, ortho_matrix, projection_matrix;
516 
517 # if DRAW_DEBUG_LINES
518  Gfx::AddDebugLine( start, end, MAKE_RGB( 200, 200, 0 ), MAKE_RGB( 200, 200, 0 ), 1 );
519 # endif // DRAW_DEBUG_LINES
520 
521  // The length of the start->end line defines the view depth of the frustum.
522  Mth::Vector splat_vector = end - start;
523  float view_depth = splat_vector.Length();
524  splat_vector.Normalize();
525 
526  // Calculate the parallel projection matrix. Generally the projection vector will tend to point downwards, so we use a
527  // random vector in the x-z plane to define the up vector for the projection. However if this splat is part of a trail,
528  // use the previous point to generate the up vector.
529  if( p_trail_details )
530  {
531  Mth::Vector up( start[X] - p_trail_details->m_last_pos[X], start[Y] - p_trail_details->m_last_pos[Y], start[Z] - p_trail_details->m_last_pos[Z], 0.0f );
532  Dbg_MsgAssert((up[X] != 0.0f) || (up[Z] != 0.0f), ("Up vector has no length"));
533  //Dbg_Message("Splat up vector (%.8f, %.8f, %.8f)", up[X], up[Y], up[Z]);
534 
535  // The height of the viewport is defined by the distance between the two points.
536  float height = up.Length() * 0.5f;
537 
538  start -= up * 0.5f;
539  end -= up * 0.5f;
540 
541  up.Normalize();
542 
543  Mth::CreateMatrixLookAt( view_matrix, start, end, up );
544  Mth::CreateMatrixOrtho( ortho_matrix, size, height, 0.1f, view_depth );
545  }
546  else if( fabsf( splat_vector[Y] ) > 0.5f )
547  {
548  float angle = ((float)rand() * 2.0f * Mth::PI ) / (float)RAND_MAX;
549  Mth::CreateMatrixLookAt( view_matrix, start, end, Mth::Vector( sinf( angle ), 0.0f, cosf( angle ), 0.0f ));
550  Mth::CreateMatrixOrtho( ortho_matrix , size, size, 0.1f, view_depth );
551  }
552  else
553  {
554  Mth::CreateMatrixLookAt( view_matrix, start, end, Mth::Vector( 0.0f, 1.0f, 0.0f, 0.0f ));
555  Mth::CreateMatrixOrtho( ortho_matrix , size, size, 0.1f, view_depth );
556  }
557 
558  projection_matrix = view_matrix * ortho_matrix;
559 
560  // Find texture info
561  Dbg_Assert(p_texture);
562  CPs2Texture *p_ps2_texture = static_cast<CPs2Texture *>(p_texture);
563  NxPs2::SSingleTexture *p_single_texture = p_ps2_texture->GetSingleTexture();
564  Dbg_Assert(p_single_texture);

830  p_details->m_lifetimes[idx] = (int)( lifetime * 1000.0f );
831 
832  p_target_verts[index + 0].pos = p_array_loop[0].pos;
833  p_target_verts[index + 1].pos = p_array_loop[1].pos;

835 
836  p_target_verts[index + 0].u = p_array_loop[0].u;
837  p_target_verts[index + 0].v = p_array_loop[0].v;
838  p_target_verts[index + 1].u = p_array_loop[1].u;
839  p_target_verts[index + 1].v = p_array_loop[1].v;
840  p_target_verts[index + 2].u = p_array_loop[2].u;
841  p_target_verts[index + 2].v = p_array_loop[2].v;
842 
843  p_target_verts[index + 0].col = p_array_loop[0].col;
844  p_target_verts[index + 1].col = p_array_loop[1].col;
845  p_target_verts[index + 2].col = p_array_loop[2].col;
846

864  }
865  }
866  }
867 
868  return false;
869 }
