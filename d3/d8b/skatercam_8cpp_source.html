535 /*******************************************************************************************************
536 logic for getTimeAdjustedSlerp()
537 
538 If the camera is lagging a distance D behind the target, and the target is moving with velocity V, and the lerp is set to L.
539 
540 Then when the camera is stable, it will be moving at the same rate as the target (V), and since the distance it will move is equal to the lerp multiplied by the distance from the target (which will be the old distance plus the new movement of the target V)
541 
542 Then:
543 
544 L*(D+V) = V
545 
546 L*D + L*V = V
547 D = V * (1 - L) / L
548 
549 Assuming in the above T = 1, then if we have a variable T, the speed of the skater will be T*V, yet we want the distance (Dt) moved to remain unchanged (D), so for a time adjusted lerp Lt
550 
551 Dt = T * V * (1-Lt)/Lt
552 
553 Since D = Dt
554 
555 V * (1 - L) / L = T * V * (1-Lt)/Lt
556 
557 V cancels out, and we get
558 
559 Lt = TL / (1 - L + TL)
560 
561 Sanity check,
562 
563 if L is 0.25, and T is 1, then Lt = 1*0.25 / (1 - 0.25 + 1*0.25) = 0.25
564 if L is 0.25, and T is 2, then Lt = 2*0.25 / (1 - 0.25 + 2*0.25) = 0.5 / 1.25 = 0.40
565 if L is 0.25, and T is 5, then Lt = 5*0.25 / (1 - 0.25 + 5*0.25) = 1.25 / 2 = 0.625
566 
567 Sounds about right.
568 
569 */

576 static float getTimeAdjustedSlerp( float slerp, float delta )
577 {
578 #if 1
579  // Mick's method, tries to guarantee that the distance from the target point
580  // will never alter at constant velocity, with changing frame rate
581  // Lt = TL / (1 - L + TL)
582 
583  float t = delta * 60.0f;
584  float Lt = t * slerp / (1.0f - slerp + t * slerp);
585  return Lt;
586 
587 #else
588  // Dave's method. simulates the effect of lerp over multiple frames
589  // curiously seems to be within 2% of the above method.
590 
591  float result = 0.0f;
592  float remainder = 1.0f;
593 
594  while( true )
595  {
596  if( delta >= ( 1.0f / 60.0f ))
597  {
598  delta -= ( 1.0f / 60.0f );
599 
600  float extra = slerp * remainder;
601  remainder -= extra;
602  result += extra;
603  }
604  else
605  {
606  // Partial fragment, or nothing at all?
607  if( delta > 0.0f )
608  {
609  // The slerp this time will be reduced.
610  slerp *= delta / ( 1.0f / 60.0f );
611 
612  float extra = slerp * remainder;
613  remainder -= extra;
614  result += extra;
615  }
616 // printf ("Result %f, Lt %f, diff = %f\n",result,Lt,result-Lt);
617  return result;
618  }
619  }
620 #endif
