75 void CRectFeeler::MergeCollisionSurfaces ( )
76 {
77  // precomputed values
78  bool p_surfaces_used[Nx::MAX_NUM_2D_COLLISIONS_REPORTED];
79  Mth::Vector p_line_directions[Nx::MAX_NUM_2D_COLLISIONS_REPORTED];
80  for (int n = m_coll_data.num_surfaces; n--; )
81  {
82  p_surfaces_used[n] = false;
83 
84  p_line_directions[n] = m_coll_data.p_surfaces[n].ends[1].point - m_coll_data.p_surfaces[n].ends[0].point;
85  p_line_directions[n].Normalize();
86  }
87 
88  // drop duplicate surfaces
89  for (int i = m_coll_data.num_surfaces; i--; )
90  {
91  Nx::S2DCollSurface& surface_i = m_coll_data.p_surfaces[i];
92 
93  if (p_surfaces_used[i]) continue;
94 
95  for (int j = i; j--; )
96  {
97  Nx::S2DCollSurface& surface_j = m_coll_data.p_surfaces[j];
98  if ((very_close(surface_i.ends[0].point, surface_j.ends[0].point) && very_close(surface_i.ends[1].point, surface_j.ends[1].point))
99  || (very_close(surface_i.ends[0].point, surface_j.ends[1].point) && very_close(surface_i.ends[1].point, surface_j.ends[0].point)))
100  {
101  if (Mth::DotProduct(surface_i.normal, surface_j.normal) > 0.99f)
102  {
103  p_surfaces_used[j] = true;
104  continue;
105  }
106  }
107  }
108  }
109 
110  // we build a merged surface list out of the collision data surface list, attempting to reduce the number of surfaces by merging those which
111  // are equivalent for the purpose of collision lines; that is, merge chains of collision lines which are parallel and have equal surface normals
112 
113  // for each surface, first attempt to absorb as yet unvisited surfaces; then, attempt to absorb the surface into a surface previously accepted into
114  // the merged surface list; if unsuccessful, add the surface to the merged surface list
115 
116  m_merged_coll_data.num_surfaces = 0;
117 
118  // loop over the collision data surfaces
119  for (int proposed_idx = m_coll_data.num_surfaces; proposed_idx--; )
120  {
121  Nx::S2DCollSurface proposed_surface = m_coll_data.p_surfaces[proposed_idx];
122 
123  if (p_surfaces_used[proposed_idx]) continue;
124 
125  // check to absorb unused surfaces into the proposed surface
126 
127  bool clean_pass;
128  do // loop over the unused surfaces until we pass through them cleanly, finding no absorption opportunities
129  {
130  clean_pass = true;
131 
132  for (int check_idx = proposed_idx; check_idx--; )
133  {
134  Nx::S2DCollSurface& check_surface = m_coll_data.p_surfaces[check_idx];
135 
136  // check that this line is not already used
137  if (p_surfaces_used[check_idx]) continue;
138 
139  // check that surfaces are parallel
140  if (Mth::DotProduct(proposed_surface.normal, check_surface.normal) < 0.99f) continue;
141 
142  // check that collision lines are parallel
143  if (Mth::Abs(Mth::DotProduct(p_line_directions[proposed_idx], p_line_directions[check_idx])) < 0.99f) continue;
144 
145  // check that collision lines connect
146  if (very_close(proposed_surface.ends[0].point, check_surface.ends[0].point))
147  {
148  proposed_surface.ends[0].point = check_surface.ends[1].point;
149  proposed_surface.ends[0].tangent_exists = check_surface.ends[1].tangent_exists;
150  proposed_surface.ends[0].tangent = check_surface.ends[1].tangent;
151 
152  }
153  else if (very_close(proposed_surface.ends[0].point, check_surface.ends[1].point))
154  {
155  proposed_surface.ends[0].point = check_surface.ends[0].point;
156  proposed_surface.ends[0].tangent_exists = check_surface.ends[0].tangent_exists;
157  proposed_surface.ends[0].tangent = check_surface.ends[0].tangent;
158  }
159  else if (very_close(proposed_surface.ends[1].point, check_surface.ends[0].point))
160  {
161  proposed_surface.ends[1].point = check_surface.ends[1].point;
162  proposed_surface.ends[1].tangent_exists = check_surface.ends[1].tangent_exists;
163  proposed_surface.ends[1].tangent = check_surface.ends[1].tangent;
164  }
165  else if (very_close(proposed_surface.ends[1].point, check_surface.ends[1].point))
166  {
167  proposed_surface.ends[1].point = check_surface.ends[0].point;
168  proposed_surface.ends[1].tangent_exists = check_surface.ends[0].tangent_exists;
169  proposed_surface.ends[1].tangent = check_surface.ends[0].tangent;
170  }
171  else
172  {
173  continue;
174  }
175 
176  // we've successfully absorbed check_surface into proposed_surface
177  p_surfaces_used[check_idx] = true;
178 
179  clean_pass = false;
180  } // END loop over unvisited surfaces
181  } while (clean_pass = false);
182 
183  // accept the poropsed surface into the reduced surface list
184 
185  m_merged_coll_data.p_surfaces[m_merged_coll_data.num_surfaces++] = proposed_surface;
186  } // END loop over proposing surfaces
187 }

189 void CRectFeeler::DebugLines ( int r, int g, int b, int num_frames ) const
190 {
191  uint32 color = MAKE_RGB(r, g, b);
192 
193  Gfx::AddDebugLine(m_corner, m_corner + m_first_edge, color, color, num_frames);
194  Gfx::AddDebugLine(m_corner + m_first_edge, m_corner + m_first_edge + m_second_edge, color, color, num_frames);
195  Gfx::AddDebugLine(m_corner + m_first_edge + m_second_edge, m_corner + m_second_edge, color, color, num_frames);
196  Gfx::AddDebugLine(m_corner + m_second_edge, m_corner, color, color, num_frames);
197 
198  // if we've merged the surfaces
199  if (m_merged_coll_data.num_surfaces > 0)
200  {
201  // draw the merged collision lines
202  for (int n = m_merged_coll_data.num_surfaces; n--; )
203  {
204  color = MAKE_RGB(r += 201, g += 99, b += 31);
205  Gfx::AddDebugLine(m_merged_coll_data.p_surfaces[n].ends[0].point, m_merged_coll_data.p_surfaces[n].ends[1].point, color, color, 1);
206  }
207  }
208  else
209  {
210  // otherwise, draw the bare collision lines
211  for (int n = m_coll_data.num_surfaces; n--; )
212  {
213  color = MAKE_RGB(r += 201, g += 99, b += 31);
214  Gfx::AddDebugLine(m_coll_data.p_surfaces[n].ends[0].point, m_coll_data.p_surfaces[n].ends[1].point, color, color, 1);
215  }
216  }
217 }
