44 void NsMatrix::translate ( NsVector * v, NsMatrix_Combine c )
45 {
46  Mtx temp;
47  switch (c) {
48  case NsMatrix_Combine_Replace:
49  MTXTrans( m_matrix, v->x, v->y, v->z );
50  break;
51  case NsMatrix_Combine_Pre:
52  MTXTrans( temp, v->x, v->y, v->z );
53  MTXConcat( m_matrix, temp, m_matrix );
54  break;
55  case NsMatrix_Combine_Post:
56  MTXTrans( temp, v->x, v->y, v->z );
57  MTXConcat( temp, m_matrix, m_matrix );
58  break;
59  default:
60  break;
61  }

173 void NsMatrix::getRotation( NsVector * axis, float * angle, NsVector * center )
174 {
175  float nTwoSinTheta, nTwoCosTheta;
176  NsVector vTwoSinThetaAxis;
177  float nRadians;
178 // Mtx mISubMat;
179 // Mtx mISubMatInverse;
180 
181 
182  nTwoCosTheta = getRight()->x + getUp()->y + getAt()->z - 1.0f;
183 
184  // Dave 08/07/01 - Unsure as to exactly why the negation of the axis fixes the slerping problem
185  // (getRotation() is called from NsSlerp::SetMatrices(), but it does. Investigate further at some point...
186 // vTwoSinThetaAxis.x = getUp()->z - getAt()->y;
187 // vTwoSinThetaAxis.y = getAt()->x - getRight()->z;
188 // vTwoSinThetaAxis.z = getRight()->y - getUp()->x;
189  vTwoSinThetaAxis.x = getAt()->y - getUp()->z;
190  vTwoSinThetaAxis.y = getRight()->z - getAt()->x;
191  vTwoSinThetaAxis.z = getUp()->x - getRight()->y;
192 
193  nTwoSinTheta = vTwoSinThetaAxis.length();
194  if (nTwoSinTheta > 0.0f)
195  {
196  float recipLength = (1.0f / (nTwoSinTheta));
197 
198  axis->scale( vTwoSinThetaAxis, recipLength );
199  }
200  else
201  {
202  axis->set( 0.0f, 0.0f, 0.0f );
203  }
204  nRadians = atan2(nTwoSinTheta, nTwoCosTheta);
205  (*angle) = nRadians * (((float) 180) / ((float) Mth::PI));
206  if ((nTwoSinTheta <= 0.01f) && (nTwoCosTheta <= 0.0f))
207  {
208  /*
209  * sin theta is 0; cos theta is -1; theta is 180 degrees
210  * vTwoSinThetaAxis was degenerate
211  * axis will have to be found another way.
212  */
213 
214  NsVector vTwoSinThetaAxis;
215 
216  /*
217  * Matrix is:
218  * [ [ 2 a_x^2 - 1, 2 a_x a_y, 2 a_x a_z, 0 ]
219  * [ 2 a_x a_y, 2 a_y^2 - 1, 2 a_y a_z, 0 ]
220  * [ 2 a_x a_z, 2 a_y a_z, 2 a_z^2 - 1, 0 ]
221  * [ 0, 0, 0, 1 ] ]
222  * Build axis scaled by 4 * component of maximum absolute value
223  */
224  if (getRight()->x > getUp()->y)
225  {
226  if (getRight()->x > getAt()->z)
227  {
228  vTwoSinThetaAxis.x = 1.0f + getRight()->x;
229  vTwoSinThetaAxis.x = vTwoSinThetaAxis.x + vTwoSinThetaAxis.x;
230  vTwoSinThetaAxis.y = getRight()->y + getUp()->x;
231  vTwoSinThetaAxis.z = getRight()->z + getAt()->x;
232  }
233  else
234  {
235  vTwoSinThetaAxis.z = 1.0f + getAt()->z;
236  vTwoSinThetaAxis.z = vTwoSinThetaAxis.z + vTwoSinThetaAxis.z;
237  vTwoSinThetaAxis.x = getAt()->x + getRight()->z;
238  vTwoSinThetaAxis.y = getAt()->y + getUp()->z;
239  }
240  }
241  else
242  {
243  if (getUp()->y > getAt()->z)
244  {
245  vTwoSinThetaAxis.y = 1.0f + getUp()->y;
246  vTwoSinThetaAxis.y = vTwoSinThetaAxis.y + vTwoSinThetaAxis.y;
247  vTwoSinThetaAxis.z = getUp()->z + getAt()->y;
248  vTwoSinThetaAxis.x = getUp()->x + getRight()->y;
249  }
250  else
251  {
252  vTwoSinThetaAxis.z = 1.0f + getAt()->z;
253  vTwoSinThetaAxis.z = (vTwoSinThetaAxis.z + vTwoSinThetaAxis.z);
254  vTwoSinThetaAxis.x = getAt()->x + getRight()->z;
255  vTwoSinThetaAxis.y = getAt()->y + getUp()->z;
256  }
257  }
258 
259  /*
260  * and normalize the axis
261  */
262 
263  axis->normalize( vTwoSinThetaAxis );
264  }

268 // * [ v - c ] * R + c = v * R + s
269 // * -> v * R + c - c * R = v * R + s
270 // * -> c * [ I - R ] = s
271 // * -> c = s * [ I - R ] ^ -1
272 // */
