96 CBaseComponent::EMemberFunctionResult CSkaterSoundComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
97 {
98  switch ( Checksum )
99  {
100  // @script | PlayJumpSound |
101  case CRCC(0xe338369, "PlayJumpSound"):
102  {
103  switch ( m_stateType )
104  {
105  case GROUND:
106  case AIR:
107  PlayJumpSound(GetObject()->GetVel().Length() / m_max_speed);
108  break;
109 
110  case RAIL:
111  PlayJumpSound(GetObject()->GetVel().Length() / m_max_speed);
112 
113  default:
114  break;
115  }
116  break;
117  }
118 
119  // @script | PlayLandSound |
120  case CRCC(0x2c779f22, "PlayLandSound"):
121  {
122  switch ( m_stateType )
123  {
124  case GROUND:
125  case AIR:
126  PlayLandSound(Mth::Abs(GetObject()->GetVel()[Y]) / m_max_speed);
127  break;
128 
129  case RAIL:
130  PlayLandSound(GetObject()->GetVel().Length() / m_max_speed);
131  break;
132 
133  default:
134  break;
135  }
136  break;
137  }
138 
139  // @script | PlayBonkSound |
140  case CRCC(0x0069e457, "PlayBonkSound"):
141  {
142  PlayBonkSound(GetObject()->GetVel().Length() / m_max_speed);
143  break;
144  }
145 
146  // @script | PlayCessSound |
147  case CRCC(0x6f5e9124, "PlayCessSound"):
148  {
149  PlayCessSound(GetObject()->GetVel().Length() / m_max_speed);
150  break;
151  }
152 
153  default:
154  return CBaseComponent::MF_NOT_EXECUTED;
155  }
156 
157  return CBaseComponent::MF_TRUE;
158 }

198 void CSkaterSoundComponent::PlayLandSound ( float speed_fraction, ETerrainType terrain )
199 {
200  Env::ETerrainActionType table;
201  if ( m_stateType == RAIL )
202  {
203  table = m_is_rail_sliding ? Env::vTABLE_SLIDELAND : Env::vTABLE_GRINDLAND;
204  }
205  else
206  {
207  table = Env::vTABLE_LAND;
208  }
209 
210  speed_fraction *= 100.0f;
211 
212  #ifdef __NOPT_ASSERT__
213  if (Script::GetInteger(CRCD(0xb9ba2d27, "debug_skater_triggered_sounds")))
214  {
215  Dbg_Message("Playing sound [land]: %.1f", speed_fraction);
216  }
217  #endif
218 
219  Env::CTerrainManager::sPlaySound(table, terrain, GetObject()->GetPos(), speed_fraction * m_vol_mult, speed_fraction, speed_fraction);
220 }
