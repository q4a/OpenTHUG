114 bool CSkatetrisGoal::Update()
115 {
116  int wait_to_add_tricks = 0;
117  mp_params->GetInteger( CRCD(0x7824963c,"wait_to_add_tricks"), &wait_to_add_tricks, Script::NO_ASSERT );
118  if ( wait_to_add_tricks )
119  {
120  // bail out...
121  return CGoal::Update();
122  }
123 
124  int minTimeLeft = 1;
125  if ( !IsSingleComboSkatetris() )
126  {
127  if ( mp_params->GetInteger( CRCD(0xa6d96603,"time_to_stop_adding_tricks"), &minTimeLeft, Script::NO_ASSERT ) )
128  minTimeLeft *= 1000;
129  }
130 
131  // check for win conditions
132  if ( IsTricktris() )
133  {
134  if ( IsActive() && !IsPaused() )
135  {
136  // tricktris is won when you've completed the pre-determined number
137  // of tricks
138  int tricks_completed = 0;
139  mp_params->GetInteger( CRCD(0x8d958f01,"tricks_completed"), &tricks_completed, Script::NO_ASSERT );
140  int total_to_win;
141  mp_params->GetInteger( CRCD(0xbfd3a831,"tricktris_total_to_win"), &total_to_win, Script::ASSERT );
142  // printf("tricks_completed = %i, total_to_win = %i\n", tricks_completed, total_to_win);
143  if ( tricks_completed >= total_to_win )
144  {
145  CGoalManager* pGoalManager = GetGoalManager();
146  pGoalManager->WinGoal( GetGoalId() );
147  return true;
148  }
149  }
150  }
151  else if ( IsActive() && (int)m_timeLeft < minTimeLeft && !IsPaused() && AllTricksCleared() )
152  {
153  // win the goal if they've cleared all tricks and there's no time to add more
154  CGoalManager* pGoalManager = GetGoalManager();
155  pGoalManager->WinGoal( GetGoalId() );
156  return true;
157  }
158 
159  // update faded tricks if we're active - even if the time has run out
160  if ( IsActive() && !IsPaused() )
161  {
162  UpdateFadedTricks();
163  }
164 
165  if ( IsTricktris() )
166  {
167  if ( ( m_unlimitedTime || (int)m_timeLeft > 0.0f ) && !IsPaused() )
168  {
169  int tricks_completed = 0;
170  mp_params->GetInteger( CRCD(0x8d958f01,"tricks_completed"), &tricks_completed, Script::NO_ASSERT );
171  int total_to_win;
172  mp_params->GetInteger( CRCD(0xbfd3a831,"tricktris_total_to_win"), &total_to_win, Script::ASSERT );
173 
174  // count number of valid entries and add if necessary
175  int total_in_stack = 0;
176  for ( int i = 0; i < Game::vMAXTETRISTRICKS; i++ )
177  {
178  if ( m_tetrisTricks[i].invalid == false )
179  total_in_stack++;
180  }
181  int desired_stack_size;
182  mp_params->GetInteger( CRCD(0x45a375c9,"tricktris_block_size"), &desired_stack_size, Script::ASSERT );
183  if ( total_in_stack < desired_stack_size && total_in_stack + tricks_completed < total_to_win )
184  {
185  int num_to_add = desired_stack_size - total_in_stack;
186  if ( total_to_win - tricks_completed - total_in_stack < num_to_add )
187  num_to_add = total_to_win - tricks_completed - total_in_stack;
188  AddTetrisTrick( num_to_add );
189  }
190  }
191  }
192  else if ( IsActive() && ( m_unlimitedTime || (int)m_timeLeft >= minTimeLeft ) && !IsPaused() )
193  {
194  m_tetrisTime += (int)( Tmr::FrameLength() * 1000 );
195 
196  if ( IsSingleComboSkatetris() )
197  {
198  // throw in a delay so the single combo doesn't appear right away
199  // during the goal intro movie
200  if ( !mp_params->ContainsFlag( CRCD(0xc93da683,"single_combo_added") ) && m_tetrisTime > 50 )
201  {
202  int combo_size;
203  mp_params->GetInteger( CRCD(0x67dd90ca,"combo_size"), &combo_size, Script::ASSERT );
204  AddTetrisTrick( combo_size );
205  mp_params->AddChecksum( NONAME, CRCD(0xc93da683,"single_combo_added") );
206  }
207  }
208  else
209  {
210  int trickTime;
211  mp_params->GetInteger( CRCD(0x648d8d2a,"adjusted_trick_time"), &trickTime, Script::ASSERT );
212  if ( m_tetrisTime > trickTime )
213  {
214  m_tetrisTime = 0;
215  if ( IsComboSkatetris() )
216  {
217  int combo_size;
218  mp_params->GetInteger( CRCD(0x67dd90ca,"combo_size"), &combo_size, Script::ASSERT );
219  AddTetrisTrick( combo_size );
220  }
221  else
222  AddTetrisTrick();
223  }
224  }
225  }
226  return CGoal::Update();
227 }

294 void CSkatetrisGoal::AddTetrisTrick( int num_to_add )
295 {
296  int max_tricks;
297  mp_params->GetInteger( CRCD(0x89473db7,"max_tricks"), &max_tricks, Script::ASSERT );
298 
299  int tricks_in_group = num_to_add;
300  int possible_num_tricks = m_numTetrisTricks + tricks_in_group;
301 
302  bool add_combo = false;
303  if ( IsComboSkatetris() || IsSingleComboSkatetris() )
304  add_combo = true;
305 
306 /* if ( add_combo )
307  {
308  mp_params->GetInteger( CRCD(0x67dd90ca,"combo_size"), &tricks_in_group, Script::ASSERT );
309  possible_num_tricks += tricks_in_group;
310  }
311  else
312  {
313  possible_num_tricks++;
314  }
315 */
316 
317  // check that they haven't lost
318  if ( possible_num_tricks > max_tricks )
319  {
320  uint32 goalId = GetGoalId();
321  Dbg_Assert( goalId );
322  CGoalManager* pGoalManager = GetGoalManager();
323  Dbg_Assert( pGoalManager );
324  pGoalManager->LoseGoal( goalId );
325  }
326  else
327  {
328  // set the group id and increment number of trick groups
329  int group_id = 0;
330  if ( add_combo )
331  {
332  for ( int i = 0; i < Game::vMAXTETRISTRICKS; i++ )
333  {
334  if ( !m_validGroups[i] )
335  {
336  group_id = i;
337  m_validGroups[group_id] = true;
338  break;
339  }
340  }
341  Dbg_Assert( !( group_id == Game::vMAXTETRISTRICKS ) );
342  }
343 
344  // add as many tricks as we need (1 by default)
345  while ( tricks_in_group > 0 )
346  {
347  Front::CScreenElementManager* p_screen_elem_man = Front::CScreenElementManager::Instance();
348 
349  // grab list of valid key combos
350  Script::CArray* p_keyCombos = NULL;
351  mp_params->GetArray( CRCD(0x7be1e689,"goal_tetris_key_combos"), &p_keyCombos, Script::NO_ASSERT );
352 
353  uint32 key_combo = 0;
354  uint32 trick_name_checksum = 0;
355  char trick_text[Game::NEW_STRING_LENGTH];
356 
357  // find the first empty trick
358  int index = 0;
359 
360  for ( int i = 0; i < Game::vMAXTETRISTRICKS; i++ )
361  {
362  if ( m_tetrisTricks[i].invalid == true )
363  {
364  index = i;
365  m_tetrisTricks[index].invalid = false;
366  m_tetrisTricks[index].group_id = group_id;
367  m_tetrisTricks[index].keyCombo = 0;
368  m_tetrisTricks[index].trickNameChecksum = 0;
369  m_tetrisTricks[index].altTrickNameChecksum = 0;
370  m_tetrisTricks[index].spin_mult = 0;
371  m_tetrisTricks[index].num_taps = 1;
372  m_tetrisTricks[index].require_perfect = false;
373  break;
374  }
375  }
376  Dbg_Assert( !( index == Game::vMAXTETRISTRICKS ) );
377 
378  // check the type of the structure...this will tell us if this is a standard
379  // skatetris goal (array of key combos), or a special one (array of structures
380  // containing the name and text of the trick).
381  const char* p_key_combo_text = NULL;
382  int num_taps = 1;
383  if ( !p_keyCombos )
384  {
385  // check for a trick array
386  Script::CArray* p_tetris_tricks = NULL;
387  mp_params->GetArray( CRCD(0xeb79fb49,"goal_tetris_tricks"), &p_tetris_tricks, Script::ASSERT );
388 
389  // grab a random trick
390  // no safety checks, since we don't care if the trick is mapped
391  Script::CStruct* p_trick = p_tetris_tricks->GetStructure( Mth::Rnd( p_tetris_tricks->GetSize() ) );
392  const char* p_trick_display_text;
393  p_trick->GetString( CRCD(0x270f56e1,"trick"), &p_trick_display_text, Script::ASSERT );
394 
395  if ( p_trick->ContainsFlag( CRCD( 0x255ed86f, "grind" ) ) )
396  {
397  // store FS and BS versions of the trick
398  char grind_string[128];
399  strcpy( grind_string, "FS " );
400  strcat( grind_string, p_trick_display_text );
401  m_tetrisTricks[index].trickNameChecksum = Script::GenerateCRC( grind_string );
402 
403  strcpy( grind_string, "BS " );
404  strcat( grind_string, p_trick_display_text );
405  m_tetrisTricks[index].altTrickNameChecksum = Script::GenerateCRC( grind_string );
406  }
407  else
408  {
409  m_tetrisTricks[index].trickNameChecksum = Script::GenerateCRC( p_trick_display_text );
410  }
411 
412  const char* p_text;
413  p_trick->GetString( CRCD(0xc4745838,"text"), &p_text, Script::ASSERT );
414  strcpy( trick_text, p_text );
415 
416  if ( !p_trick->GetChecksum( CRCD(0xacfdb27a,"key_combo"), &key_combo, Script::NO_ASSERT ) )
417  p_trick->GetString( CRCD(0xacfdb27a,"key_combo"), &p_key_combo_text, Script::NO_ASSERT );
418 
419  // printf("setting trick checksum: %s\n", Script::FindChecksumName( trick_name_checksum ) );
420  }
421  else
422  {
423  // see if this is a valid key_combo
424  int key_combo_index = GetRandomIndexFromKeyCombos( p_keyCombos );
425  if ( key_combo_index == -1 )
426  {
427  // couldn't find one - the goal is going to be decativated.
428  return;
429  }
430  if ( p_keyCombos->GetType() == ESYMBOLTYPE_NAME )
431  {
432  key_combo = p_keyCombos->GetChecksum( key_combo_index );
433  }
434  else
435  {
436  Dbg_MsgAssert( p_keyCombos->GetType() == ESYMBOLTYPE_STRUCTURE, ( "goal_tetris_key_combos array has bad type" ) );
437  Script::CStruct* p_subStruct = p_keyCombos->GetStructure( key_combo_index );
438  p_subStruct->GetChecksum( CRCD(0xacfdb27a,"key_combo"), &key_combo, Script::ASSERT );
439 
440  int spin_mult = 0;
441  if ( p_subStruct->GetInteger( CRCD(0x8c8abd19,"random_spin"), &spin_mult, Script::NO_ASSERT ) )
442  {
443  // make sure it's a multiple of 180
444  Dbg_MsgAssert( spin_mult % 180 == 0, ( "Skatetris goal %s has bad spin_mult value %i", Script::FindChecksumName( GetGoalId() ), spin_mult ) );
445  if (spin_mult < 360)
446  {
447  spin_mult=360;
448  }
449 
450  // K: Make it so that spins of 180 are excluded, since they are too hard to do on vert.
451  // Ie:
452  // spin_mult=360 gives spins of 0, or 360
453  // spin_mult=540 gives spins of 0, 360 or 540
454  // spin_mult=720 gives spins of 0, 360, 540 or 720
455  // etc ...
456  m_tetrisTricks[index].spin_mult = Mth::Rnd(spin_mult / 180) + 1;
457  if (m_tetrisTricks[index].spin_mult == 1)
458  {
459  m_tetrisTricks[index].spin_mult=0;
460  }
461  }
462  if ( p_subStruct->GetInteger( CRCD(0xedf5db70,"spin"), &spin_mult, Script::NO_ASSERT ) )
463  {
464  // make sure it's a multiple of 180
465  Dbg_MsgAssert( spin_mult % 180 == 0, ( "Skatetris goal %s has bad spin_mult value %i", Script::FindChecksumName( GetGoalId() ), spin_mult ) );
466  m_tetrisTricks[index].spin_mult = spin_mult / 180;
467  }
468 
469  if ( p_subStruct->ContainsFlag( CRCD(0x1c39f1b9,"perfect") ) )
470  m_tetrisTricks[index].require_perfect = true;
471 
472  if ( p_subStruct->GetInteger( CRCD(0xa4bee6a1,"num_taps"), &num_taps, Script::NO_ASSERT ) )
473  m_tetrisTricks[index].num_taps = num_taps;
474  }
475 
476  // get the global trick mapping
477  Mdl::Skate* skate_mod = Mdl::Skate::Instance();
478  Obj::CSkaterProfile* p_SkaterProfile = skate_mod->GetCurrentProfile();
479  Script::CStruct* p_trickMappings = p_SkaterProfile->GetTrickMapping( CRCD(0xd544aa2d,"trick_mapping") );
480 
481  const char* p_text = NULL;
482 
483  // if it's not a regular assignment, grab cat info
484  if ( !p_trickMappings->GetChecksum( key_combo, &trick_name_checksum, Script::NO_ASSERT ) )
485  {
486  int cat_index;
487  p_trickMappings->GetInteger( key_combo, &cat_index, Script::ASSERT );
488  Dbg_Assert( cat_index >= 0 && cat_index < vMAX_CREATED_TRICKS );
489  Obj::CSkater* pSkater = skate_mod->GetLocalSkater();
490  if ( pSkater )
491  {
492  Game::CCreateATrick* pCreatedTrick = pSkater->m_created_trick[cat_index];
493  pCreatedTrick->mp_other_params->GetString( CRCD( 0xa1dc81f9, "name" ), &p_text, Script::ASSERT );
494 
495  // mark it
496  // no cat tricks have double versions!
497  m_tetrisTricks[index].num_taps = 1;
498  }
499  }
500 
501  if ( !p_text )
502  {
503  Script::CStruct* p_trick_structure = NULL;
504  p_trick_structure = Script::GetStructure( trick_name_checksum, Script::NO_ASSERT );
505  if ( !p_trick_structure )
506  {
507  Script::CArray* p_trick_array = Script::GetArray( trick_name_checksum, Script::ASSERT );
508  p_trick_structure = p_trick_array->GetStructure( 0 );
509 
510  Dbg_MsgAssert( p_trick_structure, ( "Could not find trick structure" ) );
511  }
512  Script::CStruct* p_trick_params;
513  p_trick_structure->GetStructure( CRCD(0x7031f10c,"Params"), &p_trick_params, Script::ASSERT );
514 
515 
516  // get any multiple tap info
517  while ( num_taps > 1 )
518  {
519  num_taps--;
520 
521  uint32 extra_trick;
522  if ( p_trick_params->GetChecksum( CRCD(0x6e855102,"ExtraTricks"), &extra_trick, Script::NO_ASSERT ) )
523  {
524  Script::CArray* p_extra_trick_array = Script::GetArray( extra_trick, Script::ASSERT );
525  Dbg_MsgAssert( p_extra_trick_array->GetType() == ESYMBOLTYPE_STRUCTURE, ( "ExtraTricks array %s has wrong type", Script::FindChecksumName( extra_trick ) ) );
526  Script::CStruct* p_extra_trick_struct = p_extra_trick_array->GetStructure( 0 );
527  p_extra_trick_struct->GetStructure( CRCD(0x7031f10c,"Params"), &p_trick_params, Script::ASSERT );
528  }
529  else
530  {
531  num_taps = 1;
532  m_tetrisTricks[index].num_taps = 1;
533  }
534  }
535 
536  p_trick_params->GetLocalText( CRCD(0xa1dc81f9,"Name"), &p_text, Script::ASSERT );
537  }
538 
539  // build the text string
540  strcpy( trick_text, "" );
541 
542  // spin value
543  if ( m_tetrisTricks[index].spin_mult != 0 )
544  {
545  // perfect modifier
546  if ( m_tetrisTricks[index].require_perfect )
547  sprintf( trick_text, "Perfect " );
548 
549  sprintf( trick_text, "\\c5%i\\c0 ", ( m_tetrisTricks[index].spin_mult * 180 ) );
550  }
551  // trick text
552  strcat( trick_text, p_text );
553 
554  // store key combo
555  m_tetrisTricks[index].keyCombo = key_combo;
556  }
557 
558  // create the container for this element
559  Script::CStruct *p_container_params = new Script::CStruct();
560  Front::CContainerElement *p_container = new Front::CContainerElement();
561  uint32 container_id = Obj::CBaseManager::vNO_OBJECT_ID;
562  p_container->SetID( container_id );
563  p_screen_elem_man->RegisterObject( *p_container );
564 
565  // figure the dims
566  int y_dim = 20;
567  if ( add_combo && tricks_in_group == 1 )
568  {
569  y_dim *= 2;
570  }
571  p_container_params->AddPair( CRCD(0x34a68574,"dims"), 100, y_dim );
572 
573  p_container->SetProperties( p_container_params );
574 
575  // set the parent
576  Front::CScreenElement* p_vmenu = p_screen_elem_man->GetElement(CRCD(0xe35ec715,"tetris_tricks_menu"), Front::CScreenElementManager::ASSERT );
577  p_vmenu->SetChildLockState( Front::CScreenElement::UNLOCK );
578  p_screen_elem_man->SetParent( p_vmenu, p_container );
579

581

583  Script::CStruct *p_elem_params = new Script::CStruct();
584  p_elem_params->AddString( CRCD(0xc4745838,"text"), trick_text );
