17 CRefAsset::CRefAsset(CAsset *p_asset)
18 {
19 
20  mp_asset = p_asset;
21  mp_sibling = NULL;
22  if (p_asset->mp_ref_asset == NULL)
23  {
24  // this is the first reference, so just stick it in mp_ref_asset
25  p_asset->mp_ref_asset = this;
26  }
27  else
28  {
29  // it's a new asset, so insert it as the head
30  // of siblings that refer to this
31  CRefAsset *p_other_ref = p_asset->mp_ref_asset;
32  p_asset->mp_ref_asset = this;
33  mp_sibling = p_other_ref;
34  }
35  mp_data = NULL; // Ref Assets explicitly have no data
36  m_permanent = p_asset->m_permanent; // inherit the permanence flag
37 }

41 CRefAsset::~CRefAsset()
42 {
43 
44 // unhook the references
45 // from the assman table, the asset it refers to, and other assets
46 
47  if (mp_asset)
48  {
49  CRefAsset * p_prev = mp_asset->mp_ref_asset;
50  if (p_prev == this)
51  {
52  mp_asset->mp_ref_asset = mp_sibling;
53  }
54  else
55  {
56  while (p_prev && p_prev->mp_sibling != this)
57  {
58  p_prev = p_prev->mp_sibling;
59  }
60  Dbg_MsgAssert(p_prev,("Reference not listed in its parent asset"));
61  p_prev->mp_sibling = mp_sibling;
62  }
63  mp_asset = NULL;
64  mp_sibling = NULL;
65  }
66 
67 
68  Dbg_MsgAssert(mp_asset == NULL,("Destroying ref with non-null asset"));
69  Dbg_MsgAssert(mp_sibling == NULL,("Destroying ref with non-null sibling"));
70  Dbg_MsgAssert(mp_data == NULL,("Ref Asset has data!"));
71 
72 }
