469  // Replace translation component.

472  mp_instance->SetTransform( rot_mat );

646 /******************************************************************/
647 /* */
648 /* */
649 /******************************************************************/
650 bool CXboxGeom::plat_set_material_color( uint32 mat_name_checksum, int pass, Image::RGBA rgba )
651 {
652  bool something_changed = false;
653 
654  if( mp_instance && mp_instance->GetScene())
655  {
656  for( int m = 0; m < mp_instance->GetScene()->m_num_mesh_entries; ++m )
657  {
658  NxXbox::sMesh *p_mesh = mp_instance->GetScene()->m_meshes[m];
659  NxXbox::sMaterial *p_mat = p_mesh->mp_material;
660  if( p_mat )
661  {
662  bool want_this_material = false;
663  int adjusted_pass = pass;
664 
665  // We are searching for materials with a matching name. However there is a caveat in that the
666  // conversion process sometimes creates new materials for geometry flagged as 'render as separate', or for
667  // geometry which is mapped with only certain passes of a multipass material (or both cases).
668  // In such a case, the new material name checksum will differ from the original material name checksum,
669  // with the following bits having significance (i.e. consider bitflags = new_namechecksum - old_namechecksum ):
670  //
671  // Bits 3->6 Pass flags indicating which passes of the original material this material uses
672  // Bits 0->2 Absolute number ('render as separate' flagged geometry) indicating which single pass of the material this material represents.
673  if( p_mat->m_name_checksum == mat_name_checksum )
674  want_this_material = true;
675  else if(( p_mat->m_name_checksum > mat_name_checksum ) && (( p_mat->m_name_checksum - mat_name_checksum ) <= 0x7F ))
676  {
677  uint32 checksum_diff = p_mat->m_name_checksum - mat_name_checksum;
678  uint32 render_separate_pass = checksum_diff & 0x07;
679  uint32 pass_flags = checksum_diff >> 3;
680 
681  if( render_separate_pass )
682  {
683  if( render_separate_pass == ( pass + 1 ))
684  {
685  want_this_material = true;
686 
687  // Readjust the pass to zero, since this material was formed as a single pass of a multipass material.
688  adjusted_pass = 0;
689  }
690  }
691  else if( pass_flags )
692  {
693  // This material was created during scene conversion from another material with more passes.
694  if( pass_flags & ( 1 << pass ))
695  {
696  want_this_material = true;
697  for( int p = 0; p < pass; ++p )
698  {
699  if(( pass_flags & ( 1 << p )) == 0 )
700  {
701  // Readjust the pass down by 1, since this material was created as a subset of another material.
702  --adjusted_pass;
703  }
704  }
705  }
706  }
707  }
708 
709  if( want_this_material )
710  {
711  if((uint32)adjusted_pass < p_mat->m_passes )
712  {
713  if( !( p_mat->m_flags[adjusted_pass] & MATFLAG_PASS_COLOR_LOCKED ))
714  {
715  p_mat->m_color[adjusted_pass][0] = (float)rgba.r / 255.0f;
716  p_mat->m_color[adjusted_pass][1] = (float)rgba.g / 255.0f;
717  p_mat->m_color[adjusted_pass][2] = (float)rgba.b / 255.0f;
718 
719  something_changed = true;
720  }
721  }
722  }
723  }
724  }
725  }
726  return something_changed;
727 }

1129 /* */
1130 /******************************************************************/
1131 bool CXboxGeom::plat_set_uv_matrix( uint32 mat_name_checksum, int pass, const Mth::Matrix& mat )
1132 {
1133  if( mp_instance && mp_instance->GetScene())
1134  {
1135  for( int m = 0; m < mp_instance->GetScene()->m_num_mesh_entries; ++m )
1136  {
1137  NxXbox::sMesh *p_mesh = mp_instance->GetScene()->m_meshes[m];
1138  NxXbox::sMaterial *p_mat = p_mesh->mp_material;
1139  if( p_mat )
1140  {
1141  bool want_this_material = false;
1142  int adjusted_pass = pass;
1143 
1144  // We are searching for materials with a matching name. However there is a caveat in that the
1145  // conversion process sometimes creates new materials for geometry flagged as 'render as separate', or for
1146  // geometry which is mapped with only certain passes of a multipass material (or both cases).
1147  // In such a case, the new material name checksum will differ from the original material name checksum,
1148  // with the following bits having significance (i.e. consider bitflags = new_namechecksum - old_namechecksum ):
1149  //
1150  // Bits 3->6 Pass flags indicating which passes of the original material this material uses
1151  // Bits 0->2 Absolute number ('render as separate' flagged geometry) indicating which single pass of the material this material represents.
1152  if( p_mat->m_name_checksum == mat_name_checksum )
1153  {
1154  want_this_material = true;
1155  }
1156  else if(( p_mat->m_name_checksum > mat_name_checksum ) && (( p_mat->m_name_checksum - mat_name_checksum ) <= 0x7F ))
1157  {
1158  uint32 checksum_diff = p_mat->m_name_checksum - mat_name_checksum;
1159  uint32 render_separate_pass = checksum_diff & 0x07;
1160  uint32 pass_flags = checksum_diff >> 3;
1161 
1162  if( render_separate_pass )
1163  {
1164  if( render_separate_pass == ( pass + 1 ))
1165  {
1166  want_this_material = true;
1167 
1168  // Readjust the pass to zero, since this material was formed as a single pass of a multipass material.
1169  adjusted_pass = 0;
1170  }
1171  }
1172  else if( pass_flags )
1173  {
1174  // This material was created during scene conversion from another material with more passes.
1175  if( pass_flags & ( 1 << pass ))
1176  {
1177  want_this_material = true;
1178  for( int p = 0; p < pass; ++p )
1179  {
1180  if(( pass_flags & ( 1 << p )) == 0 )
1181  {
1182  // Readjust the pass down by 1, since this material was created as a subset of another material.
1183  --adjusted_pass;
1184  }
1185  }
1186  }
1187  }
1188  }
1189 
1190  if( want_this_material )
1191  {
1192  if((uint32)adjusted_pass < p_mat->m_passes )
1193  {
1194  // Create the wibble params if they don't exist already.
1195  if( p_mat->mp_UVWibbleParams[adjusted_pass] == NULL )
1196  {
1197  p_mat->mp_UVWibbleParams[adjusted_pass] = new NxXbox::sUVWibbleParams;
1198 
1199  // Need to set flags to indicate that uv wibble is now in effect.
1200  p_mat->m_uv_wibble = true;
1201  p_mat->m_flags[adjusted_pass] |= MATFLAG_UV_WIBBLE | MATFLAG_EXPLICIT_UV_WIBBLE;
1202  }
1203 
1204  // Also need to switch vertex shaders if the current vertex shader does not support UV Transforms.
1205  if( p_mesh->m_vertex_shader[0] == WeightedMeshVS_VXC_1Weight )
1206  {
1207  p_mesh->m_vertex_shader[0] = WeightedMeshVS_VXC_1Weight_UVTransform;
1208  }
1209  else if( p_mesh->m_vertex_shader[0] == WeightedMeshVS_VXC_2Weight )
1210  {
1211  p_mesh->m_vertex_shader[0] = WeightedMeshVS_VXC_2Weight_UVTransform;
1212  }
1213  else if( p_mesh->m_vertex_shader[0] == WeightedMeshVS_VXC_3Weight )
1214  {
1215  p_mesh->m_vertex_shader[0] = WeightedMeshVS_VXC_3Weight_UVTransform;
1216  }
1217 
1218  // Set the matrix values.
1219  p_mat->mp_UVWibbleParams[adjusted_pass]->m_UVMatrix[0] = mat[0][0];
1220  p_mat->mp_UVWibbleParams[adjusted_pass]->m_UVMatrix[1] = mat[0][1];
1221  p_mat->mp_UVWibbleParams[adjusted_pass]->m_UVMatrix[2] = mat[3][0];
1222  p_mat->mp_UVWibbleParams[adjusted_pass]->m_UVMatrix[3] = mat[3][1];
1223  }
1224  }
1225  }
1226  }
1227  }
1228  return true;
1229 }
