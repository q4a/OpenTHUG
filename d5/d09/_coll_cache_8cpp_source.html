80 void CCollCache::Update(const Mth::CBBox &bbox)
81 {
82 #if PRINT_TIMES
83  static uint64 s_total_time = 0, s_num_collisions = 0;
84  uint64 start_time = Tmr::GetTimeInUSeconds();
85 #endif
86 
87  // Clear old cache first
88  Clear();
89 
90  // Make line
91  Mth::Line is(bbox.GetMin(), bbox.GetMax());
92 
93  // initialize starting "max" distance to be max collision distance
94  SSec::Manager *ss_man;
95  ss_man = Nx::CEngine::sGetNearestSuperSectorManager(is);
96 // Dbg_Assert(ss_man);
97  if (!ss_man)
98  {
99  return;
100  }
101 
102  // Copy bounding box
103  m_bbox = bbox;
104 
105 
106 
107  CCollStatic** p_coll_obj_list = ss_man->GetIntersectingCollSectors( is );
108 
109  // Bloat the collision line only for the purposes of deciding against which world
110  // sectors to test the actual collision line
111  Mth::CBBox line_bbox;
112  Mth::Vector extend;
113 
114  extend = bbox.GetMax();
115  extend[X] += CCollObj::sLINE_BOX_EXTENT;
116  extend[Y] += CCollObj::sLINE_BOX_EXTENT;
117  extend[Z] += CCollObj::sLINE_BOX_EXTENT;
118  line_bbox.AddPoint(extend);
119 
120  extend = bbox.GetMin();
121  extend[X] -= CCollObj::sLINE_BOX_EXTENT;
122  extend[Y] -= CCollObj::sLINE_BOX_EXTENT;
123  extend[Z] -= CCollObj::sLINE_BOX_EXTENT;
124  line_bbox.AddPoint(extend);
125 
126  // Look for static collision
127  CCollStatic* p_coll_obj;
128  while((p_coll_obj = *p_coll_obj_list))
129  {
130  // TODO: Come up with a cleaner BBox check
131  //Dbg_Assert(p_coll_obj->GetGeometry());
132  //if(line_bbox.Intersect(p_coll_obj->GetGeometry()->GetBBox()))
133  if (p_coll_obj->WithinBBox(line_bbox))
134  {
135  add_static_collision(p_coll_obj);
136  }
137  p_coll_obj_list++;
138  }
139 
140  // Now look for movable collision
141  Lst::Node< CCollObj > *p_movable_node = CMovableCollMan::sGetCollisionList()->GetNext();
142  while(p_movable_node)
143  {
144  CCollObj *p_coll_obj = p_movable_node->GetData();
145  if (p_coll_obj && !(p_coll_obj->m_Flags & (mSD_NON_COLLIDABLE | mSD_KILLED )))
146  {
147  if (p_coll_obj->WithinBBox(line_bbox))
148  {
149  add_movable_collision(p_coll_obj);
150  }
151  }
152  p_movable_node = p_movable_node->GetNext();
153  }
154 
155 #if PRINT_TIMES
156  uint64 end_time = Tmr::GetTimeInUSeconds();
157  s_total_time += end_time - start_time;
158 
159  if (++s_num_collisions >= 1000)
160  {
161  Dbg_Message("Cache Update time %d us", s_total_time);
162  s_total_time = s_num_collisions = 0;
163  }
164 #endif
165 }

172 void CCollCache::add_static_collision(CCollStatic *p_collision)
173 {
174  Dbg_Assert(p_collision);
175  Dbg_MsgAssert(m_num_movable_coll == 0, ("Can't add static collision to cache after movable collision"));
176  Dbg_MsgAssert(m_array_size < MAX_COLLISION_OBJECTS, ("Collision cache full."));
177 
178  m_collision_array[m_array_size].mp_bbox = &(p_collision->GetGeometry()->GetBBox());
179  Dbg_MsgAssert(m_collision_array[m_array_size].mp_bbox, ("No bounding box found for the static collision"));
180 
181  m_collision_array[m_array_size++].mp_coll_obj = p_collision;
182 
183  m_num_static_coll++;
184 }

209 void CCollCache::delete_static_collision(CCollStatic *p_collision)
210 {
211  for (int i = 0; i < m_num_static_coll; i++)
212  {
213  if (m_collision_array[i].mp_coll_obj == p_collision)
214  {
215  // Must copy the whole block down
216  for (int j = i + 1; j < m_array_size; j++)
217  {
218  m_collision_array[j - 1] = m_collision_array[j];
219  }
220  m_num_static_coll--;
221  m_array_size--;
222  break;
223  }
224  }
225 }

232 void CCollCache::delete_movable_collision(CCollObj *p_collision)
233 {
234  for (int i = m_num_static_coll; i < m_array_size; i++)
235  {
236  if (m_collision_array[i].mp_coll_obj == p_collision)
237  {
238  m_collision_array[i] = m_collision_array[--m_array_size];
239  m_num_movable_coll--;
240  break;
241  }
242  }
243 }

281 void CCollCacheManager::sDestroyCollCache(CCollCache *p_cache)
282 {
283  bool found = false;
284 
285  // Take entry out of array
286  for (int i = 0; i < s_num_coll_caches; i++)
287  {
288  if (sp_coll_cache_array[i] == p_cache)
289  {
290  // Move last in array to here
291  sp_coll_cache_array[i] = sp_coll_cache_array[--s_num_coll_caches];
292 
293  found = true;
294  break;
295  }
296  }
297 
298  // And do the actual deletion
299  delete p_cache;
300 
301  Dbg_Assert(found);
302 }
