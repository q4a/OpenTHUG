68  for (int i=0; i<MAX_NESTED_LOOPS; ++i)

70  loop_is_ok[i]=false;

72 
73  Script::CArray *p_ok_scripts=Script::GetArray(CRCD(0x8b860fef,"DoNotAssertForInfiniteLoopsInTheseScripts"));
74  Script::CArray *p_blocking_functions=Script::GetArray(CRCD(0x215d13b9,"BlockingFunctions"));

77  while (*p_token!=ESCRIPTTOKEN_KEYWORD_ENDSCRIPT)
78  {
79  switch (*p_token)
80  {
81  case ESCRIPTTOKEN_KEYWORD_BEGIN:
82  loop_is_ok[loop_index++]=false; // It's guilty until proven innocent!
83  Dbg_MsgAssert(loop_index<=100,("Too many nested loops"));
84  p_token=SkipToken(p_token);
85  break;
86  case ESCRIPTTOKEN_KEYWORD_BREAK:
87  {
88  Dbg_MsgAssert(loop_index,("Zero loop_index"));
89  int i=loop_index-1;
90  while (i>=0)
91  {
92  loop_is_ok[i]=true;
93  --i;
94  }
95  p_token=SkipToken(p_token);
96  break;
97  }
98  case ESCRIPTTOKEN_KEYWORD_REPEAT:
99  p_token=SkipToken(p_token);
100 
101  Dbg_MsgAssert(loop_index,("Zero loop_index"));
102  if (*p_token==ESCRIPTTOKEN_ENDOFLINE || *p_token==ESCRIPTTOKEN_ENDOFLINENUMBER)
103  {
104  if (!loop_is_ok[loop_index-1])
105  {
106  bool allow=false;
107  if (p_ok_scripts)
108  {
109  for (uint32 i=0; i<p_ok_scripts->GetSize(); ++i)
110  {
111  if (p_ok_scripts->GetChecksum(i)==scriptName)
112  {
113  allow=true;
114  }
115  }
116  }
117 
118  if (allow)
119  {
120  }
121  else
122  {
123  Dbg_MsgAssert(0, ("Warning! Possible infinite loop: Line %d of %s\n",Script::GetLineNumber(p_token),p_fileName));
124  }
125  }
126  }
127  else
128  {
129  // If repeat is not followed by an end-of-line then it is probably followed by some
130  // count value, so it is not an infinite loop.
131  }
132

134  break;
135 
136  case ESCRIPTTOKEN_NAME:

151  if (loop_index)

156  loop_is_ok[i]=true;

158  }
159  }
160  break;
161  }
162  }
163  }
164  break;
165  }
166

175 // Used for evaluating the ESCRIPTTOKEN_KEYWORD_RANDOM_RANGE operator.
176 static uint8 *sCalculateRandomRange(uint8 *p_token, CComponent *p_comp, bool useRnd2)
177 {
178  // RandomRange token must be followed by a pair.
179  Dbg_MsgAssert(*p_token==ESCRIPTTOKEN_PAIR,("RandomRange operator must be followed by a pair of values in parentheses, File %s, line %d",GetSourceFile(p_token),GetLineNumber(p_token)));
180  ++p_token;
181  // Get the range values, which are floats cos it's a pair.
182  float x=Read4Bytes(p_token).mFloat;
183  p_token+=4;
184  float y=Read4Bytes(p_token).mFloat;
185  p_token+=4;
186 
187  // Get the integer values.
188  // Offset the floats up or down by 0.01 to ensure they round down to the expected integer values.
189  int a, b;
190  a=(int)( (x<0) ? x-0.1f:x+0.1f);
191  b=(int)( (y<0) ? y-0.1f:y+0.1f);
192 
193  bool they_are_integers=true;
194  if (fabs(x-a) > 0.00001f || fabs(y-b) > 0.00001f)
195  {
196  they_are_integers=false;
197  }
198 
199  if (they_are_integers)
200  {
201  // Make sure that a<=b
202  if (a>b)
203  {
204  // Swap them.
205  int temp=a;
206  a=b;
207  b=temp;
208  }
209  // Get val, which is between a and b inclusive.
210  // This assert is to make sure the calculations in Rnd don't overflow.
211  Dbg_MsgAssert(b-a+1<=32767,("File %s, line %d: RandomRange limits are too far apart, max is 32767",GetSourceFile(p_token),GetLineNumber(p_token)));
212  int val;
213  if (useRnd2)
214  {
215  val=a+Mth::Rnd2(b-a+1);
216  }
217  else
218  {
219  val=a+Mth::Rnd(b-a+1);
220  }
221  // Make totally sure the value is OK, cos could lead to hard-to-track-down bugs otherwise.
222 // Dbg_MsgAssert(val>=a && val<=b,("File %s, line %d: Internal error in RandomRange (%f %f %f), fire Ken.",GetSourceFile(p_token),GetLineNumber(p_token),x,y,val));
223 
224  p_comp->mType=ESYMBOLTYPE_INTEGER;
225  p_comp->mIntegerValue=val;
226  }
227  else
228  {
229  // Make sure that x<=y
230  if (x>y)
231  {
232  // Swap them.
233  float temp=x;
234  x=y;
235  y=temp;
236  }
237  // Get val, which is between x and y inclusive.
238  float val;
239  // Don't make FLOAT_RES bigger than 32766 because the calculations in Rnd will overflow otherwise.
240  #define FLOAT_RES 32760.0f
241  if (useRnd2)
242  {
243  val=x+((float)Mth::Rnd2((int)(FLOAT_RES+1.0f))*(y-x))/FLOAT_RES;
244  }
245  else
246  {
247  val=x+((float)Mth::Rnd((int)(FLOAT_RES+1.0f))*(y-x))/FLOAT_RES;
248  }
249 
250  // Make totally sure the value is OK, cos could lead to hard-to-track-down bugs otherwise.
251 // Dbg_MsgAssert(val>=x && val<=y,("File %s, line %d: Internal error in RandomRange (%f %f %f), fire Ken.",GetSourceFile(p_token),GetLineNumber(p_token),x,y,val));
252 
253  p_comp->mType=ESYMBOLTYPE_FLOAT;
254  p_comp->mFloatValue=val;
255  }
256 
257  return p_token;
258 }

260 // Skips over a data type token. Eg, if p_token points to a structure, it will
261 // skip over the whole structure.
262 static uint8 *sSkipType(uint8 *p_token)
263 {
264  switch (*p_token)
265  {
266  case ESCRIPTTOKEN_NAME:
267  case ESCRIPTTOKEN_INTEGER:
268  case ESCRIPTTOKEN_HEXINTEGER:
269  case ESCRIPTTOKEN_FLOAT:
270  p_token+=1+4;
271  break;
272  case ESCRIPTTOKEN_PAIR:
273  p_token+=1+2*4;
274  break;
275  case ESCRIPTTOKEN_VECTOR:
276  p_token+=1+3*4;
277  break;
278  case ESCRIPTTOKEN_STRING:
279  case ESCRIPTTOKEN_LOCALSTRING:
280  p_token+=1+4+Read4Bytes(p_token+1).mUInt;
281  break;
282  case ESCRIPTTOKEN_STARTSTRUCT:
283  {
284  int StructCount=1;
285  while (StructCount)
286  {
287  p_token=SkipToken(p_token);
288  if (*p_token==ESCRIPTTOKEN_STARTSTRUCT)
289  {
290  ++StructCount;
291  }
292  else if (*p_token==ESCRIPTTOKEN_ENDSTRUCT)
293  {
294  --StructCount;
295  }
296  }
297  ++p_token;
298  break;
299  }
300  case ESCRIPTTOKEN_STARTARRAY:
301  {
302  int ArrayCount=1;
303  while (ArrayCount)
304  {
305  p_token=SkipToken(p_token);
306  if (*p_token==ESCRIPTTOKEN_STARTARRAY)
307  {
308  ++ArrayCount;
309  }
310  else if (*p_token==ESCRIPTTOKEN_ENDARRAY)
311  {
312  --ArrayCount;
313  }
314  }
315  ++p_token;
316  break;
317  }
318  default:
319  Dbg_MsgAssert(0,("Unrecognized type"));
320  break;
321  }
322  return p_token;
323 }

340 
341  // Remember the start, since we're going to do a first pass through to determine the array type and size.
342  uint8 *p_start=p_token;
343 
344  // Skip over the startarray token.
345  ++p_token;
346  // Execute any random operators.
347  p_token=DoAnyRandomsOrJumps(p_token);
348 
349  ESymbolType type=ESYMBOLTYPE_NONE;
350  uint32 size=0;
351 
352  while (*p_token!=ESCRIPTTOKEN_ENDARRAY)
353  {
354  // Determine type.
355  switch (*p_token)
356  {
357  case ESCRIPTTOKEN_ENDOFLINE:
358  case ESCRIPTTOKEN_ENDOFLINENUMBER:
359  case ESCRIPTTOKEN_COMMA:
360  break;
361  case ESCRIPTTOKEN_NAME:
362  type=ESYMBOLTYPE_NAME;
363  break;
364  case ESCRIPTTOKEN_INTEGER:
365  case ESCRIPTTOKEN_HEXINTEGER:
366  // Integers don't override floats.
367  if (type!=ESYMBOLTYPE_FLOAT)
368  {
369  type=ESYMBOLTYPE_INTEGER;
370  }
371  break;
372  case ESCRIPTTOKEN_FLOAT:
373  type=ESYMBOLTYPE_FLOAT;
374  break;
375  case ESCRIPTTOKEN_PAIR:
376  type=ESYMBOLTYPE_PAIR;
377  break;
378  case ESCRIPTTOKEN_VECTOR:
379  type=ESYMBOLTYPE_VECTOR;
380  break;
381  case ESCRIPTTOKEN_STRING:
382  type=ESYMBOLTYPE_STRING;
383  break;
384  case ESCRIPTTOKEN_LOCALSTRING:
385  type=ESYMBOLTYPE_LOCALSTRING;
386  break;
387  case ESCRIPTTOKEN_STARTSTRUCT:
388  type=ESYMBOLTYPE_STRUCTURE;
389  break;
390  case ESCRIPTTOKEN_STARTARRAY:
391  type=ESYMBOLTYPE_ARRAY;
392  break;
393  default:
394  Dbg_MsgAssert(0,("Unrecognized data type in array, File %s, line %d\n",GetSourceFile(p_token),GetLineNumber(p_token)));
395  break;
396  }
397 
398  // Update the size and advance p_token.
399  switch (*p_token)
400  {
401  case ESCRIPTTOKEN_ENDOFLINE:
402  case ESCRIPTTOKEN_ENDOFLINENUMBER:
403  case ESCRIPTTOKEN_COMMA:
404  p_token=SkipToken(p_token);
405  break;
406  case ESCRIPTTOKEN_NAME:
407  case ESCRIPTTOKEN_INTEGER:
408  case ESCRIPTTOKEN_HEXINTEGER:
409  case ESCRIPTTOKEN_FLOAT:
410  case ESCRIPTTOKEN_PAIR:
411  case ESCRIPTTOKEN_VECTOR:
412  case ESCRIPTTOKEN_STRING:
413  case ESCRIPTTOKEN_LOCALSTRING:
414  case ESCRIPTTOKEN_STARTSTRUCT:
415  case ESCRIPTTOKEN_STARTARRAY:
416  ++size;
417  p_token=sSkipType(p_token);
418  break;
419  default:
420  Dbg_MsgAssert(0,("Unrecognized data type in array, File %s, line %d\n",GetSourceFile(p_token),GetLineNumber(p_token)));
421  break;
422  }
423 
424  // Execute any random operators. This has to be done each time p_token is advanced.
425  p_token=DoAnyRandomsOrJumps(p_token);
426  }
427 
428  if (type==ESYMBOLTYPE_NONE)
429  {
430  // Skip over the ESCRIPTTOKEN_ENDARRAY
431  ++p_token;
432  // Finished, empty array.
433  return p_token;
434  }

437  p_token=p_start;
438 
439  // Now that the array size and type are known, set up the CArray and fill it in.
440 
441  // Make sure we're using the script heap, because the CArray is not hard-wired to
442  // allocate it's buffer off it.

452  #endif

459

888 // Adds one component to p_dest.
889 // The component is given the name nameChecksum, and it's type and value is defined by whatever
890 // p_token is pointing to.
891 // p_args contains the structure defining the value of any args referred to using the <,> operators.
892 // Returns a pointer to the next token after parsing the value pointed to by p_token.
893 static uint8 *sAddComponentFromQB(CStruct *p_dest, uint32 nameChecksum, uint8 *p_token, CStruct *p_args)
894 {
895  Dbg_MsgAssert(p_dest,("NULL p_dest"));
896  Dbg_MsgAssert(p_token,("NULL p_token"));
897 
898  bool use_rnd2=false;
899 
900  switch (*p_token++)
901  {
902  case ESCRIPTTOKEN_NAME:
903  {
904  uint32 checksum=Read4Bytes(p_token).mChecksum;
905  p_token+=4;
906  p_dest->AddChecksum(nameChecksum,checksum);
907  break;
908  }
909  case ESCRIPTTOKEN_INTEGER:
910  {
911  int integer=Read4Bytes(p_token).mInt;
912  p_token+=4;
913  p_dest->AddInteger(nameChecksum,integer);
914  break;
915  }
916  case ESCRIPTTOKEN_KEYWORD_RANDOM_RANGE2:
917  use_rnd2=true;
918  // Intentional lack of a break here.
919  case ESCRIPTTOKEN_KEYWORD_RANDOM_RANGE:
920  {
921  CComponent *p_new_component=new CComponent;
922  p_token=sCalculateRandomRange(p_token,p_new_component,use_rnd2);
923  p_new_component->mNameChecksum=nameChecksum;
924  p_dest->AddComponent(p_new_component);
925  break;
926  }
927 
928  case ESCRIPTTOKEN_FLOAT:
929  {
930  float float_val=Read4Bytes(p_token).mFloat;
931  p_token+=4;
932  p_dest->AddFloat(nameChecksum,float_val);
933  break;
934  }
935  case ESCRIPTTOKEN_VECTOR:
936  {
937  float x=Read4Bytes(p_token).mFloat;
938  p_token+=4;
939  float y=Read4Bytes(p_token).mFloat;
940  p_token+=4;
941  float z=Read4Bytes(p_token).mFloat;
942  p_token+=4;
943 
944  // Alert! This next 'if' is kind of game specific ...
945  // It should probably be the exporter that does not include the angles in the node array if they are
946  // close to zero, rather than having this remove them from all structures.
947  // Note: We can't just remove zero angles from the node array after it has loaded, because there
948  // is not enough memory to store them initially.
949  if (nameChecksum==CRCD(0x9d2d0915,"Angles"))
950  {
951  // If the vector is close to (0,0,0) then don't bother adding it.
952  // This is a memory optimization. All nodes in levels are getting exported with
953  // an angles component, which most of the time is (0,0,0).
954  // This will work OK so long as whenever GetVector is used in the code to get the angles,
955  // the vector being loaded into is initialised with (0,0,0).
956  // That way, if the vector is not there, it will remain as (0,0,0) as required.
957  float dx=x;
958  if (dx<0.0f) dx=-dx;
959  float dy=y;
960  if (dy<0.0f) dy=-dy;
961  float dz=z;
962  if (dz<0.0f) dz=-dz;
963  if (dx<0.01f && dy<0.01f && dz<0.01f)
964  {
965  break;
966  }
967  }
968 
969  p_dest->AddVector(nameChecksum,x,y,z);
970  break;
971  }
972  case ESCRIPTTOKEN_PAIR:
973  {
974  float x=Read4Bytes(p_token).mFloat;
975  p_token+=4;
976  float y=Read4Bytes(p_token).mFloat;
977  p_token+=4;
978  p_dest->AddPair(nameChecksum,x,y);
979  break;
980  }
981  case ESCRIPTTOKEN_STRING:
982  {
983  int len=Read4Bytes(p_token).mInt;
984  p_token+=4;
985  p_dest->AddString(nameChecksum,(const char *)p_token);
986  p_token+=len;
987  break;
988  }
989  case ESCRIPTTOKEN_LOCALSTRING:
990  {
991  int len=Read4Bytes(p_token).mInt;
992  p_token+=4;
993  p_dest->AddLocalString(nameChecksum,(const char *)p_token);
994  p_token+=len;
995  break;
996  }
997  case ESCRIPTTOKEN_STARTSTRUCT:
998  {
999  // Rewind p_token to point to ESCRIPTTOKEN_STARTSTRUCT, because the
1000  // sAddComponentsWithinCurlyBraces call requires it.
1001  --p_token;
1002 
1003  // No need to set which heap we're using, cos CStruct's and their CComponent's come off
1004  // their own pools.
1005  CStruct *p_structure=new CStruct;
1006  p_token=sAddComponentsWithinCurlyBraces(p_structure,p_token,p_args);
1007  p_dest->AddStructurePointer(nameChecksum,p_structure);
1008  // Note: Not deleting p_structure, because it has been given to p_dest, which will delete
1009  // it in its destructor.
1010  break;
1011  }
1012  case ESCRIPTTOKEN_STARTARRAY:
1013  {
1014  // Rewind p_token to point to ESCRIPTTOKEN_STARTARRAY, because the CArray::Parse call requires it.
1015  --p_token;
1016

1018  // to make sure we're using the script heap here.
1019  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
1020  CArray *p_array=new CArray;
1021  Mem::Manager::sHandle().PopContext();
1022 
1023  p_token=sInitArrayFromQB(p_array,p_token,p_args);
1024  p_dest->AddArrayPointer(nameChecksum,p_array);
1025  // Note: Not deleting p_array, because it has been given to p_dest, which will delete
1026  // it in its destructor.
1027  break;
1028  }
1029  case ESCRIPTTOKEN_KEYWORD_SCRIPT:
1030  {
1031  // When adding a local script defined in a structure, we'd expect the nameChecksum passed to
1032  // this function to be zero, since the name comes after the script keyword.
1033  Dbg_MsgAssert(nameChecksum==0,("nameChecksum expected to be 0 for a local script ??, File %s, line %d\n",GetSourceFile(p_token-1),GetLineNumber(p_token-1)));
1034

1036  ++p_token; // Skip over the ESCRIPTTOKEN_NAME
1037  nameChecksum=Read4Bytes(p_token).mChecksum;
1038  p_token+=4; // Skip over the name checksum
1039 
1040  // Skip p_token over the script tokens, and calculate the size.
1041  const uint8 *p_script=p_token;
1042  p_token=SkipOverScript(p_token);
1043  uint32 size=p_token-p_script;
1044 
1045  // This will create a new buffer and copy in the script data.
1046  p_dest->AddScript(nameChecksum,p_script,size);
1047  break;
1048  }
1049  case ESCRIPTTOKEN_ARG:
1050  {
1051  Dbg_MsgAssert(*p_token==ESCRIPTTOKEN_NAME,("Expected '<' token to be followed by a name, File %s, line %d",GetSourceFile(p_token-1),GetLineNumber(p_token-1)));
1052  ++p_token;
1053  uint32 arg_checksum=Read4Bytes(p_token).mChecksum;
1054  p_token+=4;
1055  if (p_args)
1056  {
1057  // Look for a parameter named arg_checksum in p_args, recursing into any referenced structures
1058  CComponent *p_comp=p_args->FindNamedComponentRecurse(arg_checksum);
1059  if (p_comp)
1060  {
1061  if (nameChecksum==0 && p_comp->mType==ESYMBOLTYPE_STRUCTURE)
1062  {
1063  // Trying to add an unnamed structure, in which case it's contents
1064  // should be merged onto p_dest.
1065  p_dest->AppendStructure(p_comp->mpStructure);
1066  // Note: OK to call AppendStructure with NULL, so no need to check mpStructure.
1067  }
1068  else
1069  {
1070  // Create a copy of the component and add it to p_dest.
1071 
1072  CComponent *p_new_component=new CComponent;
1073  // Note: There is no copy-constructor for CComponent, because that would cause a
1074  // cyclic dependency between component.cpp/h and struct.cpp/h.
1075  // Instead, use the CopyComponent function defined in struct.cpp
1076  CopyComponent(p_new_component,p_comp);
1077  // Then change it's name to nameChecksum
1078  p_new_component->mNameChecksum=nameChecksum;
1079  // And add it to p_dest
1080  p_dest->AddComponent(p_new_component);
1081  }
1082  }
1083  }
1084  // Don't assert if p_args is NULL, cos it will be when this function is used to skip over
1085  // parameter lists in FindReferences
1086  break;
1087  }
1088  case ESCRIPTTOKEN_KEYWORD_ALLARGS:
1089  {
1090  // Don't assert if p_args is NULL, cos it will be when this function is used to skip over
1091  // parameter lists in FindReferences
1092  if (p_args)
1093  {
1094  if (nameChecksum)
1095  {
1096  // If it's a named structure, copy the passed arguments into a new structure and insert a
1097  // pointer to it.
1098  CStruct *p_structure=new CStruct;
1099  *p_structure=*p_args;
1100  p_dest->AddStructurePointer(nameChecksum,p_structure);
1101  }
1102  else
1103  {
1104  // Otherwise just merge in the passed parameters.
1105  // Un-named structures are always considered 'part of' the structure that they are in,
1106  // eg {x=9 {y=3}} is the same as {x=9 y=3}
1107  *p_dest+=*p_args;
1108  }
1109  }
1110  break;
1111  }
1112  case ESCRIPTTOKEN_COMMA:
1113  break;
1114  default:
1115  --p_token;
1116  Dbg_MsgAssert(0,("\nBad token '%s' encountered when creating component, File %s, line %d",GetTokenName((EScriptToken)*p_token),GetSourceFile(p_token),GetLineNumber(p_token)));
1117  break;
1118  }
1119  return p_token;

1122 // Parses the components in QB format pointed to by p_token, and adds them to p_dest until the
1123 // close-curly-brace token is reached.
1124 // p_token must initially point to an open-curly-brace token.
1125 // Any end-of-lines in between are ignored.
1126 // For example, p_token may point to: {x=9 foo="Blaa"}
1127 // Used when creating sub-structures and global structures.
1128 // p_args contains the structure defining the value of any args referred to using the <,> operators.
1129 // Returns a pointer to the next token after parsing the tokens pointed to by p_token.
1130 static uint8 *sAddComponentsWithinCurlyBraces(CStruct *p_dest, uint8 *p_token, CStruct *p_args)
1131 {
1132  Dbg_MsgAssert(p_dest,("NULL p_dest"));
1133  Dbg_MsgAssert(p_token,("NULL p_token"));
1134  Dbg_MsgAssert(*p_token==ESCRIPTTOKEN_STARTSTRUCT,("p_token expected to point to ESCRIPTTOKEN_STARTSTRUCT, File %s, line %d",GetSourceFile(p_token),GetLineNumber(p_token)));
1135  // Skip over the ESCRIPTTOKEN_STARTSTRUCT
1136  ++p_token;
1137 
1138  while (true)
1139  {
1140  p_token=SkipEndOfLines(p_token);
1141  // Execute any random operators. This has to be done each time p_token is advanced.
1142  p_token=DoAnyRandomsOrJumps(p_token);
1143  if (*p_token==ESCRIPTTOKEN_ENDSTRUCT)
1144  {
1145  break;
1146  }
1147 
1148  switch (*p_token)
1149  {
1150  case ESCRIPTTOKEN_NAME:
1151  {
1152  uint8 *p_name_token=p_token;
1153  ++p_token;
1154  uint32 name_checksum=Read4Bytes(p_token).mChecksum;
1155  p_token+=4;
1156  p_token=SkipEndOfLines(p_token);
1157  p_token=DoAnyRandomsOrJumps(p_token);
1158  if (*p_token==ESCRIPTTOKEN_EQUALS)
1159  {
1160  ++p_token;
1161  p_token=SkipEndOfLines(p_token);
1162  p_token=DoAnyRandomsOrJumps(p_token);
1163 
1164  if (*p_token==ESCRIPTTOKEN_OPENPARENTH)
1165  {
1166  CComponent *p_comp=new CComponent;
1167  p_token=Evaluate(p_token,p_args,p_comp);
1168  if (p_comp->mType!=ESYMBOLTYPE_NONE)
1169  {
1170  p_comp->mNameChecksum=name_checksum;
1171  p_dest->AddComponent(p_comp);
1172  }
1173  else
1174  {
1175  delete p_comp;
1176  }
1177  }
1178  else
1179  {
1180  p_token=sAddComponentFromQB(p_dest,name_checksum,p_token,p_args);
1181  }
1182  }
1183  else
1184  {
1185  p_token=sAddComponentFromQB(p_dest,NO_NAME,p_name_token,p_args);
1186  }
1187  break;
1188  }
1189  case ESCRIPTTOKEN_STARTSTRUCT:
1190  p_token=sAddComponentsWithinCurlyBraces(p_dest,p_token,p_args);
1191  break;
1192  case ESCRIPTTOKEN_COMMA:
1193  ++p_token;
1194  break;
1195  case ESCRIPTTOKEN_ENDOFLINE:
1196  case ESCRIPTTOKEN_ENDOFLINENUMBER:
1197  p_token=SkipEndOfLines(p_token);
1198  break;
1199  default:
1200  {
1201  if (*p_token==ESCRIPTTOKEN_OPENPARENTH)
1202  {
1203  CComponent *p_comp=new CComponent;
1204  p_token=Evaluate(p_token,p_args,p_comp);
1205 
1206  if (p_comp->mType==ESYMBOLTYPE_STRUCTURE)
1207  {
1208  // The CStruct::AddComponent function does not allow unnamed structure
1209  // components to be added, so merge in the contents of the structure instead.
1210  p_dest->AppendStructure(p_comp->mpStructure);
1211  // Now p_comp does have to be cleaned up and deleted, because it has not
1212  // been given to p_dest.
1213  CleanUpComponent(p_comp);
1214  delete p_comp;
1215  }
1216  else
1217  {
1218  if (p_comp->mType!=ESYMBOLTYPE_NONE)
1219  {
1220  p_comp->mNameChecksum=NO_NAME;
1221  p_dest->AddComponent(p_comp);
1222  }
1223  else
1224  {
1225  delete p_comp;
1226  }
1227  }
1228  }
1229  else
1230  {
1231  p_token=sAddComponentFromQB(p_dest,NO_NAME,p_token,p_args);
1232  }
1233  break;
1234  }
1235  }
1236  }
1237 
1238  // Skip over the ESCRIPTTOKEN_ENDSTRUCT
1239  ++p_token;
1240  return p_token;
1241 }

1441 // Given a pointer to an un-preprocessed script, this will parse through it converting function names
1442 // to either cfunction pointers or member function tokens.
1443 void PreProcessScript(uint8 *p_token)
1444 {
1445  // Skip over the default params
1446  p_token=SkipToStartOfNextLine(p_token);
1447 
1448  while (*p_token!=ESCRIPTTOKEN_KEYWORD_ENDSCRIPT)
1449  {
1450  switch (*p_token)
1451  {
1452  case ESCRIPTTOKEN_KEYWORD_IF:
1453  ++p_token;
1454  break;
1455 
1456  case ESCRIPTTOKEN_NAME:
1457  {
1458  ++p_token;
1459  uint32 name_checksum=Read4Bytes(p_token).mChecksum;
1460  p_token+=4;
1461 
1462  // Look up the name to see if it is a cfunction or member function.
1463  CSymbolTableEntry *p_entry=Resolve(name_checksum);
1464  // Must not assert if p_entry is NULL, cos they might just be loading in
1465  // a qb file that refers to a script that has not been written yet.
1466  if (p_entry)
1467  {
1468  if (p_entry->mType==ESYMBOLTYPE_CFUNCTION)
1469  {
1470  // Change the token type and replace the checksum with the
1471  // function pointer to save having to resolve it later.
1472  *(p_token-5)=ESCRIPTTOKEN_RUNTIME_CFUNCTION;
1473  Dbg_MsgAssert(p_entry->mpCFunction,("NULL p_entry->mpCFunction"));
1474  Write4Bytes(p_token-4, (uint32)p_entry->mpCFunction);
1475  }
1476  else if (p_entry->mType==ESYMBOLTYPE_MEMBERFUNCTION)
1477  {
1478  // Saves having to look up the checksum later to find out that it is
1479  // a member function.
1480  *(p_token-5)=ESCRIPTTOKEN_RUNTIME_MEMBERFUNCTION;
1481  }
1482  }
1483  p_token=SkipToStartOfNextLine(p_token);
1484  break;
1485  }
1486  default:
1487  p_token=SkipToStartOfNextLine(p_token);
1488  break;
1489  }
1490  }
