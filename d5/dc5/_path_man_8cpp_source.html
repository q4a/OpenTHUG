120 // Called just after a new NodeArray is loaded, in ScriptLoadNodeArray in cfuncs.cpp
121 // Scans through the node array inserting PathNum members to all nodes that are on a path.
122 // Enables quick look up of what path a car is on, & hence whether it is on the same path as
123 // another car.
124 void CPathMan::AddPathInfoToNodeArray()
125 {
126 // Dbg_Assert( mp_path_object_trackers );
127  if (!mp_path_object_trackers)
128  {
129  return;
130  }
131 
132  // printf("CPathMan::AddPathInfoToNodeArray\n");
133  Script::CArray *p_node_array=Script::GetArray(CRCD(0xc472ecc5,"NodeArray"),Script::ASSERT);
134 
135  ClearPathObjectTrackers();
136 
137  // The path number assigned to the nodes is just an integer, starting at 0
138  m_num_paths=0;
139 
140  int array_size=p_node_array->GetSize();
141  for ( uint16 i = 0; i < array_size; ++i )
142  {
143  Script::CStruct *p_start_node=p_node_array->GetStructure( i );
144 
145  // Rail nodes are linked into a path, but don't need path numbers, so don't bother
146  // adding them to save memory.
147  uint32 class_type=0;
148  p_start_node->GetChecksum( CRCD(0x12b4e660,"Class"), &class_type );
149  if ( class_type == CRCD( 0x8e6b02ad, "RailNode" ) ||
150  class_type == CRCD(0x30c19600, "ClimbingNode") )
151  {
152  continue;
153  }
154 
155  // make sure this isn't a pedai waypoint
156  uint32 type;
157  if ( class_type == CRCD( 0x4c23a77e, "Waypoint" )
158  && p_start_node->GetChecksum( CRCD( 0x7321a8d6, "type" ), &type, Script::NO_ASSERT )
159  && type == CRCD( 0xcba10ffa, "PedAI" ) )
160  {
161  continue;
162  }
163 
164  int num_links=SkateScript::GetNumLinks( p_start_node );
165  int node_path_number = 0;
166  // Check if the node is linked to something, and has not already had a PathNum
167  // given to it.
168  if ( ( num_links || class_type == CRCD( 0x4c23a77e, "WayPoint" ) ) &&
169  !p_start_node->ContainsComponentNamed( CRCD( 0x9c91c3ca, "PathNum" ) ) )
170  {
171  // printf("found a non PedAI waypoint\n");
172  // The node and subsequent nodes that it is linked to require a PathNum
173  Script::CStruct *p_node=p_start_node;
174 
175  // The path number that is going to be written in is m_num_paths to start with,
176  // but that might change if we hit a node that already has a path number. In that
177  // case we'll rewind and write that number in instead.
178  int path_number_to_write=m_num_paths;
179 
180  while (true)
181  {
182  if (p_node->GetInteger( CRCD( 0x9c91c3ca, "PathNum" ),&node_path_number))
183  {
184  // We've hit a node that has already had a PathNum written in.
185 
186  if (node_path_number==path_number_to_write)
187  {
188  // The path num matches what we started with, so must have looped
189  // back to the start, so finished.
190  break;
191  }
192 
193  // We've hit a node that has a different PathNum. So we need to rewind
194  // back to where we started and write in that path number instead.
195  // This happens due to the fact that the nodes could be in any order in
196  // the node array, so paths can start being traversed from any point, not
197  // necessarily the start.
198  path_number_to_write=node_path_number;
199  p_node=p_start_node;
200  }
201 
202  // Wack in the path number
203  p_node->AddInteger(0x9c91c3ca/*PathNum*/,path_number_to_write);
204 
205  // Check if this is a terminating node of the path, and stop if so.
206  int num_links=SkateScript::GetNumLinks(p_node);
207  if (num_links==0)
208  {
209  break;
210  }
211 
212  // On to the next node. Note the if the path branches, it will only
213  // follow the first branch. The other branches will end up getting
214  // their own path numbers. Don't know if this will be a problem yet.
215  p_node=p_node_array->GetStructure(SkateScript::GetLink( p_node, 0 ));
216  }
217 
218  // If path_number got used, increment it ready for the next path.
219  if (path_number_to_write==m_num_paths)
220  {
221  ++m_num_paths;
222  Dbg_MsgAssert(m_num_paths<MAX_PATHS,("Too many paths, max=%d",MAX_PATHS));
223  }
224  }
225  }
226 }
