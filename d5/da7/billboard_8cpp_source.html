34 static int sort_batches_by_draw_order( const void *p1, const void *p2 )
35 {
36  sBillboardMaterialBatch *p_batch0 = *((sBillboardMaterialBatch**)p1 );
37  sBillboardMaterialBatch *p_batch1 = *((sBillboardMaterialBatch**)p2 );
38 
39  // Deal with NULL pointers first (caused by removing batches from the list).
40  if( p_batch0 == NULL )
41  {
42  if( p_batch1 == NULL )
43  {
44  return 0;
45  }
46  else
47  {
48  return 1;
49  }
50  }
51  else if( p_batch1 == NULL )
52  {
53  return -1;
54  }
55 
56  return p_batch0->GetDrawOrder() < p_batch1->GetDrawOrder() ? -1 : p_batch0->GetDrawOrder() > p_batch1->GetDrawOrder() ? 1 : 0;
57 }

115 void sBillboardManager::RemoveEntry( sMesh *p_mesh )
116 {
117  if( p_mesh->mp_material )
118  {
119  sBillboardMaterialBatch *p_batch = GetBatch( p_mesh->mp_material );
120  if( p_batch )
121  {
122  // Removes the entry if it exists.
123  p_batch->RemoveEntry( p_mesh );
124 
125  // If the batch is now empty, remove it.
126  if( p_batch->IsEmpty())
127  {
128  for( int i = 0; i < m_num_batches; ++i )
129  {
130  if( mp_batches[i] == p_batch )
131  {
132  delete p_batch;
133  mp_batches[i] = NULL;
134  break;
135  }
136  }
137 
138  // Resort the batches (will move the NULL pointer to the end).
139  SortBatches();
140 
141  // Important not to decrement this until after the resort has been performed.
142  --m_num_batches;
143  }
144  }
145  }
