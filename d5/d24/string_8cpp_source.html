5 // Code for managing allocation and deallocation of strings within the script system.
6 // For example it manages the 'permanent string' heap, which is for strings that get created
7 // when all the qb's are loaded and never get deleted. They can therefore all be squashed
8 // together for optimum use of memory.
9 //
10 // Note: Not defining a CString class, because it would have to have a char pointer so
11 // would just be an unneccessary layer of indirection that would waste space.
12 //
13 // Instead CreateString and DeleteString are used to allocate and deallocate strings.
14 //
15 // Could add cleverer memory management here later, eg a pool of fixed sized strings
16 // for strings that need to be alloc'd and dealloc'd fast.
17 //
19 
20 // Note: Small memory saving possible by deleting the sp_permanent_string_checksums array
21 // once you know that no more permanent strings will need to be allocated. Will save 32K ish,
22 // depending on maxStrings.
23 // (There's no function here yet for doing that, but easy to add)

42 // Array of string checksums to allow quick checking to see if a particular string
43 // is already in the permanent string buffer without having to do string comparisons.
44 // Needed because when loading several thousand strings, several million string comparisons
45 // would have to be done, causing a significant pause in loading the game.
46 struct SSpecialStringChecksum
47 {
48  uint32 mChecksum;
49  char *mpString;
50 };
51 static SSpecialStringChecksum *sp_permanent_string_checksums=NULL;
52 static uint32 s_num_permanent_string_checksums=0;
53 static uint32 s_max_permanent_string_checksums=0;

55 // Needs to be called once at the start of the game.
56 // For THPS3, maxSize was 48000 for English, 60500 for other languages. maxStrings was 4000
57 void AllocatePermanentStringHeap(uint32 maxSize, uint32 maxStrings)
58 {
59  // Uses whatever the current heap is set to, which is set to the script heap in Script::AllocatePools
60  // which is where this function is called from.
61  s_permanent_string_heap_size=maxSize;
62  s_max_permanent_string_checksums=maxStrings;
63 
64  // Allocate the string buffer.
65  Dbg_MsgAssert(sp_permanent_string_heap==NULL,("sp_permanent_string_heap not NULL ???"));
66  sp_permanent_string_heap=(char*)Mem::Malloc(s_permanent_string_heap_size);
67  sp_permanent_string_heap_top=sp_permanent_string_heap;
68 
69  // Allocate the array of checksums.
70  Dbg_MsgAssert(sp_permanent_string_checksums==NULL,("sp_permanent_string_checksums not NULL ???"));
71  sp_permanent_string_checksums=(SSpecialStringChecksum*)Mem::Malloc(s_max_permanent_string_checksums*sizeof(SSpecialStringChecksum));
72  s_num_permanent_string_checksums=0;
73 }

93 // Adds the contents of the passed pString to the special string heap, and returns
94 // the pointer to the string within the heap.
95 // If the string is already found in the heap, it won't add it again.
96 static char *sAddToSpecialStringHeap(const char *p_string)
97 {
98  Dbg_MsgAssert(p_string,("NULL p_string"));
99 
100  // Check to see if it already exists in the special string heap ...
101 
102  // Calculate the checksum of the string, then search for this in the array of
103  // checksums of all the special strings so far.
104  // Quicker than doing byte comparisons, probably due to the long time it takes to
105  // read single bytes from memory.
106  // Needs to be fast, otherwise it can add a significant delay to the startup time.
107  // This method adds about 0.3 seconds altogether (in THPS3, couple of thousand strings)
108  uint32 len=strlen(p_string);
109  uint32 checksum=Crc::GenerateCRCCaseSensitive(p_string,len);
110 
111  Dbg_MsgAssert(sp_permanent_string_checksums,("NULL sp_permanent_string_checksums ??"));
112  SSpecialStringChecksum *p_ch=sp_permanent_string_checksums;
113  for (uint32 i=0; i<s_num_permanent_string_checksums; ++i)
114  {
115  if (p_ch->mChecksum==checksum)
116  {
117  return p_ch->mpString;
118  }
119  ++p_ch;
120  }
121 
122  char *p_heap_string=NULL;
123 
124  // Not found, so add to the pile of strings.
125  Dbg_MsgAssert(sp_permanent_string_heap,("NULL sp_permanent_string_heap ??"));
126 
127  Dbg_MsgAssert(s_permanent_string_heap_size-(sp_permanent_string_heap_top-sp_permanent_string_heap)>=len+1,("Out of special string heap"));
128  p_heap_string=sp_permanent_string_heap_top;
129  strcpy(p_heap_string,p_string);
130  // Update sp_permanent_string_heap_top
131  sp_permanent_string_heap_top+=len;
132  ++sp_permanent_string_heap_top; // Skip over the terminator
133 
134  // Store the checksum too.
135  Dbg_MsgAssert(s_num_permanent_string_checksums<s_max_permanent_string_checksums,("Ran out of permanent-string checksums"));
136  sp_permanent_string_checksums[s_num_permanent_string_checksums].mChecksum=checksum;
137  sp_permanent_string_checksums[s_num_permanent_string_checksums].mpString=p_heap_string;
138  ++s_num_permanent_string_checksums;
139 
140  return p_heap_string;
141 }

148 void UseRegularStringHeap()
149 {
150  s_use_permanent_string_heap=false;

152  // 56K memory saving, though this assumes that UsePermanentStringHeap() won't be called
153  // again after UseRegularStringHeap(). If it is, the code will assert as soon as a string
154  // is attempted to be created.
155  // The current code does not use the permanent string heap again, so use the savings.
156  // (End of THPS6 and memory is tight)
157  if (sp_permanent_string_checksums)
158  {
159  Mem::Free(sp_permanent_string_checksums);
160  sp_permanent_string_checksums=NULL;
161  }
162 }

164 // Allocates space for the passed string and copies it in, and returns a pointer to the new string,
165 // so the original string can safely be deleted.
166 // I guess this should really return a const char*, but if it did then DeleteString would
167 // have to take a const char* too, and it can't because then it would not be able to call
168 // Mem::Free on it.
169 char *CreateString(const char *p_string)
170 {
171  Dbg_MsgAssert(p_string,("NULL p_string"));
172 
173  if (s_use_permanent_string_heap)
174  {
175  return sAddToSpecialStringHeap(p_string);
176  }
177 
178  // Just do a regular allocation.
179  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
180 
181  char *p_new_string=(char*)Mem::Malloc(strlen(p_string)+1); // +1 for terminator
182  strcpy(p_new_string,p_string);
183 
184  Mem::Manager::sHandle().PopContext();
185 
186  return p_new_string;
187 }

189 void DeleteString(char *p_string)
190 {
191  Dbg_MsgAssert(p_string,("NULL p_string"));
192 
193  // Check whether the string is in the permanent-string buffer.
194  Dbg_MsgAssert(sp_permanent_string_heap,("NULL sp_permanent_string_heap ?"));
195  if (p_string>=sp_permanent_string_heap && p_string<sp_permanent_string_heap+s_permanent_string_heap_size)
196  {
197  // It is in the permanent-string buffer, so nothing to do.
198  }
199  else
200  {
201  // Must have been allocated the usual way, so free it.
202  Mem::Free(p_string);
203  }
204 }
