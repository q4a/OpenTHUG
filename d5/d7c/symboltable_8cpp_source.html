208 // Resolves a checksum into a symbol table entry. Does any stepping over of chains of symbols linked by
209 // name. Eg, Resolving a where a=b, b=c, c=d, d=6 will return the symbol d, with value 6.
210 // Returns NULL if the checksum cannot be resolved.
211 CSymbolTableEntry *Resolve(uint32 checksum)
212 {
213  // Look up the checksum.
214  CSymbolTableEntry *p_entry=LookUpSymbol(checksum);
215  if (p_entry && p_entry->mType==ESYMBOLTYPE_NAME)
216  {
217  // If the symbol is of type name, then it may be that the name is the name of another symbol,
218  // in which case we need to find that.
219  // Eg we could be trying to resolve a, where a=b, b=c, c=7, in which case we need to find 7.
220  // So loop until either a non-name symbol is reached, or no further symbol is reached.
221 
222  #ifdef __NOPT_ASSERT__
223  // An infinite loop could occur, for example if we have a global a=b where b=a
224  // This counter will make an assert go off if we try and loop a suspicious number of times.
225  uint32 num_loops=0;
226  #endif
227 
228  while (true)
229  {
230  CSymbolTableEntry *p_new=LookUpSymbol(p_entry->mChecksum);
231  if (!p_new)
232  {
233  // Reached the end of the chain, so return the last entry, which will have type name.
234  return p_entry;
235  }
236 
237  if (p_new->mType!=ESYMBOLTYPE_NAME)
238  {
239  // Reached something not of type name, so return that.
240  return p_new;
241  }
242 
243  // Repeat.
244  p_entry=p_new;
245  Dbg_MsgAssert(num_loops++<10,("Possible infinite recursion of the Resolve function when trying to resolve '%s', bailing out!",FindChecksumName(checksum)));
246  }
247  }
248 
249  return p_entry;
250 }

257 // Removes the symbol from the table.
258 // Used when reloading a .qb file.
259 // Note: Will not delete any entities referred to by the symbol, and will assert if any of the
260 // pointers are not NULL.
261 // This is to avoid a cyclic dependency between symboltable and struct.
262 // It is up to the caller to have deleted these before calling this.
263 void RemoveSymbol(CSymbolTableEntry *p_sym)
264 {
265  Dbg_MsgAssert(p_sym,("NULL p_sym"));
266  Dbg_MsgAssert(p_sym->mUnion==0,("CSymbolTableEntry still contains data, possibly an undeleted pointer. Type='%s'",GetTypeName(p_sym->mType)));
267  Dbg_MsgAssert(p_sym->mUsed,("Tried to call RemoveSymbol on an unused CSymbolTableEntry"));
268  Dbg_MsgAssert(sp_hash_table!=NULL,("sp_hash_table is NULL ?"));
269 
270  // Get the head pointer of the list that p_sym is in (or should be in)
271  CSymbolTableEntry *p_head=&sp_hash_table[ p_sym->mNameChecksum & ((1<<NUM_HASH_BITS)-1) ];
272 
273  if (p_sym==p_head)
274  {
275  // p_sym is in the hash table itself, so it must not be deleted because it is an element of
276  // a contiguous array, not allocated off the pool.
277 
278  // Check if there are further entries in the list. These will have been allocated off the pool.
279  if (p_sym->mpNext)
280  {
281  // There are later entries in the list, so copy the first of these down
282  // to fill the hole.
283  CSymbolTableEntry *p_source=p_sym->mpNext;
284 
285  // Copy the contents. Not using the assignement operator because that won't link.
286  // It is declared but not defined so as to cause it not to link, to prevent any problems
287  // with copying pointers if it gets used anywhere else.
288  uint32 *p_source_word=(uint32*)p_source;
289  uint32 *p_dest_word=(uint32*)p_sym;
290  Dbg_MsgAssert((sizeof(CSymbolTableEntry)&3)==0,("sizeof(CSymbolTableEntry) not a multiple of 4 ?"));
291  for (uint32 i=0; i<sizeof(CSymbolTableEntry)/4; ++i)
292  {
293  *p_dest_word++=*p_source_word++;
294  }
295 
296  // Need to zero the union so that the destructor does not think there is an undeleted
297  // pointer in it, which will make it assert.
298  p_source->mUnion=0;
299  delete p_source;
300  }
301  else
302  {
303  // Just reset the contents of p_sym.
304  p_sym->Reset();
305  }
306  }
307  else
308  {
309  // p_sym is not in the hash table, so it must have been allocated off the pool.
310  // In order to unlink it from the list we need to find the previous entry in the list.
311  // So search through the list until p_sym is found.
312 
313  // Start with mpNext, cos we know p_sym is not p_head
314  CSymbolTableEntry *p_previous=p_head;
315  CSymbolTableEntry *p_search=p_head->mpNext;
316  while (p_search)
317  {
318  if (p_search==p_sym)
319  {
320  break;
321  }
322  p_previous=p_search;
323  p_search=p_search->mpNext;
324  }
325 
326  Dbg_MsgAssert(p_search,("p_sym not found in list ? (checksum='%s')",FindChecksumName(p_sym->mNameChecksum)));
327 
328  // Unlink p_sym and delete it.
329  p_previous->mpNext=p_sym->mpNext;
330  delete p_sym;
331  }
332 }

334 // Adds a new checksum to the directory and returns a pointer to the new entry.
335 CSymbolTableEntry *CreateNewSymbolEntry(uint32 checksum)
336 {
337  #ifdef __NOPT_ASSERT__
338  CSymbolTableEntry *p_entry=LookUpSymbol(checksum);
339  Dbg_MsgAssert(p_entry==NULL,("Symbol '%s' defined twice.",FindChecksumName(checksum)));
340  #endif
341 
342  // Get the head pointer of the list where the new symbol needs to go.
343  Dbg_MsgAssert(sp_hash_table!=NULL,("sp_hash_table is NULL ?"));
344  CSymbolTableEntry *p_sym=&sp_hash_table[ checksum & ((1<<NUM_HASH_BITS)-1) ];
345 
346  // If nothing in here, use this one.
347  if (!p_sym->mUsed)
348  {
349  p_sym->Reset();
350  // Set the checksum.
351  p_sym->mNameChecksum=checksum;
352  // Flag it as used.
353  p_sym->mUsed=true;
354  // Set the mGotReloaded flag so that other game-specific code can see
355  // that the symbol has changed in value.
356  p_sym->mGotReloaded=true;
357  return p_sym;
358  }
359 
360  // Find the last element of the linked list, checking for any checksum
361  // clashes on the way.
362  CSymbolTableEntry *p_previous=NULL;
363  while (p_sym)
364  {
365  Dbg_MsgAssert(p_sym->mNameChecksum!=checksum,("Checksum clash in symbol table: '%s'",FindChecksumName(checksum)));
366  p_previous=p_sym;
367  p_sym=p_sym->mpNext;
368  }
369 
370  // Get a new entry from the pool.
371  CSymbolTableEntry *p_new=new CSymbolTableEntry;
372 
373  // Set the checksum & next pointer.
374  p_new->mNameChecksum=checksum;
375  // Flag it as used.
376  p_new->mUsed=true;
377  // Set the mGotReloaded flag so that other game-specific code can see
378  // that the symbol has changed in value.
379  p_new->mGotReloaded=true;
380 
381  p_new->mpNext=NULL;
382 
383  // This assert should never go off, but do it just in case.
384  Dbg_MsgAssert(p_previous,("NULL p_previous ???"));
385  // Stick pNew on the end of the list.
386  p_previous->mpNext=p_new;
387 
388  return p_new;
389 }

391 // This function provides an easy way to loop through all the symbols in the symbol table by
392 // hiding all the messy logic for stepping through the hash table.
393 //
394 // If passed NULL, it will return the first entry in the symbol table (or NULL if there is nothing in it)
395 // If passed a pointer to a CSymbolTableEntry, it will return the next one, or NULL if none left.
396 //
397 // So if you want to loop through all the symbols just do something like:
398 // CSymbolTableEntry *p_sym=GetNextSymbolTableEntry(); // No need to pass NULL, it defaults to it.
399 // while (p_sym)
400 // {
401 // // ...
402 // p_sym=GetNextSymbolTableEntry(p_sym);
403 // }
404 //
405 CSymbolTableEntry *GetNextSymbolTableEntry(CSymbolTableEntry *p_sym)
406 {
407  // static's for keeping track of where we are in the hash table.
408  static uint32 s_current_hash_index=0;
409  static CSymbolTableEntry *sp_last=NULL;
410 
411  if (p_sym==NULL)
412  {
413  // They want the first symbol.
414  s_current_hash_index=0;
415  p_sym=sp_hash_table;
416 
417  // Step through the hash table entries until a used one is found.
418  while (!p_sym->mUsed)
419  {
420  ++s_current_hash_index;
421  ++p_sym;
422 
423  if (s_current_hash_index>=(1<<NUM_HASH_BITS))
424  {
425  // Run out of hash table.
426  s_current_hash_index=0;
427  sp_last=NULL;
428  return NULL;
429  }
430  }
431 
432  sp_last=p_sym;
433  return p_sym;
434  }
435  else
436  {
437  // Check that p_sym is the same as the last one returned by this function.
438 
439  // Currently this function assumes that the user is going to be calling it on
440  // consecutive elements, such as in the example loop given in the comment for this function.
441  // In theory it could be made to work when passed an arbitrary member of the symbol table, but this
442  // would require it to search forward from the start each time in order to find out what index the
443  // list is at.
444  // It was easier and faster to just store the index as a static and assert if called on
445  // non-consecutive symbols. (Could fix it if it becomes a problem)
446  Dbg_MsgAssert(p_sym==sp_last,("Non-consecutive call to GetNextSymbolTableEntry"));
447  }
448 
449  p_sym=p_sym->mpNext;
450  if (!p_sym)
451  {
452  // We've gone off the end of this linked list, so go onto the next entry in the hash table.
453  ++s_current_hash_index;
454  p_sym=sp_hash_table+s_current_hash_index;
455 
456  if (s_current_hash_index>=(1<<NUM_HASH_BITS))
457  {
458  // Run out of hash table.
459  s_current_hash_index=0;
460  sp_last=NULL;
461  return NULL;
462  }
463 
464  // Step through the hash table entries until a used one is found.
465  while (!p_sym->mUsed)
466  {
467  ++s_current_hash_index;
468  ++p_sym;
469 
470  if (s_current_hash_index>=(1<<NUM_HASH_BITS))
471  {
472  s_current_hash_index=0;
473  sp_last=NULL;
474  return NULL;
475  }
476  }
477  }
478 
479  sp_last=p_sym;
480  return p_sym;
481 }

483 float GetFloat(uint32 checksum, EAssertType assert)
484 {
485  CSymbolTableEntry *p_entry=Resolve(checksum);
486  if (!p_entry)
487  {
488  Dbg_MsgAssert(!assert,("Float '%s' not found.",FindChecksumName(checksum)));
489  return 0.0f;
490  }
491 
492  switch (p_entry->mType)
493  {
494  case ESYMBOLTYPE_FLOAT:
495  return p_entry->mFloatValue;
496  break;
497  case ESYMBOLTYPE_INTEGER:
498  return (float)p_entry->mIntegerValue;
499  break;
500  default:
501  Dbg_MsgAssert(0,("Symbol '%s' of type '%s' cannot be converted to a float.",FindChecksumName(checksum),GetTypeName(p_entry->mType)));
502  return 0.0f;
503  break;
504  }
505 
506  return 0.0f;
507 }

704 {
705  CSymbolTableEntry *p_entry=Resolve(checksum);

710  }
711 
712  switch (p_entry->mType)

715  Dbg_MsgAssert(p_entry->mpCFunction,("NULL p_entry->pCFunction ?"));

718
