147 /*---------------------------------------------------------------------------*
148  Name: hwPadRead
149 
150  Description: Calls PADRead() and perform clamping. Get information
151  of button down/up and sets them into extended field.
152  This function also checks whether controllers are
153  actually connected.
154 
155  Arguments: None
156 
157  Returns: None
158  *---------------------------------------------------------------------------*/
159 void hwPadRead( void )
160 {
161  s32 i;
162  u32 ResetReq = 0; // for error handling
163 
164  // Read current PAD status
165  PADRead( padData );
166 
167  // Clamp analog inputs
168  PADClamp( padData );
169 
170  hwNumValidPads = 0;
171  for ( i = 0 ; i < PAD_MAX_CONTROLLERS ; i++ )
172  {
173  // Connection check
174  if ( padData[i].err == PAD_ERR_NONE ||
175  padData[i].err == PAD_ERR_TRANSFER )
176  {
177  ++hwNumValidPads;
178  }
179  else if ( padData[i].err == PAD_ERR_NO_CONTROLLER )
180  {
181  ResetReq |= PadChanMask[i];
182  }
183 
184 // hwPadCopy( &padData[i], &hwPad[i] );
185  }
186 
187  // Try resetting pad channels which have been not valid
188  if ( ResetReq )
189  {
190  // Don't care return status
191  // If FALSE, then reset again in next hwPadRead.
192  PADReset( ResetReq );
193  }
194 
195  return;
196 }

330 /*---------------------------------------------------------------------------*
331  Name: hwPadQueueRead
332  Description: Read gamepad state and set to hwPad[].
333  No need to care controller error.
334  (When error, PADState is set to zero in PADRead.)
335  Arguments: s32 flag: control block/noblock mode.
336  when flag == OS_MESSAGE_BLOCK,
337  wait next padinput if queue is empty.
338  when flag == OS_MESSAGE_NOBLOCK,
339  return immediately if queue is empty.
340 
341  Returns: BOOL: FALSE if queue was empty.
342  TRUE if get queued data.
343  *---------------------------------------------------------------------------*/
344 BOOL hwPadQueueRead( s32 flag )
345 {
346  PADStatus *padContainer;
347  BOOL isQueued;
348  u32 i;
349 
350  // Get pad data from valid data queue
351  isQueued = OSReceiveMessage( &PadValidMsgQ,
352  (OSMessage *)&padContainer, OS_MESSAGE_NOBLOCK );
353 
354  // If queue is empty, wait and sleep until coming pad input on v-retrace
355  if ( !isQueued )
356  {
357  if ( flag == OS_MESSAGE_BLOCK )
358  {
359  OSReceiveMessage( &PadValidMsgQ,
360  (OSMessage *)&padContainer, OS_MESSAGE_BLOCK );
361  }
362  else
363  {
364  return FALSE;
365  }
366  }
367 
368  // Copy status to hwPad
369  for ( i = 0 ; i < PAD_MAX_CONTROLLERS ; i ++ )
370  {
371  hwPadCopy( &padContainer[i], &hwPad[i] );
372  }
373 
374  // Release pad container
375  if ( !OSSendMessage( &PadEmptyMsgQ,
376  (OSMessage)padContainer, OS_MESSAGE_NOBLOCK ) )
377  {
378  // The valid queue never be full.
379  ASSERTMSG( 0, "Logical Error: Empty QUEUE is full." );
380  }
381 
382  return isQueued;
383 }
