121 bool Client::ConnectToServer( int ip, unsigned short port )
122 {
123  int result;
124  bool connected;
125 
126  memset( &m_server_address, 0, sizeof(m_server_address));
127  m_server_address.sin_family = AF_INET;
128  m_server_address.sin_port = htons( port );
129  m_server_address.sin_addr.s_addr = ip;
130 #ifdef __PLAT_NGC__
131  m_server_address.len = sizeof( sockaddr_in );
132 #else
133  memset ( &m_server_address.sin_zero, 0, 8 );
134 #endif
135 
136  Dbg_Printf( "%s Connecting to server\n", GetName());
137  connected = false;
138  do
139  {
140  if(( result = connect( m_socket, (struct sockaddr *)&m_server_address,
141  sizeof( m_server_address ))) == SOCKET_ERROR )
142  {
143 #if defined( __PLAT_WN32__ ) || defined( __PLAT_XBOX__ )
144  if( WSAGetLastError() != WSAEWOULDBLOCK )
145 #else
146 #ifdef __PLAT_NGPS__
147  if( sn_errno( m_socket ) != EWOULDBLOCK )
148 #endif
149 #endif
150  {
151  ReportError();
152  return false;
153  }
154  }
155  else
156  {
157  connected = true;
158  }
159  } while( !connected );
160 
161  // "Connecting" to INADDR_ANY disassociates our socket from all addresses
162  if( ip == INADDR_ANY )
163  {
164  m_connected = false;
165  }
166  else
167  {
168  m_connected = true;
169  }
170  return true;
171 }

178 void Client::SendEnqueuedMessages( Conn* conn )
179 {
180  int buf_len;
181  MsgPacketStamp msg;
182  bool buffer_full;
183 
184  // Should we send this frame?
185  if( ShouldSendThisFrame( conn ) == false )
186  {
187  return;
188  }
189 
190  buffer_full = false;
191  while( !buffer_full && MessagesToSend( conn ) && !BandwidthExceeded( conn ))
192  {
193  // Tack on a timestamp (for acking) if we are sending important
194  // messages in this packet
195  if( ImportantMessagesToSend( conn ))
196  {
197  MsgDesc msg_desc;
198 #ifdef __PLAT_NGPS__
199  SignalSema( m_send_semaphore_id );
200 #endif
201  msg.m_Packetstamp = (unsigned short) conn->m_latest_sent_packet_stamp;
202  //msg.m_Handle = conn->GetHandle();
203 
204  msg_desc.m_Data = &msg;
205  msg_desc.m_Length = sizeof( MsgPacketStamp );
206  msg_desc.m_Id = MSG_ID_TIMESTAMP;
207  msg_desc.m_Priority = HIGHEST_PRIORITY;
208  EnqueueMessage( conn->GetHandle(), &msg_desc );
209 #ifdef __PLAT_NGPS__
210  WaitSema( m_send_semaphore_id );
211 #endif
212  }
213 
214  // Send Latency Tests/Responses if applicable
215  if( ( conn->IsRemote()) &&
216  ( conn->IsForeign() == false ) &&
217  ( m_flags.TestMask( App::mDYNAMIC_RESEND )))
218  {
219  MsgDesc msg_desc;
220  MsgTimestamp latency_msg;
221  unsigned int cur_time;
222 
223  cur_time = m_Timestamp;
224  latency_msg.m_Timestamp = cur_time;
225  // send out a new latency test, keeping track of the time at which
226  // we sent it
227  if( ( conn->m_latency_test.m_SendTime == 0 ) ||
228  ( ( cur_time - conn->m_latency_test.m_SendTime ) > App::MAX_LATENCY ))
229  {
230  // If we never got a response, simulate an increased latency
231  if( conn->m_latency_test.m_SendTime > conn->m_latency_test.m_ReceiveTime )
232  {
233  unsigned int latency_value;
234 
235  latency_value = conn->GetAveLatency();
236  latency_value += 100;
237 
238  if( latency_value > App::MAX_LATENCY )
239  {
240  latency_value = App::MAX_LATENCY;
241  }
242  conn->m_latency_test.InputLatencyValue( latency_value );
243  }
244 
245  conn->m_latency_test.m_SendTime = cur_time;
246 
247 #ifdef __PLAT_NGPS__
248  SignalSema( m_send_semaphore_id );
249 #endif
250  msg_desc.m_Data = &latency_msg;
251  msg_desc.m_Id = MSG_ID_PING_TEST;
252  msg_desc.m_Length = sizeof( MsgTimestamp );
253  EnqueueMessage( conn->GetHandle(), &msg_desc );
254 
255 #ifdef __PLAT_NGPS__
256  WaitSema( m_send_semaphore_id );
257 #endif
258  }
259  }
260 
261  if( !BuildMsgStream( conn, QUEUE_DEFAULT ))
262  {
263  buffer_full = true;
264  }
265  // First, use up our bandwidth with re-sends or else they might remain un-sent
266  // indefinitely, causing a bad backup on the client if he's waiting on
267  // one particular sequence
268  if( !BuildMsgStream( conn, QUEUE_SEQUENCED, true ))
269  {
270  buffer_full = true;
271  }
272  if( !BuildMsgStream( conn, QUEUE_IMPORTANT, true ))
273  {
274  buffer_full = true;
275  }
276  if( !BuildMsgStream( conn, QUEUE_SEQUENCED ))
277  {
278  buffer_full = true;
279  }
280  if( !BuildMsgStream( conn, QUEUE_IMPORTANT ))
281  {
282  buffer_full = true;
283  }
284 
285  buf_len = conn->m_write_ptr - conn->m_write_buffer;
286  // If there is data to send
287  if( ( buf_len > 0 ) &&
288  ( conn->IsRemote()))
289  {
290  if( m_connected )
291  {
292  if( Send( conn->m_write_buffer, buf_len, 0 ))
293  {
294  m_TotalBytesOut += buf_len;
295  conn->GetOutboundMetrics()->AddPacket( buf_len + vUDP_PACKET_OVERHEAD, m_Timestamp );
296  conn->m_last_send_time = m_Timestamp;
297  conn->SetForceSendThisFrame( false );
298  }
299  else
300  {
301  // If it didn't send the messages properly, flag them for resend next frame
302  conn->FlagMessagesForResending( conn->m_latest_sent_packet_stamp );
303  conn->m_write_ptr = conn->m_write_buffer;
304  break;
305  }
306 
307  conn->m_write_ptr = conn->m_write_buffer;
308  }
309  else
310  {
311  if( SendTo( conn->GetIP(), conn->GetPort(), conn->m_write_buffer, buf_len, 0 ))
312  {
313  m_TotalBytesOut += buf_len;
314  conn->GetOutboundMetrics()->AddPacket( buf_len + vUDP_PACKET_OVERHEAD, m_Timestamp );
315  conn->m_last_send_time = m_Timestamp;
316  conn->SetForceSendThisFrame( false );
317  }
318  else
319  {
320  Dbg_Printf( "*** SendTo Error: Flagging messages for resending\n" );
321  // If it didn't send the messages properly, flag them for resend next frame
322  conn->FlagMessagesForResending( conn->m_latest_sent_packet_stamp );
323  conn->m_write_ptr = conn->m_write_buffer;
324  break;
325  }
326  conn->m_write_ptr = conn->m_write_buffer;
327  }
328  }
329 
330  conn->m_latest_sent_packet_stamp++;
331  }

339 void Client::ReceiveData( void )
340 {
341  Conn *conn;
342  int num_bytes, actual_data_len;
343  struct sockaddr from_address;
344 #ifdef __PLAT_NGC__
345  from_address.len = sizeof( sockaddr );
346 #else
347  int addr_len = sizeof( from_address );
348 #endif
349  struct sockaddr_in *foreign_address;
350 
351 
352 
353 
354  // Local clients never really receive. Their data is automatically "received" when
355  // the local server writes its data into the client's receive buffer automatically
356  if( IsLocal())
357  {
358  return;
359  }
360  // read data into local buffers
361  do
362  {
363  if( !m_connected )
364  {
365  num_bytes = recvfrom( m_socket, m_in_packet_buffer,
366  Manager::vMAX_PACKET_SIZE, 0, &from_address, &addr_len );
367  }
368  else
369  {
370  num_bytes = recv( m_socket, m_in_packet_buffer, Manager::vMAX_PACKET_SIZE, 0 );
371  }
372  if ( num_bytes < 0 )//== SOCKET_ERROR )
373  {
374 #if defined( __PLAT_WN32__ ) || defined( __PLAT_XBOX__ )
375  if( WSAGetLastError() != WSAEWOULDBLOCK )
376 #else
377 #ifdef __PLAT_NGPS__
378  if( sn_errno( m_socket ) != EWOULDBLOCK )
379 #else
380 #ifdef __PLAT_NGC__
381  if( num_bytes != SO_EWOULDBLOCK )
382 #endif
383 #endif
384 #endif
385  {
386 #ifdef NET_DEBUG_MESSAGES
387  Dbg_Printf( "Client Receive Error :" );
388 #ifdef __PLAT_WN32__
389  OutputDebugString( "Client Receive Error!\n" );
390 #endif
391  ReportError();
392 #endif
393  }
394  break;
395  }
396  else
397  {
398 #ifdef __PLAT_NGPS__
399  WaitSema( m_receive_semaphore_id );
400 #endif // __PLAT_NGPS__
401  m_TotalBytesIn += num_bytes;
402  if( m_flags & App::mSECURE )
403  {
404  actual_data_len = num_bytes - sizeof( unsigned short ); // total size - CRC size
405  }
406  else
407  {
408  actual_data_len = num_bytes;
409  }
410 
411  conn = GetConnectionByAddress( m_server_address.sin_addr.s_addr,
412  ntohs( m_server_address.sin_port ));
413  if( conn )
414  {
415  if(( conn->m_read_ptr + actual_data_len ) < ( conn->m_read_buffer + Conn::vREAD_BUFFER_LENGTH ))
416  {
417  Tmr::Time cur_time;
418 
419  cur_time = Tmr::GetTime();
420  if( !validate_and_copy_stream( m_in_packet_buffer, conn->m_read_ptr, num_bytes ))
421  {
422 #ifdef __PLAT_NGPS__
423  SignalSema( m_receive_semaphore_id );
424 #endif // __PLAT_NGPS__
425  // If the game would like to handle this foreign data, allow it to do so now
426  if( m_foreign_handler )
427  {
428  m_foreign_handler( m_in_packet_buffer, num_bytes, &from_address );
429  }
430  continue;
431  }
432  conn->m_read_ptr += actual_data_len;
433  conn->GetInboundMetrics()->AddPacket( num_bytes + vUDP_PACKET_OVERHEAD, m_Timestamp );
434  if( cur_time > conn->m_last_comm_time )
435  {
436  conn->m_last_comm_time = cur_time;
437  }
438  }
439  else
440  {
441 #ifdef __PLAT_WN32__
442  char message[1024];
443  sprintf( message, "****** %d: Dropped Packet\n", m_Timestamp );
444  OutputDebugString( message );
445 #endif // __PLAT_WN32__
446  }
447  }
448  else if( !m_connected )
449  {
450  Conn* foreign_conn;
451 
452  foreign_address = (struct sockaddr_in*) &from_address;
453  // Foreign connection. Create temporary connection to process this foreign message
454  foreign_conn = NewConnection( foreign_address->sin_addr.s_addr,
455  ntohs( foreign_address->sin_port ),
456  Conn::mREMOTE |
457  Conn::mFOREIGN );
458  if( foreign_conn )
459  {
460  foreign_conn->m_app = this;
461  //memcpy( foreign_conn->m_read_buffer, m_read_buffer, actual_data_len );
462  if( !validate_and_copy_stream( m_in_packet_buffer, foreign_conn->m_read_ptr, num_bytes ))
463  {
464 #ifdef __PLAT_NGPS__
465  SignalSema( m_receive_semaphore_id );
466 #endif // __PLAT_NGPS__
467  // If the game would like to handle this foreign data, allow it to do so now
468  if( m_foreign_handler )
469  {
470  m_foreign_handler( m_in_packet_buffer, num_bytes, &from_address );
471  }
472  delete foreign_conn;
473  continue;
474  }
475  foreign_conn->m_read_ptr = foreign_conn->m_read_buffer + actual_data_len;
476  foreign_conn->Invalidate();
477  }
478  }
479 #ifdef __PLAT_NGPS__
480  SignalSema( m_receive_semaphore_id );
481 #endif // __PLAT_NGPS__
482  }
483 
484  } while( num_bytes > 0 );
485 }

509 unsigned char Client::GetObjectAlias( int obj_id, int cur_time )
510 {
511  int hash, i;
512  bool quit;
513 
514  hash = obj_id & 0xff; // mask off all but lowest byte and use as a hash into the table
515  hash %= vNUM_ALIASES;
516  quit = false;
517  i = hash;
518  do
519  {
520  if( m_alias_table[i].m_Id == obj_id )
521  {
522  if( cur_time > m_alias_table[i].m_Expiration )
523  {
524  return (unsigned char) vNO_ALIAS;
525  }
526  return i;
527  }
528 
529  i = ( i + 1 ) % vNUM_ALIASES;
530  if( i == hash ) // complete revolution
531  {
532  break;
533  }
534  } while( !quit );
535 
536  return (unsigned char) vNO_ALIAS;
537 }

569  if( i == hash ) // complete revolution
570  {
571  break;
572  }
573  } while( !quit );
574 
575  // no free aliases
