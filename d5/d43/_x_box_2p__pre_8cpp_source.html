102 // Decode an LZSS encoded stream
103 // Runs at approx 12MB/s on PS2 without scratchpad (which slows it down in C)
104 // a 32x CD would run at 4.8MB/sec, although we seem to get a lot less than this
105 // with our current file system, more like 600K per seconds.....
106 // Need to write a fast streaming file system....
107 
108 void DecodeLZSS( unsigned char *pIn, unsigned char *pOut, int Len ) /* Just the reverse of Encode(). */
109 {
110  int i, j, k, r, c;
111 // uint64 LongWord;
112 // int bytes = 0;
113 // unsigned char *pScratch;
114  unsigned int flags;
115 
116 
117 
118 // int basetime = (int) Tmr::ElapsedTime(0);
119 // int len = Len;
120 
121 // int OutBytes = 4;
122 // int OutWord = 0;
123 
124 # if USE_BUFFER
125  for (i = 0; i < RINGBUFFERSIZE - MATCHLIMIT; i++)
126  text_buf[i] = ' ';
127  r = RINGBUFFERSIZE - MATCHLIMIT;
128 # else
129  r = RINGBUFFERSIZE - MATCHLIMIT;
130 # endif
131  flags = 0;
132  for ( ; ; )
133  {
134  if (((flags >>= 1) & 256) == 0)
135  {
136  ReadInto(c);
137  flags = c | 0xff00; /* uses higher byte cleverly */
138  } /* to count eight */
139  if (flags & 1)
140  {
141  ReadInto(c);
142  // putc(c, outfile);
143  WriteOut(c);
144  #if USE_BUFFER
145  text_buf[r++] = c;
146  r &= (RINGBUFFERSIZE - 1);
147  #else
148  r++;
149 // r &= (RINGBUFFERSIZE - 1); // don't need to wrap r until it is used
150  #endif
151  }
152  else
153  {
154  ReadInto(i);
155  ReadInto2(j); // note, don't need to check len on this one....
156 
157  i |= ((j & 0xf0) << 4); // i is 12 bit offset
158 
159  #if !USE_BUFFER
160  j = (j & 0x0f) + THRESHOLD+1; // j is 4 bit length (above the threshold)
161  unsigned char *pStream;
162  r &= (RINGBUFFERSIZE - 1); // wrap r around before it is used
163  pStream = pOut - r; // get base of block
164  if (i>=r) // if offset > r, then
165  pStream -= RINGBUFFERSIZE; // it's the previous block
166  pStream += i; // add in the offset to the base
167  r+=j; // add size to r
168  while (j--) // copy j bytes
169  WriteOut(*pStream++);
170  #else
171 
172  j = (j & 0x0f) + THRESHOLD; // j is 4 bit length (above the threshold)
173  for (k = 0; k <= j; k++) // just copy the bytes
174  {
175  c = text_buf[(i+k) & (RINGBUFFERSIZE - 1)];
176  WriteOut(c);
177  text_buf[r++] = c;
178  r &= (RINGBUFFERSIZE - 1);
179  }
180  #endif
181  }
182  }
183 // int Time = (int) Tmr::ElapsedTime(basetime);
184 // if (Time > 5)
185 // {
186 // printf("decomp time is %d ms, for %d bytes, %d bytes/second\n", Time,len, len * 1000 /Time );
187 // }
188 
189 }

203 PreFile::PreFile(uint8 *p_file_buffer)
204 {
205  mp_table = new Lst::StringHashTable<_File>(4);
206 
207  Dbg_AssertPtr(p_file_buffer);
208 
209  mp_buffer = p_file_buffer;
210  uint version = *((int *) (mp_buffer + 4));
211  Dbg_MsgAssert(version == CURRENT_PRE_VERSION,( "PRE file version not current"));
212  m_numEntries = *((int *)(mp_buffer + 8));
213 
214  uint8 *pEntry = mp_buffer + 12;
215  for (int i = 0; i < m_numEntries; i++)
216  {
217  int data_size = *((int *) pEntry);
218  int compressed_data_size = *((int *) (pEntry + 4));
219  int text_size = *((int *) (pEntry + 8));
220  int actual_data_size = (compressed_data_size != 0) ? compressed_data_size : data_size;
221 
222  char *pName = (char *) pEntry + 12;
223  uint8 *pCompressedData = pEntry + 12 + text_size;
224 
225  _File *pFile = new (Mem::Manager::sHandle().TopDownHeap()) _File;
226 
227  if (!mp_table->GetItem(pName))
228  {
229  // file is not in table, safe to add
230  mp_table->PutItem(pName, pFile);
231 
232  pFile->compressedDataSize = compressed_data_size;
233  pFile->pCompressedData = pCompressedData;
234  pFile->pData = NULL;
235  pFile->sky.POS = 0;
236  pFile->sky.SOF = data_size;
237  }
238  else
239  // Somehow, file is already in table, just kill it
240  // Later, I'll want to add an assert
241  delete pFile;
242 
243 # if DEBUG_PRE
244  printf(" %s, size %d\n", pName, data_size);
245 # endif
246 
247  pEntry += 12 + text_size + ((actual_data_size + 3) & (~3));
248  }
249 
250 # if DEBUG_PRE
251  printf("Done loading PRE\n");
252 # endif
253 
254  mp_activeFile = NULL;
255 }

331 // allocate memory and load file directly from a pre file, if it is there
332 // This avoids the problem of having to have the decompressed file in memory twice
333 // when we are loading directly, like with Pip::Load()
334 // using this enables us to actually load network game, where there is 1MB less heap during loading
335 //
336 // returns a pointer to the file in memory
337 // or NULL if the file is not in this pre file.
338 void *PreFile::LoadContainedFile( const char *pName, int *p_size, void *p_dest )
339 {
340  // The dest parameter is used only on Ps2.
341  Dbg_Assert( p_dest == NULL );
342 
343  _File *pFile = mp_table->GetItem( pName, false );
344  if( !pFile )
345  {
346  return NULL;
347  }
348 
349  *p_size = pFile->sky.SOF;
350 
351  p_dest = Mem::Malloc( pFile->sky.SOF );
352 
353  // Do we need to decompress file data?
354  if( !pFile->pData )
355  {
356  if( pFile->compressedDataSize )
357  {
358  // Need to uncompress data.
359  DecodeLZSS( pFile->pCompressedData, (uint8*)p_dest, pFile->compressedDataSize );
360  }
361  else
362  {
363  memcpy( p_dest, (void*)pFile->pCompressedData, pFile->sky.SOF );
364  }
365  }
366  else
367  {
368  memcpy( p_dest, (void*)pFile->pData, pFile->sky.SOF );
369  }
370  return p_dest;
371 }

386 {
387 
388  Dbg_AssertPtr( mp_activeFile );
389 
390  int seek_offs = mp_activeFile->sky.POS;
391  unsigned int limit = mp_activeFile->sky.SOF - seek_offs;
392  int copy_number = (count <= limit) ? count : limit;
393  if (mp_activeFile->compressedDataSize)
394  {
395  Dbg_MsgAssert(mp_activeFile->pData,( "file not uncompressed"));
396  memcpy( addr, mp_activeFile->pData + mp_activeFile->sky.POS, copy_number );
397  }
398  else
399  {
400  memcpy(addr, mp_activeFile->pCompressedData + mp_activeFile->sky.POS, copy_number);
401  }
402 
403  mp_activeFile->sky.POS += copy_number;
404 
405 # if DEBUG_PRE
406  printf("PRE: read %d bytes from file, handle 0x%x\n", copy_number, (int) mp_activeFile->pData);
407 # endif
408  return copy_number;

446 int PreFile::Seek(long offset, int origin)
447 {
448  int32 old_pos = mp_activeFile->sky.POS;
449 
450  // SEEK_CUR, SEEK_END, SEEK_SET
451  switch(origin)
452  {
453  case SEEK_CUR:
454  mp_activeFile->sky.POS += offset;
455  break;
456  case SEEK_END:
457  mp_activeFile->sky.POS = mp_activeFile->sky.SOF - offset;
458  break;
459  case SEEK_SET:
460  mp_activeFile->sky.POS = offset;
461  break;
462  default:
463  return -1;
464  }
465 
466  if (mp_activeFile->sky.POS < 0 || mp_activeFile->sky.POS > mp_activeFile->sky.SOF)
467  {
468  mp_activeFile->sky.POS = old_pos;
469  return -1;
470  }
471 
472  return 0;
473 }

581 {
582 
583  Dbg_AssertPtr(pHandle);

617 void PreMgr::loadPre(const char *pFilename, bool dont_assert)
618 {
619  m_blockPreLoading = (bool) Script::GetInt("block_pre_loading", false);
620 
621  if (m_blockPreLoading)
622  return;
623 
624  if( Pcm::UsingCD() )
625  {
626  Dbg_MsgAssert( 0,( "File access forbidden while PCM audio is in progress." ));
627  return;
628  }
629 
630  // Moved this to below the Pcm::UsingCD() call as that is used (bad!!) to turn off
631  // music and streams.
632 # ifdef __PLAT_NGC__
633  OSReport ( "Warning: Attempting to load pre file: %s (ignored)\n", pFilename );
634  return;
635 # endif // __PLAT_NGC__
636 
637 # ifdef __PLAT_NGPS__
638 // scePrintf("Loading PRE file %s...\n", pFilename);
639 # endif
640 
641  char fullname[256];
642  sprintf(fullname, "pre\\%s", pFilename);
643 
644  // Replace the .pre extension with .prx for Xbox PRE file.
645  fullname[strlen( fullname ) - 1] = 'x';
646 
647  // Create a lower-case version.
648  char lower[128];
649  for( unsigned int lp = 0; lp < strlen( fullname ) + 1; lp++ ) lower[lp] = tolower( fullname[lp] );
650 
651  Tmr::Time basetime = Tmr::ElapsedTime(0);
652 
653  int file_size;
654  uint8 *pFile = NULL;
655 
656  file_size = CanFileBeLoadedQuickly( fullname );
657  if ( file_size )
658  {
659  pFile = new (Mem::Manager::sHandle().TopDownHeap()) uint8[file_size];
660  bool fileLoaded= LoadFileQuicklyPlease( fullname, pFile );
661  if ( !fileLoaded )
662  {
663  printf( "pre file %s failed to load quickly.\n", fullname );
664  Dbg_MsgAssert( 0,( "Fire Matt - pre file didn't load quickly." ));
665  }
666  }
667  else
668  {
669 #ifdef __PLAT_NGC__
670  NsFile file;
671  if ( file.exist( fullname ) )
672  {
673  file.open( fullname );
674 #else
675  void *fp = File::Open(fullname, "rb");
676  if (!fp)
677  {
678 #endif // __PLAT_NGC__
679  // always run the code below if CD build
680  if (dont_assert || Config::CD())
681  {
682  printf("couldn't open %s\n", fullname);
683  return;
684  }
685  Dbg_MsgAssert(0,( "couldn't open %s\n", fullname));
686  }
687 
688 
689 #ifdef __PLAT_NGC__
690  file.read( &file_size, 4 );
691  file_size = nLtEn32(file_size);
692 #else
693  File::Read(&file_size, 4, 1, fp);
694  Dbg_MsgAssert(file_size > 0,( "%s has incorrect file size\n", fullname));
695 #endif // __PLAT_NGC__
696  if (Config::CD())
697  {
698  if (file_size <= 0) printf("%s has incorrect file size\n", fullname);
699  }
700 
701 
702  // Now allocates the .PRE file from the top of the heap, to avoid fragmentation.
703  pFile = new (Mem::Manager::sHandle().TopDownHeap()) uint8[file_size];
704  //uint8 *pFile = new uint8[file_size];
705 #ifdef __PLAT_NGC__
706  file.read( pFile + 4, file_size );
707  file.close();
708 #else
709  File::Read(pFile + 4, 1, file_size, fp);
710  File::Close(fp);
711 #endif // __PLAT_NGC__
712  }
713 
714  printf("load time for file size %d is %d ms\n", file_size, (int) Tmr::ElapsedTime(basetime));
715 
716  // the PRE file object winds up at the top of the heap, too. This is fine because
717  // it will be unloaded at the same time as the big file buffer
718  if (!mp_table->PutItem(pFilename, new (Mem::Manager::sHandle().TopDownHeap()) PreFile(pFile)))
719  Dbg_MsgAssert(0,( "PRE %s loaded twice", pFilename));
720 }
