67 void CStaticVehicleComponent::Finalize()
68 {
69  CModelComponent* p_model_component = static_cast< CModelComponent* >(GetModelComponentFromObject(GetObject()));
70  Dbg_Assert(p_model_component);
71  Nx::CModel* p_model = p_model_component->GetModel();
72  Dbg_Assert(p_model);
73  Nx::CHierarchyObject* p_hierarchy_objects = p_model->GetHierarchy();
74  Dbg_Assert(p_hierarchy_objects);
75 
76  Mth::Vector p_wheels [ CVehicleComponent::vVP_NUM_WHEELS ];
77 
78  for (int n = CVehicleComponent::vVP_NUM_WHEELS; n--; )
79  {
80  Mth::Vector& wheel_pos = p_wheels[n];
81 
82  Mth::Matrix wheel_matrix = (p_hierarchy_objects + 2 + n)->GetSetupMatrix();
83 
84  // rotate out of max coordinate system
85  wheel_pos[X] = -wheel_matrix[W][X];
86  wheel_pos[Y] = wheel_matrix[W][Z];
87  wheel_pos[Z] = -wheel_matrix[W][Y];
88  wheel_pos[W] = 1.0f;
89  }
90 
91  Mth::Matrix body_matrix = (p_hierarchy_objects + 1)->GetSetupMatrix();
92  Mth::Vector body_pos;
93  body_pos[X] = -body_matrix[W][X];
94  body_pos[Y] = body_matrix[W][Z];
95  body_pos[Z] = -body_matrix[W][Y];
96  body_pos[W] = 1.0f;
97 
98  CSkeletonComponent* p_skeleton_component = static_cast< CSkeletonComponent* >(GetSkeletonComponentFromObject(GetObject()));
99  Dbg_Assert(p_skeleton_component);
100  Dbg_Assert(p_skeleton_component->GetSkeleton());
101  Mth::Matrix* p_matrices = p_skeleton_component->GetSkeleton()->GetMatrices();
102  for (int i = 0; i < p_skeleton_component->GetSkeleton()->GetNumBones(); i++)
103  {
104  Mth::Matrix& matrix = p_matrices[i];
105 
106  // setup the matrix for each bone in the skeleton
107 
108  // shadow
109  if (i == 0)
110  {
111  matrix.Zero();
112  }
113 
114  // body
115  else if (i == 1)
116  {
117  matrix.Zero();
118  matrix[X][X] = 1.0f;
119  matrix[Y][Z] = -1.0f;
120  matrix[Z][Y] = 1.0f;
121  matrix[W] = body_pos;
122  }
123 
124  // wheel
125  else
126  {
127  Mth::Vector& wheel_pos = p_wheels[i - 2];
128 
129  matrix.Zero();
130  matrix[X][X] = -1.0f;
131  matrix[Y][Z] = 1.0f;
132  matrix[Z][Y] = 1.0f;
133  matrix[W] = wheel_pos;
134  }
135  }
136 }
