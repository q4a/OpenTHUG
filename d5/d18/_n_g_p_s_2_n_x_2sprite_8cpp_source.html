432  int j,k;
433 
434  uint8 *old_dma_loc = dma::pLoc;
435 
436  // initialise base pointers
437  if (IsInVRAM())
438  {
439  NextTBP = FontVramBase;
440  } else {
441  NextTBP = FontVramStart;
442  }
443 
444  // bits per texel and palette size
445  switch (PSM)
446  {
447  case PSMCT32:
448  BitsPerTexel = 32;
449  PaletteSize = 0;
450  break;
451  case PSMCT24:
452  BitsPerTexel = 24;
453  PaletteSize = 0;
454  break;
455  case PSMCT16:
456  BitsPerTexel = 16;
457  PaletteSize = 0;
458  break;
459  case PSMT8:
460  BitsPerTexel = 8;
461  PaletteSize = 256;
462  break;
463  case PSMT4:
464  BitsPerTexel = 4;
465  PaletteSize = 16;
466  break;
467  default:
468  printf("Unknown PSM %d in texture file\n", PSM);
469  exit(1);
470  }
471 
472  // bits per clut entry
473  if (BitsPerTexel < 16)
474  switch (CPSM)
475  {
476  case PSMCT32:
477  BitsPerClutEntry = 32;
478  break;
479  case PSMCT16:
480  BitsPerClutEntry = 16;
481  break;
482  default:
483  printf("Unknown CPSM %d in texture file\n", PSM);
484  exit(1);
485  }
486  else
487  BitsPerClutEntry = 0;
488 
489  // rearrange 256-colour cluts according to requirements of CSM1
490  if (!copy && PSM==PSMT8)
491  {
492  uint32 temp32;
493  uint16 temp16;
494  for (j=0; j<256; j+=32)
495  for (k=0; k<8; k++)
496  if (CPSM==PSMCT32)
497  {
498  temp32 = ((uint32 *)pData)[j+k+8];
499  ((uint32 *)pData)[j+k+8] = ((uint32 *)pData)[j+k+16];
500  ((uint32 *)pData)[j+k+16] = temp32;
501  }
502  else
503  {
504  temp16 = ((uint16 *)pData)[j+k+8];
505  ((uint16 *)pData)[j+k+8] = ((uint16 *)pData)[j+k+16];
506  ((uint16 *)pData)[j+k+16] = temp16;
507  }
508  }
509 
510  // calculate texture dimensions
511  Width = (TW>=0 ? 1<<TW : 0);
512  Height = (TH>=0 ? 1<<TH : 0);
513  TBW = (Width+63) >> 6;
514  if (BitsPerTexel<16 && TBW<2)
515  TBW = 2;
516 
517  // Calculate memory needed
518  int min_width = Width, min_height = Height;
519  TexMemLayout.getMinSize(min_width, min_height, PSM);
520  m_NumTexVramBlocks = TexMemLayout.getImageSize(min_width, min_height, PSM);
521  NumVramBytes = m_NumTexVramBlocks * 256;
522  //NumVramBytes = (TBW << 6) * Height * BitsPerTexel >> 3;
523 
524  // calculate sizes within file
525  NumClutBytes = PaletteSize * BitsPerClutEntry >> 3;
526  NumClutQWords = NumClutBytes >> 4;
527  NumTexBytes = Width * Height * BitsPerTexel >> 3;
528  NumTexQWords = NumTexBytes >> 4;
529 
530  //Dbg_Message("Texture size (%d, %d) of bitdepth %d and BufferWidth %d", Width, Height, BitsPerTexel, TBW);
531  //Dbg_MsgAssert(calcVramBytes <= NumVramBytes, ("Need more memory: %d vs. needed %d", NumVramBytes, calcVramBytes));
532 
533  // allocate memory for vif stream
534 #if STEX_USE_CALL
535  m_VifSize = 272;
536 #else
537  m_VifSize = 1024 + NumClutBytes + NumTexBytes; //256; //1216 + (((Width*Height+15)>>4)<<4);
538 #endif // STEX_USE_CALL
539  mp_VifData = (uint8 *)Mem::Malloc(m_VifSize);
540  Dbg_MsgAssert(mp_VifData, ("couldn't malloc memory for vif data"));
541  Dbg_MsgAssert(((uint32)mp_VifData & 15) == 0, ("vif data not qword aligned"));
542 
543  // build dma data in this buffer
544  dma::pLoc = mp_VifData;
545 
546  // calculate TBP
547  m_TBP = NextTBP;
548

550  //NextTBP = (((TBP<<8)+NumVramBytes+0x1FFF) & 0xFFFFE000) >> 8;
551  NextTBP = m_TBP + m_NumTexVramBlocks;

556  m_NumClutVramBlocks = 0;
557  else if (BitsPerTexel == 4)
558  m_NumClutVramBlocks = 4;

601 #else
602  // clut upload header
603  gif::Tag2(gs::A_D,1,PACKED,0,0,0,4);
604  mp_ClutBitBltBuf = gs::Reg2_pData(gs::BITBLTBUF,PackBITBLTBUF(0,0,0,m_CBP,1,CPSM));
