384 bool CBettingGuy::EndBetAttempt( uint32 goalId )
385 {
386  // this acts like a secondary deactivate
387  if ( IsActive() && m_currentMinigame == goalId && m_inAttempt && !m_shouldMove )
388  {
389  //printf("CBettingGuy::EndBetAttempt\n");
390  // printf("step 2\n");
391  m_inAttempt = false;
392 
393  Script::CStruct* pBetParams = GetBetParams();
394 
395  // check that they haven't exceeded the number of tries
396  int max_attempts;
397  pBetParams->GetInteger( "tries", &max_attempts, Script::ASSERT );
398 
399  // we're challenging them to break a record
400  CGoalManager* pGoalManager = GetGoalManager();
401  Dbg_Assert( pGoalManager );
402  Script::CStruct* pMinigameParams = pGoalManager->GetGoal( m_currentMinigame )->GetParams();
403  Dbg_Assert( pMinigameParams );
404  if ( m_currentChallenge )
405  {
406  // printf("step 3\n");
407  int last_attempt;
408  if ( pMinigameParams->GetInteger( "last_attempt", &last_attempt, Script::NO_ASSERT ) )
409  {
410  printf("got a last attempt of %i\n", last_attempt);
411  printf("m_currentChallenge is %i\n", m_currentChallenge);
412  if ( last_attempt >= m_currentChallenge )
413  {
414  // printf("you won the bet!\n");
415  // beat the bet and call the minigame's bet_success script
416  this->Win();
417  return true;
418  }
419  }
420  }
421  // check that this isn't their last try
422  m_numTries++;
423  if ( m_numTries >= max_attempts )
424  {
425  Lose();
426  return true;
427  }
428  // if it's not a challenge minigame, it's probably a trickspot
429  // minigame. In that case, we'll be told when the game is won
430  // the the gap script.
431  }
432  return false;
433 }
