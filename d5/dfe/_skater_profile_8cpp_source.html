310 uint32 CSkaterProfile::WriteToBuffer(uint8 *pBuffer, uint32 BufferSize, bool ignoreFaceData )
311 {
312  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
313 
314  uint32 totalSize = 0;
315  uint32 chunkSize = 0;
316 
317  chunkSize = m_Appearance.WriteToBuffer(pBuffer, BufferSize, ignoreFaceData);
318  totalSize += chunkSize;
319 
320  // write out trick mappings
321  BufferSize -= chunkSize;
322  pBuffer += chunkSize;
323 // compress_trick_mappings( "trick_mapping" );
324  chunkSize = Script::WriteToBuffer(GetTrickMapping( CRCD(0xd544aa2d,"trick_mapping") ), pBuffer, BufferSize );
325  totalSize += chunkSize;
326 // decompress_trick_mappings( "trick_mapping" );
327 
328  // write out special tricks... these could
329  // theoretically be compressed, but not using
330  // the same method as the regular tricks
331  // (because they're in a different format)
332  BufferSize -= chunkSize;
333  pBuffer += chunkSize;
334  Script::CStruct* pSpecialsStructure = GetSpecialTricksStructure();
335  chunkSize = Script::WriteToBuffer(pSpecialsStructure, pBuffer, BufferSize);
336  totalSize += chunkSize;
337 
338  // write out any extra info that needs to go across
339  // (i.e. stats, name, etc.)
340  // (some data, such as your default specials
341  // and your default appearance, aren't needed)
342  BufferSize -= chunkSize;
343  pBuffer += chunkSize;
344  chunkSize = write_extra_info_to_buffer(pBuffer, BufferSize);
345  totalSize += chunkSize;
346 
347  Mem::Manager::sHandle().PopContext();
348 
349  return totalSize;
350 }

388 uint32 CSkaterProfile::write_extra_info_to_buffer(uint8 *pBuffer, uint32 BufferSize)
389 {
390  Script::CStruct* pTempStructure = new Script::CStruct;
391 
392  Dbg_Assert( pTempStructure );
393 
394  // for each component in info, see if it's needed in-game...
395  // such as the stats, trick style, etc.
396  // (most of the other information is only needed in the front end)
397  Script::CComponent *pComp = m_Info.GetNextComponent();
398  while ( pComp )
399  {
400  Script::CComponent* pNextComp = m_Info.GetNextComponent( pComp );
401 
402  switch ( pComp->mNameChecksum )
403  {
404  case 0xa1dc81f9: // name
405  case 0x7d02bcc3: // stance
406  case 0xc15dbf86: // pushstyle
407  pTempStructure->AddComponent( pComp->mNameChecksum, ESYMBOLTYPE_NAME, (int)pComp->mChecksum );
408  break;
409 
410  case 0x2ab66cb8: // display_name
411  case 0x2820e997: // file_name
412  case 0x562e3ecd: // first_name
413  pTempStructure->AddComponent( pComp->mNameChecksum, ESYMBOLTYPE_STRING, pComp->mpString );
414  break;
415 
416  case 0x439f4704: // air
417  case 0xaf895b3f: // run
418  case 0x9b65d7b8: // ollie
419  case 0xf0d90109: // speed
420  case 0xedf5db70: // spin
421  case 0x9016b4e7: // switch
422  case 0xf73a13e3: // rail_balance
423  case 0xae798769: // lip_balance
424  case 0xb1fc0722: // manual_balance
425  case 0xd82f8ac8: // is_pro
426  case 0x6dcb497c: // flip_speed
427  case 0x3f813177: // is_male
428  pTempStructure->AddComponent( pComp->mNameChecksum, ESYMBOLTYPE_INTEGER, (int)pComp->mIntegerValue );
429  break;
430 
431  default:
432  // doesn't need to go across
433  break;
434  }
435 
436  pComp = pNextComp;
437  }
438 
439  uint32 totalSize = Script::WriteToBuffer(pTempStructure, pBuffer, BufferSize);
440 
441  delete pTempStructure;
442 
443  return totalSize;
444 }

469 void CSkaterProfile::compress_trick_mappings( const char* pTrickMappingName )
470 {
471 
472  Mdl::Skate* skate_mod = Mdl::Skate::Instance();
473  CTrickChecksumTable* pTrickChecksumTable = skate_mod->GetTrickChecksumTable();
474 
475  Script::CStruct* pTempStructure;
476  if ( !m_Info.GetStructure( pTrickMappingName, &pTempStructure ) )
477  {
478  Dbg_MsgAssert( 0, ( "Trick mapping %s not found... already compressed?", pTrickMappingName ) );
479  }
480 
481  int compressed = 0;
482  pTempStructure->GetInteger( CRCD(0x68a9b4e1,"compressed"), &compressed );
483  if ( compressed )
484  {
485  Dbg_MsgAssert( 0, ( "This structure is already compressed" ) );
486  return;
487  }
488 
489  // take all the integers, and convert them to checksums
490  Script::CComponent *pComp = pTempStructure->GetNextComponent();
491  while ( pComp )
492  {
493  Script::CComponent* pNextComp = pTempStructure->GetNextComponent( pComp );
494 
495 
496  if (pComp && pComp->mType==ESYMBOLTYPE_NAME && pComp->mNameChecksum != CRCD(0x68a9b4e1,"compressed") )
497  {
498  int compressedIndex = pTrickChecksumTable->GetIndexFromChecksum( pComp->mChecksum );
499  if ( compressedIndex != -1 )
500  {
501  uint32 nameChecksum = pComp->mNameChecksum; // Mick, need to store this now, as we are about to remove pComp
502  pTempStructure->RemoveComponent( nameChecksum );
503  pTempStructure->AddComponent( nameChecksum, ESYMBOLTYPE_INTEGER, compressedIndex );
504  }
505  }
506  pComp = pNextComp;
507  }
508 
509  // add the compressed flag
510  pTempStructure->AddComponent( CRCD(0x68a9b4e1,"compressed"), ESYMBOLTYPE_INTEGER, 1 );
511 }

518 void CSkaterProfile::decompress_trick_mappings( const char* pTrickMappingName )
519 {
520 
521  Mdl::Skate* skate_mod = Mdl::Skate::Instance();
522  CTrickChecksumTable* pTrickChecksumTable = skate_mod->GetTrickChecksumTable();
523 
524  Script::CStruct* pTempStructure;
525  if ( !m_Info.GetStructure( pTrickMappingName, &pTempStructure ) )
526  {
527  Dbg_MsgAssert( 0, ( "Trick mapping %s not found... already compressed?", pTrickMappingName ) );
528  }
529 
530  int compressed = 0;
531  pTempStructure->GetInteger( CRCD(0x68a9b4e1,"compressed"), &compressed );
532  if ( !compressed )
533  {
534 #ifdef __NOPT_ASSERT__
535  Script::PrintContents(&m_Info);
536  Script::PrintContents(pTempStructure);
537 #endif
538  Dbg_MsgAssert( 0, ( "This structure is not compressed" ) );
539  return;
540  }
541 
542  // take all the integers, and convert them to checksums
543  Script::CComponent *pComp = pTempStructure->GetNextComponent();
544  while ( pComp )
545  {
546  Script::CComponent* pNextComp = pTempStructure->GetNextComponent( pComp );
547 
548  if (pComp && pComp->mType==ESYMBOLTYPE_INTEGER && pComp->mNameChecksum != CRCD(0x68a9b4e1,"compressed") )
549  {
550  int uncompressedChecksum = pTrickChecksumTable->GetChecksumFromIndex( pComp->mIntegerValue );
551  uint32 nameChecksum = pComp->mNameChecksum; // Mick, need to store this now, as we are about to remove pComp
552  pTempStructure->RemoveComponent( nameChecksum );
553  pTempStructure->AddComponent( nameChecksum, ESYMBOLTYPE_NAME, uncompressedChecksum );
554  }
555 
556  pComp = pNextComp;
557  }
558 
559  // reset the compressed flag
560  pTempStructure->AddComponent( CRCD(0x68a9b4e1,"compressed"), ESYMBOLTYPE_INTEGER, 0 );
561 }

765 bool CSkaterProfile::SetSpecialTrickInfo( int slot_num, const SSpecialTrickInfo& theInfo, bool update_mappings )
766 {
767  Script::CStruct* pStructure = GetSpecialTricksStructure();
768  Dbg_Assert( pStructure );
769 
770  Script::CArray* pArray;
771  pStructure->GetArray( NONAME, &pArray, true );
772  Dbg_MsgAssert( pArray->GetSize() == vMAXSPECIALTRICKSLOTS, ( "Expected %i elements in specials array instead of %d", vMAXSPECIALTRICKSLOTS, pArray->GetSize() ) );
773 
774  if ( !(slot_num >= 0 && slot_num < (int)pArray->GetSize()) )
775  {
776  // out of range index...
777  // this is possible if you press the square button while on the "Done" element
778  // this isn't the best place to screen this out, but it's the safest...
779  return false;
780  }
781 
782  uint32 trickName = theInfo.m_TrickName;
783  uint32 trickSlot = theInfo.m_TrickSlot;
784 
785  if ( theInfo.IsUnassigned() )
786  {
787  // if the incoming trick is "unassigned"
788  // then clear that slot
789  trickName = CRCD(0xf60c9090,"unassigned");
790  trickSlot = CRCD(0xf60c9090,"unassigned");
791  }
792 
793  Script::CStruct* pSubStructure = pArray->GetStructure( slot_num );
794  pSubStructure->AddComponent( CRCD(0x5b077ce1,"trickname"), ESYMBOLTYPE_NAME, (int)trickName );
795  pSubStructure->AddComponent( CRCD(0xa92a2280,"trickslot"), ESYMBOLTYPE_NAME, (int)trickSlot );
796 
797  if ( theInfo.m_isCat )
798  {
799  pSubStructure->AddInteger( CRCD(0xb56a8816,"isCat"), 1 );
800  }
801  else
802  pSubStructure->RemoveComponent( CRCD(0xb56a8816,"isCat") );
803 
804  // now that a trick config has changed,
805  // update the trick mappings on any existing skaters
806  if ( update_mappings )
807  {
808  Mdl::Skate * pSkate = Mdl::Skate::Instance();
809  Obj::CSkater* pSkater = pSkate->GetLocalSkater();
810  if ( pSkater )
811  {
812  Obj::CTrickComponent* pTrickComponent = GetTrickComponentFromObject(pSkater);
813  Dbg_Assert( pTrickComponent );
814  pTrickComponent->UpdateTrickMappings( this );
815  }
816  }
817  Script::RunScript( CRCD(0xfab5c6eb,"disable_replays") );
818  return true;
819 }
