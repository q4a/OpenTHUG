144 void CModelBuilder::BuildModel( CModelAppearance* pAppearance, Nx::CModel* pModel, Gfx::CSkeleton* pSkeleton, uint32 buildScriptName )
145 {
146 // Tmr::Time baseTime = Tmr::ElapsedTime(0);
147 
148  Dbg_Assert( pAppearance );
149  Dbg_Assert( pModel );
150 
151  // don't destroy the existing model...
152  // leave it up to the particular buildscript
153 // pModel->ClearGeoms();
154 
155  Dbg_MsgAssert( !mp_model, ( "Static temporary model already exists" ) );
156  mp_model = pModel;
157 
158  Dbg_MsgAssert( !mp_skeleton, ( "Static skeleton already exists" ) );
159  mp_skeleton = pSkeleton;
160 
161  Dbg_MsgAssert( !mp_appearance, ( "Static temporary appearance already exists" ) );
162  mp_appearance = new CModelAppearance;
163 
164  // clear it out from the previous use
165  mp_appearance->Init();
166 
167  // GJ 9/8/03: since it's a temporary variable, we don't want
168  // it on the skater geom heap (which would fragment it on the
169  // PS2 with 17K worth of face texture data)
170  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
171 
172  // copy over the desired appearance
173  // (must make a copy, so that we can
174  // make changes to it)
175  *mp_appearance = *pAppearance;
176 
177  Mem::Manager::sHandle().PopContext();
178 
179  // this script is responsible for looping
180  // through all the supported partChecksums
181  // and running all the supported operations
182  // on them. Which partChecksums and operations
183  // are "supported" should be completely in
184  // script.
185 // Dbg_Message( "BuildModel - %s", Script::FindChecksumName(buildScriptName) );
186  Script::RunScript( buildScriptName, NULL, this );
187 
188  mp_model = NULL;
189  mp_skeleton = NULL;
190 
191  delete mp_appearance;
192  mp_appearance = NULL;
193 
194 // Dbg_Message( "BuildModel took %d ms", Tmr::ElapsedTime( baseTime ) );
195 }

202 bool CModelBuilder::CallMemberFunction( uint32 checksum, Script::CStruct* pParams, Script::CScript* pScript )
203 {
204  switch ( checksum )
205  {
206  case 0x30e2951d: // DebugPrintAppearance
207  {
208  // print out the appearance to make sure
209  // it's coming through correctly in net games
210  mp_appearance->PrintContents();
211  }
212  return true;
213 
214  case 0x53da94f1: // ModelAddGeom
215  {
216  uint32 partChecksum;
217  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
218  model_add_geom(partChecksum);
219  }
220  return true;
221 
222  case 0x681a03af: // ModelHideGeom
223  {
224  uint32 partChecksum;
225  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
226 
227  int hidden;
228  pParams->GetInteger( NONAME, &hidden, Script::ASSERT );
229 
230  model_hide_geom(partChecksum, hidden);
231  }
232  return true;
233 
234  case 0xaee6e915: // GeomModulateColor
235  {
236  geom_modulate_color( pParams );
237  }
238  return true;
239 
240  case 0x18924da1: // GeomSetUVOffset
241  {
242  uint32 partChecksum;
243  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
244 
245  uint32 material;
246  pParams->GetChecksum( CRCD(0x83418a6a,"material"), &material, Script::ASSERT );
247 
248  int pass;
249  pParams->GetInteger( CRCD(0x318f2bdb,"pass"), &pass, Script::ASSERT );
250 
251  geom_set_uv_offset(partChecksum, material, pass);
252  }
253  return true;
254 
255  case 0x3bf41299: // GeomAllocateUVMatrixParams
256  {
257  uint32 material;
258  pParams->GetChecksum( CRCD(0x83418a6a,"material"), &material, Script::ASSERT );
259 
260  int pass;
261  pParams->GetInteger( CRCD(0x318f2bdb,"pass"), &pass, Script::ASSERT );
262 
263  geom_allocate_uv_matrix_params(material, pass);
264  }
265  return true;
266 
267  case 0xaa42a104: // GeomReplaceTexture
268  {
269  uint32 partChecksum;
270  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
271  geom_replace_texture(partChecksum);
272  }
273  return true;
274 
275  case 0x9041e901: // ModelRemovePolys
276  {
277  model_remove_polys();
278  }
279  return true;
280 
281  case 0x50f1285b: // ModelApplyObjectScale
282  {
283  Script::CStruct* pBodyShapeStructure = NULL;
284  if ( pParams->GetStructure( vCHECKSUM_BODY_SHAPE, &pBodyShapeStructure, false ) )
285  {
286  Mth::Vector vec = mp_model->GetScale();
287 
288  Mth::Vector theScale( 1.0f, 1.0f, 1.0f, 1.0f );
289  if ( Gfx::GetScaleFromParams( &theScale, pBodyShapeStructure ) )
290  {
291  vec[X] *= theScale[X];
292  vec[Y] *= theScale[Y];
293  vec[Z] *= theScale[Z];
294  vec[W] *= theScale[W];
295 
296  // if the body shape has a scale
297  mp_model->SetScale( vec );
298  }
299  }
300  else
301  {
302  model_apply_object_scale();
303  }
304  }
305  return true;
306 
307  case 0x21aec583: // ModelApplyBoneScale
308  {
309  if ( pParams->ContainsComponentNamed( CRCD(0x3f5f5bc2,"bone_scaling") ) )
310  {
311  uint32 partChecksum;
312  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
313  model_apply_bone_scale( partChecksum );
314  }
315  }
316  return true;
317 
318  case 0xc0bc8271: // ModelApplyBodyShape
319  {
320  Script::CStruct* pBodyShapeStructure = NULL;
321  if ( pParams->GetStructure( vCHECKSUM_BODY_SHAPE, &pBodyShapeStructure, false ) )
322  {
323  // GJ: PATCH TO FIX STEVE CABALLERO'S INCORRECT WEIGHTING IN BIGHEAD MODE
324  Script::CStruct* pActualStruct = mp_appearance->GetActualDescStructure( CRCD(0x650fab6d,"skater_m_head") );
325  if ( pActualStruct && pParams->ContainsFlag( CRCD(0x3c4d849d,"is_bighead_cheat") ) )
326  {
327  pActualStruct->GetStructure( CRCD(0x1ade0dd8,"bighead_scale_info"), &pBodyShapeStructure );
328  }
329 
330  // if a body shape was explicitly specified
331  // (for kid and gorilla modes)
332  if ( mp_skeleton )
333  {
334  return mp_skeleton->ApplyBoneScale( pBodyShapeStructure );
335  }
336  }
337  else
338  {
339  model_apply_body_shape();
340  }
341  }
342  return true;
343 
344  case 0x73ab02bc: // ModelApplyFaceTexture
345  {
346 #ifdef __PLAT_NGPS__
347  bool success = true;
348 
349  Gfx::CFaceTexture* pFaceTexture = mp_appearance->GetFaceTexture() ;
350 
351  if ( pFaceTexture && pFaceTexture->IsValid() )
352  {
353  if ( mp_model )
354  {
355  const char* pSrcTexture;
356  pParams->GetText( CRCD(0x9fbbdb72,"src"), &pSrcTexture, Script::ASSERT );
357 
358  // by default, it searches globally in the model for the correct texture
359  uint32 partChecksumToReplace = Nx::CModel::vREPLACE_GLOBALLY;
360  pParams->GetChecksum( CRCD(0xa01371b1,"in"), &partChecksumToReplace, Script::NO_ASSERT );
361 
362  success = mp_model->ApplyFaceTexture( pFaceTexture, pSrcTexture, partChecksumToReplace );
363  }
364  }
365 
366  return success;
367 #else
368  // X-box and NGC don't have to implement these
369  // because they don't do face textures...
370  return true;
371 #endif
372  }
373 
374  case 0x6516b484: // AppearanceAllowScalingCheat
375  {
376  Script::CStruct* pActualStruct = mp_appearance->GetActualDescStructure( vCHECKSUM_BODY );
377  if ( pActualStruct )
378  {
379  return !pActualStruct->ContainsFlag( vCHECKSUM_NO_SCALING_ALLOWED );
380  }
381 
382  return true;
383  }
384  break;
385 
386  case 0xcdeea636: // ModelResetScale
387  {
388  model_reset_scale();
389  }
390  return true;
391 
392  case 0xbff957a9: // ModelClearGeom
393  {
394  uint32 partChecksum;
395  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
396  model_clear_geom( partChecksum );
397  }
398  return true;
399 
400  case 0x0ecf0248: // ModelClearAllGeoms
401  {
402  model_clear_all_geoms();
403  }
404  return true;
405 
406  case 0xb1c39a54: // ModelRunScript
407  {
408  uint32 partChecksum;
409  pParams->GetChecksum( vCHECKSUM_PART, &partChecksum, Script::ASSERT );
410 
411  uint32 targetChecksum;
412  pParams->GetChecksum( vCHECKSUM_TARGET, &targetChecksum, Script::ASSERT );
413 
414  // runs an embedded script
415  model_run_script(partChecksum, targetChecksum);
416  }
417  return true;
418 
419  case 0xcefe8478: // ModelFinalize
420  {
421  model_finalize();
422  }
423  return true;
424  }
425 
426  return Obj::CObject::CallMemberFunction( checksum, pParams, pScript );
427 }

457 bool CModelBuilder::model_add_geom( uint32 partChecksum )
458 {
459  Dbg_Assert( mp_model );
460  Dbg_Assert( mp_appearance );
461 
462  Script::CStruct* pActualStruct = mp_appearance->GetActualDescStructure( partChecksum );
463  if ( pActualStruct )
464  {
465  int supports_multiple_colors = 0;
466  pActualStruct->GetInteger( CRCD(0x92b43e79,"multicolor"), &supports_multiple_colors, Script::NO_ASSERT );
467 
468 #ifdef __PLAT_NGPS__
469  Gfx::CFaceTexture* pFaceTexture = mp_appearance->GetFaceTexture() ;
470 
471  // GJ: THPS5 KLUDGE
472  // WillEventuallyHaveFaceTexture is a flag used in net games...
473  // this is needed because there is no face texture pointer
474  // in net games (because it's handled in a separate net packet)...
475 
476  if ( ( pFaceTexture && pFaceTexture->IsValid() ) || ( mp_appearance->WillEventuallyHaveFaceTexture() ) )
477  {
478  // if the special parameter doesn't exist, then
479  // fall through to the normal mesh-loading code...
480 
481  const char* pMeshName;
482  if ( pActualStruct->GetString( CRCD(0xc9aed80f,"mesh_if_facemapped"), &pMeshName, Script::NO_ASSERT ) )
483  {
484  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
485  return true;
486  }
487  }
488 #endif
489  {
490  const char* pMeshName;
491  if ( pActualStruct->GetString( vCHECKSUM_MESH, &pMeshName, Script::NO_ASSERT ) )
492  {
493  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
494  }
495  if ( pActualStruct->GetString( vCHECKSUM_MESH1, &pMeshName, Script::NO_ASSERT ) )
496  {
497  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
498  }
499  if ( pActualStruct->GetString( vCHECKSUM_MESH2, &pMeshName, Script::NO_ASSERT ) )
500  {
501  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
502  }
503  if ( pActualStruct->GetString( vCHECKSUM_MESH3, &pMeshName, Script::NO_ASSERT ) )
504  {
505  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
506  }
507  if ( pActualStruct->GetString( vCHECKSUM_MESH4, &pMeshName, Script::NO_ASSERT ) )
508  {
509  mp_model->AddGeom( pMeshName, partChecksum, m_useAssetManager, m_texDictOffset, false, supports_multiple_colors );
510  }
511  }
512  }
513 
514  return true;
515 }

600 bool CModelBuilder::model_apply_bone_scale( uint32 partChecksum )
601 {
602  Dbg_Assert( mp_model );
603  Dbg_Assert( mp_skeleton );
604  Dbg_Assert( mp_appearance );
605 
606  Script::CStruct* pVirtualStruct = mp_appearance->GetVirtualDescStructure( partChecksum );
607  if ( pVirtualStruct )
608  {
609  if ( !pVirtualStruct->ContainsComponentNamed( vCHECKSUM_USE_DEFAULT_SCALE ) )
610  {
611  return false;
612  }
613 
614  int use_default_scale;
615  pVirtualStruct->GetInteger( vCHECKSUM_USE_DEFAULT_SCALE, &use_default_scale );
616 
617  if ( use_default_scale )
618  {
619  return false;
620  }
621 
622  int x, y, z;
623  if ( pVirtualStruct->GetInteger( vCHECKSUM_X, &x, false )
624  && pVirtualStruct->GetInteger( vCHECKSUM_Y, &y, false )
625  && pVirtualStruct->GetInteger( vCHECKSUM_Z, &z, false ) )
626  {
627  Script::CArray* pBoneGroupArray = Script::GetArray( partChecksum, Script::NO_ASSERT );
628  if ( pBoneGroupArray )
629  {
630  Mth::Vector theBoneScaleVector;
631 // theBoneScaleVector[X] = (float)x / 100.0f;
632 // theBoneScaleVector[Y] = (float)y / 100.0f;
633 // theBoneScaleVector[Z] = (float)z / 100.0f;
634  theBoneScaleVector[W] = 1.0f;
635 
636  for ( uint32 i = 0; i < pBoneGroupArray->GetSize(); i++ )
637  {
638  uint32 boneChecksum = pBoneGroupArray->GetChecksum(i);
639 
640  // get the existing scale...
641  mp_skeleton->GetBoneScale( boneChecksum, &theBoneScaleVector );
642 
643  // then combine it with the new desired bone scale...
644  theBoneScaleVector[X] *= ( (float)x / 100.0f );
645  theBoneScaleVector[Y] *= ( (float)y / 100.0f );
646  theBoneScaleVector[Z] *= ( (float)z / 100.0f );
647 
648  bool isLocalScale = true;
649 
650  // GJ: The shoulder bones are handled slightly differently
651  // so that the female skaters aren't so broad-shouldered.
652  Script::CArray* pShoulderScalingArray = Script::GetArray( CRCD(0x20d9ac2f,"nonlocal_bones"), Script::NO_ASSERT );
653  if ( pShoulderScalingArray )
654  {
655  for ( uint32 i = 0; i < pShoulderScalingArray->GetSize(); i++ )
656  {
657  if ( pShoulderScalingArray->GetChecksum(i) == boneChecksum )
658  {
659  isLocalScale = false;
660  }
661  }
662  }
663 
664  mp_skeleton->SetBoneScale( boneChecksum, theBoneScaleVector, isLocalScale );
665  }
666  }
667  }
668  }
669 
670  return false;
671 }

680  Dbg_Assert( mp_model );
