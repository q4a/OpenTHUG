55 // Deletes any entity pointed to by p_comp, so that p_comp can safely be deleted afterwards.
56 void CleanUpComponent(CComponent *p_comp)
57 {
58  Dbg_MsgAssert(p_comp,("NULL p_comp"));
59 
60  switch (p_comp->mType)
61  {
62  case ESYMBOLTYPE_STRING:
63  case ESYMBOLTYPE_LOCALSTRING:
64  if (p_comp->mpString)
65  {
66  DeleteString(p_comp->mpString);
67  }
68  break;
69  case ESYMBOLTYPE_PAIR:
70  if (p_comp->mpPair)
71  {
72  delete p_comp->mpPair;
73  }
74  break;
75  case ESYMBOLTYPE_VECTOR:
76  if (p_comp->mpVector)
77  {
78  delete p_comp->mpVector;
79  }
80  break;
81  case ESYMBOLTYPE_STRUCTURE:
82  if (p_comp->mpStructure)
83  {
84  delete p_comp->mpStructure;
85  }
86  break;
87  case ESYMBOLTYPE_ARRAY:
88  if (p_comp->mpArray)
89  {
90  CleanUpArray(p_comp->mpArray);
91  delete p_comp->mpArray;
92  }
93  break;
94  case ESYMBOLTYPE_QSCRIPT:
95  if (p_comp->mpScript)
96  {
97  Mem::Free(p_comp->mpScript);
98  }
99  p_comp->mScriptSize=0;
100  break;
101  default:
102  break;
103  }
104  // This will zero the union, which includes all the above pointers.
105  p_comp->mUnion=0;
106 
107  p_comp->mType=ESYMBOLTYPE_NONE;
108 }

110 // Copies the contents of p_source into p_dest, but without copying any pointers over.
111 // Eg, if p_source contains an array, a new array will be created for p_dest.
112 void CopyComponent(CComponent *p_dest, const CComponent *p_source)
113 {
114  Dbg_MsgAssert(p_dest,("NULL p_dest"));
115  Dbg_MsgAssert(p_source,("NULL p_source"));
116 
117  // Make sure p_dest is cleaned up first.
118  if (p_dest->mUnion) // This 'if' is just a small speed optimization to avoid an unnecessay call
119  {
120  CleanUpComponent(p_dest);
121  }
122 
123  p_dest->mType=p_source->mType;
124  p_dest->mNameChecksum=p_source->mNameChecksum;
125 
126  switch (p_source->mType)
127  {
128  case ESYMBOLTYPE_NONE:
129  break;
130 
131  case ESYMBOLTYPE_INTEGER:
132  case ESYMBOLTYPE_FLOAT:
133  case ESYMBOLTYPE_NAME:
134  p_dest->mUnion=p_source->mUnion;
135  break;
136 
137  case ESYMBOLTYPE_STRING:
138  Dbg_MsgAssert(p_source->mpString,("NULL p_source->mpString ?"));
139  p_dest->mpString=CreateString(p_source->mpString);
140  break;
141 
142  case ESYMBOLTYPE_LOCALSTRING:
143  Dbg_MsgAssert(p_source->mpLocalString,("NULL p_source->mpLocalString ?"));

145  break;

147  case ESYMBOLTYPE_PAIR:

150  p_dest->mpPair->mX=p_source->mpPair->mX;

152  break;

154  case ESYMBOLTYPE_VECTOR:

157  p_dest->mpVector->mX=p_source->mpVector->mX;

159  p_dest->mpVector->mZ=p_source->mpVector->mZ;
160  break;

164  p_dest->mpStructure=new CStruct;
165  *p_dest->mpStructure=*p_source->mpStructure;
166  break;

172  break;

190  *p_to++=*p_from++;
191  }
192 
193  break;
194  }
195

197  Dbg_MsgAssert(0,("Bad p_source->mType of '%s'",GetTypeName(p_source->mType)));

312 // Copies the contents of p_source into p_dest. No pointers will be copied over, new entities will be
313 // created for p_dest. So p_source can safely be deleted afterwards.
314 void CopyArray(CArray *p_dest, const CArray *p_source)
315 {
316  Dbg_MsgAssert(p_source,("NULL p_source ?"));
317  Dbg_MsgAssert(p_dest,("NULL p_dest ?"));
318 
319  // Make sure that p_dest is cleaned up first.
320  CleanUpArray(p_dest);
321 
322  ESymbolType type=p_source->GetType();
323  uint32 size=p_source->GetSize();
324 
325  p_dest->SetSizeAndType(size,type);
326 
327  if (size==0)
328  {
329  // Finished.
330  return;
331  }
332 
333  uint32 *p_source_array=p_source->GetArrayPointer();
334  Dbg_MsgAssert(p_source_array,("NULL p_source_array ?"));
335 
336  uint32 *p_dest_array=p_dest->GetArrayPointer();
337  Dbg_MsgAssert(p_dest_array,("NULL p_dest_array ?"));
338 
339  switch (type)
340  {
341  case ESYMBOLTYPE_INTEGER:
342  case ESYMBOLTYPE_FLOAT:
343  case ESYMBOLTYPE_NAME:
344  {
345  uint32 *p_source_word=p_source_array;
346  uint32 *p_dest_word=p_dest_array;
347  for (uint32 i=0; i<size; ++i)
348  {
349  *p_dest_word++=*p_source_word++;
350  }
351  break;
352  }
353 
354  case ESYMBOLTYPE_STRING:
355  case ESYMBOLTYPE_LOCALSTRING:
356  {
357  char **pp_source_string=(char**)p_source_array;
358  char **pp_dest_string=(char**)p_dest_array;
359  for (uint32 i=0; i<size; ++i)
360  {
361  char *p_source_string=*pp_source_string;
362  if (p_source_string)
363  {
364  // Create a new copy of the source string.
365  *pp_dest_string=CreateString(p_source_string);
366  }
367  ++pp_source_string;
368  ++pp_dest_string;
369  }
370  break;
371  }
372 
373  case ESYMBOLTYPE_PAIR:
374  {
375  CPair **pp_source_pair=(CPair**)p_source_array;
376  CPair **pp_dest_pair=(CPair**)p_dest_array;
377  for (uint32 i=0; i<size; ++i)
378  {
379  CPair *p_source_pair=*pp_source_pair;
380  if (p_source_pair)
381  {
382  CPair *p_new_pair=new CPair;
383  // The default assignement operator works OK for CPair since it contains no pointers.
384  *p_new_pair=*p_source_pair;
385  *pp_dest_pair=p_new_pair;
386  }
387  ++pp_source_pair;
388  ++pp_dest_pair;
389  }
390  break;
391  }
392 
393  case ESYMBOLTYPE_VECTOR:
394  {
395  CVector **pp_source_vector=(CVector**)p_source_array;
396  CVector **pp_dest_vector=(CVector**)p_dest_array;
397  for (uint32 i=0; i<size; ++i)
398  {
399  CVector *p_source_vector=*pp_source_vector;
400  if (p_source_vector)
401  {
402  CVector *p_new_vector=new CVector;
403  // The default assignement operator works OK for CVector since it contains no pointers.
404  *p_new_vector=*p_source_vector;
405  *pp_dest_vector=p_new_vector;
406  }
407  ++pp_source_vector;
408  ++pp_dest_vector;
409  }
410  break;
411  }
412 
413  case ESYMBOLTYPE_STRUCTURE:
414  {
415  CStruct **pp_source_structure=(CStruct**)p_source_array;
416  CStruct **pp_dest_structure=(CStruct**)p_dest_array;
417  for (uint32 i=0; i<size; ++i)
418  {
419  CStruct *p_source_structure=*pp_source_structure;
420  if (p_source_structure)
421  {
422  CStruct *p_new_structure=new CStruct;
423  // Copy the contents using CStruct's assignement operator.
424  *p_new_structure=*p_source_structure;
425  *pp_dest_structure=p_new_structure;
426  }
427  ++pp_source_structure;
428  ++pp_dest_structure;
429  }
430  break;
431  }
432 
433  case ESYMBOLTYPE_ARRAY:
434  {
435  CArray **pp_source_carray=(CArray**)p_source_array;
436  CArray **pp_dest_carray=(CArray**)p_dest_array;
437  for (uint32 i=0; i<size; ++i)
438  {
439  CArray *p_source_carray=*pp_source_carray;
440  if (p_source_carray)
441  {
442  CArray *p_new_carray=new CArray;
443  // No assignement operator defined for CArray (cos it would cause cyclic dependencies)
444  // so need to use CopyArray.
445  CopyArray(p_new_carray,p_source_carray);
446  *pp_dest_carray=p_new_carray;
447  }
448  ++pp_source_carray;
449  ++pp_dest_carray;
450  }
451  break;
452  }
453 
454  default:
455  Dbg_MsgAssert(0,("Bad source array type of '%s'",GetTypeName(type)));
456  break;
457  }

603  {
604  if (last[i] == this)

883 // Adds the component p_comp to the end of the list of components, removing any existing components
884 // that have the same name and type, and asserting if any components exist with the same name but different
885 // type.
886 // However, if p_comp is a flag (an un-named component of type ESYMBOLTYPE_NAME) then it will always add it,
887 // because otherwise we would not be able to have more than one flag in a structure.
888 //
889 // Note: Even two identical flags are allowed, because sometimes this is handy. For example I use a list
890 // of words in a structure as a way for Scott to represent trick button sequences (see airtricks.q) and
891 // sometimes two of the words might need to be identical.
892 //
893 // Another note: This function needs to be pretty fast because it will need to be called many times for
894 // each line of script executed. Optimize later.
895 //
896 void CStruct::AddComponent(CComponent *p_comp)
897 {
898  Dbg_MsgAssert(p_comp,("NULL p_comp"));
899 
900  // Unnamed structure components are not allowed, because to support them would require
901  // modifying the search_for function. No need, since when structures are created from lists
902  // of tokens any substructures get expanded and added component by component at that stage.
903  // So in theory no components of type structure should get added.
904  Dbg_MsgAssert(!(p_comp->mNameChecksum==0 && p_comp->mType==ESYMBOLTYPE_STRUCTURE),("Tried to add an un-named structure component ..."));
905 
906  Dbg_MsgAssert(p_comp->mType!=ESYMBOLTYPE_NONE,("Tried to add a structure component with no type, name='%s' ...",FindChecksumName(p_comp->mNameChecksum)));
907 
908  CComponent *p_last=NULL;
909  CComponent *p_scan=mp_components;
910  bool remove=false;
911 
912  while (p_scan)
913  {
914  if (p_scan->mNameChecksum==p_comp->mNameChecksum)
915  {
916  // Same name ...
917 
918  remove=false;
919 
920  if (p_scan->mType==p_comp->mType)
921  {
922  if (p_scan->mType==ESYMBOLTYPE_NAME &&
923  p_scan->mNameChecksum==0 &&
924  p_scan->mChecksum!=p_comp->mChecksum)
925  {
926  // Allow multiple un-named checksums if their values are different, because
927  // these are flags.
928  }
929  else
930  {
931  remove=true;
932  }
933  }
934  else
935  {
936  // Consider floats and ints to be the same type, so that x=3.1 overrides x=3 and vice versa
937  if ((p_scan->mType==ESYMBOLTYPE_FLOAT && p_comp->mType==ESYMBOLTYPE_INTEGER) ||
938  (p_scan->mType==ESYMBOLTYPE_INTEGER && p_comp->mType==ESYMBOLTYPE_FLOAT))
939  {
940  remove=true;
941  }
942  }
943 
944  if (remove)
945  {
946  // Remove this component.
947  if (p_last)
948  {
949  p_last->mpNext=p_scan->mpNext;
950  CleanUpComponent(p_scan);
951  delete p_scan;
952  p_scan=p_last->mpNext;
953  }
954  else
955  {
956  mp_components=p_scan->mpNext;
957  CleanUpComponent(p_scan);
958  delete p_scan;
959  p_scan=mp_components;
960  }
961  }
962  else
963  {
964  // Same name but different type.
965  // The old code would assert in this case, but the new code allows it.
966  // This is because there is no problem if a structure contains {x=7 x="Blaa"}
967  // Getting an integer called x from it would give 7, getting a string called x would give "Blaa"
968  // Also, it would allow {x=7 x=foo} where foo may be defined elsewhere to be a different integer,
969  // which is useful.
970  p_last=p_scan;
971  p_scan=p_scan->mpNext;
972  }
973  }
974  else
975  {
976  p_last=p_scan;
977  p_scan=p_scan->mpNext;
978  }
979  }
980 
981  // Now, p_last points to the last component in the list, and may be NULL.
982  // Tag p_comp onto the end of the list.
983  if (p_last)
984  {
985  p_last->mpNext=p_comp;
986  }
987  else
988  {
989  mp_components=p_comp;
990  }
991  p_comp->mpNext=NULL;
992 }

1345 bool CStruct::search_for(uint32 nameChecksum, ESymbolType type, SWhatever *p_value) const
1346 {
1347  Dbg_MsgAssert(s_num_search_for_recursions<10,("Possible infinite recursion of the CStruct::search_for function, bailing out!\nLast search was for parameter named '%s' of type '%s'",FindChecksumName(nameChecksum),GetTypeName(type)));
1348  #ifdef __NOPT_ASSERT__
1349  ++s_num_search_for_recursions;
1350  #endif
1351

1353

1355 
1356  CComponent *p_comp=mp_components;
1357  while (p_comp)
1358  {
1359  if (p_comp->mNameChecksum==nameChecksum)
1360  {
1361  // The name of the component matches that required.
1362  if (p_comp->mType==type)
1363  {
1364  // The type also matches, so the required component has been found.
1365  // Only set a flag though. Need to keep searching in case the structure
1366  // has the form {x=7 x=foo} where foo is a global defined to be an integer,
1367  // and which must override the previous 7.
1368 
1369  p_value->mUnion=p_comp->mUnion;
1370  found=true;
1371  }
1372  else
1373  {
1374  // The type does not match.
1375  if (p_comp->mType==ESYMBOLTYPE_NAME)
1376  {
1377  // The name of the component matches what we're looking for, but the type
1378  // does not and is of type name.
1379  // For example, we might be searching for an integer called x,
1380  // but we've found a component x=foo
1381 
1382  // So, see if foo is the name of a global defined somewhere.
1383  CSymbolTableEntry *p_global=Resolve(p_comp->mChecksum);
1384  if (p_global)
1385  {
1386  // It is! So check if this global has the type that we're looking for ...
1387  if (p_global->mType==type)
1388  {
1389  // It does! So we have a match.
1390  // Just set the found flag though, because we need to keep searching in
1391  // case the structure has the form {x=foo x=7} where the later 7 needs to
1392  // override what we've just found.
1393  p_value->mUnion=p_global->mUnion;
1394  found=true;
1395  }
1396  }
1397  }
1398  }
1399  }
