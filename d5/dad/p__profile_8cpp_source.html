161 void NsProfile::histogram ( float x0, float y0, float x1, float y1, GXColor color )
162 {
163  float ms;
164  float bar;
165  float xpos;
166  unsigned int lp;
167  int he;
168 
169  // Quick reject test.
170  assertf ( m_pHistoryBuffer, ( "No history buffer set up." ) );
171 
172  // Update history buffer.
173  ms = (float)OSTicksToMilliseconds( ((float)m_accumulated) );
174 // if ( ms > ( 1000.0f / 60.0f ) ) ms = ( 1000.0f / 60.0f );
175  m_pHistoryBuffer[m_historyEntry] = ms;
176  m_historyEntry++;
177  m_historyEntry %= m_historyBufferSize;
178 
179  // Draw background.
180  NsPrim::box ( x0-1.0f, y0-1.0f, x1 + 1.0f, y1 + 1.0f, (GXColor){255,255,255,128} );
181  NsPrim::box ( x0, y0, x1, y1, (GXColor){0,0,0,128} );
182 
183  // Draw bars.
184  he = m_historyEntry;
185  for ( lp = 0; lp < m_historyBufferSize; lp++ ) {
186  ms = m_pHistoryBuffer[he];
187  bar = ms * ( y1 - y0 ) / ( 1000.0f / 60.0f );
188  last_bar[lp] = bar;
189  xpos = x0 + ( ( (float)lp * ( x1 - x0 ) ) / (float)m_historyBufferSize );
190  NsPrim::line ( xpos, y1, xpos, y1 - bar, color );
191  he++;
192  he %= m_historyBufferSize;
193  }
194 
195  // Reset the stopwatch.
196  m_accumulated = 0;
197 
198  // Save for appends.
199  last_x0 = x0;
200  last_x1 = x1;
201  last_y0 = y0;
202  last_y1 = y1;
203 }

205 void NsProfile::append ( GXColor color, bool update )
206 {
207  float ms;
208  float bar;
209  float xpos;
210  unsigned int lp;
211  int he;
212  float x0 = last_x0;
213  float x1 = last_x1;
214  float y0 = last_y0;
215  float y1 = last_y1;
216 
217  // Quick reject test.
218  assertf ( m_pHistoryBuffer, ( "No history buffer set up." ) );
219 
220  // Update history buffer.
221  ms = (float)OSTicksToMilliseconds( ((float)m_accumulated) );
222 // if ( ms > ( 1000.0f / 60.0f ) ) ms = ( 1000.0f / 60.0f );
223  m_pHistoryBuffer[m_historyEntry] = ms;
224  m_historyEntry++;
225  m_historyEntry %= m_historyBufferSize;
226 
227  // Draw bars.
228  he = m_historyEntry;
229  for ( lp = 0; lp < m_historyBufferSize; lp++ ) {
230  ms = m_pHistoryBuffer[he];
231  bar = ms * ( y1 - y0 ) / ( 1000.0f / 60.0f );
232  xpos = x0 + ( ( (float)lp * ( x1 - x0 ) ) / (float)m_historyBufferSize );
233  NsPrim::line ( xpos, y1 - ( bar + last_bar[lp] ), xpos, y1 - last_bar[lp], color );
234  he++;
235  he %= m_historyBufferSize;
236  if ( update ) last_bar[lp] += bar;
237  }
238 
239  // Reset the stopwatch.
240  m_accumulated = 0;
241 }
