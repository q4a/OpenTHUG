17 int sort_by_material_draw_order( const void *p1, const void *p2 )
18 {
19  sMesh *p_mesh1 = *((sMesh**)p1 );
20  sMesh *p_mesh2 = *((sMesh**)p2 );
21  sMaterial *p_material1 = p_mesh1->mp_material;
22  sMaterial *p_material2 = p_mesh2->mp_material;
23 
24  Dbg_Assert( p_material1 != NULL );
25  Dbg_Assert( p_material2 != NULL );
26 
27  if( p_material1->m_draw_order == p_material2->m_draw_order )
28  {
29  // Have to do some special case processing for the situation where two or more meshes have the same draw order, but only some are
30  // marked as being dynamically sorted. In such a situation the non dynamically sorted ones should come first.
31  if( p_material1->m_sorted == p_material2->m_sorted )
32  {
33  if( p_material1 == p_material2 )
34  {
35  // Same material, no further sorting required.
36  return 0;
37  }
38  else
39  {
40  // If the pixel shaders are the same, sort by material address, otherwise sort by pixel shader value.
41  if( p_mesh1->m_pixel_shader == p_mesh2->m_pixel_shader )
42  {
43  return ((uint32)p_material1 > (uint32)p_material2 ) ? 1 : -1;
44  }
45  else
46  {
47  return ((uint32)p_mesh1->m_pixel_shader > (uint32)p_mesh2->m_pixel_shader ) ? 1 : -1;
48  }
49  }
50  }
51  else if( p_material1->m_sorted )
52  {
53  return 1;
54  }
55  return -1;
56  }
57  return ( p_material1->m_draw_order > p_material2->m_draw_order ) ? 1 : -1;
58 }

347 void sScene::HidePolys( uint32 mask, sCASData *p_cas_data, uint32 num_entries )
348 {
349  if(( num_entries == 0 ) || ( mask == 0 ))
350  {
351  return;
352  }
353 
354 // Tmr::Time start = Tmr::GetTimeInUSeconds();
355 
356  int indices_added[64];
357  ZeroMemory( indices_added, sizeof( int ) * 64 );
358 
359  // Check against every CAS entry.
360  sCASData *p_cas_entry = p_cas_data;
361  for( uint32 entry = 0; entry < num_entries; ++entry, ++p_cas_entry )
362  {
363  // Check this CAS entry has the correct mask...
364  if( p_cas_entry->mask & mask )
365  {
366  // Get the mesh this entry references.
367  uint32 load_order = p_cas_entry->data0 >> 16;
368  sMesh *p_mesh = GetMeshByLoadOrder( load_order );
369 
370  Dbg_Assert( load_order < 32 );
371 
372  if( p_mesh == NULL )
373  {
374  continue;
375  }
376 
377  // Get the indices of the poly referenced.
378 // uint32 i0 = ( p_cas_entry->data0 & 0xFFFF ) - p_mesh->m_index_base;
379  uint32 i0 = p_cas_entry->data0 & 0xFFFF;
380 // uint32 i1 = ( p_cas_entry->data1 >> 16 ) - p_mesh->m_index_base;
381  uint32 i1 = p_cas_entry->data1 >> 16;
382 // uint32 i2 = ( p_cas_entry->data1 & 0xFFFF ) - p_mesh->m_index_base;
383  uint32 i2 = p_cas_entry->data1 & 0xFFFF;
384 
385  // For every vertex index in this mesh...
386  uint16 *p_indices = p_mesh->mp_index_buffer[0];
387  uint32 index0 = p_indices[p_cas_entry->start_index] & 0x7FFF;
388  uint32 index1 = p_indices[p_cas_entry->start_index + 1] & 0x7FFF;
389  for( uint32 i = p_cas_entry->start_index + 2; i < p_mesh->m_num_indices[0]; ++i )
390  {
391  uint32 index2 = p_indices[i] & 0x7FFF;
392  if(( index0 == i0 ) && ( index1 == i1 ) && ( index2 == i2 ))
393  {
394  // Indicate this mesh will need rebuilding.
395  indices_added[load_order] = indices_added[load_order] + 1;
396 
397  // Flag the second index of the poly.
398  p_indices[i - 1] |= 0x8000;
399  break;
400  }
401  index0 = index1;
402  index1 = index2;
403  }
404  }
405  }
406 
407  // Now rebuild those meshes that need it.
408  for( int m = 0; m < m_num_mesh_entries; ++m )
409  {
410  sMesh *p_mesh = m_meshes[m];
411  if( indices_added[p_mesh->m_load_order] > 0 )
412  {
413  uint16 *p_indices = p_mesh->mp_index_buffer[0];
414  uint32 new_indices_index = 0;
415  for( uint32 i = 0; i < p_mesh->m_num_indices[0]; ++i )
416  {
417  uint16 index = p_indices[i];
418 
419  Dbg_Assert( new_indices_index < INDEX_WORKBUFFER_SIZE );
420  index_workbuffer[new_indices_index++] = index & 0x7FFF;
421 
422  if( index & 0x8000 )
423  {
424  Dbg_Assert( new_indices_index < INDEX_WORKBUFFER_SIZE );
425  index_workbuffer[new_indices_index++] = index & 0x7FFF;
426  Dbg_Assert( new_indices_index < INDEX_WORKBUFFER_SIZE );
427  index_workbuffer[new_indices_index++] = index & 0x7FFF;
428  }
429  }
430 
431  // Create new index buffer, on the same heap as the existing index buffer.
432  Mem::Allocator::BlockHeader* p_bheader = Mem::Allocator::BlockHeader::sRead( p_mesh->mp_index_buffer[0] );
433  Mem::Allocator* p_allocater = p_bheader->mpAlloc;
434  Mem::Manager::sHandle().PushContext( p_allocater );
435 
436  delete [] p_mesh->mp_index_buffer[0];
437  p_mesh->m_num_indices[0] = (uint16)new_indices_index;
438  p_mesh->mp_index_buffer[0] = new uint16[p_mesh->m_num_indices[0]];
439 
440  Mem::Manager::sHandle().PopContext();
441 
442  // And copy in the new indices from the workbuffer.
443  CopyMemory( p_mesh->mp_index_buffer[0], index_workbuffer, sizeof( uint16 ) * p_mesh->m_num_indices[0] );
444  }
445  }
