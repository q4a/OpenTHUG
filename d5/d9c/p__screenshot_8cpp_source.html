68 static void CopyoutEFB( u8* image_buffer, u32 width, u32 height )
69 {
70  u16 i, j;
71  u32 color;
72 
73  for( j=0; j<height; j++ ) {
74  for( i=0; i<width; i++ ) {
75  GX::PeekARGB( i, j, &color );
76  image_buffer[2] = (u8)(color & 0x000000FF);
77  image_buffer[1] = (u8)((color & 0x0000FF00) >> 8);
78  image_buffer[0] = (u8)((color & 0x00FF0000) >> 16);
79  image_buffer += 3;
80  }
81  }
82 }

86 static void CopyoutPortionEFB( u8* image_buffer, u32 width, u32 startByte, u32 totalBytes )
87 {
88  u32 color;
89 
90  u16 x = (u16)((u32)(startByte/3) % width);
91  u16 y = (u16)((u32)(startByte/3) / width);
92  u32 component = startByte%3;
93 
94  u32 count = 0;
95  while( count < totalBytes )
96  {
97  GX::PeekARGB( x, y, &color );
98 
99  if( component == 0 ) {
100  count++;
101  image_buffer[0] = (u8)((color & 0x00FF0000) >> 16);
102  }
103  if( component <= 1 && count < totalBytes ) {
104  count++;
105  image_buffer[1 - component] = (u8)((color & 0x0000FF00) >> 8);
106  }
107  if( component <= 2 && count < totalBytes ) {
108  count++;
109  image_buffer[2 - component] = (u8)(color & 0x000000FF);
110  }
111  image_buffer += 3 - component;
112  component = 0;
113 
114  x++;
115  if( x >= width ) {
116  x = 0;
117  y++;
118  }
119  }

171 static GRAB_STATUS GrabChunk( u32 chunk,
172  void* bufferXFB,
173  SCREENSHOTAllocator allocator,
174  SCREENSHOTDeallocator deallocator )
175 {
176  if( chunk >= XFB_BASE && chunk < XFB_BASE + XFB_RANGE )
177  {
178  void* buffer = 0;
179 
180  if( chunk == XFB_BASE )
181  { //save off XFB screenshot
182  if( !g_minimize_buffer ) {
183  TakeScreenshotXFB( bufferXFB, allocator );
184  }
185  }
186 
187  //transfer chunk
188  if( g_minimize_buffer ) {
189  u8* fb = (u8*)bufferXFB;
190  buffer = (void*)( &fb[100*1024*(chunk - XFB_BASE)] );
191  }
192  else {
193  buffer = (void*)( &g_data[100*1024*(chunk - XFB_BASE)] );
194  }
195 
196  while( !HIOWrite( 0x00000500, buffer, 100*1024 ) ) {
197  //Spin until communication is successful
198  }
199  OSReport( "SCREENSHOT: Wrote chunk #%d.\n", chunk );
200 
201  //communicate that the chunk is ready to be read
202  while( !HIOWriteMailbox( chunk ) ) {
203  //Spin until communication is successful
204  }
205  OSReport( "SCREENSHOT: Notify of write chunk #%d put in mailbox.\n", g_received_signal );
206  }
207  else if( chunk >= EFB_BASE && chunk < EFB_BASE + EFB_RANGE )
208  {
209  void* buffer;
210 
211  if( chunk == EFB_BASE )
212  { //save off EFB screenshot
213  if( !g_minimize_buffer ) {
214  TakeScreenshotEFB( allocator );
215  }
216  }
217 
218  if( g_minimize_buffer )
219  {
220  WriteScreenshotPortionEFBtoUSB( chunk - EFB_BASE, allocator );
221  }
222  else
223  { //transfer chunk
224  buffer = (void*)( &g_data[100*1024*(chunk - EFB_BASE)] );
225  while( !HIOWrite( 0x00000500, buffer, 100*1024 ) ) {
226  //Spin until communication is successful
227  }
228  }
229  OSReport( "SCREENSHOT: Wrote chunk #%d.\n", chunk );
230 
231  //communicate that the chunk is ready to be read
232  while( !HIOWriteMailbox( chunk ) ) {
233  //Spin until communication is successful
234  }
235  OSReport( "SCREENSHOT: Notify of write chunk #%d put in mailbox.\n", g_received_signal );
236 
237  }
238 
239  if( chunk == XFB_BASE + XFB_RANGE - 1 ||
240  chunk == EFB_BASE + EFB_RANGE - 1 )
241  { //free screenshot buffer space
242  if( g_data ) {
243  deallocator( g_data );
244  g_data = 0;
245  }
246  return( GRAB_NOT_TRANSFERRING );
247  }
248  else {
249  return( GRAB_TRANSFERRING );
250  }

256 static void CheckMail( void* bufferXFB,
257  SCREENSHOTAllocator allocator,
258  SCREENSHOTDeallocator deallocator )
259 {
260  u32 temp = 0;
261  u32 escapeCount = 0;
262  GRAB_STATUS status = GRAB_TRANSFERRING;
263  while( status == GRAB_TRANSFERRING )
264  {
265  escapeCount++;
266  if( g_mail_waiting )
267  {
268  if( HIOReadMailbox( &temp ) ) {
269  g_mail_waiting = FALSE;
270  status = GRAB_NOT_TRANSFERRING;
271  if( temp >= PING_BASE && temp < PING_BASE + PING_RANGE )
272  { //ping message received
273  while( !HIOWriteMailbox( temp ) ) {
274  //Spin until communication is successful
275  }
276  OSReport( "SCREENSHOT: Sent ping back.\n" );
277  }
278  else if( (temp >= XFB_BASE && temp < XFB_BASE + XFB_RANGE) ||
279  (temp >= EFB_BASE && temp < EFB_BASE + EFB_RANGE) )
280  { //request for screen capture
281  escapeCount = 0;
282  g_received_signal = temp;
283  OSReport( "SCREENSHOT: Grab screenshot request.\n" );
284  status = GrabChunk( g_received_signal, bufferXFB, allocator, deallocator );
285  }
286  else if( temp == MINIMIZE_BUFFER_TRUE )
287  { //request to use minimal memory (100K), however game pauses
288  g_minimize_buffer = TRUE;
289  }
290  else if( temp == MINIMIZE_BUFFER_FALSE )
291  { //request to not use minimal memory and keep game running
292  g_minimize_buffer = FALSE;
293  }
294  }
295  }
296  if( !g_minimize_buffer )
297  { //Escape loop since were not pausing the game to transfer
298  return;
299  }
300  if( g_minimize_buffer && escapeCount > 150000000 )
301  { //The PC screenshot app has died - escape
302  if( g_data ) {
303  deallocator( g_data );
304  g_data = 0;
305  }
306  return;
307  }
308  }

314 static BOOL ConnectToUSB( void )
315 {
316  if( !g_connected )
317  {
318  if( HIOEnumDevices( HostIOEnumCallback ) ) {
319  if( g_usb_channel >= 0 ) {
320  if( HIOInit( g_usb_channel, HostIOCallback ) ) {
321  u32 temp = 0;
322  HIOReadMailbox( &temp );
323  g_connected = TRUE;
324  OSReport( "SCREENSHOT: USB connected\n" );
325  }
326  }
327  }
328  }
329 
330  return( g_connected );
