88  if( m_enabled )
89  {
90  // Use the position and orientation of the parent object to position and orient the attached camera.
91  Mth::Vector pos = GetObject()->GetPos();
92  Mth::Matrix mat = GetObject()->GetMatrix();
93 
94  // Set the Display pos of the object to the actual pos, so we can attach a model
95  // to the camera
96  GetObject()->SetDisplayMatrix(mat);
97 
98  mp_camera->SetPos( pos );
99  mp_camera->SetMatrix( mat );
100  }

109 CBaseComponent::EMemberFunctionResult CCameraComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
110 {
111  switch ( Checksum )
112  {
113 /*
114  // @script | DoSomething | does some functionality
115  case 0xbb4ad101: // DoSomething
116  DoSomething();
117  break;
118 
119  // @script | ValueIsTrue | returns a boolean value
120  case 0x769260f7: // ValueIsTrue
121  {
122  return ValueIsTrue() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
123  }
124  break;
125 */
126 
127  default:
128  return CBaseComponent::MF_NOT_EXECUTED;
129  }
130 
131  // the "default" case of the switch statement handles
132  // unrecognized functions; if we make it down here,
133  // that means that the component already handled it
134  // somehow
135  return CBaseComponent::MF_TRUE;
136 }
