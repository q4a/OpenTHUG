314 void CRecordTable::UpdateInitials(CInitials* pInitials)
315 {
316  for (int line= 0; line < m_numRecords; line++)
317  {
318  mp_records[line].UpdateInitials(pInitials);
319  }
320 }
321 
322 // given a new "value", then insert it into the table in the position
323 // that matches this value, and move other entries down (losing the final entry)
324 // returns the entery number (fisr tline is 0)
325 // or -1 if it does not make the cut.
326 int CRecordTable::MaybeNewEntry(int value, int number)
327 {
328 
329 
330  int line;
331 
332  for (line= 0; line < m_numRecords; line++)
333  {
334  mp_records[line].SetNewRecord(false);
335  }
336 
337  for (line= 0; line < m_numRecords; line++)
338  {
339  if (value > mp_records[line].GetValue())
340  {
341  // alright! this value belongs at "line"
342  // so maove down any that are below this line
343  // and insert this one
344  for (int dest = m_numRecords - 1; dest > line; dest--)
345  {
346  mp_records[dest].Set(&mp_records[dest-1]);
347  }
348  mp_records[line].SetValue(value);
349  mp_records[line].SetNumber(number);
350  mp_records[line].SetInitials("XXX");
351  mp_records[line].SetNewRecord(true);
352  return line;
353  }
354  }
355 
356  return -1;
357 }
