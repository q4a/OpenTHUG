116 CBaseComponent::EMemberFunctionResult CSkaterFlipAndRotateComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
117 {
118  switch ( Checksum )
119  {
120  // @script | Flip | flip animation
121  case CRCC(0x65011baa, "Flip"):
122  ToggleFlipState();
123  break;
124 
125  // @script | ResetSwitched | flip if we're switched
126  case CRCC(0x72289a28, "ResetSwitched"):
127  Dbg_Assert(mp_core_physics_component);
128  if (mp_core_physics_component->IsSwitched())
129  {
130  ToggleFlipState();
131  }
132  break;
133 
134  // These three just set flags, which will cause the appropriate flip/rotate
135  // when the next PlayAnim occurs.
136  // @script | FlipAfter | sets flag, which will cause the appropriate flip/rotate when the next PlayAnim occurs
137  case CRCC(0xeba11b20, "FlipAfter"):
138  mFlipAfter = true;
139  break;
140 
141  // @script | RotateAfter | sets flag, which will cause the appropriate flip/rotate when the next PlayAnim occurs
142  case CRCC(0xacac1bad, "RotateAfter"):
143  mRotateAfter = true;
144  break;
145 
146  // @script | BoardRotateAfter | sets flag, which will cause the appropriate flip/rotate when the next PlayAnim occurs
147  case CRCC(0x63e93f69, "BoardRotateAfter"):
148  mBoardRotateAfter = true;
149  break;
150 
151  // @script | IsFlipAfterSet | returns mFlipAfter
152  case CRCC(0xa6243daa, "IsFlipAfterSet"):
153  return mFlipAfter ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
154 
155  // @script | FlipAndRotate | flip Z and X, to rotate 180 degrees about y
156  case CRCC(0xd9de0c65, "FlipAndRotate"):
157  {
158  Dbg_Assert(mp_core_physics_component);
159  mp_core_physics_component->ReverseFacing();
160  ToggleFlipState();
161  break;
162  }
163 
164  // @script | BoardRotate |
165  // @flag normal | put the board back to normal (otherwise just flip)
166  case CRCC(0xe0f3a644, "BoardRotate"):
167  {
168  GameNet::PlayerInfo* player = GameNet::Manager::Instance()->GetPlayerByObjectID(GetObject()->GetID());
169  Dbg_Assert(player);
170 
171  Net::Client* client = GameNet::Manager::Instance()->GetClient(player->GetSkaterNumber());
172  Dbg_Assert(client);
173 
174  if (pParams->ContainsFlag(CRCD(0xde7a971b, "Normal")))
175  {
176  // Put the board back to normal
177  RotateSkateboard(GetObject()->GetID(), false, client->m_Timestamp, true);
178  }
179  else
180  {
181  // Otherwise flip it, flip it good.
182  RotateSkateboard(GetObject()->GetID(), !m_rotate_board, client->m_Timestamp, true);
183  }
184  break;
185  }
186 
187  // @script | BoardIsRotated |
188  case CRCC(0x79ee5ccf, "BoardIsRotated"):
189  return m_rotate_board ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
190 
191  case CRCC(0xb1e7291, "PlayAnim"):
192  // do any required flips and rotates, and then cascade the member function call
193  DoAnyFlipRotateOrBoardRotateAfters();
194  return CBaseComponent::MF_NOT_EXECUTED;
195 
196  default:
197  return CBaseComponent::MF_NOT_EXECUTED;
198  }
199  return CBaseComponent::MF_TRUE;
200 }

274 /*
275 
276 If any of mFlipAfter, mRotateAfter or mBoardRotateAfter are set
277 this will flip or rotate the skater, or rotate the board, and then
278 reset those flags.
279 
280 Quite often the skater needs to be flipped and/or rotated in order to
281 look correctly oriented, due to the way the animations are done.
282 For example, the nollie heelflip animation is actually animated backwards,
283 so to look correct the skater has to be flipped and rotated just for the
284 duration of that animation.
285 As soon as any sort of 'event' happens that could result in an animation
286 change, or requires that the orientation be correct for the physics, then
287 the skater needs to be flipped and rotated back.
288 So in the script, when running the nollie heelflip anim, Scott will flip and
289 rotate the skater first so that he looks correct, then set FlipAfter and RotateAfter.
290 
291 It used to be that I'd check the flags only in the next call to the PlayAnim script
292 member function.
293 However, this is often not early enough. For example, when the player lands whilst
294 doing a nollie heelflip, the land script will do some checks to see if the player
295 is facing backwards before doing a playanim, so it will wrongly think the player
296 is backwards, when he isn't 'really'.
297 
298 So instead I sprinkled calls to the following function whenever anything happens
299 that could cause a new script to be executed. In particular, before every SetScript,
300 and also as soon as a grind gets detected, so that the grind maths won't think the
301 skater is backwards when landing from a nollie heelflip.
302 
303 */
304

307 void CSkaterFlipAndRotateComponent::DoAnyFlipRotateOrBoardRotateAfters ( )
308 {
309  if (mFlipAfter)
310  {
311  ToggleFlipState();
312  mFlipAfter = false;
313  }
314 
315  if (mRotateAfter)
316  {
317  Dbg_Assert(mp_core_physics_component);
318  mp_core_physics_component->ReverseFacing();
319  mRotateAfter = false;
320  }
321 
322  if (mBoardRotateAfter)
323  {
324  GameNet::PlayerInfo* player = GameNet::Manager::Instance()->GetPlayerByObjectID(GetObject()->GetID());
325  Dbg_Assert(player);
326 
327  Net::Client* client = GameNet::Manager::Instance()->GetClient(player->GetSkaterNumber());
328  Dbg_Assert(client);
329 
330  // Rotate the board.
331  RotateSkateboard(GetObject()->GetID(), !m_rotate_board, client->m_Timestamp, true);
332  mBoardRotateAfter = false;
333  }
334 
335  // This makes the skater display facing the correct way again.
336  // The mFlipDisplayMatrix is only used when falling back in from a lip trick.
337  // Since it would be bad if this flag got left on, make sure it is off whenever
338  // a new anim runs. This function is a good place to put it, since it gets
339  // run whenever the anim changes.
340  mp_model_component->mFlipDisplayMatrix = false;
341 }
