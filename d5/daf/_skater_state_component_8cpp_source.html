98 CBaseComponent::EMemberFunctionResult CSkaterStateComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
99 {
100  switch ( Checksum )
101  {
102  // @script | DoingTrick | true if we're doing a trick
103  case 0x58ad903f: // DoingTrick
104  return DoingTrick() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
105  break;
106 
107  case 0xb07ac662: // HasPowerup
108  {
109  uint32 type;
110 
111  pParams->GetChecksum( NONAME, &type, true );
112  return HasPowerup( type ) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
113  break;
114  }
115 
116  case 0xe11b7ca: // PickedUpPowerup
117  {
118  uint32 type;
119 
120  pParams->GetChecksum( NONAME, &type, true );
121  PickedUpPowerup( type );
122  break;
123  }
124 
125  // @script | GetTerrain | returns the number of the terrain in 'terrain'
126  case CRCC(0x44ba5fce, "GetTerrain"):
127  pScript->GetParams()->AddInteger(CRCD(0x3789ac4e, "terrain"), m_terrain);
128  break;
129 
130  default:
131  return CBaseComponent::MF_NOT_EXECUTED;
132  }
133  return CBaseComponent::MF_TRUE;
134 }
