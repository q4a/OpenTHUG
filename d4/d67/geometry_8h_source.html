291 // for intersection test, we do X and Z first,
292 // as we war aer more likely to be intersecting in the Y
293 // as everyhting is about the same height
294 // so this eliminates thigns quicker
295 inline bool CBBox::Intersect(const CBBox &bbox) const
296 {
297  if ((m_min[X] > bbox.m_max[X]) || (bbox.m_min[X] > m_max[X]) ||
298  (m_min[Z] > bbox.m_max[Z]) || (bbox.m_min[Z] > m_max[Z]) ||
299  (m_min[Y] > bbox.m_max[Y]) || (bbox.m_min[Y] > m_max[Y]))
300  {
301  return false;
302  }
303 
304  return true;
305 }

325 // Not So Quick & Dirty Intersection w/ Triangle
326 inline bool CBBox::CouldIntersect(const Vector &v0, const Vector &v1, const Vector &v2) const
327 {
328  if ( ( v0[X] > m_max[X] ) &&
329  ( v1[X] > m_max[X] ) &&
330  ( v2[X] > m_max[X] ) ) return false;
331  if ( ( v0[X] < m_min[X] ) &&
332  ( v1[X] < m_min[X] ) &&
333  ( v2[X] < m_min[X] ) ) return false;
334  if ( ( v0[Y] > m_max[Y] ) &&
335  ( v1[Y] > m_max[Y] ) &&
336  ( v2[Y] > m_max[Y] ) ) return false;
337  if ( ( v0[Y] < m_min[Y] ) &&
338  ( v1[Y] < m_min[Y] ) &&
339  ( v2[Y] < m_min[Y] ) ) return false;
340  if ( ( v0[Z] > m_max[Z] ) &&
341  ( v1[Z] > m_max[Z] ) &&
342  ( v2[Z] > m_max[Z] ) ) return false;
343  if ( ( v0[Z] < m_min[Z] ) &&
344  ( v1[Z] < m_min[Z] ) &&
345  ( v2[Z] < m_min[Z] ) ) return false;
346 
347  return true;
