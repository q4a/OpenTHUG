70 // Returns location of Needle in the Haystack
71 const char * StrStr (const char *pHay, const char *pNeedle)
72 {
73  register char First = *pNeedle++; // get first letter in needle
74  if (!First) return 0; // if end of needle, then return NULL
75  First = Caseless(First);
76 
77  while (1)
78  {
79  while (1) // scan Haystack one letter at a time
80  {
81  char FirstHay = *pHay++; // get first letter of haystack
82  FirstHay = Caseless(FirstHay);
83  if (FirstHay == First) break; // first letter matches, so drop into other code
84  if (!FirstHay) return 0; // if end of haystack, then return NULL
85  }
86 
87  const char *pH = pHay; // second letter of hay
88  const char *pN = pNeedle; // second letter of needle
89  while (1)
90  {
91  char n = *pN++; // get a needle letter
92  if (!n) return pHay-1; // end of needle, so return pointer to start of string in haystack
93  char h = *pH++; // get a hay letter
94  if (Caseless(n) != Caseless(h)) break; // difference, so stop looping (could be end of haystack, which is valid)
95  }
96  }
97 }

181 uint DehexifyDigit(const char *pLetter)
182 {
183  uint digit = 0;
184  if (*pLetter >= '0' && *pLetter <= '9')
185  digit = *pLetter - '0';
186  else if (*pLetter >= 'a' && *pLetter <= 'v')
187  digit = *pLetter - 'a' + 10;
188  else if (*pLetter >= 'A' && *pLetter <= 'V')
189  digit = *pLetter - 'A' + 10;
190  else
191  {
192  Dbg_MsgAssert(0, ("Non Hex digit"));
193  }

199 // 0 -f a==b

204 int Compare(const char *p_a, const char *p_b)
205 {
206  for (;;)
207  {
208  char a = *p_a++;
209  char b = *p_b++;
210  a = Caseless(a);
211  b = Caseless(b);
212  if (a < b)
213  {
214  return -1;
215  }
216  if (a > b)
217  {
218  return 1;
219  }
220  if (!a && !b)
221  {
222  // strings are identical
223  return 0;
224  }
225  if (!a)
226  {
227  // a is shorter, so a < b
228  return -1;
229  }
230  if (!b)
231  {
232  // b is shorter, so b < a
233  return 1;
234  }
235  }
236 }
