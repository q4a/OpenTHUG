128  // in the collision resolution code
129  // NOTE: probably want to add some small random rotation when the skater hits the object; two reasons; one, when you hit a line of cones, right
130  // now it seems too orderly; two, when you push a cone ahead of you, the cone's dynamics should be more jerky and less smooth

226

228  if (m_num_contacts == 0)

440  {
441  m_largest_contact_extent = contact_extent_sqr;
442  }
443  }
444  m_largest_contact_extent = sqrtf(m_largest_contact_extent) + 1.0f;
445 
446  m_ignore_skater_countdown = 0.0f;
447 
448  m_pos -= m_matrix.Rotate(m_model_offset);

468 void CRigidBodyComponent::Update()
469 {
470  #ifdef __NOPT_DEBUG__
471  Dbg_MsgAssert(!s_contact_states_in_use, ("Two CRigidBodyComponents updating simultaneously"));
472  s_contact_states_in_use = true;
473  #endif
474 
475  float full_time_step = Tmr::FrameLength();
476 
477  setup_contact_states();
478 
479  // only check for skater collisions if we haven't collided with the skater recently; this insures that the stupid skater doesn't cram us
480  // into corners and the like
481  if (m_ignore_skater_countdown > 0.0f)
482  {
483  m_ignore_skater_countdown -= full_time_step;
484  }
485  else if (!m_flags.Test(PLAYER_COLLISION_DISABLED))
486  {
487  handle_skater_collisions();
488  }
489 
490  if (m_die_countdown >= 0.0f)
491  {
492  if (m_die_countdown <= full_time_step)
493  {
494  GetObject()->MarkAsDead();
495  }
496  else
497  {
498  m_die_countdown -= full_time_step;
499  }
500  }
501 
502  // if we are asleep
503  if (m_state == ASLEEP)
504  {
505  GetObject()->SetDisplayMatrix(m_matrix);
506 
507  if (s_debug_lines_on)
508  {
509  draw_debug_lines();
510  }
511 
512  // do just about nothing
513 
514  #ifdef __NOPT_DEBUG__
515  s_contact_states_in_use = false;
516  #endif
517 
518  return;
519  }
520 
521  int main_update_loop_count = 0;
522  bool last_ditch_collision = false;
523  float remaining_time_step = full_time_step;
524  do {
525  float time_step = remaining_time_step;
526 
527  // send a feeler from object's center of mass to our new center of mass; most collisions will not be caught in this manner;
528  // this is a last-ditch feeler to insure that we don't fly through objects at high speeds
529  CFeeler feeler;
530  feeler.m_start = m_pos;
531  feeler.m_end = m_pos + time_step * m_vel;
532  feeler.m_end[Y] += 0.5f * time_step * time_step * m_const_acc;
533 
534  last_ditch_collision = feeler.GetCollision(false);
535 
536  // if we have such a collision, move only to it
537  if (last_ditch_collision)
538  {
539  time_step = (feeler.GetDist() * feeler.Length() - 6.0f) / m_vel.Length();
540  if (time_step < 0.0f)
541  {
542  time_step = 0.0f;
543  }
544  }
545 
546  update_dynamic_state(time_step);
547 
548  // we do a very simple collision
549  if (last_ditch_collision)
550  {
551  m_vel -= (1.0f + m_coeff_restitution) * Mth::DotProduct(m_vel, feeler.GetNormal()) * feeler.GetNormal();
552  }
553 
554  remaining_time_step -= time_step;
555 
556  // make sure we're not in an infinite loop; a weirdly hanging box is better than a frozen game
557  if (++main_update_loop_count == 5)
558  {
559  MESSAGE("too many last-ditch contacts in a single frame; going to sleep");
560  sleep();
561  break;
562  }
563  } while (last_ditch_collision);
564 
565  if (detect_collisions())
566  {
567  resolve_collisions();
568  }
569 
570  consider_sleeping();
571 
572  // if we've gotten here, we need to update the object
573 
574  GetObject()->SetPos(m_pos + m_matrix.Rotate(m_model_offset));
575 
576  GetObject()->SetMatrix(m_matrix);
577  GetObject()->SetDisplayMatrix(m_matrix);
578 
579  GetObject()->SetVel(m_vel);
580 
581  if (s_debug_lines_on)
582  {
583  draw_debug_lines();
584  }
585 
586  #ifdef __NOPT_DEBUG__
587  s_contact_states_in_use = false;
588  #endif
589 }

596 CBaseComponent::EMemberFunctionResult CRigidBodyComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
597 {
598  switch ( Checksum )
599  {
600  // @script | RigidBody_IgnoreSkater | rigidbody ignores the skater for a given duration

688  }
689  }
690 
691  wake();
692 
693  break;
694  }
695 
696  default:
697  return CBaseComponent::MF_NOT_EXECUTED;
698  }
699  return CBaseComponent::MF_TRUE;

841 void CRigidBodyComponent::handle_skater_collisions ( )
842 {
843  for (int n = Mdl::Skate::Instance()->GetNumSkaters(); n--; )
844  {
845  CSkater& skater = *Mdl::Skate::Instance()->GetSkater(n);
846 
847  float radius_boost = skater.GetRigidBodyCollisionRadiusBoost();
848 
849  // detect collisions within a cylinder of the skater; if objects end up having significant extent in some directions and not others,
850  // this detection code may have to become more sophisticated
851 
852  // check distance to the skater in the X-Z plane
853  float distance_sqr = (m_pos[X] - skater.GetPos()[X]) * (m_pos[X] - skater.GetPos()[X])
854  + (m_pos[Z] - skater.GetPos()[Z]) * (m_pos[Z] - skater.GetPos()[Z]);
855  if (distance_sqr > Mth::Sqr(m_skater_collision_radius + radius_boost)) continue;
856 
857  // check Y-offset to the skater
858  float y_offset = skater.GetPos()[Y] - m_pos[Y];
859  if (y_offset > m_skater_collision_application_radius + radius_boost
860  || y_offset < -(m_skater_collision_application_radius + radius_boost + s_skater_head_height)) continue;
861 
862  // apply the skater collision impulse; the impulse occurs at a slight assent and in a direction halfway between of the skater's
863  // velocity and the line to the object; the magnitude is roughly proportional to the component of the skater's velocity along the
864  // line to the object; the impulse will be applied a few inches above the skater's lowest point, along the line between the object
865  // and the skater at distance equal to the collision radius of the object
866 
867  // unit vector pointing from the skater to the object in the X-Z plane
868  Mth::Vector r = m_pos;
869  r -= skater.GetPos();
870  // r[Y] = 0.0f; // optimized out as r[Y] never used
871  float eff_length = sqrtf(r[X] * r[X] + r[Z] * r[Z]);
872  r[X] /= eff_length;
873  r[Z] /= eff_length;
874 
875  // skater's relative velocity in the X-Z plane
876  Mth::Vector skater_vel = skater.GetVel();
877  skater_vel -= m_vel;
878  skater_vel[Y] = 0.0f;
879 
880  if(( fabsf( skater_vel[X] ) < 0.01f ) && ( fabsf( skater_vel[Z] ) < 0.01f ))
881  {
882  // Small enough not to worry about - smaller can cause NaN decomposition.
883  return;
884  }
885 
886  // unadjusted impulse magnitude
887  float magnitude = skater_vel[X] * r[X] + skater_vel[Z] * r[Z];
888  if (magnitude <= 0.0f) continue;
889 
890  // we adjust the impulse magnitude so the object will be vaulted in front of the skater so as to give the player a nice view of the object's
891  // dynamics; also, at low skater velocities, we want only a very small impulse
892 
893  // fast collisions
894  if (magnitude > (2.0f * vRP_SKATER_COLLISION_VELOCITY_THRESHOLD_INCREMENT))
895  {
896  // increase the impulse by a constant
897  magnitude += ((1.0f + vRP_SKATER_COLLISION_VELOCITY_THRESHOLD_INCREMENT) * vRP_SKATER_COLLISION_VELOCITY_FACTOR);
898  }
899  // medium collisions
900  else if (magnitude > vRP_SKATER_COLLISION_VELOCITY_THRESHOLD_INCREMENT) {
901  // connect the functions continuously
902  magnitude += (vRP_SKATER_COLLISION_VELOCITY_FACTOR * vRP_SKATER_COLLISION_VELOCITY_THRESHOLD_INCREMENT)
903  + (magnitude - vRP_SKATER_COLLISION_VELOCITY_THRESHOLD_INCREMENT);
904  }
905  else
906  // slow collisions
907  {
908  // increase the impulse by a factor
909  magnitude *= (1.0f + vRP_SKATER_COLLISION_VELOCITY_FACTOR);
910  }
911 
912  // impulse direction in the X-Z plane is the average of the skater velocity and r directions
913  // we know the Y component for both is zero
914  Mth::Vector impulse = skater_vel;
915  eff_length = sqrtf(impulse[X] * impulse[X] + impulse[Z] * impulse[Z]);
916  impulse[X] = impulse[X] / eff_length + r[X];
917  impulse[Z] = impulse[Z] / eff_length + r[Z];
918  eff_length = sqrtf(impulse[X] * impulse[X] + impulse[Z] * impulse[Z]);
919  impulse[X] = impulse[X] / eff_length;
920  impulse[Z] = impulse[Z] / eff_length;
921 
922  // adjust the impulse direction upwards
923  impulse[X] *= m_cos_skater_collision_assent;
924  impulse[Y] = m_sin_skater_collision_assent;
925  impulse[Z] *= m_cos_skater_collision_assent;
926 
927  // factor in the magnitude
928  impulse *= m_skater_collision_impulse_factor * magnitude;
929 
930  // the impulse's point of application in the XZ-plane is along the line between the skater and the object at the object's impulse application
931  // radius
932 
933  // we calculate the point of application with respect to the center of mass
934  Mth::Vector application_point = skater.GetPos();
935  application_point[X] -= m_pos[X];
936  application_point[Y] = 0.0f;
937  application_point[Z] -= m_pos[Z];
938  application_point.Normalize(m_skater_collision_application_radius);
939 
940  // calculate the Y-component of the point of application
941  if (y_offset > 0.0f)
942  {
943  // collide with the skater's feet
944  application_point[Y] = skater.GetPos()[Y] - m_pos[Y];
945  }
946  else if (y_offset < -s_skater_head_height)
947  {
948  // collide with the skater's head
949  application_point[Y] = skater.GetPos()[Y] + s_skater_head_height - m_pos[Y];
950  }
951  else
952  {
953  // collide with the skater's body; use an offset from the center of mass to cause rotation
954  application_point[Y] = -12.0f;
955  }
956 
957  // apply the impulse; artifically reduce the initial rotation; it looks nice when the objects tumbles more after their first bounce
958  m_vel += impulse;
959  m_rotvel += m_skater_collision_rotation_factor * 0.6f * m_mass_over_moment * Mth::CrossProduct(application_point, impulse);
960 
961  wake();
962 
963  // start a countdown; while this countdown is complete, we will ignore the skater
964  m_ignore_skater_countdown = m_ignore_skater_duration;
965 
966  // call the bounce script callback
967  if (m_script_names.collide)
968  {
969  GetObject()->SpawnScriptPlease(m_script_names.collide, mp_script_params)->Update();
970  }
971 
972  // kill the object's shadow
973  if (Nx::CSector *p_shadow_sector = Nx::CEngine::sGetMainScene()->GetSector(Crc::ExtendCRCWithString(GetObject()->GetID(), "_Shadow")))
974  {
975  p_shadow_sector->SetActive(false);
976  }
977 
978  Tmr::Time time = Tmr::GetTime();
979  if ((int) (time - m_collide_sound_allowed_time) > 0 && (int) (time - s_collide_sound_allowed_time) > 0)
980  {
981  #ifdef __NOPT_ASSERT__
982  if (m_sound_type_id && Script::GetInteger(CRCD(0xd634a297, "DynamicRigidbodySounds")))
983  {
984  get_sound_setup(m_sound_type_id);
985  }
986  #endif
987 
988  if (m_sound_setup.collide_sound)
989  {
990  float percent = (100.0f / 1000.0f) * impulse.Length();
991  m_sound_setup.collide_sound->AddFloat(CRCD(0x9e497fc6, "Percent"), percent);
992  mp_sound_component->PlayScriptedSound(m_sound_setup.collide_sound);
993 
994  // delay the next collision sound
995  m_collide_sound_allowed_time = Tmr::GetTime() + (50 + Mth::Rnd(100)) * m_sound_setup.collide_mute_delay / 100;
996  s_collide_sound_allowed_time = Tmr::GetTime() + (50 + Mth::Rnd(100)) * m_sound_setup.global_collide_mute_delay / 100;
997  }
998  }
999 
1000  // collide with only a single skater
1001  break;
1002  } // END loop over skaters
1003 }

1107 bool CRigidBodyComponent::detect_collisions ( )
1108 {
1109  CFeeler feeler;
1110 
1111  // set up a bounding box around the space within which all collision detection will occur
1112  Mth::CBBox bounding_box(m_pos - Mth::Vector(m_largest_contact_extent, m_largest_contact_extent, m_largest_contact_extent),
1113  m_pos + Mth::Vector(m_largest_contact_extent, m_largest_contact_extent, m_largest_contact_extent));
1114  s_collision_cache.Update(bounding_box);
1115  feeler.SetCache(&s_collision_cache);
1116 
1117  // loop over the contact points
1118  m_num_collisions = 0;
1119  for (int n = m_num_contacts; n--; )
1120  {
1121  SContactState& contact_state = sp_contact_states[n];
1122 
1123  // run a feeler from the object's center to the contact point
1124  feeler.m_start = m_pos;
1125  feeler.m_end = m_pos;
1126  feeler.m_end += contact_state.p_world;
1127 
1128  contact_state.collision = feeler.GetCollision(false);
1129 
1130  if (!contact_state.collision) continue;
1131 
1132  contact_state.normal = feeler.GetNormal();
1133  contact_state.normal_eff_mass_set = false;
1134  contact_state.normal_impulse_magnitude = 0.0f;
1135 
1136  contact_state.depth = Mth::DotProduct(contact_state.normal, (1.0f - feeler.GetDist()) * (feeler.m_end - feeler.m_start));
1137  contact_state.collision = contact_state.depth <= 0.0f;
1138 
1139  if (contact_state.collision)
1140  {
1141  m_num_collisions++;
1142  }
1143  }
1144 
1145  return m_num_collisions != 0;
1146 }

1166 
1167 void CRigidBodyComponent::resolve_collisions ( )
1168 {
1169  // collision resolution code of my own devise; simplified and not wholly accurate, it is good enough because we only have a single object
1170  // colliding with a (mostly) static background; multiple collisions are not resolved simultaneously; instead, an iterative procedure is used to
1171  // resolve the normal forces; this can cause objects which are colliding face-to-face with a surface to bounce off with odd rotation,
1172  // as the required collision impulse will not be spread evenly across the contacts (as would be correct); also, friction is not handled in closed
1173  // form; instead, after the collision is resolved and the normal forces are determined, we go back in and apply frictional forces (using the
