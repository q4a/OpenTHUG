137 // The component's Update() function is called from the CCompositeObject's
138 // Update() function. That is called every game frame by the CCompositeObjectManager
139 // from the s_logic_code function that the CCompositeObjectManager registers
140 // with the task manger.
141 void CSkaterProximityComponent::Update()
142 {
143  // GJ TODO: maybe this should be made more generic
144  // so that it takes a CCompositeObject* (it would
145  // make replacing the skater w/ a different player
146  // object easier)
147 
148  // check skater dist exceptions...
149  float distToSkaterSqr = 0.0f;
150 
151  if ( mInnerRadiusSqr )
152  {
153  distToSkaterSqr = GetDistToLocalSkaterSquared();
154  //distToSkaterSqr = GetDistToNearestSkaterSquared( );
155  if ( distToSkaterSqr <= mInnerRadiusSqr )
156  {
157  // If the local skater is in radius, then flag "SkaterInRadius"
158  FLAGEXCEPTION( CRCD(0xdb7413fb,"SkaterInRadius") );
159  if (!GetObject()->IsDead())
160  {
161  FLAGEXCEPTION( CRCD(0x5e8eb123,"AnySkaterInRadius") );
162  }
163  }
164  else
165  {
166  distToSkaterSqr = GetDistToNearestSkaterSquared(); // MIGHT BE A NETWORK SKATER
167  if ( distToSkaterSqr <= mInnerRadiusSqr )
168  {
169  FLAGEXCEPTION( CRCD(0x5e8eb123,"AnySkaterInRadius") );
170  }
171  distToSkaterSqr = 0.0f; // Mick: need to clear it, otherwise this value will be used for the
172  // AvoidRadius settings, which only apply to local skaters
173  }
174  }
175 
176  if ( mOuterRadiusSqr )
177  {
178  if ( distToSkaterSqr == 0.0f )
179  {
180  distToSkaterSqr = GetDistToLocalSkaterSquared();
181  }
182  if ( distToSkaterSqr >= mOuterRadiusSqr )
183  {
184  if (!GetObject()->IsDead())
185  {
186  FLAGEXCEPTION( CRCD(0xa41f5336,"SkaterOutOfRadius") );
187  }
188  }
189  }
190 
191  if ( mInnerAvoidRadiusSqr )
192  {
193  if ( distToSkaterSqr == 0.0f )
194  {
195  distToSkaterSqr = GetDistToLocalSkaterSquared();
196  }
197  if ( distToSkaterSqr <= mInnerAvoidRadiusSqr )
198  {
199  if (!GetObject()->IsDead())
200  {
201  FLAGEXCEPTION( CRCD(0xfaeec40f,"SkaterInAvoidRadius") );
202  }
203  }
204  }
205 
206  if ( mOuterAvoidRadiusSqr )
207  {
208  if ( distToSkaterSqr == 0.0f )
209  {
210  distToSkaterSqr = GetDistToLocalSkaterSquared();
211  }
212  if ( distToSkaterSqr >= mOuterAvoidRadiusSqr )
213  {
214  if (!GetObject()->IsDead())
215  {
216  FLAGEXCEPTION( CRCD(0x9c5af1a0,"SkaterOutOfAvoidRadius") );
217  }
218  }
219  }
220 }

227 // Given the "Checksum" of a script command, then possibly handle it
228 // if it's a command that this component will handle
229 CBaseComponent::EMemberFunctionResult CSkaterProximityComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
230 {
231  switch ( Checksum )
232  {
233  // @script | Obj_SetInnerRadius |
234  // @uparm 1.0 | inner radius
235  case ( 0x82a0a873 ): // Obj_SetInnerRadius
236  if ( pParams->GetFloat( NONAME, &mInnerRadiusSqr ) )
237  {
238  mInnerRadiusSqr *= FEET_TO_INCHES( 1.0f );
239  mInnerRadiusSqr *= mInnerRadiusSqr;
240  }
241  break;
242 
243  // @script | Obj_SetOuterRadius |
244  // @uparm 1.0 | outer radius
245  case ( 0x8cb5d0c0 ): // Obj_SetOuterRadius
246  if ( pParams->GetFloat( NONAME, &mOuterRadiusSqr ) )
247  {
248  mOuterRadiusSqr *= FEET_TO_INCHES( 1.0f );
249  mOuterRadiusSqr *= mOuterRadiusSqr;
250  }
251  break;
252 
253  // @script | Obj_SetInnerAvoidRadius |
254  // @uparm 1.0 | inner radius
255  case 0x50f28d36: // Obj_SetInnerAvoidRadius
256  if ( pParams->GetFloat( NONAME, &mInnerAvoidRadiusSqr ) )
257  {
258  mInnerAvoidRadiusSqr *= FEET_TO_INCHES( 1.0f );
259  mInnerAvoidRadiusSqr *= mInnerAvoidRadiusSqr;
260  }
261  break;
262 
263  // @script | Obj_SetOuterAvoidRadius |
264  // @uparm 1.0 | outer radius
265  case 0x2d703b94: // Obj_SetOuterAvoidRadius
266  if ( pParams->GetFloat( NONAME, &mOuterAvoidRadiusSqr ) )
267  {
268  mOuterAvoidRadiusSqr *= FEET_TO_INCHES( 1.0f );
269  mOuterAvoidRadiusSqr *= mOuterAvoidRadiusSqr;
270  }
271  break;
272 
273 
274 
275  default:
276  return CBaseComponent::MF_NOT_EXECUTED;
277  }
278 
279  // the "default" case of the switch statement handles
280  // unrecognized functions; if we make it down here,
281  // that means that the component already handled it
282  // somehow
283  return CBaseComponent::MF_TRUE;
284 }

349 float CSkaterProximityComponent::GetDistToNearestSkaterSquared()
350 {
351  float nearest_distance = HUGE_DISTANCE_SQUARED;
352 
353  int num_skaters = Mdl::Skate::Instance()->GetNumSkaters();
354 
355  for( int i = 0; i < num_skaters; i++ )
356  {
357  CCompositeObject* pObj = Mdl::Skate::Instance()->GetSkater( i );
358  if ( pObj && ( pObj != GetObject() ))
359  {
360  float this_dist = Mth::DistanceSqr( pObj->GetPos(), GetObject()->GetPos() );
361  if( this_dist < nearest_distance )
362  {
363  nearest_distance = this_dist;
364  }
365  }
366  }
367 
368  return nearest_distance;
369 }
