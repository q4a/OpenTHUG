131 // Split out the actual updating from Update(), so we co do specific updates
132 // of the require/cancel flags at arbitart times
133 // specifically we want to update the "Lip" requirements as soon as we land on them

136 void CSkaterGapComponent::UpdateCancelRequire(int cancel, int require)
137 {
138  CGap* pGap = static_cast< CGap* >(m_gap_list.FirstItem());
139 
140  // don't bother determining the state flags if there are no active gaps
141  if (!pGap) return;
142 
143  // update the active gaps based on the flags
144  while (pGap)
145  {
146  CGap* pNext = static_cast< CGap* >(pGap->GetNext());
147 
148  // Clear any require flags that match the require mask, so if we do all that is required, then the require part of the flags field will be cleared
149  pGap->m_flags &= ~require;
150 
151  // if any cancel flags match the cancel mask, cancel the gap
152  if (pGap->m_flags & cancel)
153  {
154 // printf ("Cancelled by %x\n",pGap->m_flags & cancel);
155  if (pGap->m_trickscript)
156  {
157  // delete the gap trick
158  CGapTrick *pGapTrick = static_cast< CGapTrick* >(m_gaptrick_list.FirstItem());
159  while (pGapTrick)
160  {
161  CGapTrick * pNext = static_cast< CGapTrick* >(pGapTrick->GetNext());
162 
163  // only delete gaptricks if you've not been awarded them, and they exactly match the gap we're cancelling
164  if (!pGapTrick->m_got
165  && pGapTrick->m_id == pGap->m_id
166  && pGapTrick->m_script == pGap->m_trickscript
167  && pGapTrick->m_node == pGap->m_node)
168  {
169  delete pGapTrick;
170  break;
171  }
172 
173  pGapTrick = pNext;
174  }
175  }
176 
177  #ifdef DEBUG_GAPS
178  MESSAGE("DELETING GAP");
179  DUMPC(pGap->m_id);
180  #endif
181 
182  // delete the gap
183  delete pGap;
184  }
185  pGap = pNext;
186  }
187 }

194 CBaseComponent::EMemberFunctionResult CSkaterGapComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
195 {
196  switch ( Checksum )
197  {
198  // @script | StartGap | Start a gap...used in conjunction with EndGap. <nl>
199  // See online doc for detailed description and examples
200  // @parmopt name | GapID | Links | the name of the gap
201  // @parmopt array | flags | | array of flags to use
202  // @parmopt name | trickscript | | Script that is run when you land a combo that includes this gap
203  // subject to the two optional conditions below
204  // @parmopt checksum | KeyCombo | | Name of a key combo, like Air_SquareD, you have to do trick with this button combo
205  // @parmopt string | TrickText | | This text substring, like 'kickflip' has to be in the combo
206  // @nextparm | flags | flags | file | gapflags.lst | done
207  // @nextparm | trickscript | script
208  // @nextparm | KeyCombo | list | file | keycombo.lst | done
209  // @parmopt checksum | combined_flags | | An optional way of specifying the cancel flags as a uint32 consisting
210  // of the flag bit values or'd together. Currently only used when the Park Editor calls StartGap from within
211  // the c-code.
212  case CRCC(0x65b9ec39, "StartGap"):
213  start_gap(pParams, pScript);
214  break;
215 
216  // @script | StartGapTrick | This adds a gap trick to the skaters
217  // current list
218  // @parm string | TrickText | text to look for in the trick text
219  // @parm name | gapscript | script to run when you land this gap trick
220  // @parmopt checksum | KeyCombo | | Name of a key combo, like Air_SquareD, you have to do trick with this button combo
221  case CRCC(0xfc4f2009, "StartGapTrick"):
222  start_gap_trick(pParams, pScript);
223  break;
224 
225  // @script | EndGap | end a gap - used with StartGap <nl>
226  // See online doc for detailed description and examples
227  // @parm name | GapID | gap gap_id
228  // @parmopt string | text | "Unnamed Gap" | text to display
229  // @parmopt int | score | | score to award for this gap
230  // @flag NetEnabled |
231  // @flag Permanent |
232  // @parmopt structure | continue | | for linking gaps <nl>
233  // e.g. continue = { GapID = SecondHalf Cancel = CANCEL_GROUND }
234  // @nextparm | score | slidernum | 0 | 999999
235  case CRCC(0xe5399fb2, "EndGap"):
236  end_gap(pParams, pScript);
237  break;
238 
239  // @script | CheckGapTricks | Check to see if there are any gap tricks,
240  // then check to see if we did the trick yet
241  // if so, then fire them off
242  // note, this does not delete gaps that you did not do
243  // so you would ahve to call ClearGapTricks
244  // in that case
245  case CRCC(0x42f80e5e, "CheckGapTricks"):
246  check_gap_tricks(pParams);
247  break;
248 
249  // @script | ClearGapTricks | clears all the gap tricks.
250  // Currently, I call CheckGapTricks then ClearGapTricks
251  // at any point in the skater's logic where he can land
252  // a trick combo. So if you just stick a "StartGapTrick"
253  // in your gap script, then it will automatically work
254  case CRCC(0x196772b, "ClearGapTricks"):
255  clear_gap_tricks(pParams);
256  break;
257 
258  // @script | DuplicateTrigger | quick way of doing two way gaps <nl>
259  // see online doc for detailed description and examples
260  case CRCC(0x4562e4a6, "DuplicateTrigger"):
261  {
262  // used in a trigger script
263  // find the first node linked to this one that has
264  // a trigger script, and execute that instead
265  {
266  int node = pScript->mNode;
267  Dbg_MsgAssert(node != -1,("DuplicateTrigger in script with no node,\n%s\n",pScript->GetScriptInfo()));
268  if (node != -1) // Failsafe
269  {
270  int links = SkateScript::GetNumLinks(node);
271  //dodgy_test(); printf("node %d has %d links\n",node,links);
272 
273  Dbg_MsgAssert(links>0,("DuplicateTrigger in node (%d) with no links,\n%s\n",node,pScript->GetScriptInfo()));
274  int i;
275  for (i=0;i<links;i++)
276  {
277  int linked_node = SkateScript::GetLink(node,i);
278  Script::CStruct *pDest = SkateScript::GetNode(linked_node);
279  uint32 script = 0;
280  //dodgy_test(); printf("node %d linked to node %d\n",node,linked_node);
281  if (pDest->GetChecksum(CRCD(0x2ca8a299, "TriggerScript"),&script))
282  {
283  // found another trigger script in the node we are linked to
284  GetObject()->SpawnAndRunScript(script,pScript->mNode);
285  break;
286  }
287  }
288  Dbg_MsgAssert(i<links,("DuplicateTrigger in node (%d) not linked to trigger,\n%s\n",node,pScript->GetScriptInfo()));
289  }
290  }
291  break;
292  }
293 
294  default:
295  return CBaseComponent::MF_NOT_EXECUTED;
296  }
297  return CBaseComponent::MF_TRUE;
298 }

427 void CSkaterGapComponent::get_state_flags ( int& cancel, int& require )
428 {
429  cancel = require = 0;
430 
431  if (!mp_physics_control_component->IsDriving())
432  {
433  if (mp_physics_control_component->IsSkating())
434  {
435  // skating
436 
437  cancel |= CANCEL_SKATE;
438  require |= REQUIRE_SKATE;
439 
440  switch (mp_core_physics_component->GetState())
441  {
442  case AIR:
443  cancel |= CANCEL_AIR;
444  require |= REQUIRE_AIR;
445  break;
446 
447  case GROUND:
448  if (!mp_core_physics_component->GetFlag(OVERRIDE_CANCEL_GROUND)
449  && !mp_balance_trick_component->GetBalanceTrickType()
450  && !mp_balance_trick_component->DoingBalanceTrick())
451  // && !mp_core_physics_component->HaveLandedThisFrame())
452  {
453  cancel |= CANCEL_GROUND;
454  require |= REQUIRE_GROUND;
455  }
456  if (mp_balance_trick_component->GetBalanceTrickType() || mp_balance_trick_component->DoingBalanceTrick())
457  {
458  cancel |= CANCEL_MANUAL;
459  require |= REQUIRE_MANUAL;
460  }
461  break;
462 
463  case RAIL:
464  cancel |= CANCEL_RAIL;
465  require |= REQUIRE_RAIL;
466  break;
467 
468  case WALL:
469  cancel |= CANCEL_WALL;
470  require |= REQUIRE_WALL;
471  break;
472 
473  case LIP:
474  cancel |= CANCEL_LIP;
475  require |= REQUIRE_LIP;
476  break;
477 
478  case WALLPLANT:
479  cancel |= CANCEL_WALLPLANT;
480  require |= REQUIRE_WALLPLANT;
481  break;
482  }
483  }
484  else
485  {
486  // walking
487 
488  cancel |= CANCEL_WALK;
489  require |= REQUIRE_WALK;
490 
491  switch (mp_walk_component->GetState())
492  {
493  case CWalkComponent::WALKING_AIR:
494  cancel |= CANCEL_AIR;
495  require |= REQUIRE_AIR;
496  break;
497 
498  case CWalkComponent::WALKING_GROUND:
499  cancel |= CANCEL_GROUND;
500  require |= REQUIRE_GROUND;
501  break;
502 
503  case CWalkComponent::WALKING_HANG:
504  cancel |= CANCEL_HANG;
505  require |= REQUIRE_HANG;
506  break;
507 
508  case CWalkComponent::WALKING_LADDER:
509  cancel |= CANCEL_LADDER;
510  require |= REQUIRE_LADDER;
511  break;
512 
513  default:
514  break;
515  }
516  }
517  }
518  else
519  {
520  // driving
521 
522  cancel |= CANCEL_DRIVE;
523  require |= REQUIRE_DRIVE;
524 
525  // driving gaps are a bit of a kludge right now
526 
527  CCompositeObject* p_vehicle = static_cast< CCompositeObject* >(CCompositeObjectManager::Instance()->GetObjectByID(CRCD(0x824c6a24, "PlayerVehicle")));
528  Dbg_Assert(p_vehicle);
529  CVehicleComponent* p_vehicle_component = GetVehicleComponentFromObject(p_vehicle);
530  Dbg_Assert(p_vehicle_component);
531 
532  if (p_vehicle_component->IsOnGround())
533  {
534  cancel |= CANCEL_GROUND;
535  require |= REQUIRE_GROUND;
536  }
537  else
538  {
539  cancel |= CANCEL_AIR;
540  require |= REQUIRE_AIR;
541  }
542  }
543 }

550 void CSkaterGapComponent::start_gap ( Script::CStruct *pParams, Script::CScript* pScript )
551 {
552  if (mp_core_physics_component->GetFlag(IS_BAILING)) return;
553 
554  uint32 gap_id = 0;
555  pParams->GetChecksum(CRCD(0x3b442e26, "GapID"), &gap_id);
556  if (!gap_id)
557  {
558  // no gap_id is assumed to indicate a "Links" gap
559  Dbg_MsgAssert(pScript->mNode != -1, ("\n%s\nLink gap with no node", pScript->GetScriptInfo()));
560  Dbg_MsgAssert(SkateScript::GetNumLinks(pScript->mNode), ("\n%s\nLink gap with no links", pScript->GetScriptInfo()));
561  }
562 
563  #ifdef DEBUG_GAPS
564  MESSAGE("STARTING GAP");
565  DUMPC(gap_id);
566  #endif
567 
568  // check to see if there are any other gaps from this node with this ID and delete them
569  // (note that this means you can only have one "Links" gap from a single node although there can be many targets)
570  CGap* pGap = static_cast< CGap* >(m_gap_list.FirstItem());
571  while (pGap)
572  {
573  CGap* pNext = static_cast< CGap* >(pGap->GetNext());
574 
575  if (pGap->m_id == gap_id && static_cast< int >(pGap->m_node) == pScript->mNode)
576  {
577  delete pGap;
578  }
579  pGap = pNext;
580  }
581 
582  // create the gap and add the info to it
583  pGap = new CGap;
584  m_gap_list.AddToHead(pGap);
585  pGap->m_id = gap_id;
586  pGap->m_node = pScript->mNode;
587  pGap->m_trickTextChecksum = 0;
588 
589  // Take a snapshot of the skater's position & orientation so that if the gap is successfully
590  // got this info can be recorded into the CGapCheck instance and used to determine a good
591  // camera position for viewing the gap in the view-gaps menu.
592  pGap->m_skater_start_pos=((CSkater*)GetObject())->GetCamera()->GetPos();
593  pGap->m_skater_start_dir=-((CSkater*)GetObject())->GetCamera()->GetMatrix()[Mth::AT];
594 
595  // if it has a trickscript, it's a gaptrick
596  uint32 trickscript;
597  if (pParams->GetChecksum(CRCD(0xa26994e6, "trickscript"), &trickscript))
598  {
599  uint32 key_combo;
600  const char* tricktext;
601  Script::CArray* p_gapTricks;
602 
603  pGap->m_trickscript = trickscript;
604 
605  Dbg_Assert(mp_score);
606 
607  if (pParams->GetChecksum(CRCD(0x95e16467,"KeyCombo"), &key_combo))
608  {
609  pGap->m_trickChecksum = key_combo;
610 
611  int spin;
612  if ( pParams->GetInteger(CRCD(0xedf5db70, "spin"), &spin))
613  {
614  pGap->m_requirePerfect = pParams->ContainsFlag(CRCD(0x1c39f1b9, "perfect"));
615  Dbg_MsgAssert(spin % 180 == 0, ("StartGap called with a spin value of %i which is not a multiple of 180", spin));
616  pGap->m_spinMult = spin / 180;
617  }
618  pParams->GetInteger( CRCD(0xa4bee6a1,"num_taps"), &pGap->m_numTaps, Script::NO_ASSERT );
619 
620  pGap->m_numTrickOccurrences = mp_score->GetPreviousNumberOfOccurrences( key_combo, pGap->m_spinMult, pGap->m_numTaps );
621  }

748 void CSkaterGapComponent::end_gap ( Script::CStruct *pParams, Script::CScript* pScript )
749 {
750  if (mp_core_physics_component->GetFlag(IS_BAILING)) return;
751 
752  uint32 gap_id = 0;
753  pParams->GetChecksum(CRCD(0x3b442e26, "GapID"), &gap_id);
754 
755  #ifdef DEBUG_GAPS
756  MESSAGE("ENDING GAP");
757  DUMPC(gap_id);
758  #endif
759 
760  Dbg_Assert(mp_score);
761 
762  // find the ending gap in our gap list
763  CGap* pNext;
764  for (CGap* pGap = static_cast< CGap* >(m_gap_list.FirstItem()); pGap; pGap = pNext)
765  {
766  pNext = static_cast< CGap* >(pGap->GetNext());
767 
768  if (pGap->m_id != gap_id) continue;
769 
770  // check if its got a gap_id, or if the gaps node is linked to this node
771  if (gap_id == 0 && !SkateScript::IsLinkedTo(pGap->m_node, pScript->mNode)) continue;
772 
773  // found the gap
774 
775  // check requirements
776  if (pGap->m_flags & REQUIRE_MASK)
777  {
778  // requirements failed, so just delete the gap
779  delete pGap;
780  continue;
781  }
782 
783  // requirements met
784 
785  // Get the gap name, this is what is displayed as the trick name for this gap.
786  // This is not required, and will default if not there.
787  const char *p_name;
788  if (!pParams->GetString(CRCD(0xc4745838, "text"), &p_name))
789  {
790  p_name = "Unnamed Gap";
791  }
792  char gap_name[256];
793  sprintf(gap_name, "\\c1%s\\c0", p_name);
794 
795 
796  // check for gap trick; this MUST occur before the gap is added to the trick string
797  if (pGap->m_trickscript != 0)
798  {
799  bool gotGapTrick = false;
800 
801  if (pGap->m_trickChecksum != 0 || pGap->m_trickTextChecksum != 0 || pGap->mp_tricks->GetSize() > 0)
802  {
803  // see if the trick has been done since the gap started
804  if ( pGap->m_trickChecksum && pGap->m_numTrickOccurrences < mp_score->GetCurrentNumberOfOccurrences( pGap->m_trickChecksum, pGap->m_spinMult, pGap->m_requirePerfect, pGap->m_numTaps ) )
805  {
806  gotGapTrick = true;
807  }
808  else if ( pGap->m_trickTextChecksum && pGap->m_numTrickOccurrences < mp_score->GetCurrentNumberOfOccurrencesByName( pGap->m_trickTextChecksum, pGap->m_spinMult, pGap->m_requirePerfect, pGap->m_numTaps ) )
809  {
810  gotGapTrick = true;
811  }
812  else if ( pGap->mp_tricks->GetSize() > 0 && mp_score->GetCurrentNumberOfOccurrences( pGap->mp_tricks, pGap->m_startGapTrickCount ) > 0 )
813  {
814  gotGapTrick = true;
815  }
816 
817  // if they get the gap trick
818  if ( gotGapTrick )
819  {
820  // get the gap trick
821  CGapTrick* pGapTrick = static_cast< CGapTrick* >(m_gaptrick_list.FirstItem());
822  while ( pGapTrick )
823  {
824  CGapTrick* pNext = static_cast< CGapTrick* >(pGapTrick->GetNext());
825 
826  if (pGapTrick->m_id == pGap->m_id)
827  {
828  pGapTrick->GetGapTrick();
829  break;
830  }
831 
832  pGapTrick = pNext;
833  }
834  } // END if they get the gap trick
835  }
836  else
837  {
838  // if no trick script, look for a no requirement gap trick
839  CGapTrick* pGapTrick = static_cast< CGapTrick* >(m_gaptrick_list.FirstItem());
840  while (pGapTrick)
841  {
842  CGapTrick* pNext = static_cast< CGapTrick* >(pGapTrick->GetNext());
843 
844  if (pGapTrick->m_id == pGap->m_id)
845  {
846  pGapTrick->GetGapTrick();
847  break;
848  }
849  pGapTrick = pNext;
850  }
851  }
852  } // END if trickscript != 0
853 
854 
855  Mdl::Skate::Instance()->GetGapChecklist()->SetInfoForCamera(p_name,pGap->m_skater_start_pos,pGap->m_skater_start_dir);
856 
857  // Get the score for this gap.
858  // If there is no score defined, then we don't award a trick.
859  int score = 0;
860  if (pParams->GetInteger(CRCD(0xcd66c8ae, "score"), &score))
861  {
862  Mdl::Score::Flags flags = Mdl::Score::vGAP;
863  if (GetSkater()->IsLocalClient())
864  {
865  mp_score->Trigger(gap_name, score, flags);
866 
867  CTrickComponent* p_trick_component = GetTrickComponentFromObject(GetObject());
868  Dbg_Assert(p_trick_component);
869  p_trick_component->SetFirstTrickStarted(true);
870 
871  Mdl::Skate::Instance()->GetGapChecklist()->GetGapByText(p_name);
872 
873  GetObject()->SpawnAndRunScript(CRCD(0x541a2485, "DefaultGapScript"));
874  }
875  }
876 
877  // execute the gap_script
878  uint32 gap_script;
879  if (pParams->GetChecksum(CRCD(0x7c9e51ab, "gapscript"), &gap_script))
880  {
881  GetObject()->SpawnAndRunScript(
882  gap_script,
883  pScript->mNode,
884  pParams->ContainsFlag(CRCD(0x20209c31, "NetEnabled")),
885  pParams->ContainsFlag(CRCD(0x23627fd7, "Permanent"))
886  );
887  }
888 
889  // handle for continue script parameter
890  Script::CStruct * p_continue = NULL;
891  if (pParams->GetStructure(CRCD(0xec1cd520, "continue"), &p_continue))
892  {
893  start_gap(p_continue, pScript);
894  }
895 
896  // Broadcast an event saying we got the gap
897  // won't happen very often, and broadcasing is cheap)
898  // we also pass in the parameters, just in case we want to do
899  uint32 id_got = Crc::ExtendCRCWithString(gap_id,"_Success");
900  Obj::CTracker::Instance()->LaunchEvent( id_got, 0xffffffff, GetObject()->GetID(), pParams, true /*, radius */ );
901 
902  // Script functions we might have called could have deleted this gap and maybe the next gap.
903  // The safest things is to check to see if this one is still there, delete it, then reset back to the start of the list.
904  CGap* pKill = static_cast< CGap* >(m_gap_list.FirstItem());
905  while (pKill)
906  {
907  if (pKill == pGap)
908  {
909  delete pGap;
910  break;
911  }
912  pKill = static_cast< CGap* >(pKill->GetNext());
913  }
914 
915  // restart our pass through the gaps
916  pNext = static_cast< CGap* >(m_gap_list.FirstItem());
917  }
918 }
