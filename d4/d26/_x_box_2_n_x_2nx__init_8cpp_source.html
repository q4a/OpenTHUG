372 void FatalFileError( uint32 error )
373 {
374  static char* p_error_message_english[2] = { "There's a problem with the disc you're using.",
375  "It may be dirty or damaged." };
376  static char* p_error_message_french[2] = { "Le disque utilis prsente une anomalie.",
377  "Il est peut-tre sale ou endommag." };
378  static char* p_error_message_german[2] = { "Bei der benutzten CD ist ein Problem aufgetreten.",
379  "Mglicherweise ist sie verschmutzt oder beschdigt." };
380 
381  // Turn off the loading bar if it is active.
382  if( EngineGlobals.loadingbar_timer_event != 0 )
383  {
384  timeKillEvent( EngineGlobals.loadingbar_timer_event );
385  EngineGlobals.loadingbar_timer_event = 0;
386  }
387 
388  // Ensure the graphics device has been initialised at this point.
389  if( EngineGlobals.p_Device == NULL )
390  {
391  InitialiseEngine();
392  }
393 
394  // Wait for any rendering to complete.
395  EngineGlobals.p_Device->BlockUntilIdle();
396 
397  char* p_error_message[2];
398  switch( Config::GetLanguage())
399  {
400  case Config::LANGUAGE_FRENCH:
401  {
402  p_error_message[0] = p_error_message_french[0];
403  p_error_message[1] = p_error_message_french[1];
404  break;
405  }
406  case Config::LANGUAGE_GERMAN:
407  {
408  p_error_message[0] = p_error_message_german[0];
409  p_error_message[1] = p_error_message_german[1];
410  break;
411  }
412  default:
413  {
414  p_error_message[0] = p_error_message_english[0];
415  p_error_message[1] = p_error_message_english[1];
416  break;
417  }
418 
419  }
420 
421  // Set up the text string used for the error message.
422  SText error_text;
423  error_text.mp_font = (SFont*)EngineGlobals.p_memory_resident_font;
424  error_text.m_xpos = 48.0f;
425  error_text.m_ypos = 128.0f;
426  error_text.m_xscale = 0.8f;
427  error_text.m_yscale = 1.0f;
428  error_text.m_rgba = 0x80808080;
429  error_text.mp_next = NULL;
430 
431  set_texture( 1, NULL );
432  set_texture( 2, NULL );
433  set_texture( 3, NULL );
434 
435  // Want an infinite loop here.
436  while( true )
437  {
438  D3DDevice_Swap( D3DSWAP_DEFAULT );
439 
440  // Now that the swap instruction has been pushed, clear the buffer for next frame.
441  D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, 0x00000000, 1.0f, 0 );
442 
443  set_blend_mode( vBLEND_MODE_BLEND );
444  set_texture( 0, NULL );
445 
446  set_render_state( RS_UVADDRESSMODE0, 0x00010001UL );
447  set_render_state( RS_ZBIAS, 0 );
448  set_render_state( RS_ALPHACUTOFF, 1 );
449  set_render_state( RS_ZWRITEENABLE, 0 );
450 
451  D3DDevice_SetTextureStageState( 0, D3DTSS_COLORSIGN, D3DTSIGN_RUNSIGNED | D3DTSIGN_GUNSIGNED | D3DTSIGN_BUNSIGNED );
452  D3DDevice_SetTextureStageState( 0, D3DTSS_TEXTURETRANSFORMFLAGS, D3DTTFF_DISABLE );
453  D3DDevice_SetTextureStageState( 0, D3DTSS_TEXCOORDINDEX, D3DTSS_TCI_PASSTHRU | 0 );
454 
455  error_text.mp_string = p_error_message[0];
456  error_text.m_ypos = error_text.m_ypos - 16.0f;
457  error_text.BeginDraw();
458  error_text.Draw();
459  error_text.EndDraw();
460 
461  error_text.mp_string = p_error_message[1];
462  error_text.m_ypos = error_text.m_ypos + 16.0f;
463  error_text.BeginDraw();
464  error_text.Draw();
465  error_text.EndDraw();
466  }
