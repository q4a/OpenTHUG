41 // This will rotate pos-rotateCentre by degrees about Y, add the result to newCentre, and return the result.
42 // Needed by the clipboard when displaying the copied rails as they rotate with the cursor.
43 static Mth::Vector s_rotate_and_translate(Mth::Vector& pos, Mth::Vector& rotateCentre, Mth::Vector& newCentre, float degrees)
44 {
45  Mth::Vector d=pos-rotateCentre;
46 
47  float rad=degrees*3.141592654f/180.0f;
48  float s=sinf(rad);
49  float c=cosf(rad);
50 
51  Mth::Vector rotated;
52  rotated[X]=c*d[X]+s*d[Z];
53  rotated[Y]=d[Y];
54  rotated[Z]=c*d[Z]-s*d[X];
55 
56  return newCentre+rotated;
57 }

86 static void s_generate_end_vert_positions(Mth::Vector &pos, Mth::Vector &dir,
87  Mth::Vector *p_geom_verts,
88  int *p_vert_indices,
89  int numIndices)
90 {
91  // Need to copy the origin into a local var cos the source is about to be modified.
92  Mth::Vector origin=p_geom_verts[p_vert_indices[0]];
93 
94  Mth::Vector up(0.0f,1.0f,0.0f);
95  Mth::Vector u=dir;
96  u.Normalize();
97 
98  Mth::Vector w;
99  if (fabs(u[Y]) > 0.99999f)
100  {
101  w.Set(0.0f,0.0f,1.0f);
102  }
103  else
104  {
105  w=Mth::CrossProduct(up,u);
106  w.Normalize();
107  }
108 
109  Mth::Vector v=Mth::CrossProduct(u,w);
110  v.Normalize();
111  Mth::Matrix rot;
112  rot[Mth::RIGHT]=u;
113  rot[Mth::UP]=v;
114  rot[Mth::AT]=w;
115  rot[3][3]=1.0f;
116 
117  for (int i=0; i<numIndices; ++i)
118  {
119  Mth::Vector *p_geom_vert=&p_geom_verts[p_vert_indices[i]];
120  Mth::Vector off=*p_geom_vert-origin;
121  off=off*rot;
122  *p_geom_vert=off+pos;
123  }

135 // TODO: Fix bug where end of rail shrinks when rail is very steep.
136 static void s_calculate_rail_sector_vertex_coords(Mth::Vector &lastPos, Mth::Vector &startPos, Mth::Vector &endPos,
137  Nx::CSector *p_clonedSector, uint32 sourceSectorChecksum,
138  Nx::CSector *p_lastRailSector, bool update_collision)
139 {
140  // These arrays are figured out manually by printf'ing the vertex coords.
141  // Each end of the rail has 5 vertices, a top one, 2 middle ones and 2 bottom ones, forming a house shape.
142  // The top one has the highest y coord. All the verts for one end have the same x coord.
143  // Some of the vertices are duplicated however, which is why there is more than one top coord, & more than 2 middle, etc.
144  // For each vertex in p_end_verts_a, the corresponding vertex in p_end_verts_b must have the same y and z coords.
145  // This is so that the start of one rail piece can join up with the end of the last by tieing the vertices together.
146 
147  // TODO: These indices will need to be different on Xbox and GameCube
148 # ifdef __PLAT_NGC__
149  int p_end_verts_a[]=
150  {
151  2, 13, // Top
152  0, 4, 11, 15, // Middle
153  1, 3, 17, 19, // Bottom
154  };
155  int p_end_verts_b[]=
156  {
157  7, 12, // Top
158  5, 9, 10, 14, // Middle
159  6, 8, 16, 18, // Bottom
160  };
161 #else
162 # ifdef __PLAT_XBOX__
163  int p_end_verts_a[]=
164  {
165  1,9, // Top
166  3,10,14,19, // Middle
167  0,13,2,16, // Bottom
168  };
169  int p_end_verts_b[]=
170  {
171  6,11, // Top
172  7,8,15,18, // Middle
173  4,12,5,17, // Bottom
174  };
175 # else
176  int p_end_verts_a[]=
177  {
178  2,19, // Top
179  0,4,11,17,21, // Middle
180  1,3,13,15, // Bottom
181  };
182  int p_end_verts_b[]=
183  {
184  7,18, // Top
185  9,5,10,16,20, // Middle
186  8,6,12,14, // Bottom
187  };
188 # endif
189 # endif
190 
191  Dbg_MsgAssert(sizeof(p_end_verts_a)==sizeof(p_end_verts_b),("End vert array size mismatch!"));
192  int num_indices=sizeof(p_end_verts_a)/sizeof(int);
193 
194  // Get the default coords of the sectors vertices. These will be relative to the sectors origin.
195  Nx::CSector *p_source_sector=Nx::CEngine::sGetSector(sourceSectorChecksum);
196  Nx::CGeom *p_source_geom=p_source_sector->GetGeom();
197  Dbg_MsgAssert(p_source_geom,("NULL p_source_geom ?"));
198 
199  int num_render_verts=p_source_geom->GetNumRenderVerts();
200  Dbg_MsgAssert(num_render_verts==num_indices*2,("Unexpected extra vertices in rail sector, expected %d, got %d",2*num_indices,num_render_verts));
201 
202  // SPEEDOPT: If necessary, could use a static buffer for the verts
203  Mth::Vector *p_modified_render_verts=(Mth::Vector*)Mem::Malloc(num_render_verts * sizeof(Mth::Vector));
204  p_source_geom->GetRenderVerts(p_modified_render_verts);
205 
206  #ifdef __NOPT_ASSERT__
207  for (int i=0; i<num_indices; ++i)
208  {
209  Dbg_MsgAssert(p_modified_render_verts[p_end_verts_a[i]].GetY()==p_modified_render_verts[p_end_verts_b[i]].GetY(),("Rail sector vertex Y mismatch between vertices %d and %d",p_end_verts_a[i],p_end_verts_b[i]));
210  Dbg_MsgAssert(p_modified_render_verts[p_end_verts_a[i]].GetZ()==p_modified_render_verts[p_end_verts_b[i]].GetZ(),("Rail sector vertex Z mismatch between vertices %d and %d",p_end_verts_a[i],p_end_verts_b[i]));
211  }
212 
213  float rail_height=p_modified_render_verts[p_end_verts_a[0]].GetY()-p_modified_render_verts[p_end_verts_a[num_indices-1]].GetY();
214  Dbg_MsgAssert(rail_height==RAIL_SECTOR_HEIGHT,("Need to update RAIL_SECTOR_HEIGHT to be %f",rail_height));
215  #endif
216 
217  Mth::Vector dir;
218  if (p_lastRailSector)
219  {
220  dir=((startPos-lastPos)+(endPos-startPos)) / 2.0f;
221 
222  s_generate_end_vert_positions(startPos, dir, p_modified_render_verts, p_end_verts_a, num_indices);
223 
224  // Set the last sector's end verts equal to the just calculated set of coords too.
225  Nx::CGeom *p_last_geom=p_lastRailSector->GetGeom();
226  Dbg_MsgAssert(p_last_geom,("NULL p_last_geom ?"));
227 
228  int last_num_render_verts=p_last_geom->GetNumRenderVerts();
229  // SPEEDOPT: If necessary, could use a static buffer for the verts
230  Mth::Vector *p_last_verts=(Mth::Vector*)Mem::Malloc(last_num_render_verts * sizeof(Mth::Vector));
231  p_last_geom->GetRenderVerts(p_last_verts);
232 
233  // Tie the end verts to the new start verts.
234  for (int i=0; i<num_indices; ++i)
235  {
236  Dbg_MsgAssert(p_end_verts_b[i] < last_num_render_verts,("Bad index into p_last_verts"));
237  p_last_verts[p_end_verts_b[i]]=p_modified_render_verts[p_end_verts_a[i]];
238  }
239 
240  p_last_geom->SetRenderVerts(p_last_verts);
241 
242  Mem::Free(p_last_verts);
243  }
244  else
245  {
246  dir=endPos-startPos;
247  s_generate_end_vert_positions(startPos, dir, p_modified_render_verts, p_end_verts_a, num_indices);
248  }
249 
250  dir=endPos-startPos;
251  s_generate_end_vert_positions(endPos, dir, p_modified_render_verts, p_end_verts_b, num_indices);
252 
253 
254  // Write the vertex coords just calculated into the cloned sector.
255  Dbg_MsgAssert(p_clonedSector,("NULL p_clonedSector"));
256  Nx::CGeom *p_geom=p_clonedSector->GetGeom();
257  Dbg_MsgAssert(p_geom,("NULL p_geom ?"));
258  Dbg_MsgAssert(p_geom->GetNumRenderVerts()==p_source_geom->GetNumRenderVerts(),("Source geom num verts mismatch"));
259  p_geom->SetRenderVerts(p_modified_render_verts);
260 
262  // Now update the collision verts
264 
265  if (update_collision) // Uncomment this once the rail pieces have been reexported with correct collision
266  {
267  // Get the source render verts again so that we can match up the source collision verts with them, and hence get
268  // the index of the modified render vert to use for each collision vert.
269  Mth::Vector *p_source_render_verts=(Mth::Vector*)Mem::Malloc(num_render_verts * sizeof(Mth::Vector));
270  p_source_geom->GetRenderVerts(p_source_render_verts);
271  // The source render verts are in relative coords.
272 
273  // Get the source collision verts, which will also be in relative coords.
274  Nx::CCollObjTriData *p_source_col_data=p_source_geom->GetCollTriData();
275  int num_source_col_verts=p_source_col_data->GetNumVerts();
276 
277  Mth::Vector *p_collision_verts=(Mth::Vector*)Mem::Malloc(num_source_col_verts * sizeof(Mth::Vector));
278  p_source_col_data->GetRawVertices(p_collision_verts);
279 
280  // For each of the collision verts, look for a matching coord in the source render verts,
281  // and if found, write in the world coords calculated for it earlier.
282  for (int i=0; i<num_source_col_verts; ++i)
283  {
284  Mth::Vector pos=p_collision_verts[i];
285 
286  // All but two of the collision verts should be found by this loop.
287  bool found_exact_match=false;
288  for (int j=0; j<num_render_verts; ++j)
289  {
290  if (p_source_render_verts[j][X] == pos[X] &&
291  p_source_render_verts[j][Y] == pos[Y] &&
292  p_source_render_verts[j][Z] == pos[Z])
293  {
294  p_collision_verts[i]=p_modified_render_verts[j];
295  found_exact_match=true;
296  break;
297  }
298  }
299 
300  if (!found_exact_match)
301  {
302  bool found_match=false;
303 
304  // The two verts that have no corresponding render vert are the two bottom vertices
305  // of the collision poly that hangs below the rail.
306  // For these, find the render verts that match the x & z, and use their world position,
307  // but just drop the y down the same distance as it was below it in the original relative coords.
308  for (int j=0; j<num_render_verts; ++j)
309  {
310  if (p_source_render_verts[j][X] == pos[X] &&
311  p_source_render_verts[j][Z] == pos[Z])
312  {
313  // Get the y difference in the original relative coords ...
314  float y_off=pos[Y]-p_source_render_verts[j][Y];
315  // Set the collision vertex equal to the previously calculated world coords of the vertex ...
316  p_collision_verts[i]=p_modified_render_verts[j];
317  // Then move it down by the y difference again.
318  p_collision_verts[i][Y]+=y_off;
319 
320  found_match=true;
321  break;
322  }
323  }
324 
325  if (!found_match)
326  {
327  // If no match was found, then maybe the sector got re-exported in a strange manner.
328  pos.PrintContents();
329  Dbg_MsgAssert(0,("Could not find a render-vert match for collision vert %d",i));
330  }
331  }
332  }
333 
334  // Write the new collision verts into the cloned sector.
335  #ifdef __NOPT_ASSERT__
336  Nx::CCollObjTriData *p_dest_col_data=p_geom->GetCollTriData();
337  Dbg_MsgAssert(p_dest_col_data->GetNumVerts()==num_source_col_verts,("Bad p_dest_col_data->GetNumVerts() ?"));
338  #endif
339 
340  p_clonedSector->SetRawCollVertices(p_collision_verts);
341  Mem::Free(p_collision_verts);
342  Mem::Free(p_source_render_verts);
343  }
344 
345  Mem::Free(p_modified_render_verts);
346 }

384 {

432 void CEditedRailPoint::UpdateRailGeometry(Mth::Vector& rotateCentre, Mth::Vector& newCentre, float degrees)
433 {
434  // Note: The last point that was placed is actually the next in the list, since they get added to the front.
435  if (!mpNext)
436  {
437  // There is no last point, so a rail can't be created because there is nothing to join to.
438 
439  // TODO: Could maybe still make a small section of rail, so that the first post of a rail has
440  // a bit of rail overhanging it. Would look better than having the rail stop right above the post.
441 
442  // Make sure any existing rail is destroyed.

447  // Clone the sector if necessary
448  if (!mpClonedRailSector)

451  }

453  Mth::Vector a, b, c;
454 
455  // Don't tie the end vertices of the two rail sections together if they meet at too great
456  // an angle, cos otherwise one of the sections gets too thin.
457  bool join_to_last_sector=true;
458  if (mpNext->mpNext)
459  {
460  a=mpNext->mPos-mpNext->mpNext->mPos;
461  b=mPos-mpNext->mPos;

465  {
466  join_to_last_sector=false;
467  }
468 
469  a=s_rotate_and_translate(mpNext->mpNext->mPos, rotateCentre, newCentre, degrees);
470  b=s_rotate_and_translate(mpNext->mPos, rotateCentre, newCentre, degrees);
471  c=s_rotate_and_translate(mPos, rotateCentre, newCentre, degrees);
472 
473  s_calculate_rail_sector_vertex_coords(a, b, c,
474  mpClonedRailSector, CRCD(0x66fe99bf,"Sk5Ed_RA_Dynamic"),

476  }

482  a=s_rotate_and_translate(mpNext->mPos, rotateCentre, newCentre, degrees);

490 // Creates the post sector if it does not exist already, then writes the correct world
491 // positions into the vertices.
492 // This is called when the post is first created, and also gets called for all posts whenever a cell in
493 // the park is raised or lowered, so that the posts maintain contact with the ground.
494 
495 // rotateCentre, newCentre, and degrees allow the post to be displayed rotated and translated from its original
496 // position. This is used when displaying a rail section on the clipboard cursor in the park editor.
497 void CEditedRailPoint::UpdatePostGeometry(Mth::Vector& rotateCentre, Mth::Vector& newCentre, float degrees)
498 {
499  if (!mHasPost)
500  {
501  DestroyPostGeometry();
502  return;
503  }
504 
505  // Don't place a post if the rail is so close to the ground that there is not enough room for a post.
506  // This often happens when snapping a rail point to the edge of some level geometry for example.
507  if (mHeightAboveGround < RAIL_SECTOR_HEIGHT)
508  {
509  DestroyPostGeometry();
510  return;
511  }
512 
513  // Get the coords of the source sectors vertices. These will be relative to the sectors origin,
514  // which is in the middle of the post.
515  Nx::CSector *p_source_sector=Nx::CEngine::sGetSector(CRCD(0x2756c52d,"Sk5Ed_RAp_Dynamic"));
516  Nx::CGeom *p_source_geom=p_source_sector->GetGeom();
517  Dbg_MsgAssert(p_source_geom,("NULL p_source_geom ?"));
518 
519  int num_verts=p_source_geom->GetNumRenderVerts();
520  // SPEEDOPT: If necessary, could use a static buffer for the verts
521  Mth::Vector *p_verts=(Mth::Vector*)Mem::Malloc(num_verts * sizeof(Mth::Vector));
522  p_source_geom->GetRenderVerts(p_verts);
523 
524  // Find the y coords of the top and bottom
525  float min_y=1000000.0f;
526  float max_y=-1000000.0f;
527  for (int i=0; i < num_verts; ++i)
528  {
529  float y=p_verts[i].GetY();
530  if (y < min_y)
531  {
532  min_y=y;
533  }
534  if (y > max_y)
535  {
536  max_y=y;
537  }
538  }
539 
540  // Check that max_y and min_y are the same, just opposite in sign.
541  Dbg_MsgAssert(fabs(max_y-(-min_y)) < 0.01f,("Expected post geometry to have origin in centre, but min_y=%f, max_y=%f",min_y,max_y));
542  Dbg_MsgAssert(max_y > 0.0f,("max_y < 0 ?? (max_y=%f)",max_y));
543 
544  float half_post_height=max_y;
545 
546  float rail_sector_height=RAIL_SECTOR_HEIGHT;
547  // If the next post is at a different height, the height of the rail be a bit smaller than normal
548  // due to the end face being tilted to point towards the next post.
549  if (mpNext && mpPrevious)
550  {
551  Mth::Vector a=mpPrevious->mPos-mPos;
552  Mth::Vector b=mPos-mpNext->mPos;
553  // The end-face's normal will be the average of the two rails
554  Mth::Vector av=(a+b)/2.0f;
555  av.Normalize();
556  // av[Y] is the sine of the angle, but we want the cosine
557  rail_sector_height *= sqrtf(1.0f-av[Y]*av[Y]);
558  }
559 
560  Mth::Vector translated_rail_point=s_rotate_and_translate(mPos, rotateCentre, newCentre, degrees);
561 
562  // Translate the post so that it is centred on the rail point, then shift the y's down so that the
563  // top of the post is at the bottom of the rail piece.
564  // The height of the rail section needs to be added too because the rail point (mPos) is at the apex of the
565  // rail section.
566  float shift_down_amount=half_post_height+rail_sector_height;
567  for (int i=0; i < num_verts; ++i)
568  {
569  p_verts[i]+=translated_rail_point;
570  p_verts[i][Y]-=shift_down_amount;
571  }
572 
573  // The top of the post is now in the correct position.
574  // Now we need to move the bottom vertices down to ground level.
575 
576  // This array figured out manually, by printing out the vertex coords.
577  // TODO: These indices will need to be different on Xbox and GameCube
578 # ifdef __PLAT_NGC__
579  int p_bottom_vert_indices[]=
580  {
581  0,2,4,6, // Post bottom
582  8,9,10,11 // Base plate
583  };
584 #else
585 # ifdef __PLAT_XBOX__
586  int p_bottom_vert_indices[]=
587  {
588  8,9,10,11, // Post bottom
589  1,2,5,6 // Base plate
590  };
591 # else
592  int p_bottom_vert_indices[]=
593  {
594  1,3,5,7,9, // Post bottom
595  10,11,12,13 // Base plate
596  };
597 # endif
598 # endif
599  int num_bottom_vertices=sizeof(p_bottom_vert_indices)/sizeof(int);
600  Dbg_MsgAssert(num_bottom_vertices <= num_verts,("Too many bottom vertices!"));
601 
602  // The amount that the bottom vertices need to be moved down is the height, minus the
603  // previous shift_down_amount cos we've already moved them by that much, minus
604  // half the height of the post so that the bottom is on the ground.
605  shift_down_amount=mHeightAboveGround-shift_down_amount-half_post_height;
606 
607  for (int i=0; i<num_bottom_vertices; ++i)
608  {
609  p_verts[p_bottom_vert_indices[i]][Y]-=shift_down_amount;
610  }
611 
612  // Now rotate the post about the y so that it is aligned with
613  // the previous rail section.
614  if (mpNext)
615  {
616  Mth::Vector u=mPos-mpNext->mPos;
617  u[Y]=0.0f;
618  u.Normalize();
619  float ux=u[X];
620  float uz=u[Z];
621  float vx=-uz;
622  float vz=ux;
623 
624  for (int i=0; i < num_verts; ++i)
625  {
626  float dx=p_verts[i][X]-translated_rail_point[X];
627  float dz=p_verts[i][Z]-translated_rail_point[Z];
628  p_verts[i][X]=ux*dx+vx*dz+translated_rail_point[X];
629  p_verts[i][Z]=uz*dx+vz*dz+translated_rail_point[Z];
630  }
631  }
632 
633  // Clone the sector if necessary and write in the vertex coords just calculated.
634  if (!mpPostSector)
635  {
636  mpPostSector=s_clone_sector(CRCD(0x2756c52d,"Sk5Ed_RAp_Dynamic"));
637  }
638 
639  Nx::CGeom *p_geom=mpPostSector->GetGeom();
640  Dbg_MsgAssert(p_geom,("NULL p_geom ?"));
641  Dbg_MsgAssert(p_geom->GetNumRenderVerts()==p_source_geom->GetNumRenderVerts(),("Source geom num verts mismatch"));
642  p_geom->SetRenderVerts(p_verts);
643  Mem::Free(p_verts);
644 }
