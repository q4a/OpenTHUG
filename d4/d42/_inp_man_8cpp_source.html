122 void Manager::process_handlers( const Tsk::Task< Manager >& task )
123 {
124 
125 
126  int i;
127  Manager& manager = task.GetData();
128 
129  for( i = 0; i < SIO::vMAX_DEVICES; i++ )
130  {
131  // See if we hit the end of our record/playback buffers
132  if( ( manager.m_server[i].m_data_in ) &&
133  ( manager.m_server[i].m_data_in >= manager.m_server[i].m_data_in_end ))
134  {
135  manager.m_server[i].m_data_in = NULL;
136  }
137  if( ( manager.m_server[i].m_data_out ) &&
138  ( manager.m_server[i].m_data_out >= manager.m_server[i].m_data_out_end ))
139  {
140  manager.m_server[i].m_data_out = NULL;
141  }
142 
143  manager.m_server[i].service_handlers();
144  }
145 }
146 
147 // Note: The order of these checksums must match the corresponding order of the PAD_ defines
148 // for the buttons in inpman.h
149 static uint32 ButtonChecksums[PAD_NUMBUTTONS]=
150 {
151  0x0, // Nowt
152  0xbc6b118f, // Up
153  0xe3006fc4, // Down
154  0x85981897, // Left
155  0x4b358aeb, // Right
156  0xb7231a95, // UpLeft
157  0xa50950c5, // UpRight
158  0xd8847efa, // DownLeft
159  0x786b8b68, // DownRight
160  0x2b489a86, // Circle
161  0x321c9756, // Square
162  0x7323e97c, // X
163  0x20689278, // Triangle
164  0x26b0c991, // L1
165  0xbfb9982b, // L2
166  0xc8bea8bd, // L3
167  0xf2f1f64e, // R1
168  0x6bf8a7f4, // R2
169  0x1cff9762, // R3
170  0x767a45d7, // Black
171  0xbd30325b, // White
172  0x9d2d8850, // Z
173 };
174 
175 uint32 GetButtonChecksum( int whichButton )
176 {
177 
178  Dbg_MsgAssert( ( whichButton > 0 ) && ( whichButton < PAD_NUMBUTTONS ), ( "button %d out of range", whichButton ) );
179  return ( ButtonChecksums[ whichButton ] );
180 }
181 
182 int GetButtonIndex(uint32 Checksum)
183 {
184 
185  for (int i=1; i<PAD_NUMBUTTONS; ++i)
186  {
187  if (ButtonChecksums[i]==Checksum)
188  {
189  return i;
190  }
191  }
192  Dbg_MsgAssert(0,("Bad button name '%s' sent to GetButtonIndex",Script::FindChecksumName(Checksum)));
193  return 0;
194 }
