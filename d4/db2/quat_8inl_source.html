68 inline Quat::Quat( const Vector& axis, float angle )
69 {
70 
71 
72  float mod = axis.Length();
73  float ang = angle / 2.0f;
74 
75  mod = ( mod > 0.0f ) ? ( 1.0f / mod ) : 0.0f;
76  mod *= sinf( ang );
77 
78  SetVector( mod * axis[X], mod * axis[Y], mod * axis[Z] );
79  SetScalar( cosf ( ang ));
80 }

101 inline Quat::Quat( const Matrix& m )
102 {
103 
104 
105  float t = m[X][X] + m[Y][Y] + m[Z][Z];
106  float trace;
107 
108  if ( t > 0.0f )
109  {
110  trace = sqrtf( t + 1.0f );
111 
112  SetScalar( trace * 0.5f );
113  trace = 0.5f / trace;
114 
115  SetVector(( m[Z][Y] - m[Y][Z] ) * trace,
116  ( m[X][Z] - m[Z][X] ) * trace,
117  ( m[Y][X] - m[X][Y] ) * trace );
118  }
119  else
120  {
121  // find greatest element in Matrix diagonal
122  sint i = X;
123  if ( m[Y][Y] > m[X][X] ) i = Y;
124  if ( m[Z][Z] > m[i][i] ) i = Z;
125 
126  sint j = ( i + 1 ) % W;
127  sint k = ( j + 1 ) % W;
128 
129  trace = sqrtf(( m[i][i] - (m[j][j] + m[k][k] )) + 1.0f );
130 
131  quat[i] = ( trace * 0.5f );
132  trace = 0.5f / trace;
133  quat[j] = ( m[j][i] + m[i][j] ) * trace;
134  quat[k] = ( m[k][i] + m[i][k] ) * trace;
135  quat[W] = ( m[k][j] - m[j][k] ) * trace;
136  }
137 }

382 inline void Quat::GetMatrix ( Matrix& mat ) const
383 {
384 
385 
386  float xs, ys, zs,
387  wx, wy, wz,
388  xx, xy, xz,
389  yy, yz, zz, ss;
390 
391 
392 #if 0
393  // our original version. Seems essentially broken, as LengthSqr ignores W
394  ss = 2.0f / quat.LengthSqr();
395 #else
396  // version suggested by Andre at Left Field
397  // uses proper Modulus, and clamps 2.0/0.0f to zero
398  ss = ModulusSqr();
399  ss = ( ss>0.0f ? 2.0f/ss : 0.0f);
400 #endif
401 
402  xs = quat[X] * ss;
403  ys = quat[Y] * ss;
404  zs = quat[Z] * ss;
405 
406  wx = quat[W] * xs;
407  wy = quat[W] * ys;
408  wz = quat[W] * zs;
409 
410  xx = quat[X] * xs;
411  xy = quat[X] * ys;
412  xz = quat[X] * zs;
413 
414  yy = quat[Y] * ys;
415  yz = quat[Y] * zs;
416 
417  zz = quat[Z] * zs;
418 
419  mat[X][X] = 1.0f - (yy + zz);
420  mat[Y][X] = xy + wz;
421  mat[Z][X] = xz - wy;
422 
423  mat[X][Y] = xy - wz;
424  mat[Y][Y] = 1.0f - (xx + zz);
425  mat[Z][Y] = yz + wx;
426 
427  mat[X][Z] = xz + wy;
428  mat[Y][Z] = yz - wx;
429  mat[Z][Z] = 1.0f - (xx + yy );
430 }

534 inline Quat Slerp ( const Quat& q1, const Quat& q3, float t )
535 {
536  float sclp, sclq;
537 
538  // GJ: Need to possibly negate the second quaternion,
539  // to avoid spinning in the wrong direction.
540  Quat q2 = q3;
541  if ( DotProduct( q1, q3 ) < 0.0f )
542  {
543  q2 = -q2;
544  }
545 
546  float coso = q1.GetScalar() * q2.GetScalar() +
547  DotProduct ( q1.GetVector(), q2.GetVector());
548 
549  Quat q;
550 
551  if (( 1.0f + coso + 1.0f ) > EPSILON )
552  {
553  if (( 1.0f - coso ) > EPSILON ) // slerp
554  {
555  float omega = acosf ( coso );
556  float sino = sinf ( omega );
557 
558  sclp = sinf (( 1.0f - t ) * omega ) / sino;
559  sclq = sinf ( t * omega ) / sino;
560  }
561  else // lerp ( angle tends to 0 )
562  {
563  sclp = 1.0f - t;
564  sclq = t;
565  }
566 
567  q = ( sclp * q1 ) + ( sclq * q2 );
568  }
569  else // angle tends to 2*PI
570  {
571  q.SetVector ( -q1[Y], q1[X], -q1[W] );
572  q.SetScalar ( q1[Z] );
573 
574  sclp = sinf (( 1.0f - t ) * PI / 2.0f );
575  sclq = sinf ( t * PI / 2.0f );
576 
577  q.SetVector (( sclp * q1.GetVector()) + ( sclq * q.GetVector()));
578  }
579 
580  return q;
581 }

617 inline void QuatVecToMatrix( Mth::Quat* pQ, Mth::Vector* pT, Mth::Matrix* pMatrix )
618 {
619  Dbg_Assert( pQ );
620  Dbg_Assert( pT );
621  Dbg_Assert( pMatrix );
622 
623  pQ->Invert();
624 
625  Mth::Vector square;
626  Mth::Vector cross;
627  Mth::Vector wimag;
628 
629  square[X] = (*pQ)[X] * (*pQ)[X];
630  square[Y] = (*pQ)[Y] * (*pQ)[Y];
631  square[Z] = (*pQ)[Z] * (*pQ)[Z];
632 
633  cross[X] = (*pQ)[Y] * (*pQ)[Z];
634  cross[Y] = (*pQ)[Z] * (*pQ)[X];
635  cross[Z] = (*pQ)[X] * (*pQ)[Y];
636 
637  wimag[X] = (*pQ)[W] * (*pQ)[X];
638  wimag[Y] = (*pQ)[W] * (*pQ)[Y];
639  wimag[Z] = (*pQ)[W] * (*pQ)[Z];
640 
641  (*pMatrix)[Mth::RIGHT][X] = 1 - 2 * (square[Y] + square[Z]);
642  (*pMatrix)[Mth::RIGHT][Y] = 2 * (cross[Z] + wimag[Z]);
643  (*pMatrix)[Mth::RIGHT][Z] = 2 * (cross[Y] - wimag[Y]);
644  (*pMatrix)[Mth::RIGHT][W] = 0.0f;
645 
646  (*pMatrix)[Mth::UP][X] = 2 * (cross[Z] - wimag[Z]);
647  (*pMatrix)[Mth::UP][Y] = 1 - 2 * (square[X] + square[Z]);
648  (*pMatrix)[Mth::UP][Z] = 2 * (cross[X] + wimag[X]);
649  (*pMatrix)[Mth::UP][W] = 0.0f;
650 
651  (*pMatrix)[Mth::AT][X] = 2 * (cross[Y] + wimag[Y]);
652  (*pMatrix)[Mth::AT][Y] = 2 * (cross[X] - wimag[X]);
653  (*pMatrix)[Mth::AT][Z] = 1 - 2 * (square[X] + square[Y]);
654  (*pMatrix)[Mth::AT][W] = 0.0f;
655 
656  (*pMatrix)[Mth::POS] = *pT;
657  (*pMatrix)[Mth::POS][W] = 1.0f;
658 }

660 inline void SCacheQuatVecToMatrix( Mth::Quat* pQ, Mth::Vector* pT, Mth::Matrix* pMatrix )
661 {
662  Dbg_Assert( pQ );
663  Dbg_Assert( pT );
664  Dbg_Assert( pMatrix );
665 
666  pQ->Invert();
667 
668  Mth::Vector *p_square = ((Mth::Vector*)0x7000000)+0;
669  Mth::Vector *p_cross = ((Mth::Vector*)0x7000000)+1;
670  Mth::Vector *p_wimag = ((Mth::Vector*)0x7000000)+2;
671 
672  (*p_square)[X] = (*pQ)[X] * (*pQ)[X];
673  (*p_square)[Y] = (*pQ)[Y] * (*pQ)[Y];
674  (*p_square)[Z] = (*pQ)[Z] * (*pQ)[Z];
675 
676  (*p_cross)[X] = (*pQ)[Y] * (*pQ)[Z];
677  (*p_cross)[Y] = (*pQ)[Z] * (*pQ)[X];
678  (*p_cross)[Z] = (*pQ)[X] * (*pQ)[Y];
679 
680  (*p_wimag)[X] = (*pQ)[W] * (*pQ)[X];
681  (*p_wimag)[Y] = (*pQ)[W] * (*pQ)[Y];
682  (*p_wimag)[Z] = (*pQ)[W] * (*pQ)[Z];
683 
684  (*pMatrix)[Mth::RIGHT][X] = 1 - 2 * ((*p_square)[Y] + (*p_square)[Z]);
685  (*pMatrix)[Mth::RIGHT][Y] = 2 * ((*p_cross)[Z] + (*p_wimag)[Z]);
686  (*pMatrix)[Mth::RIGHT][Z] = 2 * ((*p_cross)[Y] - (*p_wimag)[Y]);
687  (*pMatrix)[Mth::RIGHT][W] = 0.0f;
688 
689  (*pMatrix)[Mth::UP][X] = 2 * ((*p_cross)[Z] - (*p_wimag)[Z]);
690  (*pMatrix)[Mth::UP][Y] = 1 - 2 * ((*p_square)[X] + (*p_square)[Z]);
691  (*pMatrix)[Mth::UP][Z] = 2 * ((*p_cross)[X] + (*p_wimag)[X]);
692  (*pMatrix)[Mth::UP][W] = 0.0f;
693 
694  (*pMatrix)[Mth::AT][X] = 2 * ((*p_cross)[Y] + (*p_wimag)[Y]);
695  (*pMatrix)[Mth::AT][Y] = 2 * ((*p_cross)[X] - (*p_wimag)[X]);
696  (*pMatrix)[Mth::AT][Z] = 1 - 2 * ((*p_square)[X] + (*p_square)[Y]);
697  (*pMatrix)[Mth::AT][W] = 0.0f;
698 
699  (*pMatrix)[Mth::POS] = *pT;
700  (*pMatrix)[Mth::POS][W] = 1.0f;
701 }

718 inline float counter_warp(float t, float cos_alpha)
719 {
720  const float ATTENUATION = 0.82279687f;
721  const float WORST_CASE_SLOPE = 0.58549219f;
722 
723  float factor = 1.0f - ATTENUATION * cos_alpha;
724  factor *= factor;
725  float k = WORST_CASE_SLOPE * factor;
726 
727  return t*(k*t*(2.0f*t - 3.0f) + 1.0f + k);
728 }

772 inline Mth::Quat EulerToQuat( const Mth::Vector& euler )
773 {
774  // This code is a modified version of Left Field's
775  // MakeQuatFromEuler() function.
776 
777  Mth::Quat q;
778 
779  float roll = euler[X];
780  float pitch = euler[Y];
781  float yaw = euler[Z];
782 
783  float cyaw, cpitch, croll, syaw, spitch, sroll;
784  float cyawcpitch, syawspitch, cyawspitch, syawcpitch;
785 
786  cyaw = cosf(0.5f * yaw);
787  cpitch = cosf(0.5f * pitch);
788  croll = cosf(0.5f * roll);
789 
790  syaw = sinf(0.5f * yaw);
791  spitch = sinf(0.5f * pitch);
792  sroll = sinf(0.5f * roll);
793 
794  cyawcpitch = cyaw * cpitch;
795  syawspitch = syaw * spitch;
796  cyawspitch = cyaw * spitch;
797  syawcpitch = syaw * cpitch;
798
