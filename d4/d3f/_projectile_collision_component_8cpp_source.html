169 // The component's Update() function is called from the CCompositeObject's
170 // Update() function. That is called every game frame by the CCompositeObjectManager
171 // from the s_logic_code function that the CCompositeObjectManager registers
172 // with the task manger.
173 void CProjectileCollisionComponent::Update()
174 {
175  float radii;
176 
177  radii = m_radius + vSKATER_RADIUS;
178  if (!mp_suspend_component->SkipLogic() && !m_dying )
179  {
180  float dist;
181  bool team_game;
182  Mdl::Skate* skate_mod = Mdl::Skate::Instance();
183  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
184  uint32 NumSkaters = skate_mod->GetNumSkaters();
185  GameNet::PlayerInfo* target_player, *src_player;
186 
187  team_game = skate_mod->GetGameMode()->IsTeamGame();
188  for( uint32 i = 0; i < NumSkaters; i++ )
189  {
190  Obj::CSkater *pSkater = skate_mod->GetSkater(i);
191  Obj::CSkaterScoreComponent* p_score_comp;
192  Mdl::Score* score;
193 
194  // Can't hit yourself with your own projectile
195  if( pSkater->GetID() == m_owner_id )
196  {
197  continue;
198  }
199 
200  src_player = gamenet_man->GetPlayerByObjectID( m_owner_id );
201  target_player = gamenet_man->GetPlayerByObjectID( pSkater->GetID());
202 
203  // Allow players to get up after being knocked down
204  if( target_player->IsVulnerable() == false )
205  {
206  continue;
207  }
208 
209  p_score_comp = GetSkaterScoreComponentFromObject( pSkater );
210  score = p_score_comp->GetScore();
211 
212  // Don't target dead players
213  if( score->GetTotalScore() <= 0 )
214  {
215  continue;
216  }
217 
218  if( team_game )
219  {
220  if( src_player->m_Team == target_player->m_Team )
221  {
222  uint32 enabled;
223 
224  enabled = gamenet_man->GetNetworkPreferences()->GetPreferenceChecksum( CRCD(0xe959c43a,"friendly_fire"), CRCD(0x21902065,"checksum"));
225  if( enabled != CRCD(0xf81bc89b,"boolean_true"))
226  {
227  continue;
228  }
229  }
230  }
231 
232  dist = (pSkater->GetPos() - GetObject()->GetPos()).Length();
233  if( dist < radii )
234  {
235  GameNet::MsgProjectileHit hit_msg;
236  Net::MsgDesc msg_desc;
237  Net::Server* server;
238  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
239 
240  server = gamenet_man->GetServer();
241 
242  //Dbg_Printf( "COLLISION: Skater %d: owner: %d\n", pSkater->GetID(), m_owner_id);
243  GetObject()->MarkAsDead();
244  if( m_death_script != 0 )
245  {
246  Script::CStruct* params;
247  Mth::Vector pos;
248 
249  pos = GetObject()->GetPos();
250  params = new Script::CStruct;
251 
252  params->AddVector( CRCD(0x7f261953,"pos"), pos );
253  params->AddVector( CRCD(0xc4c809e,"vel"), m_vel );
254  params->AddFloat( CRCD(0x13b9da7b,"scale"), m_scale );
255 
256  Script::RunScript( m_death_script, params );
257  delete params;
258  }
259 
260  // a message explaining "You've been hit down by ID for N damage"
261  hit_msg.m_Id = m_owner_id;
262  hit_msg.m_Damage = get_damage_amount();
263 
264  /*if (CInputComponent* p_input_component = GetInputComponentFromObject(GetObject()))
265  {
266  p_input_component->DisableInput();
267  }
268  p_player->MarkAsNonCollidable();*/
269 
270  msg_desc.m_Id = GameNet::MSG_ID_SKATER_HIT_BY_PROJECTILE;
271  msg_desc.m_Data = &hit_msg;
272  msg_desc.m_Length = sizeof(GameNet::MsgProjectileHit);
273  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
274  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
275  target_player->SetHitTime( Tmr::GetTime());
276  server->EnqueueMessage( target_player->GetConnHandle(), &msg_desc );
277 
278  // basically a message explaining "You hit someone"
279  hit_msg.m_Id = pSkater->GetID();
280  msg_desc.m_Id = GameNet::MSG_ID_SKATER_PROJECTILE_HIT_TARGET;
281  server->EnqueueMessage( src_player->GetConnHandle(), &msg_desc );
282 
283 
284  if(( score->GetTotalScore() - get_damage_amount() ) <= 0 )
285  {
286  // If this shot is killing them...
287  if( score->GetTotalScore() > 0 )
288  {
289  if( target_player->IsLocalPlayer() == false )
290  {
291  Script::CStruct* params;
292 
293  Dbg_Printf( "*** Elimination\n" );
294 
295  params = new Script::CStruct;
296  params->AddString( CRCD(0xa1dc81f9,"name"), pSkater->GetDisplayName());
297  Script::RunScript( CRCD(0x9b043179,"announce_elimination"), params );
298  delete params;
299  skate_mod->HideSkater( pSkater, true );
300  }
301  }
302  score->SetTotalScore( 0 );
303  }
304  else
305  {
306  score->SetTotalScore( score->GetTotalScore() - get_damage_amount());
307  }
308 
309  break;
310  }
311  }
312  }
313 }
