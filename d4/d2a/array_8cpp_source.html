44 // Deletes the array buffer if it exists, asserting if it contains any non-NULL pointers.
45 // Sets type to NONE and size to 0.
46 void CArray::Clear()
47 {
48  if (m_size==1)
49  {
50  // Memory optimization:
51  // Special case for size 1. In this case, no memory block has been allocated.
52 
53  if (m_union)
54  {
55  // The element is not zero ...
56 
57  #ifdef __NOPT_ASSERT__
58  // Check that no references to things remain in the array.
59  switch (m_type)
60  {
61  case ESYMBOLTYPE_INTEGER:
62  case ESYMBOLTYPE_FLOAT:
63  case ESYMBOLTYPE_NAME:
64  // No need for the user to have zeroed these.
65  break;
66 
67  case ESYMBOLTYPE_STRING:
68  case ESYMBOLTYPE_LOCALSTRING:
69  case ESYMBOLTYPE_PAIR:
70  case ESYMBOLTYPE_VECTOR:
71  case ESYMBOLTYPE_STRUCTURE:
72  case ESYMBOLTYPE_ARRAY:
73  {
74  // The array contains a pointer to something.
75  // The CArray cannot delete it itself because this would cause cyclic dependencies.
76  Dbg_MsgAssert(0,("Tried to delete a CArray that still contains non-NULL data: size=%d type='%s'",m_size,GetTypeName(m_type)));
77  break;
78  }
79 
80  default:
81  Dbg_MsgAssert(0,("Bad CArray::m_type of '%s'",GetTypeName(m_type)));
82  break;
83  }
84  #endif
85 
86  m_union=0;
87  }
88  }
89  else
90  {
91  if (mp_array_data)
92  {
93  #ifdef __NOPT_ASSERT__
94  // Check that no references to things remain in the array.
95  switch (m_type)
96  {
97  case ESYMBOLTYPE_INTEGER:
98  case ESYMBOLTYPE_FLOAT:
99  case ESYMBOLTYPE_NAME:
100  // No need for the user to have zeroed these.
101  break;
102 
103  case ESYMBOLTYPE_STRING:
104  case ESYMBOLTYPE_LOCALSTRING:
105  case ESYMBOLTYPE_PAIR:
106  case ESYMBOLTYPE_VECTOR:
107  case ESYMBOLTYPE_STRUCTURE:
108  case ESYMBOLTYPE_ARRAY:
109  {
110  // The array is of pointers, so make sure that the user of CArray has deleted and zeroed these before deleting the array.
111  // The CArray cannot delete them itself because this would cause cyclic dependencies.
112  for (uint32 i=0; i<m_size; ++i)
113  {
114  Dbg_MsgAssert(mp_array_data[i]==0,("Tried to delete a CArray that still contains non-NULL data: size=%d type='%s'",m_size,GetTypeName(m_type)));
115  }
116  break;
117  }
118 
119  default:
120  Dbg_MsgAssert(0,("Bad CArray::m_type of '%s'",GetTypeName(m_type)));
121  break;
122  }
123  #endif
124 
125  Mem::Free(mp_array_data);
126  mp_array_data=NULL;
127  }
128  }
129 
130  m_type=ESYMBOLTYPE_NONE;
131  m_size=0;
132 }

134 // Calls Clear, sets the size and type, allocates the buffer if necessary and initialises it to all zeroes.
135 void CArray::SetSizeAndType(int size, ESymbolType type)
136 {
137  switch (type)
138  {
139  case ESYMBOLTYPE_INTEGER:
140  case ESYMBOLTYPE_FLOAT:
141  case ESYMBOLTYPE_NAME:
142  case ESYMBOLTYPE_STRING:
143  case ESYMBOLTYPE_LOCALSTRING:
144  case ESYMBOLTYPE_PAIR:
145  case ESYMBOLTYPE_VECTOR:
146  case ESYMBOLTYPE_STRUCTURE:
147  case ESYMBOLTYPE_ARRAY:
148  break;
149 
150  case ESYMBOLTYPE_NONE:
151  Dbg_MsgAssert(size==0,("Array type of NONE with non-zero size sent to CArray::SetSizeAndType"));
152  break;
153 
154  default:
155  Dbg_MsgAssert(0,("Bad type of '%s' sent to CArray::SetSizeAndType",GetTypeName(type)));
156  break;
157  }
158 
159  Clear();
160  m_type=type;
161  m_size=size;
162 
163  if (size)
164  {
165  if (size==1)
166  {
167  Dbg_MsgAssert(m_union==0,("m_union not zero ?"));
168  // Nothing to do. No memory block is allocated for arrays of 1 element, to save memory.
169  }
170  else
171  {
172  Dbg_MsgAssert(mp_array_data==NULL,("mp_array_data not NULL ?"));
173 
174  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
175  mp_array_data=(uint32*)Mem::Malloc(m_size*sizeof(uint32));
176  Mem::Manager::sHandle().PopContext();
177 
178  uint32 *p_long_word=mp_array_data;
179  for (uint32 i=0; i<m_size; ++i)
180  {
181  *p_long_word++=0;
182  }
183  }
184  }
185  else
186  {
187  // Make all zero size arrays have type none.
188  // Seems a reasonable thing to do, and in particular this fixed an assert in WriteToBuffer
189  // after CSkaterCareer::WriteIntoStructure had added an empty array of structures. (The Gaps array)
190  m_type=ESYMBOLTYPE_NONE;
191  }
192 }

194 void CArray::Resize(uint32 newSize)
195 {
196  if (newSize==m_size)
197  {
198  // Nothing to do
199  return;
200  }
201  Dbg_MsgAssert(newSize>m_size,("Tried to resize CArray to a smaller size, not supported yet ..."));
202  // TODO: Make it able to make the CArray smaller, if a need arises. To do, factor out some of the
203  // code from CleanUpArray so that the leftover bit can be cleaned up.
204  Dbg_MsgAssert(newSize>1,("Resizing arrays to size 1 not supported yet ..."));
205  // TODO: Support the above if need be. Need to not allocate a new buffer in that case.
206 
207  // Allocate the new buffer.
208  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
209  uint32 *p_new_buffer=(uint32*)Mem::Malloc(newSize*sizeof(uint32));
210  Mem::Manager::sHandle().PopContext();
211 
212  // Copy the contents of the old buffer into the new.
213  uint32 *p_source=GetArrayPointer();
214  // Note: Does not support resizing zero size arrays because it does not know what the type of the
215  // new array should be.
216  Dbg_MsgAssert(p_source,("NULL array pointer ?"));
217  uint32 *p_dest=p_new_buffer;
218  uint32 i;
219  for (i=0; i<m_size; ++i)
220  {
221  *p_dest++=*p_source++;
222  }
223  // Zero the remainder of the new buffer.
224  uint32 remainder=newSize-m_size;
225  for (i=0; i<remainder; ++i)
226  {
227  *p_dest++=0;
228  }
229 
230  // Only free mp_array_data if the old size was bigger than 1.
231  // mp_array_data is not allocated for sizes of 1 as a memory optimization.
232  if (m_size > 1)
233  {
234  Mem::Free(mp_array_data);
235  }
236  mp_array_data=p_new_buffer;
237 
238  m_size=newSize;
239  Dbg_MsgAssert(m_size>1,("Expected array size to be > 1 ??")); // Just to be sure
240 }
