99 void CCameraLookAroundComponent::Update()
100 {
101  CControlPad * p_control_pad;
102 
103  if (Script::GetInt(CRCD(0x702247a5,"Viewer_controller"), false))
104  {
105  p_control_pad = &mp_input_component->GetControlPad2();
106  }
107  else
108  {
109  p_control_pad = &mp_input_component->GetControlPad();
110  }
111 
112 
113  // Handle pressing of "Select" button
114  //
115  if (!Script::GetInt(CRCD(0xf3e055e1,"select_shift")))
116  {
117  if (p_control_pad->m_select.GetTriggered())
118  {
119  printf ("Camera Detected Select Pressed\n");
120  p_control_pad->m_select.ClearTrigger();
121  // Depending on which skatercam we are, run one of the UserSelectSelect scripts
122  if (GetObject()->GetID() == CRCD(0x967c138c,"SkaterCam0"))
123  {
124  Script::RunScript(CRCD(0x60871393,"UserSelectSelect"));
125  }
126  else
127  {
128  Script::RunScript(CRCD(0xa1b1146d,"UserSelectSelect2"));
129  }
130  }
131  }
132 
133  if (p_control_pad->m_R3.GetTriggered())
134  {
135  mLookaroundLock = !mLookaroundLock;
136  p_control_pad->m_R3.ClearTrigger();
137  }
138 
139  float frame_length = Tmr::FrameLength();
140 
141  if (!mLookaroundLock && !mLookaroundOverride)
142  {
143  float target = p_control_pad->m_scaled_rightY;
144  target = -1.4f * (target < 0.0f ? (0.7f * target) : (0.5f * target));
145  if (Mth::Abs(target - mLookaroundTilt) > 0.001f)
146  {
147  mLookaroundTilt += (target - mLookaroundTilt) * 3.75f * frame_length;
148  }
149  else
150  {
151  mLookaroundTilt = target;
152  }
153 
154  target = 3.0f * p_control_pad->m_scaled_rightX;
155  if (Mth::Abs(target - mLookaroundHeading) > 0.001f)
156  {
157  mLookaroundHeading += (target - mLookaroundHeading) * 3.75f * frame_length;
158  }
159  else
160  {
161  mLookaroundHeading = target;
162  }
163  }
164 
165  float delta_time = Tmr::FrameLength();
166 
167  if (mLookaroundOverride)
168  {
169  if (mLookaroundDeltaTimer > 0.0f)
170  {
171  if (mLookaroundDeltaTimer > delta_time)
172  {
173  mLookaroundHeading += mLookaroundHeadingDelta * delta_time;
174  mLookaroundTilt += mLookaroundTiltDelta * delta_time;
175  mLookaroundDeltaTimer -= delta_time;
176  }
177  else
178  {
179  mLookaroundHeading = mLookaroundHeadingTarget;
180  mLookaroundTilt = mLookaroundTiltTarget;
181  mLookaroundDeltaTimer = 0.0f;
182  }
183  }
184  }
185 }

192 CBaseComponent::EMemberFunctionResult CCameraLookAroundComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
193 {
194  switch ( Checksum )
195  {
196  // @script | SC_SetSkaterCamOverride |
197  // @parm float | heading |
198  // @parm float | tilt |
199  // @parm float | time |
200  // @parm float | zoom |
201  case 0xb8f23899: // SC_SetSkaterCamOverride
202  {
203  float heading, tilt, time;
204  float zoom = 1.0f;
205  pParams->GetFloat( CRCD(0xfd4bc03e,"heading"), &heading, Script::ASSERT );
206  pParams->GetFloat( CRCD(0xe3c07609,"tilt"), &tilt, Script::ASSERT );
207  pParams->GetFloat( CRCD(0x906b67ba,"time"), &time, Script::ASSERT );
208  pParams->GetFloat( CRCD(0x48d4868b,"zoom"), &zoom );
209 
210  SetLookaround( heading, tilt, time, zoom );
211 
212  break;
213  }
214 
215  // @script | SC_ClearSkaterCamOverride |
216  case 0xe3b327c0: // SC_ClearSkaterCamOverride
217  ClearLookaround();
218  break;
219 
220  default:
221  return CBaseComponent::MF_NOT_EXECUTED;
222  }
223  return CBaseComponent::MF_TRUE;
224 }

245 void CCameraLookAroundComponent::SetLookaround( float heading, float tilt, float time, float zoom )
246 {
247  mLookaroundOverride = true;
248 
249  mLookaroundHeadingTarget = heading;
250  mLookaroundTiltTarget = tilt;
251 
252  // Calculate delta - the amount to move in 1 second.
253  if (time > 0.0f)
254  {
255  mLookaroundHeadingDelta = (heading - mLookaroundHeading) / time;
256  mLookaroundTiltDelta = (tilt - mLookaroundTilt) / time;
257  }
258 
259  mLookaroundDeltaTimer = time;
260  mLookaroundZoom = zoom;
261 }
