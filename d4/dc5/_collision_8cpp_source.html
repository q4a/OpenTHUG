89 bool CCollObj::s_found_collision(const Mth::Line *p_is, CCollObj *p_coll_object, SCollSurface *p_collSurface,
90  float distance, CollData *p_collData)
91 {
92  Dbg_Assert( p_collData );
93 
94  bool good = false;
95  bool updated = false;
96 
97  if (s_check_for_far)
98  {
99  if( distance >= p_collData->dist )
100  {
101  good = true;
102  }
103  }
104  else
105  {
106  //Dbg_Message("Checking new dist %f against old dist %f", distance, p_collData->dist);
107  if( distance <= p_collData->dist )
108  {
109  good = true;
110  }
111  }
112  CollData old;
113 
114 
115  if (!good)
116  {
117  old = *p_collData;
118  }
119 
120 
121  // Get the user plugin offset
122  int index = p_collSurface->index; // the index of the surface in the object
123 
124  // get the face flags from within this
125  // note that this uses the "index" field, which is
126  // an index of a face within the sector
127  uint32 flags = p_coll_object->GetFaceFlags(index);
128 
129  // Test flags against the ignore_1 (ignore if a bit is set in flags)
130  // and against ignore_0 (ignore if the bit is clear in flags)
131 
132  //Dbg_Message("*Flags %x ignore_0 %x ignore_1 %x", flags, p_collData->ignore_0, p_collData->ignore_1);
133  if (!(flags & p_collData->ignore_1) && !(~flags & p_collData->ignore_0))
134  {
135  // Setup the collData members
136 
137  p_collData->coll_found = true; // found at least one (might be overrridden, but found at least one)
138  p_collData->surface = *p_collSurface; // Store the collision surface
139  p_collData->dist = distance; // set distance (will end up being the smallest dist)
140  p_collData->flags = flags; // Store face flags
141  p_collData->terrain = (ETerrainType) p_coll_object->GetFaceTerrainType(index);
142  p_collData->p_coll_object = p_coll_object;
143  p_collData->trigger = (flags& mFD_TRIGGER); // set flag if it's a trigger
144 
145  p_collData->node_name = p_coll_object->GetChecksum();
146  // Determine the point of intersection on the triangle
147  // which will be in the direction of the line's vector at "distance" length
148  Mth::Vector coll_point, line_vec;
149  line_vec = p_is->m_end - p_is->m_start;
150 
151  coll_point = line_vec.Scale(distance);
152  coll_point = p_is->m_start + coll_point;
153  coll_point[W] = 1.0f; // Hack to force into homogenious space. Even though the math is correct here,
154  // many times the line data fed to it isn't.
155  p_collData->surface.point = coll_point; // After looking at their implementation, I don't
156  // trust "point" so I recalculate it
157 
158  // check for additional per-face callback
159 
160  if (p_collData->callback)
161  {
162  // call the callback function, and pass it this col data
163  p_collData->callback(p_collData);
164  }
165 
166  updated = true;
167  }
168 
169  if (!good)
170  {
171  *p_collData = old;
172  }
173 
174  return (good && updated);
175 }

456 {
457  // determines collision between two lines (p and q) in 2D; reports the interval along p at which the collision occurs
458 
459  // Antonio's method; Real-Time Rendering, p 337
460 
461  const float EPSILON = 0.00000000001f;
462 
463  float a_X = q_delta_X;
464  float a_Y = q_delta_Y;
465 
466  float b_X = p_delta_X;
467  float b_Y = p_delta_Y;
468 
469  float c_X = p_start_X - q_start_X;
470  float c_Y = p_start_Y - q_start_Y;
471 
472  float d = c_X * -a_Y + c_Y * a_X;
473 
474  float e = c_X * -b_Y + c_Y * b_X;
475 
476  float f = a_X * -b_Y + a_Y * b_X;
477 
478  if (f > 0.0f)
479  {
480  if (d < 0.0f || d > f) return false;
481  }
482  else
483  {
484  if (d > 0.0f || d < f) return false;
485  }
486 
487  if (f > 0.0f)
488  {
489  if (e < 0.0f || e > f) return false;
490  }
491  else
492  {
493  if (e > 0.0f || e < f) return false;
494  }
495 
496  if (Mth::Abs(f) < EPSILON)
497  {
498  *s = 0.0f;
499  return true;
500  }
501 
502  *s = d / f;
503 
504  return true;
505 }

512 bool CCollObj::sRectangleTriangleCollision(const Mth::Rectangle *rect, const Mth::Vector *v0, const Mth::Vector *v1, const Mth::Vector *v2, Mth::Vector p_collision_endpoints[2], ETriangleEdgeType p_collision_triangle_edges[2] )
513 {
514  // ERIT method; Real-Time Rendering, p 317
515 
516  const float EPSILON = 0.00001f;
517 
518  // calculate the plane equation of the rectangle
519  Mth::Vector rectangle_n = Mth::CrossProduct(rect->m_first_edge, rect->m_second_edge);
520  float rectangle_d = -Mth::DotProduct(rectangle_n, rect->m_corner);
521 
522  // calculate the distance (times rectangle_n squared) of the triangle vertices from the rectangle's plane
523  float d0 = Mth::DotProduct(rectangle_n, *v0) + rectangle_d;
524  float d1 = Mth::DotProduct(rectangle_n, *v1) + rectangle_d;
525  float d2 = Mth::DotProduct(rectangle_n, *v2) + rectangle_d;
526 
527  // if all vertices are on the same side of the plane, we reject
528  if (d0 > 0.0f && d1 > 0.0f && d2 > 0.0f) return false;
529  if (d0 < 0.0f && d1 < 0.0f && d2 < 0.0f) return false;
530 
531  // snap verticies near the plane to the plane
532  if (Mth::Abs(d0) < EPSILON)
533  {
534  d0 = 0.0f;
535  }
536  if (Mth::Abs(d1) < EPSILON)
537  {
538  d1 = 0.0f;
539  }
540  if (Mth::Abs(d2) < EPSILON)
541  {
542  d2 = 0.0f;
543  }
544 
545  // find the two points (p and q) on the plane of the rectangle which are the endpoints of the triangle's interception with the rectangle's plane
546  Mth::Vector p, q;
547  ETriangleEdgeType p_edge, q_edge;
548  if ((d0 <= 0.0f && d1 >= 0.0f) || (d0 >= 0.0f && d1 <= 0.0f))
549  {
550  p_edge = TRIANGLE_EDGE_V0_V1;
551  if (Mth::Abs(d0 - d1) > EPSILON)
552  {
553  p = *v0 + (*v1 - *v0) * d0 / (d0 - d1);
554  }
555  else
556  {
557  p = *v0;
558  }
559 
560  if ((d1 <= 0.0f && d2 >= 0.0f) || (d1 >= 0.0f && d2 <= 0.0f))
561  {
562  q_edge = TRIANGLE_EDGE_V1_V2;
563  if (Mth::Abs(d1 - d2) > EPSILON)
564  {
565  q = *v1 + (*v2 - *v1) * d1 / (d1 - d2);
566  }
567  else
568  {
569  q = *v1;
570  }
571  }
572  else
573  {
574  q_edge = TRIANGLE_EDGE_V2_V0;
575  if (Mth::Abs(d2 - d0) > EPSILON)
576  {
577  q = *v2 + (*v0 - *v2) * d2 / (d2 - d0);
578  }
579  else
580  {
581  q = *v2;
582  }
583  }
584  }
585  else
586  {
587  p_edge = TRIANGLE_EDGE_V1_V2;
588  if (Mth::Abs(d1 - d2) > EPSILON)
589  {
590  p = *v1 + (*v2 - *v1) * d1 / (d1 - d2);
591  }
592  else
593  {
594  p = *v1;
595  }
596 
597  q_edge = TRIANGLE_EDGE_V2_V0;
598  if (Mth::Abs(d0 - d2) > EPSILON)
599  {
600  q = *v0 + (*v2 - *v0) * d0 / (d0 - d2);
601  }
602  else
603  {
604  q = *v0;
605  }
606  }
607 
608  // project the rectangle and the intersection endpoints into the coordinate plane which maximizes the area of the rectangle's projection
609 
610  float r_corner_S, r_corner_T;
611  float r_first_edge_S, r_first_edge_T;
612  float r_second_edge_S, r_second_edge_T;
613  float p_S, p_T;
614  float q_S, q_T;
615 
616  if (Mth::Abs(rectangle_n[X]) < Mth::Abs(rectangle_n[Z]) && Mth::Abs(rectangle_n[Y]) < Mth::Abs(rectangle_n[Z]))
617  {
618  // project onto XY plane
619  r_corner_S = rect->m_corner[X];
620  r_corner_T = rect->m_corner[Y];
621  r_first_edge_S = rect->m_first_edge[X];
622  r_first_edge_T = rect->m_first_edge[Y];
623  r_second_edge_S = rect->m_second_edge[X];
624  r_second_edge_T = rect->m_second_edge[Y];
625  p_S = p[X];
626  p_T = p[Y];
627  q_S = q[X];
628  q_T = q[Y];
629  }
630  else if (Mth::Abs(rectangle_n[X]) < Mth::Abs(rectangle_n[Y]) && Mth::Abs(rectangle_n[Z]) < Mth::Abs(rectangle_n[Y]))
631  {
632  // project onto XZ plane
633  r_corner_S = rect->m_corner[X];
634  r_corner_T = rect->m_corner[Z];
635  r_first_edge_S = rect->m_first_edge[X];
636  r_first_edge_T = rect->m_first_edge[Z];
637  r_second_edge_S = rect->m_second_edge[X];
638  r_second_edge_T = rect->m_second_edge[Z];

642  q_T = q[Z];
643  }
644  else
645  {
646  // project onto YZ plane
647  r_corner_S = rect->m_corner[Y];
648  r_corner_T = rect->m_corner[Z];
649  r_first_edge_S = rect->m_first_edge[Y];
650  r_first_edge_T = rect->m_first_edge[Z];
651  r_second_edge_S = rect->m_second_edge[Y];
652  r_second_edge_T = rect->m_second_edge[Z];
653  p_S = p[Y];
654  p_T = p[Z];
655  q_S = q[Y];
656  q_T = q[Z];
657  }
658 
659  // test p and q against the four edges of the rectangle; if both lie outside any edge, we reject
660 
661  // these variables define a rectangle's edge in 2D; as + bt + c = 0
662  float a, b, c;
663 
664  float pd, qd;
665  float inside_sign;
666 
667  // not really convinced that this bitmap method of faster
668  // char within = 0;
669  // enum
670  // {
671  // P_TRIANGLE_EDGE_0_MASK = 1 << 0,
672  // Q_TRIANGLE_EDGE_0_MASK = 1 << 1,
673  // P_TRIANGLE_EDGE_1_MASK = 1 << 2,
674  // Q_TRIANGLE_EDGE_1_MASK = 1 << 3,
675  // P_TRIANGLE_EDGE_2_MASK = 1 << 4,
676  // Q_TRIANGLE_EDGE_2_MASK = 1 << 5,
677  // P_TRIANGLE_EDGE_3_MASK = 1 << 6,
678  // Q_TRIANGLE_EDGE_3_MASK = 1 << 7,
679  // ALL_P_MASKS = P_TRIANGLE_EDGE_0_MASK | P_TRIANGLE_EDGE_1_MASK | P_TRIANGLE_EDGE_2_MASK | P_TRIANGLE_EDGE_3_MASK,
680  // ALL_Q_MASKS = Q_TRIANGLE_EDGE_0_MASK | Q_TRIANGLE_EDGE_1_MASK | Q_TRIANGLE_EDGE_2_MASK | Q_TRIANGLE_EDGE_3_MASK
681  // };
682  bool p_within[4] = { false, false, false, false };
683  bool q_within[4] = { false, false, false, false };
684 
685  // edge 0: corner to corner + first_edge
686  // second_edge defines inside direction
687  a = r_first_edge_T;
688  b = -r_first_edge_S;
689  c = r_corner_T * r_first_edge_S - r_corner_S * r_first_edge_T;
690  pd = a * p_S + b * p_T + c;
691  qd = a * q_S + b * q_T + c;
692  inside_sign = a * r_second_edge_S + b * r_second_edge_T;
693  if (inside_sign > 0.0f)
694  {
695  // within = (P_TRIANGLE_EDGE_0_MASK * (pd >= 0.0f)) | (Q_TRIANGLE_EDGE_0_MASK * (qd >= 0.0f));
696  p_within[0] = pd >= 0.0f;
697  q_within[0] = qd >= 0.0f;
698  }
699  else
700  {
701  // within = (P_TRIANGLE_EDGE_0_MASK * (pd <= 0.0f)) | (Q_TRIANGLE_EDGE_0_MASK * (qd <= 0.0f));
702  p_within[0] = pd <= 0.0f;
703  q_within[0] = qd <= 0.0f;
704  }
705  // if (!(within & (P_TRIANGLE_EDGE_0_MASK | Q_TRIANGLE_EDGE_0_MASK))) return false;
706  if (!p_within[0] && !q_within[0]) return false;
707 
708  // edge 1: corner + second_edge to corner + first_edge + second_edge
709  // second_edge defines outside direction
710  c = (r_corner_T + r_second_edge_T) * r_first_edge_S - (r_corner_S + r_second_edge_S) * r_first_edge_T;
711  pd = a * p_S + b * p_T + c;
712  qd = a * q_S + b * q_T + c;
713  if (inside_sign < 0.0f)
714  {
715  // within |= (P_TRIANGLE_EDGE_1_MASK * (pd >= 0.0f)) | (Q_TRIANGLE_EDGE_1_MASK * (qd >= 0.0f));
716  p_within[1] = pd >= 0.0f;
717  q_within[1] = qd >= 0.0f;
718  }
719  else
720  {
721  // within |= (P_TRIANGLE_EDGE_1_MASK * (pd <= 0.0f)) | (Q_TRIANGLE_EDGE_1_MASK * (qd <= 0.0f));
722  p_within[1] = pd <= 0.0f;
723  q_within[1] = qd <= 0.0f;
724  }
725  // if (!(within & (P_TRIANGLE_EDGE_1_MASK | Q_TRIANGLE_EDGE_1_MASK))) return false;
726  if (!p_within[1] && !q_within[1]) return false;
727 
728  // edge 2: corner to corner + second_edge
729  // first_edge defines inside direction
730  a = r_second_edge_T;
731  b = -r_second_edge_S;
732  c = r_corner_T * r_second_edge_S - r_corner_S * r_second_edge_T;
733  pd = a * p_S + b * p_T + c;
734  qd = a * q_S + b * q_T + c;
735  inside_sign = a * r_first_edge_S + b * r_first_edge_T;
736  if (inside_sign > 0.0f)
737  {
738  // within |= (P_TRIANGLE_EDGE_2_MASK * (pd >= 0.0f)) | (Q_TRIANGLE_EDGE_2_MASK * (qd >= 0.0f));
739  p_within[2] = pd >= 0.0f;
740  q_within[2] = qd >= 0.0f;
741  }
742  else
743  {
744  // within |= (P_TRIANGLE_EDGE_2_MASK * (pd <= 0.0f)) | (Q_TRIANGLE_EDGE_2_MASK * (qd <= 0.0f));
745  p_within[2] = pd <= 0.0f;
746  q_within[2] = qd <= 0.0f;
747  }
748  // if (!(within & (P_TRIANGLE_EDGE_2_MASK | Q_TRIANGLE_EDGE_2_MASK))) return false;
749  if (!p_within[2] && !q_within[2]) return false;
750 
751  // edge 3: corner + first_edge to corner + first_edge + second_edge
752  // first_edge defines outside direction
753  c = (r_corner_T + r_first_edge_T) * r_second_edge_S - (r_corner_S + r_first_edge_S) * r_second_edge_T;
754  pd = a * p_S + b * p_T + c;
755  qd = a * q_S + b * q_T + c;
756  if (inside_sign < 0.0f)
757  {
758  // within |= (P_TRIANGLE_EDGE_3_MASK * (pd >= 0.0f)) | (Q_TRIANGLE_EDGE_3_MASK * (qd >= 0.0f));
759  p_within[3] = pd >= 0.0f;
760  q_within[3] = qd >= 0.0f;
761  }
762  else
763  {
764  // within |= (P_TRIANGLE_EDGE_3_MASK * (pd <= 0.0f)) | (Q_TRIANGLE_EDGE_3_MASK * (qd <= 0.0f));
765  p_within[3] = pd <= 0.0f;
766  q_within[3] = qd <= 0.0f;
767  }
768  // if (!(within & (P_TRIANGLE_EDGE_3_MASK | Q_TRIANGLE_EDGE_3_MASK))) return false;
769  if (!p_within[3] && !q_within[3]) return false;
770 
771  int next_collision_endpoint_idx;
772 
773  // if p or q is within all edges, it is the endpoint of the collision
774  // if ((within & ALL_P_MASKS) == ALL_P_MASKS)
775  if (p_within[0] && p_within[1] && p_within[2] && p_within[3])
776  {
777  p_collision_endpoints[0] = p;
778  p_collision_triangle_edges[0] = p_edge;
779  // if ((within & ALL_Q_MASKS) == ALL_Q_MASKS)
780  if (q_within[0] && q_within[1] && q_within[2] && q_within[3])
781  {
782  p_collision_endpoints[1] = q;
783  p_collision_triangle_edges[1] = q_edge;
784  return true;
785  }
786  next_collision_endpoint_idx = 1;
787  }
788  // else if ((within & ALL_Q_MASKS) == ALL_Q_MASKS)
789  else if (q_within[0] && q_within[1] && q_within[2] && q_within[3])
790  {
791  p_collision_endpoints[0] = q;
792  p_collision_triangle_edges[0] = q_edge;
793  next_collision_endpoint_idx = 1;
794  }
795  else
796  {
797  next_collision_endpoint_idx = 0;
798  }
799 
800  // we've found any endpoints of the collision which are within the rectangle; further collision endpoints will lie on the edges of the rectangle;
801 
802  // we check each edge of the rectangle for intersection with the triangle's plane-intersection line segment
803 
804  float s;
805 
806  // edge 0: corner to corner + first_edge
807  if (s2DLineLineCollision(r_corner_S, r_corner_T, r_first_edge_S, r_first_edge_T, p_S, p_T, q_S - p_S, q_T - p_T, &s))
808  {
809  p_collision_endpoints[next_collision_endpoint_idx] = rect->m_first_edge;
810  p_collision_endpoints[next_collision_endpoint_idx] *= s;
811  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_corner;
812  p_collision_triangle_edges[next_collision_endpoint_idx] = NO_TRIANGLE_EDGE;
813  if (next_collision_endpoint_idx == 1) return true;
814  next_collision_endpoint_idx = 1;
815  }
816 
817  // edge 1: corner + second_edge to corner + first_edge + second_edge
818  if (s2DLineLineCollision(r_corner_S + r_second_edge_S, r_corner_T + r_second_edge_T, r_first_edge_S, r_first_edge_T, p_S, p_T, q_S - p_S, q_T - p_T, &s))
819  {
820  p_collision_endpoints[next_collision_endpoint_idx] = rect->m_first_edge;
821  p_collision_endpoints[next_collision_endpoint_idx] *= s;
822  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_corner;
823  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_second_edge;
824  p_collision_triangle_edges[next_collision_endpoint_idx] = NO_TRIANGLE_EDGE;
825  if (next_collision_endpoint_idx == 1) return true;
826  next_collision_endpoint_idx = 1;
827  }
828 
829  // edge 2: corner to corner + second_edge
830  if (s2DLineLineCollision(r_corner_S, r_corner_T, r_second_edge_S, r_second_edge_T, p_S, p_T, q_S - p_S, q_T - p_T, &s))
831  {
832  p_collision_endpoints[next_collision_endpoint_idx] = rect->m_second_edge;
833  p_collision_endpoints[next_collision_endpoint_idx] *= s;
834  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_corner;
835  p_collision_triangle_edges[next_collision_endpoint_idx] = NO_TRIANGLE_EDGE;
836  if (next_collision_endpoint_idx == 1) return true;
837  next_collision_endpoint_idx = 1;
838  }
839 
840  // edge 3: corner + first_edge to corner + first_edge + second_edge
841  if (s2DLineLineCollision(r_corner_S + r_first_edge_S, r_corner_T + r_first_edge_T, r_second_edge_S, r_second_edge_T, p_S, p_T, q_S - p_S, q_T - p_T, &s))
842  {
843  if (next_collision_endpoint_idx == 1)
844  {
845  p_collision_endpoints[next_collision_endpoint_idx] = rect->m_second_edge;
846  p_collision_endpoints[next_collision_endpoint_idx] *= s;
847  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_corner;
848  p_collision_endpoints[next_collision_endpoint_idx] += rect->m_first_edge;
849  p_collision_triangle_edges[next_collision_endpoint_idx] = NO_TRIANGLE_EDGE;
850  return true;
851  }
852  }
853 
854  // no collision could be found
855  return false;
856 }
