170 /******************************************************************/
171 /* */
172 /* */
173 /******************************************************************/
174 void sShatterInstanceDetails::UpdateParameters( int index, float timestep )
175 {
176  Dbg_Assert( index < m_num_triangles );
177 
178  mp_positions[index] += mp_velocities[index] * timestep;
179 
180  if(( mp_positions[index][Y] < m_bounce_level ) && ( mp_velocities[index][Y] < 0.0f ))
181  {
182  // Hit the floor. Bounce back up.
183  mp_positions[index][Y] = m_bounce_level + ( m_bounce_level - mp_positions[index][Y] );
184  mp_velocities[index][Y] = mp_velocities[index][Y] * -m_bounce_amplitude;
185 
186  // And figure a new rotation matrix.
187  Mth::Vector axis( -1.0f + ( 2.0f * (float)rand() / RAND_MAX ), -1.0f + ( 2.0f * (float)rand() / RAND_MAX ), -1.0f + ( 2.0f * (float)rand() / RAND_MAX ));
188  axis.Normalize();
189  mp_matrices[index].Ident();
190  mp_matrices[index].Rotate( axis, 0.1f * ((float)rand() / RAND_MAX ));
191  }
192 
193  mp_velocities[index][Y] -= m_gravity * timestep;
194 }

461 bool TextureSplat( Mth::Vector& splat_start, Mth::Vector& splat_end, float size, float lifetime, const char *p_texture_name, uint32 trail )
462 {
463  Replay::WriteTextureSplat(splat_start,splat_end,size,lifetime,p_texture_name,trail);
464 
465  // If this is a flagged as a trail splat, see if we have a trail instance already that matches.
466  sSplatTrailInstanceDetails *p_trail_details = NULL;
467  if( trail > 0 )
468  {
469  uint32 time = Tmr::GetTime();
470 
471  p_splat_trail_details_table->IterateStart();
472  p_trail_details = p_splat_trail_details_table->IterateNext();
473  while( p_trail_details )
474  {
475  sSplatTrailInstanceDetails *p_delete = NULL;
476 
477  if(( time < p_trail_details->m_last_pos_added_time ) || (( time - p_trail_details->m_last_pos_added_time ) > 500 ))
478  {
479  // This trail is over half a second old - remove it.
480  p_delete = p_trail_details;
481  }
482  else
483  {
484  // This trail had a point added within the last half second, so check it for bone ID.
485  if( p_trail_details->m_trail_id == trail )
486  {
487  // ID's match, so check it for proximity. Seems like fast moving reverts can cause successive calls to be up to 5.5 feet away.
488  float sq_dist = ( splat_start - p_trail_details->m_last_pos ).LengthSqr();
489  if( sq_dist < ( 66.0f * 66.0f ))
490  {
491  // This trail had a point added within the last half second that is within 3 feet of this point. Select this trail.
492  break;
493  }
494  }
495  }
496 
497  p_trail_details = p_splat_trail_details_table->IterateNext();
498 
499  if( p_delete )
500  {
501  p_splat_trail_details_table->FlushItem((uint32)p_delete );
502  delete p_delete;
503  }
504  }
505 
506  // If there were no trail details, create a new instance and just return.
507  if( p_trail_details == NULL )
508  {
509  p_trail_details = new sSplatTrailInstanceDetails;
510  p_trail_details->m_trail_id = trail;
511  p_trail_details->m_last_pos = splat_start;
512  p_trail_details->m_last_pos_added_time = time;
513  p_splat_trail_details_table->PutItem((uint32)p_trail_details, p_trail_details );
514  return true;
515  }
516  }
517 
518  // Convert the name string to a checksum.
519  uint32 texture_checksum = Crc::GenerateCRCFromString( p_texture_name );
520 
521  // Obtain a pointer to the texture.
522  Nx::CTexture *p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( texture_checksum );
523  Dbg_MsgAssert( p_texture, ("Couldn't find texture %s in particle texture dictionary", p_texture_name) );
524 
525  Mth::Line is( splat_start, splat_end );
526 
527  // Make initial line bounding box.
528  Mth::CBBox line_bbox( is.m_start );
529  line_bbox.AddPoint( is.m_end );
530 
531  // Get a list of all collision sectors within all super sectors that the line intersects.
532  // This is a very fast call, but likely to provide many collision sectors that are not actually within the
533  // bounding box of the line.
534  SSec::Manager *ss_man = Nx::CEngine::sGetNearestSuperSectorManager( is );
535  Nx::CCollStatic **pp_coll_obj_list = ss_man->GetIntersectingCollSectors( is );
536 
537  // Can't create blood splat if no objects found.
538  if( !pp_coll_obj_list )
539  {
540  return false;
541  }
542 
543  // The sector table is used to pass a platform independent set of CSector pointers to the below p-line code.
544  // Garrett: Also added static collision table, just in case CSector points to a CCollMulti
545  #define SECTOR_TABLE_SIZE 127
546  static CSector* sector_table[SECTOR_TABLE_SIZE + 1];
547  static Nx::CCollStatic* collision_table[SECTOR_TABLE_SIZE + 1];
548 
549  int sector_table_index = 0;
550  Nx::CCollStatic *p_coll_obj;
551  while(( p_coll_obj = *pp_coll_obj_list ))
552  {
553  if( p_coll_obj->GetGeometry())
554  {
555  // Determine whether this collision sector actually falls within the bounding box of our line.
556  if( p_coll_obj->GetGeometry()->GetBBox().Intersect( line_bbox ))
557  {
558  uint32 checksum = p_coll_obj->GetChecksum();
559 
560  // From the checksum we can get the CSector (the renderable version) from the hash table of the scene.
561  Nx::CSector *p_sector = Nx::CEngine::sGetSector( checksum );
562 
563  // Store the pointer in the table.
564  Dbg_Assert( sector_table_index < SECTOR_TABLE_SIZE );
565  collision_table[sector_table_index] = p_coll_obj;
566  sector_table[sector_table_index++] = p_sector;
567  }
568  ++pp_coll_obj_list;
569  }
570  }
571 
572  // Mark the end of the sector table with a NULL pointer.
573  if( sector_table_index > 0 )
574  {
575  sector_table[sector_table_index] = NULL;
576  collision_table[sector_table_index] = NULL;
577 
578  // Make sure we can can calculate an up vector. If not, just move on.
579  bool rv;
580  if ((trail == 0) || (splat_start[X] != p_trail_details->m_last_pos[X]) || (splat_start[Z] != p_trail_details->m_last_pos[Z]))
581  {
582  rv = plat_texture_splat( sector_table, collision_table, splat_start, splat_end, size, lifetime, p_texture, p_trail_details );
583  } else {
584  rv = false;
585  }
586 
587  // Update the trail details if present.
588  if( p_trail_details )
589  {
590  p_trail_details->m_last_pos = splat_start;
591  p_trail_details->m_last_pos_added_time = Tmr::GetTime();
592  }
593 
594  return rv;
595  }
596 
597  return false;
598 }
