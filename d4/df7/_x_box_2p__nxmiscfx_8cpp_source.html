172 bool subdivide_tri_stack( uint8 **p_write, NxXbox::sMesh *p_mesh )
173 {
174  // Three temporary buffers.
175  static uint8 v0[128];
176  static uint8 v1[128];
177  static uint8 v2[128];
178 
179  // Three more temporary buffers.
180  static uint8 i01[128];
181  static uint8 i12[128];
182  static uint8 i20[128];
183 
184  // If there are elements on the stack, pop off the top three vertices and subdivide if necessary.
185  if( triSubdivideStack.IsEmpty())
186  {
187  return false;
188  }
189 
190  D3DXVECTOR3 *p_v0 = (D3DXVECTOR3*)v0;
191  D3DXVECTOR3 *p_v1 = (D3DXVECTOR3*)v1;
192  D3DXVECTOR3 *p_v2 = (D3DXVECTOR3*)v2;
193 
194  // Stack is LIFO, so Pop() off in reverse order.
195  triSubdivideStack.Pop( p_v2 );
196  triSubdivideStack.Pop( p_v1 );
197  triSubdivideStack.Pop( p_v0 );
198 
199  // Calculate the area of this tri.
200  Mth::Vector p( p_v1->x - p_v0->x, p_v1->y - p_v0->y, p_v1->z - p_v0->z );
201  Mth::Vector q( p_v2->x - p_v0->x, p_v2->y - p_v0->y, p_v2->z - p_v0->z );
202  Mth::Vector r(( p[Y] * q[Z] ) - ( q[Y] * p[Z] ), ( p[Z] * q[X] ) - ( q[Z] * p[X] ), ( p[X] * q[Y] ) - ( q[X] * p[Y] ));
203  float area_squared = r.LengthSqr();
204 
205  if( area_squared > shatterAreaTest )
206  {
207  // We need to subdivide this tri. Calculate the three intermediate points.
208  int block_size = triSubdivideStack.GetBlockSize();
209 
210  memcpy( i01, v0, block_size );
211  memcpy( i12, v1, block_size );
212  memcpy( i20, v2, block_size );
213 
214  // Deal with positions (always present).
215  ((D3DXVECTOR3*)i01 )->x = p_v0->x + (( p_v1->x - p_v0->x ) * 0.5f );
216  ((D3DXVECTOR3*)i01 )->y = p_v0->y + (( p_v1->y - p_v0->y ) * 0.5f );
217  ((D3DXVECTOR3*)i01 )->z = p_v0->z + (( p_v1->z - p_v0->z ) * 0.5f );
218 
219  ((D3DXVECTOR3*)i12 )->x = p_v1->x + (( p_v2->x - p_v1->x ) * 0.5f );
220  ((D3DXVECTOR3*)i12 )->y = p_v1->y + (( p_v2->y - p_v1->y ) * 0.5f );
221  ((D3DXVECTOR3*)i12 )->z = p_v1->z + (( p_v2->z - p_v1->z ) * 0.5f );
222 
223  ((D3DXVECTOR3*)i20 )->x = p_v2->x + (( p_v0->x - p_v2->x ) * 0.5f );
224  ((D3DXVECTOR3*)i20 )->y = p_v2->y + (( p_v0->y - p_v2->y ) * 0.5f );
225  ((D3DXVECTOR3*)i20 )->z = p_v2->z + (( p_v0->z - p_v2->z ) * 0.5f );
226 
227  // Deal with colors (not always present).
228  if( p_mesh->m_diffuse_offset > 0 )
229  {
230  uint8 *p_v0col = (uint8*)( v0 + p_mesh->m_diffuse_offset );
231  uint8 *p_v1col = (uint8*)( v1 + p_mesh->m_diffuse_offset );
232  uint8 *p_v2col = (uint8*)( v2 + p_mesh->m_diffuse_offset );
233  uint8 *p_i01col = (uint8*)( i01 + p_mesh->m_diffuse_offset );
234  uint8 *p_i12col = (uint8*)( i12 + p_mesh->m_diffuse_offset );
235  uint8 *p_i20col = (uint8*)( i20 + p_mesh->m_diffuse_offset );
236 
237  for( int i = 0; i < 4; ++i )
238  {
239  p_i01col[i] = p_v0col[i] + (((int)p_v1col[i] - (int)p_v0col[i] ) / 2 );
240  p_i12col[i] = p_v1col[i] + (((int)p_v2col[i] - (int)p_v1col[i] ) / 2 );
241  p_i20col[i] = p_v2col[i] + (((int)p_v0col[i] - (int)p_v2col[i] ) / 2 );
242  }
243  }
244 
245  // Deal with uv0 (not always present).
246  if( p_mesh->m_uv0_offset > 0 )
247  {
248  float *p_v0uv = (float*)( v0 + p_mesh->m_uv0_offset );
249  float *p_v1uv = (float*)( v1 + p_mesh->m_uv0_offset );
250  float *p_v2uv = (float*)( v2 + p_mesh->m_uv0_offset );
251  float *p_i01uv = (float*)( i01 + p_mesh->m_uv0_offset );
252  float *p_i12uv = (float*)( i12 + p_mesh->m_uv0_offset );
253  float *p_i20uv = (float*)( i20 + p_mesh->m_uv0_offset );
254 
255  for( int i = 0; i < 2; ++i )
256  {
257  p_i01uv[i] = p_v0uv[i] + (( p_v1uv[i] - p_v0uv[i] ) * 0.5f );
258  p_i12uv[i] = p_v1uv[i] + (( p_v2uv[i] - p_v1uv[i] ) * 0.5f );
259  p_i20uv[i] = p_v2uv[i] + (( p_v0uv[i] - p_v2uv[i] ) * 0.5f );
260  }
261  }
262 
263  // Push the four new tris onto the stack.
264  triSubdivideStack.Push( v0 );
265  triSubdivideStack.Push( i01 );
266  triSubdivideStack.Push( i20 );
267 
268  triSubdivideStack.Push( i01 );
269  triSubdivideStack.Push( v1 );
270  triSubdivideStack.Push( i12 );
271 
272  triSubdivideStack.Push( i01 );
273  triSubdivideStack.Push( i12 );
274  triSubdivideStack.Push( i20 );
275 
276  triSubdivideStack.Push( i20 );
277  triSubdivideStack.Push( i12 );
278  triSubdivideStack.Push( v2 );
279  }
280  else
281  {
282  // Don't need to subdivide this tri.
283  int block_size = triSubdivideStack.GetBlockSize();
284 
285  // Just copy the tri into the next available slot.
286  memcpy( *p_write, v0, block_size );
287  *p_write += block_size;
288  memcpy( *p_write, v1, block_size );
289  *p_write += block_size;
290  memcpy( *p_write, v2, block_size );
291  *p_write += block_size;
292  }
293  return true;
294 }

471 static inline bool line_segment_intersection( float x1, float y1, float x2, float y2, float x3, float y3, float x4, float y4 )
472 {
473  float ax = x2 - x1;
474  float bx = x3 - x4;
475  float ay = y2 - y1;
476  float by = y3 - y4;
477  float cx = x1 - x3;
478  float cy = y1 - y3;
479  float d = by * cx - bx * cy; // Alpha numerator.
480  float f = ay * bx - ax * by; // Both denominator.
481 
482  // Alpha tests.
483  if( f > 0.0f )
484  {
485  if( d < 0.0f || d > f )
486  return false;
487  }
488  else
489  {
490  if( d > 0 || d < f )
491  return false;
492  }
493 
494  float e = ax * cy - ay * cx; // Beta numerator.
495 
496  // Beta tests.
497  if( f > 0.0f )
498  {
499  if( e < 0.0f || e > f )
500  return false;
501  }
502  else
503  {
504  if( e > 0 || e < f )
505  return false;
506  }
507 
508  return true;
509 }

513 /******************************************************************/
514 /* */
515 /* */
516 /******************************************************************/
517 static inline bool tri_texture_intersect( float u0, float v0, float u1, float v1, float u2, float v2 )
518 {
519  // Trivial check to see if all three points are outside range of texture.
520  if(( u0 < -1.0f ) && ( u1 < -1.0f ) && ( u2 < -1.0f ))
521  return false;
522  if(( u0 > 1.0f ) && ( u1 > 1.0f ) && ( u2 > 1.0f ))
523  return false;
524  if(( v0 < -1.0f ) && ( v1 < -1.0f ) && ( v2 < -1.0f ))
525  return false;
526  if(( v0 > 1.0f ) && ( v1 > 1.0f ) && ( v2 > 1.0f ))
527  return false;
528 
529  // Perform the check to see if any line segment forming the tri intersects any line segment forming the texture.
530  Mth::Vector texture_square[4] = { Mth::Vector( -1.0f, -1.0f, 0.0f, 0.0f ),
531  Mth::Vector( 1.0f, -1.0f, 0.0f, 0.0f ),
532  Mth::Vector( 1.0f, 1.0f, 0.0f, 0.0f ),
533  Mth::Vector( -1.0f, 1.0f, 0.0f, 0.0f )};
534  for( int p = 0; p < 4; ++p )
535  {
536  int q = ( p + 1 ) % 4;
537  if( line_segment_intersection( u0, v0, u1, v1, texture_square[p][X], texture_square[p][Y], texture_square[q][X], texture_square[q][Y] ))
538  return true;
539  if( line_segment_intersection( u1, v1, u2, v2, texture_square[p][X], texture_square[p][Y], texture_square[q][X], texture_square[q][Y] ))
540  return true;
541  if( line_segment_intersection( u2, v2, u0, v0, texture_square[p][X], texture_square[p][Y], texture_square[q][X], texture_square[q][Y] ))
542  return true;
543  }
544 
545  // If we reach this point there are three remaining possibilities:
546  // 1) That the tri lies entirely within the texture
547  // 2) That the texture lies entirely within the tri
548  // 3) That there is no space shared by tri and texture
549 
550  // 1) Perform a trivial check to see whether a corner of the tri lies within the texture.
551  if(( u0 >= -1.0f ) && ( u0 <= 1.0f ) && ( v0 >= -1.0f ) && ( v0 <= 1.0f ))
552  return true;
553  if(( u1 >= -1.0f ) && ( u1 <= 1.0f ) && ( v1 >= -1.0f ) && ( v1 <= 1.0f ))
554  return true;
555  if(( u2 >= -1.0f ) && ( u2 <= 1.0f ) && ( v2 >= -1.0f ) && ( v2 <= 1.0f ))
556  return true;
557 
558  // 2) Check that at least one corner of the texture falls within the tri.
559  Mth::Vector a( u0, v0, 0.0f );
560  Mth::Vector b( u1, v1, 0.0f );
561  Mth::Vector c( u2, v2, 0.0f );
562  for( int p = 0; p < 4; ++p )
563  {
564  if( point_in_triangle( texture_square[p], a, b, c ))
565  {
566  return true;
567  }
568  }
569 
570  // 3) No space shared.
571  return false;
572 }

580 bool plat_texture_splat( Nx::CSector **pp_sectors, Nx::CCollStatic **pp_collision, Mth::Vector& start, Mth::Vector& end, float size, float lifetime, Nx::CTexture *p_texture, Nx::sSplatTrailInstanceDetails *p_trail_details )
581 {
582  XGMATRIX view_matrix, ortho_matrix, projection_matrix;
583 
584 # if DRAW_DEBUG_LINES
585  Gfx::AddDebugLine( start, end, MAKE_RGB( 200, 200, 0 ), MAKE_RGB( 200, 200, 0 ), 1 );
586 # endif // DRAW_DEBUG_LINES
587 
588  // The length of the start->end line defines the view depth of the frustum.
589  Mth::Vector splat_vector = end - start;
590  float view_depth = splat_vector.Length();
591  splat_vector.Normalize();
592 
593  // Calculate the parallel projection matrix. Generally the projection vector will tend to point downwards, so we use a
594  // random vector in the x-z plane to define the up vector for the projection. However if this splat is part of a trail,
595  // use the previous point to generate the up vector.
596  Mth::Vector up;
597  if( p_trail_details )
598  {
599  up.Set( start[X] - p_trail_details->m_last_pos[X], start[Y] - p_trail_details->m_last_pos[Y], start[Z] - p_trail_details->m_last_pos[Z] );
600 
601  // The height of the viewport is defined by the distance between the two points.
602  float height = up.Length() * 0.5f;
603 
604  // Now we move start and end halfway back along the up vector.
605  start -= up * 0.5f;
606  end -= up * 0.5f;
607 
608  up.Normalize();
609 
610  XGMatrixLookAtRH( &view_matrix, (XGVECTOR3*)( &start[X] ), (XGVECTOR3*)( &end[X] ), (XGVECTOR3*)( &up[X] ));
611  XGMatrixOrthoRH( &ortho_matrix, size, height, 0.1f, view_depth );
612  }
613  else if( fabsf( splat_vector[Y] ) > 0.5f )
614  {
615  float angle = ((float)rand() * 2.0f * Mth::PI ) / (float)RAND_MAX;
616  up.Set( sinf( angle ), 0.0f, cosf( angle ));
617  XGMatrixLookAtRH( &view_matrix, (XGVECTOR3*)( &start[X] ), (XGVECTOR3*)( &end[X] ), (XGVECTOR3*)( &up[X] ));
618  XGMatrixOrthoRH( &ortho_matrix, size, size, 0.1f, view_depth );
619  }
620  else
621  {
622  up.Set( 0.0f, 1.0f, 0.0f );
623  XGMatrixLookAtRH( &view_matrix, (XGVECTOR3*)( &start[X] ), (XGVECTOR3*)( &end[X] ), (XGVECTOR3*)( &up[X] ));
624  XGMatrixOrthoRH( &ortho_matrix, size, size, 0.1f, view_depth );
625  }
626 
627  XGMatrixMultiply( &projection_matrix, &view_matrix, &ortho_matrix );
628 
629  // Pointer to the mesh we will be modifying. (Don't want to set the pointer up until we know for
630  // sure that we will be adding some polys).
631  sXboxSplatInstanceDetails *p_details = NULL;
632  sXboxSplatVert *p_target_verts = NULL;
633 
634  Nx::CSector *p_sector;
635 
636  while( p_sector = *pp_sectors )
637  {
638  Nx::CXboxGeom *p_xbox_geom = static_cast<Nx::CXboxGeom*>( p_sector->GetGeom());
639 
640  if( p_xbox_geom )
641  {
642 # if DRAW_DEBUG_LINES
643  Mth::Vector min = p_xbox_geom->GetBoundingBox().GetMin();
644  Mth::Vector max = p_xbox_geom->GetBoundingBox().GetMax();
645 
646  Mth::Vector box[8];
647  box[0] = box[1] = box[2] = box[3] = max;
648  box[1][X] = min[X];
649  box[2][Y] = min[Y];
650  box[3][Z] = min[Z];
651  box[5] = box[6] = box[7] = box[4] = min;;
652  box[5][X] = max[X];
653  box[6][Y] = max[Y];
654  box[7][Z] = max[Z];
655 
656  for ( int i = 1; i < 4; i++ )
657  {
658  Gfx::AddDebugLine( box[0], box[i], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
659  }
660  for ( int i = 5; i < 8; i++ )
661  {
662  Gfx::AddDebugLine( box[4], box[i], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
663  }
664  Gfx::AddDebugLine( box[1], box[6], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
665  Gfx::AddDebugLine( box[1], box[7], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
666  Gfx::AddDebugLine( box[2], box[5], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
667  Gfx::AddDebugLine( box[2], box[7], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
668  Gfx::AddDebugLine( box[3], box[5], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
669  Gfx::AddDebugLine( box[3], box[6], MAKE_RGB( 200, 0, 0 ), MAKE_RGB( 200, 0, 0 ), 1 );
670 # endif // DRAW_DEBUG_LINES
671 
672  // For each mesh in the geom...
673  for( uint32 m = 0; m < p_xbox_geom->m_num_mesh; ++m )
674  {
675  NxXbox::sMesh *p_mesh = p_xbox_geom->m_mesh_array[m];
676 
677  // Not allowed on meshes which are flagged not to shadow.
678  if( p_mesh->m_flags & NxXbox::sMesh::MESH_FLAG_NO_SKATER_SHADOW )
679  continue;
680 
681  // Check the bounding box of this mesh falls within the scope of the line.
682 
683  // Transform the mesh bounding box to see whether it falls within the projection frustum.
684 
685  // If it falls within the projection frustum, we need to explicitly transform all the vertices.
686  BYTE *p_vert_data;
687  p_mesh->mp_vertex_buffer[0]->Lock( 0, 0, &p_vert_data, D3DLOCK_READONLY );
688 
689  // Now scan through each non-degenerate tri, checking the verts to see if they are within scope.
690  uint32 index0;
691  uint32 index1 = p_mesh->mp_index_buffer[0][0];
692  uint32 index2 = p_mesh->mp_index_buffer[0][1];
693  for( uint32 i = 2; i < p_mesh->m_num_indices[0]; ++i )
694  {
695  // Wrap the indices round.
696  index0 = index1;
697  index1 = index2;
698  index2 = p_mesh->mp_index_buffer[0][i];
699 
700  if(( index0 != index1 ) && ( index0 != index2 ) && ( index1 != index2 ))
701  {
702  XGVECTOR3 uvprojections[3];
703  XGVec3TransformCoord( &uvprojections[0], (XGVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index0 )), &projection_matrix );
704  XGVec3TransformCoord( &uvprojections[1], (XGVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index1 )), &projection_matrix );
705  XGVec3TransformCoord( &uvprojections[2], (XGVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index2 )), &projection_matrix );
706 
707  // Check the z-values here, everything else is checked in tri_texture_intersect().
708  if(( uvprojections[0].z < 0.0f ) && ( uvprojections[1].z < 0.0f ) && ( uvprojections[2].z < 0.0f ))
709  continue;
710  if(( uvprojections[0].z > 1.0f ) && ( uvprojections[1].z > 1.0f ) && ( uvprojections[2].z > 1.0f ))
711  continue;
712 
713  // Okay, this tri lies within the projection frustum. Now check that it intersects the texture
714  if( !tri_texture_intersect( uvprojections[0].x, uvprojections[0].y,
715  uvprojections[1].x, uvprojections[1].y,
716  uvprojections[2].x, uvprojections[2].y ))
717  {
718  continue;
719  }
720 
721  // Get a pointer to the mesh used for rendering texture splats with the given texture.
722  // (Note this will create a new instance to handle texture splats of this texture if one does not already exist).
723  if( p_target_verts == NULL )
724  {
725  CXboxTexture *p_xbox_texture = static_cast<CXboxTexture*>( p_texture );
726  p_details = getDetailsForTextureSplat( p_xbox_texture->GetEngineTexture());
727  p_target_verts = p_details->m_verts;
728  Dbg_Assert( p_target_verts );
729  }
730 
731  // Scan through the lifetimes, finding a 'dead' poly (lifetime == 0), or the oldest.
732  uint32 idx = p_details->GetOldestSplat();
733 
734  // Convert lifetime from seconds to milliseconds.
735  p_details->m_lifetimes[idx] = (int)( lifetime * 1000.0f );
736 
737  // Set up the corresponding vertices. First write the positions...
738  uint32 index = idx * 3;
739  p_target_verts[index + 0].pos = *(D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index0 ));
740  p_target_verts[index + 1].pos = *(D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index1 ));
741  p_target_verts[index + 2].pos = *(D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index2 ));
742 
743  // ...then the uv's...
744  p_target_verts[index + 0].u = ( uvprojections[0].x + 1.0f ) * 0.5f;
745  p_target_verts[index + 0].v = ( uvprojections[0].y + 1.0f ) * 0.5f;
746  p_target_verts[index + 1].u = ( uvprojections[1].x + 1.0f ) * 0.5f;
747  p_target_verts[index + 1].v = ( uvprojections[1].y + 1.0f ) * 0.5f;
748  p_target_verts[index + 2].u = ( uvprojections[2].x + 1.0f ) * 0.5f;
749  p_target_verts[index + 2].v = ( uvprojections[2].y + 1.0f ) * 0.5f;
750 
751  // ...then the vertex colors.
752  p_target_verts[index + 0].col = ( *(D3DCOLOR*)( p_vert_data + ( p_mesh->m_vertex_stride * index0 ) + p_mesh->m_diffuse_offset ) & 0xFFFFFFUL ) | 0x80000000UL;
753  p_target_verts[index + 1].col = ( *(D3DCOLOR*)( p_vert_data + ( p_mesh->m_vertex_stride * index1 ) + p_mesh->m_diffuse_offset ) & 0xFFFFFFUL ) | 0x80000000UL;
754  p_target_verts[index + 2].col = ( *(D3DCOLOR*)( p_vert_data + ( p_mesh->m_vertex_stride * index2 ) + p_mesh->m_diffuse_offset ) & 0xFFFFFFUL ) | 0x80000000UL;
755 
756  D3DXVECTOR3 *p_v0 = &( p_target_verts[index + 0].pos );
757  D3DXVECTOR3 *p_v1 = &( p_target_verts[index + 1].pos );
758  D3DXVECTOR3 *p_v2 = &( p_target_verts[index + 2].pos );
759  Mth::Vector pv( p_v1->x - p_v0->x, p_v1->y - p_v0->y, p_v1->z - p_v0->z );
760  Mth::Vector qv( p_v2->x - p_v0->x, p_v2->y - p_v0->y, p_v2->z - p_v0->z );
761  Mth::Vector r(( pv[Y] * qv[Z] ) - ( qv[Y] * pv[Z] ), ( pv[Z] * qv[X] ) - ( qv[Z] * pv[X] ), ( pv[X] * qv[Y] ) - ( qv[X] * pv[Y] ));
762  float area_squared = r.LengthSqr();
763 
764  // Set the shatter test to ensure that we don't subdivide too far. Note that each successive subdivision will quarter
765  // the area of each triangle, which means the area *squared* of each triangle will become 1/16th of the previous value.
766  shatterAreaTest = area_squared / 128.0f;
767 
768  triSubdivideStack.Reset();
769  triSubdivideStack.SetBlockSize( sizeof( sXboxSplatVert ));
770  triSubdivideStack.Push( &p_target_verts[index + 0] );
771  triSubdivideStack.Push( &p_target_verts[index + 1] );
772  triSubdivideStack.Push( &p_target_verts[index + 2] );
773 
774  // Allocate a block of memory into which the subdivision stack will write the results.
775  uint8 *p_array = new uint8[8 * 1024];
776  uint8 *p_array_start = p_array;
777  uint8 *p_array_loop = p_array;
778  memset( p_array, 0, 8 * 1024 );
779 
780  NxXbox::sMesh *p_dummy_mesh = new NxXbox::sMesh();
781  p_dummy_mesh->m_diffuse_offset = 12;
782  p_dummy_mesh->m_uv0_offset = 16;
783 
784  while( subdivide_tri_stack( &p_array, p_dummy_mesh ));
785 
786  // Ensure we haven't overrun the buffer.
787  Dbg_Assert((uint32)p_array - (uint32)p_array_loop < ( 8 * 1024 ));
788 
789  bool subdivided_tri_added = false;
790 
791  while( p_array_loop != p_array )
792  {
793  // Add this triangle, *if* it is valid.
794  if( tri_texture_intersect((((sXboxSplatVert*)p_array_loop )[0].u * 2.0f ) - 1.0f,
795  (((sXboxSplatVert*)p_array_loop )[0].v * 2.0f ) - 1.0f,
796  (((sXboxSplatVert*)p_array_loop )[1].u * 2.0f ) - 1.0f,
797  (((sXboxSplatVert*)p_array_loop )[1].v * 2.0f ) - 1.0f,
798  (((sXboxSplatVert*)p_array_loop )[2].u * 2.0f ) - 1.0f,
799  (((sXboxSplatVert*)p_array_loop )[2].v * 2.0f ) - 1.0f ))
800  {
801  // We have added at least one subdivided tri.
802  subdivided_tri_added = true;
803 
804  // Convert lifetime from seconds to milliseconds.
805  p_details->m_lifetimes[idx] = (int)( lifetime * 1000.0f );
806 
807  p_target_verts[index + 0].pos = ((sXboxSplatVert*)p_array_loop )[0].pos;
808  p_target_verts[index + 1].pos = ((sXboxSplatVert*)p_array_loop )[1].pos;
809  p_target_verts[index + 2].pos = ((sXboxSplatVert*)p_array_loop )[2].pos;
810 
811  p_target_verts[index + 0].u = ((sXboxSplatVert*)p_array_loop )[0].u;
812  p_target_verts[index + 0].v = ((sXboxSplatVert*)p_array_loop )[0].v;
813  p_target_verts[index + 1].u = ((sXboxSplatVert*)p_array_loop )[1].u;
814  p_target_verts[index + 1].v = ((sXboxSplatVert*)p_array_loop )[1].v;
815  p_target_verts[index + 2].u = ((sXboxSplatVert*)p_array_loop )[2].u;
816  p_target_verts[index + 2].v = ((sXboxSplatVert*)p_array_loop )[2].v;
817 
818  p_target_verts[index + 0].col = (((sXboxSplatVert*)p_array_loop )[0].col & 0xFFFFFFUL ) | 0x80000000UL;
819  p_target_verts[index + 1].col = (((sXboxSplatVert*)p_array_loop )[1].col & 0xFFFFFFUL ) | 0x80000000UL;
820  p_target_verts[index + 2].col = (((sXboxSplatVert*)p_array_loop )[2].col & 0xFFFFFFUL ) | 0x80000000UL;
821 
822  idx = p_details->GetOldestSplat();
823  index = idx * 3;
824  }
825 
826  p_array_loop += ( sizeof( sXboxSplatVert ) * 3 );
827  }
828 
829  if( !subdivided_tri_added )
830  {
831  // No subdivided tris were added. This means we still have the large tri sitting in the list, which we don't want.
832  p_details->m_lifetimes[idx] = 0;
833  plat_texture_splat_reset_poly( p_details, idx );
834  }
835 
836  delete p_dummy_mesh;
837  delete [] p_array_start;
838 
839 # if DRAW_DEBUG_LINES
840  D3DXVECTOR3* p_d3dvert;
841  p_d3dvert = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index0 ));
842  Mth::Vector v0( p_d3dvert->x, p_d3dvert->y, p_d3dvert->z );
843  p_d3dvert = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index1 ));
844  Mth::Vector v1( p_d3dvert->x, p_d3dvert->y, p_d3dvert->z );
845  p_d3dvert = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index2 ));
846  Mth::Vector v2( p_d3dvert->x, p_d3dvert->y, p_d3dvert->z );
847  Gfx::AddDebugLine( v0, v1, MAKE_RGB( 0, 200, 200 ), MAKE_RGB( 0, 200, 200 ), 1 );
848  Gfx::AddDebugLine( v1, v2, MAKE_RGB( 0, 200, 200 ), MAKE_RGB( 0, 200, 200 ), 1 );
849  Gfx::AddDebugLine( v2, v0, MAKE_RGB( 0, 200, 200 ), MAKE_RGB( 0, 200, 200 ), 1 );
850 # endif // DRAW_DEBUG_LINES
851  }
852  }
853  }
854  }
855  ++pp_sectors;
856  }
857  return false;
858 }

939 /******************************************************************/
940 /* */
941 /* */
942 /******************************************************************/
943 void plat_shatter( CGeom *p_geom )
944 {
945  CXboxGeom *p_xbox_geom = static_cast<CXboxGeom*>( p_geom );
946 
947  // For each mesh in the geom...
948  for( uint32 m = 0; m < p_xbox_geom->m_num_mesh; ++m )
949  {
950  NxXbox::sMesh *p_mesh = p_xbox_geom->m_mesh_array[m];
951 
952  if( p_mesh->m_num_indices[0] >= 3 )
953  {
954  // Set the block size for this mesh.
955  triSubdivideStack.SetBlockSize( p_mesh->m_vertex_stride );
956 
957  // Get a pointer to the renderable data.
958  BYTE *p_vert_data;
959  p_mesh->mp_vertex_buffer[0]->Lock( 0, 0, &p_vert_data, D3DLOCK_READONLY );
960 
961  // First scan through each non-degenerate tri, counting them to see how many verts we'll need.
962  // We also have to figure the area of the tris here, since we need to calculate the worst case given the requirements for subdivision.
963  uint32 valid_tris = 0;
964  uint32 index0;
965  uint32 index1 = p_mesh->mp_index_buffer[0][0];
966  uint32 index2 = p_mesh->mp_index_buffer[0][1];
967  for( uint32 i = 2; i < p_mesh->m_num_indices[0]; ++i )
968  {
969  // Wrap the indices round.
970  index0 = index1;
971  index1 = index2;
972  index2 = p_mesh->mp_index_buffer[0][i];
973 
974  if(( index0 != index1 ) && ( index0 != index2 ) && ( index1 != index2 ))
975  {
976  ++valid_tris;
977 
978  D3DXVECTOR3 *p_vert0 = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index0 ));
979  D3DXVECTOR3 *p_vert1 = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index1 ));
980  D3DXVECTOR3 *p_vert2 = (D3DXVECTOR3*)( p_vert_data + ( p_mesh->m_vertex_stride * index2 ));
981 
982  // Push this tri onto the stack.
983  triSubdivideStack.Push( p_vert0 );
984  triSubdivideStack.Push( p_vert1 );
985  triSubdivideStack.Push( p_vert2 );
986 
987  // Figure the area of this tri.
988  Mth::Vector p( p_vert1->x - p_vert0->x, p_vert1->y - p_vert0->y, p_vert1->z - p_vert0->z );
989  Mth::Vector q( p_vert2->x - p_vert0->x, p_vert2->y - p_vert0->y, p_vert2->z - p_vert0->z );
990  Mth::Vector r(( p[Y] * q[Z] ) - ( q[Y] * p[Z] ), ( p[Z] * q[X] ) - ( q[Z] * p[X] ), ( p[X] * q[Y] ) - ( q[X] * p[Y] ));
991  float area_squared = r.LengthSqr();
992  if( area_squared > shatterAreaTest )
993  {
994  // We will need to subdivide - each subdivision will result in an area one quarter the previous area
995  // (and thusly the square of the area will be one sixteenth the previous area).
996  int num_extra_tris = 1;
997  while( area_squared > shatterAreaTest )
998  {
999  num_extra_tris *= 4;
1000  area_squared *= ( 1.0f / 16.0f );
1001  }
1002 
1003  // This original tri will not be added...
1004  --valid_tris;
1005 
1006  // ...however, the subdivided versions will.
1007  valid_tris += num_extra_tris;
1008  }
1009  }
1010  }
1011 
1012  if( valid_tris == 0 )
1013  {
1014  continue;
1015  }
1016 
1017  // Create a tracking structure for this mesh.
1018  sXboxShatterInstanceDetails *p_details = new sXboxShatterInstanceDetails( valid_tris, p_mesh );
1019  uint8 *p_write_vertex = p_details->mp_vertex_buffer;
1020  uint32 details_index = 0;
1021 
1022  Mth::Vector spread_center = shatterVelocity * -shatterSpreadFactor;
