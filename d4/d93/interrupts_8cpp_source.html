14 /*
15 
16  Mick Notes:
17 
18  All interrupts are set up in the "InitialiseEngine" function of nx_init.cpp
19 
20  There are three types of interrupt covered here
21 
22  1) GIF Interrupt - Used to signal the completion of texture upload to VRAM
23  2) GS Interrupt - Used to signal the completion of GS rendering (drawing polygons)
24  3) VIF Interrupt - NOT USED, could be used like the GS interrupt, but unreliable.
25 
26  There are two types of activity going on in parallel: Upload and Render
27 
28  Whenever one of these finishes, it causes an interrupt.
29 
30  There is no way of knowing in advance what order the tasks will finish in
31  so we must handle all possible cases.
32 
33  At any time, either upload or render might be "Stalled". In this case we are waiting
34  for something before we can start rendering or uploading a new group. Stalls are not
35  good, as the relevent processor is sat idle. We want to minimize stalls as much as
36  possible.
37 
38  Stalls happen when:
39 
40  - The initial render group is waiting for its textures to upload.
41  (?) The renderer is always stalled on the first group
42  - An upload group want to use some VRAM that is used by a group
43  that is currently rendering.
44  - A render group is waiting for it's textures to upload. This could
45  happen if a group renders quicker than the next groups uploads.
46 
47 */

180 int GsHandler(int Cause)
181 {
182  // mask GS interrupts
183  *GS_IMR = PackIMR(1,1,1,1);
184 
185  // reset SIGNAL event bit
186  *GS_CSR = *GS_CSR & ~(uint64)0x30F | 1;
187 
188  // reset signal bit
189  *GS_SIGLBLID &= ~(uint64)1;
190 
191  // maybe it was a false alarm...
192  if (!sGroup::pRenderGroup) // if nothing currently rendering, then we've finished!!
193  goto EXIT;
194 
195  // we've finished rendering a group, so find the next group to render
196  sGroup::pRenderGroup = sGroup::pRenderGroup->pNext;
197  while (sGroup::pRenderGroup && !sGroup::pRenderGroup->Used[!render::Field]) // if exists, but not used
198  {
199  sGroup::pRenderGroup = sGroup::pRenderGroup->pNext; // then skip groups until reach end, or find one that is used
200  }
201 
202  # ifdef __USE_PROFILER__
203  Sys::VUProfiler->PopContext();
204  # endif // __USE_PROFILER__
205 
206 
207  // is there another group to render?
208  if (sGroup::pRenderGroup)
209  {
210  // was there a stalled upload?
211  if (UploadStalled)
212  {
213  // is the next render waiting for the stalled upload?
214  // (i.e., they both are the same group)
215  if (sGroup::pRenderGroup == sGroup::pUploadGroup)
216  {
217  // kick off the stalled upload
218 # ifdef __USE_PROFILER__
219  Sys::DMAProfiler->PushContext( sGroup::pUploadGroup->profile_color );
220 # endif // __USE_PROFILER__
221  START_UPLOAD();
222 
223  // but stall the next render
224  // as this upload group is needed for this render
225  RenderStalled = 2;
226  }
227  else
228  {
229  // kick off the next render, as it's from a previous group
230 # ifdef __USE_PROFILER__
231  Sys::VUProfiler->PushContext( sGroup::pRenderGroup->profile_color );
232 # endif // __USE_PROFILER__
233  START_RENDER();
234 
235  // is the next upload safe?
236  if (NO_VRAM_CONFLICT())
237  {
238  // kick off the next upload
239 # ifdef __USE_PROFILER__
240  Sys::DMAProfiler->PushContext( sGroup::pUploadGroup->profile_color );
241 # endif // __USE_PROFILER__
242  START_UPLOAD();
243  }
244  }
245  }
246 
247  // is there an upload in progress?
248  // (Mick: Does it make sense to also check to see if the upload group is the same as the render group?
249  // or is that always the case, and that's why Mike is not checking it?)
250  else if (sGroup::pUploadGroup)
251  {
252  // stall the next render
253  RenderStalled = 1;
254  }
255 
256  // no uploads left
257  else
258  {
259  // kick off the final render
260 # ifdef __USE_PROFILER__
261  Sys::VUProfiler->PushContext( sGroup::pRenderGroup->profile_color );
262 # endif // __USE_PROFILER__
263  START_RENDER();
264  }
265  }
266 
267 EXIT:
268  // re-enable GS interrupts
269  *GS_IMR = PackIMR(0,1,1,1);
270 
271  ExitHandler();
272  return 0;
273 }
