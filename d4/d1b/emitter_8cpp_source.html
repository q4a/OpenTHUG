313 void CEmitterObject::InitGeometry(Nx::CSector *p_sector)
314 {
315  mp_sector = p_sector;
316 
317  if (!mp_sector)
318  {
319  // No Geometry
320  Dbg_MsgAssert(m_shape_type == CEmitterObject::vSHAPE_SPHERE, ("EmitterObject with shape %d has no geometry.", m_shape_type));
321  return;
322  }
323 
324 #ifdef DEBUG_EMITTER
325  Dbg_Message("Found Geometry for EmitterObject %s at address %x", Script::FindChecksumName(m_name), mp_sector);
326 #endif
327 
328  switch (m_shape_type)
329  {
330  case CEmitterObject::vSHAPE_SPHERE:
331  Dbg_MsgAssert(0, ("EmitterObject with sphere shape has geometry."));
332  break;
333 
334  case CEmitterObject::vSHAPE_BBOX:
335  m_bbox = mp_sector->GetBoundingBox();
336 #ifdef DEBUG_EMITTER
337  Dbg_Message("Bounding box of geometry (%f, %f, %f) - (%f, %f, %f)",
338  m_bbox.GetMin()[X], m_bbox.GetMin()[Y], m_bbox.GetMin()[Z],
339  m_bbox.GetMax()[X], m_bbox.GetMax()[Y], m_bbox.GetMax()[Z]);
340 #endif
341  break;
342 
343  case CEmitterObject::vSHAPE_NON_AA_BBOX:
344  Dbg_MsgAssert(0, ("EmitterObject with non axis aligned bounding box shape not supported yet."));
345  break;
346  }
347 }

354 bool CEmitterObject::CheckInside(const Mth::Vector &object_pos) const
355 {
356  Mth::Vector node_to_object = object_pos - m_pos;
357  float dist_squared = node_to_object.LengthSqr();
358 
359  bool inside = false;
360 
361  switch (m_shape_type)
362  {
363  case vSHAPE_SPHERE:
364  inside = (dist_squared <= m_radius_squared);
365  break;
366 
367  case vSHAPE_BBOX:
368  inside = m_bbox.Within(object_pos);
369 #ifdef DEBUG_EMITTER
370  m_bbox.DebugRender(0xff0000ff);
371 #endif // DEBUG_EMITTER
372  break;
373 
374  default:
375  Dbg_MsgAssert(0, ("Can't handle EmitterObject type %d", m_shape_type));
376  break;
377  }
378 
379  return inside;
380 }

387 Mth::Vector CEmitterObject::GetClosestPoint(const Mth::Vector &object_pos) const
388 {
389  // Return original position if within the object
390  if (CheckInside(object_pos))
391  {
392  return object_pos;
393  }
394 
395  Mth::Vector intersect_point;
396 
397  switch (m_shape_type)
398  {
399  case vSHAPE_SPHERE:
400  {
401  // Same for either within or outside sphere
402  Mth::Vector direction = object_pos - m_pos;
403  direction.Normalize();
404  direction *= m_radius;
405 
406  intersect_point = m_pos + direction;
407  }
408  break;
409 
410  case vSHAPE_BBOX:
411  m_bbox.GetClosestIntersectPoint(object_pos, intersect_point);
412  break;
413 
414  default:
415  Dbg_MsgAssert(0, ("Can't handle EmitterObject type %d", m_shape_type));
416  break;
417  }
418 
419 #ifdef DEBUG_EMITTER
420  Gfx::AddDebugStar(intersect_point);
421 #endif // DEBUG_EMITTER
422 
423  return intersect_point;
424 }
