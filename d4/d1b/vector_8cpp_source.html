91 Vector& Vector::RotateY90( ERot90 angle )
92 {
93  float temp;
94 
95  switch (angle)
96  {
97  case ROT_0:
98  break;
99 
100  case ROT_90:
101  temp = col[X];
102  col[X] = col[Z];
103  col[Z] = -temp;
104  break;
105 
106  case ROT_180:
107  col[X] = -col[X];
108  col[Z] = -col[Z];
109  break;
110 
111  case ROT_270:
112  temp = col[X];
113  col[X] = -col[Z];
114  col[Z] = temp;
115  break;
116 
117  default:
118  Dbg_MsgAssert(0, ("Vector::RotateY90() out of range: %d", angle));
119  break;
120  }
121 
122  return *this;
123 }

141 Vector& Vector::RotateToPlane(const Vector& normal)
142 {
143  // get the length of the vector
144  float length = Length();
145 
146  // Project the vector onto the plane
147  ProjectToPlane(normal);
148 
149  // now we've projected it ontot he plane
150  // we need to handle the case where it dissapears into a point
151  // which would indicate that we were perpendicular to the plane
152  // so need to get an arbitary vector in the plane
153  float projected_length = Length();
154  if (projected_length == 0.0f) // is this a valid comparision?
155  {
156  // Rotate vector through -90 degrees about Y then +90 about X
157  col[X] = -normal[Z];
158  col[Y] = normal[X];
159  col[Z] = -normal[Y];
160  }
161 
162  // get unit vector in this direction
163  Normalize();
164 
165  // multiply by original speed to rotate velocity onto plane
166  *this *= length;
167 
168  col[W] = 0.0f; // clean up W, otherwise multilications will accumelate over time...
169 
170  return *this;
171 }

195 {

257 float GetAngle( const Matrix &currentOrientation, const Vector &desiredHeading, int headingAxis, int rotAxis )
258 {
259 
260 
261  Dbg_MsgAssert( headingAxis == X || headingAxis == Y || headingAxis == Z,( "Improper heading axis." ));
262  Dbg_MsgAssert( rotAxis == X || rotAxis == Y || rotAxis == Z,( "Improper rot axis." ));
263  Dbg_MsgAssert( rotAxis != headingAxis,( "Can't use heading axis as rot axis." ));
264 
265  Mth::Vector tempHeading = desiredHeading;
266  tempHeading.ProjectToPlane( currentOrientation[ rotAxis ] );
267  if ( !tempHeading.Length( ) )
268  {
269  // If our rot axis is along Y, for example, the
270  // target heading is right above us... Can't pick
271  // a rotation!
272  return ( 0.0f );
273  }
274  tempHeading.Normalize( );
275 
276  // while we have these two vectors, find the angle between them...
277  float angCos = DotProduct( currentOrientation[ headingAxis ], tempHeading );
278 
279  // Mick: contrain Dot product to range -1.0f to +1.0f, since FP innacuracies might
280  // make it go outside this range
281  // (previously this was done only on NGC. Not sure why it has started failing now
282  // but it seems logical that it would fail occasionally, so this check is necessary)
283  float ang = angCos;
284  if ( ang > 1.0f ) ang = 1.0f;
285  if ( ang < -1.0f ) ang = -1.0f;
286  float retVal = -acosf( ang );
287 
288  Mth::Vector temp = Mth::CrossProduct( currentOrientation[ headingAxis ], tempHeading );
289 
290  // check to see if the cross product is in the same quatrant as our rot axis...
291  // if so, gots to rotate the other way and shit like that...
292  int whichAxis;
293  float tempMax = temp.GetMaxAxis( &whichAxis );
294  if ( ( tempMax > 0 ) == ( currentOrientation[ rotAxis ][ whichAxis ] > 0 ) )
295  {
296  return ( -retVal );
297  }
298  return ( retVal );
299 }

313 // Returns true if the angle between a and b is greater than the passed Angle.
314 bool AngleBetweenGreaterThan(const Vector& a, const Vector& b, float Angle)
315 {
316  float dot=Mth::DotProduct(a,b);
317  float len=a.Length();
318  if (len<0.5f)
319  {
320  return false;
321  }
322  dot/=len;
323 
324  len=b.Length();
325  if (len<0.5f)
326  {
327  return false;
328  }
329  dot/=len;
330 
331  float TestAngleCosine=cosf(Mth::DegToRad(Angle));
332 
333  if (dot>=0.0f)
334  {
335  return dot<TestAngleCosine;
336  }
337  else
338  {
339  if (TestAngleCosine>=0.0f)
340  {
341  return true;
342  }
343  else
344  {
345  return dot<TestAngleCosine;
346  }
347  }
348 }

350 // Mick:
351 // Given two vectors, v1 and v2 and an Axis vector
352 // return the amount that v1 would have to rotate about the Axis
353 // in order to be pointing in the same direction as v2
354 // when viewed along the Axis
355 //
356 // Typically, "Axis" will be the Y axis of the object (Like the skater)
357 // and v1 will be the Z direction
358 // which we want to rotate toward v2 (about Y)
359 //
360 // retunr value will be in the range -PI .. PI (-180 to 180 degrees)
361 float GetAngleAbout(Vector &v1, Vector &v2, Vector &Axis)
362 {
363  Axis.Normalize();
364  v1.ProjectToPlane(Axis);
365  v2.ProjectToPlane(Axis);
366  v1.Normalize();
367  v2.Normalize();
368  Vector v3 = Mth::CrossProduct(Axis,v1);
369  float v1_dot = Mth::DotProduct(v1,v2);
370  float v3_dot = Mth::DotProduct(v3,v2);
371 
372  float angle = acosf(Mth::Clamp(v1_dot, -1.0f, 1.0f));
373 
374  if (v3_dot < 0.0f)
375  {
376  angle = -angle;
377  }
378  return angle;
379 }
