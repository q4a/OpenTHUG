132 void CCarPhysicsComponent::init_car_skeleton()
133 {
134  // Does some sanity checks
135  Obj::CSkeletonComponent* pSkeletonComponent = (Obj::CSkeletonComponent*)(GetObject()->GetComponent( CRC_SKELETON ));
136  Gfx::CSkeleton* pSkeleton = NULL;
137  if ( pSkeletonComponent )
138  {
139  pSkeleton = pSkeletonComponent->GetSkeleton();
140  }
141 
142  Obj::CModelComponent* pModelComponent = (Obj::CModelComponent*)(GetObject()->GetComponent( CRC_MODEL ));
143  Nx::CModel* pModel = NULL;
144  if ( pModelComponent )
145  {
146  pModel = pModelComponent->GetModel();
147  }
148 
149  Dbg_MsgAssert( pSkeleton, ("Car has no skeleton") );
150  Dbg_MsgAssert( pModel, ( "Model hasn't been set up yet" ) );
151 
152  Dbg_MsgAssert( pModel->GetHierarchy(), ("veh flagged as car has no hierarchy from node %s", Script::FindChecksumName(GetObject()->GetID())));
153  Nx::CHierarchyObject* pHierarchyObjects = pModel->GetHierarchy();
154  Dbg_MsgAssert(( pHierarchyObjects + 0 ),("Missing matrix in car"));
155  Dbg_MsgAssert(( pHierarchyObjects + 2 ),("Missing matrix in car"));
156  Dbg_MsgAssert(( pHierarchyObjects + 4 ),("Missing matrix in car"));
157 
158  // TODO: This 5-bone hierarchy shouldn't be so hardcoded...
159  // We should somehow make this generic so that we
160  // can use different skeletons with the car physics.
161 
162  Mth::Matrix frontTire = ( pHierarchyObjects + 2 )->GetSetupMatrix();
163  frontTire *= ( pHierarchyObjects + 0 )->GetSetupMatrix();
164  Mth::Matrix backTire = ( pHierarchyObjects + 4 )->GetSetupMatrix();
165  backTire *= ( pHierarchyObjects + 0 )->GetSetupMatrix();
166 
167  Mth::Vector dist = frontTire[Mth::POS] - backTire[Mth::POS];
168  m_distanceBetweenTires = fabs( dist.Length() );
169 }
