77 inline void get_rotation_from_key( Gfx::CAnimQKey* p_in, Mth::Quat* pQuat, bool isHiRes )
78 {
79  if ( isHiRes )
80  {
81  (*pQuat)[X] = ((Gfx::CHiResAnimQKey*)p_in)->qx;
82  (*pQuat)[Y] = ((Gfx::CHiResAnimQKey*)p_in)->qy;
83  (*pQuat)[Z] = ((Gfx::CHiResAnimQKey*)p_in)->qz;
84 // (*pQuat)[W] = ((Gfx::CHiResAnimQKey*)p_in)->qw;
85  }
86  else
87  {
88  (*pQuat)[X] = quatDown( ((Gfx::CStandardAnimQKey*)p_in)->qx );
89  (*pQuat)[Y] = quatDown( ((Gfx::CStandardAnimQKey*)p_in)->qy );
90  (*pQuat)[Z] = quatDown( ((Gfx::CStandardAnimQKey*)p_in)->qz );
91 // (*pQuat)[W] = quatDown( ((Gfx::CStandardAnimQKey*)p_in)->qw );
92  }
93 
94  float qx = (*pQuat)[X];
95  float qy = (*pQuat)[Y];
96  float qz = (*pQuat)[Z];
97 
98  // Dave note: added 09/12/02 - a simple check to ensure we don't try to take the square root of a negative
99  // number, which will hose Nan-sensitive platforms later on...
100  float sum = 1.0f - qx * qx - qy * qy - qz * qz;
101  (*pQuat)[W] = sqrtf (( sum < 0.0f ) ? 0.0f : sum );
102 
103  if ( p_in->signBit )
104  {
105  (*pQuat)[W] = -(*pQuat)[W];
106  }
107 }

114 inline void get_translation_from_key( Gfx::CAnimTKey* p_in, Mth::Vector* pVector, bool isHiRes )
115 {
116  if ( isHiRes )
117  {
118  (*pVector)[X] = ((Gfx::CHiResAnimTKey*)p_in)->tx;
119  (*pVector)[Y] = ((Gfx::CHiResAnimTKey*)p_in)->ty;
120  (*pVector)[Z] = ((Gfx::CHiResAnimTKey*)p_in)->tz;
121  (*pVector)[W] = 1.0f;
122  }
123  else
124  {
125  (*pVector)[X] = transDown( ((Gfx::CStandardAnimTKey*)p_in)->tx, 32.0f );
126  (*pVector)[Y] = transDown( ((Gfx::CStandardAnimTKey*)p_in)->ty, 32.0f );
127  (*pVector)[Z] = transDown( ((Gfx::CStandardAnimTKey*)p_in)->tz, 32.0f );
128  (*pVector)[W] = 1.0f;
129  }
130 }

137 inline float get_alpha( float timeStamp1, float timeStamp2, float time )

139  Dbg_MsgAssert(timeStamp1 <= time && timeStamp2 >= time, ( "%f should be within [%f %f]", time, timeStamp1, timeStamp2 ));

149 inline void interpolate_q_frame(Mth::Quat* p_out, Gfx::CAnimQKey* p_in1, Gfx::CAnimQKey* p_in2, float alpha, bool isHiRes)
150 {
151  Dbg_Assert(p_out);
152  Dbg_Assert(p_in1);
153  Dbg_Assert(p_in2);
154 
155  Mth::Quat qIn1;
156  get_rotation_from_key( p_in1, &qIn1, isHiRes );
157 
158  if ( alpha == 0.0f )
159  {
160  // don't need to slerp, because it's the start time
161  *p_out = qIn1;
162  return;
163  }
164 
165  Mth::Quat qIn2;
166  get_rotation_from_key( p_in2, &qIn2, isHiRes );
167 
168  if ( alpha == 1.0f )
169  {
170  // don't need to slerp, because it's the end time
171  *p_out = qIn2;
172  return;
173  }
174 
175  // fast slerp, stolen from game developer magazine
176  *p_out = Mth::FastSlerp( qIn1, qIn2, alpha );
177 }

184 inline void interpolate_t_frame(Mth::Vector* p_out, Gfx::CAnimTKey* p_in1, Gfx::CAnimTKey* p_in2, float alpha, bool isHiRes)
185 {
186  Dbg_Assert(p_out);
187  Dbg_Assert(p_in1);
188  Dbg_Assert(p_in2);
189 
190  // INTERPOLATE T-COMPONENT
191  Mth::Vector tIn1;
192  get_translation_from_key( p_in1, &tIn1, isHiRes );
193 
194  if ( alpha == 0.0f )
195  {
196  // don't need to lerp, because it's the start time
197  *p_out = tIn1;
198  return;
199  }
200 
201  Mth::Vector tIn2;
202  get_translation_from_key( p_in2, &tIn2, isHiRes );
203 
204  if ( alpha == 1.0f )
205  {
206  // don't need to slerp, because it's the end time
207  *p_out = tIn2;
208  return;
209  }
210 
211  /* Linearly interpolate positions */
212  *p_out = Mth::Lerp( tIn1, tIn2, alpha );
213 }
