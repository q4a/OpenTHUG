79 void CSkaterEndRunComponent::Update()
80 {
81  if (!Mdl::Skate::Instance()->IsMultiplayerGame())
82  {
83  Suspend(true);
84  return;
85  }
86 
87  if( static_cast< CSkater* >(GetObject())->IsLocalClient())
88  {
89  if (m_flags.Test(FINISHED_END_OF_RUN) && !GameNet::Manager::Instance()->HaveSentEndOfRunMessage())
90  {
91  GameNet::Manager::Instance()->SendEndOfRunMessage();
92  }
93  }
94 }

101 CBaseComponent::EMemberFunctionResult CSkaterEndRunComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
102 {
103  switch ( Checksum )
104  {
105  // @script | EndOfRunDone |
106  case CRCC(0x4c58771e, "EndOfRunDone"):
107  m_flags.Set(FINISHED_END_OF_RUN);
108  m_flags.Clear(IS_ENDING_RUN);
109  break;
110 
111  case CRCC(0xb866bf9b, "RunStarted"):
112  m_flags.Clear(STARTED_END_OF_RUN);
113  m_flags.Clear( STARTED_GOAL_END_OF_RUN );
114  m_flags.Clear(FINISHED_END_OF_RUN);
115  m_flags.Clear( FINISHED_GOAL_END_OF_RUN );
116  m_flags.Clear(IS_ENDING_RUN);
117  m_flags.Clear(IS_ENDING_GOAL);
118  break;
119 
120  case CRCC(0x962dea01, "EndOfRunStarted"):
121  m_flags.Set(STARTED_END_OF_RUN);
122  break;
123 
124  case CRCC(0xafc6a7,"Goal_EndOfRunStarted"):
125  m_flags.Set(STARTED_GOAL_END_OF_RUN);
126  break;
127 
128  // @script | Goal_EndOfRunDone |
129  case CRCC(0x69a9e37f, "Goal_EndOfRunDone"):
130  m_flags.Set(FINISHED_GOAL_END_OF_RUN);
131  m_flags.Clear(IS_ENDING_GOAL);
132  break;
133 
134  case CRCC(0x95bdcfcd, "IsInEndOfRun"):
135  return (m_flags.Test(STARTED_END_OF_RUN) && !m_flags.Test(FINISHED_END_OF_RUN)) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
136 
137  default:
138  return CBaseComponent::MF_NOT_EXECUTED;
139  }
140  return CBaseComponent::MF_TRUE;
141 }

169 void CSkaterEndRunComponent::EndRun ( bool force_end )
170 {
171  if (force_end || Mdl::Skate::Instance()->GetGameMode()->ShouldStopAtZero())
172  {
173  if (!m_flags.Test(IS_ENDING_RUN) && !m_flags.Test(FINISHED_END_OF_RUN))
174  {
175  //DumpUnwindStack( 20, 0 );
176  m_flags.Set(IS_ENDING_RUN);
177  if (static_cast< CSkater* >(GetObject())->IsLocalClient())
178  {
179  Script::RunScript(CRCD(0xf4ce3e97, "ForceEndOfRun"), NULL, GetObject());
180  }
181  }
182  }
183  else
184  {
185  //DumpUnwindStack( 20, 0 );
186  FLAGEXCEPTION(CRCD(0x822e13a9, "RunHasEnded"));
187  m_flags.Clear(FINISHED_END_OF_RUN);
188  }
189 }
