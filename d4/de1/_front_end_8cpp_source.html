210 // This function JUST pauses the game
211 // this mainly consists of pausing all the tasks (for objects, cars, peds), but
212 // first it pauses the skate module (for music), and the sio_manager (pad vibration)
213 //
214 // The pausing of stuff is fairly comvoluted, here we:
215 //
216 // Pause the PCM music
217 // pause the pad vibration
218 // pause all messages, but allow future messages to be launched
219 // pause all spawned scripts, apart from the current one, and future spawned scritps
220 //
221 // pause the skater, but still allow the reading of the pad from script
222 //
223 // Overall, a rather mixed bag of things, which might be best refactored
224 // but should do for now.....

228 void FrontEnd::PauseGame(bool paused)
229 {
230 
231  // if already in the correct "paused" state, then just return
232  if (paused == m_paused)
233  {
234  return;
235  }
236 
237  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
238  if (paused)
239  {
240  // make game be paused
241  skate_mod->Pause( );
242 
243  // Ken: Pause the controller vibrations.
244  SIO::Manager * sio_man = SIO::Manager::Instance();
245  sio_man->Pause();
246  }
247  else
248  {
249  // make game be unpaused
250  // Ken: Un-pause the controller vibrations.
251  SIO::Manager * sio_man = SIO::Manager::Instance();
252  sio_man->UnPause();
253  }
254 
255  // Pause currently spawned scripts (any script with PauseGame in it will unpause itself)
256  Script::PauseSpawnedScripts(paused);
257 
258  // Pause or unpause all the skaters
259  uint32 NumSkaters = skate_mod->GetNumSkaters();
260 
261  for (uint32 i = 0; i < NumSkaters; ++i)
262  {
263  Obj::CSkater *pSkater = skate_mod->GetSkater(i);
264  if (pSkater) // Hmm, assert instead?
265  {
266  if (paused)
267  {
268  pSkater->Pause();
269  }
270  else
271  {
272  pSkater->UnPause();
273  }
274  }
275  }
276 
277  // Pause all composite objects
278  Obj::CCompositeObjectManager::Instance()->Pause(paused);
279 
280 
281  // pause (or unpause) any screen elements in process of morphing
282  Front::CScreenElementManager* p_screen_element_manager = Front::CScreenElementManager::Instance();
283  p_screen_element_manager->SetPausedState(paused);
284 
285  m_paused = paused;
286 
287 
288  if (m_paused)
289  {
290  // make game be paused
291  Mlp::Manager * mlp_manager = Mlp::Manager::Instance();
292  mlp_manager->SetLogicMask(Mlp::Manager::mGAME_OBJECTS | Mlp::Manager::mGFX_MANAGER);
293  Tmr::StoreTimerInfo( );
294  }
295  else
296  {
297  // make game be unpaused
298  Mlp::Manager * mlp_manager = Mlp::Manager::Instance();
299  mlp_manager->SetLogicMask(Mlp::Manager::mNONE);
300 // rw_viewer->DisableMainLogic(false);
301  // unpause music and soundfx:
302  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
303  skate_mod->Pause( false );
304  Tmr::RecallTimerInfo( );
305  }
306 }

312 {

314  const char *p_array_name = "xbox";

327  {
328  "left_trigger_full",

346  };
347  #else
348  #ifdef __PLAT_NGC__
349  const char *p_button_names[] =
350  {
351  "left_trigger_full",
352  "right_trigger_full",
353  "left_trigger_half",
354  "right_trigger_half",
355  "y",
356  "x",
357  "a",

371  #else
372  const char *p_button_names[] =
373  {
374  "left_trigger2",
375  "right_trigger2",
376  "left_trigger1",
377  "right_trigger1",
378  "triangle",
379  "circle",
380  "x",
381  "square",
382  "select",
383  "left_stick_button",
384  "right_stick_button"
385  };
386  #endif
387  #endif
388 
389  // Cleaner way of allowing varying button names array per platform.
390  int num_mapped_buttons = sizeof( p_button_names ) / sizeof( char* );
391 
392  uint32 null_crc = Script::GenerateCRC("null");
393 
394  /*
395  Indices:
396 
397  i: into script array
398  j: into button name array
399  x: into map array
400  */
401 
402  Script::CArray *p_map;
403  if (!pParams->GetArray(p_array_name, &p_map))
404  return;
405 
406  for (uint i = 0; i < p_map->GetSize(); i++)
407  {
408  Script::CArray *p_entry = p_map->GetArray(i);
409 
410  uint32 button_crc = p_entry->GetChecksum(0);
411  uint32 event_crc = p_entry->GetChecksum(1);
412 
413  for (uint j = 0; j < (uint) num_mapped_buttons; j++)
414  {
415  if (button_crc == Script::GenerateCRC(p_button_names[j]) || button_crc == null_crc)
416  {
417  #ifdef __NOPT_ASSERT__
418  bool removing_entries = (button_crc == null_crc || event_crc == null_crc);
419  #endif
420  int x = 0;
421  for (; x < MAX_BUTTON_EVENT_MAP_ENTRIES; x++)
422  {
423  if (button_crc == null_crc && m_digital_button_event_map[x].mEventType == event_crc)
424  {
425  // in this case, I am removing existing entry (or entries)
426  m_digital_button_event_map[x].mEventType = DEAD_ENTRY;
427  }
428  else if (event_crc == null_crc && m_digital_button_event_map[x].mDigitalButtonIndex == j)
429  {
430  // in this case, I am removing existing entry (or entries)
431  m_digital_button_event_map[x].mEventType = DEAD_ENTRY;
432  }
433  else if (m_digital_button_event_map[x].mEventType == DEAD_ENTRY)
434  {
435  // in this case, I am adding a new entry, so need empty slot
436  m_digital_button_event_map[x].mDigitalButtonIndex = j;
437  m_digital_button_event_map[x].mEventType = event_crc;
438  break;
439  }
440  }
441  Dbg_MsgAssert(x < MAX_BUTTON_EVENT_MAP_ENTRIES || removing_entries, ("out of entries, increase size of MAX_BUTTON_EVENT_MAP_ENTRIES"));
442 
443  break;
444  }
445  }
446  }

588 void FrontEnd::s_handle_keyboard_code( const Tsk::Task< FrontEnd >& task )
589 {
590  int i, num_chars;
591  char makes[256];
592  FrontEnd& mdl = task.GetData();
593 
594  num_chars = SIO::KeyboardRead( makes );
595 
596  for( i = 0; i < num_chars; i++ )
597  {
598  if((( makes[i] >= 32 ) && ( makes[i] <= 126 ) ) || ( makes[i] == SIO::vKB_ENTER ) || ( makes[i] == SIO::vKB_BACKSPACE ))
599  {
600  Script::CStruct* pParams;
601 
602  pParams = new Script::CStruct;
603  if( makes[i] == SIO::vKB_ENTER )
604  {
605  pParams->AddChecksum( NONAME, Script::GenerateCRC( "got_enter" ));
606  }
607  else if( makes[i] == SIO::vKB_BACKSPACE )
608  {
609  pParams->AddChecksum( NONAME, Script::GenerateCRC( "got_backspace" ));
610  }
611  else
612  {
613  char text_string[2];
614 
615  text_string[0] = makes[i];
616  text_string[1] = '\0';
617  pParams->AddString( "text", text_string );
618  }
619 
620  pParams->AddInteger( "max_length", mdl.m_max_kb_string_length );
621  Script::RunScript( "handle_keyboard_input", pParams );
622  delete pParams;
623  }
624  }
625 
626  // Don't let anyone else handle the keypresses
627  SIO::KeyboardClear();
628 }

726 uint32 FrontEnd::grab_timed_event_type(uint mask, int index)
727 {
728  char makes[256];
729  int i, num_chars;
730  uint32 event_type = 0;
731 
732  /*
733  We check for these types of pad presses first, because of the timing element
734  */
735  if (mask & Inp::Data::mD_LEFT)
736  event_type = Obj::CEvent::TYPE_PAD_LEFT;
737  else if (mask & Inp::Data::mD_RIGHT)
738  event_type = Obj::CEvent::TYPE_PAD_RIGHT;
739  else if (mask & Inp::Data::mD_UP)
740  event_type = Obj::CEvent::TYPE_PAD_UP;
741  else if (mask & Inp::Data::mD_DOWN)
742  event_type = Obj::CEvent::TYPE_PAD_DOWN;
743 
744  // Let the pad take precedence over the keyboard
745  if(( index == 0 ) && ( event_type == 0 ))
746  {
747  // First check for keyboard events, if there were any
748  num_chars = SIO::KeyboardRead( makes );
749 
750  if( num_chars > 0 )
751  {
752  for( i = 0; i < num_chars; i++ )
753  {
754  if( makes[i] == SIO::vKB_LEFT )
755  {
756  event_type = Obj::CEvent::TYPE_PAD_LEFT;
757  break;
758  }
759  else if( makes[i] == SIO::vKB_RIGHT )
760  {
761  event_type = Obj::CEvent::TYPE_PAD_RIGHT;
762  break;
763  }
764  else if( makes[i] == SIO::vKB_UP )
765  {
766  event_type = Obj::CEvent::TYPE_PAD_UP;
767  break;
768  }
769  else if( makes[i] == SIO::vKB_DOWN )
770  {
771  event_type = Obj::CEvent::TYPE_PAD_DOWN;
772  break;
773  }
774  }
775  }
776  }
777 
778  if (m_pad_info[index].mState == NOT_DOWN)
779  {
780  if (event_type)
781  {
782  // The directional pad was freshly pressed, so we will send an event.
783  // Begin next countdown until the pad input "counts" again.
784  m_pad_info[index].mAutorepeatCountdown = m_auto_repeat_time[0];
785  m_pad_info[index].mState = SLOW_REPEAT;
786  }
787  }
788  else if (event_type)
789  {
790  // The directional pad was held down last frame (and this one)
791 
792  // update countdown time
793  Tmr::Time frame_time = (Tmr::Time) (Tmr::FrameLength() * Tmr::vRESOLUTION);
794  if (frame_time < m_pad_info[index].mAutorepeatCountdown)
795  m_pad_info[index].mAutorepeatCountdown -= frame_time;
796  else
797  m_pad_info[index].mAutorepeatCountdown = 0;
798 
799  if (m_pad_info[index].mAutorepeatCountdown == 0)
800  {
801  // countdown finished, time to fire event, reset countdown time
802  if (m_pad_info[index].mState == SLOW_REPEAT)
803  m_pad_info[index].mState = FAST_REPEAT;
804  m_pad_info[index].mAutorepeatCountdown = m_auto_repeat_time[1];
805  }
806  else
807  // don't fire pad event, countdown hasn't finished
808  event_type = 0;
809  }
810  else
811  {
812  m_pad_info[index].mState = NOT_DOWN;
813  }
814 
815  if (m_using_auto_input && index == 0)
816  {
817  event_type = 0;
818 
819  Script::CScript *p_script = Script::GetScriptWithUniqueId(m_auto_input_script_id);
820  if (p_script)
821  {
822  Script::UnpauseSpawnedScript(p_script);
823  }
824  else
825  {
826  m_using_auto_input = false;
827  }
828  }
829 
830  return event_type;
831 }

837 {
