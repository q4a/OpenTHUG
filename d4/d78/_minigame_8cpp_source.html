242 bool CMinigame::AwardCash( int amount )
243 {
244  if ( m_cashLimit <= 0 )
245  {
246  Script::RunScript( "minigame_cash_depleted" );
247  return false;
248  }
249 
250  // clamp it

252  amount = m_cashLimit;

254  CGoalManager* pGoalManager = GetGoalManager();
255  Dbg_Assert( pGoalManager );
256  if ( pGoalManager )
257  {
258  pGoalManager->AddCash( amount );
259  m_cashLimit -= amount;
260  Script::CStruct* pScriptParams = new Script::CStruct();
261  pScriptParams->AddInteger( "amount", amount );
262  Script::RunScript( "minigame_got_cash", pScriptParams );
263  delete pScriptParams;
264  return true;
265  }
266  return false;
