401 void CProximNode::SetActive(bool active)
402 {
403  // Check for changes
404  if (m_active != active)
405  {
406  if (m_active)
407  {
408  // Turning off node, so treat like were going outside
409  if ( m_proxim_object )

413  printf ("----- SETTING ProximNode INACTIVE, killing %s running script %s \n",Script::FindChecksumName(m_proxim_object),Script::FindChecksumName(m_script));
414 #endif
415  int object_id = 0;
416  while (m_active_objects)
417  {
418  if (m_active_objects & 0x1) // Check if this ID is turned on

423  p_obj->MarkAsDead();
424  }
425  }
426 
427  // Advance ID
428  m_active_objects = m_active_objects >> 1;
429  object_id++;
430  }
431  }
432  else
433  {
434 #ifdef DEBUG_PROXIM
435  printf ("----- SETTING ProximNode INACTIVE, running script %s \n",Script::FindChecksumName(m_script));
436 #endif
437  CProximManager* proxim_manager = Mdl::Skate::Instance()->mpProximManager;
438 
439  if (proxim_manager)
440  {
441  int object_id = 0;
442  while (m_active_scripts)
443  {
444  if (m_active_scripts & 0x1) // Check if this ID is turned on
445  {
446  Obj::CObject *p_obj = Obj::CCompositeObjectManager::Instance()->GetObjectByID(object_id);
447  Dbg_MsgAssert(p_obj, ("Can't find object to play script on"));
448  proxim_manager->m_inside_flag = false;
449 
450  proxim_manager->m_inside_flag_valid = true;
451  p_obj->SpawnAndRunScript( m_script, m_node );
452  proxim_manager->m_inside_flag_valid = false;
453  }
454 
455  // Advance ID
456  m_active_scripts = m_active_scripts >> 1;
457  object_id++;
458  }
459  }
460  }
461  }
462 
463  // Make everything outside
464  m_outside_flags = 0xffffffff;
465 
466  m_active = active;
467  }

533 bool CProximNode::CheckInside(const Mth::Vector &object_pos) const
534 {
535  Mth::Vector node_to_object = object_pos - m_pos;
536  float dist_squared = node_to_object.LengthSqr();
537 
538  bool inside = false;
539 
540  switch (m_shape)
541  {
542  case vSHAPE_SPHERE:
543  inside = (dist_squared <= m_radius_squared);
544  break;
545 
546  case vSHAPE_BBOX:
547  inside = m_bbox.Within(object_pos);
548 #ifdef DEBUG_PROXIM
549  m_bbox.DebugRender(0xff0000ff);
550  {
551  Mth::Vector closest_point;
552  m_bbox.GetClosestIntersectPoint(object_pos, closest_point);
553 // Gfx::AddDebugStar(closest_point);
554  }
555 #endif // DEBUG_PROXIM
556  break;
557 
558  default:
559  Dbg_MsgAssert(0, ("Can't handle ProximNode type %d", m_shape));
560  }
561 
562  return inside;
563 }
