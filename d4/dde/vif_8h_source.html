455 // Patch up the NUM field of an UNPACK generated using BeginUNPACK().
456 // Alignment is necessary because some data formats won't fill out to a 4-byte boundary.

458 inline void vif::EndUNPACK(void)
459 {
460  UnpackSize = ((dma::pLoc - pVifCode - 4) << 3) / (Dimension * BitLength);
461  dma::Align(0,4);
462  if (UnpackSize==0)
463  dma::pLoc -= 4; // no data, rewind over vifcode
464  else if (UnpackSize < 256)
465  pVifCode[2] = UnpackSize; // normal usage
466  else if (UnpackSize == 256)
