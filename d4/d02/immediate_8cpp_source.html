27 uint64 CImmediateMode::sGetTextureBlend( uint32 blend_checksum, int fix )
28 {
29  uint64 rv = PackALPHA(0,0,0,0,0);
30  switch ( blend_checksum )
31  {
32  case 0x54628ed7: // Blend
33  rv = PackALPHA(0,1,0,1,0);
34  break;
35  case 0x02e58c18: // Add
36  rv = PackALPHA(0,2,0,1,0);
37  break;
38  case 0xa7fd7d23: // Sub
39  case 0xdea7e576: // Subtract
40  rv = PackALPHA(2,0,0,1,0);
41  break;
42  case 0x40f44b8a: // Modulate
43  rv = PackALPHA(1,2,0,2,0);
44  break;
45  case 0x68e77f40: // Brighten
46  rv = PackALPHA(1,2,0,1,0);
47  break;
48  case 0x18b98905: // FixBlend
49  rv = PackALPHA(0,1,2,1,fix);
50  break;
51  case 0xa86285a1: // FixAdd
52  rv = PackALPHA(0,2,2,1,fix);
53  break;
54  case 0x0d7a749a: // FixSub
55  case 0x0eea99ff: // FixSubtract
56  rv = PackALPHA(2,0,2,1,fix);
57  break;
58  case 0x90b93703: // FixModulate
59  rv = PackALPHA(1,2,2,2,fix);
60  break;
61  case 0xb8aa03c9: // FixBrighten
62  rv = PackALPHA(1,2,2,1,fix);
63  break;
64  case 0x515e298e: // Diffuse
65  case 0x806fff30: // None
66  rv = PackALPHA(0,0,0,0,0);
67  break;
68  default:
69  Dbg_MsgAssert(0,("Illegal blend mode specified. Please use (fix)blend/add/sub/modulate/brighten or diffuse/none."));
70  break;
71  }
72  return rv;
73 }

75 void CImmediateMode::sViewportInit()
76 {
77  // start a cnt tag
78  dma::BeginTag(dma::cnt, 0);
79  vif::STROW((int)render::RowRegI[0], (int)render::RowRegI[1], (int)render::RowRegI[2], (int)render::RowRegI[3]);
80 
81  // VU context, uploading the view transform data
82  vu1::BeginPrim(ABS, VU1_ADDR(L_VF09)); // Begin packed register upload to VF10
83  vu1::StoreVec(*(Vec *)&render::AltFrustum); // VF09
84  vu1::StoreVec(*(Vec *)&render::InverseViewportScale); // VF10
85  vu1::StoreVec(*(Vec *)&render::InverseViewportOffset); // VF11
86 // vu1::StoreMat(*(Mat *)&render::AdjustedWorldToFrustum); // VF12-15
87  vu1::StoreMat(*(Mat *)&render::AdjustedWorldToViewport);// VF12-15
88  vu1::EndPrim(0); // End upload
89 
90  // GS context
91  gs::BeginPrim(ABS,0,0);
92  gs::Reg1(gs::CLAMP_1, PackCLAMP(CLAMP,CLAMP,0,0,0,0)); // may want to make REPEAT an option
93  gs::Reg1(gs::FRAME_1, PackFRAME(FRAME_START,HRES/64,PSMCT32,0xFF000000));
94  gs::Reg1(gs::SCISSOR_1, render::reg_SCISSOR);
95  gs::Reg1(gs::XYOFFSET_1, render::reg_XYOFFSET);
96  gs::Reg1(gs::TEST_1, PackTEST(0,0,0,0,0,0,1,ZGEQUAL));
97  gs::Reg1(gs::COLCLAMP, PackCOLCLAMP(1));
98  gs::Reg1(gs::ZBUF_1, PackZBUF(ZBUFFER_START,PSMZ24,1));
99  gs::EndPrim(0);
100 
101  // end the dma tag
102  dma::EndTag();
103 }
