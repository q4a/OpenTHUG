68 // Axis-Aligned BSP tree node

70 class CCollBSPNode

72 public:
73  CCollBSPNode();
74 
75  // These functions allow us to have "virtual" functionality with an instance. We can safely
76  // cast a pointer to CCollBSPLeaf if isLeaf() returns true.
77  uint8 GetSplitAxis() const { return m_leaf.m_split_axis & 0x3; }
78  void SetSplitAxis(int axis);
79  bool IsNode() const { return (GetSplitAxis() != 3); }
80  bool IsLeaf() const { return (GetSplitAxis() == 3); }
81 
82  CCollBSPNode * GetLessBranch() { Dbg_Assert(IsNode()); return m_node.m_children.GetLessBranch(); }
83  CCollBSPNode * GetGreaterBranch() { Dbg_Assert(IsNode()); return m_node.m_children.GetGreaterBranch(); }
84 
85  FaceIndex * GetFaceIndexArray() { Dbg_Assert(IsLeaf()); return m_leaf.mp_face_idx_array; }
86 
87  int GetSplitPoint() const { return m_node.m_split_point >> NUM_AXIS_BITS; }
88  void SetSplitPoint(int ipoint) { m_node.m_split_point = (ipoint << NUM_AXIS_BITS) | GetSplitAxis(); }
89  float GetFSplitPoint() const { return ((float) GetSplitPoint()) * COLLISION_RECIPROCAL_SUB_INCH_PRECISION; }
90  void SetFSplitPoint(float point) { SetSplitPoint(point * COLLISION_SUB_INCH_PRECISION); }
91 
92 private:
94  // Basically, a class for a pointer that uses the lowest two bits for flags.
95  // It also assumes that the pointer points to two consecutive nodes: the left
96  // node and the right node.
97  //
98  class CCollBSPChildren
99  {
100  public:
101 
102  void Init() { m_left_child_and_flags = 0; }
103 
104  // Actual branches
105  CCollBSPNode * GetLeftBranch() const { return GetBasePointer(); }
106  CCollBSPNode * GetRightBranch() const { return GetBasePointer() + 1; }
107 
108  // Figures out which branch is which
109  CCollBSPNode * GetLessBranch() const { return IsLeftGreater() ? GetRightBranch() : GetLeftBranch(); }
110  CCollBSPNode * GetGreaterBranch() const { return IsLeftGreater() ? GetLeftBranch() : GetRightBranch(); }
111 
112  // These two are needed for the cloning function
113  CCollBSPNode * GetBasePointer() const { return (CCollBSPNode *) (m_left_child_and_flags & ~0x3); }
114  void SetBasePointer(CCollBSPNode *p_base) { m_left_child_and_flags = ((uint32) p_base) | (m_left_child_and_flags & 0x3); }
115 
116  bool IsLeftGreater() const { return m_left_child_and_flags & mLEFT_IS_GREATER; }
117  void SetLeftGreater(bool greater);
118 
119  private:
120 
121  // Constants
122  enum
123  {
124  mLEFT_IS_GREATER = 0x01, // Indicates that the left branch is the greater branch
125  };
126 
127  uint32 m_left_child_and_flags; // points to left branch, right branch one node over
128  };

548 inline Mth::Vector CCollObjTriData::GetRawVertexPos(int vert_idx) const
549 {
550 #if !defined(FIXED_POINT_VERTICES)
551 
552 #ifdef __PLAT_NGC__
553  Mth::Vector v;
554 
555  if ( mp_cloned_vert_pos )
556  {
557  v[X] = mp_cloned_vert_pos[vert_idx].x;
558  v[Y] = mp_cloned_vert_pos[vert_idx].y;
559  v[Z] = mp_cloned_vert_pos[vert_idx].z;
560  v[W] = 0.0f;
561  }
562  else
563  {
564  v[X] = mp_raw_vert_pos[vert_idx].x;
565  v[Y] = mp_raw_vert_pos[vert_idx].y;
566  v[Z] = mp_raw_vert_pos[vert_idx].z;
567  v[W] = 0.0f;
568  }
569 
570  return v;
571 #else
572  return mp_vert_pos[vert_idx];
573 #endif // __PLAT_NGC__
574 
575 #else
576 
577  if (m_use_fixed_verts)
578  {
579  Mth::Vector pos(m_bbox.GetMin());
580 
581  pos[X] += ((float) mp_fixed_vert[vert_idx].m_pos[X]) * COLLISION_RECIPROCAL_SUB_INCH_PRECISION;
582  pos[Y] += ((float) mp_fixed_vert[vert_idx].m_pos[Y]) * COLLISION_RECIPROCAL_SUB_INCH_PRECISION;
583  pos[Z] += ((float) mp_fixed_vert[vert_idx].m_pos[Z]) * COLLISION_RECIPROCAL_SUB_INCH_PRECISION;
584  pos[W] = 0.0f;
585 
586  return pos;
587  }
588  else
589  {
590  Mth::Vector pos(Mth::Vector::NO_INIT);
591  pos[X] = mp_float_vert[vert_idx].m_pos[X];
592  pos[Y] = mp_float_vert[vert_idx].m_pos[Y];
593  pos[Z] = mp_float_vert[vert_idx].m_pos[Z];
594  pos[W] = 0.0f;
595  return pos;
596  }
597 #endif
598 }

629  pos[Z] += ((float) mp_fixed_vert[vert_idx].m_pos[Z]) * COLLISION_RECIPROCAL_SUB_INCH_PRECISION;
