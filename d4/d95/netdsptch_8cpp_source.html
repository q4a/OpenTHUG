154 int Dispatcher::DispatchMsgHandlers( Conn *conn, int flags )
155 {
156  MsgHandlerContext msg_context;
157  char msg_data[MAX_STREAM_LENGTH];
158  int result;
159  char *data;
160 
161  Dbg_Assert( conn != NULL );
162 
163  msg_context.m_Conn = conn;
164  msg_context.m_App = conn->m_app;
165  msg_context.m_PacketFlags = flags;
166  msg_context.m_Msg = msg_data;
167  msg_context.m_MsgLength = 0;
168 
169  data = conn->m_read_buffer;
170 
171  // loop through received data and handle messages
172  while( data < conn->m_read_ptr )
173  {
174  bool size_known;
175 
176  // *** IMPORTANT. Because of alignment issues, I now copy the data from the byte stream
177  // into an aligned message buffer. Indexing into the buffer and casting to Msg's caused
178  // load errors
179  memcpy( &msg_context.m_MsgId, data, sizeof( unsigned char ));
180  data += sizeof( unsigned char );
181 
182  // Check to see if the size of this message type is stored after the msg id
183  size_known = (( conn->m_app->GetManager()->GetMessageFlags( msg_context.m_MsgId ) & mMSG_SIZE_UNKNOWN ) == 0 );
184  //if( conn->IsRemote())
185  //{
186  //Dbg_Printf( "Got message %s with size_known of %d\n", conn->m_app->GetManager()->GetMessageName(msg_context.m_MsgId ), size_known);
187  //}
188 
189  if( size_known )
190  {
191  unsigned short temp;
192  memcpy( &temp, data, sizeof( unsigned short ));
193  msg_context.m_MsgLength = (unsigned long)temp;
194 
195  data += sizeof( unsigned short );
196  }
197 
198  if( msg_context.m_MsgId == MSG_ID_SEQUENCED )
199  {
200  unsigned char embedded_msg_id;
201  unsigned short embedded_msg_length;
202 
203  Dbg_Assert( size_known );
204 
205  memcpy( &embedded_msg_id, data + 5, sizeof( unsigned char ));
206  embedded_msg_length = msg_context.m_MsgLength - Manager::vMSG_SEQ_HEADER_LENGTH;
207 
208  conn->GetInboundMetrics()->AddMessage( embedded_msg_id, msg_context.m_MsgLength );
209 #ifdef NET_PRINT_MESSAGES
210  if( conn->IsRemote())
211  {
212  unsigned char group_id;
213  unsigned int seq_id;
214 
215  memcpy( &group_id, data, sizeof( unsigned char ));
216  memcpy( &seq_id, data + 1, sizeof( unsigned int ));
217  Dbg_Printf( "(%d) Dispatching Sequenced Message (%d) Length (%d) Emb. Length (%d) : %s. Seq: %d Group: %d\n",
218  conn->m_app->m_FrameCounter,
219  embedded_msg_id,
220  msg_context.m_MsgLength,
221  embedded_msg_length,
222  conn->m_app->GetManager()->GetMessageName( embedded_msg_id ),
223  seq_id,
224  group_id );
225  }
226 #endif NET_PRINT_MESSAGES
227  }
228 
229  // If we know the size, just copy the message into the aligned buffer
230  // otherwise, copy the maximum size of a message into the buffer and have the
231  // handler tell us the size of the message by changint the context's m_MsgLength member
232  if( size_known )
233  {
234  if( msg_context.m_MsgLength > 0 )
235  {
236  memcpy( msg_context.m_Msg, data, msg_context.m_MsgLength );
237  }
238  }
239  else
240  {
241  memcpy( msg_context.m_Msg, data, Manager::vMAX_PACKET_SIZE );
242  }
243 
244  if( conn->IsForeign())
245  {
246  msg_context.m_PacketFlags |= mHANDLE_FOREIGN;
247  }
248  result = DispatchMessage( &msg_context );
249  if( ( result == HANDLER_ERROR ) ||
250  ( result == HANDLER_HALT ))
251  {
252  // output error indicating mal-formmated data
253  Dbg_Printf( "Stream processing halted on message %d\n", msg_context.m_MsgId );
254  conn->m_read_ptr = conn->m_read_buffer;
255  return HANDLER_HALT;
256  }
257 
258  // By this time we know the message size, whether it was explicit or implicit
259  // so we can do the metrics calculation and the debug printout
260  if( msg_context.m_MsgId != MSG_ID_SEQUENCED )
261  {
262  conn->GetInboundMetrics()->AddMessage( msg_context.m_MsgId, msg_context.m_MsgLength );
263 #ifdef NET_PRINT_MESSAGES
264  /*if( conn->IsRemote())
265  {
266  Dbg_Printf( "(%d) Dispatching Message (%d) Length (%d) : %s\n",
267  conn->m_app->m_FrameCounter,
268  msg_context.m_MsgId,
269  msg_context.m_MsgLength,
270  conn->m_app->GetManager()->GetMessageName( msg_context.m_MsgId ));
271  }*/
272 #endif
273  }
274 
275  data += msg_context.m_MsgLength;
276  }
277 
278  conn->m_read_ptr = conn->m_read_buffer;
279  return HANDLER_CONTINUE;
280 }

283 /* Dispatch a message to its appropriate handler. Stop if higher */
284 /* priority handlers say to */

287 int Dispatcher::DispatchMessage( MsgHandlerContext *context )
288 {
289  Lst::Search< MsgHandler > sh;
290  MsgHandler *handler, *next;
291  int result;
292 
293 
294 
295  Dbg_Assert( context );
296 
297  // If this is a message we want to handle, call the handler
298  for( handler = sh.FirstItem( m_handler_list[context->m_MsgId] );
299  handler; handler = next )
300  {
301  next = sh.NextItem();
302 
303  if(( handler->m_flags & context->m_PacketFlags ) != context->m_PacketFlags )
304  {
305  Dbg_Printf( "*** Conn %d packetflag mismatch on id %d : 0x%x 0x%x\n", context->m_Conn->GetHandle(), context->m_MsgId, handler->m_flags, context->m_PacketFlags );
306  continue;
307  }
308 
309  context->m_Data = handler->m_data;
310  result = handler->m_code( context );
311  // Check the result from the handler
312  if( ( result == HANDLER_ERROR ) ||
313  ( result == HANDLER_HALT ) ||
314  ( result == HANDLER_MSG_DESTROYED ))
315  {
316  // Stop processing the buffer
317  return result;
318  }
319  // A higher int handler has chosen to mask this message from the rest
320  if( ( result == HANDLER_MSG_DONE ))
321  {
322  return HANDLER_CONTINUE;
323  }
324  }
325 
326  return HANDLER_CONTINUE;
327 }
