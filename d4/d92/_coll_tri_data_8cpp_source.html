247 // Garrett: This is an unconventional way of cloning the data. We are assuming that all the
248 // tree nodes are in one contiguous block of memory. I normally wouldn't write code like
249 // this, but we need to avoid lots of small allocations.
250 CCollBSPNode * CCollBSPNode::clone(bool instance)
251 {
252  Dbg_MsgAssert(!instance, ("CCollBSPNode::clone() with instances not implemented yet"));
253  //Dbg_MsgAssert(IsNode(), ("Called CCollBSPNode::clone() on CCollBSPLeaf()"));
254 
255  // So that we don't make lots of little allocations, allocate the whole BSP array at once
256  int num_nodes = count_bsp_nodes();
257  int num_face_indices = count_bsp_face_indices();
258  if (num_nodes > 0)
259  {
260  Dbg_MsgAssert(num_face_indices, ("Didn't find any face indices in the BSP tree"));
261  FaceIndex * p_orig_face_index_array = find_bsp_face_index_array_start();
262 
263  CCollBSPNode * p_new_bsp_array = new CCollBSPNode[num_nodes];
264  FaceIndex * p_new_face_index_array = new FaceIndex[num_face_indices];
265 
266  // Just memcpy it first, since we know the source tree is in an array
267  memcpy(p_new_bsp_array, this, num_nodes * sizeof(CCollBSPNode));
268  memcpy(p_new_face_index_array, p_orig_face_index_array, num_face_indices * sizeof(FaceIndex));
269 
270 #ifdef __NOPT_ASSERT__
271  // Look through new bsp array that still has the old face index pointers in it.
272  for (int node_idx = 0; node_idx < num_nodes; node_idx++)
273  {
274  if (p_new_bsp_array[node_idx].IsLeaf())
275  {
276  Dbg_MsgAssert(p_orig_face_index_array <= p_new_bsp_array[node_idx].GetFaceIndexArray(), ("find_bsp_face_index_array_start() failed"));
277  }
278  }
279 #endif // __NOPT_ASSERT__
280 
281  // Now adjust the pointers by finding the differences
282  int node_address_diff = (int) p_new_bsp_array - (int) this;
283  int face_address_diff = (int) p_new_face_index_array - (int) p_orig_face_index_array;
284  for (int i = 0; i < num_nodes; i++)
285  {
286  if (p_new_bsp_array[i].IsNode())
287  {
288  // Adjust branch pointers
289  p_new_bsp_array[i].m_node.m_children.SetBasePointer((CCollBSPNode *)((int) p_new_bsp_array[i].m_node.m_children.GetBasePointer() + node_address_diff));
290  }
291  else
292  {
293  // Adjust face index pointer
294  p_new_bsp_array[i].m_leaf.mp_face_idx_array = (FaceIndex *) ((int) p_new_bsp_array[i].m_leaf.mp_face_idx_array + face_address_diff);
295  }
296  }
297 
298  return p_new_bsp_array;
299  }
300  else
301  {
302  return NULL;
303  }
304 }

407 void CCollBSPNode::rotate_y(const Mth::Vector & world_origin, Mth::ERot90 rot_y, bool root_node)
408 {
409  // Take it to the origin
410  if (root_node)
411  {
412  translate(-world_origin);
413  }
414 
415  // Rotate node
416  if (IsNode())
417  {
418  int axis = GetSplitAxis();
419 
420  if (axis != Y) // Y won't change
421  {
422  int other_axis = (axis == X) ? Z : X; // So we know what axis we potentially flip to
423 
424  switch (rot_y)
425  {
426  case Mth::ROT_0:
427  break;
428 
429  case Mth::ROT_90:
430  if (axis == X)
431  {
432  SetSplitPoint(-GetSplitPoint());
433  m_node.m_children.SetLeftGreater(!m_node.m_children.IsLeftGreater());
434  }
435  SetSplitAxis(other_axis);
436  break;
437 
438  case Mth::ROT_180:
439  SetSplitPoint(-GetSplitPoint());
440  m_node.m_children.SetLeftGreater(!m_node.m_children.IsLeftGreater());
441  break;
442 
443  case Mth::ROT_270:
444  if (axis == Z)
445  {
446  SetSplitPoint(-GetSplitPoint());
447  m_node.m_children.SetLeftGreater(!m_node.m_children.IsLeftGreater());
448  }
449  SetSplitAxis(other_axis);
450  break;
451 
452  default:
453  Dbg_MsgAssert(0, ("CCollBSPNode::rotate_y() out of range: %d", rot_y));
454  break;
455  }
456  }
457 
458  // Traverse the tree
459  GetLessBranch()->rotate_y(world_origin, rot_y, false);
460  GetGreaterBranch()->rotate_y(world_origin, rot_y, false);
461  }
462  // Leaf does nothing
463 
464  // Put it back
465  if (root_node)
466  {
467  translate(world_origin);
468  }
469 }

604 bool CCollObjTriData::calc_split_faces(uint axis, float axis_distance, FaceIndex *p_face_indexes,
605  uint num_faces, uint & less_faces, uint & greater_faces,
606  FaceIndex *p_less_face_indexes, FaceIndex *p_greater_face_indexes)
607 {
608  less_faces = greater_faces = 0;
609 
610  for (uint i = 0; i < num_faces; i++)
611  {
612  bool less = false, greater = false;
613 
614  // Check the face
615  for (uint j = 0; j < 3; j++)
616  {
617  uint vidx = GetFaceVertIndex(i, j);
618 #if defined(FIXED_POINT_VERTICES) || defined(__PLAT_NGC__)
619  if (GetRawVertexPos(vidx)[axis] < axis_distance)
620  {
621  less = true;
622  } else if (GetRawVertexPos(vidx)[axis] >= axis_distance)
623  {
624  greater = true;
625  }
626 #else
627  if (mp_vert_pos[vidx][axis] < axis_distance)
628  {
629  less = true;
630  } else if (mp_vert_pos[vidx][axis] >= axis_distance)
631  {
632  greater = true;
633  }
634 #endif
635  }
636 
637  Dbg_Assert(less || greater);
638 
639  // Increment counts and possibly put in new array
640  if (less)
641  {
642  if (p_less_face_indexes)
643  {
644  p_less_face_indexes[less_faces] = p_face_indexes[i];
645  }
646  less_faces++;
647  }
648  if (greater)
649  {
650  if (p_greater_face_indexes)
651  {
652  p_greater_face_indexes[greater_faces] = p_face_indexes[i];
653  }
654  greater_faces++;
655  }
656  }
657 
658  return true;
659 }

690 CCollBSPNode * CCollObjTriData::create_bsp_tree(const Mth::CBBox & bbox, FaceIndex *p_face_indexes, uint num_faces, uint level)
691 {
692  if ((num_faces <= s_max_face_per_leaf) || (level == s_max_tree_levels)) // Check if this should be a leaf
693  {
694  return create_bsp_leaf(p_face_indexes, num_faces);
695  } else { // Create Node
696 
697  // Find initial splits on the three axis
698  Mth::Vector mid_width((bbox.GetMax() - bbox.GetMin()) * 0.5f);
699  Mth::Vector mid_split(mid_width + bbox.GetMin());
700 
701  // Find the weighting of the three potential splits
702  uint less_faces[3], greater_faces[3];
703  calc_split_faces(X, mid_split[X], p_face_indexes, num_faces, less_faces[X], greater_faces[X]);
704  calc_split_faces(Y, mid_split[Y], p_face_indexes, num_faces, less_faces[Y], greater_faces[Y]);
705  calc_split_faces(Z, mid_split[Z], p_face_indexes, num_faces, less_faces[Z], greater_faces[Z]);
706 
707  // Figure out best split
708  int best_axis = -1;
709  float best_diff = -1;
710  const int duplicate_threshold = (num_faces * 7) / 10; // tunable
711  for (uint axis = X; axis <= Z; axis++)
712  {
713  float new_diff = (float)fabs((float)(less_faces[axis] - greater_faces[axis]));
714  int duplicates = less_faces[axis] + greater_faces[axis] - num_faces;
715 
716  if (duplicates >= duplicate_threshold)
717  continue;
718 
719  new_diff += duplicates; // tunable
720  new_diff /= mid_width[axis]; // tunable
721

738  FaceIndex *p_less_face_indexes = new FaceIndex[less_faces[best_axis]];
739  FaceIndex *p_greater_face_indexes = new FaceIndex[greater_faces[best_axis]];
740 
741  Mem::Manager::sHandle().PopContext();
742

744  calc_split_faces(best_axis, mid_split[best_axis], p_face_indexes, num_faces,

746  p_less_face_indexes, p_greater_face_indexes);

751  Mth::Vector greater_min = greater_bbox.GetMin();

753  greater_min[best_axis] = mid_split[best_axis];

766  delete p_greater_face_indexes;

1280 
1281 void CCollObjTriData::get_float_array_from_data(Mth::Vector *p_float_array) const
1282 {
1283 #ifdef __PLAT_NGC__
1284  if ( mp_cloned_vert_pos )
1285  {

1287  for ( int i = 0; i < m_num_verts; i++ )

1294  }
1295  else

1297  // We need to pull the verts out of the main render vert list.
1298  int lp;
1299  int lp2;
1300  int index;
1301  unsigned char used[(MAX_USED/8)];
1302 
1303  // Build a bit list of used verts.
1304  for ( lp = 0; lp < (MAX_USED/8); lp++ ) used[lp] = 0;
1305 
1306  for ( lp = 0; lp < m_num_faces; lp++ )
1307  {
1308  index = mp_faces[lp].m_vertex_index[0];
1309  used[(index>>3)] |= (1<<(index&7));
1310  index = mp_faces[lp].m_vertex_index[1];
1311  used[(index>>3)] |= (1<<(index&7));
1312  index = mp_faces[lp].m_vertex_index[2];
1313  used[(index>>3)] |= (1<<(index&7));
1314  }
1315 
1316  // Copy verts out in order.
1317  int num_copied = 0;
1318  for ( lp = 0; lp < (MAX_USED/8); lp++ )
1319  {
1320  if ( used[lp] )
1321  {
1322  for ( lp2 = 0; lp2 < 8; lp2++ )
1323  {
1324  if ( used[lp] & (1<<lp2) )
1325  {
1326  index = (lp<<3) + lp2;
1327  p_float_array[num_copied] = GetRawVertexPos(index);
1328  num_copied++;
1329  }
1330  }
1331  }
1332  }
1333  }
1334 #else
