327 void * LoadVertices(void *pFile, sMesh *pMesh, sMaterial *pMat, sGroup *pGroup, int& skinnedIndexCount)
328 {
329  uint32 NumVertices;
330  uint REGS, NREG, PRIM, ADDR;
331  sTexture *pTex;
332  uint32 i;
333  uint8 VertexData[64];
334  int STOffset, ColourOffset, NormalOffset, SkinOffset, XYZOffset, VertexSize;
335  int texture_width=0, texture_height=0;
336 
337  // get number of vertices for this mesh
338  MEM_Read(&NumVertices, sizeof(uint32), 1, pFile);
339 
340  // skip mesh if it has no vertices
341  if (NumVertices==0)
342  return pFile;
343 
344  // work out giftag fields
345  REGS = gs::XYZF2;
346  NREG = 1;
347  PRIM = TRISTRIP|ABE|FGE;
348  ADDR = VU1_ADDR(Proj);
349  if (pMesh->Flags & MESHFLAG_COLOURS)
350  {
351  REGS = REGS<<4 | gs::RGBAQ;
352  NREG++;
353  }
354  if (pMesh->Flags & MESHFLAG_NORMALS)
355  {
356  if (pMat->Flags & MATFLAG_ENVIRONMENT)
357  {
358  ADDR = VU1_ADDR(Refl);
359  REGS = REGS<<4 | gs::ST;
360  PRIM |= TME;
361  }
362  else
363  {
364  REGS = REGS<<4 | gs::NOP;
365  }
366  NREG++;
367  }
368  if ((pMesh->Flags & MESHFLAG_TEXTURE) && !(pMat->Flags & MATFLAG_ENVIRONMENT))
369  {
370  REGS = REGS<<4 | gs::ST;
371  NREG++;
372  PRIM |= TME;
373  ADDR = VU1_ADDR(PTex);
374  }
375  if (pMat->Flags & MATFLAG_SMOOTH)
376  PRIM |= IIP;
377 
378  // override everything for skinned models!
379  if (pMesh->Flags & MESHFLAG_SKINNED)
380  {
381  if (pMat->RegCLAMP & PackCLAMP(3,3,0,0,0,0))
382  {
383  // clamped case, send texture coords as 32 bits floating point STs
384  REGS = gs::XYZ2<<16 | gs::RGBAQ<<12 | gs::NOP<<8 | gs::NOP<<4 | gs::ST;
385  NREG = 5;
386  PRIM |= TME;
387  ADDR = VU1_ADDR(Skin);
388  }
389  else
390  {
391  // standard case, non-clamped, compress texture coords to 16 bit fixed point UVs
392  REGS = gs::XYZ2<<16 | gs::RGBAQ<<12 | gs::NOP<<8 | gs::NOP<<4 | gs::UV;
393  NREG = 5;
394  PRIM |= TME|FST;
395  ADDR = VU1_ADDR(Skin);
396  }
397  }
398 
399  // output GS context for material
400  gs::BeginPrim(REL,0,0);
401  gs::Reg1(gs::ALPHA_1, pMat->RegALPHA);
402  gs::Reg1(gs::TEST_1, PackTEST(1,AGEQUAL,pMat->Aref,KEEP,0,0,1,ZGEQUAL));
403 
404  // texture registers if necessary
405  if (pMat->Flags & MATFLAG_TEXTURED)
406  {
407  pTex = pMat->pTex;
408  gs::Reg1(gs::TEX0_1, pTex->RegTEX0);
409  gs::Reg1(gs::TEX1_1, pMat->RegTEX1);
410  gs::Reg1(gs::CLAMP_1, pMat->RegCLAMP);
411  if (pTex->MXL > 0)
412  {
413  gs::Reg1(gs::MIPTBP1_1,pTex->RegMIPTBP1);
414  if (pTex->MXL > 3)
415  {
416  gs::Reg1(gs::MIPTBP2_1,pTex->RegMIPTBP2);
417  }
418  }
419 
420  // extract texture dimensions
421  texture_width = 1 << ((pTex->RegTEX0 >> 26) & 0xF);
422  texture_height = 1 << ((pTex->RegTEX0 >> 30) & 0xF);
423  }
424  gs::EndPrim(0);
425 
426  // set maximum vu buffer size
427  vu1::MaxBuffer = pMesh->Flags&MESHFLAG_SKINNED ? 240 : 240;
428 
429  // begin a batch of vertices
430  BeginModelMultiPrim(REGS, NREG, PRIM, NumVertices, ADDR);
431 
432  // work out vertex size and data offsets
433  VertexSize = (pMesh->Flags & MESHFLAG_SKINNED) ? 8 : 16;
434  XYZOffset = SkinOffset = NormalOffset = ColourOffset = STOffset = 0;
435  if (pMesh->Flags & MESHFLAG_SKINNED)
436  {
437  VertexSize += 8;
438  XYZOffset += 8;
439  }
440  if (pMesh->Flags & MESHFLAG_NORMALS)
441  {
442  VertexSize += 4;
443  XYZOffset += 4;
444  SkinOffset += 4;
445  }
446  if (pMesh->Flags & MESHFLAG_COLOURS)
447  {
448  VertexSize += 4;
449  XYZOffset += 4;
450  SkinOffset += 4;
451  NormalOffset += 4;
452  }
453  if (pMesh->Flags & MESHFLAG_TEXTURE)
454  {
455  int size = (pMesh->Flags & MESHFLAG_SKINNED) ? 4 : 8;
456  VertexSize += size;
457  XYZOffset += size;
458  SkinOffset += size;
459  NormalOffset += size;
460  ColourOffset += size;
461  }
462  // loop over vertices
463  for (i=0; i<NumVertices; i++)
464  {
465  UnpackOffset=1;
466 
467  MEM_Read(VertexData, 1, VertexSize, pFile);
468 
469  BeginVertex(pMesh, pMat);
470 
471  // special case for skinned models
472  if (pMesh->Flags & MESHFLAG_SKINNED)
473  {
474  // texture coords
475  UnpackOffset = 1;
476  if ((pMesh->Flags & MESHFLAG_TEXTURE) && !(pMat->Flags & MATFLAG_ENVIRONMENT))
477  {
478  if (pMat->RegCLAMP & PackCLAMP(3,3,0,0,0,0))
479  {
480  VertexSTFloat(VertexData+STOffset);
481  }
482  else
483  {
484  VertexUV(VertexData+STOffset, texture_width<<4, texture_height<<4);
485  }
486  }

490  VertexWeights(VertexData+SkinOffset);

492  // normal and transform offsets
493  UnpackOffset = 3;
494  VertexSkinNormal(VertexData+NormalOffset, VertexData+SkinOffset+4);
495 
496  // colour

499  {
500  VertexRGBA(VertexData+ColourOffset);
501  }
502 
503  // position
504  UnpackOffset = 5;
505  VertexXYZ(VertexData+XYZOffset, true, skinnedIndexCount);
506  }
507  else

511  {
512  VertexST16(VertexData+STOffset);
513  }
514

517  {
518  VertexNormal(VertexData+NormalOffset);

520 
521  // colour
