269 // Registering components simply adds them to the list of components that
270 // the compositeobject manager can create
271 // All that is passed in is a 32 bit id (of the component, e.g. CRC_MODEL)
272 // and the callback function pointer to the s_create function
273 
274 void CCompositeObjectManager::RegisterComponent(uint32 id, CBaseComponent *(p_create_function)(), void(p_register_function)())
275 {
276  Dbg_MsgAssert(m_num_components < vMAX_COMPONENTS,("Too many components (%d)",vMAX_COMPONENTS));
277  m_registered_components[m_num_components].mComponentID = id;
278  m_registered_components[m_num_components].mpCreateFunction = p_create_function;
279  m_num_components++;
280 
281  // I'm letting the component manager control calling the "register" function
282  // no good reason, just better encapsulation. Prevents bugs.
283  if (p_register_function)
284  {
285  p_register_function();
286  }

313 void CCompositeObjectManager::Update()
314 {
315  Lst::Search< CObject > sh;
316 
317  uint32 stamp_mask = m_stamp_bit_manager.RequestBit();
318 
319  // clear the appropriate stamp bit
320  for (CObject* pObject = sh.FirstItem(m_object_list); pObject; pObject = sh.NextItem())
321  {
322  pObject->ClearStampBit(stamp_mask);
323  }
324 
325  // traverse the object list
326  do
327  {
328  // clear the appropriate list changed bit
329  m_list_changed &= ~stamp_mask;
330 
331  CObject* pNextObject = sh.FirstItem(m_object_list);
332 
333  while (pNextObject)
334  {
335  CObject* pObject = pNextObject;
336  pNextObject = sh.NextItem();
337 
338  if (!pObject->CheckStampBit(stamp_mask))
339  {
340  pObject->SetStampBit(stamp_mask);
341 
342  #ifdef __NOPT_ASSERT__
343  Tmr::CPUCycles time_before = Tmr::GetTimeInCPUCycles();
344  CSmtPtr< CObject > p_smart_object = pObject;
345  uint32 obj_id = pObject->GetID();
346  #endif
347 
348  static_cast< CCompositeObject* >(pObject)->Update();
349 
350  #ifdef __NOPT_ASSERT__
351  Dbg_MsgAssert(p_smart_object, ("Object %s has deleted itself in its Update() function", Script::FindChecksumName(obj_id)));
352  // Convert to microseconds by dividing by 150
353  static_cast< CCompositeObject* >(pObject)->SetUpdateTime((Tmr::GetTimeInCPUCycles() - time_before) / 150);
354  #endif
355 
356  if (m_list_changed & stamp_mask) break;
357  }
358  }
359  }
360  while (m_list_changed & stamp_mask);
361 
362  m_stamp_bit_manager.ReturnBit(stamp_mask);
363 }

471 void CCompositeObjectManager::RemoveComponentByType( CBaseComponent *p_component )
472 {
473  uint32 id = p_component->GetType();
474 
475  for( uint32 i = 0; i < m_num_components; ++i )
476  {
477  if( m_registered_components[i].mComponentID == id )
478  {
479  if( mp_components_by_type[i] == p_component )
480  {
481  mp_components_by_type[i] = p_component->GetNextSameType();
482  p_component->SetNextSameType( NULL );
483  return;
484  }
485  else
486  {
487  CBaseComponent *p_loop = mp_components_by_type[i];
488  while( p_loop )
489  {
490  if( p_loop->GetNextSameType() == p_component )
491  {
492  p_loop->SetNextSameType( p_component->GetNextSameType());
493  p_component->SetNextSameType( NULL );
494  return;
495  }
496  p_loop = p_loop->GetNextSameType();
497  }
498  }
499  }
500  }
501  Dbg_MsgAssert( 0, ( "Component %s not registered", Script::FindChecksumName( id )));
502 }
