380 NsMatrix * NsCamera::getViewMatrix ( void )
381 {
382  float scale;
383  NsVector vVector;
384  NsVector temp;
385  NsVector cpos;
386  NsMatrix * cameraLTM;
387 
388  cameraLTM = getFrame()->getModelMatrix();
389  cpos.set( cameraLTM->getPosX(), cameraLTM->getPosY(), cameraLTM->getPosZ() );
390 
391  /*INDENT OFF */
392 
393  /*
394  * Builds the concatenation of the following matrices:
395  *
396  * [ [ -Right_X, Up_X, At_X, 0 ] re-orient
397  * [ -Right_Y, Up_Y, At_Y, 0 ]
398  * [ -Right_Z, Up_Z, At_Z, 0 ]
399  * [ pos Right, -(pos Up), -(pos At), 1 ] ]
400  * [ [ 1, 0, 0, 0 ] offset eye
401  * [ 0, 1, 0, 0 ]
402  * [ off_x, off_y, 1, 0 ]
403  * [ -off_x, -off_y, 0, 1 ] ]
404  * [ [ 0.5 / width, 0, 0, 0 ] scale for view window
405  * [ 0, 0.5 / height, 0, 0 ]
406  * [ 0, 0, 1, 0 ]
407  * [ 0, 0, 0, 1 ] ]
408  * [ [ 1, 0, 0, 0 ] project & flip y
409  * [ 0, -1, 0, 0 ]
410  * [ 0, 0, 1, 1 ] DIFFERS FROM PARALLEL
411  * [ 0, 0, 0, 0 ] ]
412  * [ [ 1, 0, 0, 0 ] xform XY
413  * [ 0, 1, 0, 0 ] from [-0.5..0.5]^2
414  * [ 0, 0, 1, 0 ] to [0..1]^2
415  * [ 0.5, 0.5, 0, 1 ] ]
416  */
417 
418  /*INDENT ON */
419 
420  /* At */
421  scale = (view1x * ((float) ((-0.5))));
422  vVector.scale( *cameraLTM->getRight(), scale );
423 
424  scale = ( 0.5f - (scale * 0.0f) );
425  temp.scale( *cameraLTM->getAt(), scale );
426  vVector.add( temp );
427 
428  m_view.setRight( &vVector );
429 
430  m_view.setRightX( vVector.x );
431  m_view.setUpX( vVector.y );
432  m_view.setAtX( vVector.z );
433  m_view.setPosX ( 0.5f - ( scale + cpos.dot( vVector ) ) );
434 
435  /* Up */
436  scale = ( view1y * -0.5f );
437  vVector.scale( *cameraLTM->getUp(), scale );
438 
439  scale = ( 0.5f + (scale * 0.0f) );
440  temp.scale( *cameraLTM->getAt(), scale );
441  vVector.add( temp );
442 
443  m_view.setRightY( vVector.x );
444  m_view.setUpY( vVector.y );
445  m_view.setAtY( vVector.z );
446  m_view.setPosY ( 0.5f - ( scale + cpos.dot( vVector ) ) );
447 
448  /* At */
449  m_view.setRightZ(cameraLTM->getAt()->x );
450  m_view.setUpZ( cameraLTM->getAt()->y );
451  m_view.setAtZ( cameraLTM->getAt()->z );
452  m_view.setPosZ ( -cpos.dot( *cameraLTM->getAt() ) );
453 
454  return &m_view;
455 }
