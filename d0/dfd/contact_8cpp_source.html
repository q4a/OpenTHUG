44 // call this once per frame, to calculate the movement of the contact point
45 bool CContact::Update(const Mth::Vector &pos)
46 {
47  if (!mp_moving_object.Convert())
48  {
49  return false;
50  }
51 
52  m_movement = mp_moving_object->GetPos() - m_object_last_pos;
53 
54  // so far we have only accounted for the movement of the object's origin
55  // (mp_moving_object->GetPos()), and we need to see if the object has rotated
56  // which will give us an additional movement vector
57 
58  if (m_object_last_matrix != mp_moving_object->GetMatrix())
59  {
60  m_rotated = true;
61 // printf ("Object has rotated\n");
62  // Get the position relative to the origin on the object
63  Mth::Vector old_offset = pos - m_object_last_pos;
64  Mth::Vector offset = old_offset;
65 
66 // printf ("Offset (%.2f,%.2f,%.2f)\n",offset[X],offset[Y],offset[Z]);
67 
68  // transform by the inverse of the original matrix
69  Mth::Matrix back = m_object_last_matrix;
70  // just to be safe
71  back[W] = Mth::Vector(0,0,0,1);
72  back[X][W] = 0.0f;
73  back[Y][W] = 0.0f;
74  back[Z][W] = 0.0f;
75  back.InvertUniform();
76 
77  // transform to the new matrix
78  m_rotation = mp_moving_object->GetMatrix();
79  // just to be safe
80  m_rotation[W] = Mth::Vector(0,0,0,1);
81  m_rotation[X][W] = 0.0f;
82  m_rotation[Y][W] = 0.0f;
83  m_rotation[Z][W] = 0.0f;
84 
85  m_rotation *= back;
86 
87  Mth::Vector new_offset = m_rotation.Transform(offset);
88 
89 // printf ("new Offset (%.2f,%.2f,%.2f)\n",new_offset[X],new_offset[Y],new_offset[Z]);
90 
91  // Calculate the movment due to rotation
92  Mth::Vector rotation_movement = (new_offset - old_offset);
93 
94  // add it to the movement, so it is included in the skater's movement
95  m_movement += rotation_movement;
96 
97  m_object_last_matrix = mp_moving_object->GetMatrix();
98 
99  }
100  else
101  {
102  m_rotated = false;
103  }
104 
105  m_object_last_pos = mp_moving_object->GetPos();
106 
107  return true;
108 }
