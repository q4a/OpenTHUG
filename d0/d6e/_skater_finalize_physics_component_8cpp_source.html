92 void CSkaterFinalizePhysicsComponent::Update()
93 {
94  // setup the state component
95 
96  // Logic for setting/not setting the flag for telling the camera whether to look down on the skater or not.
97  if (mp_core_physics_component->m_began_frame_in_lip_state && mp_state_component->GetState() != LIP)
98  {
99  // This flag is sort of badly named now, it really means we changed from the lip state to something other than GROUND or LIP.
100  mp_state_component->mJumpedOutOfLipTrick = false;
101  if (mp_state_component->GetState() == AIR)
102  {
103  // we only want to set this flag if we jumped straight up; meaning the x and z velocities are close to zero
104  if (Mth::Abs(GetObject()->m_vel[X]) < 1.0f && Mth::Abs(GetObject()->m_vel[Z]) < 1.0f)
105  {
106  mp_state_component->mJumpedOutOfLipTrick = true;
107  }
108  }
109  }
110 
111  // this flag needs clearing whenever we get out of the air
112  if (mp_state_component->GetState() != AIR)
113  {
114  mp_state_component->mJumpedOutOfLipTrick = false;
115  }
116 
117  mp_state_component->m_camera_display_normal = mp_core_physics_component->m_display_normal;
118  mp_state_component->m_camera_current_normal = mp_core_physics_component->m_current_normal;
119 
120  // make sure the matrices don't get distorted
121  GetObject()->m_matrix.OrthoNormalizeAbout(Y);
122  mp_core_physics_component->m_lerping_display_matrix.OrthoNormalizeAbout(Y);
123 
124  // if any part of the matrix has collapsed, then we will neet to patch it up
125  // Note, this is a very rare occurence; probably only occurs when you hit things perfectly at right angles, so
126  // you attempt to orthonormalize about an axis that that is now coincident with another axis
127  // would not happen if we rotated the matrix, or used quaternions
128  if (GetObject()->m_matrix.PatchOrthogonality())
129  {
130  mp_core_physics_component->ResetLerpingMatrix();
131  }
132 
133  // Extract the informations from the physics object that we need for rendering
134  GetObject()->SetDisplayMatrix(mp_core_physics_component->m_lerping_display_matrix);
135 
136  #ifdef __USER_DAN__
137  // Gfx::AddDebugArrow(GetObject()->GetPos(), GetObject()->GetPos() + 60.0f * GetObject()->GetDisplayMatrix()[Z], RED, 0, 1);
138  // Gfx::AddDebugArrow(GetObject()->GetPos(), GetObject()->GetPos() + 60.0f * GetObject()->GetDisplayMatrix()[X], BLUE, 0, 1);
139  // Gfx::AddDebugArrow(GetObject()->GetPos(), GetObject()->GetPos() + 60.0f * GetObject()->GetDisplayMatrix()[Y], GREEN, 0, 1);
140  #endif
141 
142  // update the sound components' state
143 
144  Obj::CSkaterSoundComponent *pSoundComponent = GetSkaterSoundComponentFromObject( GetObject() );
145  Dbg_Assert( pSoundComponent );
146 
147  pSoundComponent->SetIsRailSliding( mp_state_component->GetFlag(RAIL_SLIDING) );
148  pSoundComponent->SetTerrain( mp_state_component->GetTerrain() );
149 
150  Obj::CSkaterLoopingSoundComponent *pLoopingSoundComponent = GetSkaterLoopingSoundComponentFromObject( GetObject() );
151  Dbg_Assert( pLoopingSoundComponent );
152 
153  float speed_fraction = sqrtf( GetObject()->GetVel()[X] * GetObject()->GetVel()[X] + GetObject()->GetVel()[Z] * GetObject()->GetVel()[Z] ) / GetSkater()->GetScriptedStat(CRCD(0xcc5f87aa, "Skater_Max_Max_Speed_Stat") );
154  pLoopingSoundComponent->SetSpeedFraction( speed_fraction );
155  pLoopingSoundComponent->SetIsBailing(mp_state_component->GetFlag(IS_BAILING));
156  pLoopingSoundComponent->SetIsRailSliding( mp_state_component->GetFlag(RAIL_SLIDING) );
157  pLoopingSoundComponent->SetTerrain( mp_state_component->GetTerrain() );
158  pLoopingSoundComponent->SetState( mp_state_component->GetState() );
159 }
