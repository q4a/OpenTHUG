65 void CNgcModelLights::plat_update_engine( Mth::Vector & pos, bool add_scene_light )
66 {
67  if( m_flags & mUSE_MODEL_AMBIENT )
68  {
69  // Use the local ambient color, modulate it with the local ambient brightness.
70  int difr = (int)((float)m_ambient_color.r * m_ambient_brightness);
71  int difg = (int)((float)m_ambient_color.g * m_ambient_brightness);
72  int difb = (int)((float)m_ambient_color.b * m_ambient_brightness);
73  NxNgc::EngineGlobals.ambient_light_color.r = difr > 255 ? 255 : (u8)difr;
74  NxNgc::EngineGlobals.ambient_light_color.g = difg > 255 ? 255 : (u8)difg;
75  NxNgc::EngineGlobals.ambient_light_color.b = difb > 255 ? 255 : (u8)difb;
76  }
77  else
78  {
79  // Use the default ambient color, but modulate it with the local ambient brightness.
80  Image::RGBA amb = CLightManager::sGetLightAmbientColor();
81  int difr = (int)((float)amb.r * m_ambient_brightness);
82  int difg = (int)((float)amb.g * m_ambient_brightness);
83  int difb = (int)((float)amb.b * m_ambient_brightness);
84  NxNgc::EngineGlobals.ambient_light_color.r = difr > 255 ? 255 : (u8)difr;
85  NxNgc::EngineGlobals.ambient_light_color.g = difg > 255 ? 255 : (u8)difg;
86  NxNgc::EngineGlobals.ambient_light_color.b = difb > 255 ? 255 : (u8)difb;
87  }
88 
89 // NxNgc::EngineGlobals.ambient_light_color.r = NxNgc::EngineGlobals.ambient_light_color.r < 128 ? NxNgc::EngineGlobals.ambient_light_color.r * 2 : 255;
90 // NxNgc::EngineGlobals.ambient_light_color.g = NxNgc::EngineGlobals.ambient_light_color.g < 128 ? NxNgc::EngineGlobals.ambient_light_color.g * 2 : 255;
91 // NxNgc::EngineGlobals.ambient_light_color.b = NxNgc::EngineGlobals.ambient_light_color.b < 128 ? NxNgc::EngineGlobals.ambient_light_color.b * 2 : 255;
92 
93  for( int i = 0; i < 2; ++i )
94  {
95  if( m_flags & (( i == 0 ) ? mUSE_MODEL_DIFFUSE_0 : mUSE_MODEL_DIFFUSE_1 ))
96  {
97  // Use the local directional color, modulate it with the local directional brightness.
98  int difr = (int)((float)m_diffuse_color[i].r * m_diffuse_brightness[i] * 0.5f);
99  int difg = (int)((float)m_diffuse_color[i].g * m_diffuse_brightness[i] * 0.5f);
100  int difb = (int)((float)m_diffuse_color[i].b * m_diffuse_brightness[i] * 0.5f);
101  NxNgc::EngineGlobals.diffuse_light_color[i].r = difr > 255 ? 255 : (u8)difr;
102  NxNgc::EngineGlobals.diffuse_light_color[i].g = difg > 255 ? 255 : (u8)difg;
103  NxNgc::EngineGlobals.diffuse_light_color[i].b = difb > 255 ? 255 : (u8)difb;
104 
105  // Use the local direction.
106  m_diffuse_direction[i].Normalize();
107  NxNgc::EngineGlobals.light_x[i] = m_diffuse_direction[i][X] * 200000.0f;
108  NxNgc::EngineGlobals.light_y[i] = m_diffuse_direction[i][Y] * 200000.0f;
109  NxNgc::EngineGlobals.light_z[i] = m_diffuse_direction[i][Z] * 200000.0f;
110  }
111  else
112  {
113  // Use the default directional color, but modulate it with the local directional brightness.
114  Image::RGBA dif = CLightManager::sGetLightDiffuseColor( i );
115  int difr = (int)((float)dif.r * m_diffuse_brightness[i] * 0.5f);
116  int difg = (int)((float)dif.g * m_diffuse_brightness[i] * 0.5f);
117  int difb = (int)((float)dif.b * m_diffuse_brightness[i] * 0.5f);
118  NxNgc::EngineGlobals.diffuse_light_color[i].r = difr > 255 ? 255 : (u8)difr;
119  NxNgc::EngineGlobals.diffuse_light_color[i].g = difg > 255 ? 255 : (u8)difg;
120  NxNgc::EngineGlobals.diffuse_light_color[i].b = difb > 255 ? 255 : (u8)difb;
121 
122  // Use the default direction.
123  Mth::Vector dir = CLightManager::sGetLightDirection( i );
124  dir.Normalize();
125  NxNgc::EngineGlobals.light_x[i] = dir[X] * 200000.0f;
126  NxNgc::EngineGlobals.light_y[i] = dir[Y] * 200000.0f;
127  NxNgc::EngineGlobals.light_z[i] = dir[Z] * 200000.0f;
128  }
129 // NxNgc::EngineGlobals.diffuse_light_color[i].r = NxNgc::EngineGlobals.diffuse_light_color[i].r < 128 ? NxNgc::EngineGlobals.diffuse_light_color[i].r * 2 : 255;
130 // NxNgc::EngineGlobals.diffuse_light_color[i].g = NxNgc::EngineGlobals.diffuse_light_color[i].g < 128 ? NxNgc::EngineGlobals.diffuse_light_color[i].g * 2 : 255;
131 // NxNgc::EngineGlobals.diffuse_light_color[i].b = NxNgc::EngineGlobals.diffuse_light_color[i].b < 128 ? NxNgc::EngineGlobals.diffuse_light_color[i].b * 2 : 255;
132  }
133 
134  if( add_scene_light )
135  {
136  Nx::CSceneLight *p_scene_light = CLightManager::sGetOptimumSceneLight( pos );
137  if( p_scene_light )
138  {
139  Mth::Vector light_pos = p_scene_light->GetLightPosition();
140 
141  float dist = Mth::Distance( pos, light_pos );
142  float ratio = dist * p_scene_light->GetLightReciprocalRadius();
143 
144  light_pos = ( pos - light_pos ).Normalize();
145 
146  // Figure the direction...
147  NxNgc::EngineGlobals.light_x[2] = -light_pos[X] * 200000.0f;
148  NxNgc::EngineGlobals.light_y[2] = -light_pos[Y] * 200000.0f;
149  NxNgc::EngineGlobals.light_z[2] = -light_pos[Z] * 200000.0f;
150 
151  // ...and the color.
152  ratio = sqrtf( 1.0f - ratio ) * p_scene_light->GetLightIntensity();
153  int difr = (int)((float)p_scene_light->GetLightColor().r * ratio * 0.5f);
154  int difg = (int)((float)p_scene_light->GetLightColor().g * ratio * 0.5f);
155  int difb = (int)((float)p_scene_light->GetLightColor().b * ratio * 0.5f);
156  NxNgc::EngineGlobals.diffuse_light_color[2].r = difr > 255 ? 255 : (u8)difr;
157  NxNgc::EngineGlobals.diffuse_light_color[2].g = difg > 255 ? 255 : (u8)difg;
158  NxNgc::EngineGlobals.diffuse_light_color[2].b = difb > 255 ? 255 : (u8)difb;
159  }
160  else
161  {
162  // Disbale this light by setting zero color.
163  NxNgc::EngineGlobals.diffuse_light_color[2].r = 0;
164  NxNgc::EngineGlobals.diffuse_light_color[2].g = 0;
165  NxNgc::EngineGlobals.diffuse_light_color[2].b = 0;
166  }
167  }
