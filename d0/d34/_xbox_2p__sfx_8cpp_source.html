298 //-----------------------------------------------------------------------------
299 // Name: Open()
300 // Desc: Opens an existing chunk.
301 //-----------------------------------------------------------------------------
302 bool CRiffChunk::Open()
303 {
304  RIFFHEADER rhRiffHeader;
305  LONG lOffset = 0;
306 
307  // Seek to the first byte of the parent chunk's data section
308  if( m_pParentChunk )
309  {
310  lOffset = m_pParentChunk->m_dwDataOffset;
311 
312  // Special case the RIFF chunk
313  if( FOURCC_RIFF == m_pParentChunk->m_fccChunkId )
314  {
315  lOffset += sizeof(FOURCC);
316  }
317  }
318 
319  // Read each child chunk header until we find the one we're looking for
320 // RwFileFunctions* fs = RwOsGetFileInterface();
321  for( ;; )
322  {
323 // if( fs->rwfseek( m_hFile, lOffset, SEEK_SET ) != 0 )
324  File::Seek( m_hFile, lOffset, SEEK_SET );
325 // {
326 // return false;
327 // }
328 
329  DWORD dwRead;
330 // dwRead = fs->rwfread( &rhRiffHeader, sizeof( rhRiffHeader ), 1, m_hFile );
331  dwRead = File::Read( &rhRiffHeader, sizeof( rhRiffHeader ), 1, m_hFile );
332  if( dwRead != sizeof( rhRiffHeader ))
333  {
334  return false;
335  }
336 
337  // Check if we found the one we're looking for.
338  if( m_fccChunkId == rhRiffHeader.fccChunkId )
339  {
340  // Save the chunk size and data offset.
341  m_dwDataOffset = lOffset + sizeof( rhRiffHeader );
342  m_dwDataSize = rhRiffHeader.dwDataSize;
343 
344  // Success.
345  m_dwFlags |= RIFFCHUNK_FLAGS_VALID;
346 
347  return true;
348  }
349  lOffset += sizeof( rhRiffHeader ) + rhRiffHeader.dwDataSize;
350  }
351  return false;
352 }

473 //-----------------------------------------------------------------------------
474 // Name: GetFormat()
475 // Desc: Gets the wave file format
476 //-----------------------------------------------------------------------------
477 bool CWaveFile::GetFormat( WAVEFORMATEX* pwfxFormat, DWORD dwFormatSize )
478 {
479  DWORD dwValidSize = m_FormatChunk.GetDataSize();
480 
481  if( NULL == pwfxFormat || 0 == dwFormatSize )
482  return false;
483 
484  // Read the format chunk into the buffer
485  bool hr = m_FormatChunk.ReadData( 0, pwfxFormat, min( dwFormatSize, dwValidSize ));
486 
487  if( hr )
488  {
489  // Zero out remaining bytes, in case enough bytes were not read
490  if( dwFormatSize > dwValidSize )
491  {
492  ZeroMemory( (BYTE*)pwfxFormat + dwValidSize, dwFormatSize - dwValidSize );
493  }
494  }
495 
496  return hr;
497 }

591 //-----------------------------------------------------------------------------
592 // Name: Create()
593 // Desc: Creates the sound. Sound is buffered to memory allocated internally
594 // by DirectSound.
595 //-----------------------------------------------------------------------------
596 bool CXBSound::Create( const CHAR* strFileName, DWORD dwFlags )
597 {
598  bool br;
599 
600  // Open the .wav file
601  CWaveFile waveFile;
602  br = waveFile.Open( strFileName );
603  if( !br )
604  {
605  return false;
606  }
607 
608  // Get the WAVEFORMAT structure for the .wav file
609  br = waveFile.GetFormat( &m_WaveFormat.wfx, sizeof( WAVEFORMATEX ));
610  if( !br )
611  {
612  return false;
613  }
614 
615  // Required if the sound is Xbox ADPCM, will be ignored otherwise.
616  m_WaveFormat.wSamplesPerBlock = 64;
617 
618  // Get the size of the .wav file
619  waveFile.GetDuration( &m_dwBufferSize );
620 
621  // See if the sound is looping.
622  m_Loops = waveFile.ContainsLoop();
623 
624  // Create the sound buffer
625  br = Create( &m_WaveFormat, dwFlags, NULL, m_dwBufferSize );
626  if( !br )
627  {
628  return false;
629  }
630 
631  // Lock the buffer so it can be filled
632  VOID* pLock1 = NULL;
633  VOID* pLock2 = NULL;
634  DWORD dwLockSize1 = 0L;
635  DWORD dwLockSize2 = 0L;
636 
637  pLock1 = new char[m_dsbd.dwBufferBytes];
638 
639  // Read the wave file data into the buffer
640  br = waveFile.ReadSample( 0L, pLock1, m_dsbd.dwBufferBytes, NULL );
641  if( !br )
642  {
643  return false;
644  }
645 
646  // Store pointer to raw sound data.
647  m_pRawData = pLock1;
648 
649  return true;
650 }

655 //-----------------------------------------------------------------------------
656 // Name: Create()
657 // Desc: Creates the sound and tells DirectSound where the sound data will be
658 // stored. If pBuffer is NULL, DirectSound handles buffer creation.
659 //-----------------------------------------------------------------------------
660 bool CXBSound::Create( const XBOXADPCMWAVEFORMAT* pwfxFormat, DWORD dwFlags, VOID* pBuffer, DWORD dwBytes )
661 {
662  // Setup the sound buffer description.
663  ZeroMemory( &m_dsbd, sizeof(DSBUFFERDESC) );
664  m_dsbd.dwSize = sizeof(DSBUFFERDESC);
665  m_dsbd.dwFlags = dwFlags;
666 
667  if( pwfxFormat->wfx.wFormatTag == WAVE_FORMAT_XBOX_ADPCM )
668  {
669  m_dsbd.lpwfxFormat = (LPWAVEFORMATEX)&m_WaveFormat;
670  }
671  else
672  {
673  m_dsbd.lpwfxFormat = (LPWAVEFORMATEX)&pwfxFormat->wfx;
674  }
675 
676  // If pBuffer is non-NULL, dwBufferBytes will be zero, which informs
677  // DirectSoundCreateBuffer that we will presently be using SetBufferData().
678  // Otherwise, we set dwBufferBytes to the size of the WAV data, potentially
679  // including alignment bytes.
680  if( pBuffer == NULL )
681  {
682  m_dsbd.dwBufferBytes = ( 0 == m_WaveFormat.wfx.nBlockAlign ) ? dwBytes :
683  dwBytes - ( dwBytes % m_WaveFormat.wfx.nBlockAlign );
684  }
685 
686  // If buffer specified, tell DirectSound to use it
687  if( pBuffer != NULL )
688  {
689  // Store pointer to raw sound data.
690  m_pRawData = pBuffer;
691  }
692 
693  return true;
694 }

762  }
763

766  {
767  VoiceSimulator[i].p_buffer = NULL;
768  }
769 
770  // Create the DirectSound interface.

773 // HANDLE hFile = CreateFile( "d:\\data\\sounds\\bin\\dsstdfx.bin", GENERIC_READ, 0, NULL, OPEN_EXISTING, 0, NULL );

775  if( hFile == INVALID_HANDLE_VALUE )
776  {
777  Dbg_Assert( 0 );
778  }
779  else
780  {
781  // Determine the size of the scratch image by seeking to the end of the file.
782  int size = SetFilePointer( hFile, 0, NULL, FILE_END );

784 
785  // Allocate memory to read the scratch image from disk
786  BYTE* p_buffer = new BYTE[size];
787 
788  // Read the image in

790  BOOL bResult = ReadFile( hFile, p_buffer, size, &dwBytesRead, 0 );
791  if( !bResult )
792  {
793  Dbg_Assert( 0 );
794  }
795  else
796  {
797  // Call DSound API to download the image.

1068 //void SetVoiceParameters( int voice, float volL, float volR, float pitch )
1069 void SetVoiceParameters( int voice, sVolume *p_vol, float pitch )
1070 {
1071  if( VoiceSimulator[voice].p_buffer )
1072  {
1073  // This array will hold individual volumes for the five speakers.
1074  // In order, they are: front left, center, front right, rear right, rear left.
1075  float volumes[5];
1076  int i_volumes[5], max_i_volume;
1077 
1078  memset( volumes, 0, sizeof( float ) * 5 );
1079 
1080  if( p_vol->IsSilent())
1081  {
1082  // Pointless doing any more work.
1083  }
1084  else
1085  {
1086  switch( p_vol->GetVolumeType())
1087  {
1088  case VOLUME_TYPE_5_CHANNEL_DOLBY5_1:
1089  {
1090  // All the work has been done already - just copy volumes over.
1091  volumes[0] = p_vol->GetChannelVolume( 0 );
1092  volumes[1] = p_vol->GetChannelVolume( 4 );
1093  volumes[2] = p_vol->GetChannelVolume( 1 );
1094  volumes[3] = p_vol->GetChannelVolume( 2 );
1095  volumes[4] = p_vol->GetChannelVolume( 3 );
1096  break;
1097  }
1098 
1099  case VOLUME_TYPE_BASIC_2_CHANNEL:
1100  {
1101  // Simple two channel sound, so just setup as such.
1102  volumes[0] = p_vol->GetChannelVolume( 0 );
1103  volumes[1] = ( p_vol->GetChannelVolume( 0 ) + p_vol->GetChannelVolume( 1 )) * 0.5f;
1104  volumes[2] = p_vol->GetChannelVolume( 1 );
1105  volumes[3] = p_vol->GetChannelVolume( 0 ) * 0.5f;
1106  volumes[4] = p_vol->GetChannelVolume( 1 ) * 0.5f;
1107  break;
1108  }
1109 
1110  case VOLUME_TYPE_2_CHANNEL_DOLBYII:
1111  default:
1112  {
1113  // Wes should not get any of these.
1114  Dbg_Assert( 0 );
1115 
1116 # if 0
1117  // Get the length of the vector here which will be used to multiply out the normalised speaker volumes.
1118  Mth::Vector test( fabsf( p_vol->GetChannelVolume( 0 )), fabsf( p_vol->GetChannelVolume( 1 )), 0.0f, 0.0f );
1119  float amplitude = test.Length();
1120 
1121  // Look just at the normalized right component to figure the sound angle from Matt's calculations.
1122  test.Normalize();
1123 
1124  float angle;
1125  angle = asinf( test[Y] );
1126  angle = ( angle * 2.0f ) - ( Mth::PI * 0.5f );
1127  angle = ( p_vol->GetChannelVolume( 0 ) < 0.0f ) ? ( Mth::PI - angle ) : angle;
1128 
1129 
1130  // Now figure volumes based on speaker coverage.
1131  angle = Mth::RadToDeg( angle );
1132 
1133  // Left front channel.
1134  if(( angle >= 225.0f ) || ( angle <= 45.0f ))
1135  {
1136  // Because of the discontinuity, shift this angle round to the [0,180] range.
1137  float shift_angle = angle + 135;
1138  shift_angle = ( shift_angle >= 360.0f ) ? ( shift_angle - 360.0f ) : shift_angle;
1139  volumes[0] = ( shift_angle / 180.0f ) * Mth::PI;
1140  volumes[0] = sqrtf( sinf( volumes[0] ));
1141  }
1142 
1143  // Center channel.
1144  if(( angle >= -60.0f ) && ( angle <= 60.0f ))
1145  {
1146  // Scale this into [0,PI] space so we can get smooth fadeout.
1147  volumes[1] = (( angle + 60.0f ) / 120.0f ) * Mth::PI;
1148  volumes[1] = sqrtf( sinf( volumes[1] ));
1149  }
1150 
1151  // Right front channel.
1152  if(( angle >= -45.0f ) && ( angle <= 135.0f ))
1153  {
1154  // Scale this into [0,PI] space so we can get smooth fadeout.
1155  volumes[2] = (( angle + 45.0f ) / 180.0f ) * Mth::PI;
1156  volumes[2] = sqrtf( sinf( volumes[2] ));
1157  }
1158 
1159  // Right rear channel.
1160  if(( angle >= 45.0f ) && ( angle <= 225.0f ))
1161  {
1162  // Scale this into [0,PI] space so we can get smooth fadeout.
1163  volumes[3] = (( angle - 45.0f ) / 180.0f ) * Mth::PI;
1164  volumes[3] = sqrtf( sinf( volumes[3] ));
1165  }
1166 
1167  // Left rear channel.
1168  if(( angle >= 135.0f ) || ( angle <= -45.0f ))
1169  {
1170  // Because of the discontinuity, shift this angle round to the [0,180] range.
1171  float shift_angle = angle + 225;
1172  shift_angle = ( shift_angle >= 360.0f ) ? ( shift_angle - 360.0f ) : shift_angle;
1173  volumes[4] = ( shift_angle / 180.0f ) * Mth::PI;
1174  volumes[4] = sqrtf( sinf( volumes[4] ));
1175  }
1176 
1177  // Scale back up to original amplitude.
1178  for( int v = 0; v < 5; ++v )
1179  {
1180  volumes[v] *= amplitude;
1181  }
1182 # endif
1183  break;
1184  }
1185  }
1186  }
1187 
1188  // Now readjust the relative values...
1189  for( int v = 0; v < 5; ++v )
1190  {
1191  // Adjust for SFX volume level, and scale into limits.
1192  volumes[v] = fabsf( PERCENT( gSfxVolume, volumes[v] ));
1193 
1194  if( volumes[v] > 100.0f )
1195  volumes[v] = 100.0f;
1196  }
1197 
1198  // Now figure the attenuation of the sound. To convert to a decibel value, figure the ratio of requested
1199  // volume versus max volume, then calculate the log10 and multiply by (10 * 2). (The 2 is because sound
1200  // power varies as square of pressure, and squaring doubles the log value).
1201  max_i_volume = DSBVOLUME_EFFECTIVE_MIN;
1202  for( int v = 0; v < 5; ++v )
1203  {
1204  if( volumes[v] > 0.0f )
1205  {
1206  float attenuation = 20.0f * log10f( volumes[v] * 0.01f );
1207  i_volumes[v] = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
1208  if( i_volumes[v] < DSBVOLUME_EFFECTIVE_MIN )
1209  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
1210 
1211  if( i_volumes[v] > max_i_volume )
1212  max_i_volume = i_volumes[v];
1213  }
1214  else
1215  {
1216  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
1217  }
1218  }
1219 
1220  // Set individual mixbins for panning.
1221  DSMIXBINS dsmixbins;
1222  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
1223 
1224  dsmixbins.dwMixBinCount = 0;
1225  dsmixbins.lpMixBinVolumePairs = dsmbvp;
1226 
1227  if( i_volumes[0] > DSBVOLUME_EFFECTIVE_MIN )
1228  {
1229  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_LEFT;
1230  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[0];
1231  dsmixbins.dwMixBinCount++;
1232  }
1233 
1234  if( i_volumes[1] > DSBVOLUME_EFFECTIVE_MIN )
1235  {
1236  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_FRONT_CENTER;
1237  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[1];
1238  dsmixbins.dwMixBinCount++;
1239  }
1240 
1241  if( i_volumes[2] > DSBVOLUME_EFFECTIVE_MIN )
1242  {
1243  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_RIGHT;
1244  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[2];
1245  dsmixbins.dwMixBinCount++;
1246  }
1247 
1248  if( i_volumes[3] > DSBVOLUME_EFFECTIVE_MIN )
1249  {
1250  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_RIGHT;
1251  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[3];
1252  dsmixbins.dwMixBinCount++;
1253  }
1254 
1255  if( i_volumes[4] > DSBVOLUME_EFFECTIVE_MIN )
1256  {
1257  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_LEFT;
1258  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[4];
1259  dsmixbins.dwMixBinCount++;
1260  }
1261 
1262  if( dsmixbins.dwMixBinCount > 0 )
1263  {

1265  dsmbvp[dsmixbins.dwMixBinCount].lVolume = max_i_volume;

1269  dsmbvp[dsmixbins.dwMixBinCount].lVolume = DSBVOLUME_MAX;
1270  dsmixbins.dwMixBinCount++;
1271  }

1274  HRESULT hr = VoiceSimulator[voice].p_buffer->SetMixBinVolumes( &dsmixbins );

1283  hr = VoiceSimulator[voice].p_buffer->SetVolume( DSBVOLUME_EFFECTIVE_MIN );
1284
