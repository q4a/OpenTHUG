207 int sSweepPruneArray::BinarySearchFirstBigger( float value )
208 {
209  int first_bigger = -1;
210  int l = 0;
211  int u = m_num_entries - 1;
212  while( l <= u )
213  {
214  int m = ( l + u ) / 2;
215  if( m_array[m].GetValue( m_axis ) <= value )
216  {
217  // x[m] <= t
218  l = m + 1;
219  }
220  else
221  {
222  // x[m] > t
223  first_bigger = m;
224  u = m - 1;
225  }
226  }
227  return first_bigger;
228 }

261 /******************************************************************/
262 /* */
263 /* */
264 /******************************************************************/
265 int sSweepPruneArray::BuildCompositeIntersectingArray( void )
266 {
267  // Now build the composite array of intersections that appear in all three axis lists.
268  int final_intersections = 0;
269  int source, check1, check2;
270 
271  // Use the smallest list to check against.
272  if(( perAxisIntersections[0] <= perAxisIntersections[1] ) && ( perAxisIntersections[0] <= perAxisIntersections[2] ))
273  {
274  source = 0;
275  check1 = 1;
276  check2 = 2;
277  }
278  else if(( perAxisIntersections[1] <= perAxisIntersections[0] ) && ( perAxisIntersections[1] <= perAxisIntersections[2] ))
279  {
280  source = 1;
281  check1 = 0;
282  check2 = 2;
283  }
284  else
285  {
286  source = 2;
287  check1 = 0;
288  check2 = 1;
289  }
290 
291  for( int i = 0; i < perAxisIntersections[source]; ++i )
292  {
293  bool found;
294 
295  // Is this intersection also in the first check list?
296  found = false;
297  for( int j = 0; j < perAxisIntersections[check1]; ++j )
298  {
299  if( perAxisWorkspace[source][i] == perAxisWorkspace[check1][j] )
300  {
301  found = true;
302  break;
303  }
304  }
305 
306  // If it wasn't in the first check list, there's no intersection.
307  if( !found )
308  {
309  break;
310  }
311 
312  // Is this intersection also in the second check list?
313  found = false;
314  for( int j = 0; j < perAxisIntersections[check2]; ++j )
315  {
316  if( perAxisWorkspace[source][i] == perAxisWorkspace[check2][j] )
317  {
318  found = true;
319  break;
320  }
321  }
322 
323  // If it wasn't in the second check list, there's no intersection.
324  if( !found )
325  {
326  break;
327  }
328 
329  // At this stage the intersection appeared in all three axis lists, so it is a valid intersection.
330  // As a final check, make sure it isn't already in the final list.
331  found = false;
332  for( int j = 0; j < final_intersections; ++j )
333  {
334  if( combinedAxisWorkspace[j] == perAxisWorkspace[source][i] )
335  {
336  found = true;
337  break;
338  }
339  }
340 
341  if( !found )
342  {
343  combinedAxisWorkspace[final_intersections] = perAxisWorkspace[source][i];
344  ++final_intersections;
345  }
346  }
347  return final_intersections;
348 }

356 CNearComponent** CNearComponent::GetIntersectingNearComponents( Mth::Vector &bbmin, Mth::Vector &bbmax, int *p_num_intersecting )
357 {
358  // Zero axis intersection count array.
359  perAxisIntersections[0] = perAxisIntersections[1] = perAxisIntersections[2] = 0;
360 
361  for( int axis = 0; axis < 3; ++axis )
362  {
363  // Find the start index.
364  int start_index = m_sweep_prune_array[axis].BinarySearchFirstBigger( bbmin[axis] );
365  if( start_index == -1 )
366  {
367  // Nothing was found, therefore there can be no intersections.
368  *p_num_intersecting = 0;
369  return NULL;
370  }
371 
372  // Find the end index.
373  int end_index = m_sweep_prune_array[axis].BinarySearchLastSmaller( bbmax[axis] );
374  if( end_index == -1 )
375  {
376  // Nothing was found, therefore there can be no intersections.
377  *p_num_intersecting = 0;
378  return NULL;
379  }
380 
381  Dbg_Assert( end_index > start_index );
382 
383  // Build up the workspace array for this axis.
384  int intersections_for_this_axis = 0;
385  for( int i = start_index; i <= end_index; ++i )
386  {
387  // Don't want to include the calling component in the list.
388  if( m_sweep_prune_array[axis].m_array[i].mp_component != this )
389  {
390  perAxisWorkspace[axis][intersections_for_this_axis++] = m_sweep_prune_array[axis].m_array[i].mp_component;
391  }
392  }
393 
394  // If we found no intersections for this axis, there can be no components that intersect.
395  if( intersections_for_this_axis == 0 )
396  {
397  *p_num_intersecting = 0;
398  return NULL;
399  }
400 
401  // Store off the number of intersections for this axis.
402  perAxisIntersections[axis] = intersections_for_this_axis;
403 
404  }
405 
406  // Build the composite array.
407  int final_intersections = sSweepPruneArray::BuildCompositeIntersectingArray();
408 
409  *p_num_intersecting = final_intersections;
410  return ( final_intersections > 0 ) ? &combinedAxisWorkspace[0] : NULL;
411 }

419 CNearComponent** CNearComponent::GetIntersectingNearComponents( int *p_num_intersecting )
420 {
421  // Zero axis intersection count array.
422  perAxisIntersections[0] = perAxisIntersections[1] = perAxisIntersections[2] = 0;
423 
424  // Build up each workspace array.
425  for( int axis = 0; axis < 3; ++axis )
426  {
427  int i;
428  for( i = 0; i < m_sweep_prune_array[axis].m_num_entries; ++i )
429  {
430  if( m_sweep_prune_array[axis].m_array[i].mp_component == this )
431  {
432  // We have found the start of the span for this component.
433  Dbg_Assert( m_sweep_prune_array[axis].m_array[i].m_min_max == 0 );
434  ++i;
435  break;
436  }
437  }
438 
439  // Now add any other components we come across before we hit the end of the span.
440  while(( i < m_sweep_prune_array[axis].m_num_entries ) && ( m_sweep_prune_array[axis].m_array[i].mp_component != this ))
441  {
442  perAxisWorkspace[axis][perAxisIntersections[axis]] = m_sweep_prune_array[axis].m_array[i].mp_component;
443  ++perAxisIntersections[axis];
444  ++i;
445  }
446 
447  // If we found no intersections for this axis, there can be no components that intersect.
448  if( perAxisIntersections[axis] == 0 )
449  {
450  *p_num_intersecting = 0;
451  return NULL;
452  }
453  }
454 
455  // Build the composite array.
456  int final_intersections = sSweepPruneArray::BuildCompositeIntersectingArray();
457 
458  *p_num_intersecting = final_intersections;
459  return ( final_intersections > 0 ) ? &combinedAxisWorkspace[0] : NULL;
460 }

537 /* */
538 /* */
539 /******************************************************************/
540 void sSweepPruneArray::ResortComponent( CNearComponent *p_component )
541 {
542  // Does a simple check to see whether the component is still in the correct order.
543  // If it isn't, sorting is applied.
544 
545  // Check min first.
546  int i;
547  for( i = 0; i < m_num_entries; ++i )
548  {
549  if( m_array[i].mp_component == p_component )
550  {
551  // This should be the min entry.
552  Dbg_Assert( m_array[i].m_min_max == 0 );
553 
554  // Set the new min value.
555  m_array[i].SetValues();
556 
557  // Check this is bigger than or equal to the previous value, if there is one.
558  while(( i > 0 ) && ( m_array[i].GetValue( m_axis ) < m_array[i - 1].GetValue( m_axis )))
559  {
560  // Move this entry down in the array.
561  sSweepPruneEntry swap = m_array[i];
562  m_array[i] = m_array[i - 1];
563  m_array[i - 1] = swap;
564  --i;
565  }
566 
567  // Check this is smaller than or equal to the next value, if there is one.
568  while(( i < ( m_num_entries - 1)) && ( m_array[i].GetValue( m_axis ) > m_array[i + 1].GetValue( m_axis )))
569  {
570  // Move this entry up in the array.
571  sSweepPruneEntry swap = m_array[i];
572  m_array[i] = m_array[i + 1];
573  m_array[i + 1] = swap;
574  ++i;
575  }
576  break;
577  }
578  }
579 
580  // Now check max.
581  ++i;
582  for( ; i < m_num_entries; ++i )
583  {
584  if( m_array[i].mp_component == p_component )
585  {
586  // This should be the max entry.
587  Dbg_Assert( m_array[i].m_min_max == 1 );
588 
589  // Set the new max value.
590  m_array[i].SetValues();
591 
592  // Check this is bigger than or equal to the previous value, if there is one.
593  while(( i > 0 ) && ( m_array[i].GetValue( m_axis ) < m_array[i - 1].GetValue( m_axis )))
594  {
595  // Move this entry down in the array.
596  sSweepPruneEntry swap = m_array[i];
597  m_array[i] = m_array[i - 1];
598  m_array[i - 1] = swap;
599  --i;
600  }
601 
602  // Check this is smaller than or equal to the next value, if there is one.
603  while(( i < ( m_num_entries - 1)) && ( m_array[i].GetValue( m_axis ) > m_array[i + 1].GetValue( m_axis )))
604  {
605  // Move this entry up in the array.
606  sSweepPruneEntry swap = m_array[i];
607  m_array[i] = m_array[i + 1];
608  m_array[i + 1] = swap;
609  ++i;
610  }
611  break;
612  }
613  }
614 }

622 void sSweepPruneArray::InsertComponent( CNearComponent *p_component )
623 {
624  float axis_min = p_component->GetMin()[m_axis];
625  float axis_max = p_component->GetMax()[m_axis];
626 
627  // Scan through until we find an entry that is bigger than the minimum for this axis, or we come to the end of the list.
628  for( int i = 0; i <= m_num_entries; ++i )
629  {
630  if(( i == m_num_entries ) || ( m_array[i].mp_component->GetMin()[m_axis] > axis_min ))
631  {
632  // Insert this entry here.
633  PushUp( i );
634  m_array[i].m_min_max = 0;
635  m_array[i].mp_component = p_component;
636 
637  // Set the new min value.
638  m_array[i].SetValues();
639  break;
640  }
641  }
642 
643  // Now scan through until we find an entry that is bigger than the maximum for this axis, or we come to the end of the list.
644  for( int i = 0; i <= m_num_entries; ++i )
645  {
646  if(( i == m_num_entries ) || ( m_array[i].mp_component->GetMax()[m_axis] > axis_max ))
647  {
648  // Insert this entry here.
649  PushUp( i );
650  m_array[i].m_min_max = 1;
651  m_array[i].mp_component = p_component;
652 
653  // Set the new min value.
654  m_array[i].SetValues();
655  break;
656  }
657  }
658 }
