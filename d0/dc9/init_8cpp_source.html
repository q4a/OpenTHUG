131 // Adds all the c-functions listed in the passed array to the symbol table.
132 void RegisterCFunctions(SCFunction *p_cFunctions, uint32 numFunctions)
133 {
134  Dbg_MsgAssert(p_cFunctions,("NULL p_cFunctions"));
135 
136  for (uint32 i=0; i<numFunctions; ++i)
137  {
138  uint32 name_checksum=Crc::GenerateCRCFromString(p_cFunctions[i].mpName);
139 
140  AddChecksumName(name_checksum,p_cFunctions[i].mpName);
141 
142  // Check that there is no symbol already with this name.
143  #ifdef __NOPT_ASSERT__
144  CSymbolTableEntry *p_existing_entry=LookUpSymbol(name_checksum);
145  Dbg_MsgAssert(p_existing_entry==NULL,(
146  "The C-function named '%s' is already defined in the file %s where it has type '%s'",
147  p_cFunctions[i].mpName,
148  p_existing_entry->mType==ESYMBOLTYPE_CFUNCTION || p_existing_entry->mType==ESYMBOLTYPE_MEMBERFUNCTION ? "ftables.cpp":FindChecksumName(p_existing_entry->mSourceFileNameChecksum),
149  GetTypeName(p_existing_entry->mType)));
150  #endif
151 
152  CSymbolTableEntry *p_new=CreateNewSymbolEntry(name_checksum);
153  Dbg_MsgAssert(p_new,("NULL p_new ??"));
154 
155  p_new->mType=ESYMBOLTYPE_CFUNCTION;
156  p_new->mpCFunction=p_cFunctions[i].mpFunction;
157  // Note: Not setting p_new->mSourceFileNameChecksum to anything, just leaving it as 0, since there is no
158  // source qb file.
159  }
160 }
