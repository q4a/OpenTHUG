330 static inline bool line_segment_intersection( float x1, float y1, float x2, float y2, float x3, float y3, float x4, float y4 )
331 {
332  float den = (( y4 - y3 ) * ( x2 - x1 )) - (( x4 - x3 ) * ( y2 - y1 ));
333 
334  if( den == 0.0f )
335  {
336  // Parallel lines.
337  return false;
338  }
339 
340  float num_a = (( x4 - x3 ) * ( y1 - y3 )) - (( y4 - y3 ) * ( x1 - x3 ));
341  float num_b = (( x2 - x1 ) * ( y1 - y3 )) - (( y2 - y1 ) * ( x1 - x3 ));
342 
343  num_a /= den;
344  num_b /= den;
345 
346  if(( num_a <= 1.0f ) && ( num_b <= 1.0f ))
347  return true;
348 
349  return false;
350 }

356 static inline bool tri_texture_intersect( float u0, float v0, float u1, float v1, float u2, float v2 )
357 {
358  // Trivial check to see if all three points are outside range of texture.
359  if(( u0 < -1.0f ) && ( u1 < -1.0f ) && ( u2 < -1.0f ))
360  return false;
361  if(( u0 > 1.0f ) && ( u1 > 1.0f ) && ( u2 > 1.0f ))
362  return false;
363  if(( v0 < -1.0f ) && ( v1 < -1.0f ) && ( v2 < -1.0f ))
364  return false;
365  if(( v0 > 1.0f ) && ( v1 > 1.0f ) && ( v2 > 1.0f ))
366  return false;
367 
368  // Check that at least one corner of the texture falls within the tri.
369  Mth::Vector texture_square[4] = { Mth::Vector( -1.0f, -1.0f, 0.0f, 0.0f ),
370  Mth::Vector( 1.0f, -1.0f, 0.0f, 0.0f ),
371  Mth::Vector( 1.0f, 1.0f, 0.0f, 0.0f ),
372  Mth::Vector( -1.0f, 1.0f, 0.0f, 0.0f )};
373 
374  Mth::Vector a( u0, v0, 0.0f );
375  Mth::Vector b( u1, v1, 0.0f );
376  Mth::Vector c( u2, v2, 0.0f );
377 
378  for( int p = 0; p < 4; ++p )
379  {
380  if( point_in_triangle( texture_square[p], a, b, c ))
381  {
382  return true;
383  }
384  }
385 
386  // No corners of the texture fall within the tri. There are 3 possible explanations:
387  // 1) The tri intersects the texture, but does not contain a texture corner.
388  // 2) The texture lies entirely outside of the tri.
389  // 3) The tri falls entirely within the texture.
390  // Given the relatively small size of the textures, case (3) is extremely unlikely.
391 
392  // Perform a trivial check to see whether a corner of the tri lies within the texture. This will catch (3) and sometimes (1).
393  if(( u0 >= -1.0f ) && ( u0 <= 1.0f ) && ( v0 >= -1.0f ) && ( v0 <= 1.0f ))
394  return true;
395  if(( u1 >= -1.0f ) && ( u1 <= 1.0f ) && ( v1 >= -1.0f ) && ( v1 <= 1.0f ))
396  return true;
397  if(( u2 >= -1.0f ) && ( u2 <= 1.0f ) && ( v2 >= -1.0f ) && ( v2 <= 1.0f ))
398  return true;
399 
400  // Perform the complete check to see if any line segment forming the tri intersects any line segment forming the texture.
401  for( int p = 0; p < 4; ++p )
402  {
403  int q = ( p + 1 ) % 4;
404  if( line_segment_intersection( u0, v0, u1, v1, texture_square[p][X], texture_square[p][Y], texture_square[q][X], texture_square[q][Y] ))
405  return true;
406  }
407 
408  return false;
409 }
