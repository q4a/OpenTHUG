36 bool CPs2Mesh::build_casdata_table(uint8* p_casData)
37 {
38  #ifdef __NOPT_ASSERT__
39  uint32 version = *((uint32*)p_casData);
40  #endif
41  p_casData += sizeof(uint32);
42 
43  Dbg_MsgAssert( version >= 2, ( "Obsolete version of CAS file. Please re-export." ) )
44 
45  m_CASRemovalMask = *((uint32*)p_casData);
46  p_casData += sizeof(uint32);
47 
48  m_numCASData = *((int*)p_casData);
49  p_casData += sizeof(int);
50 
51  // temporary lookup table, for the duration of the scene loading
52  NxPs2::CreateCASDataLookupTable();
53 
54  if ( m_numCASData > 0 )
55  {
56  // create-a-skater flags
57  mp_CASData = new NxPs2::sCASData[m_numCASData];
58 
59  for ( int i = 0; i < m_numCASData; i++ )
60  {
61  mp_CASData[i].mask = *((uint32*)p_casData);
62  p_casData += sizeof(uint32);
63 
64  // this will get converted into a pointer to the DMA data
65  int vertIndex = *((int*)p_casData);
66  p_casData += sizeof(int);
67 
68  NxPs2::SetCASDataLookupData( vertIndex, &mp_CASData[i] );
69  }
70  }
71 
72  return ( m_numCASData > 0 );
73 }

80 bool CPs2Mesh::build_casdata_table(const char* pFileName)
81 {
82  void *pFile = File::Open(pFileName, "rb");
83  Dbg_MsgAssert(pFile, ("Couldn't open CAS data file %s\n", pFileName));
84 
85  int file_size = File::GetFileSize(pFile);
86 
87  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
88  uint8* p_fileBuffer = (uint8*)Mem::Malloc( file_size );
89  Mem::Manager::sHandle().PopContext();
90 
91  File::Read(p_fileBuffer, file_size, 1, pFile);
92  File::Close( pFile );
93 
94  bool success = build_casdata_table( p_fileBuffer );
95 
96  Mem::Free(p_fileBuffer);
97 
98  // some debugging information for the artists
99  if ( Script::GetInteger( "DebugCAS", Script::NO_ASSERT ) )
100  {
101  printf( "%s removes %08x: [", pFileName, m_CASRemovalMask );
102  for ( int i = 0; i < 32; i++ )
103  {
104  uint32 mask = ( 1 << i );
105  if ( m_CASRemovalMask & mask )
106  {
107  printf( " %d", i );
108  }
109  }
110  printf( "]\n" );
111  }
112 
113  return success;
114 }

146 CPs2Mesh::CPs2Mesh(uint32* pModelData, int modelDataSize, uint8* pCASData, Nx::CTexDict* pTexDict, uint32 texDictOffset, bool isSkin, bool doShadowVolume)
147 {
148  Dbg_Assert( pTexDict );
149 
150  mp_texDict = pTexDict;
151 
152  // polys should only be hidden once
153  m_polysAlreadyHidden = false;
154 
155  m_isPipped = false;
156  m_numCASData = 0;
157 
158  // don't do this for models
159  if ( isSkin && pCASData )
160  {
161  build_casdata_table( pCASData );
162  }
163 
164  // GJ: The geomnodes blow off the texDictOffset,
165  // which means it won't work with skater models,
166  // or the 10 board gameobjs in the boardshop.
167  if ( !isSkin && texDictOffset == 0 )
168  {
169  // since we're not actually opening a PIP file
170  // we don't need to keep track of it
171  m_isPipped = false;
172 
173  // this should already be in GEOM format,
174  // by the time the data arrives here...
175  // (we're not loading up the Pip through the
176  // normal Pip::Load process, so we need
177  // to remember the pointer to the pip data
178  // so that we can delete it later)
179  mp_pipData = (uint8*)Mem::Malloc( modelDataSize );
180  Dbg_Assert( mp_pipData );
181 
182  memcpy( mp_pipData, pModelData, modelDataSize );
183 
184  // process data in place
185  // final arg (0) says to hold it in a database to be rendered later
186  // (as opposed to rendering it unconditionally, LOADFLAG_RENDERNOW)
187  mp_geomNode = NxPs2::CGeomNode::sProcessInPlace( mp_pipData, 0 );
188 
189  // Get hierarchy array
190  mp_hierarchyObjects = (CHierarchyObject*) NxPs2::CGeomNode::sGetHierarchyArray(mp_pipData, m_numHierarchyObjects);
191  }
192  else
193  {
194  NxPs2::sScene* pTextureDictionary = ((Nx::CPs2TexDict*)mp_texDict)->GetEngineTextureDictionary();
195 
196  NxPs2::LoadScene( pModelData, modelDataSize, pTextureDictionary, true, true, texDictOffset, doShadowVolume );
197 
198  mp_geomNode = NULL;
199  }
200 
201  // temporary lookup table is no longer needed
202  NxPs2::DestroyCASDataLookupTable();
203 }

332 void CPs2Mesh::HidePolys( uint32 mask )
333 {
334  int count = 0;
335 
336  // this function should only be called once per mesh
337  // because after the ADC bits are squeezed,
338  // the pointers are no longer valid...
339  // (was causing a crash in the secret level
340  // with random peds that were doing poly removal)
341  if ( m_polysAlreadyHidden )
342  {
343  return;
344  }
345 
346 // printf( "Hiding polys with mask 0x%08x\n", mask);
347 
348  for ( int j = 0; j < m_numCASData; j++ )
349  {
350  Dbg_Assert( mp_CASData );
351 
352  if ( mp_CASData[j].mask & mask )
353  {
354  if ( mp_CASData[j].pADCBit )
355  {
356  *mp_CASData[j].pADCBit = 0xC000;
357  }
358  count++;
359  }
360  }
361 
362 // printf( "Hiding %d polys\n", count );
363 
364  m_polysAlreadyHidden = true;
365 }
