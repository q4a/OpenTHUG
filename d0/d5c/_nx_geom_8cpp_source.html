555 CGeom* CGeom::Clone(bool instance, CScene* pDestScene)
556 {
557 #ifdef __PLAT_NGPS__
558  // Garrett: Want to be able to disable copy just in case it opens up crashes on the Park Editor
559  if (Script::GetInt( "disable_clone_copy", false ))
560  {
561  instance = true;
562  }
563 #endif
564 
565  // Create new instance and clone stuff below p-line
566  CGeom* p_new_geom = plat_clone(instance, pDestScene);
567 
568  if (p_new_geom)
569  {
570  p_new_geom->m_cloned = (instance) ? vINSTANCE : vCOPY;
571 
572  if (mp_coll_tri_data)
573  {
574 #if 1
575  // Garrett: Just copying the pointer now, since the CGeom version is just used for reference
576  p_new_geom->mp_coll_tri_data = mp_coll_tri_data;
577 #else
578  // Make NULL for now until we can figure out a better way of cloning collision with CSector
579  p_new_geom->mp_coll_tri_data = NULL;
580  //p_new_geom->mp_coll_tri_data = mp_coll_tri_data->Clone(instance);
581  //Dbg_Assert(p_new_geom->mp_coll_tri_data);
582 #endif
583  }
584  // Mick: bit of a patch - attempt to fix crash from deleting thigns twice
585  p_new_geom->mp_orig_render_colors = NULL;
586  }
587 
588  return p_new_geom;
589 }
