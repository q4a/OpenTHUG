45 CPs2ParticleNewFlat::CPs2ParticleNewFlat( uint32 checksum, int max_streams, uint32 texture_checksum, uint32 blendmode_checksum, int fix )
46 {
47  m_checksum = checksum;
48  m_max_streams = max_streams;
49  m_num_streams = 0;
50 
51  // Get the texture.
52  Nx::CTexture *p_texture;
53  Nx::CPs2Texture *p_ps2_texture;
54  mp_engine_texture = NULL;
55 
56  p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( texture_checksum );
57  p_ps2_texture = static_cast<Nx::CPs2Texture*>( p_texture );
58  if ( p_ps2_texture )
59  {
60  mp_engine_texture = p_ps2_texture->GetSingleTexture();
61  }
62 
63  // Set blendmode.
64  m_blend = NxPs2::CImmediateMode::sGetTextureBlend( blendmode_checksum, fix );
65 
66  // Allocate space for span information & streams.
67  m_num_spans = 1; // Currently defaulting to 1 span.
68  mp_span = new CParticleSpan[m_num_spans];
69  mp_stream = new CParticleStream[m_max_streams];
70 
71  mp_newest_stream = mp_stream + m_max_streams - 1;
72  mp_oldest_stream = mp_stream;
73  m_life = 0.5f * (m_life_min + m_life_max);
74  m_emitting = false;
75 
76 
77  // set up system dma data, apart from new coefficients, which are calculated in plat_build_path()
78  m_systemDmaData.m_GScontext[0] = 0x00008004;
79  m_systemDmaData.m_GScontext[1] = 0x10000000;
80  m_systemDmaData.m_GScontext[2] = 0x0000000E;
81  m_systemDmaData.m_GScontext[3] = 0x00000000;
82 
83  *(uint64 *)&m_systemDmaData.m_GScontext[4] = m_blend;
84  m_systemDmaData.m_GScontext[6] = 0x00000042;
85 
86  if (mp_engine_texture)
87  {
88  m_systemDmaData.m_u0 = 8;
89  m_systemDmaData.m_v0 = (mp_engine_texture->GetHeight()<<4) - 8;
90  m_systemDmaData.m_u1 = (mp_engine_texture->GetWidth() <<4) - 8;
91  m_systemDmaData.m_v1 = 8;
92  //m_systemDmaData.m_GScontext[16] = ((mp_engine_texture->GetHeight()<<4) - 8) << 16 | 8;
93 
94  *(uint64 *)&m_systemDmaData.m_GScontext[8] = mp_engine_texture->m_RegTEX0;
95  *(uint64 *)&m_systemDmaData.m_GScontext[12] = mp_engine_texture->m_RegTEX1;
96  }
97  m_systemDmaData.m_GScontext[10] = 0x00000006;
98  m_systemDmaData.m_GScontext[14] = 0x00000014;
99 
100  m_systemDmaData.m_GScontext[16] = 0x0005000B /* | m_alphaCutoff<<4 */; //101 0000 xxxx xxxx 1011
101  m_systemDmaData.m_GScontext[18] = 0x00000047;
102 
103  m_systemDmaData.m_tagy = 0x60AB4000;
104  m_systemDmaData.m_tagz = 0x00434312;
105 
106 
107  m_mid_time = -1.0f;
108 }

175 void CPs2ParticleNewFlat::plat_render( void )
176 {
177  CParticleStream *p_stream;
178  int i;
179 
180 
181  //--------------------------------------------------------------------------------------------------------------------
182  // process the streams
183  if (m_emit_rate && (!m_emitting || (m_emit_rate != mp_newest_stream->m_rate)))
184  {
185  if (m_num_streams < m_max_streams)
186  {
187  // add new stream to cyclic buffer
188  m_num_streams++;
189  mp_newest_stream++;
190  if (mp_newest_stream == mp_stream + m_max_streams)
191  {
192  mp_newest_stream = mp_stream;
193  }
194 
195  // initialise new stream
196  mp_newest_stream->m_rate = m_emit_rate;
197  mp_newest_stream->m_interval = 1.0f/m_emit_rate;
198  mp_newest_stream->m_oldest_age = 0.0f;
199  mp_newest_stream->m_num_particles = 0;
200  mp_newest_stream->m_seed[0] = rand();
201  mp_newest_stream->m_seed[1] = rand();
202  mp_newest_stream->m_seed[2] = rand();
203  mp_newest_stream->m_seed[3] = rand();
204  m_emitting = true;
205  }
206  else
207  {
208  m_emitting = false;
209  }
210  }
211  else
212  {
213  m_emitting = m_emit_rate;
214  }
215 
216  if (!m_num_streams)
217  return;
218 
219  // age all streams
220  for (i=0, p_stream=mp_oldest_stream; i<m_num_streams; i++)
221  {
222  // increase age of oldest particle
223  p_stream->m_oldest_age += 1.0f/60.0f;
224 
225  // step pointer within cyclic buffer
226  p_stream++;
227  if (p_stream == mp_stream + m_max_streams)
228  {
229  p_stream = mp_stream;
230  }
231  }
232 
233  // births into newest stream
234  if (m_emitting)
235  {
236  // how many particles so far emitted?
237  mp_newest_stream->m_num_particles = (int)(mp_newest_stream->m_oldest_age * mp_newest_stream->m_rate + 1.0f);
238  }
239 
240  // deaths from oldest stream
241  if (mp_oldest_stream->m_oldest_age > m_life)
242  {
243  // work out number dead
244  int particles_dead = (int)((mp_oldest_stream->m_oldest_age - m_life) * mp_oldest_stream->m_rate + 1.0f);
245 
246  // remove dead particles
247  mp_oldest_stream->m_num_particles -= particles_dead;
248 
249  // should we keep processing the oldest stream?
250  if (mp_oldest_stream->m_num_particles>0 || (m_num_streams==1 && m_emitting))
251  {
252  // adjust age of oldest particle
253  mp_oldest_stream->m_oldest_age -= (float)particles_dead * mp_oldest_stream->m_interval;
254 
255  // advance seed
256  mp_oldest_stream->AdvanceSeed(particles_dead);
257  }
258  else
259  {
260  // remove oldest stream and wrap in cyclic buffer if necessary
261  m_num_streams--;
262  mp_oldest_stream++;
263  if (mp_oldest_stream == mp_stream + m_max_streams)
264  {
265  mp_oldest_stream = mp_stream;
266  }
267  if (!m_num_streams)
268  return;
269  }
270  }
271 
272 
273  //--------------------------------------------------------------------------------------------------------------------
274  // now render the streams
275 
276  if (mp_engine_texture)
277  {
278  // Mick: If this is textured, we reset the TEX0, TEX1 regs as the texture might move
279  // Since 2D textures are dynamically packed every frame
280  *(uint64 *)&m_systemDmaData.m_GScontext[8] = mp_engine_texture->m_RegTEX0;
281  *(uint64 *)&m_systemDmaData.m_GScontext[12] = mp_engine_texture->m_RegTEX1;
282  }
283 
284  // matrix is system rotation times view transform
285  m_systemDmaData.m_matrix = m_rotation * NxPs2::render::WorldToIntViewport;
286 
287  // in the per-frame setup dma we must have:
288  // dma tag
289  // base
290  // offset
291 
292  // set the group
293  NxPs2::dma::SetList(NxPs2::sGroup::pParticles);
294 
295  // the system and streams will be loaded to a double-buffered input area of VUMem1
296  // ref the system data, and include an unpack for the system and streams:
297  NxPs2::dma::Tag(NxPs2::dma::ref, (sizeof(CSystemDmaDataFlat)>>4), (uint)&m_systemDmaData);
298  NxPs2::vif::STCYCL(1,1);
299  NxPs2::vif::UNPACK(0, V4_32, (sizeof(CSystemDmaDataFlat)>>4)+m_num_streams*2, REL, SIGNED, 0);
300 
301  // construct a packet with data for each stream
302  NxPs2::dma::BeginTag(NxPs2::dma::cnt, 0);
303  for (i=0,p_stream=mp_oldest_stream; i<m_num_streams; i++,p_stream++)
304  {
305  Dbg_MsgAssert(m_num_particles<65536, ("particle limit reached (65536)"));
306 
307  // wrap at end of cyclic buffer
308  if (p_stream == mp_stream + m_max_streams)
309  {
310  p_stream = mp_stream;
311  }
312 
313  ((float *)NxPs2::dma::pLoc)[0] = p_stream->m_oldest_age;
314  ((float *)NxPs2::dma::pLoc)[1] = p_stream->m_interval;
315  ((uint32 *)NxPs2::dma::pLoc)[2] = p_stream->m_num_particles;
316  ((uint32 *)NxPs2::dma::pLoc)[3] = (p_stream==mp_newest_stream) ? 0x8000 : 0;
317  ((uint32 *)NxPs2::dma::pLoc)[4] = p_stream->m_seed[0];
318  ((uint32 *)NxPs2::dma::pLoc)[5] = p_stream->m_seed[1];
319  ((uint32 *)NxPs2::dma::pLoc)[6] = p_stream->m_seed[2];
320  ((uint32 *)NxPs2::dma::pLoc)[7] = p_stream->m_seed[3];
321 
322  //printf("stream %d, num particles=%d, oldest age=%f\n", s, p_stream->m_num_particles, p_stream->m_oldest_age);
323 
324  // step dma pointer
325  NxPs2::dma::pLoc += 8*4;
326 
327  // count particles
328  NxPs2::render::sTotalNewParticles += p_stream->m_num_particles;
329  }
330 
331  NxPs2::vif::MSCAL(10); // sprites
332  NxPs2::dma::EndTag();
333  NxPs2::dma::SetList(NxPs2::sGroup::pEpilogue);
334 }

342 void CPs2ParticleNewFlat::plat_build_path( void )
343 {
344  float midTime;
345 
346  // set m_life
347  m_life = 0.5f * (m_life_min + m_life_max);
348 
349  // set clamped local midtime variable
350  if (m_mid_time <= 0.0f)
351  {
352  midTime = 0.0f;
353  }
354  else if (m_mid_time >= 100.0f)
355  {
356  midTime = m_life;
357  }
358  else
359  {
360  midTime = m_mid_time * 0.01f * m_life;
361  }
362 
363  // x-component will be overwritten by vu1 code, so can store midtime there
364  *(float *)&m_systemDmaData.m_tagx = midTime;
365 
366 
367  // constant part
368  m_systemDmaData.m_s0x = m_emit_w;
369  m_systemDmaData.m_s0y = 0.0f;
370  m_systemDmaData.m_s0z = m_emit_h;
371  m_systemDmaData.m_s0w = 0.0f;
372 
373  m_systemDmaData.m_c0x = m_pos[0] - 1.5f * m_systemDmaData.m_s0x;
374  m_systemDmaData.m_c0y = m_pos[1] - 1.5f * m_systemDmaData.m_s0y;
375  m_systemDmaData.m_c0z = m_pos[2] - 1.5f * m_systemDmaData.m_s0z;
376  m_systemDmaData.m_c0w = m_sw - 1.5f * m_systemDmaData.m_s0w;
377 
378 
379  // linear part
380  Mth::Vector vel(m_tx,m_ty,m_tz,0.0f);
381  float speed = 60.0f * 0.5 * (m_speed_min + m_speed_max);
382  vel.Normalize();
383  vel *= speed;
384  m_systemDmaData.m_s1x = speed * Mth::PI/2.0f * m_emit_angle_spread;
385  m_systemDmaData.m_s1y = 60.0f * (m_speed_max - m_speed_min);
386  m_systemDmaData.m_s1z = speed * Mth::PI/2.0f * m_emit_angle_spread;
387  m_systemDmaData.m_s1w = 0.0f;
388 
389  m_systemDmaData.m_c1x = vel[0] - 1.5f * m_systemDmaData.m_s1x;
390  m_systemDmaData.m_c1y = vel[1] - 1.5f * m_systemDmaData.m_s1y;
391  m_systemDmaData.m_c1z = vel[2] - 1.5f * m_systemDmaData.m_s1z;
392  m_systemDmaData.m_c1w = (m_ew-m_sw) / m_life - 1.5f * m_systemDmaData.m_s1w;
393 
394 
395  // quadratic part
396  m_systemDmaData.m_s2x = 0.0f;
397  m_systemDmaData.m_s2y = 0.0f;
398  m_systemDmaData.m_s2z = 0.0f;
399  m_systemDmaData.m_s2w = 0.0f;
400 
401  m_systemDmaData.m_c2x = m_fx*0.5f*(60.0f*60.0f) - 1.5f * m_systemDmaData.m_s2x;
402  m_systemDmaData.m_c2y = m_fy*0.5f*(60.0f*60.0f) - 1.5f * m_systemDmaData.m_s2y;
403  m_systemDmaData.m_c2z = m_fz*0.5f*(60.0f*60.0f) - 1.5f * m_systemDmaData.m_s2z;
404  m_systemDmaData.m_c2w = 0.0f - 1.5f * m_systemDmaData.m_s2w;
405 
406 
407  // cubic part
408  //m_systemDmaData.m_s3x = 0.0f;
409  //m_systemDmaData.m_s3y = 0.0f;
410  //m_systemDmaData.m_s3z = 0.0f;
411  //m_systemDmaData.m_s3w = 0.0f;
412 
413  //m_systemDmaData.m_c3x = 0.0f - 1.5f * m_systemDmaData.m_s3x;
414  //m_systemDmaData.m_c3y = 0.0f - 1.5f * m_systemDmaData.m_s3y;
415  //m_systemDmaData.m_c3z = 0.0f - 1.5f * m_systemDmaData.m_s3z;
416  //m_systemDmaData.m_c3w = 0.0f - 1.5f * m_systemDmaData.m_s3w;
417 
418 
419  // colour
420  if (m_mid_time >= 0.0f) // if using mid-colour
421  {
422  float q0 = 1.0f / midTime;
423  float q1 = 1.0f / (m_life - midTime);
424 
425  m_systemDmaData.m_c0r = ((float)m_mid_color.r - (float)m_start_color.r) * q0;
426  m_systemDmaData.m_c0g = ((float)m_mid_color.g - (float)m_start_color.g) * q0;
427  m_systemDmaData.m_c0b = ((float)m_mid_color.b - (float)m_start_color.b) * q0;
428  m_systemDmaData.m_c0a = ((float)m_mid_color.a - (float)m_start_color.a) * q0;
429 
430  m_systemDmaData.m_c1r = (float)m_mid_color.r;
431  m_systemDmaData.m_c1g = (float)m_mid_color.g;
432  m_systemDmaData.m_c1b = (float)m_mid_color.b;
433  m_systemDmaData.m_c1a = (float)m_mid_color.a;
434 
435  m_systemDmaData.m_c2r = ((float)m_end_color.r - (float)m_mid_color.r) * q1;
436  m_systemDmaData.m_c2g = ((float)m_end_color.g - (float)m_mid_color.g) * q1;
437  m_systemDmaData.m_c2b = ((float)m_end_color.b - (float)m_mid_color.b) * q1;
438  m_systemDmaData.m_c2a = ((float)m_end_color.a - (float)m_mid_color.a) * q1;
439  }
440  else // else suppress mid-colour
441  {
442  float q = 1.0f / m_life;
443 
444  m_systemDmaData.m_c1r = (float)m_start_color.r;
445  m_systemDmaData.m_c1g = (float)m_start_color.g;
446  m_systemDmaData.m_c1b = (float)m_start_color.b;
447  m_systemDmaData.m_c1a = (float)m_start_color.a;
448 
449  m_systemDmaData.m_c2r = ((float)m_end_color.r - (float)m_start_color.r) * q;
450  m_systemDmaData.m_c2g = ((float)m_end_color.g - (float)m_start_color.g) * q;
451  m_systemDmaData.m_c2b = ((float)m_end_color.b - (float)m_start_color.b) * q;
452  m_systemDmaData.m_c2a = ((float)m_end_color.a - (float)m_start_color.a) * q;
453  }
454 
455 
456  // rotation matrix, hardwired to the identity for now
457  m_rotation.Identity();
458 
459  // temp test
460  //m_rotation[0][0] = 0.8;
461  //m_rotation[0][1] = 0.6;
462  //m_rotation[1][0] = -0.6;
463  //m_rotation[1][1] = 0.8;
464 
465  // invert rotation and apply to spatial params
466  // leaving this code a bit shoddy and slow until full transition to new-style params
467  Mth::Matrix mat;
468  mat=m_rotation;
469  mat.Transpose();
470  Mth::Vector vec;
471 
472  vec[0] = m_systemDmaData.m_c0x + 1.5f * m_systemDmaData.m_s0x;
473  vec[1] = m_systemDmaData.m_c0y + 1.5f * m_systemDmaData.m_s0y;
474  vec[2] = m_systemDmaData.m_c0z + 1.5f * m_systemDmaData.m_s0z;
475  vec[3] = 0;
476  vec *= mat;
477  m_systemDmaData.m_c0x = vec[0] - 1.5f * m_systemDmaData.m_s0x;
478  m_systemDmaData.m_c0y = vec[1] - 1.5f * m_systemDmaData.m_s0y;
479  m_systemDmaData.m_c0z = vec[2] - 1.5f * m_systemDmaData.m_s0z;
480 
481  vec[0] = m_systemDmaData.m_c1x + 1.5f * m_systemDmaData.m_s1x;
482  vec[1] = m_systemDmaData.m_c1y + 1.5f * m_systemDmaData.m_s1y;
483  vec[2] = m_systemDmaData.m_c1z + 1.5f * m_systemDmaData.m_s1z;
484  vec[3] = 0;
485  vec *= mat;
486  m_systemDmaData.m_c1x = vec[0] - 1.5f * m_systemDmaData.m_s1x;
487  m_systemDmaData.m_c1y = vec[1] - 1.5f * m_systemDmaData.m_s1y;
488  m_systemDmaData.m_c1z = vec[2] - 1.5f * m_systemDmaData.m_s1z;
489 
490  vec[0] = m_systemDmaData.m_c2x + 1.5f * m_systemDmaData.m_s2x;
491  vec[1] = m_systemDmaData.m_c2y + 1.5f * m_systemDmaData.m_s2y;
492  vec[2] = m_systemDmaData.m_c2z + 1.5f * m_systemDmaData.m_s2z;
493  vec[3] = 0;
494  vec *= mat;
495  m_systemDmaData.m_c2x = vec[0] - 1.5f * m_systemDmaData.m_s2x;
496  m_systemDmaData.m_c2y = vec[1] - 1.5f * m_systemDmaData.m_s2y;
497  m_systemDmaData.m_c2z = vec[2] - 1.5f * m_systemDmaData.m_s2z;
498
