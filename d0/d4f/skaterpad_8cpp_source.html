256 float CSkaterPad::GetScaledAnalogStickMagnitude ( float analog_x, float analog_y, float analog_angle )
257 {
258  float x = analog_x / (analog_x > 0.0f ? 127.0f : 128.0f);
259  float y = analog_y / (analog_y > 0.0f ? 127.0f : 128.0f);
260 
261  #ifndef __PLAT_NGC__
262 
263  float angle;
264  if (analog_angle > (-Mth::PI / 4.0f) && analog_angle < (Mth::PI / 4.0f))
265  {
266  // top quadrant
267  angle = analog_angle;
268  }
269  else if (analog_angle > (Mth::PI / 4.0f) && analog_angle < (3.0f * Mth::PI / 4.0f))
270  {
271  // right quadrant
272  angle = analog_angle - (Mth::PI / 2.0f);
273  }
274  else if (Mth::Abs(analog_angle) > (3.0f * Mth::PI / 4.0f))
275  {
276  // bottom quadrant
277  angle = Mth::Abs(analog_angle) - Mth::PI;
278  }
279  else
280  {
281  // left quadrant
282  angle = analog_angle + (Mth::PI / 2.0f);
283  }
284 
285  return sqrtf(x * x + y * y) * Mth::Abs(cosf(angle));
286 
287  #else
288 
289  return sqrtf(x * x + y * y);
290 
291  #endif
292 }
