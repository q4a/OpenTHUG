101 {
102  switch (checksum)
103  {
104  case 0xef082878: // standard
105  return DROPOFF_FUNC_STANDARD;
106 
107  case 0xf221474c: // linear
108  return DROPOFF_FUNC_LINEAR;
109 
110  case 0x9706104b: // exponential

112 
113  case 0xca64c41: // inv_exponential
114  return DROPOFF_FUNC_INV_EXPONENTIAL;
115 
116  default:
117  Dbg_MsgAssert(0, ("Invalid dropoff function type %s (%x)", Script::FindChecksumName(checksum), checksum));

565 /******************************************************************/
566 /* */
567 /* Returns true if sound is still playing. */
568 /* Adjust the pitch for doppler, and the volume and pan for */
569 /* position. */
570 /* */
571 /******************************************************************/
572 bool CSfxManager::AdjustObjectSound( Obj::CSoundComponent *pObj, VoiceInfo *pVoiceInfo, Tmr::Time gameTime )
573 {
574  Dbg_Assert( pVoiceInfo );
575 
576  ObjectSoundInfo *pInfo = &pVoiceInfo->info;
577 
578  float dist;
579 
580  // Get dropoff distance if sent to Obj_PlaySound. If not, get default (set up in LoadSound).
581  float dropoffDist = ( pInfo->dropoffDist == 0.0f ) ? DefaultDropoffDist : pInfo->dropoffDist;
582 
583  Gfx::Camera *pCamera = Nx::CViewportManager::sGetClosestCamera( pObj->GetPosition(), &dist );
584  if( !pCamera )
585  {
586  Dbg_Message( "Warning: Positional sound, but no camera... Sound won't be updated." );
587 
588  // If there is no camera, just don't update the sound but keep it playing the same as before.
589  // (If we return false, the object will delete its sound).
590  // We might not be able to find the camera at the start of the level or perhaps when changing levels.
591  return true;
592  }
593 
594  Mth::Vector dropoff_pos;
595  Mth::Vector *p_dropoff_pos = NULL;
596  if (pObj->GetClosestDropoffPos(pCamera, dropoff_pos))
597  {
598  p_dropoff_pos = &dropoff_pos;
599  }
600  // See if the object sound could be turned off (far enough away from the camera).
601  else if( pInfo->timeForNextDistCheck <= gameTime )
602  {
603 // Dbg_Message( "checking if we need to turn %s off", Script::FindChecksumName( pInfo->checksum ) );
604  if( dist >= ( dropoffDist + DIST_FROM_DROPOFF_AT_WHICH_TO_STOP_SOUND ))
605  {
606 // Dbg_Message( "sound %s auto off", Script::FindChecksumName( pInfo->checksum ) );
607 
608  // If the sound is looping, let the scripts continue to think the
609  // sound is still around and keep pitch/volume adjusted for the action.
610  if( pVoiceInfo->waveIndex->platformWaveInfo.looping )
611  {
612  // Turn the sound off, monitor the sound from the object update.
613  pObj->SoundOutOfRange( pInfo, gameTime );
614  }
615 
616  // Should cause this sound to be stopped and voice cleared!
617  return false;
618  }
619  // Still okay, sound still in range. Continue playing for x more milliseconds.
620  pInfo->timeForNextDistCheck = gameTime + 2000;
621  }
622 
623  float pitch = 100.0f;
624  sVolume vol;
625  vol.SetSilent();
626 
627  // If dist is beyond dropoff dist, leave volume at zero.
628  if( ( p_dropoff_pos || (dist < dropoffDist) ) )
629  {
630  Mth::Vector obj_pos(pObj->GetClosestEmitterPos(pCamera));
631 
632  if( pVoiceInfo->waveIndex->flags & SFX_FLAG_POSITIONAL_UPDATE_WITH_DOPPLER )
633  {
634  // Even if the object didn't move, the camera could have!
635  AdjustPitchForDoppler( &pitch, obj_pos, pObj->GetClosestOldEmitterPos(pCamera), Tmr::FrameLength(), pCamera );
636  }
637  SetVolumeFromPos( &vol, obj_pos, dropoffDist, pInfo->dropoffFunction, pCamera, p_dropoff_pos );
638  }
639 
640  if( !Config::CD())
641  {
642  if( Script::GetInteger( 0x6d2e270e /* DebugSoundFxUpdate */, Script::NO_ASSERT ) && !vol.IsSilent())
643  {
644  Dbg_Message( "Updating sound %s with volL = %f volR = %f", Script::FindChecksumName( pInfo->checksum ), vol.GetChannelVolume(0), vol.GetChannelVolume(1));
645  }
646  }
647 
648  return UpdateLoopingSound( pVoiceInfo->uniqueID, &vol, pitch );
649 }

1068  pInfo->origVolume = pInfo->targetVolume = pInfo->currentVolume;
1069  pInfo->deltaPitch = 0.0f;
1070  pInfo->deltaVolume = 0.0f;
1071  pInfo->dropoffFunction = pUpdateInfo->dropoffFunction;

1073  else

1076  pInfo->currentVolume = 100.0f;
1077  pInfo->dropoffDist = 0.0f;
1078  pInfo->dropoffFunction = DROPOFF_FUNC_STANDARD;

1396 /* */
1397 /* Calculates multipliers for front left, front right, rear left, */
1398 /* rear right and center channel speakers. */
1399 /* */
1400 /******************************************************************/
1401 void CSfxManager::Get5ChannelMultipliers( float angle, float *p_multipliers )
1402 {
1403  static float speakers[5][3] = { { 330.00f, 240.01f, 359.99f }, // Front left max angle, min angle0, min angle1
1404  { 30.00f, 0.01f, 119.99f }, // Front right
1405  { 240.00f, 120.01f, 329.99f }, // Rear left
1406  { 120.0f, 30.01f, 239.99f }, // Rear right
1407  { 0.0f, 330.01f, 29.99f }}; // Center
1408 
1409  // Ensure angle is in a suitable range.
1410  angle = ( angle < 0.0f ) ? ( 360.0f + angle ) : angle;
1411 
1412  // Go through and calculate the relative volumes for each speaker.
1413  for( int spkr = 0; spkr < 5; ++spkr )
1414  {
1415  float amin0 = speakers[spkr][1];
1416  float amin1 = speakers[spkr][2];
1417  float amax = speakers[spkr][0];
1418  float mul = 0.0f;
1419 
1420  if( amin0 < amax )
1421  {
1422  // Regular test.
1423  if(( angle > amin0 ) && ( angle <= amax ))
1424  {
1425  // Angle lies between amin0 and amax.
1426  mul = ( angle - amin0 ) / ( amax - amin0 );
1427  }
1428  else if(( angle > amax ) && ( angle < amin1 ))
1429  {
1430  // Angle lies between amax and amin1.
1431  mul = 1.0f - (( angle - amax ) / ( amin1 - amax ));
1432  }
1433  }
1434  else
1435  {
1436  // Non regular test (center channel). Assumes center at 0.0.
1437  if( angle > amin0 )
1438  {
1439  mul = ( angle - amin0 ) / ( 360.0f - amin0 );
1440  }
1441  else if( angle < amin1 )
1442  {
1443  mul = ( amin1 - angle ) / ( amin1 );
1444  }
1445  }
1446 
1447  // Angle is within scope of this speaker. Figure multiplier.
1448  Dbg_Assert( mul <= 1.0f );
1449  p_multipliers[spkr] = sinf( mul * Mth::PI * 0.5f );
1450  }
1451 }
