91 Script::CStruct* GetOptionStructure( uint32 partChecksum, uint32 lookFor, bool assertOnFail )
92 {
93  Script::CArray* pArray;
94  pArray = Script::GetArray( partChecksum, Script::ASSERT );
95 
96  Script::CStruct* pReturnStructure = NULL;
97 
98  if ( lookFor == 0 )
99  {
100  // special checksum, meaning to return NULL
101  // (handy when preprocessing random descs)
102  return NULL;
103  }
104 
105  for ( uint32 i = 0; i < pArray->GetSize(); i++ )
106  {
107  Script::CStruct* pCurrDesc = pArray->GetStructure( i );
108  uint32 desc_id = get_desc_id_from_structure( pCurrDesc );
109  if ( desc_id == lookFor )
110  {
111  pReturnStructure = pCurrDesc;
112  break;
113  }
114  }
115 
116  if ( assertOnFail )
117  {
118  Dbg_MsgAssert( pReturnStructure, ( "Couldn't find %s %s", Script::FindChecksumName(partChecksum), Script::FindChecksumName(lookFor) ) );
119  }
120 
121  return pReturnStructure;
122 }

142 void BuildRandomSetList( uint32 partChecksum, uint32 random_set, Script::CStruct** ppReturnStructs, int* pReturnNumStructs )
143 {
144  Script::CArray* pArray;
145  pArray = Script::GetArray( partChecksum, Script::ASSERT );
146 
147  // reset the return value
148  *pReturnNumStructs = 0;
149 
150  int part_array_size = pArray->GetSize();
151  for ( int i = 0; i < part_array_size; i++ )
152  {
153  Script::CStruct* pStruct = pArray->GetStructure( i );
154  int allowed_to_pick = 0;
155  pStruct->GetInteger( CRCD(0x355f9467,"allowed_to_pick"), &allowed_to_pick, Script::ASSERT );
156 
157  if ( allowed_to_pick )
158  {
159  // if a random_set was specified, then
160  // make sure that this item works with it
161  // (used for matching skin tones)
162  uint32 this_random_set = 0;
163  pStruct->GetChecksum( CRCD(0x0d7260fd,"random_set"), &this_random_set, Script::NO_ASSERT );
164 
165  if ( this_random_set == random_set )
166  {
167  *ppReturnStructs = pStruct;
168  ppReturnStructs++;
169 
170  (*pReturnNumStructs)++;
171  }
172  }
173  }
174 }

181 Script::CStruct* GetRandomOptionStructure( uint32 partChecksum, uint32 random_set )
182 {
183  Script::CArray* pArray;
184  pArray = Script::GetArray( partChecksum, Script::ASSERT );
185 
186  const int vMAX_CAS_ITEMS = 32;
187  Script::CStruct* pAllowedStructs[vMAX_CAS_ITEMS];
188  bool alreadySelected[vMAX_CAS_ITEMS];
189  int allowRandomCount = 0;
190 
191  int part_array_size = pArray->GetSize();
192  for ( int i = 0; i < part_array_size; i++ )
193  {
194  Script::CStruct* pStruct = pArray->GetStructure( i );
195  int allow_random = !pStruct->ContainsFlag( CRCD(0xf6f8e158,"no_random") );
196  pStruct->GetInteger( CRCD(0xf1e3cd22,"allow_random"), &allow_random, Script::ASSERT );
197 
198  if ( random_set )
199  {
200  // if a random_set was specified, then
201  // make sure that this item works with it
202  // (used for matching skin tones)
203  uint32 this_random_set = 0;
204  if ( pStruct->GetChecksum( CRCD(0x0d7260fd,"random_set"), &this_random_set, Script::NO_ASSERT ) )
205  {
206  if ( this_random_set != random_set )
207  {
208  allow_random = false;
209  }
210  }
211  }
212 
213  if ( allow_random )
214  {
215  Dbg_Assert( allowRandomCount < vMAX_CAS_ITEMS );
216  pAllowedStructs[allowRandomCount] = pStruct;
217 
218  int already_selected;
219  pStruct->GetInteger( CRCD(0x92bddfd9,"already_selected"), &already_selected, true );
220  alreadySelected[allowRandomCount] = already_selected;
221 
222  allowRandomCount++;
223  }
224  }
225 
226  // make sure there's at least one allow_random,
227  // or else we'll get into an infinite loop
228  Dbg_MsgAssert( allowRandomCount > 0, ( "No allow_random was found in this array %s (Tell Gary!)", Script::FindChecksumName(partChecksum) ) );
229 
230  // count how many of them are already unselected
231  int unselected_count = 0;
232  for ( int i = 0; i < allowRandomCount; i++ )
233  {
234  if ( !alreadySelected[i] )
235  {
236  unselected_count++;
237  }
238  }
239 
240  // the following makes sure that all the "allow_random" parts
241  // are used at least once, before duplicates begin.
242  if ( unselected_count == 0 )
243  {
244  for ( int i = 0; i < allowRandomCount; i++ )
245  {
246  alreadySelected[i] = false;
247  }
248  }
249 
250  // now get rid of all the already selected items
251  int currentCount = 0;
252  for ( int i = 0; i < allowRandomCount; i++ )
253  {
254  if ( !alreadySelected[i] )
255  {
256  pAllowedStructs[currentCount] = pAllowedStructs[i];
257  alreadySelected[currentCount] = alreadySelected[i];
258  currentCount++;
259  }
260  }
261  allowRandomCount = currentCount;
262 
263  Dbg_MsgAssert( allowRandomCount >= 1, ( "No part to pick" ) );
264  int index = Mth::Rnd( allowRandomCount );
265  Dbg_Assert( index >= 0 && index < allowRandomCount );
266  Script::CStruct* pReturnStruct = pAllowedStructs[index];
267 
268  Dbg_Assert( pReturnStruct );
269  pReturnStruct->AddInteger( CRCD(0x92bddfd9,"already_selected"), 1 );
270 
271 // uint32 checksum;
272 // pStruct->GetChecksum( "desc_id", &checksum, true );
273 // Dbg_Message( "Choosing random part %s", Script::FindChecksumName( checksum ) );
274 
275  return pReturnStruct;
276 }

283 Script::CStruct* GetRandomOptionStructureByIndex( uint32 partChecksum, int desired_index, uint32 random_set )
284 {
285  Script::CArray* pArray;
286  pArray = Script::GetArray( partChecksum, Script::ASSERT );
287 
288  int part_array_size = pArray->GetSize();
289 
290  if ( !random_set )
291  {
292  for ( int i = 0; i < part_array_size; i++ )
293  {
294  Script::CStruct* pStruct = pArray->GetStructure( i );
295 
296  int random_index = -1;
297  if ( pStruct->GetInteger( CRCD(0x4b833e64,"random_index"), &random_index, Script::ASSERT ) )
298  {
299  if ( random_index == desired_index )
300  {
301  return pStruct;
302  }
303  }
304  }
305  }
306  else
307  {
308  // GJ FIX FOR SK5:TT12579: "NJ - Goal peds being created with mismached skin colors"
309  // if a random_set (skintone) was specified, then find the n-th item that actually works with this
310  // random set
311  while ( 1 )
312  {
313  #ifdef __NOPT_ASSERT__
314  int current_random_index = desired_index;
315  #endif
316 
317  for ( int i = 0; i < part_array_size; i++ )
318  {
319  Script::CStruct* pStruct = pArray->GetStructure( i );
320 
321  int allow_random = !pStruct->ContainsFlag( CRCD(0xf6f8e158,"no_random") );
322  pStruct->GetInteger( CRCD(0xf1e3cd22,"allow_random"), &allow_random, Script::ASSERT );
323 
324  uint32 this_random_set = 0;
325  pStruct->GetChecksum( CRCD(0x0d7260fd,"random_set"), &this_random_set, Script::NO_ASSERT );
326  bool is_universal_item = ( this_random_set == 0 );
327 
328  if ( allow_random && ( ( this_random_set == random_set ) || is_universal_item ) )
329  {
330  if ( desired_index <= 0 )
331  {
332  return pStruct;
333  }
334 
335  desired_index--;
336  }
337  }
338 
339  // once we get to the end of the array, we need to start over
340  // (basically, we're doing a modulo of the desired_index)...
341  // the following checks for an infinite loop in which there
342  // were no items found that works with this random set
343  Dbg_MsgAssert( current_random_index != desired_index, ( "Couldn't find 'random_index' %d for ped part %s (are there any that fit this skintone %s?)", desired_index, Script::FindChecksumName(partChecksum), Script::FindChecksumName(random_set) ) );
344  }
345  }
346 
347  Dbg_MsgAssert( 0, ( "Couldn't find 'random_index' %d for ped part %s", desired_index, Script::FindChecksumName(partChecksum) ) );
348  return NULL;
349 }
