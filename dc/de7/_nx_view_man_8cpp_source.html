127 CViewport* CViewportManager::sGetActiveViewport(int view)
128 {
129  Dbg_MsgAssert(view < s_num_active_viewports[s_screen_mode], ("Active viewport number is invalid %d", view));
130 
131  int found_views = 0;
132 
133  // Go through array, and don't count invalid viewports
134  for ( uint32 i = 0; i < vNUM_VIEWPORTS; i++ )
135  {
136  if (sp_active_viewport_indexes[i] >= 0)
137  {
138  if (found_views == view)
139  {
140  return sp_viewports[sp_active_viewport_indexes[i]];
141  }
142  found_views++;
143  }
144  }
145 
146  Dbg_MsgAssert(0, ("Cant find active viewport # %d", view));
147 
148  return NULL;
149 }

259 Gfx::Camera* CViewportManager::sGetClosestCamera(const Mth::Vector &pos, float *distParam)
260 {
261  int i;
262  int numCams = sGetNumActiveCameras( );
263  Gfx::Camera *p_camera;
264  Gfx::Camera *p_closestCamera = NULL;
265  float closestDist = -1.0f;
266  float tempDist;
267 
268  for ( i = 0; i < numCams; i++ )
269  {
270  p_camera = sGetActiveCamera( i );
271  if ( p_camera )
272  {
273  tempDist = Mth::DistanceSqr( pos, p_camera->GetPos() );
274  if ( ( closestDist < 0 ) || ( tempDist < closestDist ) )
275  {
276  closestDist = tempDist;
277  p_closestCamera = p_camera;
278  }
279  }
280  }
281 
282  //Dbg_MsgAssert( p_closestCamera,( "No cameras or what?" ));
283  if ( p_closestCamera == NULL )
284  {
285  return NULL;
286  }
287 
288  if ( distParam )
289  {
290  *distParam = sqrtf(closestDist);
291  }
292  return p_closestCamera;
293 }

350 void CViewportManager::sSetScreenMode( ScreenMode mode )
351 {
352  // Update active viewport indexes
353  sp_active_viewport_indexes = s_screen_mode_viewport_indexes[mode];
354 
355  s_screen_mode = mode;
356 
357  s_num_active_viewports_cached = s_num_active_viewports[s_screen_mode];
358 
359 
360  // Now update cameras
361  for ( int i = 0; i < s_num_active_viewports[mode]; i++ )
362  {
363  CViewport *p_viewport = sGetActiveViewport(i);
364  Dbg_MsgAssert(p_viewport, ("SetScreenMode: Couldn't find active viewport # %d for mode %d", i, mode));
365 
366  // no camera has been bound
367  Gfx::Camera *p_camera = p_viewport->GetCamera();
368  if ( !p_camera )
369  {
370  //return;
371  continue; // just continue, even if there is no camera set up yet.....
372  }
373 
374  //Dbg_Message("******************* SetScreenMode: Setting Camera %d (%x) of mode %d", i, p_camera, mode);
375 
376  // set up proper aspect ratio for camera
377  switch (mode)
378  {
379  case vSPLIT_V:
380  // camera "window" at same height, half width as full screen camera
381  //p_camera->SetFocalLength ( Script::GetFloat("Skater_Cam_Focal_Length"), vBASE_ASPECT_RATIO / 2.0f);
382  p_camera->SetHFOV(Script::GetFloat("Skater_Cam_Horiz_FOV") / 2.0f);
383  break;
384  case vSPLIT_H:
385  // camera "window" at same width, half height as full screen camera
386  //p_camera->SetFocalLength ( Script::GetFloat("Skater_Cam_Focal_Length") * 2.0f, vBASE_ASPECT_RATIO * 2.0f);
387  p_camera->SetHFOV(Script::GetFloat("Skater_Cam_Horiz_FOV"));
388  break;
389  default:
390  // camera "window" at same width, height as full screen camera
391  //p_camera->SetFocalLength ( Script::GetFloat("Skater_Cam_Focal_Length"), vBASE_ASPECT_RATIO);
392  p_camera->SetHFOV(Script::GetFloat("Skater_Cam_Horiz_FOV"));
393  break;
394  }
395  }

409 // some fundamental things about the TV
410 //
411 // s_screen_aspect is the aspect ratio of the TV
412 // this is the physical aspect ratio, and not the pixel ratio
413 // thus it will be either 4:3 (4.0f/3.0f) or 16:9 (16.0f/9.0f)
414 //
415 // s_screen_angle is the default view angle for a full screen viewport
416 // currently (2/8/02) it is set at 72 degrees when screen_aspect is 4:3
417 // when we have a wider aspect ratio, we want a wider screen angle
418 // however, we don't want to go too wide, or we will render too much
419 // 80 degrees is okay, providing us with reduced vertical coverage, but increased
420 // horizontal coverage (so skater seems bigger on screen, and you can see more of the level)
421

471 void CViewportManager::sSetScreenAngle(float angle)
472 {
473  if (angle == 0.0f)
474  {
475  angle = s_default_angle;
476  }
477 
478  Dbg_MsgAssert(angle > 0.01f && angle < 180.0f,("Unlikely value (%f) for screen angle",angle));
479  s_screen_angle = angle;
480 
481  s_screen_angle_factor = tanf(Mth::DegToRad(angle / 2.0f)) / tanf(Mth::DegToRad(s_default_angle / 2.0f));
482 
483  //Dbg_Message("Screen angle %f; Screen angle factor %f", s_screen_angle, s_screen_angle_factor);
484 
485  // Update cameras
486  // Garrett: This isn't the right way to do it since it won't touch all the possible cameras.
487  // It would be better to go through a list of all the cameras.
488  for (int view_idx = 0; view_idx < sGetNumActiveViewports(); view_idx++)
489  {
490  CViewport *p_cur_viewport = CViewportManager::sGetActiveViewport(view_idx);
491  if (p_cur_viewport)
492  {
493  Gfx::Camera *p_cur_camera = p_cur_viewport->GetCamera();
494 
495  if (p_cur_camera)
496  {
497  p_cur_camera->UpdateAdjustedHFOV();
498  }
499  }
500  }

554 // @script | SetScreenMode | Sets the way the screen will be split up into separate
555 // viewports in the game. If there is more than one viewport, and multiple skaters
556 // are active, then each skater will get his own viewport
557 // @uparm One_Camera | the mode to use - one of the following: <nl>
558 // One_Camera <nl>
559 // Split_Vertical <nl>
560 // Split_Horizontal <nl>

562 bool ScriptSetScreenMode(Script::CScriptStructure *pParams, Script::CScript *pScript)
563 {
564 
565 
566  uint32 mode;
567  pParams->GetChecksum(NONAME, &mode, true);
568 
569  switch (mode)
570  {
571  case 0x6f70ea65: // "one_camera"
572  CViewportManager::sSetScreenMode(vONE_CAM);
573  break;
574  case 0x9b62e5a6: // "split_vertical"
575  CViewportManager::sSetScreenMode(vSPLIT_V);
576  break;
577  case 0xb07b580e: // "split_horizontal"
578  CViewportManager::sSetScreenMode(vSPLIT_H);
579  break;
580  case 0xb6624a73: // "split_quarters"
581  CViewportManager::sSetScreenMode(vSPLIT_QUARTERS);
582  break;
583  case 0x6f2d1231: // "horse1"
584  CViewportManager::sSetScreenMode(vHORSE1);
585  break;
586  case 0xf624438b: // "horse2"
587  CViewportManager::sSetScreenMode(vHORSE2);
