336 bool CTerrainManager::s_get_terrain_actions_struct(ETerrainType terrain, Script::CStruct *p_actions)
337 {
338  Dbg_MsgAssert(p_actions, ("s_get_terrain_actions_struct given NULL CStruct"));
339 
340  Script::CStruct *p_default_actions = NULL;
341  Script::CStruct *p_terrain_actions = NULL;
342  Script::CStruct *p_level_default_actions = NULL;
343  Script::CStruct *p_level_terrain_actions = NULL;
344 
345  uint32 terrain_checksum;
346  ETerrainType terrain_type;
347 
348  // Search Default Terrain array first
349  Script::CArray *pDefaultTerrainArray = Script::GetArray(CRCD(0x1c2c4218,"standard_terrain"));
350  Dbg_MsgAssert(pDefaultTerrainArray, ("standard_terrain not found"));
351 
352  for (uint i = 0; (!p_default_actions || !p_terrain_actions) && i < pDefaultTerrainArray->GetSize(); i++)
353  {
354  // Get next terrain struct
355  Script::CStruct *pTerrainStruct = pDefaultTerrainArray->GetStructure(i);
356  Dbg_MsgAssert(pTerrainStruct, ("terrain array not made of structures"));
357 
358  pTerrainStruct->GetChecksum(CRCD(0x3789ac4e,"Terrain"), &terrain_checksum, Script::ASSERT);
359  terrain_type = sGetTerrainFromChecksum(terrain_checksum);
360 
361  if (terrain_type == vTERRAIN_DEFAULT)
362  {
363  pTerrainStruct->GetStructure(CRCD(0x5c969c50,"SoundActions"), &p_default_actions);
364  //Dbg_Message("Found default terrain:");
365  //Script::PrintContents(p_default_actions);
366  }
367  else if (terrain_type == terrain)
368  {
369  pTerrainStruct->GetStructure(CRCD(0x5c969c50,"SoundActions"), &p_terrain_actions);
370  //Dbg_Message("Found terrain:");
371  //Script::PrintContents(p_terrain_actions);
372  }
373 
374  if (p_default_actions && p_terrain_actions)
375  {
376  break;
377  }
378  }
379 
380  // Search Level Terrain array
381  Script::CArray *pLevelTerrainArray = Script::GetArray(CRCD(0x96218951,"level_terrain"));
382  if (pLevelTerrainArray)
383  {
384  for (uint i = 0; i < pLevelTerrainArray->GetSize(); i++)
385  {
386  // Get next terrain struct
387  Script::CStruct *pTerrainStruct = pLevelTerrainArray->GetStructure(i);
388  Dbg_MsgAssert(pTerrainStruct, ("terrain array not made of structures"));
389 
390  pTerrainStruct->GetChecksum(CRCD(0x3789ac4e,"Terrain"), &terrain_checksum, Script::ASSERT);
391  terrain_type = sGetTerrainFromChecksum(terrain_checksum);
392 
393  if (terrain_type == vTERRAIN_DEFAULT)
394  {
395  pTerrainStruct->GetStructure(CRCD(0x5c969c50,"SoundActions"), &p_level_default_actions);
396  } else if (terrain_type == terrain)
397  {
398  pTerrainStruct->GetStructure(CRCD(0x5c969c50,"SoundActions"), &p_level_terrain_actions);
399  }
400 
401  if (p_level_default_actions && p_level_terrain_actions)
402  {
403  break;
404  }
405  }
406  }
407 
408  // Combine them together
409  *p_actions = *p_default_actions;
410 
411  if (p_level_default_actions)
412  {
413  *p_actions += *p_level_default_actions;
414  }
415 
416  if (p_terrain_actions)
417  {
418  *p_actions += *p_terrain_actions;
419  }
420 
421  if (p_level_terrain_actions)
422  {
423  *p_actions += *p_level_terrain_actions;
424  }
425 
426  return true;
427 }

434 Script::CStruct * CTerrainManager::s_choose_sound_action(Script::CStruct *p_sound_action_struct)
435 {
436  Script::CStruct* p_sound;
437 
438  // soundAction is a single stucture, so just use it
439  if (p_sound_action_struct->GetStructure(CRCD(0x25f654c8, "soundAction"), &p_sound))
440  {
441  Dbg_MsgAssert(!p_sound->ContainsComponentNamed(CRCD(0xdbc4c4db, "chance")),
442  ("A sound may only include a chance parameter when it is an element of a random-choice list of sounds"));
443  return p_sound;
444  }
445 
446  // soundAction is an array of structures, so choose one randomly based on their given chances
447  Script::CArray* p_random_indexed_sound_array;
448  p_sound_action_struct->GetArray(CRCD(0x25f654c8, "soundAction"), &p_random_indexed_sound_array, Script::ASSERT);
449 
450  int total_chance = 0;
451  for (int sound_idx = p_random_indexed_sound_array->GetSize(); sound_idx--; )
452  {
453  p_sound = p_random_indexed_sound_array->GetStructure(sound_idx);
454 
455  int chance;
456  p_sound->GetInteger(CRCD(0xdbc4c4db, "chance"), &chance, Script::ASSERT);
457 
458  total_chance += chance;
459  }
460 
461  int choice = Mth::Rnd(total_chance);
462 
463  for (int sound_idx = p_random_indexed_sound_array->GetSize(); sound_idx--; )
464  {
465  p_sound = p_random_indexed_sound_array->GetStructure(sound_idx);
466 
467  int chance;
468  p_sound->GetInteger(CRCD(0xdbc4c4db, "chance"), &chance, Script::ASSERT);
469 
470  choice -= chance;
471  if (choice < 0)
472  {
473  return p_sound;
474  }
475  }
476 
477  Dbg_Assert(false);
478  return NULL;
479 }

486 Script::CStruct * CTerrainManager::s_get_action_sound_struct(Script::CStruct *p_actions, ETerrainActionType action, float soundChoice)
487 {
488  Dbg_MsgAssert(p_actions, ("s_get_action_sound_struct given NULL action CStruct"));
489 
490  Script::CStruct *p_sound;
491 
492  uint32 checksum = s_get_action_checksum(action);
493 
494  // sound is a single sound structure, so just use it
495  if (p_actions->GetStructure(checksum, &p_sound))
496  {
497  return p_sound;
498  }
499 
500  // sound is an array of structures, one of which must be choosen depending upon speed
501  Script::CArray* p_float_indexed_sound_array;
502  p_actions->GetArray(checksum, &p_float_indexed_sound_array, Script::ASSERT);
503 
504  // check to insure that the final structure in the array has no useUpTo parameter and thus operates as a default choice
505  Dbg_MsgAssert(!p_float_indexed_sound_array->GetStructure(p_float_indexed_sound_array->GetSize() - 1)->ContainsComponentNamed(CRCD(0x9ee9a09a, "useUpTo")),
506  ("Final sound in a speed-indexed sound list must not include useUpTo"));
507 
508  for (uint32 sound_action_idx = 0; sound_action_idx < p_float_indexed_sound_array->GetSize(); sound_action_idx++)
509  {
510  // each structure contains a useUpTo parameter which is checked against speed in turn; if found to be larger
511  // than speed, the sound is used; a structure lacking useUpTo is always used
512  Script::CStruct* p_sound_action_struct = p_float_indexed_sound_array->GetStructure(sound_action_idx);
513 
514  float up_to_speed;
515  if (!p_sound_action_struct->GetFloat(CRCD(0x9ee9a09a, "useUpTo"), &up_to_speed))
516  {
517  Dbg_MsgAssert(sound_action_idx == p_float_indexed_sound_array->GetSize() - 1,
518  ("Only the final sound in a float-indexed sound list may fail to include useUpTo"));
519  return s_choose_sound_action(p_sound_action_struct);
520  }
521 
522  if (soundChoice <= up_to_speed)
523  {
524  return s_choose_sound_action(p_sound_action_struct);
525  }
526  }
527 
528  Dbg_Assert(false);
529  return NULL;
530 }

537 bool CTerrainManager::s_get_sound_info(Script::CStruct *p_sound, STerrainSoundInfo *p_info)
538 {
539  Dbg_MsgAssert(p_info, ("s_get_sound_info given NULL STerrainSoundInfo"));
540 
541  // Set defaults
542  p_info->mp_soundName = NULL;
543  p_info->m_soundChecksum = 0;
544  p_info->m_maxPitch = 100.0f;
545  p_info->m_minPitch = 100.0f;
546  p_info->m_maxVol = 100.0f;
547  p_info->m_minVol = 0.0f;
548  p_info->m_loadPitch = 100.0f;
549  p_info->m_loadVol = 100.0f;
550 
551  p_sound->GetString( CRCD(0x7713c7b,"sound"), &p_info->mp_soundName );
552  if ( !p_info->mp_soundName )
553  {
554  Dbg_MsgAssert(0, ( "Need the name of a sound in s_get_sound_info" ));
555  return false;
556  }
557 
558  // Calculate checksum
559  const char *pNameMinusPath = p_info->mp_soundName;
560  int stringLength = strlen( p_info->mp_soundName );
561  for( int i = 0; i < stringLength; i++ )
562  {
563  if(( p_info->mp_soundName[i] == '\\' ) || ( p_info->mp_soundName[i] == '/' ))
564  pNameMinusPath = &p_info->mp_soundName[i + 1];
565  }
566 
567  p_info->m_soundChecksum = Script::GenerateCRC( pNameMinusPath );
568 
569  p_sound->GetFloat( CRCD(0xfa3e14c5,"maxPitch"), &p_info->m_maxPitch );
570  p_sound->GetFloat( CRCD(0x1c5ebb24,"minPitch"), &p_info->m_minPitch );
571 
572  p_sound->GetFloat( CRCD(0x693daaf,"maxVol"), &p_info->m_maxVol );
573  p_sound->GetFloat( CRCD(0x4391992d,"minVol"), &p_info->m_minVol );
574 
575  p_sound->GetFloat( CRCD(0xf573682f,"loadPitch"), &p_info->m_loadPitch );
576  p_sound->GetFloat( CRCD(0x312ccbcf,"loadVol"), &p_info->m_loadVol );
577 
578  return true;
579 }
