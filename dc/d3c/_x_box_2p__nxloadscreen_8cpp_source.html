45 void CALLBACK loadingBarTimerCallback( UINT uTimerID, UINT uMsg, DWORD_PTR dwUser, DWORD_PTR dw1, DWORD_PTR dw2 )
46 {
47  if( NxXbox::EngineGlobals.loadingbar_timer_event != 0 )
48  {
49  loadingBarCurrentSeconds += loadingBarDeltaSeconds;
50  float mult = loadingBarCurrentSeconds / loadingBarTotalSeconds;
51  mult = ( mult > 1.0f ) ? 1.0f : mult;
52  int bar_width = (int)( loadingBarWidth * mult );
53 
54  // Get pointer to front buffer memory.
55  IDirect3DSurface8 *p_buffer;
56  D3DLOCKED_RECT locked_rect;
57  NxXbox::EngineGlobals.p_Device->GetBackBuffer( -1, D3DBACKBUFFER_TYPE_MONO, &p_buffer );
58  p_buffer->LockRect( &locked_rect, NULL, D3DLOCK_TILED );
59  uint32 *p_screen = (uint32*)locked_rect.pBits;
60 
61  // ----------------------------------
62  // | |||||||||||||||||||||| |
63  // ----------------------------------
64  // ^ ^ ^ ^
65  // a b c d
66  //
67  // a = surround start
68  // b = bar start
69  // c = bar end
70  // d = surround end
71  int bar_start = ( 640 - loadingBarWidth ) / 2;
72  int surround_start = bar_start - 5;
73  int bar_end = bar_start + bar_width;
74  int surround_end = surround_start + loadingBarWidth + 10;
75 
76  const int HDTV_OFFSET = 48;
77 
78  bar_start = (int)SCREEN_CONV_X( bar_start );
79  surround_start = (int)SCREEN_CONV_X( surround_start );
80  bar_end = (int)SCREEN_CONV_X( bar_end );
81  surround_end = (int)SCREEN_CONV_X( surround_end );
82 
83  if( NxXbox::EngineGlobals.backbuffer_width > 640 )
84  {
85  bar_start += HDTV_OFFSET;
86  surround_start += HDTV_OFFSET;
87  bar_end += HDTV_OFFSET;
88  surround_end += HDTV_OFFSET;
89  }
90 
91  int base_y = (int)SCREEN_CONV_Y( 410 );
92 
93  for( int i = 0; i < 20; ++i )
94  {
95  uint32 *p_loop = p_screen + (( base_y + i ) * NxXbox::EngineGlobals.backbuffer_width );
96 
97  if(( i < 5 ) || ( i >= 15 ))
98  {
99  for( int j = surround_start; j < surround_end; ++j )
100  p_loop[j] = loadingBarBorderColor;
101  }
102  else
103  {
104  for( int j = surround_start; j < ( surround_start + 5 ); ++j )
105  p_loop[j] = loadingBarBorderColor;
106 
107  for( int j = bar_end; j < surround_end; ++j )
108  p_loop[j] = loadingBarBorderColor;
109 
110  for( int j = bar_start; j < bar_end; ++j )
111  {
112  uint32 idx = ( j - bar_start >= 1279 ) ? 1279 : ( j - bar_start );
113  uint32 write_value = 0x80000000UL | ( loadingBarColors[idx][0] << 16 ) | ( loadingBarColors[idx][1] << 8 ) | ( loadingBarColors[idx][2] << 0 );
114  p_loop[j] = write_value;
115  }
116  }
117  }
118  }
119 }

142 void CLoadScreen::s_plat_display(const char* filename, bool just_freeze, bool blank)
143 {
144  // Wait for asyncronous rendering to finish.
145  NxXbox::EngineGlobals.p_Device->BlockUntilIdle();
146 
147  if( !just_freeze )
148  {
149  if( blank )
150  {
151  D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET, 0x00000000UL, 1.0f, 0 );
152  D3DDevice_Swap( D3DSWAP_DEFAULT );
153  }
154  else
155  {
156  // Engine stuff
157  Dbg_Assert(!sp_load_screen_texture);
158  Dbg_Assert(!sp_load_screen_sprite);
159 
160  sp_load_screen_texture = new CXboxTexture();
161 
162 # ifdef __PAL_BUILD__
163  switch( Config::GetLanguage())
164  {
165  case Config::LANGUAGE_FRENCH:
166  {
167  char t_filename[200];
168  sprintf( t_filename, "PALImages/FRImages/%s", filename );
169  if( !sp_load_screen_texture->LoadTexture( t_filename, true ))
170  {
171  Dbg_Error( "Can't load texture %s", t_filename );
172  return;
173  }
174  break;
175  }
176  case Config::LANGUAGE_GERMAN:
177  {
178  char t_filename[200];
179  sprintf( t_filename, "PALImages/GRImages/%s", filename );
180  if( !sp_load_screen_texture->LoadTexture( t_filename, true ))
181  {
182  Dbg_Error( "Can't load texture %s", t_filename );
183  return;
184  }
185  break;
186  }
187  default:
188  {
189  if( !sp_load_screen_texture->LoadTexture( filename, true ))
190  {
191  Dbg_Error( "Can't load texture %s", filename );
192  return;
193  }
194  break;
195  }
196  }
197 # else
198  if( !sp_load_screen_texture->LoadTexture( filename, true ))
199  {
200  Dbg_Error( "Can't load texture %s", filename );
201  return;
202  }
203 # endif // __PAL_BUILD__
204 
205  // Copy into frame buffer.
206  float x_offset = 0.0f;
207  float y_offset = 0.0f;
208  float x_scale = 1.0f;
209  float y_scale = 1.0f;
210  float alpha_level = 1.0f;
211 
212  D3DDevice_SetTextureStageState( 0, D3DTSS_COLOROP, D3DTOP_SELECTARG1 );
213  D3DDevice_SetTextureStageState( 0, D3DTSS_ALPHAOP, D3DTOP_SELECTARG1 );
214  D3DDevice_SetTextureStageState( 0, D3DTSS_COLORARG1, D3DTA_TEXTURE );
215  D3DDevice_SetTextureStageState( 0, D3DTSS_ALPHAARG1, D3DTA_TEXTURE );
216 
217  // Store the stage zero minfilter, since it may be anisotropic.
218  DWORD stage_zero_minfilter;
219  D3DDevice_GetTextureStageState( 0, D3DTSS_MINFILTER, &stage_zero_minfilter );
220 
221  // Turn on texture filtering when scaling...
222  D3DDevice_SetTextureStageState( 0, D3DTSS_MAGFILTER, D3DTEXF_LINEAR );
223  D3DDevice_SetTextureStageState( 0, D3DTSS_MINFILTER, D3DTEXF_LINEAR );
224 
225  // Turn on clamping so that the linear textures work
226  NxXbox::set_render_state( RS_UVADDRESSMODE0, 0x00010001UL );
227 
228  D3DDevice_SetRenderState( D3DRS_LIGHTING, FALSE );
229 
230  // Use a default vertex shader
231  NxXbox::set_pixel_shader( 0 );
232  NxXbox::set_vertex_shader( D3DFVF_XYZRHW | D3DFVF_TEX1 );
233  NxXbox::set_blend_mode( NxXbox::vBLEND_MODE_DIFFUSE );
234 
235  // Select the texture (flush first, since the screen texture is linear).
236  NxXbox::set_texture( 0, NULL );
237  NxXbox::set_texture( 0, sp_load_screen_texture->GetEngineTexture()->pD3DTexture, sp_load_screen_texture->GetEngineTexture()->pD3DPalette );
238 
239  // Setup up the vertices.
240  typedef struct
241  {
242  float sx,sy,sz;
243  float rhw;
244  float tu,tv;
245  }
246  LOADSCREEN_VERT;
247 
248  // Get the width height from the texture itself, since we may be using a texture not designed especially
249  // for this screen dimension.
250  float tex_w = (float)sp_load_screen_texture->GetEngineTexture()->ActualWidth;
251  float tex_h = (float)sp_load_screen_texture->GetEngineTexture()->ActualHeight;
252  float scr_w = (float)NxXbox::EngineGlobals.backbuffer_width;
253  float scr_h = (float)NxXbox::EngineGlobals.backbuffer_height;
254 
255  LOADSCREEN_VERT vertices[4];
256 
257  // The texture coordinate addressing will differ depending on whether this is a linear texture or not.
258  if( is_power_of_two( sp_load_screen_texture->GetEngineTexture()->ActualWidth ) &&
259  is_power_of_two( sp_load_screen_texture->GetEngineTexture()->ActualHeight ))
260  {
261  // Not a linear texture, will be swizzled, max uv is (1.0, 1.0).
262  vertices[0].sx = x_offset;
263  vertices[0].sy = y_offset;
264  vertices[0].sz = 0.0f;
265  vertices[0].rhw = 0.0f;
266  vertices[0].tu = 0.0f;
267  vertices[0].tv = 1.0f;
268  vertices[1] = vertices[0];
269  vertices[1].sy = y_offset + ( scr_h * y_scale );
270  vertices[1].tv = 0.0f;
271  vertices[2] = vertices[0];
272  vertices[2].sx = x_offset + ( scr_w * x_scale );
273  vertices[2].tu = 1.0f;
274  vertices[3] = vertices[2];
275  vertices[3].sy = vertices[1].sy;
276  vertices[3].tv = 0.0f;
277  }
278  else
279  {
280  // Linear texture, won't be swizzled, max uv is (tex_w, tex_h).
281  vertices[0].sx = x_offset;
282  vertices[0].sy = y_offset;
283  vertices[0].sz = 0.0f;
284  vertices[0].rhw = 0.0f;
285  vertices[0].tu = -0.5f;
286  vertices[0].tv = tex_h - 0.5f;
287  vertices[1] = vertices[0];
288  vertices[1].sy = y_offset + ( scr_h * y_scale );
289  vertices[1].tv = -0.5f;
290  vertices[2] = vertices[0];
291  vertices[2].sx = x_offset + ( scr_w * x_scale );
292  vertices[2].tu = tex_w - 0.5f;
293  vertices[3] = vertices[2];
294  vertices[3].sy = vertices[1].sy;
295  vertices[3].tv = -0.5f;
296  }
297 
298  // Draw the vertices, and make sure they're displayed.
299  D3DDevice_DrawVerticesUP( D3DPT_TRIANGLESTRIP, 4, vertices, sizeof( LOADSCREEN_VERT ));
300  D3DDevice_Swap( D3DSWAP_DEFAULT );
301 
302  // Done with texture
303  delete sp_load_screen_texture;
304  sp_load_screen_texture = NULL;
305 
306  // Reflush linear texture out.
307  NxXbox::set_texture( 0, NULL );
308 
309  // Restore the stage zero minfilter.
310  D3DDevice_SetTextureStageState( 0, D3DTSS_MINFILTER, stage_zero_minfilter );
311  }
312  }
313 
314  // Indicate that the loading screen is visible, to stop any more rendering until it is hidden.
315  NxXbox::EngineGlobals.loadingscreen_visible = true;
316 
317 }

407  for( int c = 0; c < 3; ++c )
408  {
409  loadingBarColors[i][c] = loadingBarStartColor[c] + ((( loadingBarEndColor[c] - loadingBarStartColor[c] ) * i ) / last_color );
410  }
411  }
