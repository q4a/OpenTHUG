303 
304  // Extract contact point.
305  float x = p_face->point[X];
306  float y = p_face->point[Y];
307  float z = p_face->point[Z];
308 
309  Dbg_MsgAssert(p_object->IsTriangleCollision(),("Not triangle collision !!!"));
310 
311  Nx::CCollObjTriData *p_tri_data=p_object->GetGeometry();
312  Dbg_MsgAssert(p_tri_data,("NULL p_tri_data"));
313 
314  // Get all the vertices into a nice simple array of (x,y,z,v).
315  for( i = 0; i < 3; ++i )
316  {
317  //uint16 *vert_idxs = p_object->GetFaceVertIndicies(face_index);
318 
319  const Mth::Vector & vertex = p_object->GetVertexPos(p_tri_data->GetFaceVertIndex(face_index,i));
320 
321  // Check that the vertices match.
322  point[i].p.Set( vertex.GetX() - x, vertex.GetY() - y, vertex.GetZ() - z );
323 
324  // Here is where the conversion is made from values [0,255] to [0,1].
325 #ifdef __PLAT_NGC__
326  point[i].v = (float)( p_tri_data->GetVertexIntensity(face_index, i)) * ( 1.0f / 255.0f );
327 #else
328  point[i].v = (float)( p_tri_data->GetVertexIntensity(p_tri_data->GetFaceVertIndex(face_index, i)) ) * ( 1.0f / 255.0f );
329 #endif // __PLAT_NGC__
330  }
331 
332  // Calculate the intersection of the line from Point 0 through the origin, with the line 1-2.
333  float mua, mub;
334  Mth::Line line1( point[0].p, Mth::Vector( 0.0f, 0.0f, 0.0f ));
335  Mth::Line line2( point[1].p, point[2].p );
336  Mth::Vector a, b;
337 
338  Mth::LineLineIntersect( line1, line2, &a, &b, &mua, &mub, false );
339 
340  float l0, l1;
341  Mth::Vector length;
342 
343  // First interpolation a = i( 1, 2 ).
344  length = a - point[1].p;
345  l0 = length.Length();
346  length = point[2].p - point[1].p;
347  l1 = length.Length();
348  float va = point[1].v + (( point[2].v - point[1].v ) * ( l0 / l1 ));
349 
350  // Second interpolation b = i( 0, a ).
351  l0 = point[0].p.Length();
352  length = a - point[0].p;
353  l1 = length.Length();
354  brightness = point[0].v + (( va - point[0].v ) * ( l0 / l1 ));
355  }
356 
357 # ifdef __PLAT_XBOX__
358  // Xbox vertex colors run on a different scale to Ps2.
359  brightness *= 0.5f;
360 # endif
361 
362  return brightness;
363 }
