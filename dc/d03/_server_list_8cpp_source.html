175  {
176  break;
177  }

198  //Dbg_Printf( "Serverlist State: Starting MotD\n" );

322  man.m_server_list_wait_time = 0;
323  break;
324  }
325  case vSERVER_LIST_STATE_STARTING_LOBBY_LIST:
326  {
327  // First, show the dialog box
328  Script::RunScript( "CreateGettingLobbyListDialog" );
329 
330  // Free up the previous lobby list
331  man.mpLobbyMan->StopLobbyList();
332 
333  // Now spawn the lobby list -- this function also changes the state
334  man.mpLobbyMan->StartLobbyList();
335  break;
336  }
337  case vSERVER_LIST_STATE_GETTING_LOBBY_LIST:
338  {
339 
340  //Dbg_Printf( "Serverlist State: Getting Lobby List\n" );
341 
342  man.SetServerListState( vSERVER_LIST_STATE_WAIT );
343  man.m_server_list_wait_time = 0;
344  break;
345  }
346  case vSERVER_LIST_STATE_FAILED_LOBBY_LIST:
347  {
348  Script::CScriptStructure* pStructure;
349 
350  //Dbg_Printf( "Failed lobby list\n" );
351  man.mpLobbyMan->StopLobbyList();
352 
353  pStructure = new Script::CScriptStructure;
354  pStructure->AddComponent( Script::GenerateCRC( "message" ), ESYMBOLTYPE_STRING, Script::GetLocalString( "net_status_gamespy_no_connect" ));
355 
356  Script::RunScript( "CreateFailedLobbyListDialog", pStructure );
357 
358  delete pStructure;
359 
360  man.SetServerListState( vSERVER_LIST_STATE_SHUTDOWN );
361  break;
362  }
363  case vSERVER_LIST_STATE_GOT_LOBBY_LIST:
364  {
365  // If we didn't get any lobbies, that either means the matchmaker's down or their
366  // connection isn't valid
367  if( man.mpLobbyMan->GetLobbyInfo( 0 ) == NULL )
368  {
369  Script::CScriptStructure* pStructure = new Script::CScriptStructure;
370 
371  pStructure->AddComponent( Script::GenerateCRC( "message" ), ESYMBOLTYPE_STRING, Script::GetLocalString( "net_status_gamespy_no_connect" ));
372 
373  Script::RunScript( "CreateFailedLobbyListDialog", pStructure );
374 
375  delete pStructure;
376 
377  man.SetServerListState( vSERVER_LIST_STATE_SHUTDOWN );
378  }
379  else
380  {
381  man.SetServerListState( vSERVER_LIST_STATE_FILL_LOBBY_LIST );
382  }
383 
384  Pcm::PauseStream( 0 );
385  break;
386  }
387  case vSERVER_LIST_STATE_FILL_LOBBY_LIST:
388  {
389  man.mpLobbyMan->FillLobbyList();
390  man.SetServerListState( vSERVER_LIST_STATE_SHUTDOWN );
391  break;
392  }
393 
394  default:
395  break;
396  }

406 void Manager::s_auto_refresh_code( const Tsk::Task< Manager >& task )
407 {
408  Manager& man = task.GetData();
409  static Tmr::Time s_refresh_time = 0;
410 
411  if(( Tmr::GetTime() - s_refresh_time ) > vTIME_BETWEEN_AUTO_REFRESH )
412  {
413  s_refresh_time = Tmr::GetTime();
414 #ifdef __PLAT_XBOX__
415  man.RefreshServerList( true );
416 #endif
417  }
418  else
419  {
420  if( s_refresh_pending )
421  {
422  if(( Tmr::GetTime() - s_last_refresh_time ) > vLAN_SERVER_LIST_TIMEOUT )
423  {
424  Lst::Search< ServerInfo > sh;
425  ServerInfo* server, *next;
426 
427  man.ClearServerList();
428 
429  for( server = sh.FirstItem( man.m_temp_servers ); server; server = next )
430  {
431  next = sh.NextItem();
432  server->Remove();
433 
434  man.AddServerToMenu( server, man.m_servers.CountItems());
435  man.m_servers.AddToTail( server );
436  }
437  Script::RunScript( "update_lobby_server_list" );
438 
439  s_refresh_pending = false;
440  }
441  }
442  }

454 int Manager::s_handle_server_response( Net::MsgHandlerContext* context )
455 {
456  MsgServerDescription* msg;
457  ServerInfo *server_info;
458  Manager* manager = (Manager*) context->m_Data;
459  int i;
460 
461  //Dbg_Printf( "GOT SERVER RESPONSE!\n" );
462 
463  msg = (MsgServerDescription*) context->m_Msg;
464 
465  if( context->m_MsgLength != sizeof( MsgServerDescription ))
466  {
467  return Net::HANDLER_MSG_DONE;
468  }
469 
470 #ifdef __PLAT_XBOX__
471  if( manager->ServerAlreadyInList( msg->m_Name, &msg->m_XboxAddr ))
472  {
473  return Net::HANDLER_MSG_DONE;
474  }
475 #else
476  if( manager->ServerAlreadyInList( msg->m_Name, context->m_Conn->GetIP()))
477  {
478  return Net::HANDLER_MSG_DONE;
479  }
480 #endif
481 
482 
483 #ifdef __PLAT_XBOX__
484  if( memcmp( msg->m_Nonce, context->m_App->m_Nonce, 8 ))
485  {
486  return Net::HANDLER_MSG_DONE;
487  }
488 #endif
489 
490  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().NetworkHeap());
491 
492  server_info = new ServerInfo;
493 
494  strcpy( server_info->m_Name, msg->m_Name );
495  server_info->m_Ip = context->m_Conn->GetIP();
496  server_info->m_Latency = Tmr::GetTime() - msg->m_Timestamp;
497  strcpy( server_info->m_Level, msg->m_Level );
498  strcpy( server_info->m_Mode, msg->m_Mode );
499  server_info->m_MaxPlayers = msg->m_MaxPlayers;
500  server_info->m_NumPlayers = msg->m_NumPlayers;
501  server_info->m_MaxObservers = msg->m_MaxObservers;
502  server_info->m_NumObservers = msg->m_NumObservers;
503  server_info->m_Password = msg->m_Password;
504  server_info->m_GameStarted = msg->m_GameStarted;
505  server_info->m_HostMode = msg->m_HostMode;
506  server_info->m_Ranked = msg->m_Ranked;
507  server_info->m_SkillLevel = msg->m_SkillLevel;
508  server_info->m_Port = vHOST_PORT;
509 #ifdef __PLAT_XBOX__
510  memcpy( &server_info->m_XboxKeyId, &msg->m_XboxKeyId, sizeof( XNKID ));
511  memcpy( &server_info->m_XboxKey, &msg->m_XboxKey, sizeof( XNKEY ));
512  memcpy( &server_info->m_XboxAddr, &msg->m_XboxAddr, sizeof( XNADDR ));
513 #endif
514 
515  for( i = 0; i < server_info->m_NumPlayers; i++ )
516  {
517  server_info->AddPlayer( msg->m_PlayerNames[i] );
518  }
519 
520  //manager->AddServerToMenu( server_info, manager->m_servers.CountItems());
521 
522  //manager->m_servers.AddToTail( server_info );
523  manager->m_temp_servers.AddToTail( server_info );
524 
525  Mem::Manager::sHandle().PopContext();
526 
527  return Net::HANDLER_CONTINUE;
528 }
