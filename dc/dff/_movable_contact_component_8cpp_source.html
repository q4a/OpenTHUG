121 bool CMovableContactComponent::CheckForMovableContact ( CFeeler& feeler )
122 {
123  // we might now be in contact with something new
124  if (feeler.IsMovableCollision())
125  {
126  if (HaveContact())
127  {
128  if (mp_contact->GetObject() != feeler.GetMovingObject())
129  {
130  LoseAnyContact();
131  }
132  }
133 
134  if (!HaveContact())
135  {
136  // only create a new contact point if there was not one before, or if we just deleted one
137  mp_contact = new CContact(feeler.GetMovingObject());
138  return true;
139  }
140  }
141  else
142  {
143  // when we land on the ground, then we can't be in contact with anything, so delete any existing contact
144  if (HaveContact())
145  {
146  LoseAnyContact();
147  }
148  }
149 
150  return false;
151 }

174 CCompositeObject* CMovableContactComponent::CheckInsideObjects ( Mth::Vector& pos, const Mth::Vector& safe_pos )
175 {
176  // not movable contact functionality per say, but useful a service related to movable collision nonetheless
177 
178  // Check to see if the skater is inside any moving object and move him outside
179  // Algorithm
180  // start at a point 100 feet above the skater
181  // check collision down to the skater
182  // while there is a collision if skater has a moving object collision in the 100 feet above him
183  // and no collisions (with that object) going up to that point, then we are inside the object
184 
185  Mth::Vector top;
186  CCompositeObject *p_object = find_inside_object(pos, top);
187  if (!p_object) return NULL;
188 
189  // check if the last position is now fine...
190  Mth::Vector dummy;
191  if (!find_inside_object(safe_pos, dummy))
192  {
193  // first choice: move back to old pos
194  pos = safe_pos;
195  }
196  else
197  {
198  // recheck last position here, like adding in the contact movement
199 
200  Mth::Vector test = pos + p_object->GetVel() * Tmr::FrameLength() * 2.0f;
201 
202  if (!find_inside_object(test, dummy))
203  {
204  // first choice: move back to old pos
205  pos = test;
206  }
207  else
208  {
209  // last resort, snap up
210  pos = top;
211 
212  // when snapping up, put him an inch above the object; otherwise, you will not hit it next frame, as the point will be in the plane
213  pos[Y] += 1.0f;
214  }
215  }
216 
217  return p_object;
218 }

225 CCompositeObject* CMovableContactComponent::find_inside_object ( const Mth::Vector& pos, Mth::Vector& p_return_pos )
226 {
227  // Given a position (pos), then find out if this position is inside this moving object
228  // Assumes anything within 6 inches below the top of the object is actually outside as the code assumes
229 
230  // create a line that goes from 400 inches above this point to six inches above
231  CFeeler down_feeler;
232  down_feeler.m_start = pos;
233  down_feeler.m_start[Y] += 400.0f;
234  down_feeler.m_end = pos;
235  down_feeler.m_end[Y] += 6.0f;
236 
237  CFeeler up_feeler;
238  int loop_detect = 0;
239  while (true)
240  {
241  // return on final builds to handle highly obscure cases I've not forseen....
242  if (++loop_detect == 10) return NULL;
243 
244  // nothing above
245  if (!down_feeler.GetCollision()) return NULL;
246 
247  // something above, but not movable
248  if (!down_feeler.IsMovableCollision()) return NULL;
249 
250  // something movable above
251 
252  // Now we get the collision point, and check UP to that point from m_pos to a point just above that
253  up_feeler.m_start = pos;
254  up_feeler.m_end = down_feeler.GetPoint();
255  up_feeler.m_end[Y] += 0.1f;
256 
257  if (!up_feeler.GetCollision())
258  {
259  p_return_pos = down_feeler.GetPoint();
260  return down_feeler.GetMovingObject();
261  }
262  else
263  {
264  down_feeler.m_start = down_feeler.GetPoint();
265  down_feeler.m_start[Y] -= 0.1f;
266  }
267  }
268 }
