189 # endif // __USE_PROFILER__
190 #endif
191 
192 # ifdef __PLAT_NGC__
193  if( start_render_hook )
194  {
195  start_render_hook->Call();
196  }
197  if( !display_tasks_paused )
198  {
199  display_task_stack.Process( currently_profiling );
200  }
201 # else
202  if (!display_tasks_paused)
203  {
204  // If paused don't call the start_render_hook
205  // as this clears the none-visible frame buffer
206  // which will result in a flash when display is unpaused
207  if ( start_render_hook ) // set up for rendering
208  {
209  start_render_hook->Call();
210  }
211 
212 // printf ("\nTiming render\n");
213 
214  display_task_stack.Process(currently_profiling); // service rendering routines
215  }
216 # endif // __PLAT_NGC__
217 
218 #ifndef __PLAT_WN32__
219 # ifdef __USE_PROFILER__
220  Sys::CPUProfiler->PopContext();
221  Sys::CPUProfiler->PushContext( 255, 255, 0 );
222 # endif // __USE_PROFILER__
223 #endif
224 
225  if ( end_render_hook ) // end rendering
226  {
227  end_render_hook->Call();
228  }
229 #ifndef __PLAT_WN32__
230 # ifdef __USE_PROFILER__
231  Sys::CPUProfiler->PopContext();
232 # endif // __USE_PROFILER__
233 #endif

240 void Manager::MainLoop( void )
241 {
242 
243 
244  bool old_flag = done; // push the current done flag
245 
246  done = false;
247 
248  while ( !done )
249  {
250 
251  if (trigger_profiling)
252  {
253  printf ("\nProfiling.....Start of main loop\n\n");
254  trigger_profiling--;
255  currently_profiling = 1;
256  }
257 
258 
259 #ifndef __PLAT_WN32__
260 
261 #ifdef __PLAT_NGPS__
262 // for profiling on the PS2, we do a bit of the code from sPreRender here
263 // so we can time the final waiting for GS
264 # ifdef __USE_PROFILER__
265  Sys::CPUProfiler->PushContext( 0,0,255 ); // blue = wait for VU/DMA/GPU to finish
266 # endif
267  NxPs2::WaitForRendering(); // PS2 Specific, wait for prior frame's DMA to finish
268  NxPs2::StuffAfterGSFinished(); // PS2 Specific, DMA altering stuff run after previous frame's DMA finished
269 # ifdef __USE_PROFILER__
270  Sys::CPUProfiler->PopContext( );
271  Sys::CPUProfiler->PushContext( 128,0,0 ); // red = wait for vblank
272 # endif
273  int framerate = Script::GetInteger(0x3214c818/*"lock_framerate"*/);
274  if (framerate) // normal vsync
275  {
276  static uint64 next_vblanks= 0;
277  // Still call VSync, as it updates some functions
278  Tmr::VSync();
279  while (Tmr::GetVblanks()<next_vblanks)
280  {
281  // just hanging until vblanks gets big enough
282  }
283  next_vblanks = Tmr::GetVblanks() + framerate;
284  }
285  else
286  {
287  Tmr::VSync1();
288  }
289 # ifdef __USE_PROFILER__
290  Sys::CPUProfiler->PopContext( );
291 # endif
292 #endif
293 #ifndef __PLAT_NGC__
294 # ifdef __USE_PROFILER__
295  Sys::Profiler::sStartFrame();
296 # endif
297 #endif
298 
299  Nx::CEngine::sPreRender(); // start rendering previous frame's DMA list
300 
301 #ifdef __PLAT_NGPS__
302  Sfx::CSpuManager::sUpdateStatus(); // Garrett: This should go into some system task, but I'll put it here for now
303 #endif
304 
305 // Sys::Profiler::sStartFrame();
306 #endif
307 
308 #ifdef __PLAT_NGC__
309 # ifdef __USE_PROFILER__
310  Sys::Profiler::sStartFrame();
311 # endif
312 # endif
313 
314  service_system();
315 
316 #if defined(__PLAT_NGPS__) && defined(BATCH_TRI_COLLISION)
317  // Enable VU0 collision
318  bool got_vu0 = Nx::CBatchTriCollMan::sUseVU0Micro();
319  Dbg_Assert(got_vu0);
320 #endif
321 
322  #ifdef __PLAT_NGPS__
323 // snProfSetRange( -1, (void*)0, (void*)-1);
324 // snProfSetFlagValue(0x01);
325  #endif
326 
327  game_logic();
328 
329  #ifdef __PLAT_NGPS__
330 // snProfSetRange( 4, (void*)NULL, (void*)-1);
331  #endif
332 
333 
334 #if defined(__PLAT_NGPS__) && defined(BATCH_TRI_COLLISION)
335  // Disable VU0 collision
336  if (got_vu0)
337  {
338  Nx::CBatchTriCollMan::sDisableVU0Micro();
339  }
340 #endif
341 
342 #ifndef __PLAT_WN32__
343  // Display the memory contents, (if memview is active)
344  MemView_Display();
345 # ifdef __USE_PROFILER__
346  Sys::CPUProfiler->PushContext( 255, 255, 0 ); // yellow = render world
347 # endif
348 
349  #ifdef __PLAT_NGPS__
350 // snProfSetRange( -1, (void*)0, (void*)-1);
351 // snProfSetFlagValue(0x01);
352  #endif
353 
354  Nx::CEngine::sRenderWorld();
355 #ifdef __PLAT_NGC__
356 #ifdef __USE_PROFILER__
357  Sys::Render_Profiler();
358 #endif
359 #endif
360  // Mick: bit of a patch here, we need some better debug hooks
361  Mdl::Rail_DebugRender();
362 # ifdef __USE_PROFILER__
363  Sys::CPUProfiler->PushContext( 0, 0, 0 ); // Black (Under Yellow) = sPostRender
364 # endif // __USE_PROFILER__
365  Nx::CEngine::sPostRender(); // Previous frames profiler is rendered here
366 # ifdef __USE_PROFILER__
367  Sys::CPUProfiler->PopContext( );
368 # endif
369 
370 
371 
372 
373 # ifdef __USE_PROFILER__
374  Sys::CPUProfiler->PopContext( );
375 # endif
376  Tmr::OncePerRender();
377  #ifdef __PLAT_NGPS__
378 // snProfSetRange( 4, (void*)NULL, (void*)-1);
379  #endif
380 #endif
381 
382  currently_profiling = false;
383 
384  }
385 
386  done = old_flag;
387 }
