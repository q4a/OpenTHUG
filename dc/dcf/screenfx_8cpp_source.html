237 static HRESULT calculate_depth_mapping( float fDepth0, float fDepth1, float* pfAlphaOffset, float* pfAlphaSlope, float* pfBlueOffset, float* pfBlueSlope )
238 {
239  // Check range of args
240  if( fDepth0 < 0.0f ) fDepth0 = 0.0f;
241  if( fDepth0 > 1.0f ) fDepth0 = 1.0f;
242  if( fDepth1 < 0.0f ) fDepth1 = 0.0f;
243  if( fDepth1 > 1.0f ) fDepth1 = 1.0f;
244 
245  if( fDepth1 < fDepth0 )
246  {
247  // Swap depth to make fDepth0 <= fDepth1
248  float t = fDepth1;
249  fDepth1 = fDepth0;
250  fDepth0 = t;
251  }
252 
253  // Calculate quantized values
254  float fAlpha0, fBlue0;
255  float fQuantizedDepth0 = FQuantizedDepth(fDepth0, &fAlpha0, &fBlue0);
256  float fAlpha1, fBlue1;
257  float fQuantizedDepth1 = FQuantizedDepth(fDepth1, &fAlpha1, &fBlue1);
258 
259  // Calculate offset and slopes
260  float fScale = 1.0f / (fQuantizedDepth1 - fQuantizedDepth0);
261  if( fScale > g_fPixelShaderScale )
262  {
263  fScale = g_fPixelShaderScale; // This is the steepest slope we can handle
264  fDepth0 = 0.5f * (fDepth0 + fDepth1) - 0.5f / fScale; // Move start so that peak is in middle of fDepth0 and fDepth1
265  fDepth1 = fDepth0 + 1.0f / fScale;
266  fQuantizedDepth0 = FQuantizedDepth(fDepth0, &fAlpha0, &fBlue0);
267  fQuantizedDepth1 = FQuantizedDepth(fDepth1, &fAlpha1, &fBlue1);
268  }
269 
270  (*pfAlphaOffset) = fAlpha0;
271  (*pfAlphaSlope) = fScale / g_fPixelShaderScale;
272  (*pfBlueSlope) = fScale * (1.0f/255.0f); // blue ramp adds more levels to the ramp
273 
274  // Align peak of map to center by calculating the quantized alpha value
275 // *pfBlueOffset = 0.5f; // zero biased up by 0.5f
276 // float fZeroDesired = (fQuantizedDepth0 - fDepth0) / (fDepth1 - fDepth0);
277 // float fZero = FUnitMap(fAlpha0, fBlue0, *pfAlphaOffset, *pfAlphaSlope, *pfBlueOffset, *pfBlueSlope);
278 // float fOneDesired = (fQuantizedDepth1 - fDepth0) / (fDepth1 - fDepth0);
279 // float fOne = FUnitMap(fAlpha1, fBlue1, *pfAlphaOffset, *pfAlphaSlope, *pfBlueOffset, *pfBlueSlope);
280 // *pfBlueOffset = 0.5f * (fZeroDesired-fZero + fOneDesired-fOne) + 0.5f; // biased up by 0.5f
281  (*pfBlueOffset) = fBlue0;
282 
283  return S_OK;
284 }

412  EngineGlobals.p_Device->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_ONE );

429  set_vertex_shader( D3DFVF_XYZRHW | D3DFVF_TEX4 );

432  float x0 = -0.5f;
433  float y0 = -0.5f;

448  { x1, y0, 1.0f, 1.0f, },
449  { x0, y1, 1.0f, 1.0f, },
450  { x1, y1, 1.0f, 1.0f, }
451  };

454  xx = 0; // current texture stage
455  FLOAT fOffsetScaleX, fOffsetScaleY; // convert destination coords to source texture coords

462  fOffsetScaleY = (FLOAT)dwSuperSampleY * fHeightScale;

466  v1 = (FLOAT)descSrc.Height * fHeightScale;

471  fOffsetScaleY = (FLOAT)dwSuperSampleY;
472  u0 = 0.0f;
473  v0 = 0.0f;
474  u1 = (FLOAT)descSrc.Width;
475  v1 = (FLOAT)descSrc.Height;
476  }

478  DWORD rPSInput[4];
479  for( DWORD dwSample = 0; dwSample < dwNumSamples; dwSample++ )
480  {
481  // Set filter coefficients
482  FLOAT fValue = rSample[dwSample].fValue;
483 // float rf[4] = {fValue, fValue, fValue, fValue};
484 // EngineGlobals.p_Device->SetPixelShaderConstant(xx, rf, 1); // positive coeff
485

489  rPSInput[xx] = PS_INPUTMAPPING_SIGNED_NEGATE | ((xx % 2) ? PS_REGISTER_C1 : PS_REGISTER_C0);
490  }
491  else
492  {
493  rColor[xx] = D3DXCOLOR(fValue, fValue, fValue, fValue);
494  rPSInput[xx] = PS_INPUTMAPPING_SIGNED_IDENTITY | ((xx % 2) ? PS_REGISTER_C1 : PS_REGISTER_C0);
495  }
496

498  FLOAT fOffsetX = rSample[dwSample].fOffsetX * fOffsetScaleX;
499  FLOAT fOffsetY = rSample[dwSample].fOffsetY * fOffsetScaleY;
500  aQuad[0].tex[xx].u = u0 + fOffsetX;
501  aQuad[0].tex[xx].v = v0 + fOffsetY;
502  aQuad[1].tex[xx].u = u1 + fOffsetX;
503  aQuad[1].tex[xx].v = v0 + fOffsetY;
504  aQuad[2].tex[xx].u = u0 + fOffsetX;
505  aQuad[2].tex[xx].v = v1 + fOffsetY;
506  aQuad[3].tex[xx].u = u1 + fOffsetX;
507  aQuad[3].tex[xx].v = v1 + fOffsetY;
508

510  if( xx == 4 || dwSample == dwNumSamples - 1 ) // max texture stages or last sample
511  {
512  // zero out unused texture stage coefficients
513  // (only for last filter sample, when number of samples is not divisible by 4)
514  for( ; xx < 4; xx++ )
515  {
516  set_texture( xx, NULL );
517  rColor[xx] = 0;
518  rPSInput[xx] = PS_INPUTMAPPING_UNSIGNED_IDENTITY | PS_REGISTER_ZERO;
519  }
520

522  EngineGlobals.p_Device->SetRenderState( D3DRS_PSCONSTANT0_0, rColor[0] );
523  EngineGlobals.p_Device->SetRenderState( D3DRS_PSCONSTANT1_0, rColor[1] );
524  EngineGlobals.p_Device->SetRenderState( D3DRS_PSCONSTANT0_1, rColor[2] );

529  PS_COMBINERINPUTS( rPSInput[0] | PS_CHANNEL_RGB, PS_REGISTER_T0 | PS_CHANNEL_RGB | PS_INPUTMAPPING_SIGNED_IDENTITY,
530  rPSInput[1] | PS_CHANNEL_RGB, PS_REGISTER_T1 | PS_CHANNEL_RGB | PS_INPUTMAPPING_SIGNED_IDENTITY ) );
531  EngineGlobals.p_Device->SetRenderState( D3DRS_PSALPHAINPUTS0,
532  PS_COMBINERINPUTS( rPSInput[0] | PS_CHANNEL_ALPHA, PS_REGISTER_T0 | PS_CHANNEL_ALPHA | PS_INPUTMAPPING_SIGNED_IDENTITY,
533  rPSInput[1] | PS_CHANNEL_ALPHA, PS_REGISTER_T1 | PS_CHANNEL_ALPHA | PS_INPUTMAPPING_SIGNED_IDENTITY ) );
534  EngineGlobals.p_Device->SetRenderState( D3DRS_PSRGBINPUTS1,
