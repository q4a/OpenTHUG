241 void CVehicleSoundComponent::update_engine_sounds ( )
242 {
243  float frame_length = Tmr::FrameLength();
244 
245  CVehicleComponent::SWheel* p_wheels = mp_vehicle_component->mp_wheels;
246 
247  // determine the smallest of all the drive wheels' rotational velocity
248  float min_vel = 1e20f;
249  for (int n = CVehicleComponent::vVP_NUM_WHEELS; n--; )
250  {
251  if (!p_wheels[n].drive) continue;
252  min_vel = Mth::Min(min_vel, Mth::Abs(p_wheels[n].rotvel) * p_wheels[n].radius);
253  }
254  float frame_effective_speed = IPS_TO_MPH(min_vel);
255 
256  // adjust effective speed based on throttle position
257  bool throttle = (mp_vehicle_component->m_controls.throttle || mp_vehicle_component->m_controls.reverse) && !mp_vehicle_component->m_controls.brake;
258  if (!throttle)
259  {
260  frame_effective_speed *= get_global_param(CRCD(0xec67bc97, "idle_engine_effective_speed_factor"));
261  }
262 
263  // smooth out frame-to-frame fluxuations
264  m_effective_speed = Mth::Lerp(frame_effective_speed, m_effective_speed, get_global_param(CRCD(0x2a7095b, "effective_speed_lerp_rate")) * frame_length);
265  if (m_effective_speed < frame_effective_speed)
266  {
267  m_effective_speed = Mth::ClampMax(m_effective_speed + get_global_param(CRCD(0xca5efd91, "effective_speed_adjust_up_rate")) * frame_length, m_effective_speed);
268  }
269  else
270  {
271  m_effective_speed = Mth::ClampMin(m_effective_speed - get_global_param(CRCD(0x77c8e7c8, "effective_speed_adjust_down_rate")) * frame_length, m_effective_speed);
272  }
273  DEBUG_ENGINE(("speed: %.2f mph", m_effective_speed));
274 
275  // determine the appropriate gear for this frame
276  if (mp_vehicle_component->m_num_wheels_in_contact != 0)
277  {
278  if (m_effective_gear < m_num_gears - 1 && m_effective_speed > m_gears[m_effective_gear].upshift_point)
279  {
280  m_effective_gear++;
281  m_gear_shift_time_stamp = Tmr::GetTime();
282  DEBUG_ENGINE(("upshift to gear: %i", m_effective_gear));
283  }
284  else if (m_effective_gear > 0 && m_effective_speed < m_gears[m_effective_gear].downshift_point)
285  {
286  m_effective_gear--;
287  m_gear_shift_time_stamp = Tmr::GetTime();
288  DEBUG_ENGINE(("downshift to gear: %i", m_effective_gear));
289  }
290  }
291  else if (mp_vehicle_component->m_air_time > get_global_param(CRCD(0xc0f451d4, "engine_airborne_delay")))
292  {
293  // when airborne, drop to lowest gear and use no gear shift velocity damping
294  m_effective_gear = 0;
295  }
296  DEBUG_ENGINE(("gear: %i", m_effective_gear + 1));
297 
298  // calculate a volume adjustment based on the time since the last gear shift
299  float gear_shift_vol_factor = Mth::ClampMax(
300  Mth::LinearMap(
301  get_global_param(CRCD(0xbfa32092, "gear_shift_min_vol_factor")),
302  1.0f,
303  Tmr::ElapsedTime(m_gear_shift_time_stamp),
304  0.0f,
305  get_global_param(CRCD(0xfb6bd20e, "gear_shift_vol_adjustment_duration"))
306  ),
307  1.0f
308  );
309 
310  // calculate this frame's engine rpm based on the current gear and speed
311  float frame_engine_rpm = Mth::LinearMap(
312  m_gears[m_effective_gear].bottom_rpm,
313  m_gears[m_effective_gear].top_rpm,
314  m_effective_speed,
315  m_effective_gear ? m_gears[m_effective_gear - 1].upshift_point : 0.0f,
316  m_gears[m_effective_gear].upshift_point
317  );
318  // DEBUG_ENGINE(("speed top: %.2f", m_gears[m_effective_gear].upshift_point));
319  // DEBUG_ENGINE(("speed bottom: %.2f", m_effective_gear ? m_gears[m_effective_gear - 1].upshift_point : 0.0f));
320  // DEBUG_ENGINE(("rpm top: %.2f", m_gears[m_effective_gear].top_rpm));
321  // DEBUG_ENGINE(("rpm bottom: %.2f", m_gears[m_effective_gear].bottom_rpm));
322 
323  // smooth out the engine rpm changes (causes nice shifting effects)
324  m_effective_engine_rpm = Mth::Lerp(
325  m_effective_engine_rpm,
326  frame_engine_rpm,
327  get_global_param(CRCD(0x3dd9c040, "engine_rpm_lerp_rate")) * frame_length
328  );
329  DEBUG_ENGINE(("engine RPM: %.2f", m_effective_engine_rpm));
330 
331  // calculate a number between 0.0f and 1.0f which will scale the engine sound's pitch and volume
332  float sound_factor = Mth::Clamp(
333  Mth::LinearMap(
334  0.0f,
335  1.0f,
336  m_effective_engine_rpm,
337  get_param(CRCD(0x05ac84a3, "MinEngineRPM")),
338  get_param(CRCD(0x95df9449, "MaxEngineRPM"))
339  ),
340  0.0f,
341  1.0f
342  );
343 
344  // setup volume
345  Sfx::sVolume volume;
346  Sfx::CSfxManager::Instance()->SetVolumeFromPos(&volume, GetObject()->GetPos(), Sfx::CSfxManager::Instance()->GetDropoffDist(m_engine_sound_checksum));
347  float volume_percent = gear_shift_vol_factor * Mth::Lerp(
348  get_param(CRCD(0xb8f81277, "MinEngineVol")),
349  get_param(CRCD(0x288b029d, "MaxEngineVol")),
350  sound_factor
351  );
352  volume.PercentageAdjustment(volume_percent);
353  DEBUG_ENGINE(("engine volume: %.0f", volume_percent));
354 
355  // setup pitch
356  float pitch = Mth::Lerp(
357  get_param(CRCD(0x2660af3b, "MinEnginePitch")),
358  get_param(CRCD(0x9f46344a, "MaxEnginePitch")),
359  sound_factor
360  );
361  DEBUG_ENGINE(("engine pitch: %.0f", pitch));
362 
363  // play or adjust sound
364  if (!m_engine_sound_id)
365  {
366  m_engine_sound_id = Sfx::CSfxManager::Instance()->PlaySound(m_engine_sound_checksum, &volume, pitch, NULL);
367  }
368  else
369  {
370  Sfx::CSfxManager::Instance()->UpdateLoopingSound(m_engine_sound_id, &volume, pitch);
371  }
372 }

379 void CVehicleSoundComponent::update_tire_sounds ( )
380 {
381  CVehicleComponent::SWheel* p_wheels = mp_vehicle_component->mp_wheels;
382 
383  // determine the largest of all the wheels' slip velocity
384  float max_slip_vel = 0.0f;
385 
386  if (mp_vehicle_component->m_controls.brake || mp_vehicle_component->m_controls.handbrake)
387  {
388  for (int n = CVehicleComponent::vVP_NUM_WHEELS; n--; )
389  {
390  // handbrake skid states do not depend on the tire's slip velocity, so we check the slip velocity ourselves
391  if ((p_wheels[n].state == CVehicleComponent::SWheel::SLIPPING || p_wheels[n].state == CVehicleComponent::SWheel::SKIDDING)
392  || ((p_wheels[n].state == CVehicleComponent::SWheel::HANDBRAKE_LOCKED || p_wheels[n].state == CVehicleComponent::SWheel::HANDBRAKE_THROTTLE)
393  && p_wheels[n].slip_vel > p_wheels[n].max_static_velocity))
394  {
395  max_slip_vel = Mth::Max(max_slip_vel, p_wheels[n].slip_vel);
396  }
397  }
398  }
399 
400  m_effective_tire_slip_vel = Mth::Lerp(
401  m_effective_tire_slip_vel,
402  max_slip_vel,
403  get_global_param(CRCD(0x95fbe774, "tire_slip_lerp_rate")) * Tmr::FrameLength()
404  );
405  DEBUG_TIRES(("tire slip velocity: %.2f", m_effective_tire_slip_vel));
406 
407  float sound_factor = Mth::Clamp(
408  m_effective_tire_slip_vel / get_param(CRCD(0x28f1aad0, "FullTireSlipVelocity")),
409  0.0f,
410  1.0f
411  );
412 
413  // setup volume
414  Sfx::sVolume volume;
415  Sfx::CSfxManager::Instance()->SetVolumeFromPos(&volume, GetObject()->GetPos(), Sfx::CSfxManager::Instance()->GetDropoffDist(m_tire_sound_checksum));
416  float volume_percent = Mth::Lerp(
417  get_param(CRCD(0x43e6cdf2, "MinTireVol")),
418  get_param(CRCD(0x42ea5746, "MaxTireVol")),
419  sound_factor
420  );
421  volume.PercentageAdjustment(volume_percent);
422  DEBUG_TIRES(("tire volume: %.0f", volume_percent));
423 
424  // setup pitch
425  float pitch = Mth::Lerp(
426  get_param(CRCD(0x76f8f76e, "MinTirePitch")),
427  get_param(CRCD(0xe68be784, "MaxTirePitch")),
428  sound_factor
429  );
430  DEBUG_TIRES(("tire pitch: %.0f", pitch));
431 
432  if (!volume.IsSilent())
433  {
434  if (!m_tire_sound_id)
435  {
436  m_tire_sound_id = Sfx::CSfxManager::Instance()->PlaySound(m_tire_sound_checksum, &volume, pitch, NULL);
437  }
438  else
439  {
440  Sfx::CSfxManager::Instance()->UpdateLoopingSound(m_tire_sound_id, &volume, pitch);
441  }
442  }
443  else if (m_tire_sound_id)
444  {
445  Sfx::CSfxManager::Instance()->StopSound(m_tire_sound_id);
446  m_tire_sound_id = 0;
447  }
448 }

455 void CVehicleSoundComponent::update_collision_sounds ( )
456 {
457  if (Tmr::ElapsedTime(m_latest_collision_sound_time_stamp) < get_global_param(CRCD(0x71b4b9fe, "collision_mute_delay"))) return;
458 
459  float sound_factor = mp_vehicle_component->m_max_normal_collision_impulse / get_param(CRCD(0xcea59d0b, "FullCollision"));
460  if (sound_factor > get_global_param(CRCD(0xf6981837, "collision_cutoff_factor")))
461  {
462  DEBUG_COLLISION(("collision: %.2f", mp_vehicle_component->m_max_normal_collision_impulse));
463 
464  m_collide_sound_struct->AddFloat(CRCD(0x9e497fc6, "Percent"), 100.0f * sound_factor);
465  mp_sound_component->PlayScriptedSound(m_collide_sound_struct);
466 
467  m_latest_collision_sound_time_stamp = Tmr::GetTime();
468 
469  Script::CStruct* p_params = new Script::CStruct;
470  p_params->AddFloat(CRCD(0x29461e27, "Strength"), Mth::Clamp(sound_factor, 0.0f, 1.0f));
471  mp_vehicle_component->mp_skater->SelfEvent(CRCD(0xc4a60423, "Vehicle_BodyCollision"), p_params);
472  delete p_params;
473  }

481 void CVehicleSoundComponent::fetch_script_parameters ( )
482 {
483  // get sound setup structure
484  Script::CStruct* sound_setups_struct = Script::GetStructure(CRCD(0x3c2da7f5, "PlayerVehicleSounds"), Script::ASSERT);
485  sound_setups_struct->GetStructure(mp_vehicle_component->GetSoundSetupChecksum(), &m_sound_setup_struct, Script::ASSERT);
486 
487  // extract sound checksum
488  m_sound_setup_struct->GetChecksum(CRCD(0x306f8a9f, "EngineSound"), &m_engine_sound_checksum);
489  m_sound_setup_struct->GetChecksum(CRCD(0x667fa96a, "TireSound"), &m_tire_sound_checksum);
490 
491  // extract gear information
492 
493  Script::CArray* gear_array;
494  m_sound_setup_struct->GetArray(CRCD(0x83273e6b, "Gears"), &gear_array, Script::ASSERT);
495 
496  m_num_gears = gear_array->GetSize();
497  Dbg_MsgAssert(m_num_gears <= vVS_MAX_NUM_GEARS, ("PlayerVehicleSound '%s' exceeds the maximum number of %i allowed gears", Script::FindChecksumName(mp_vehicle_component->GetSoundSetupChecksum()), vVS_MAX_NUM_GEARS));
498  for (int n = 0; n < m_num_gears; n++)
499  {
500  Script::CStruct* gear_struct = gear_array->GetStructure(n);
501 
502  gear_struct->GetFloat(CRCD(0xfad20901, "UpshiftPoint"), &m_gears[n].upshift_point, Script::ASSERT);
503  gear_struct->GetFloat(CRCD(0xfcf5c8a2, "DownshiftPoint"), &m_gears[n].downshift_point, Script::ASSERT);
504  gear_struct->GetFloat(CRCD(0x735b5ce4, "BottomRPM"), &m_gears[n].bottom_rpm, Script::ASSERT);
505  gear_struct->GetFloat(CRCD(0xe59a89a6, "TopRPM"), &m_gears[n].top_rpm, Script::ASSERT);
506 
507  Dbg_MsgAssert(m_gears[n].upshift_point > m_gears[n].downshift_point, ("In gear %i of PlayerVehicleSound '%s,' DownshiftPoint exceeds UpshiftPoint", n + 1, Script::FindChecksumName(mp_vehicle_component->GetSoundSetupChecksum())));
508  Dbg_MsgAssert(!n || m_gears[n].downshift_point < m_gears[n - 1].upshift_point, ("In gear %i of PlayerVehicleSound '%s', DownshiftPoint exceeds lower gear's UpshiftPoint", n + 1, Script::FindChecksumName(mp_vehicle_component->GetSoundSetupChecksum())));
509  }
510 
511  // extract collision sound structure
512  m_sound_setup_struct->GetStructure(CRCD(0xbf8e0ace, "CollideSound"), &m_collide_sound_struct, Script::ASSERT);
513 
514  // if we're in a create-a-goal, swap sound checksums for default sound checksums, as the true sounds may not be loaded
515  if (m_use_default_sounds)
516  {
517  Script::CStruct* global_sound_setup = Script::GetStructure(CRCD(0x38ec37ac, "PlayerVehicleSoundGlobalParameters"), Script::ASSERT);
518 
519  global_sound_setup->GetChecksum(CRCD(0xf5406e1a, "CAGEngineSound"), &m_engine_sound_checksum, Script::ASSERT);
520  global_sound_setup->GetChecksum(CRCD(0x60e93a2, "CAGTireSound"), &m_tire_sound_checksum, Script::ASSERT);
521  global_sound_setup->GetStructure(CRCD(0x22995285, "CAGCollideSound"), &m_collide_sound_struct, Script::ASSERT);
522  }
523 }
