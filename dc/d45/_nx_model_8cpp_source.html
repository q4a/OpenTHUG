279 bool CModel::Render( Mth::Matrix* pMatrix, bool no_anim, Gfx::CSkeleton* pSkeleton )
280 {
281  // don't display the skin, unless we're
282  // in the standard "textured" mode
283  bool is_textured = m_renderMode == vTEXTURED;
284 
285  int numGeoms = GetNumGeoms();
286  for ( int i = 0; i < numGeoms; i++ )
287  {
288  Dbg_Assert( GetGeomByIndex(i) );
289  GetGeomByIndex(i)->SetActive( is_textured && m_active && (m_geomActiveMask&(1<<i)) );
290  }
291 
292  // scale up the matrix by the appropriate amount...
293  if ( m_scalingEnabled )
294  {
295  pMatrix->ScaleLocal(m_scale);
296  }
297 
298  // make sure the skeleton matches up...
299  if ( pSkeleton )
300  {
301  Dbg_MsgAssert( m_numBones != 0, ( "Trying to update a model with incorrect number of bones (expected %d, found %d)", m_numBones, pSkeleton->GetNumBones() ) );
302  Dbg_Assert( mp_skeletonMatrices );
303  }
304  else
305  {
306  Dbg_MsgAssert( m_numBones == 0, ( "Trying to update a model with incorrect number of bones (expected %d, found %d)", 0, m_numBones ) );
307  Dbg_Assert( mp_skeletonMatrices == NULL );
308  }
309 
310  switch ( m_renderMode )
311  {
312  case vTEXTURED:
313  {
314  if ( pSkeleton )
315  {
316  (*pMatrix)[Mth::RIGHT][W] = 0.0f;
317  (*pMatrix)[Mth::UP][W] = 0.0f;
318  (*pMatrix)[Mth::AT][W] = 0.0f;
319  (*pMatrix)[Mth::POS][W] = 1.0f;
320 
321  int numGeoms = GetNumGeoms();
322 
323  if ( no_anim )
324  {
325  // update root position without updating bones
326  for ( int i = 0; i < numGeoms; i++ )
327  {
328  Dbg_Assert( GetGeomByIndex(i) );
329  GetGeomByIndex(i)->Render( pMatrix, NULL, 0 );
330  }
331  }
332  else
333  {
334  // update both root position AND bones
335  for ( int i = 0; i < numGeoms; i++ )
336  {
337  Dbg_Assert( GetGeomByIndex(i) );
338  GetGeomByIndex(i)->Render( pMatrix, pSkeleton->GetMatrices(), pSkeleton->GetNumBones() );
339  }
340  }
341  }
342  else
343  {
344  // the following should NOT be necessary,
345  // but it is.
346  (*pMatrix)[Mth::RIGHT][W] = 0.0f;
347  (*pMatrix)[Mth::UP][W] = 0.0f;
348  (*pMatrix)[Mth::AT][W] = 0.0f;
349  (*pMatrix)[Mth::POS][W] = 1.0f;
350 
351  int numGeoms = GetNumGeoms();
352  for ( int i = 0; i < numGeoms; i++ )
353  {
354  Dbg_Assert( GetGeomByIndex(i) );
355  GetGeomByIndex(i)->Render( pMatrix, NULL, 0 );
356  }
357  }
358  }
359  break;
360 
361  case vSKELETON:
362  {
363  if ( pSkeleton )
364  {
365  pSkeleton->Display( pMatrix, 0.5f, 0.5f, 1.0f );
366  }
367  }
368  break;
369 
370  case vBBOX:
371  default:
372  {
373  Mth::Matrix matrix = *pMatrix;
374 
375  // really, the AddDebugBox function should be
376  // able to draw with only one matrix, rather
377  // than matrix+pos.
378  matrix[Mth::POS] = Mth::Vector( 0, 0, 0 );
379 
380  Mth::Vector pos = pMatrix->GetPos();
381 
382  // set up bounding box
383  SBBox theBox;
384  theBox.m_max.Set(10.0f, 10.0f, 10.0f);
385  theBox.m_min.Set(-10.0f, -10.0f, -10.0f);
386 
387  // For now, draw a bounding box
388  Gfx::AddDebugBox( matrix, pos, &theBox, NULL, 1, NULL );
389  }
390  break;
391 
392  case vGOURAUD:
393  case vFLAT:
394  case vWIREFRAME:
395  Dbg_Assert( 0 );
396  break;
397 
398  case vNONE:
399  // draw nothing...
400  break;
401 
402  }
403 
404  return true;
405 }

437 bool CModel::AddGeom(uint32 assetName, uint32 geomName, bool supportMultipleMaterialColors)
438 {
439  // now that a new geom has been added to the model,
440  // consider the bounding sphere "dirty"
441  m_boundingSphereCached = false;
442 
443  // for now, assume that if we're using this method of creating assets,
444  // then we'll be using the asset manager and a texDictOffset of 0
445 
446  bool success = false;
447 
448  Dbg_MsgAssert( m_numGeoms < MAX_GEOMS, ( "Too many geoms for this model (max=%d)", MAX_GEOMS ) );
449 
450  // the asset must already exist in the assman by this point!
451  // note that we're not loading the asset, only getting it
452  Ass::CAssMan * ass_man = Ass::CAssMan::Instance();
453 
454  Nx::CMesh* pMesh = NULL;
455  pMesh = (Nx::CMesh*)ass_man->GetAsset( assetName, false );
456  if( !pMesh )
457  {
458  Dbg_MsgAssert( false,( "Could not get asset %08x", assetName ));
459  }
460 
461  m_preloadedMeshNames[m_numPreloadedMeshes] = assetName;
462  m_numPreloadedMeshes++;
463 
464  mp_geom[m_numGeoms] = Nx::CEngine::sInitGeom();
465 
466  if ( mp_geom[m_numGeoms] )
467  {
468  mp_geom[m_numGeoms]->LoadGeomData(pMesh, this, supportMultipleMaterialColors);
469  mp_geomTexDict[m_numGeoms] = pMesh->GetTextureDictionary();
470  mp_geom[m_numGeoms]->EnableShadow(m_shadowEnabled);
471  m_geomName[m_numGeoms] = geomName;
472  m_geomActiveMask |= (1<<m_numGeoms);
473 
474  // Set the model lights, if any
475  if (mp_model_lights)
476  {
477  mp_geom[m_numGeoms]->SetModelLights(mp_model_lights);
478  }
479 
480  m_numGeoms++;
481 
482  SetRenderMode( vTEXTURED );
483  }
484 
485  // TODO: The mesh file name represents
486  // mesh #0, but the mesh file name is
487  // already kind of kludge that will be
488  // cleaned up when CGeoms are properly
489  // implemented.
490  if ( m_numGeoms == 1 )
491  {
492  // doesn't have a filename because it came from a stream...
493  m_primaryMeshName = Script::GenerateCRC("data_from_stream");
494  }
495 
496  return success;
497 }
