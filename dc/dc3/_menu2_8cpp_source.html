387 void CBaseMenu::reposition()
388 {
389  float z = 0.0f; // functions as a virtual x or y
390  float biggest = 0.0f; // functions as virtual 'widest' or 'highest'
391 
392  bool use_regular_space = (m_regular_space_val > 0.0001f);
393 
394  if (use_regular_space)
395  {
396  // do first pass to find out biggest element, total height of all elements
397  CScreenElementPtr p_child = mp_child_list;
398  while(p_child)
399  {
400  // virtual dimensions, 'k' in the direction that menu items are added (i.e. of 'z')
401  float scaled_j;
402 
403  if (m_is_vertical_menu)
404  scaled_j = p_child->GetBaseW() * p_child->GetScaleX();
405  else
406  scaled_j = p_child->GetBaseH() * p_child->GetScaleY();
407 
408  z += m_regular_space_val;
409 
410  if (scaled_j > biggest)
411  biggest = scaled_j;
412 
413  p_child = p_child->GetNextSibling();
414  }
415  }
416  else
417  {
418  // do first pass to find out biggest element, total height of all elements
419  CScreenElementPtr p_child = mp_child_list;
420  while(p_child)
421  {
422  // virtual dimensions, 'k' in the direction that menu items are added (i.e. of 'z')
423  float scaled_j, scaled_k;
424 
425  if (m_is_vertical_menu)
426  {
427  scaled_j = p_child->GetBaseW() * m_padding_scale;
428  scaled_k = p_child->GetBaseH() * m_padding_scale;
429  }
430  else
431  {
432  scaled_j = p_child->GetBaseH() * m_padding_scale;
433  scaled_k = p_child->GetBaseW() * m_padding_scale;
434  }
435 
436  z += scaled_k + m_spacing_between;
437 
438  if (scaled_j > biggest)
439  biggest = scaled_j;
440 
441  p_child = p_child->GetNextSibling();
442  }
443  z -= m_spacing_between;
444  }
445 
446  if ((m_object_flags & vFORCED_DIMS))
447  {
448  // "grow" the menu in case of overflow (disobeying rule of above flag)
449  if (m_is_vertical_menu && z > m_base_h)
450  m_base_h = z;
451  else if (!m_is_vertical_menu && z > m_base_w)
452  m_base_w = z;
453  }
454  else
455  {
456  // this element is in control of own dimensions
457  if (m_is_vertical_menu)
458  {
459  m_base_w = biggest;
460  m_base_h = z;
461  }
462  else
463  {
464  m_base_w = z;
465  m_base_h = biggest;
466  }
467  }
468  compute_ul_pos(m_target_local_props);
469  m_object_flags |= vNEEDS_LOCAL_POS_CALC;
470 
471 
472  // use vertical justification to figure out starting position
473  if (m_is_vertical_menu)
474  z = (m_internal_just_y + 1.0f) * (m_base_h - z) / 2.0f;
475  else
476  z = (m_internal_just_x + 1.0f) * (m_base_w - z) / 2.0f;
477 
478  if (use_regular_space)
479  {
480  z += m_regular_space_val / 2.0f;
481 
482  // now, set positions of everything
483  CScreenElementPtr p_child = mp_child_list;
484  while(p_child)
485  {
486  if (m_is_vertical_menu)
487  {
488  p_child->SetJust(m_internal_just_x, 0.0f);
489  p_child->SetPos((m_internal_just_x + 1.0f) * m_base_w / 2.0f, z, FORCE_INSTANT);
490  }
491  else
492  {
493  p_child->SetJust(0.0f, m_internal_just_y);
494  p_child->SetPos(z, (m_internal_just_y + 1.0f) * m_base_h / 2.0f, FORCE_INSTANT);
495  }
496

498  p_child = p_child->GetNextSibling();
499  }
500  }
501  else // if !use_regular_space

503  // now, set positions of everything

505  while(p_child)
