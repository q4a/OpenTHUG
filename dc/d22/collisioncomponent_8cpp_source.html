70 void CCollisionComponent::InitFromStructure( Script::CStruct* pParams )
71 {
72  Dbg_MsgAssert( pParams, ( "No node data?" ) );
73  Dbg_MsgAssert( !mp_collision, ( "Set up collision twice?" ) );
74 
75  uint32 col_type_checksum = Nx::vCOLL_TYPE_NONE;
76  pParams->GetChecksum( CRCD(0x2d7e583b,"CollisionMode"), &col_type_checksum );
77 
78  Nx::CollType col_type = Nx::CCollObj::sConvertTypeChecksum( col_type_checksum );
79  if ( col_type != Nx::vCOLL_TYPE_NONE )
80  {
81  uint32 class_checksum = 0;
82  pParams->GetChecksum("class", &class_checksum);
83  if ( class_checksum == CRCD(0xb7b3bd86,"LevelObject") )
84  {
85  uint32 name;
86  pParams->GetChecksum(CRCD(0xa1dc81f9,"name"), &name, Script::ASSERT);
87  Nx::CSector *p_sector = Nx::CEngine::sGetSector(name);
88  Dbg_MsgAssert( p_sector, ( "WARNING: sGetSector(0x%x) returned NULL (%s)\n", name, Script::FindChecksumName(name) ) );
89  Nx::CCollObjTriData *p_coll_tri_data = p_sector->GetCollSector()->GetGeometry();
90  InitCollision(col_type, p_coll_tri_data);
91  }
92  else
93  {
94  InitCollision(col_type);
95  }
96  }
97 }

104 void CCollisionComponent::InitCollision( Nx::CollType type, Nx::CCollObjTriData *p_coll_tri_data )
105 {
106  Dbg_Assert( !mp_collision );
107 
108  if ( p_coll_tri_data )
109  {
110  mp_collision = Nx::CCollObj::sCreateMovableCollision(type, p_coll_tri_data, 1, GetObject());
111  }
112  else
113  {
114  Obj::CModelComponent* pModelComponent = GetModelComponentFromObject( GetObject() );
115  Dbg_MsgAssert( pModelComponent, ( "Initing collision on something with no model component" ) );
116 
117  Nx::CModel* pModel = pModelComponent->GetModel();
118  Dbg_MsgAssert( pModel, ( "Initing collision on something with no model" ) );
119 
120  Ass::CAssMan * ass_man = Ass::CAssMan::Instance();
121  Nx::CMesh* pMesh = (Nx::CMesh*)ass_man->GetAsset(pModel->GetFileName(), false);
122  if (pMesh)
123  {
124  mp_collision = Nx::CCollObj::sCreateMovableCollision(type, pMesh->GetCollisionTriDataArray(), pMesh->GetCollisionTriDataArraySize(), GetObject());
125  }
126  else
127  {
128  mp_collision = Nx::CCollObj::sCreateMovableCollision(type, NULL, 0, GetObject());
129  }
130  }
131 }

164 void CCollisionComponent::Update()
165 {
166 // GJ: On THPS4, we did the following test first before
167 // updating the collision; I think this will be handled
168 // by the suspend component, so we shouldn't have to do
169 // this here...
170 // if ( (m_object_flags & vINVISIBLE) && !initialize )
171 // {
172 // return;
173 // }
174 
175  // get the collision object
176  Nx::CCollObj* p_coll = GetCollision();
177 
178  // Update collision
179  // Might be problem with things being LODed off, but still needing collision?
180  if ( p_coll )
181  {
182  // GJ TODO: Somehow remove the collision object's dependency on the model!
183 
184  Nx::CModel* p_model = NULL;
185  Nx::CHierarchyObject* p_hierarchy = NULL;
186  Obj::CModelComponent* p_model_component = GetModelComponentFromObject( GetObject() );
187  if ( p_model_component )
188  {
189  p_model = p_model_component->GetModel();
190  Dbg_MsgAssert( p_model, ( "No model?!?" ) );
191  p_hierarchy = p_model->GetHierarchy();
192  }
193 
194  if (p_hierarchy)
195  {
196  Mth::Vector translation(p_hierarchy->GetSetupMatrix()[W]);
197  translation[W] = 0.0f; // turn into vector
198  p_hierarchy->GetSetupMatrix().Transform(translation); // Re-orients the translation
199 
200  p_coll->SetWorldPosition( GetObject()->GetPos() + translation );
201  p_coll->SetOrientation( p_hierarchy->GetSetupMatrix() * GetObject()->GetDisplayMatrix() );
202  }
203  else
204  {
205  p_coll->SetWorldPosition( GetObject()->GetPos() );
206  p_coll->SetOrientation( GetObject()->GetDisplayMatrix() );
207  }
208  }
209 }
