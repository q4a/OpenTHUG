42 void CHorseGoal::CheckScore()
43 {
44  // get the right flag
45  Script::CArray* pHorseSpots;
46  mp_params->GetArray( "horse_spots", &pHorseSpots, Script::ASSERT );
47 
48  int size = pHorseSpots->GetSize();
49  int i;
50  int score;
51  uint32 flag;
52  for ( i = 0; i < size; i++ )
53  {
54  Script::CStruct* pSpot = pHorseSpots->GetStructure( i );
55  pSpot->GetChecksum( "flag", &flag, Script::ASSERT );
56  int flag_value;
57  mp_goalFlags->GetInteger( flag, &flag_value, Script::ASSERT );
58 
59  // find the first unset flag
60  if ( flag_value == 0 )
61  {
62  if ( !pSpot->GetInteger( "score", &score, Script::NO_ASSERT ) )
63  mp_params->GetInteger( "score", &score, Script::ASSERT );
64  break;
65  }
66  }
67 
68  // make sure they're not all set...
69  if ( i == size )
70  return;
71 
72  // check the score
73  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
74  Obj::CSkater* pSkater = skate_mod->GetLocalSkater();
75  Mdl::Score* pScore = pSkater->GetScoreObject();
76 
77  int last_score_landed = pScore->GetLastScoreLanded();
78  if ( last_score_landed > 0 )
79  {
80  uint32 goalId = GetGoalId();
81  CGoalManager* pGoalManager = GetGoalManager();
82  if ( last_score_landed > score )
83  {
84  pGoalManager->SetGoalFlag( goalId, flag, 1 );
85  pGoalManager->NextTourSpot( goalId );
86  }
87  else
88  pGoalManager->LoseGoal( goalId );
89  }
90  mp_params->AddInteger( "should_check_trick", 0 );
91 }
