201 bool CMovingObject::LookAtObject_Init( Script::CStruct* pParams, Script::CScript* pScript )
202 {
203  uint32 typeChecksum;
204  if ( pParams->GetChecksum( 0x7321a8d6, &typeChecksum ) ) // "type"
205  {
206  int type;
207  switch ( typeChecksum )
208  {
209  case 0x88c21962: // car
210  type = SKATE_TYPE_CAR;
211  break;
212  case 0x61a741e: // ped
213  type = SKATE_TYPE_PED;
214  break;
215  case 0x5b8ab877: // skater
216  type = SKATE_TYPE_SKATER;
217  break;
218  case 0x3b5737a6: // gameobj
219  type = SKATE_TYPE_GAME_OBJ;
220  break;
221  default:
222  Dbg_MsgAssert( 0,( "\n%s\nUnknown type %s", pScript->GetScriptInfo( ), Script::FindChecksumName( typeChecksum ) ));
223  return false;
224  break;
225  }
226  CCompositeObject* pClosestObject = GetClosestObjectOfType( type );
227  if ( !pClosestObject )
228  {
229  Dbg_Message( "\n%s\nWarning: Looking at object that doesn't exist.", pScript->GetScriptInfo( ) );
230  return false;
231  }
232  return GetMotionComponent()->LookAt_Init( pParams, pClosestObject->m_pos );
233  }
234 
235  // find a named object:
236  uint32 nameChecksum;
237  if ( pParams->GetChecksum( 0xa1dc81f9, &nameChecksum ) ) // "name"
238  {
239  CCompositeObject* pNext;
240  CCompositeObject* pObj;
241  Lst::Search< CObject > sh;
242  Dbg_MsgAssert(mp_manager,("NULL mp_Manager in MovingObject"));
243  pNext = (Obj::CCompositeObject*) sh.FirstItem( mp_manager->GetRefObjectList() );
244  while ( pNext )
245  {
246  Dbg_AssertType( pNext, CCompositeObject );
247  pObj = pNext;
248  pNext = (Obj::CCompositeObject*) sh.NextItem();
249  // find objects spawned from this node, return true if out of this radius...
250  if ( ( pObj != this ) && ( pObj->GetID() == nameChecksum ) )
251  {
252  return GetMotionComponent()->LookAt_Init( pParams, pObj->m_pos );
253  }
254  }
255  Dbg_Message( "\n%s\nWarning: Looking at object that doesn't exist.", pScript->GetScriptInfo( ) );
256  return false;
257  }
258  Dbg_Message( "\n%s\nWarning: No object specified for Obj_LookAtObject", pScript->GetScriptInfo( ) );
259  return false;
260 }

309 float CMovingObject::GetSiteBoxMaxRadiusSquared( SiteBox* pBox )
310 {
311  float maxBoxDist;
312  if ( pBox->maxDistInitialized )
313  {
314  return ( pBox->maxDistSquared );
315  }
316  maxBoxDist = fabsf( pBox->dist );
317  if ( fabsf( pBox->height ) > maxBoxDist )
318  {
319  maxBoxDist = fabsf( pBox->height );
320  }
321  if ( fabsf( pBox->width ) > maxBoxDist )
322  {
323  maxBoxDist = fabsf( pBox->width );
324  }
325 
326  float maxOffset;
327  Mth::Vector* pOffset;
328  pOffset = &pBox->bbox.centerOfGravity;
329  maxOffset = fabsf( pOffset->GetZ( ) );
330  if ( fabsf( pOffset->GetY( ) ) > maxOffset )
331  {
332  maxOffset = fabsf( pOffset->GetY( ) );
333  }
334  if ( fabsf( pOffset->GetX( ) ) > maxOffset )
335  {
336  maxOffset = fabsf( pOffset->GetX( ) );
337  }
338 
339  pBox->maxDistSquared = maxOffset + maxBoxDist;
340  pBox->maxDistSquared *= pBox->maxDistSquared;
341  pBox->maxDistInitialized = true;
342  return ( pBox->maxDistSquared );
343 }

350 void CMovingObject::InitializeSiteBox( SiteBox* pBox )
351 {
352  if ( pBox->initialized )
353  {
354  // already initialized!
355  return;
356  }
357 
358  pBox->matrix = m_matrix;
359  if ( pBox->angle )
360  {
361  pBox->matrix = pBox->matrix.RotateYLocal( pBox->angle );
362  }
363  pBox->matrix.Invert( );
364 
365  SBBox* pBBox = &pBox->bbox;
366  // start by setting min/max to half the dimentions:
367  pBBox->m_max.Set( pBox->width / 2.0f, pBox->height / 2.0f, pBox->dist );
368  pBBox->m_min = -pBBox->m_max;
369  pBBox->m_min[ Z ] = 0.0f;
370 
371  // using the vector in pBox->bbox.centerOfGravity to store offset
372  // sent by the player! MISNOMER:
373  pBBox->m_max += pBBox->centerOfGravity;
374  pBBox->m_min += pBBox->centerOfGravity;
375 
376  // gots ta move the site box up so that it comes from the object's
377  // center of gravity (as the object's position isn't usually the center
378  // of gravity):
379 /* if ( mp_model )
380  {
381  //Mth::Vector temp3;
382  //temp3 = m_pos + centerOfGravity;
383  //Gfx::AddDebugLine( m_pos, temp3 );
384  pBBox->m_max += m_bbox.centerOfGravity;
385  pBBox->m_min += m_bbox.centerOfGravity;
386  }*/
387 
388 #ifndef __PLAT_NGC__
389 #ifdef __NOPT_ASSERT__
390  if ( pBox->debug )
391  {
392  pBBox->centerOfGravity.Set( ); // zero that out for the render...
393  Mth::Vector drawOffset;
394 // drawOffset.Set( 0, m_ground_offset, 0 );
395  drawOffset.Set( 0, 0, 0 );
396  if ( pBox->angle )
397  {
398  Mth::Vector angles;
399  angles.Set( 0, pBox->angle, 0 );
400  DrawBoundingBox( pBBox, &drawOffset, 10, &angles );
401  }
402  else
403  {
404  DrawBoundingBox( pBBox, &drawOffset, 10 );
405  }
406  }
407 #endif
408 #endif
409 
410  // Phew... now we can quickly check objects by rotating their position
411  // into our world, and seeing if they're in the motherfuckin' box:
412  pBox->initialized = true;
413 }

420 bool CMovingObject::ObjInSiteBox( CMovingObject* pObj, SiteBox* pBox )
421 {
422  // K: There was a bug whereby the initial distance check below was not taking
423  // into account the offset of the box.
424  // The offset is stored in centerOfGravity which may get zeroed in InitializeSiteBox,
425  // so store it here.
426  Mth::Vector off=pBox->bbox.centerOfGravity;
427 
428  // initialize the site box and the BBox contained therein:
429  InitializeSiteBox( pBox );
430 
431  // sorta trivial radius check and return...
432  float distSquared;
433  distSquared = Mth::DistanceSqr( pObj->m_pos+off, m_pos );
434  float boxRadiusSquared = GetSiteBoxMaxRadiusSquared( pBox );
435  if ( distSquared > boxRadiusSquared )
436  {
437  return false;
438  }
439 
440  Mth::Vector objPos;
441  objPos = pObj->m_pos;
442 
443  // center of the object...
444  objPos[ Y ] += ( pObj->m_bbox.m_max[ Y ] - pObj->m_bbox.m_min[ Y ] ) / 2.0f;
445 
446 #ifndef __PLAT_NGC__
447 #ifdef __NOPT_ASSERT__
448  Mth::Vector preTransform = objPos;
449 #endif
450 #endif
451 
452  objPos -= m_pos;
453  // now we have the head and toes at the right distance from us and from each other...
454  // rotate them into our world:
455  objPos = pBox->matrix.Transform( objPos );
456 
457 #ifndef __PLAT_NGC__
458 #ifdef __NOPT_ASSERT__
459  if ( pBox->debug )
460  {
461  if ( Gfx::PointInsideBox( objPos, pBox->bbox.m_max, pBox->bbox.m_min ) )
462  {
463  Gfx::AddDebugLine( m_pos, preTransform, MAKE_RGB( 150, 50, 50 ), MAKE_RGB( 150, 50, 50 ), 10 );
464  return true;
465  }
466  else
467  {
468  Gfx::AddDebugLine( m_pos, preTransform, MAKE_RGB( 50, 50, 100 ), MAKE_RGB( 50, 50, 100 ), 10 );
469  return false;
470  }
471  }
472 #endif
473 #endif
474  return ( Gfx::PointInsideBox( objPos, pBox->bbox.m_max, pBox->bbox.m_min ) );
475 }

632 bool CMovingObject::ObjectWithinRange( Script::CStruct* pParams, Script::CScript* pScript, SiteBox* pBox )
633 {
634  float radiusSqr = 0.0f;
635 
636  if ( !pBox )
637  {
638  if( !( pParams->GetFloat( 0xc48391a5, &radiusSqr ) ) )//"radius"
639  {
640  Dbg_MsgAssert( 0,( "\n%s\nNo radius specified in Obj_*InRadius.", pScript->GetScriptInfo( ) ));
641  return false;
642  }
643  radiusSqr = FEET_TO_INCHES( radiusSqr );
644  radiusSqr *= radiusSqr;
645  }
646 
647  Script::CArray* pArray=NULL;
648  uint32 typeChecksum;
649  if ( pParams->GetChecksum( 0x7321a8d6, &typeChecksum ) ) // "type"
650  {
651  return ( ObjTypeInRange( pParams, pScript, radiusSqr, pBox, typeChecksum ) );
652  }
653  else if ( pParams->GetArray( 0x7321a8d6, &pArray, Script::ASSERT ) ) // "type"
654  {
655  Dbg_MsgAssert( pArray->GetType( ) == ESYMBOLTYPE_NAME,( "\n%s\nObjectWithinRange: Array must be of names",pScript->GetScriptInfo()));
656  uint32 i;
657  for ( i = 0; i < pArray->GetSize( ); ++i )
658  {
659  if ( ObjTypeInRange( pParams, pScript, radiusSqr, pBox, pArray->GetChecksum( i ) ) )
660  {
661  return true;
662  }
663  }
664  return false;
665  }
666 
667  // find a named object:
668  uint32 nameChecksum;
669  if ( pParams->GetChecksum( 0xa1dc81f9, &nameChecksum ) ) // "name"
670  {
671  int nodeNum = SkateScript::FindNamedNode( nameChecksum );
672  return ( ObjectFromNodeWithinRange( nodeNum, radiusSqr, pBox ) );
673  }
674  const char* pPrefix;
675  if ( pParams->GetText( 0x6c4e7971, &pPrefix ) ) // checksum 'prefix'
676  {
677  // Create with a prefix specified:
678  uint16 numNodes = 0;
679  const uint16* pMatchingNodes = SkateScript::GetPrefixedNodes( pPrefix, &numNodes );
680  int i;
681  for ( i = 0; i < numNodes; i++ )
682  {
683  if ( ObjectFromNodeWithinRange( pMatchingNodes[ i ], radiusSqr, pBox ) )
684  {
685  //printf("Oof old\n");
686  return true;
687  }
688  }
689  return false;
