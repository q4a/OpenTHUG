89 void CSkaterScoreComponent::Update()
90 {
91  // NOTE: in fact, non-local skaters may not need a score component at all
92  if (!GetSkater()->IsLocalClient())
93  {
94  Suspend(true);
95  return;
96  }
97 
98  mp_score->Update();
99 
100  if (GetSkater()->m_always_special || Mdl::Skate::Instance()->GetCareer()->GetCheat(CRCD(0xeefecf1b, "CHEAT_ALWAYS_SPECIAL")))
101  {
102  // Here, set the flag. It may seem redundant, but the above line is very likely
103  // to be hacked by gameshark. They probably won't notice this one, which will
104  // set the flags as if they had actually enabled the cheat -- which enables us
105  // to detect that it has been turned on more easily.
106  Mdl::Skate::Instance()->GetCareer()->SetGlobalFlag( Script::GetInteger(CRCD(0xeefecf1b, "CHEAT_ALWAYS_SPECIAL")));
107  mp_score->ForceSpecial();
108  g_CheatsEnabled = true;
109  }
110 }

117 CBaseComponent::EMemberFunctionResult CSkaterScoreComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
118 {
119  switch ( Checksum )
120  {
121  // @script | LastScoreLandedGreaterThan | Multiplayer-safe version
122  // of SkaterLastScoreLandedGreaterThan
123  // Example: if LastScoreLandedGreaterThan 2000 --do cool stuff-- endif
124  // @uparm 1 | Score (int)
125  case CRCC(0xe0112aab, "LastScoreLandedGreaterThan"):
126  {
127  int score;
128  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
129  return mp_score->GetLastScoreLanded() > score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
130  }
131 
132  // @script | LastScoreLandedLessThan | Multiplayer-safe version
133  // of SkaterLastScoreLandedLessThan
134  // Example: if LastScoreLandedLessThan 2000 --do cool stuff-- endif
135  // @uparm 1 | Score (int)
136  case CRCC(0x3c445a82, "LastScoreLandedLessThan"):
137  {
138  int score;
139  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
140  return mp_score->GetLastScoreLanded() < score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
141  }
142 
143  // @script | TotalScoreGreaterThan | Multiplayer-safe version
144  // of SkaterTotalScoreGreaterThan
145  // Example: if TotalScoreGreaterThan 2000 --do cool stuff-- endif
146  // @uparm 1 | Score (int)
147  case CRCC(0x8bf43831, "TotalScoreGreaterThan"):
148  {
149  int score;
150  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
151  return mp_score->GetTotalScore() > score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
152  }
153 
154  // @script | TotalScoreLessThan | Multiplayer-safe version
155  // of SkaterTotalScoreLessThan
156  // Example: if TotalScoreLessThan 2000 --do cool stuff-- endif
157  // @uparm 1 | Score (int)
158  case CRCC(0xdf14bd98, "TotalScoreLessThan"):
159  {
160  int score;
161  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
162  return mp_score->GetTotalScore() < score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
163  }
164 
165  // @script | CurrentScorePotGreaterThan | Multiplayer-safe version
166  // of SkaterCurrentScorePotGreaterThan
167  // Example: if CurrentScorePotGreaterThan 2000 --do cool stuff-- endif
168  // @uparm 1 | Score (int)
169  case CRCC(0xd19bdb50, "CurrentScorePotGreaterThan"):
170  {
171  int score;
172  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
173  return mp_score->GetScorePotValue() > score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
174  }
175 
176  // @script | GetTotalScore | returns the total score of the skater
177  // in TotalScore
178  case CRCC( 0x8f2fe3d3, "GetTotalScore" ):
179  {
180  pScript->GetParams()->AddInteger( CRCD( 0xee5b2b48, "TotalScore" ), mp_score->GetTotalScore() );
181  return CBaseComponent::MF_TRUE;
182  }
183 
184  // @script | CurrentScorePotLessThan | Multiplayer-safe version
185  // of SkaterCurrentScorePotLessThan
186  // Example: if CurrentScorePotLessThan 2000 --do cool stuff-- endif
187  // @uparm 1 | Score (int)
188  case CRCC(0x56645fc6, "CurrentScorePotLessThan"):
189  {
190  int score;
191  pParams->GetInteger(NO_NAME, &score, Script::ASSERT);
192  return mp_score->GetScorePotValue() < score ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
193  }
194 
195  // @script | GetNumberOfNonGapTricks | returns number of non-gap tricks currently in the combo
196  case CRCC(0xc068ea59, "GetNumberOfNonGapTricks"):
197  pScript->GetParams()->AddInteger(CRCD(0xffe7e02c, "NumberOfNonGapTricks"), mp_score->GetNumberOfNonGapTricks());
198  break;
199 
200  // @script | GotSpecial | true if special is active
201  case CRCC(0x5589b902, "GotSpecial"):
202  return mp_score->GetSpecialState() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
203 
204  // @script | TweakTrick |
205  // @uparm 0 | tweak
206  case CRCC(0xbe3e5463, "TweakTrick"):
207  {
208  int Tweak = 0;
209  pParams->GetInteger(NO_NAME, &Tweak);
210 
211  mp_score->TweakTrick(Tweak);
212  break;
213  }
214 
215  // @script | IsLatestTrick |
216  case CRCC(0x2877b61a, "IsLatestTrick"):
217  {
218  uint32 key_combo;
219  if (pParams->GetChecksum(CRCD(0x95e16467, "KeyCombo"), &key_combo))
220  {
221  int spin = 0;
222  if (pParams->GetInteger(CRCD(0xedf5db70, "Spin"), &spin))
223  {
224  Dbg_MsgAssert(spin % 180 == 0, ("IsLatestTrick called with a spin value of %i which is not a multiple of 180", spin));
225  spin /= 180;
226  }
227 
228  int num_taps = 1;
229  pParams->GetInteger(CRCD(0xc82cf71b, "NumTaps"), &num_taps);
230 
231  return mp_score->IsLatestTrick(key_combo, spin, false, num_taps) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
232  }
233  else
234  {
235  const char* trick_text;
236  if (pParams->GetString(CRCD(0x3eafa520, "TrickText"), &trick_text))
237  {
238  int spin = 0;
239  if (pParams->GetInteger(CRCD(0xedf5db70, "Spin"), &spin))
240  {
241  Dbg_MsgAssert(spin % 180 == 0, ("IsLatestTrick called with a spin value of %i which is not a multiple of 180", spin));
242  spin /= 180;
243  }
244 
245  int num_taps = 1;
246  pParams->GetInteger(CRCD(0xc82cf71b, "NumTaps"), &num_taps);
247 
248  return mp_score->IsLatestTrickByName(Script::GenerateCRC(trick_text), spin, false, num_taps) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
249  }
250  else
251  {
252  Dbg_MsgAssert(false, ("IsLatestTrick must have either a KeyCombo or a TrickText parameter"));
253  }
254  }
255  }
256 
257  default:
258  return CBaseComponent::MF_NOT_EXECUTED;
259  }
260  return CBaseComponent::MF_TRUE;
261 }
