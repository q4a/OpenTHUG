97 void* Pool::allocate( size_t size, bool assert_on_fail )
98 {
99 
100 
101  Dbg_MsgAssert ( size <= m_size,( "Failed to allocate: Size (%x) > Pool Element (%x)", size, m_size ));
102 
103  if ( mp_free_list )
104  {
105  Dbg_AssertType( mp_free_list, BlockHeader );
106 
107  BlockHeader* p_header = mp_free_list;
108 
109  mp_free_list = p_header->mpNext;
110  p_header->mpAlloc = this;
111 
112  MemDbg_AllocateBlock ( p_header );
113 
114  return (void*)((uint)p_header + BlockHeader::sSize);
115  }
116 
117  if ( assert_on_fail )
118  {
119  Dbg_MsgAssert ( false,( "Failed to allocate %d bytes, no free slots", size ));
120  }
121 
122  return NULL;
123 }

146 Pool::Pool( Region* region, size_t size, uint count, Direction dir )
147 : Allocator( region, dir, "Pool" ), m_count( count), m_size( size )
148 {
149 
150 
151  if ( !nAlignedBy( m_size, 4 ))
152  {
153 #ifdef __PLAT_NGC__
154  Dbg_Warning ( "size %x must be 32 byte aligned ( setting to %x )", m_size, nAlignUpBy( (void*)m_size, 5 ));
155  m_size = (size_t)nAlignUpBy( m_size, 5 );
156 #else
157  Dbg_Warning ( "size %x must be 16 byte aligned ( setting to %x )", m_size, nAlignUpBy( (void*)m_size, 4 ));
158  m_size = (size_t)nAlignUpBy( m_size, 4 );
159 #endif
160  }
161 
162  BlockHeader* p_freeblock = (BlockHeader*)mp_region->Allocate( this, ( m_size + BlockHeader::sSize ) * m_count );
163 
164  Dbg_MsgAssert ( p_freeblock,( "Failed to allocate Memory Pool from Region" ));
165 
166  mp_free_list = p_freeblock;
167 
168  for ( uint i = 0; i < ( m_count - 1 ); i++ )
169  {
170  new ((void*)p_freeblock) BlockHeader( this, m_size );
171 
172  BlockHeader* p_nextblock = (BlockHeader*)( (uint)p_freeblock + BlockHeader::sSize + m_size );
173  p_freeblock->mpNext = p_nextblock;
174  p_freeblock = p_nextblock;
175  }
176 
177  new ((void*)p_freeblock) BlockHeader( this, m_size );
178  p_freeblock->mpNext = NULL;
179 }
