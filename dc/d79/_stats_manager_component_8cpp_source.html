771 // Given the "Checksum" of a script command, then possibly handle it
772 // if it's a command that this component will handle
773 CBaseComponent::EMemberFunctionResult CStatsManagerComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
774 {
775  switch ( Checksum )
776  {
777 /*
778  // @script | DoSomething | does some functionality
779  case 0xbb4ad101: // DoSomething
780  DoSomething();
781  break;
782 
783  // @script | ValueIsTrue | returns a boolean value
784  case 0x769260f7: // ValueIsTrue
785  {
786  return ValueIsTrue() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
787  }
788  break;
789 */
790 
791  // @script | StatsManager_ReInit | reinitializes stats manger. used when starting new career/story
792  case 0x82ce30ce: // StatsManager_ReInit
793  {
794  RefreshFromStructure( pParams );
795  }
796  break;
797 
798  // @script | StatsManager_Reset | resets stats manger. used when reinitializing skater
799  case 0x57879c96: //StatsManager_Reset
800  {
801  Bail();
802  }
803  break;
804 
805  // @script | StatsManager_ActivateGoals | activates stats goals
806  case 0xee59073c: // StatsManager_ActivateGoals
807  {
808  stat_goals_on = true;
809  Script::RunScript( Script::GenerateCRC( "console_unhide" ) );
810  }
811  break;
812 
813  // @script | StatsManager_DeactivateGoals | deactivates stats goals
814  case 0xf1745991: // StatsManager_DeactivateGoals
815  {
816  stat_goals_on = false;
817  Script::RunScript( Script::GenerateCRC( "console_hide" ) );
818  }
819  break;
820 
821  // @script | StatsManager_GetGoalStatus | returns status of stats goal with provided index
822  case 0x78e7d855: // StatsManager_GetGoalStatus
823  {
824  int index;
825  int status;
826  pParams->GetInteger( NONAME, &index, Script::ASSERT );
827  status = stat_goals[index].complete;
828  pScript->GetParams()->AddInteger( CRCD(0x84ff9ae3,"status"), status );
829  }
830  break;
831 
832  // @script | StatsManager_UnlockAmGoals | unlocks amateur stats goals
833  case 0xb341b5a5: // StatsManager_UnlockAmGoals
834  {
835  for (int i=0; i < NUM_STATS_GOALS; i++)
836  {
837  if ( stat_goals[i].complete == 2 )
838  {
839  stat_goals[i].complete = 0;
840  }
841  }
842  }
843  break;
844 
845  // @script | StatsManager_UnlockProGoals | unlocks pro stats goals
846  case 0xb7fac894: // StatsManager_UnlockProGoals
847  {
848  for (int i=0; i < NUM_STATS_GOALS; i++)
849  {
850  if ( stat_goals[i].complete == 3 )
851  {
852  stat_goals[i].complete = 0;
853  }
854  }
855  }
856  break;
857 
858  default:
859  return CBaseComponent::MF_NOT_EXECUTED;
860  }
861 
862  // the "default" case of the switch statement handles
863  // unrecognized functions; if we make it down here,
864  // that means that the component already handled it
865  // somehow
866  return CBaseComponent::MF_TRUE;
867 }
