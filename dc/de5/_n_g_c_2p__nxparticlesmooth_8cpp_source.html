163 void CNgcParticleSmooth::plat_render( void )
164 {
165  NxNgc::sMaterialHeader mat;
166  NxNgc::sMaterialPassHeader pass;
167 
168  // Header.
169  mat.m_checksum = 0xa9db601e; // particle
170  mat.m_passes = 1;
171  mat.m_alpha_cutoff = 1;
172  mat.m_flags = (1<<1); // 2 sided.
173 // mat.m_shininess = 0.0f;
174 
175  // Pass 0.
176  pass.m_texture.p_data = mp_engine_texture;
177  pass.m_flags = ( mp_engine_texture ? (1<<0) : 0 ) | (1<<5) | (1<<6); // textured, clamped.
178  pass.m_filter = 0;
179  pass.m_blend_mode = (unsigned char)m_blend;
180  pass.m_alpha_fix = (unsigned char)m_fix;
181  pass.m_k = 0;
182  pass.m_color.r = 128;
183  pass.m_color.g = 128;
184  pass.m_color.b = 128;
185  pass.m_color.a = 255;
186 
187  NxNgc::multi_mesh( &mat, &pass, true, true );
188 
189  GX::SetTexCoordGen( GX_TEXCOORD0, GX_TG_MTX2x4, GX_TG_TEX0, GX_FALSE, GX_PTIDENTITY );
190  GX::SetCurrMtxPosTex03( GX_PNMTX0, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY );
191 
192  // Draw the particles.
193 
194  // Used to figure the right and up vectors for creating screen-aligned particle quads.
195  NsMatrix* p_matrix = &NxNgc::EngineGlobals.camera;
196 
197  // Concatenate p_matrix with the emmission angle to create the direction.
198 
199  NsVector up( 0.0f, 1.0f, 0.0f );
200 
201  // Get the 'right' vector as the cross product of camera 'at and world 'up'.
202  NsVector screen_right;
203  NsVector screen_up;
204  screen_right.cross( *p_matrix->getAt(), up );
205  screen_up.cross( screen_right, *p_matrix->getAt());
206 
207  screen_right.normalize();
208  screen_up.normalize();
209 
210  int lp;
211  CParticleEntry *p_particle;
212  float *p_v;
213 
214  if ( mp_engine_texture )
215  {
216  GX::SetVtxDesc( 3, GX_VA_POS, GX_DIRECT, GX_VA_CLR0, GX_DIRECT, GX_VA_TEX0, GX_DIRECT );
217  }
218  else
219  {
220  GX::SetVtxDesc( 2, GX_VA_POS, GX_DIRECT, GX_VA_CLR0, GX_DIRECT );
221  }
222 
223  for ( lp = 0, p_particle = mp_particle_array, p_v = mp_vertices; lp < m_num_particles; lp++, p_particle++, p_v += 3 )
224  {
225  float terp = p_particle->m_time / p_particle->m_life;
226 
227  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
228  float h = p_particle->m_sh + ( ( p_particle->m_eh - p_particle->m_sh ) * terp );
229 
230  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
231  Mth::Vector pos( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
232  Mth::Vector ss_right, ss_up; //, ss_pos;
233  Mth::Vector tmp[4];
234 
235  VECScale((Vec*)&screen_right, (Vec*)&ss_right, w );
236  VECScale((Vec*)&screen_up, (Vec*)&ss_up, h );
237 
238  tmp[0] = pos - ss_right + ss_up;
239  tmp[1] = pos + ss_right + ss_up;
240  tmp[2] = pos + ss_right - ss_up;
241  tmp[3] = pos - ss_right - ss_up;
242 
243  GXColor color[2];
244  GXColor *p_col0;
245  GXColor *p_col1;
246 
247  if ( m_mid_time >= 0.0f )
248  {
249  if ( terp < m_mid_time )
250  {
251  p_col0 = m_start_color;
252  p_col1 = m_mid_color;
253  // Adjust interpolation for this half of the color blend.
254  terp = terp / m_mid_time;
255  }
256  else
257  {
258  p_col0 = m_mid_color;
259  p_col1 = m_end_color;
260  // Adjust interpolation for this half of the color blend.
261  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
262  }
263  }
264  else
265  {
266  // No mid color specified.
267  p_col0 = m_start_color;
268  p_col1 = m_end_color;
269  }
270 
271  for ( int c = 0; c < 2; c++ )
272  {
273  GXColor start = *p_col0++;
274  GXColor end = *p_col1++;
275 
276  color[c].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
277  color[c].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
278  color[c].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
279  color[c].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
280  }
281  GX::Begin( GX_TRIANGLEFAN, GX_VTXFMT0, 6 );
282 
283  GX::Position3f32( pos[X], pos[Y], pos[Z] );
284  GX::Color1u32( *((uint32*)&color[0]) );
285  if ( mp_engine_texture ) GX::TexCoord2f32( 0.5f, 0.5f );
286 
287  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
288  GX::Color1u32( *((uint32*)&color[1]) );
289  if ( mp_engine_texture ) GX::TexCoord2f32( 0.0f, 0.0f );
290 
291  GX::Position3f32( tmp[1][X], tmp[1][Y], tmp[1][Z] );
292  GX::Color1u32( *((uint32*)&color[1]) );
293  if ( mp_engine_texture ) GX::TexCoord2f32( 1.0f, 0.0f );
294 
295  GX::Position3f32( tmp[2][X], tmp[2][Y], tmp[2][Z] );
296  GX::Color1u32( *((uint32*)&color[1]) );
297  if ( mp_engine_texture ) GX::TexCoord2f32( 1.0f, 1.0f );
298 
299  GX::Position3f32( tmp[3][X], tmp[3][Y], tmp[3][Z] );
300  GX::Color1u32( *((uint32*)&color[1]) );
301  if ( mp_engine_texture ) GX::TexCoord2f32( 0.0f, 1.0f );
302 
303  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
304  GX::Color1u32( *((uint32*)&color[1]) );
305  if ( mp_engine_texture ) GX::TexCoord2f32( 0.0f, 0.0f );
306 
307  GX::End();
308  }
309 }
