171 void SlerpInterpolator::getMatrix( Matrix * result, float delta )
172 {
173  /* Cap and floor the delta */
174  /* If we are at one end the solution is easy */
175  if (delta <= 0.0f)
176  {
177  // delta = 0.0f;
178  *result = m_start;
179  return;
180  }
181  else if (delta >= 1.0f)
182  {
183  // delta = 1.0f;
184  *result = m_end;
185  return;
186  }
187 
188 #if 0
189  // GJ: always lerp, used while slerp was being debugged
190 // m_useLerp = true;
191 #endif
192 
193  /* Do the lerp if we are, else... */
194  if ( m_useLerp )
195  {
196  /* Get the lerp matrix */
197  Matrix lerp;
198  Vector lpos;
199  Vector spos;
200  Vector epos;
201  Vector rpos;
202 
203  lerp.Ident();
204 
205  spos = m_start[Mth::POS];
206  epos = m_end[Mth::POS];
207 
208  lerp[Mth::RIGHT] = m_end[Mth::RIGHT] - m_start[Mth::RIGHT];
209  lerp[Mth::UP] = m_end[Mth::UP] - m_start[Mth::UP];
210  lerp[Mth::AT] = m_end[Mth::AT] - m_start[Mth::AT];
211  lpos = epos - spos;
212 
213  /* Do lerp */
214  lerp[Mth::RIGHT].Scale( delta );
215  lerp[Mth::UP].Scale( delta );
216  lerp[Mth::AT].Scale( delta );
217  lpos.Scale( delta );
218 
219  (*result)[Mth::RIGHT] = m_start[Mth::RIGHT] + lerp[Mth::RIGHT];
220  (*result)[Mth::UP] = m_start[Mth::UP] + lerp[Mth::UP];
221  (*result)[Mth::AT] = m_start[Mth::AT] + lerp[Mth::AT];
222  rpos = spos + lpos;
223 
224  (*result)[Mth::RIGHT].Normalize();
225  (*result)[Mth::UP].Normalize();
226  (*result)[Mth::AT].Normalize();
227 
228  (*result)[Mth::POS] = rpos;
229  }
230  else
231  {
232  Vector rpos;
233  Vector spos;
234  Vector epos;
235  Vector lpos;
236 
237  spos = m_start[Mth::POS];
238  epos = m_end[Mth::POS];
239 
240  /* Remove the translation for now */
241  *result = m_start;
242  (*result)[Mth::POS] = Mth::Vector( 0.0f, 0.0f, 0.0f );
243 
244  /* Rotate the new matrix */
245 // m_axis[W] = 0.0f;
246  result->Rotate( m_axis, m_radians * delta );
247 
248  /* Do linear interpolation on position */
249  lpos = epos - spos;
250  lpos.Scale( delta );
251  rpos = spos + lpos;
252 
253  (*result)[Mth::POS] = rpos;
254  }
255 }
