101 static int cmp( const void *p1, const void *p2 )
102 {
103  // Zero the score for any occlusion poly that is no longer available. This will force it out of the stack.
104  if(((sOccluder*)p1)->p_poly->available == false )
105  ((sOccluder*)p1)->score = 0;
106 
107  if(((sOccluder*)p2)->p_poly->available == false )
108  ((sOccluder*)p2)->score = 0;
109 
110  return((sOccluder*)p1)->score < ((sOccluder*)p2)->score ? 1 : ((sOccluder*)p1)->score > ((sOccluder*)p2)->score ? -1 : 0;
111 }

130 void sOccluder::tidy_stack( void )
131 {
132  if( NumOccluders > 0 )
133  {
134  // Sort in descending score order.
135  sort_stack();
136 
137 
138  int total = 0;
139  int demoted = 0;
140  // Count backwards so we know we get all the bad occluders.
141  for( int i = NumOccluders - 1; i >= 0; --i )
142  {
143  // If we have hit an occluder with zero meshes culled, cut off the stack at this point.
144  // also, if the stack is full, then demote the lowest scoring 4
145  if( Occluders[i].score == 0 || i > (int)(MAX_OCCLUDERS-MIN_NEW_OCCLUSION_POLYS_TO_CHECK_PER_FRAME))
146  {
147  // No longer using this poly.
148  Occluders[i].p_poly->in_use = false;
149 
150  // One less occluder to worry about.
151  --NumOccluders;
152  demoted++;
153  }
154  else
155  {
156  total += Occluders[i].score;
157  // Reset the good occluders.
158  Occluders[i].score = 0;
159  }
160  }
161 // printf ("\nOccluded %d objects with %d Occluders (demoted %d)\n",total,NumOccluders,demoted);
162  }

179 void AddOcclusionPoly( Mth::Vector &v0, Mth::Vector &v1, Mth::Vector &v2, Mth::Vector &v3, uint32 checksum )
180 {
181 // printf (" Adding occlusion polygons, currently %d\n",NumOcclusionPolys);
182  Dbg_MsgAssert( NumOcclusionPolys < MAX_OCCLUSION_POLYS,("Too many (%d) occlusion polygons",NumOcclusionPolys) );
183 
184  OcclusionPolys[NumOcclusionPolys].in_use = false;
185  OcclusionPolys[NumOcclusionPolys].available = true;
186  OcclusionPolys[NumOcclusionPolys].checksum = checksum;
187  OcclusionPolys[NumOcclusionPolys].verts[0] = v0;
188  OcclusionPolys[NumOcclusionPolys].verts[1] = v1;
189  OcclusionPolys[NumOcclusionPolys].verts[2] = v2;
190  OcclusionPolys[NumOcclusionPolys].verts[3] = v3;
191  OcclusionPolys[NumOcclusionPolys].normal = Mth::CrossProduct( v1 - v0, v3 - v0 );
192  OcclusionPolys[NumOcclusionPolys].normal.Normalize();
193 
194  ++NumOcclusionPolys;
195 
196  sCopiedOccludersOffline = false; // current scores now invalid
197 }

237 /******************************************************************/
238 /* */
239 /* */
240 /******************************************************************/
241 void CheckForOptimalOccluders( Mth::Vector &cam_pos, Mth::Vector &view_direction )
242 {
243  if( NumOcclusionPolys > 0 )
244  {
245  uint32 added = 0;
246  uint32 checked = 0;
247 
248  while( added < MAX_NEW_OCCLUSION_POLYS_TO_CHECK_PER_FRAME )
249  {
250  // Given the current position of the camera, check through the unused occlusion polys to see if one scores higher
251  // than the lowest scoring occlusion poly in use.
252  sOcclusionPoly *poly_to_check = &OcclusionPolys[NextOcclusionPolyToCheck++];
253  if(( !poly_to_check->in_use ) && ( poly_to_check->available ))
254  {
255  sOccluder::add_to_stack( poly_to_check );
256  ++added;
257  }
258  ++checked;
259 
260  // Ensure we are always checking within bounds.
261  if( NextOcclusionPolyToCheck >= NumOcclusionPolys )
262  {
263  NextOcclusionPolyToCheck = 0;
264  }
265 
266  // Quit out if we have less available occluders than spaces to fill.
267  if( checked >= NumOcclusionPolys )
268  {
269  break;
270  }
271  }
272  }
273 }

283 void BuildOccluders( Mth::Vector *p_cam_pos )
284 {
285 #ifdef OCCLUDER_USES_VU0_MACROMODE
286  if (sCopiedOccludersOffline)
287  {
288  sOccluder * p_occluder = &sOccluder::Occluders[0];
289  uint128 * p_vu0_mem = (uint128*)0x11004060; // Start of first score
290 
291  for( uint32 o = sOccluder::NumOccluders; o > 0 ; --o )
292  {
293  p_occluder->score = *((uint32*) (p_vu0_mem));
294  p_occluder++;
295  p_vu0_mem += 6;
296  }
297 
298  sCopiedOccludersOffline = false; // we are now up-to-date
299  }
300 #endif
301  for( uint32 i = 0; i < NumOcclusionPolys; ++i )
302  {
303 // Gfx::AddDebugLine( OcclusionPolys[i].verts[0], OcclusionPolys[i].verts[1], 0xFFFFFFFFUL, 0, 1 );
304 // Gfx::AddDebugLine( OcclusionPolys[i].verts[1], OcclusionPolys[i].verts[2], 0xFFFFFFFFUL, 0, 1 );
305 // Gfx::AddDebugLine( OcclusionPolys[i].verts[2], OcclusionPolys[i].verts[3], 0xFFFFFFFFUL, 0, 1 );
306 // Gfx::AddDebugLine( OcclusionPolys[i].verts[3], OcclusionPolys[i].verts[0], 0xFFFFFFFFUL, 0, 1 );
307  }
308 
309  // Tidy up from last frame.
310  sOccluder::tidy_stack();
311 
312  // Cyclically add more occluders for checking.
313  CheckForOptimalOccluders( *p_cam_pos, *p_cam_pos );
314 
315  // Build all 5 planes for each occluder.
316  Mth::Vector u0, u1, p;
317 
318  // The order in which the verts are used to build tha planes depends upon where the camera is in relation to the occlusion
319  // poly. We use the default order when the viewpoint is on the side of the poly on which the default poly normal faces.
320  for( uint32 i = 0; i < sOccluder::NumOccluders; ++i )
321  {
322  sOcclusionPoly *p_poly = sOccluder::Occluders[i].p_poly;
323  if( Mth::DotProduct( *p_cam_pos - p_poly->verts[0], p_poly->normal ) >= 0.0f )
324  {
325  // Start with left (0->1->2)...
326  u0 = p_poly->verts[0] - *p_cam_pos;
327  u1 = p_poly->verts[1] - *p_cam_pos;
328  p = Mth::CrossProduct( u0, u1 );
329  p.Normalize();
330  p[W] = -Mth::DotProduct( p, *p_cam_pos );
331  sOccluder::Occluders[i].planes[0] = p;
332 
333  // ...then right (0->3->4)...
334  u0 = p_poly->verts[2] - *p_cam_pos;
335  u1 = p_poly->verts[3] - *p_cam_pos;
336  p = Mth::CrossProduct( u0, u1 );
337  p.Normalize();
338  p[W] = -Mth::DotProduct( p, *p_cam_pos );
339  sOccluder::Occluders[i].planes[1] = p;
340 
341  // ...then top (0->2->3)...
342  u0 = p_poly->verts[1] - *p_cam_pos;
343  u1 = p_poly->verts[2] - *p_cam_pos;
344  p = Mth::CrossProduct( u0, u1 );
345  p.Normalize();
346  p[W] = -Mth::DotProduct( p, *p_cam_pos );
347  sOccluder::Occluders[i].planes[2] = p;
348 
349  // ...then bottom (0->4->1)...
350  u0 = p_poly->verts[3] - *p_cam_pos;
351  u1 = p_poly->verts[0] - *p_cam_pos;
352  p = Mth::CrossProduct( u0, u1 );
353  p.Normalize();
354  p[W] = -Mth::DotProduct( p, *p_cam_pos );
355  sOccluder::Occluders[i].planes[3] = p;
356 
357  // ...then front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
358  // are considered occluded. (1->3->4)...
359  u0 = p_poly->verts[2] - p_poly->verts[0];
360  u1 = p_poly->verts[3] - p_poly->verts[0];
361  p = Mth::CrossProduct( u0, u1 );
362  p.Normalize();
363  p[W] = -Mth::DotProduct( p, p_poly->verts[0] );
364  sOccluder::Occluders[i].planes[4] = p;
365  }
366  else
367  {
368  // Start with left (0->2->1)...
369  u0 = p_poly->verts[1] - *p_cam_pos;
370  u1 = p_poly->verts[0] - *p_cam_pos;
371  p = Mth::CrossProduct( u0, u1 );
372  p.Normalize();
373  p[W] = -Mth::DotProduct( p, *p_cam_pos );
374  sOccluder::Occluders[i].planes[0] = p;
375 
376  // ...then right (0->4->3)...
377  u0 = p_poly->verts[3] - *p_cam_pos;
378  u1 = p_poly->verts[2] - *p_cam_pos;
379  p = Mth::CrossProduct( u0, u1 );
380  p.Normalize();
381  p[W] = -Mth::DotProduct( p, *p_cam_pos );
382  sOccluder::Occluders[i].planes[1] = p;
383 
384  // ...then top (0->3->2)...
385  u0 = p_poly->verts[2] - *p_cam_pos;
386  u1 = p_poly->verts[1] - *p_cam_pos;
387  p = Mth::CrossProduct( u0, u1 );
388  p.Normalize();
389  p[W] = -Mth::DotProduct( p, *p_cam_pos );
390  sOccluder::Occluders[i].planes[2] = p;
391 
392  // ...then bottom (0->1->4)...
393  u0 = p_poly->verts[0] - *p_cam_pos;
394  u1 = p_poly->verts[3] - *p_cam_pos;
395  p = Mth::CrossProduct( u0, u1 );
396  p.Normalize();
397  p[W] = -Mth::DotProduct( p, *p_cam_pos );
398  sOccluder::Occluders[i].planes[3] = p;
399 
400  // ...then front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
401  // are considered occluded. (1->4->3)...
402  u0 = p_poly->verts[3] - p_poly->verts[0];
403  u1 = p_poly->verts[2] - p_poly->verts[0];
404  p = Mth::CrossProduct( u0, u1 );
405  p.Normalize();
406  p[W] = -Mth::DotProduct( p, p_poly->verts[0] );
407  sOccluder::Occluders[i].planes[4] = p;
408  }
409  }
410 
411 
412  if (sOccluder::sUseScratchPad)
413  {
414  // Copy the occlusion polygons to the scratchpad
415  // as we will read them a zillion times during rendering
416  sOccluder * p_occluder = &sOccluder::Occluders[0];
417  sOccluder * p_scratch = (sOccluder*)0x70000000;
418  // sOccluder * p_scratch = (sOccluder*)scratch;
419  for( uint32 o = sOccluder::NumOccluders; o > 0 ; --o )
420  {
421  *p_scratch = *p_occluder;
422  p_occluder++;
423  p_scratch++;
424  }
425  Dbg_MsgAssert((void*)p_scratch < (void*)0x70004000,("Scratchpad overflow")); // 16k scratchpad. Nice.
426  }
427 
428 #ifdef OCCLUDER_USES_VU0_MACROMODE
429  sOccluder * p_occluder = &sOccluder::Occluders[0];
430  uint128 * p_vu0_mem = (uint128*)0x11004000; // VU0 data memory
431 
432  // Copies the number of occulders into X
433  *((uint32*) (p_vu0_mem++)) = sOccluder::NumOccluders;
434 
435  for( uint32 o = sOccluder::NumOccluders; o > 0 ; --o )
436  {
437  *(p_vu0_mem++) = *((uint128*) &(p_occluder->planes[0]));
438  *(p_vu0_mem++) = *((uint128*) &(p_occluder->planes[1]));
439  *(p_vu0_mem++) = *((uint128*) &(p_occluder->planes[2]));
440  *(p_vu0_mem++) = *((uint128*) &(p_occluder->planes[3]));
441  *(p_vu0_mem++) = *((uint128*) &(p_occluder->planes[4]));
442  *((uint32*) (p_vu0_mem++)) = p_occluder->score;
443  p_occluder++;
444  }
445 
446  FlushCache(WRITEBACK_DCACHE);
447 
448  Dbg_MsgAssert((void*)p_vu0_mem < (void*)0x11005000,("VU0 memory overflow")); // 4k vu0 mem. Nice.
449 
450  sCopiedOccludersOffline = true; // so we know to copy the scores back (technically, this should be in
451  // TestSphereAgainstOccluders(), but we only want to write this var once
452 #endif
453

461 // Mick: removed this to fix a warning in final build.

463  Mth::Vector * p_plane;
