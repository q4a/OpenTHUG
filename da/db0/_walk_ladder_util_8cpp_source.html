236 bool CWalkComponent::FilterLadderUpRail ( const Mth::Vector& bottom_pos, const Mth::Vector& top_pos, const Mth::Matrix& orientation, SLadderRailData& rail_data )
237 {
238  // check that we're at the bottom of the ladder
239  if (bottom_pos[Y] > top_pos[Y]) return false;
240 
241  // check that the proposed ladder rail is vertical
242  Mth::Vector ladder_direction = top_pos - bottom_pos;
243  ladder_direction.Normalize();
244  if (ladder_direction[Y] < 0.995f) return false;
245 
246  // check that we're facing the ladder
247  if (Mth::DotProduct(m_facing, orientation[Z]) < cosf(Mth::DegToRad(s_get_param(CRCD(0x456f216d, "max_onto_ladder_angle"))))) return false;
248 
249  // calculate the climb position
250  rail_data.climb_point = bottom_pos + ladder_direction * s_get_param(CRCD(0x5296e3fd, "ladder_bottom_offset_up"));
251  rail_data.climb_point -= s_get_param(CRCD(0x35b3bbda, "ladder_climb_offset")) * orientation[Z];
252 
253  // send feeler to climb position
254  CFeeler feeler;
255  feeler.m_start = m_pos;
256  feeler.m_end = rail_data.climb_point - 6.0f * orientation[Z];
257  #ifdef __USER_DAN__
258  if (Script::GetInteger(CRCD(0xaf90c5fd, "walking_debug_lines")))
259  {
260  feeler.DebugLine(0, 0, 255, 0);
261  }
262  #endif
263  if (feeler.GetCollision()) return false;
264 
265  // store the climb facing
266  rail_data.climb_facing = orientation[Z];
267 
268  return true;
269 }

315 bool CWalkComponent::FilterLadderAirGrabRail ( const Mth::Vector& bottom_pos, const Mth::Vector& top_pos, const Mth::Matrix& orientation, const Mth::Vector& preliminary_climb_point, float along_rail_factor, SLadderRailData& rail_data )
316 {
317  // check that the proposed ladder rail is vertical
318  Mth::Vector ladder_direction = top_pos - bottom_pos;
319  ladder_direction.Normalize();
320  if (ladder_direction[Y] < 0.995f) return false;
321 
322  // insure that the target point on the ladder is above the climb off bottom point
323  float off_bottom_height = bottom_pos[Y] + ladder_direction[Y] * s_get_param(CRCD(0x5296e3fd, "ladder_bottom_offset_up"));
324  if (preliminary_climb_point[Y] < off_bottom_height) return false;
325 
326  // insure that the target point on the ladder is below the climb off top point
327  float off_top_height = top_pos[Y] - ladder_direction[Y] * s_get_param(CRCD(0x76bf49e0, "ladder_top_offset_up"));
328  if (preliminary_climb_point[Y] > off_top_height) return false;
329 
330  // move the climb point a touch back off the rail
331  rail_data.climb_point = preliminary_climb_point - s_get_param(CRCD(0x35b3bbda, "ladder_climb_offset")) * orientation[Z];
332 
333  // send a feeler to the climb position
334  CFeeler feeler;
335  feeler.m_start = m_pos;
336  feeler.m_end = rail_data.climb_point;
337  #ifdef __USER_DAN__
338  if (Script::GetInteger(CRCD(0xaf90c5fd, "walking_debug_lines")))
339  {
340  feeler.DebugLine(0, 0, 255, 0);
341  }
342  #endif
343  if (feeler.GetCollision()) return false;
344 
345  // store the climb facing
346  rail_data.climb_facing = orientation[Z];
347 
348  rail_data.along_rail_factor = along_rail_factor;
349 
350  mp_core_physics_component->SetFlagTrue(SNAPPED);
351 
352  return true;
353 }

378 void CWalkComponent::ladder_movement ( )
379 {
380  // movement up and down the ladder
381 
382  // the desired speed; this will be used to scale the animation speed
383  m_anim_effective_speed = s_get_param(CRCD(0xab2db54, "ladder_move_speed")) * m_control_magnitude * cosf(mp_input_component->GetControlPad().m_leftAngle);
384 
385  // the movement this frame
386  float movement = m_anim_effective_speed * m_frame_length;
387 
388  // get the top rail node; mp_rail_start is always the bottom node
389  const CRailNode* p_top_rail_node = mp_rail_start->GetNextLink() ? mp_rail_start->GetNextLink() : mp_rail_start->GetPrevLink();
390  Dbg_Assert(p_top_rail_node);
391 
392  // the offset from the ladder bottom to rail top
393  Mth::Vector rail = mp_rail_manager->GetPos(p_top_rail_node) - mp_rail_manager->GetPos(mp_rail_start);
394  Dbg_MsgAssert(rail.Length() > s_get_param(CRCD(0x76bf49e0, "ladder_top_offset_up")), ("Ladder too short for ladder animations"));
395 
396  // the movement in terms of the fraction of the rail
397  float delta_factor = movement / rail.Length();
398 
399  // movement is up or down
400  bool up = delta_factor > 0.0f;
401 
402  // move along the rail
403  m_along_rail_factor += delta_factor;
404 
405  float rail_length = rail.Length();
406 
407  // see if we've hit the bottom
408  float bottom_along_rail_factor = s_get_param(CRCD(0x5296e3fd, "ladder_bottom_offset_up")) / rail_length;
409  if (m_along_rail_factor < bottom_along_rail_factor)
410  {
411  off_ladder_bottom();
412  return;
413  }
414 
415  // see if we've hit the top
416  float top_along_rail_factor = (rail_length - s_get_param(CRCD(0x76bf49e0, "ladder_top_offset_up"))) / rail_length;
417  if (m_along_rail_factor > top_along_rail_factor)
418  {
419  // we've hit the top
420  off_ladder_top();
421  return;
422  }
423 
424  // still clamp for now
425  m_along_rail_factor = Mth::Clamp(m_along_rail_factor, 0.0f, 1.0f);
426 
427  if (m_along_rail_factor == 0.0f || m_along_rail_factor == 1.0f)
428  {
429  m_frame_event = CRCD(0xc84243da, "Ladder");
430  m_anim_effective_speed = 0.0f;
431  }
432  else
433  {
434  m_frame_event = up ? CRCD(0xaf5abc82, "LadderMoveUp") : CRCD(0xfec9dded, "LadderMoveDown");
435  }
436 
437  // calculate our new position
438  m_pos = mp_rail_manager->GetPos(mp_rail_start) + m_along_rail_factor * rail - s_get_param(CRCD(0x35b3bbda, "ladder_climb_offset")) * m_facing;
439  DUMP_WPOSITION
440 
441  // positive animation speed
442  m_anim_effective_speed = Mth::Abs(m_anim_effective_speed);
443 }
