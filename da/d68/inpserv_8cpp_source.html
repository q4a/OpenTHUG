133 void Server::service_handlers( void )
134 {
135 //
136 
137  unsigned char *control_data;
138  Data *handler_data;
139  Flags<SIO::Capabilities> caps;
140 
141  // If we are getting our input from recorded data, set all depended controller data from recorded data
142  if( m_data_in && m_data_in->m_valid )
143  {
144  int i;
145 
146  m_data.m_prev = m_data.m_cur;
147  m_data.m_cur = m_data_in->m_cur; // Get digital input from buffer
148  m_data.m_new = m_data.m_prev ^ m_data.m_cur;
149  m_data.m_Buttons = m_data.m_cur;
150  m_data.m_Makes = m_data.m_new & m_data.m_cur;
151  m_data.m_Breaks = m_data.m_new & ~m_data.m_cur;
152 
153  for( i = 0; i < Data::vMAX_ANALOG_EVENTS; i++ ) // Get analog input from buffer
154  {
155  m_data.m_Event[i] = m_data_in->m_event[i];
156  }
157 
158  m_handler_stack.Process();
159 
160  m_data_in++;
161  return;
162  }
163 
164  if( ( m_device ) &&
165  ( m_device->HasValidControlData()))
166  {
167  Inp::Manager * inp_manager = Inp::Manager::Instance();
168 
169  control_data = m_device->GetControlData();
170  handler_data = &m_data;
171  caps = m_device->GetCapabilities();
172  Dbg_Assert( control_data );
173 
174  if( m_data_out )
175  {
176  m_data_out->m_valid = FALSE;
177  }
178 
179  if( control_data[0] == 0 ) // Valid controller communication
180  {
181  unsigned char controller_type;
182 
183  controller_type = control_data[1] >> 4; // top 4 bits contain controller type
184  handler_data->m_prev = handler_data->m_cur;
185  // Regular buttons
186  handler_data->m_cur = 0xFFFF ^ (( control_data[2] << 8 ) | control_data[3] );
187  // The XBox black & white and NGC Z buttons.
188  handler_data->m_cur |= (control_data[20]<<16);
189 
190 
191  handler_data->m_new = handler_data->m_prev ^ handler_data->m_cur;
192 
193  handler_data->m_Buttons = handler_data->m_cur;
194  handler_data->m_Makes = handler_data->m_new & handler_data->m_cur;
195  handler_data->m_Breaks = handler_data->m_new & ~handler_data->m_cur;
196 
197  switch( controller_type )
198  {
199  case SIO::Device::vNEGI_COM:
200  // TODO: Handle
201  Dbg_MsgAssert( 0,( "Unsupported Device Type\n" ));
202  break;
203 
204  case SIO::Device::vKONAMI_GUN:
205  // TODO: Handle
206  Dbg_MsgAssert( 0,( "Unsupported Device Type\n" ));
207  break;
208 
209  case SIO::Device::vDIGITAL_CTRL:
210  handler_data->ConvertDigitalToAnalog();
211  break;
212 
213  case SIO::Device::vNAMCO_GUN:
214  // TODO: Handle
215  Dbg_MsgAssert( 0,( "Unsupported Device Type\n" ));
216  break;
217 
218  case SIO::Device::vJOYSTICK:
219  handler_data->m_Event[Data::vA_RIGHT_X] = control_data[4];
220  handler_data->m_Event[Data::vA_RIGHT_Y] = control_data[5];
221  handler_data->m_Event[Data::vA_LEFT_X] = control_data[6];
222  handler_data->m_Event[Data::vA_LEFT_Y] = control_data[7];
223  handler_data->ConvertDigitalToAnalog();
224  if( inp_manager->ShouldAnalogStickOverride())
225  {
226  handler_data->OverrideAnalogPadWithStick();
227  }
228  break;
229 
230  case SIO::Device::vANALOG_CTRL:
231  //if( caps.TestMask( SIO::mANALOG_BUTTONS ))
232  if( m_device->GetButtonMode() == SIO::vANALOG )
233  {
234  memcpy( handler_data->m_Event, &control_data[4], Data::vMAX_ANALOG_EVENTS );
235  }
236  else
237  {
238  handler_data->m_Event[Data::vA_RIGHT_X] = control_data[4];
239  handler_data->m_Event[Data::vA_RIGHT_Y] = control_data[5];
240  handler_data->m_Event[Data::vA_LEFT_X] = control_data[6];
241  handler_data->m_Event[Data::vA_LEFT_Y] = control_data[7];
242  handler_data->ConvertDigitalToAnalog();
243  handler_data->handle_analog_tolerance();
244  if( inp_manager->ShouldAnalogStickOverride())
245  {
246  handler_data->OverrideAnalogPadWithStick();
247  }
248  }
249  break;
250 
251  case SIO::Device::vFISHING_CTRL:
252  // TODO: Handle
253  Dbg_MsgAssert( 0,( "Unsupported Device Type\n" ));
254  break;
255 
256  case SIO::Device::vJOG_CTRL:
257  // TODO: Handle
258  Dbg_MsgAssert( 0,( "Unsupported Device Type\n" ));
259  break;
260 
261  }
262 
263  if( m_data_out )
264  {
265  int i;
266 
267  m_data_out->m_cur = handler_data->m_cur;
268  m_data_out->m_valid = TRUE;
269  for( i = 0; i < Data::vMAX_ANALOG_EVENTS; i++ ) // Get analog input from buffer
270  {
271  m_data_out->m_event[i] = handler_data->m_Event[i];
272  }
273 
274  m_data_out++;
275  }
276 
277  // Make simple controller buttons accessible at global level
278  // for debugging tools
279 
280  if (m_device)
281  {
282  gDebugButtons[m_device->GetIndex()] = m_data.m_Buttons;
283  gDebugMakes[m_device->GetIndex()] = m_data.m_Makes;
284  gDebugBreaks[m_device->GetIndex()] = m_data.m_Breaks;
285 
286  #ifdef __PLAT_NGPS__
287  MemView_Input(m_data.m_Buttons, m_data.m_Makes, m_data.m_Breaks );
288 
289  // Mick: added test for "Select" which will now exit the game if in demo mode
290 
291  if (Config::Bootstrap())
292  {
293 
294  bool exit = false;
295 
296  if (m_data.m_Buttons ||
297  (handler_data->m_Event[Data::vA_RIGHT_X] !=128)||
298  (handler_data->m_Event[Data::vA_RIGHT_Y] !=128) ||
299  (handler_data->m_Event[Data::vA_LEFT_X] !=128) ||
300  (handler_data->m_Event[Data::vA_LEFT_Y] !=128))
301  {
302  //printf ("%d activity....\n", inactive_countdown);
303  inactive_countdown = inactive_time;
304  }
305 
306  if (m_device->GetIndex() == 0)
307  {
308 
309  if (inactive_countdown)
310  {
311  inactive_countdown --;
312  if (inactive_countdown == 0)
313  {
314  printf ("Exiting due to inactivity\n");
315  exit = true;
316  demo_exit_reason = SCE_DEMO_ENDREASON_PLAYABLE_INACTIVITY_TIMEOUT;
317  }
318  }
319  if (gameplay_time)
320  {
321  gameplay_time--;
322  if (gameplay_time == 0)
323  {
324  printf ("Exiting due to gameplay timer exiting\n");
325  exit = true;
326  demo_exit_reason = SCE_DEMO_ENDREASON_PLAYABLE_GAMEPLAY_TIMEOUT;
327  }
328  }
329  }
330 
331  if ((m_data.m_Buttons & Inp::Data::mD_SELECT) && (m_data.m_Buttons & Inp::Data::mD_START))
332  {
333  exit = true;
334  demo_exit_reason = SCE_DEMO_ENDREASON_PLAYABLE_QUIT;
335  }
336 
337  if ( exit )
338  {
339  Sys::ExitDemo();
340  }
341  }
342  #endif
343  }
344 
345  // At this point, we have obtained and converted valid controller information
346  // Now propogate it to clients
347  m_handler_stack.Process();
348  }
349  }

357 void Data::handle_analog_tolerance( void )
358 {
359 
360  // copy unclamped values over before we clamp them
361  m_Event[Data::vA_RIGHT_X_UNCLAMPED] = m_Event[Data::vA_RIGHT_X];
362  m_Event[Data::vA_RIGHT_Y_UNCLAMPED] = m_Event[Data::vA_RIGHT_Y];
363  m_Event[Data::vA_LEFT_X_UNCLAMPED] = m_Event[Data::vA_LEFT_X];
364  m_Event[Data::vA_LEFT_Y_UNCLAMPED] = m_Event[Data::vA_LEFT_Y];
365 
366 
367  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_RIGHT_X] ) <= vANALOGUE_TOL )
368  {
369  m_Event[Data::vA_RIGHT_X] = vANALOGUE_CENTER;
370  }
371 
372  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_RIGHT_Y] ) <= vANALOGUE_TOL )
373  {
374  m_Event[Data::vA_RIGHT_Y] = vANALOGUE_CENTER;
375  }
376 
377  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_LEFT_X] ) <= vANALOGUE_TOL )
378  {
379  m_Event[Data::vA_LEFT_X] = vANALOGUE_CENTER;
380  }
381 
382  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_LEFT_Y] ) <= vANALOGUE_TOL )
383  {
384  m_Event[Data::vA_LEFT_Y] = vANALOGUE_CENTER;
385  }
386 
387 // "unclamped", only zero the controllers if BOTH X and Y are in the dead zone
388 // that way, we can smoothly go around the periphery, and get much better aiming ability
389 // (improves the look-around camera, and the walking)
390 
391  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_RIGHT_X_UNCLAMPED] ) <= vANALOGUE_TOL
392  && Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_RIGHT_Y_UNCLAMPED] ) <= vANALOGUE_TOL )
393  {
394  m_Event[Data::vA_RIGHT_X_UNCLAMPED] = vANALOGUE_CENTER;
395  m_Event[Data::vA_RIGHT_Y_UNCLAMPED] = vANALOGUE_CENTER;
396  }
397 
398  if( Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_LEFT_X_UNCLAMPED] ) <= vANALOGUE_TOL
399  && Mth::Abs( vANALOGUE_CENTER - m_Event[Data::vA_LEFT_Y_UNCLAMPED] ) <= vANALOGUE_TOL )
400  {
401  m_Event[Data::vA_LEFT_X_UNCLAMPED] = vANALOGUE_CENTER;
402  m_Event[Data::vA_LEFT_Y_UNCLAMPED] = vANALOGUE_CENTER;
403  }

480 void Data::MaskAnalogInput( int button_mask )
481 {
482 //
483 
484 
485  int i, mask;
486 
487  mask = 1;
488  for( i = 0; i < vMAX_ANALOG_EVENTS; i++, mask <<= 1 )
489  {
490  if( button_mask & mask )
491  {
492  // Analog stick is neutral at 128, all other buttons are neutral at 0
493  if(
494  ( i == vA_RIGHT_X ) ||
495  ( i == vA_RIGHT_Y ) ||
496  ( i == vA_LEFT_X ) ||
497  ( i == vA_LEFT_Y ) ||
498  ( i == vA_RIGHT_X_UNCLAMPED ) ||
499  ( i == vA_RIGHT_Y_UNCLAMPED ) ||
500  ( i == vA_LEFT_X_UNCLAMPED ) ||
501  ( i == vA_LEFT_Y_UNCLAMPED ))
502  {
503  m_Event[i] = 128;
504  }
505  else
506  {
507  m_Event[i] = 0;
508  }
509  }
510  }
511 }

514 /* Overrides Analog Pad with Analog Stick values if pad is not in */

516 /******************************************************************/
517 
518 void Data::OverrideAnalogPadWithStick( void )
519 {
520 
521 
522  // Only override analog pad if no analog input is present
523  if( m_Buttons & ( mD_UP | mD_RIGHT | mD_DOWN | mD_LEFT ))
524  {
525  return;
526  }
527 
528  if( m_Event[vA_LEFT_X] > vANALOGUE_CENTER )
529  {
530  m_Event[vA_RIGHT] = m_Event[vA_LEFT_X] - vANALOGUE_CENTER;
531  m_Buttons |= mD_RIGHT;
532  }
533  else if( m_Event[vA_LEFT_X] < vANALOGUE_CENTER )
534  {
535  m_Event[vA_LEFT] = Mth::Abs( vANALOGUE_CENTER - m_Event[vA_LEFT_X] );
536  m_Buttons |= mD_LEFT;
537  }
538 
539  if( m_Event[vA_LEFT_Y] < vANALOGUE_CENTER )
540  {
541  m_Event[vA_UP] = m_Event[Data::vA_LEFT_Y] - vANALOGUE_CENTER;
542  m_Buttons |= mD_UP;
543  }
544  else if( m_Event[vA_LEFT_Y] > vANALOGUE_CENTER )
545  {
546  m_Event[vA_DOWN] = Mth::Abs( vANALOGUE_CENTER - m_Event[vA_LEFT_Y] );
547  m_Buttons |= mD_DOWN;
548  }
549 }
