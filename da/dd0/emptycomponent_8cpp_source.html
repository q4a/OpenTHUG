18 // #include <gel/components/watchcomponent.h>
19 // - Add it to build\gel.mkf, like:
20 // $(NGEL)/components/WatchComponent.cpp\
21 // - Fill in the OWNER (yourself) and the CREATION DATE (today's date) in the .cpp and the .h files
