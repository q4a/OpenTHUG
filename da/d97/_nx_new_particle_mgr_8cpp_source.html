98 CNewParticle* CNewParticleManager::CreateParticle( CParticleParams* params, bool generate_name )
99 {
100  CNewParticle* particle;
101 
102  particle = plat_create_particle();
103  if( particle )
104  {
105  if( generate_name )
106  {
107  uint32 name;
108 
109  name = params->m_Name;
110  // Ensure a unique name for this system
111  while(( mp_particle_table->GetItem( name, false ) != NULL ))
112  {
113  name++;
114  }
115 
116  params->m_Name = name;
117  }
118 
119  particle->Initialize( params );
120  mp_particle_table->PutItem( params->m_Name, particle );
121  }
122 
123  return particle;
124 }

173 void CNewParticleManager::Cleanup( void )
174 {
175  // Goes through and removes all particle systems.
176  mp_particle_table->IterateStart();
177  CNewParticle *p_particle = mp_particle_table->IterateNext();
178  while( p_particle )
179  {
180  CNewParticle *p_next = mp_particle_table->IterateNext();
181  p_particle->Destroy();
182  delete p_particle;
183  p_particle = p_next;
184  }
185  mp_particle_table->FlushAllItems();
186 }
