170 // Loads a pre file into memory. The name must have no path since the pre is
171 // assumed to be in the data\pre directory.
172 // Ie, a valid name would be "alc.pre"
173 void LoadPre(const char *p_preFileName)
174 {
175  Dbg_MsgAssert(p_preFileName,("NULL p_preFileName"));
176 
177  // Check to see if the pre is already loaded, and return without doing anything if it is.
178  for (int i=0; i<MAX_PRE_FILES; ++i)
179  {
180  if (spp_pre_files[i])
181  {
182  // Do a case-insensitive comparison.
183  if (stricmp(spp_pre_files[i],p_preFileName)==0)
184  {
185  // Found it! Nothing to do.
186  return;
187  }
188  }
189  }
190 
191  // Not loaded already, so need to load it.
192  // Find a spare slot ...
193  int spare_index=-1;
194  for (int pre=0; pre<MAX_PRE_FILES; ++pre)
195  {
196  if (!spp_pre_files[pre])
197  {
198  spare_index=pre;
199  break;
200  }
201  }
202  Dbg_MsgAssert(spare_index!=-1, ("Reached limit of %d pre files loaded at once.",MAX_PRE_FILES));
203 
204  // Prefix with 'pre\'
205  char p_full_pre_name[500];
206  Dbg_MsgAssert(strlen(p_preFileName)<400,("Pre name too long: '%s'",p_preFileName));
207  sprintf(p_full_pre_name,"pre\\%s",p_preFileName);
208 
209  #ifdef __NOPT_ASSERT__
210  printf("Loading pip pre '%s'\n",p_preFileName);
211  #endif
212 
213  // Calculate the space needed to store the pre name, which will get stuck before the
214  // file data.
215  int name_size=strlen(p_preFileName)+1;
216  name_size=(name_size+15)&~15;
217 
218 
219 
220  char *p_old_file_data=NULL;
221  uint32 original_file_size = File::CanFileBeLoadedQuickly( p_full_pre_name );
222  uint32 old_pre_buffer_size=0;
223 
224  if (original_file_size)
225  {
226  // Goody! It can be loaded quickly.
227 
228  old_pre_buffer_size=(original_file_size+2047)&~2047;
229  old_pre_buffer_size+=name_size;
230 
231  p_old_file_data=(char*)Mem::Malloc(old_pre_buffer_size);
232  Dbg_MsgAssert(p_old_file_data,("Could not allocate memory for file %s",p_full_pre_name));
233 
234  // Copy in the pre name at the start of the buffer.
235  strcpy(p_old_file_data,p_preFileName);
236 
237  bool fileLoaded = File::LoadFileQuicklyPlease( p_full_pre_name, (uint8*)(p_old_file_data+name_size) );
238  if ( !fileLoaded )
239  {
240  Dbg_MsgAssert( 0,( "File %s failed to load quickly.\n", p_full_pre_name));
241  Mem::Free(p_old_file_data);
242  p_old_file_data=NULL;
243  }
244  }
245  else
246  {
247  // Open the file & get its file size
248  void *p_file = File::Open(p_full_pre_name, "rb");
249 
250  if (!p_file)
251  {
252  Dbg_MsgAssert(0,("Could not open file %s",p_full_pre_name));
253  }
254 
255  original_file_size=File::GetFileSize(p_file);
256  if (!original_file_size)
257  {
258  Dbg_MsgAssert(0,("Zero file size for file %s",p_full_pre_name));
259  }
260 
261 
262  // Allocate memory.
263  // Just to be safe, make sure the buffer is at least the file size rounded up to the
264  // next multiple of 2048, cos maybe loading a file off CD will always load
265  // whole numbers of sectors.
266  // Haven't checked that though.
267  old_pre_buffer_size=(original_file_size+2047)&~2047;
268  old_pre_buffer_size+=name_size;
269 
270  p_old_file_data=(char*)Mem::Malloc(old_pre_buffer_size);
271  Dbg_MsgAssert(p_old_file_data,("Could not allocate memory for file %s",p_full_pre_name));
272 
273  // Copy in the pre name at the start of the buffer.
274  strcpy(p_old_file_data,p_preFileName);
275 
276  // Load the file into memory then close the file.
277  #ifdef __NOPT_ASSERT__
278  long bytes_read=File::Read(p_old_file_data+name_size, 1, original_file_size, p_file);
279  Dbg_MsgAssert(bytes_read<=(long)original_file_size,("bytes_read>original_file_size ?"));
280  #else
281  File::Read(p_old_file_data+name_size, 1, original_file_size, p_file);
282  #endif
283 
284  File::Close(p_file);
285  }
286 
287 
288  // Calculate the new buffer size required.
289  // Note that even if no decompression is required, the new buffer will probably need to
290  // be bigger than the old, because the contained files need to be moved so that they all start
291  // on 16 byte boundaries. This is required by the collision code for example.
292 
293  uint32 new_pre_buffer_size=name_size;
294  new_pre_buffer_size+=sizeof(SPreHeader);
295 
296  SPreHeader *p_pre_header=sSkipOverPreName(p_old_file_data);
297  uint32 num_files=p_pre_header->mNumFiles;
298  SPreContained *p_contained=(SPreContained*)(p_pre_header+1);
299  for (uint32 f=0; f<num_files; ++f)
300  {
301  // Do a quick check to make sure that all the usage values are initially zero.
302  // They should be, because the namesize member should never be bigger than 65535 ...
303  Dbg_MsgAssert(p_contained->mUsage==0,("The file %s in %s has mUsage=%d ??",p_contained->mpName,p_preFileName,p_contained->mUsage));
304 
305  new_pre_buffer_size+=(uint32)p_contained->mpName-(uint32)p_contained;
306  new_pre_buffer_size+=p_contained->mNameSize;
307 
308  new_pre_buffer_size=(new_pre_buffer_size+15)&~15;
309 
310  new_pre_buffer_size+=(p_contained->mDataSize+3)&~3;
311 
312  p_contained=sSkipToNextPreContained(p_contained,NOT_QUAD_WORD_ALIGNED);
313  }
314 
315  // Need to add a small margin to prevent decompressed data overtaking the compressed data.
316  new_pre_buffer_size+=IN_PLACE_DECOMPRESSION_MARGIN;
317 
318  // At this point we have:
319  //
320  // original_file_size The exact size of the original pre file. Will be a multiple of 4.
321  //
322  // name_size The size of the pre name, rounded up to a multiple of 4
323  //
324  // old_pre_buffer_size The size of the memory buffer pointed to by p_old_file_data
325  // Equals name_size + (original_file_size, rounded up to a multiple of 2048)
326  //
327  // new_pre_buffer_size The required size of the new buffer, to contain the decompressed pre,
328  // with all the files moved to be at 16 byte boundaries.
329  //
330 
331  #ifdef __NOPT_ASSERT__
332  printf("Decompressing and rearranging pre file ...\n");
333  #endif
334 
335  // Note: It does not matter if new_pre_buffer_size is not a multiple of 2048 any more.
336  // old_pre_buffer_size was only a multiple of 2048 to ensure that the file loading
337  // did not overrun the end of the buffer. I think file loading may only load a whole
338  // number of sectors.
339 
340 
341  // Reallocate the buffer.
342  char *p_new_file_data=NULL;
343  if (Mem::Manager::sHandle().GetContextDirection()==Mem::Allocator::vTOP_DOWN)
344  {
345  // If using the top-down heap expand the buffer downwards ...
346 
347  // p_new_file_data will become a pointer lower down in memory than p_old_file_data.
348  // The old data pointed to by p_old_file_data will still be there.
349  // The memory between p_new_file_data and p_old_file_data is all free to use. Hoorah!
350  p_new_file_data=(char*)Mem::ReallocateDown(new_pre_buffer_size,p_old_file_data);
351  }
352  else
353  {
354  // If using the top-down heap expand the buffer upwards ...
355  p_new_file_data=(char*)Mem::ReallocateUp(new_pre_buffer_size,p_old_file_data);
356  Dbg_MsgAssert(p_new_file_data,("ReallocateUp failed!"));
357 
358  // Now need to move the data up so that it can be decompressed into the gap below.
359  uint32 *p_source=(uint32*)(p_old_file_data+old_pre_buffer_size);
360  uint32 *p_dest=(uint32*)(p_new_file_data+new_pre_buffer_size);
361  // Loading backwards cos the destination overlaps the source.
362 
363  // Note: Did some timing tests to see if this was was slow, but it's not really.
364  // To load all the pre files in the game, one after the other, takes an average of
365  // 32.406 seconds when the top down heap is used. When using the bottom up, which
366  // necessitates doing this copy, it takes 33.518 seconds.
367  // So per pre file that isn't much. (There's about 60 pre files)
368  Dbg_MsgAssert((old_pre_buffer_size&3)==0,("old_pre_buffer_size not a multiple of 4 ?"));
369  uint32 num_longs=old_pre_buffer_size/4;
370  for (uint32 i=0; i<num_longs; ++i)
371  {
372  --p_source;
373  --p_dest;
374  *p_dest=*p_source;
375  }
376 
377  // Now update p_old_file_data to point where it should.
378  p_old_file_data=p_new_file_data+new_pre_buffer_size-old_pre_buffer_size;
379  }
380 
381  // Copy the pre name down.
382  for (int i=0; i<name_size; ++i)
383  {
384  p_new_file_data[i]=p_old_file_data[i];
385  }
386 
387  // Write in the new pre header
388  SPreHeader *p_source_header=(SPreHeader*)(p_old_file_data+name_size);
389  SPreHeader *p_dest_header=(SPreHeader*)(p_new_file_data+name_size);
390  p_dest_header->mNumFiles=p_source_header->mNumFiles;
391  p_dest_header->mVersion=p_source_header->mVersion;
392 
393  // Copy down each of the contained files, decompressing those that need it.
394  SPreContained *p_source_contained=(SPreContained*)(p_source_header+1);
395  SPreContained *p_dest_contained=(SPreContained*)(p_dest_header+1);
396  for (uint32 f=0; f<num_files; ++f)
397  {
398  // Copy down the contained file's header.
399  p_dest_contained->mDataSize=p_source_contained->mDataSize;
400  p_dest_contained->mChecksum=p_source_contained->mChecksum;
401  p_dest_contained->mCompressedSize=0; // The new file will not be compressed.
402  p_dest_contained->mUsage=0;
403  p_dest_contained->mNameSize=p_source_contained->mNameSize;
404  for (int i=0; i<p_source_contained->mNameSize; ++i)
405  {
406  p_dest_contained->mpName[i]=p_source_contained->mpName[i];
407  }
408 
409  // Pre-calculate p_next_source_contained because decompression may (and often will)
410  // cause the new data to overwrite the contents of p_source_contained.
411  SPreContained *p_next_source_contained=sSkipToNextPreContained(p_source_contained,NOT_QUAD_WORD_ALIGNED);
412 
413  uint8 *p_source=(uint8*)(p_source_contained->mpName+p_source_contained->mNameSize);
414  uint8 *p_dest=(uint8*)(p_dest_contained->mpName+p_dest_contained->mNameSize);
415  p_dest=(uint8*)( ((uint32)p_dest+15)&~15 );
416 
417  if (p_source_contained->mCompressedSize)
418  {
419  uint32 num_bytes_decompressed=p_dest_contained->mDataSize;
420  uint8 *p_end=DecodeLZSS(p_source,p_dest,p_source_contained->mCompressedSize);
421  Dbg_MsgAssert(p_end==p_dest+num_bytes_decompressed,("Eh? DecodeLZSS wrote %d bytes, expected it to write %d",p_end-p_dest,num_bytes_decompressed));
422 
423  // For neatness, write zero's into the pad bytes at the end, otherwise they'll
424  // be uninitialised data.
425  while (num_bytes_decompressed & 3)
426  {
427  *p_end++=0;
428  ++num_bytes_decompressed;
429  }
430  }
431  else
432  {
433  // Uncompressed, so just copy the data down.
434  uint32 *p_source_long=(uint32*)p_source;
435  uint32 *p_dest_long=(uint32*)p_dest;
436  // mDataSize is not necessarily a multiple of 4, but the actual data will be
437  // padded at the end so that it does occupy a whole number of long words.
438  // So the +3 is to ensure that n is rounded up to the next whole number of longs.
439  uint32 n=(p_source_contained->mDataSize+3)/4;
440 
441  for (uint32 i=0; i<n; ++i)
442  {
443  *p_dest_long++=*p_source_long++;
444  }
445  }
446 
447  p_source_contained=p_next_source_contained;
448 
449  p_dest_contained=sSkipToNextPreContained(p_dest_contained);
450  }
451 
452  // I don't think I ever need to reference mSize again, but might as well make it correct.
453  p_dest_header->mSize=(uint32)p_dest_contained-(uint32)p_dest_header;
454 
455  //printf("Wasted space = %d\n",new_pre_buffer_size-((uint32)p_dest_contained-(uint32)p_new_file_data));
456 
457  spp_pre_files[spare_index]=p_new_file_data;
458  #ifdef __NOPT_ASSERT__
459  printf("Done\n");
460  #endif
461 }

463 // Removes the specified pre from memory. The name must have no path since the pre is
464 // assumed to be in the data\pre directory.
465 // Ie, a valid name would be "alc.pre"
466 //
467 // Won't do anything if the pre is gone already.
468 // Asserts if any of the files in the pre are still 'open' in that they haven't had Unload called.
469 bool UnloadPre(const char *p_preFileName)
470 {
471  bool success = false;
472 
473  Dbg_MsgAssert(p_preFileName,("NULL p_preFileName"));
474 
475  for (int i=0; i<MAX_PRE_FILES; ++i)
476  {
477  if (spp_pre_files[i])
478  {
479  // Do a case-insensitive comparison.
480  if (stricmp(spp_pre_files[i],p_preFileName)==0)
481  {
482  // Found it!
483 
484  // Before unloading it, make sure none of the files within is still open.
485  #ifdef __NOPT_ASSERT__
486  SPreHeader *p_pre_header=sSkipOverPreName(spp_pre_files[i]);
487  int num_files=p_pre_header->mNumFiles;
488  SPreContained *p_contained=(SPreContained*)(p_pre_header+1);
489  for (int f=0; f<num_files; ++f)
490  {
491  Dbg_MsgAssert(!p_contained->mUsage,("Tried to unload %s when the file %s contained within it was still open! (mUsage=%d)",spp_pre_files[i],p_contained->mpName,p_contained->mUsage));
492  p_contained=sSkipToNextPreContained(p_contained);
493  }
494  #endif
495 
496  // Delete it.
497  Mem::Free(spp_pre_files[i]);
498  spp_pre_files[i]=NULL;
499 
500  // we've successfully unloaded a pre
501  success = true;
502  }
503  }
504  }
505 
506  // Not found, so nothing to do.
507  return success;
508 }

564 void* Load(const char* p_fileName)
565 {
566  uint32 filename_checksum=Crc::GenerateCRCFromString(p_fileName);
567 
568  // First, see if the file is in one of the loaded pre files.
569  SPreContained *p_contained_file=sSeeIfFileIsInAnyPre( filename_checksum );
570  if (p_contained_file)
571  {
572  ++p_contained_file->mUsage;
573 
574  Dbg_MsgAssert(p_contained_file->mCompressedSize==0,("The file '%s' is stored compressed in %s !",p_fileName,sGetPreName(p_contained_file)));
575 
576  uint32 p_data=(uint32)p_contained_file->mpName+p_contained_file->mNameSize;
577  p_data=(p_data+15)&~15;
578  return (void*)p_data;
579  }
580 
581  // Next, see if it is one of the unpreed files that is already loaded.

802 // @parmopt name | heap | 0 (checksum value) | Which heap to use.

804 // If no heap is specified it will use whatever the current heap is.
805 bool ScriptLoadPipPre(Script::CStruct *pParams, Script::CScript *pScript)
806 {
807  const char *p_filename;
808  pParams->GetString(NONAME, &p_filename, Script::ASSERT);
809 
810  uint32 chosen_heap=0;
811  pParams->GetChecksum("Heap",&chosen_heap);
812  switch (chosen_heap)
813  {
814  case 0:
815  chosen_heap = 0;
816  // Use whatever the current heap is.
817  break;
818  case 0x477fc6de: // TopDown
819  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
820  break;
821  case 0xc80bf12d: // BottomUp
822  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().BottomUpHeap());
823  break;
824  default:
825  Dbg_Warning("Heap '%s' not supported by LoadPipPre, using current heap instead.",Script::FindChecksumName(chosen_heap));
826  chosen_heap=0;
827  break;
828  }
829 
830  LoadPre(p_filename);
831 
832  if (chosen_heap)
833  {
834  Mem::Manager::sHandle().PopContext();
835  }
836 
837  return true;

916 // Load file as quickly as possible
917 // Try loading from the pre manager first, then from the CD, using the "quick filesys"
918 // then finally using the regular file loading functions
919 // returns pointer to allocated memory, or NULL if file fails to load
920 // optionally returns size of file in bytes, in *p_filesize (ignored if p_filesize is NULL)
921 // if file fails to load, *p_filesize will be 0
922 // if you pass in p_dest, you must also pass in maxSize, the size of the buffer
923 void * LoadAlloc(const char *p_fileName, int *p_filesize, void *p_dest, int maxSize)
924 {
925 
926  // Mick 2/19/2003 - Removed code that stripped project specific headers
927  // as this is now handled at the gs_file level
928 
929  int file_size = 0;
930 // Perhaps the file is in a PRE file, so try loading it directly, as that will be quickest
931  uint8 *p_file_data = (uint8*)File::PreMgr::Instance()->LoadFile(p_fileName,&file_size, p_dest);
932  if (!p_file_data)
933  {
934  // nope, so try loading "Quickly" (basically if we are on a CD)
935  file_size = File::CanFileBeLoadedQuickly( p_fileName );
936  if (file_size)
937  {
938  // It can be loaded quickly.
939  if (!p_dest)
940  {
941  p_file_data=(uint8*)Mem::Malloc((file_size+2047)&~2047);
942  }
943  else
944  {
945  p_file_data = (uint8*)p_dest;
946  }
947 
948  bool fileLoaded = File::LoadFileQuicklyPlease( p_fileName, p_file_data );
949  if ( !fileLoaded )
950  {
951  Dbg_MsgAssert( 0,( "File %s failed to load quickly.\n", p_fileName));
952  if (!p_dest)
953  {
954  Mem::Free(p_file_data);
955  }
956  p_file_data=NULL;
957  }
958  }
959  else
960  {
961  // can't load quickly, so probably loading from the PC
962  // Open the file & get its file size
963 
964  void *p_file = File::Open(p_fileName, "rb");
965  if (!p_file)
966  {
967  Dbg_MsgAssert(0,("Could not open file '%s'",p_fileName));
968  }
969 
970  file_size=File::GetFileSize(p_file);
971  if (!file_size)
972  {
973  Dbg_MsgAssert(0,("Zero file size for file %s",p_fileName));
974  }
975 
976  if (!p_dest)
977  {
978  // Allocate memory.
979  // Just to be safe, allocate a buffer of size file_size rounded up to the
980  // next multiple of 2048, cos maybe loading a file off CD will always load
981  // whole numbers of sectors.
982  // Haven't checked that though.
983  p_file_data=(uint8*)Mem::Malloc((file_size+2047)&~2047);
984  Dbg_MsgAssert(p_file_data,("Could not allocate memory for file %s",p_fileName));
985  }
986  else
987  {
988  p_file_data = (uint8*)p_dest;
989  }
990  // Load the file into memory then close the file.
991  #ifdef __NOPT_ASSERT__
992  long bytes_read=File::Read(p_file_data, 1, file_size, p_file);
993  Dbg_MsgAssert(bytes_read<=file_size,("bytes_read>file_size ?"));
994  #else
995  File::Read(p_file_data, 1, file_size, p_file);
996  #endif
997 
998  File::Close(p_file);
999  }
1000  // Shrink memory back down to accurate usage - saves 43K total in the school!!!
1001  if (!p_dest && p_file_data)
1002  {
1003  Mem::ReallocateShrink(file_size,p_file_data);
1004  }
1005  }
1006  if (p_filesize)
1007  {
1008  *p_filesize = file_size;
1009  }
1010  // If we specified a destination, then make sure we did not overflow it
1011  if (p_dest)
1012  {
1013  Dbg_MsgAssert(((file_size + 2047)&~2047) < maxSize,("file size (%d) overflows buffer (%d) for %s\n",file_size,maxSize,p_fileName));
1014  }
1015  return (void *)p_file_data;
1016 }
