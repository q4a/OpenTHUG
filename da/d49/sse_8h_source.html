54 //-----------------------------------------------------------------------------
55 // Name: SqrtEstimae_ASM
56 // Desc: SSE square root estimate, accurate to 12 significant bits of
57 // the mantissa. Note that a check for zero must be made since
58 // sqrt(0) == 0 but 1/sqrt(0) = inf
59 //-----------------------------------------------------------------------------
60 __forceinline float SqrtEstimate_ASM( const float f )
61 {
62  float recsqrt;
63  __asm movss xmm0,f // xmm0 = f
64  __asm rsqrtss xmm1, xmm0 // xmm1 = rsqrtss(f)
65  __asm mulss xmm1, xmm0 // xmm1 = rsqrtss(f) * f = sqrt(f)
66  __asm xorps xmm2, xmm2 // xmm2 = 0
67  __asm cmpneqss xmm2, xmm0 // xmm2 = (f != 0 ? 1s : 0s)
68  __asm andps xmm1, xmm2 // xmm1 = xmm1 & xmm2
69  __asm movss recsqrt, xmm1 // return xmm1
70  return recsqrt;
71 }

74 //-----------------------------------------------------------------------------
75 // Name: ReciprocalEstimateNR_ASM
76 // Desc: SSE Newton-Raphson reciprocal estimate, accurate to 23 significant
77 // bits of the mantissa
78 // One Newtown-Raphson Iteration:
79 // f(i+1) = 2 * rcpss(f) - f * rcpss(f) * rcpss(f)
80 //-----------------------------------------------------------------------------
81 __forceinline float ReciprocalEstimateNR_ASM( const float f )
82 {
83  float rec;
84  __asm rcpss xmm0, f // xmm0 = rcpss(f)
85  __asm movss xmm1, f // xmm1 = f
86  __asm mulss xmm1, xmm0 // xmm1 = f * rcpss(f)
87  __asm mulss xmm1, xmm0 // xmm2 = f * rcpss(f) * rcpss(f)
88  __asm addss xmm0, xmm0 // xmm0 = 2 * rcpss(f)
89  __asm subss xmm0, xmm1 // xmm0 = 2 * rcpss(f)
90  // - f * rcpss(f) * rcpss(f)
91  __asm movss rec, xmm0 // return xmm0
92  return rec;
93 }

101 //-----------------------------------------------------------------------------
102 // Name: ReciprocalSqrtEstimateNR_ASM
103 // Desc: SSE Newton-Raphson reciprocal square root estimate, accurate to 23
104 // significant bits of the mantissa
105 // One Newtown-Raphson Iteration:
106 // f(i+1) = 0.5 * rsqrtss(f) * (3.0 - (f * rsqrtss(f) * rsqrtss(f))
107 // NOTE: rsqrtss(f) * rsqrtss(f) != rcpss(f) (presision is not maintained)
108 //-----------------------------------------------------------------------------
109 __forceinline float ReciprocalSqrtEstimateNR_ASM( const float f )
110 {
111  float recsqrt;
112  __asm rsqrtss xmm0, f // xmm0 = rsqrtss(f)
113  __asm movss xmm1, f // xmm1 = f
114  __asm mulss xmm1, xmm0 // xmm1 = f * rsqrtss(f)
115  __asm movss xmm2, g_SqrtNRConst+4 // xmm2 = 3.0f
116  __asm mulss xmm1, xmm0 // xmm1 = f * rsqrtss(f) * rsqrtss(f)
117  __asm mulss xmm0, g_SqrtNRConst // xmm0 = 0.5f * rsqrtss(f)
118  __asm subss xmm2, xmm1 // xmm2 = 3.0f -
119  // f * rsqrtss(f) * rsqrtss(f)
120  __asm mulss xmm0, xmm2 // xmm0 = 0.5 * rsqrtss(f)
121  // * (3.0 - (f * rsqrtss(f) * rsqrtss(f))
122  __asm movss recsqrt, xmm0 // return xmm0
123  return recsqrt;
124 }

127 //-----------------------------------------------------------------------------
128 // Name: SqrtEstimateNR_ASM
129 // Desc: SSE Newton-Raphson square root estimate, accurate to 23 significant
130 // bits of the mantissa
131 // NOTE: x/sqrt(x) = sqrt(x)
132 // One Newtown-Raphson Iteration (for 1/sqrt(x)) :
133 // f(i+1) = 0.5 * rsqrtss(f) * (3.0 - (f * rsqrtss(f) * rsqrtss(f))
134 // NOTE: rsqrtss(f) * rsqrtss(f) != rcpss(f) (presision is not maintained)
135 //-----------------------------------------------------------------------------
136 __forceinline float SqrtEstimateNR_ASM( const float f )
137 {
138  float recsqrt;
139  __asm rsqrtss xmm0, f // xmm0 = rsqrtss(f)
140  __asm movss xmm1, f // xmm1 = f
141  __asm mulss xmm1, xmm0 // xmm1 = f * rsqrtss(f)
142  __asm movss xmm2, g_SqrtNRConst+4 // xmm2 = 3.0f
143  __asm mulss xmm1, xmm0 // xmm1 = f * rsqrtss(f) * rsqrtss(f)
144  __asm mulss xmm0, g_SqrtNRConst // xmm0 = 0.5f * rsqrtss(f)
145  __asm subss xmm2, xmm1 // xmm2 = 3.0f -
146  // f * rsqrtss(f) * rsqrtss(f)
147  __asm mulss xmm0, xmm2 // xmm0 = 0.5 * rsqrtss(f)
148  // * (3.0 - (f * rsqrtss(f) * rsqrtss(f))
149  __asm xorps xmm1, xmm1 // xmm1 = 0
150  __asm mulss xmm0, f // xmm0 = sqrt(f)
151  __asm cmpneqss xmm1, f // xmm1 = (f != 0 ? 1s : 0s)
152  __asm andps xmm0, xmm1 // xmm0 = xmm1 & xmm2
153  __asm movss recsqrt, xmm0 // return xmm0
154  return recsqrt;
155 }
