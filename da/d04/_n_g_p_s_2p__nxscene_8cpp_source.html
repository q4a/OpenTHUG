124 void CPs2Scene::create_sectors(NxPs2::CGeomNode *p_node)
125 {
126  if (!p_node)
127  return;
128 
129  while (p_node)
130  {
131  // Recursively traverse tree, treating the object-level nodes as leaves.
132  // From each object, create a sector.
133  // (Mick: optimized tail recursion, after suggestion from John Brandwood)
134  if (p_node->IsObject()) // Create CSector from object
135  {
136  CSector *pSector;
137  CPs2Sector *pPs2Sector; // platform dependent class
138  CPs2Geom *pGeom;
139 
140  // Find Sector
141  pSector = GetSector(p_node->GetChecksum());
142  Dbg_MsgAssert(!pSector, ("We already have a sector with checksum %x", p_node->GetChecksum()));
143 
144  // If it doesn't exist yet, create it
145  if (!pSector)
146  {
147  pSector = new CPs2Sector; // Create platform specific data
148  pSector->SetChecksum(p_node->GetChecksum());
149 
150  pGeom = new CPs2Geom;
151  pGeom->SetEngineObject(p_node);
152 
153  pSector->SetGeom(pGeom);
154 
155  AddSector(pSector);
156 
157  //Dbg_Message("Creating new sector %x", p_node->GetChecksum());
158  }
159 
160  // Access platform dependent data
161  pPs2Sector = static_cast<CPs2Sector *>(pSector);
162 
163  // Add node to sector
164  //pPs2Sector->SetEngineObject(p_node);
165  pPs2Sector->SetEngineObject(NULL);
166  } else { // Traverse children and siblings
167  create_sectors(p_node->GetChild());
168  }
169 
170  p_node = p_node->GetSibling();
171  }
172 }
