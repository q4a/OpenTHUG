13 
14  Mick Notes:
15 
16  The calls the the DMA functions suggest that you are immediately executing something.
17  You are not. You are simply building a list of DMA commands that get executed the next frame
18 
19  The DMA list is the primary mechanism for initiating rendering on the PS2.
20 
21  To render something (like a line), you "simply" need to generate the appropiate DMA
22  packets that contain GIF packets, containing GS primitives, and then link this into the
23  main DMA list.
24 
25  Things such as world sectors have pre-build DMA lists that contain commands
26  to upload lists of vertices to VU1 micro-memory, and then upload lists of
27  triangles, and then trigger the appropiate VU1 microcode to transform and render them.
28 
29  These pre-built lists are linked in each frame with the appropiate transformation
30  matricies that are needed based on the current camera position and the current
31  object position (if it is a moving object)
32 
33  Thus, each object is mostly pre-build, and so rendering it requires very little CPU time.
34 
35  Each frame, we build a series of dynamic DMA lists.
36 
37  There are two lists for each group (see group.cpp).
38 
39  The first list is the texture upload. This is pretty simple, just uploading a few textures.
40  Texture uploading happens asyncrnously with rendering, so we can be executing a DMA list
41  of rendering stuff at the same time as we are uploading the textures for the next group.
42 
43  A lot of the DMA code is in DMA.H, as it is small inline functions.
44 
45  The Runtime buffer is dynamically allocated in InitializeEngine(). Last time I looked it
46  was 512K in size (note though, this is increased to 16MB for the debugging wireframe mode)
47 
48  The Runtime DMA lists are double buffered. You are executing one whilst building the next one.
49 
50  The pRunTimeBuffer is split in two, and pointers to the base of each half are stored in
51  pList[0] and pList[1]
52 
53  The integer varaible "Field" is 0 or 1, and indicates which field (odd or even) we are rendering to.
54 
55  The DMA lists are build using a global variable char * pLoc.
56 
57  At the start of a frame, pLoc is initilized like:
58 
59  dma::pLoc = dma::pList[Field];
60 
61  So it points to the start of one of our 256K DMA buffers (remember the other is being executed)
62 
63  DMA packets are now built using calls to the functions here and in DMA.H, for example:
64 
65  dma::Tag(dma::ref, ((uint8 *)&MPGEnd-(uint8 *)&MPGStart+15)/16, (uint)&MPGStart);
66  vif::NOP();
67  vif::NOP();
68 
69  (the above simply transfers the microcode from RAM to VU1 micromem)
70 
71  Now, it's important to note that DMA execution does NOT simply start at pList[Field] and
72  then run all the way through a frame's worth of data. There are actually several DMA lists,
73  which are fired off individually. Some of these (the groups) are logically connected so that
74  the end of one group's DMA (and the GS/VU activity it triggers) causes an interrupt,
75  which the CPU handles by starting the DMA for another group (see interrupt.cpp)
76 
77  After the groups, there is the "immediate mode" DMA list, triggered by pEpilogue->pRender[Field]
78  this is simply a raw DMA list that you can put anything you want into. It starts at the value
79  of pLoc after RenderWorld has been called (a terrible hack, as noted in the code :)
80 
81  All of the "immediate mode" rendering is considered to be part of pEpilogue. It is terminated
82  by the RenderEpilogue() function, which just links in a final interrupt trigger, which will
83  get picked up by GsHandler(), and sGroup::pRenderGroup will be set to NULL, which is what
84  WaitForRendering() uses to determine if rendering has finished.
85 
86 
87 
88

195 uint8 *dma::NextTag(uint8 *pTag, bool stepInto)
196 {
197  Dbg_MsgAssert((*(uint32 *)pTag&0x80000000)==0, ("IRQ bit set in dma tag"));
198  Dbg_MsgAssert(*(uint32 *)pTag!=refe<<28 && *(uint32 *)pTag!=refs<<28, ("refe and refs not supported in dma tag!"));
199  switch (*(uint32 *)pTag>>28)
200  {
201  case cnt:
202  return pTag + 16 + ((uint)((uint16 *)pTag)[0] << 4);
203 
204  case next:
205  return ((uint8 **)pTag)[1];
206 
207  case ref:
208  return pTag + 16;
209 
210  case call:
211  Dbg_MsgAssert(sp<2, ("dma call stack overflow"));
212  if (stepInto)
213  {
214  Stack[sp++] = pTag + 16 + ((uint)((uint16 *)pTag)[0] << 4);
215  return ((uint8 **)pTag)[1];
216  }
217  else
218  {
219  return pTag + 16 + ((uint)((uint16 *)pTag)[0] << 4);
220  }
221 
222  case ret:
223  Dbg_MsgAssert(sp>0, ("dma call stack underflow"));
224  return Stack[--sp];
225 
226  case end:
227  default:
228  return NULL;
229  }

1003 void dma::SqueezeADC(uint8 *pTag)
1004 {
1005  uint8 *p_code, *p_end, *p_unpack[5];
1006  uint16 addr, vumem_size;
1007  uint32 nloop=0, nreg=0, texcrds_size, *p_giftag=NULL, i, num_squeezed, addr_diff;
1008  uint32 *p_texcrds_source, *p_texcrds_dest;
1009  uint8 *p_weights_source, *p_weights_dest;
1010  uint16 *p_normal_source, *p_normal_dest;
1011  uint32 *p_colour_source, *p_colour_dest;
1012  uint32 *p_coords_source, *p_coords_dest;
1013  int unpack_num;
1014  bool seenMSCAL;
1015 
1016  // get start and end of vifcode packet
1017  p_code = pTag + 8;
1018  p_end = pTag + 16 + (*(uint16 *)pTag << 4);
1019 
1020  // initialise state
1021  addr_diff = 0;
1022  unpack_num = -1;
1023  seenMSCAL = true;
1024 
1025  // parse vifcodes
1026  while (p_code < p_end)
1027  {
1028  if ((p_code[3] & 0x60) == 0x60) // if it's an unpack
1029  {
1030  // adjust ADDR field
1031  addr = *(uint16 *)p_code;
1032  addr -= addr_diff;
1033  addr &= 0x3FF;
1034  *(uint16 *)p_code = *(uint16 *)p_code & 0xFC00 | addr;
1035 
1036  // if we have a VU1_ADR(Jump) in a giftag, reset offset
1037  if (((uint16 *)p_code)[1]==0x6C01 && ((uint32 *)p_code)[1]==0x00008000 && ((uint32 *)p_code)[2]==0x00000052)
1038  addr_diff = 0;
1039 
1040  // look for vertex packets
1041  if (vif::WL==1 && vif::CL>1 && seenMSCAL)
1042  {
1043  // look for giftag
1044  if (unpack_num==-1)
1045  {
1046  p_giftag = (uint32 *)p_code + 1;
1047  nloop = p_giftag[0] & 0x7FFF;
1048  nreg = p_giftag[1] >> 28;
1049  }
1050 
1051  // look for vertex elements
1052  else
1053  {
1054  p_unpack[unpack_num] = p_code;
1055  }
1056 
1057  // next element
1058  unpack_num++;
1059 
1060  }
1061 
1062  }
1063  else
1064  {
1065  if ((p_code[3] & 0x7F) == 0x14) // mscal
1066  {
1067  seenMSCAL = true;
1068  }
1069  }
1070 
1071  // step to next vifcode
1072  p_code = vif::NextCode(p_code);
1073 
1074  // have we found all 5 unpacks of a vertex packet?
1075  if (unpack_num==5)
1076  {
1077  // perform compression...
1078 
1079  // set element pointers
1080  p_texcrds_source = p_texcrds_dest = (uint32 *)(p_unpack[0]+4);
1081  p_weights_source = p_weights_dest = (uint8 *)(p_unpack[1]+4);
1082  p_normal_source = p_normal_dest = (uint16 *)(p_unpack[2]+4);
1083  p_colour_source = p_colour_dest = (uint32 *)(p_unpack[3]+4);
1084  p_coords_source = p_coords_dest = (uint32 *)(p_unpack[4]+4);
1085 
1086  // set datasize for tex coords
1087  texcrds_size = ((p_unpack[0][3] & 0x07) == 0x04) ? 2 : 1;
1088 
1089  // loop over source vertices
1090  num_squeezed = 0;
1091  for (i=0; i<nloop; i++)
1092  {
1093  // skip if vertex is redundant
1094  if ( (i<=nloop-3) && (p_coords_source[1] & 0x80000000)
1095  && (p_coords_source[3] & 0x80000000)
1096  && (p_coords_source[5] & 0x80000000)
1097  || (i==nloop-2) && (p_coords_source[1] & 0x80000000)
1098  && (p_coords_source[3] & 0x80000000)
1099  || (i==nloop-1) && (p_coords_source[1] & 0x80000000))
1100  {
1101  p_texcrds_source += texcrds_size;
1102  p_weights_source += 3;
1103  p_normal_source += 3;
1104  p_colour_source += 1;
1105  p_coords_source += 2;
1106  num_squeezed++;
1107  }
1108  else if (p_coords_source != p_coords_dest)
1109  // copy vertex
1110  {
1111  *p_texcrds_dest++ = *p_texcrds_source++;
1112  if (texcrds_size==2)
1113  *p_texcrds_dest++ = *p_texcrds_source++;
1114  *p_weights_dest++ = *p_weights_source++;
1115  *p_weights_dest++ = *p_weights_source++;
1116  *p_weights_dest++ = *p_weights_source++;
1117  *p_normal_dest++ = *p_normal_source++;
1118  *p_normal_dest++ = *p_normal_source++;
1119  *p_normal_dest++ = *p_normal_source++;
1120  *p_colour_dest++ = *p_colour_source++;
1121  *p_coords_dest++ = *p_coords_source++;
1122  *p_coords_dest++ = *p_coords_source++;
1123  }
1124  else
1125  // just inc pointers
1126  {
1127  p_texcrds_source += texcrds_size;
1128  p_weights_source += 3;
1129  p_normal_source += 3;
1130  p_colour_source += 1;
1131  p_coords_source += 2;
1132  p_texcrds_dest += texcrds_size;
1133  p_weights_dest += 3;
1134  p_normal_dest += 3;
1135  p_colour_dest += 1;
1136  p_coords_dest += 2;
1137  }
1138  }
1139 
1140  // reduce nloop
1141  nloop -= num_squeezed;
1142 
1143  // make sure there are at least 2 vertices left
1144  if (nloop < 2)
1145  {
1146  num_squeezed -= 2-nloop;
1147  p_texcrds_dest += (2-nloop) * texcrds_size;
1148  p_weights_dest += (2-nloop) * 3;
1149  p_normal_dest += (2-nloop) * 3;
1150  p_colour_dest += (2-nloop) * 1;
1151  p_coords_dest += (2-nloop) * 2;
1152  nloop = 2;
1153  }
1154 
1155  // adjust SIZE fields in unpacks
1156  for (i=0; i<5; i++)
1157  {
1158  p_unpack[i][2] = nloop;
1159  }
1160 
1161  // pad the dead space after each unpack with vif NOPs
1162  for (i=0; i<num_squeezed; i++)
1163  {
1164  *p_texcrds_dest++ = 0;
1165  if (texcrds_size==2)
1166  *p_texcrds_dest++ = 0;
1167  *p_weights_dest++ = 0;
1168  *p_weights_dest++ = 0;
1169  *p_weights_dest++ = 0;
1170  *p_normal_dest++ = 0;
1171  *p_normal_dest++ = 0;
1172  *p_normal_dest++ = 0;
1173  *p_colour_dest++ = 0;
1174  *p_coords_dest++ = 0;
1175  *p_coords_dest++ = 0;
1176  }
1177 
1178  // adjust NLOOP and SIZE in giftag
1179  p_giftag[0] = p_giftag[0] & 0xFFFF8000 | nloop;
1180  p_giftag[3] = nloop*nreg;
1181 
1182  // accumulate savings
1183  addr_diff += num_squeezed * 5;
1184 
1185  // reset for next vertex packet
1186  unpack_num = -1;
1187  seenMSCAL = false;
1188  }
1189 
1190  }
1191 
1192  // adjust vu-mem size in dma tag
1193  vumem_size = ((uint16 *)pTag)[1];
1194  vumem_size -= addr_diff;
1195  vumem_size &= 0x3FF;
1196  ((uint16 *)pTag)[1] = ((uint16 *)pTag)[1] & 0xFC00 | vumem_size;
1197
