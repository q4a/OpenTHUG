105 bool normal_sensitive_collision( CFeeler& feeler, float normal_y_sensitivity )
106 {
107  // Safety check.
108  int iteration = 0;
109 
110  // The feeler should already have been set up with the start and end points.
111  while( feeler.GetCollision())
112  {
113  // If this collision is with a normal that passes ouyr sensitivity test, consider it valid.
114  if( Mth::Abs( feeler.GetNormal()[Y] ) < normal_y_sensitivity )
115  {
116  return true;
117  }
118 
119  // Otherwise move the start location on by 0.1 inch and continue.
120  feeler.m_start += ( feeler.m_end - feeler.m_start ).Normalize( 0.1f );
121 
122  ++iteration;
123  if( iteration > 8 )
124  {
125  Dbg_Message( "Multiple iterations in normal_sensitive_collision()" );
126 
127  // Err on the side of caution.
128  return true;
129  }
130  }
131  return false;
132 }

179 CNavNode* get_nearest_visible_node( Mth::Vector& pos, Mth::Vector& heading )
180 {
181  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
182  Obj::CNavManager* p_nav_man = skate_mod->GetNavManager();
183 
184  float best_dp = -0.707106f; // cosine of 135 degrees
185  CNavNode* p_best_node = NULL;
186 
187  int num_nodes = p_nav_man->GetNumNodes();
188  for( int i = 0; i < num_nodes; ++i )
189  {
190  CNavNode* p_node = p_nav_man->GetNavNodeByIndex( i );
191 
192  // Discard nodes that are not in the required direction.
193  Mth::Vector node_pos = p_node->GetPos() + Mth::Vector( 0.0f, 60.0f, 0.0f, 0.0f );
194  Mth::Vector pos_to_node = node_pos - pos;
195  pos_to_node[Y] = 0.0f;
196  pos_to_node[W] = 0.0f;
197  pos_to_node.Normalize();
198  float dp = Mth::DotProduct( pos_to_node, heading );
199 
200  // Do we already have a better node.
201  if( dp < best_dp )
202  {
203  continue;
204  }
205 
206  // Discard nodes that are not visible.
207  CFeeler feeler;
208  feeler.m_start = pos;
209  feeler.m_end = node_pos;
210  if( normal_sensitive_collision( feeler, 0.2f ))
211  {
212  continue;
213  }
214 
215  best_dp = dp;
216  p_best_node = p_node;
217  }
218  return p_best_node;
219 }

227 CNavNode* CalculateNodePath( Mth::Vector pos, Mth::Vector target_pos )
228 {
229  // Clear open and closed nodes.
230  openList.Clear();
231  closedList.Clear();
232 
233  // Find nearest visible node from current location, preferably in direction currently facing.
234  Mth::Vector heading = target_pos - pos;
235  heading[Y] = 0.0f;
236  heading[W] = 0.0f;
237  heading.Normalize();
238  CNavNode* p_nearest_visible = get_nearest_visible_node( pos, heading );
239  if( p_nearest_visible == NULL )
240  {
241  return false;
242  }
243 
244  // Initialise the A* section of the node.
245  p_nearest_visible->GetAStarNode()->m_cost_from_start = path_cost_estimate( pos, p_nearest_visible->GetPos());
246  p_nearest_visible->GetAStarNode()->m_cost_to_goal = path_cost_estimate( p_nearest_visible->GetPos(), target_pos );
247  p_nearest_visible->GetAStarNode()->m_total_cost = p_nearest_visible->GetAStarNode()->m_cost_from_start + p_nearest_visible->GetAStarNode()->m_cost_to_goal;
248  p_nearest_visible->GetAStarNode()->mp_parent = NULL;
249 
250  // Push this node on to the open list.
251  openList.Push( p_nearest_visible );
252 
253  while( !openList.IsEmpty())
254  {
255  // Pop lowest total cost node from open list.
256  CNavNode* p_this_node = openList.Pop();
257 
258  // If goal visible from this node, we're done.
259  Mth::Vector raised_node_pos = p_this_node->GetPos() + Mth::Vector( 0.0f, 60.0f, 0.0f, 0.0f );
260  CFeeler feeler;
261  feeler.m_start = raised_node_pos;
262  feeler.m_end = target_pos;
263  if( !normal_sensitive_collision( feeler, 0.2f ))
264  {
265  // Build list of nodes.
266 
267  // Empty lists.
268  openList.Empty();
269  closedList.Empty();
270 
271  return p_this_node;
272  }
273 
274  // For each node linked to this node...
275  for( int l = 0; l < p_this_node->GetNumLinks(); ++l )
276  {
277  CNavNode* p_new_node = p_this_node->GetLink( l );
278  float new_cost = p_this_node->GetAStarNode()->m_cost_from_start + traverse_cost( p_this_node, p_new_node );
279 
280  // Ignore this node if it is already in a list and provides no improvement.
281  if(( p_new_node->GetAStarNode()->InOpen() || p_new_node->GetAStarNode()->InClosed()) &&
282  p_new_node->GetAStarNode()->m_cost_from_start <= new_cost )
283  {
284  continue;
285  }
286 
287  // Store the new or improved information.
288  p_new_node->GetAStarNode()->mp_parent = p_this_node;
289  p_new_node->GetAStarNode()->m_cost_from_start = new_cost;
290  p_new_node->GetAStarNode()->m_cost_to_goal = path_cost_estimate( p_new_node->GetPos(), target_pos );
291  p_new_node->GetAStarNode()->m_total_cost = p_new_node->GetAStarNode()->m_cost_from_start + p_new_node->GetAStarNode()->m_cost_to_goal;
292 
293  // Remove new node from closed list if it's in there.
294  if( p_new_node->GetAStarNode()->InClosed())
295  {
296  closedList.Remove( p_new_node );
297  }
298 
299  // Push new node onto open list (if it's not already in there).
300  // This will re-sort the open list.
301  if( !p_new_node->GetAStarNode()->InOpen())
302  {
303  openList.Push( p_new_node );
304  }
305  }
306 
307  // Add node onto the closed list.
308  closedList.Add( p_this_node );
309  }
310 
311  // Empty lists.
312  openList.Empty();
313  closedList.Empty();
314 
315  // If we get here we have failed to find any valid path.
316  return NULL;
317 }

652

654 
655  // First iterate over the node array, and count the number of nodes needed.
656  m_num_nodes = 0;
657  for( i = 0; i < p_nodearray->GetSize(); ++i )
658  {
659  Script::CStruct* p_node_struct = p_nodearray->GetStructure( i );
660  Dbg_MsgAssert( p_node_struct, ( "Error getting node from node array for navigation" ));
661  if( !skate_mod->ShouldBeAbsentNode( p_node_struct ))
662  {
663  uint32 class_checksum = 0;
664  p_node_struct->GetChecksum( CRCD( 0x12b4e660, "Class" ), &class_checksum );
665  if( class_checksum == CRCD( 0x4c23a77e, "Waypoint" ))
666  {
667  uint32 type_checksum = 0;
668  p_node_struct->GetChecksum( CRCD( 0x7321a8d6, "Type" ), &type_checksum );
669  if( type_checksum == CRCD( 0x9784feb5, "HorseNav" ))
670  {
671  ++m_num_nodes;
672  }
673  }
674  }
675  }
676 
677  // No nodes found, so just return
678  if( m_num_nodes == 0 )
679  {
680  return;
681  }
682 
683  mp_nodes = (CNavNode*)Mem::Malloc( m_num_nodes * sizeof( CNavNode ));
684 
685  // Iterate over the nodes and add them to the array.
686  m_current_node = 0;
687 
688  for( i = 0; i < p_nodearray->GetSize(); ++i )
689  {
690  Script::CStruct *p_node_struct = p_nodearray->GetStructure( i );
691  Dbg_MsgAssert( p_node_struct, ("Error getting node from node array for navigation" ));
692 
693  if( !skate_mod->ShouldBeAbsentNode( p_node_struct ))
694  {
695  uint32 class_checksum = 0;
696  p_node_struct->GetChecksum( CRCD( 0x12b4e660, "Class" ), &class_checksum );
697  if( class_checksum == CRCD( 0x4c23a77e, "Waypoint" ))
698  {
699  uint32 type_checksum = 0;
700  p_node_struct->GetChecksum( CRCD( 0x7321a8d6, "Type" ), &type_checksum );
701  if( type_checksum == CRCD( 0x9784feb5, "HorseNav" ))
702  {
703  AddNavNode( i, p_node_struct );
704  ++m_current_node;
705  }
706  }
707  }
708  }
709 
710  // We are creating a table of all nodes, and the pointer to the CNavNode for that node, so we can do a reverse lookup.
711  int num_nodes = p_nodearray->GetSize();
712  CNavNode **pp_navnodes = (CNavNode**)Mem::Malloc( num_nodes * sizeof( CNavNode* ));
713  for( int i = 0; i < num_nodes; ++i )
714  {
715  pp_navnodes[i] = NULL;
716  }
717 
718  // Now fill it in.
719  CNavNode* p_node;
720  for( int node = 0; node < m_num_nodes; ++node )
721  {
722  p_node = &mp_nodes[node];
723  pp_navnodes[p_node->GetNode()] = p_node;
724  }
725 
726  // Now go through all the nodes.
727  for( int node = 0; node < m_num_nodes; ++node )
728  {
729  p_node = &mp_nodes[node];
730  for( int i = 0; i < p_node->m_num_links; ++i )
731  {
732  int index = (int)p_node->mp_links[i];
733 
734  Dbg_MsgAssert( index < num_nodes, ( "Node %d, Nav link node (%d) out of range (0 .. %d). Bad Node array?", p_node->m_node, index, num_nodes ));
735  Dbg_MsgAssert( pp_navnodes[index], ( "NavNode %d linked to something (node %d) that is not a NavNode", p_node->m_node, index ));
736 
737  p_node->mp_links[i] = pp_navnodes[index];
738  }
739  }
740 
741  // Don't need the temporary array anymore.
742  Mem::Free( pp_navnodes );
743 }
