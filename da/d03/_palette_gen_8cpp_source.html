233 assignindex(OctNode * root, Image::RGBA * origin, int depth, box * region,
234  int palIndex)
235 {
236  int width, dr, dg, db, da, dR, dG, dB, dA;
237  int i;
238 
239  Dbg_Assert(origin);
240  Dbg_Assert(region);
241 
242  if (!root)
243  return;
244 
245  width = 1 << depth;
246 
247  dr = origin->r - region->col1.r;
248  dg = origin->g - region->col1.g;
249  db = origin->b - region->col1.b;
250  da = origin->a - region->col1.a;
251  if (dr >= 0 || dg >= 0 || db >= 0 || da >= 0)
252  {
253  return;
254  }
255 
256  dR = region->col0.r - origin->r;
257  dG = region->col0.g - origin->g;
258  dB = region->col0.b - origin->b;
259  dA = region->col0.a - origin->a;
260  if (dR >= width || dG >= width || dB >= width || dA >= width)
261  {
262  return;
263  }
264 
265  /* wholly inside region and a leaf? */
266  if (dr <= -width && dg <= -width && db <= -width && da <= -width)
267  if (dR <= 0 && dG <= 0 && dB <= 0 && dA <= 0)
268  if (depth == 0)
269  {
270  root->Leaf.palIndex = (unsigned char) palIndex;
271  return;
272  }
273 
274  /* try children */
275  depth--;
276  for (i = 0; i < 16; i++)
277  {
278  Image::RGBA suborigin;
279 
280  suborigin.r = origin->r + (((i >> 3) & 1) << depth);
281  suborigin.g = origin->g + (((i >> 2) & 1) << depth);
282  suborigin.b = origin->b + (((i >> 1) & 1) << depth);
283  suborigin.a = origin->a + (((i >> 0) & 1) << depth);
284 
285  assignindex(root->Branch.dir[i], &suborigin, depth, region, palIndex);
286  }

410 /* We want to minimize the sum of the variances of two subboxes.
411  * The sum(c^2) terms can be ignored since their sum over both subboxes
412  * is the same (the sum for the whole box) no matter where we split.
413  * The remaining terms have a minus sign in the variance formula,
414  * so we drop the minus sign and MAXIMIZE the sum of the two terms.
415  */
416 static float
417 nMaximize(OctNode * root, box * cube, int dir, int * cut,
418  LeafNode * whole)
419 {
420  box infcube;
421  LeafNode left, right;
422  float maxsum, val, lastval;
423  int i;
424 
425  Dbg_Assert(root);
426  Dbg_Assert(cube);
427  Dbg_Assert(cut);
428  Dbg_Assert(whole);
429 
430  lastval = maxsum = 0.0f;
431  *cut = -1;
432  infcube = *cube;
433  switch (dir)
434  {
435  case RED:
436  for (i = cube->col0.r; i < cube->col1.r; i++)
437  {
438  infcube.col1.r = (unsigned char) i;
439 
440  nVol(&left, root, &infcube);
441  ColorRealSub(&right.ac, &whole->ac, &left.ac);
442  right.weight = whole->weight - left.weight;
443  if ((left.weight > 0.0f) && (right.weight > 0.0f))
444  {
445  val = ColorRealLengthSq(&left.ac) / left.weight;
446  val += ColorRealLengthSq(&right.ac) / right.weight;
447 
448  if (val > maxsum)
449  {
450  maxsum = val;
451  *cut = i;
452  }
453  else if (val < lastval)
454  {
455  /* we've past the peak */
456  break;
457  }
458 
459  lastval = val;
460  }
461  }
462  break;
463 
464  case GREEN:
465  for (i = cube->col0.g; i < cube->col1.g; i++)
466  {
467  infcube.col1.g = (unsigned char) i;
468 
469  nVol(&left, root, &infcube);

472  if ((left.weight > 0.0f) && (right.weight > 0.0f))

475  val += ColorRealLengthSq(&right.ac) / right.weight;
476 
477  if (val > maxsum)

492
