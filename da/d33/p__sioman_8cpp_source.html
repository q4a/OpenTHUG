123 Manager::Manager ( void )
124 {
125  int i, index;
126 
127  XDEVICE_PREALLOC_TYPE xdpt[] = {{ XDEVICE_TYPE_GAMEPAD, 4 },
128  { XDEVICE_TYPE_MEMORY_UNIT, 1 }};
129 
130  // Initialize the peripherals.
131  XInitDevices( sizeof( xdpt ) / sizeof( XDEVICE_PREALLOC_TYPE ), xdpt );
132 
133  // Create the keyboard queue.
134 // XInputDebugInitKeyboardQueue( &xdkp );
135 
136  m_process_devices_task = new Tsk::Task< DeviceList > ( Manager::process_devices, m_devices );
137 
138  // Pause briefly here to give the system time to enumerate the attached devices.
139  Sleep( 1000 );
140 
141  index = 0;
142  for( i = 0; i < SIO::vMAX_PORT; i++ )
143  {
144  Device* p_device;
145 
146  if(( p_device = create_device( index, i, 0 )))
147  {
148  m_devices.AddToTail( p_device->m_node );
149  p_device->Acquire();
150  index++;
151  }
152  }
153 
154  if( !Pcm::NoMusicPlease())
155  {
156  Pcm::Init();
157  }
158 
159  Dbg_Message( "Initialized Controller lib\n" );
160 }
