188 static int snAddToThreadInfo(int iThreadId)
189 {
190  /* Checks that iThreadId is not already stored in s_ThreadInfo, then */
191  /* looks for a free entry and stores it in there. */
192  /* If there is no space to store iThreadId, returns -1, else returns */
193  /* the index where iThreadId is stored in s_ThreadInfo */
194 
195  int iInterruptsWereEnabled;
196  int iThreadInfoIndex;
197 
198  /* Do this with interrupts disabled to serialise access */
199  iInterruptsWereEnabled = DI();
200 
201  /* If this thread is already registered print a warning and return ok */
202  iThreadInfoIndex = snGetThreadInfoIndexForThreadId(iThreadId);
203  if (iThreadInfoIndex >= 0)
204  {
205  if(iInterruptsWereEnabled != 0)
206  {
207  EI();
208  }
209  printf("Warning: snAddToThreadInfo() was already added\n");
210  return iThreadInfoIndex;
211  }
212 
213  /* Not already registered, so try and find a spare entry to use */
214  iThreadInfoIndex = snGetThreadInfoIndexFreeEntry();
215 
216  /* If the list is full, print error message and return error */
217  if (iThreadInfoIndex < 0)
218  {
219  if(iInterruptsWereEnabled != 0)
220  {
221  EI();
222  }
223  printf("Error: snAddToThreadInfo() list is full\n");
224  return -1;
225  }
226 
227  /* Store iThreadId in s_ThreadInfo */
228  s_ThreadInfo[iThreadInfoIndex].iThreadId = iThreadId;
229  s_ThreadInfo[iThreadInfoIndex].iErrno = 0;
230  s_ThreadInfo[iThreadInfoIndex].iHerrno = 0;
231 
232  if(iInterruptsWereEnabled != 0)
233  {
234  EI();
235  }
236  return iThreadInfoIndex;
237 }

291 static int snConvertNdkErrToNetGlueErr(int iSnErrno)
292 {
293  /* Converts NDK error number in iSnErrno into the closest NetGlue error */
294  /* number and returns it */
295 
296  int iRetval = 0;
297 
298  switch(iSnErrno)
299  {
300  case ENOBUFS: /* No memory buffers are available */
301  iRetval = NG_ENOMEM; /* Insufficient memory */
302  break;
303 
304  case ETIMEDOUT: /* Operation timed out */
305  iRetval = NG_ETIMEDOUT; /* Timeout */
306  break;
307 
308  case EISCONN: /* The socket is already connected */
309  iRetval = NG_EISCONN; /* Connection has already been made */
310  break;
311 
312  case EOPNOTSUPP: /* Operation not supported */
313  iRetval = NG_EOPNOTSUPP; /* Invalid call for socket */
314  break;
315 
316  case ECONNABORTED: /* The connection was aborted */
317  iRetval = NG_ECONNABORTED; /* Aborted by call to sceGlueAbort() */
318  break;
319 
320  case EWOULDBLOCK: /* Caller would block */
321  iRetval = -1; /* Shouldn't happen in blocking mode */
322  break;
323 
324  case ECONNREFUSED: /* The connection was refused */
325  iRetval = NG_ECONNREFUSED; /* Connection request was refused */
326  break;
327 
328  case ECONNRESET: /* The connection has been reset */
329  iRetval = NG_ECONNRESET; /* Connection was reset */
330  break;
331 
332  case ENOTCONN: /* The socket is not connected */
333  iRetval = NG_ENOTCONN; /* Connection does not exist */
334  break;
335 
336  case EALREADY: /* Operation is already in progress */
337  iRetval = -1; /* Shouldn't happen in blocking mode */
338  break;
339 
340  case EINVAL: /* Invalid parameter */
341  iRetval = NG_EINVAL; /* Invalid arguments specified */
342  break;
343 
344  case EMSGSIZE: /* Invalid message size */
345  iRetval = -1; /* Shouldn't happen (not using fn's) */
346  break;
347 
348  case EPIPE: /* Cannot send any more */
349  /* Closest NG error code is ... */
350  iRetval = NG_ENOMEM; /* Insufficient memory */
351  break;
352 
353  case EDESTADDRREQ: /* Destination address is missing */
354  /* Closest NG error code is ... */
355  iRetval = NG_EINVAL; /* Invalid arguments specified */
356  break;
357 
358  case ESHUTDOWN: /* Connection has been shut down */
359  /* Closest NG error code is ... */
360  iRetval = NG_ENOTCONN; /* Connection does not exist */
361  break;
362 
363  case ENOPROTOOPT: /* Option unknown for this protocol */
364  /* Closest NG error code is ... */
365  iRetval = NG_EINVAL; /* Invalid arguments specified */
366  break;
367 
368  case EHAVEOOB: /* Have received out-of-band data */
369  iRetval = -1; /* Shouldn't happen */
370  break;
371 
372  case ENOMEM: /* No memory available */
373  iRetval = NG_ENOMEM; /* Insufficient memory */
374  break;
375 
376  case EADDRNOTAVAIL: /* Specified address not available */
377  iRetval = NG_EADDRNOTAVAIL;/* Invalid address specified */
378  break;
379 
380  case EADDRINUSE: /* Specified address already in use */
381  iRetval = NG_EADDRINUSE; /* Tried bind to port already bound */
382  break;
383 
384  case EAFNOSUPPORT: /* Address family is not supported */
385  iRetval = NG_EAFNOSUPPORT; /* Address family is an unsupported */
386  break;
387 
388  case EINPROGRESS: /* Operation is in progress */
389  iRetval = -1; /* Shouldn't happen in blocking mode */
390  break;
391 
392  case ELOWER: /* Unused */
393  iRetval = -1; /* Shouldn't happen */
394  break;
395 
396  case EBADF: /* Socket descriptor is invalid */
397  iRetval = NG_EBADF; /* Invalid descriptor specified */
398  break;
399 
400  case SN_ENOTINIT: /* Socket API not initialised */
401  /* Closest NG error code is ... */
402  iRetval = NG_EBUSY; /* Library not yet available */
403  break;
404 
405  case SN_ETHNOTREG: /* Thread not registered with API */
406  /* Closest NG error code is ... */
407  iRetval = NG_EINVAL; /* Invalid arguments specified */
408  break;
409

411  /* Closest NG error code is ... */
412  iRetval = NG_EINVAL; /* Invalid arguments specified */
413  break;

422  iRetval = NG_ENOMEM; /* Insufficient memory */
423  break;
424 
425  case SN_EBINDFAIL: /* sceSifBindRpc failed */
426  /* Closest NG error code is ... */
427  iRetval = NG_EBUSY; /* Library not yet available */
428  break;
429

483  iRetval = -1; /* Shouldn't happen */
484  }
485 
486  /* If iRetval is set to -1 (unexpected error) has to be worth printing */
487  /* a debug error message to indicate what the SN error code was */
488 
489  if (iRetval == -1)
490  {
491  printf("Warning: snConvertNdkErrToNetGlueErr(%d) unexpected\n",iSnErrno);
492  iRetval = NG_UNSPECIFIED;
493  }
494 
495  return iRetval;

540 static void snGetConvStoreDnsError(void)
541 {
542  /* This should only be called when an SN NDK function that results in */
543  /* an error that can be read by sn_h_errno() */
544  /* */
545  /* Get the error number from sn_h_errno() and convert to a netglue */
546  /* error code and store it in s_ThreadInfo[].iHerrno for this thread id */
547 
548  int iSnHerrno;
549  int iNgHerrno = 0;
550 
551  /* Get the NDK error code */
552  iSnHerrno = sn_h_errno();
553 
554  /* Convert it to a netglue error code */
555  switch (iSnHerrno)
556  {
557  /* NDK does not support the standard BSD DNS error codes: */
558  /* HOST_NOT_FOUND, TRY_AGAIN, NO_RECOVERY, NO_DATA, NO_ADDRESS, all */
559  /* of these conditions result in NDK returning the following error */
560 
561  case SN_EDNSFAIL:
562  iNgHerrno = NG_HOST_NOT_FOUND; /* not sure what else I can do */
563  break;
564 
565  /* Internal errors in the TCP/IP stack state, or, invalid input */
566  /* can result in the following errors: */
567  /* SN_ESTKDOWN The TCP/IP Stack has not been started */
568  /* EINVAL name was too long or was a NULL pointer */
569 
570  default:
571  printf("Warning: snGetConvStoreDnsError(%d) unexpected\n",iSnHerrno);
572  iNgHerrno = NG_NETDB_INTERNAL;
573  /* The sony comment for the above Herrno says "see errno", so I */
574  /* I assume I should store something in the errno. */
575  snStoreErrno(NG_UNSPECIFIED);
576  }
577 
578  /* Store the Herrno in s_ThreadInfo[].iHerrno for this thread */
579  snStoreHerrno(iNgHerrno);
580 }

1302 int sceNetGlueThreadInit(int thread_id)
1303 {
1304  int iResult;
1305  int iDontCare;
1306  int iThisThreadId;
1307 
1308  iThisThreadId = GetThreadId();
1309 
1310  /* thread_id == 0 means this thread */
1311  if (thread_id == 0)
1312  {
1313  thread_id = iThisThreadId;
1314  }
1315 
1316  /* NDK requires each thread that is going to use the socket API to */
1317  /* register *itself* with the NDK API by calling sockAPIregthr() i.e. */
1318  /* there is currently no way for thread A to register thread B */
1319  /* so trap this and report error. */
1320 
1321  /* If this restriction turns out to be a problem in practice then it */
1322  /* will be necessary for NDK to be modified so that thread A can */
1323  /* register thread B e.g. add new fn sockAPIregother() */
1324 
1325  /* so trap this and report error. */
1326 
1327  if (iThisThreadId != thread_id)
1328  {
1329  printf("Error: sceNetGlueThreadInit() can only register own thread\n");
1330  return -1;
1331  }
1332 
1333  /* Add this to our internal list of registered threads */
1334  iResult = snAddToThreadInfo(thread_id);
1335 
1336  /* If it failed (list is full), return now with failure */
1337  if (iResult < 0)
1338  {
1339  return -1;
1340  }
1341 
1342  /* Now register with NDK */
