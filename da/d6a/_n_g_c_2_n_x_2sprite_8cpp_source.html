110 void SDraw2D::InsertDrawList()
111 {
112  if (!sp_2D_draw_list || // Empty List
113  (m_pri <= sp_2D_draw_list->m_pri)) // Start List
114  {
115  mp_next = sp_2D_draw_list;
116  sp_2D_draw_list = this;
117  } else { // Insert
118  SDraw2D *p_cur = sp_2D_draw_list;
119 
120  // Find where to insert
121  while(p_cur->mp_next)
122  {
123  if (m_pri <= p_cur->mp_next->m_pri)
124  break;
125 
126  p_cur = p_cur->mp_next;
127  }
128 
129  // Insert at this point
130  mp_next = p_cur->mp_next;
131  p_cur->mp_next = this;
132  }
133 }
