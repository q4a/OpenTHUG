72 void RotateY90( ERot90 angle, int32& x, int32& y, int32& z )
73 {
74  int32 temp;
75 
76  switch (angle)
77  {
78  case ROT_0:
79  break;
80 
81  case ROT_90:
82  temp = x;
83  x = z;
84  z = -temp;
85  break;
86 
87  case ROT_180:
88  x = -x;
89  z = -z;
90  break;
91 
92  case ROT_270:
93  temp = x;
94  x = -z;
95  z = temp;
96  break;
97 
98  default:
99  Dbg_MsgAssert(0, ("RotateY90() out of range: %d", angle));
100  break;
101  }
102 }
