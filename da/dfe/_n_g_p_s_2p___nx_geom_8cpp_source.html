85 bool CPs2Geom::plat_load_geom_data(CMesh* pMesh, CModel* pModel, bool color_per_material)
86 {
87  int numBones;
88  numBones = pModel->GetNumBones();
89 
90  Mth::Matrix temp;
91  temp.Identity();
92  Dbg_Assert( mp_oldInstance == NULL );
93 
94  // attempting to move skins over to .geom.ps2
95  mp_matrices = ((CPs2Model*)pModel)->GetMatrices();
96  if ( ((CPs2Mesh*)pMesh)->GetGeomNode() )
97  {
98  if ( numBones )
99  {
100  mp_instance = ((CPs2Mesh*)pMesh)->GetGeomNode()->CreateInstance( &m_rootMatrix, numBones, (Mth::Matrix *)mp_matrices );
101  }
102  else
103  {
104  mp_instance = ((CPs2Mesh*)pMesh)->GetGeomNode()->CreateInstance( &m_rootMatrix );
105  }
106  }
107  else
108  {
109  Dbg_Assert( pMesh );
110  Dbg_Assert( pMesh->GetTextureDictionary() );
111  NxPs2::sScene* pScene = ((Nx::CPs2TexDict*)pMesh->GetTextureDictionary())->GetEngineTextureDictionary();
112  Dbg_Assert( pScene );
113 
114  if ( numBones )
115  {
116  mp_oldInstance = new NxPs2::CInstance(pScene, temp, color_per_material, numBones, mp_matrices );
117  }
118  else
119  {
120  mp_oldInstance = new NxPs2::CInstance(pScene, temp, color_per_material);
121  }
122  }
123 
124  // remember the source mesh, so that we can do poly-hiding on it
125  mp_sourceMesh = (CPs2Mesh*)pMesh;
126 
127  return true;
128 }

182 CGeom * CPs2Geom::plat_clone(bool instance, CModel* pDestModel)
183 {
184  Dbg_MsgAssert(mp_instance == NULL, ("Wrong version of CPs2Geom::plat_clone() for CGeomNodes"));
185  Dbg_Assert(mp_oldInstance);
186 
187  // Copy into new geom
188  CPs2Geom *p_new_geom = new CPs2Geom(*this);
189 
190  Dbg_Assert(p_new_geom);
191 
192  int numBones;
193  numBones = pDestModel->GetNumBones();
194 
195  Mth::Matrix temp;
196  temp.Identity();
197 
198  p_new_geom->mp_matrices = ((CPs2Model*)pDestModel)->GetMatrices();
199 
200  NxPs2::sScene* pScene = mp_oldInstance->GetScene();
201  Dbg_Assert( pScene );
202 
203  if ( numBones )
204  {
205  p_new_geom->mp_oldInstance = new NxPs2::CInstance(pScene, temp, mp_oldInstance->HasColorPerMaterial(), numBones, p_new_geom->mp_matrices );
206  }
207  else
208  {
209  p_new_geom->mp_oldInstance = new NxPs2::CInstance(pScene, temp, mp_oldInstance->HasColorPerMaterial());
210  }
211 
212  Dbg_Assert(p_new_geom->mp_oldInstance);
213 
214  // Copy the colors from instance
215  if (mp_oldInstance->HasColorPerMaterial())
216  {
217  int num_colors = mp_oldInstance->GetScene()->NumMeshes;
218  for (int i = 0; i < num_colors; i++)
219  {
220  p_new_geom->mp_oldInstance->SetMaterialColorByIndex(i, mp_oldInstance->GetMaterialColorByIndex(i));
221  }
222  }
223  else
224  {
225  p_new_geom->mp_oldInstance->SetColor(mp_oldInstance->GetColor());
226  }
227 
228  return p_new_geom;
229 }
