117 
118  // Convert texture to 32-bit, if not already (speeds conversion)
119  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().BottomUpHeap());
120  p_face_texture->Generate32BitImage();
121  Mem::Manager::sHandle().PopContext();
122 
123  // Adjust the texture based on the model face points

130  if (sp_face_texture_overlay)

132  p_face_texture->CombineTextures(sp_face_texture_overlay, palette_gen);
133  }
134 
135  // Put 32-bit image into texture

137 
138 #if PRINT_TIMES
139  uint32 end_time = Tmr::GetTimeInUSeconds();
140  Dbg_Message("sMassageTexture Update time %d us", end_time - start_time);
141 #endif

143  return true;

151 bool CFaceTexMassager::sAdjustTextureToModel(CTexture *p_face_texture, const SFacePoints &c_texture_points,
152  bool use_fill_color, Image::RGBA fill_color)
153 {
154  Dbg_Assert(p_face_texture);
155  Dbg_MsgAssert(p_face_texture->GetWidth() == s_model_face_points.m_texture_width, ("Face texture width is not equal to the face points width"));
156  Dbg_MsgAssert(p_face_texture->GetHeight() == s_model_face_points.m_texture_height, ("Face texture height is not equal to the face points height"));
157 
158  // Make a local copy of the texture points, as we might need to modify it to make it safe
159  SFacePoints texture_points = c_texture_points;
160 
161 // Make the values "safe" (should also do this at the user interface stage)
162 
163 
164 // if eyes are flipped, then flip them back
165  int t;
166  if (texture_points.m_left_eye[X] > texture_points.m_right_eye[X])
167  {
168  t = texture_points.m_left_eye[X];
169  texture_points.m_left_eye[X] = texture_points.m_right_eye[X];
170  texture_points.m_right_eye[X] = t;
171  }
172 
173 
174 // If Left eye is below the nose then put it above it.
175  if (texture_points.m_left_eye[Y] > texture_points.m_nose[Y])
176  {
177  texture_points.m_left_eye[Y] = texture_points.m_nose[Y]-1;
178  }
179 
180 
181 // If Right eye is below the nose then put it above it.
182  if (texture_points.m_right_eye[Y] >= texture_points.m_nose[Y])
183  {
184  texture_points.m_right_eye[Y] = texture_points.m_nose[Y]-2;
185  }
186 
187 // if Lips is above the nose, then put it below
188  if (texture_points.m_lips[Y] <= texture_points.m_nose[Y])
189  {
190  texture_points.m_lips[Y] = texture_points.m_nose[Y]+2;
191  }
192 
193  // All stretching and shrinking will be done at the nose point
194  int x_axis_point = s_model_face_points.m_nose[X];
195  int y_axis_point = s_model_face_points.m_nose[Y];
196 
197  // First, offset the texture by the difference in nose points
198  int x_offset = s_model_face_points.m_nose[X] - texture_points.m_nose[X];
199  int y_offset = s_model_face_points.m_nose[Y] - texture_points.m_nose[Y];
200  p_face_texture->Offset(x_offset, y_offset);
201 
202  int model_eyeline_y = (s_model_face_points.m_left_eye[Y] + s_model_face_points.m_right_eye[Y]) >> 1;
203  int texture_eyeline_y = ((texture_points.m_left_eye[Y] + texture_points.m_right_eye[Y]) >> 1) + y_offset;
204 
205  // Calculate the ratios to find how much we need to pull/push a texture section
206  float left_eye_to_nose_width = texture_points.m_nose[X] - texture_points.m_left_eye[X];
207  float left_nose_width = texture_points.m_nose[X] + x_offset;
208  float left_eye_ratio = left_nose_width / left_eye_to_nose_width;
209 
210  float right_eye_to_nose_width = texture_points.m_right_eye[X] - texture_points.m_nose[X];
211  float right_nose_width = p_face_texture->GetWidth() - (texture_points.m_nose[X] + x_offset);
212  float right_eye_ratio = right_nose_width / right_eye_to_nose_width;
213 
214  float eyeline_to_nose_height = (texture_points.m_nose[Y] + y_offset) - texture_eyeline_y;
215  float top_nose_height = texture_points.m_nose[Y] + y_offset;
216  float eyeline_ratio = top_nose_height / eyeline_to_nose_height;
217 
218  float lips_to_nose_height = texture_points.m_lips[Y] - texture_points.m_nose[Y];
219  float bottom_nose_height = p_face_texture->GetHeight() - (texture_points.m_nose[Y] + y_offset);
220  float lips_ratio = bottom_nose_height / lips_to_nose_height;
221 
222  // Move the left eye along the X axis
223  int left_eye_pixels = s_model_face_points.m_left_eye[X] - (texture_points.m_left_eye[X] + x_offset);
224  //Dbg_Message("Left eye pixels before %d", left_eye_pixels);
225  //Dbg_Message("Left eye to_nose %f width %f", left_eye_to_nose_width, left_nose_width);
226  left_eye_pixels = (int) ((left_eye_pixels * left_eye_ratio) + 0.5f);
227  //Dbg_Message("Left eye pixels after %d", left_eye_pixels);
228  if (left_eye_pixels < 0)
229  {
230  // Stretch out
231  p_face_texture->PullToEdge(x_axis_point, X, left_eye_pixels);
232  }
233  else
234  {
235  // Shrink
236  p_face_texture->PushToPoint(x_axis_point, X, left_eye_pixels, use_fill_color, fill_color);
237  }
238 
239  // Move the right eye along the X axis
240  int right_eye_pixels = s_model_face_points.m_right_eye[X] - (texture_points.m_right_eye[X] + x_offset);
241  //Dbg_Message("Right eye pixels before %d", right_eye_pixels);
242  //Dbg_Message("Right eye to_nose %f width %f", right_eye_to_nose_width, right_nose_width);
243  right_eye_pixels = (int) ((right_eye_pixels * right_eye_ratio) + 0.5f);
244  //Dbg_Message("Right eye pixels after %d", right_eye_pixels);
245  if (right_eye_pixels >= 0)
246  {
247  // Stretch out
248  p_face_texture->PullToEdge(x_axis_point, X, right_eye_pixels);
249  }
250  else
251  {
252  // Shrink
253  p_face_texture->PushToPoint(x_axis_point, X, right_eye_pixels, use_fill_color, fill_color);
254  }
255 
256  // Move the eyeline along the Y axis
257  int eyeline_pixels = model_eyeline_y - texture_eyeline_y;
258  eyeline_pixels = (int) ((eyeline_pixels * eyeline_ratio) + 0.5f);
259  if (eyeline_pixels < 0)
260  {
261  // Stretch out
262  p_face_texture->PullToEdge(y_axis_point, Y, eyeline_pixels);
263  }
264  else
265  {
266  // Shrink
267  p_face_texture->PushToPoint(y_axis_point, Y, eyeline_pixels, use_fill_color, fill_color);
268  }
269 
270  // Move the lips along the Y axis
271  int lips_pixels = s_model_face_points.m_lips[Y] - (texture_points.m_lips[Y] + y_offset);
272  lips_pixels = (int) ((lips_pixels * lips_ratio) + 0.5f);
273  if (lips_pixels >= 0)
274  {
275  // Stretch out
276  p_face_texture->PullToEdge(y_axis_point, Y, lips_pixels);
277  }
278  else
279  {
280  // Shrink
281  p_face_texture->PushToPoint(y_axis_point, Y, lips_pixels, use_fill_color, fill_color);
282  }
283 
284  return true;
285 }

378 bool GetFacePointsStruct(SFacePoints &face_points, Script::CStruct *p_struct)
379 {
380  Dbg_Assert(p_struct);
381 
382  Script::CArray *p_left_eye = NULL;
383  Script::CArray *p_right_eye = NULL;
384  Script::CArray *p_nose = NULL;
385  Script::CArray *p_lips = NULL;
386 
387  // Left eye
388  p_struct->GetArray( CRCD(0x08bf1d3f,"left_eye"), &p_left_eye );
389  if (p_left_eye)
390  {
391  face_points.m_left_eye[X] = p_left_eye->GetInteger(0);
392  face_points.m_left_eye[Y] = p_left_eye->GetInteger(1);
393  }
394  else
395  {
396  Dbg_MsgAssert(0, ("Couldn't find member left_eye"))
397  return false;
398  }
399 
400  // Right eye
401  p_struct->GetArray( CRCD(0xb0b44396,"right_eye"), &p_right_eye );
402  if (p_right_eye)
403  {
404  face_points.m_right_eye[X] = p_right_eye->GetInteger(0);
405  face_points.m_right_eye[Y] = p_right_eye->GetInteger(1);
406  }
407  else
408  {
409  Dbg_MsgAssert(0, ("Couldn't find member right_eye"))
410  return false;
411  }
412 
413  // Nose
414  p_struct->GetArray( CRCD(0x7f03932c,"nose"), &p_nose );
415  if (p_nose)
416  {
417  face_points.m_nose[X] = p_nose->GetInteger(0);
418  face_points.m_nose[Y] = p_nose->GetInteger(1);
419  }
420  else
421  {
422  Dbg_MsgAssert(0, ("Couldn't find member nose"))
423  return false;
424  }
425 
426  // Lips
427  p_struct->GetArray( CRCD(0x0e7ec187,"lips"), &p_lips );
428  if (p_lips)
429  {
430  face_points.m_lips[X] = p_lips->GetInteger(0);
431  face_points.m_lips[Y] = p_lips->GetInteger(1);
432  }
433  else
434  {
435  Dbg_MsgAssert(0, ("Couldn't find member lips"))
436  return false;
437  }
438 
439  // Grab color adjustment values, if any
440  face_points.m_adjust_hsv = p_struct->GetFloat(CRCD(0x6e94f918,"h"), &face_points.m_h_offset);
441  face_points.m_adjust_hsv = p_struct->GetFloat(CRCD(0xe4f130f4,"s"), &face_points.m_s_scale) && face_points.m_adjust_hsv;
442  face_points.m_adjust_hsv = p_struct->GetFloat(CRCD(0x949bc47b,"v"), &face_points.m_v_scale) && face_points.m_adjust_hsv;
443 
444  if (face_points.m_adjust_hsv)
445  {
446  Dbg_MsgAssert((face_points.m_h_offset >= 0.0f) && (face_points.m_h_offset <= 360.0f), ("h must be in the range of 0-360"));
447  Dbg_MsgAssert(face_points.m_s_scale >= 0.0f, ("s cannot be negative"));
448  Dbg_MsgAssert(face_points.m_v_scale >= 0.0f, ("v cannot be negative"));
449  }
450 
451  // Width and height (some functions may not care, so init them to 0)
452  face_points.m_texture_width = 0;
453  face_points.m_texture_height = 0;
454  p_struct->GetInteger( CRCD(0x73e5bad0,"width"), &face_points.m_texture_width );
455  p_struct->GetInteger( CRCD(0x0ab21af0,"height"), &face_points.m_texture_height );
456 
457  return true;
458 }
