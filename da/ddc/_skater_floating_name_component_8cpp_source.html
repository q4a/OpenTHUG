114 void CSkaterFloatingNameComponent::Update()
115 {
116  if (!GameNet::Manager::Instance()->InNetGame() || !GameNet::Manager::Instance()->ShouldDrawPlayerNames())
117  {
118  Suspend(true);
119  return;
120  }
121 
122  if (!GetSkater()->IsInWorld()) return;
123 
124  GameNet::PlayerInfo* player = GameNet::Manager::Instance()->GetPlayerByObjectID(GetObject()->GetID());
125 
126  float offset;
127  if (GameNet::Manager::Instance()->GetCurrentlyObservedPlayer() == player)
128  {
129  offset = FEET(8.0f);
130  }
131  else
132  {
133  offset = FEET(10.0f);
134  }
135 
136  Script::CStruct* pParams = new Script::CStruct;
137  pParams->AddChecksum(CRCD(0x40c698af, "id"), m_screen_element_id);
138 
139  int color_index;
140  if (Mdl::Skate::Instance()->GetGameMode()->IsTeamGame())
141  {
142  color_index = player->m_Team + 2;
143  }
144  else
145  {
146  color_index = GetObject()->GetID() + 2;
147  }
148 
149  char text[64];
150  sprintf(text, "\\c%d%s", color_index, GetSkater()->GetDisplayName());
151  pParams->AddString(CRCD(0xc4745838, "text"), text);
152  pParams->AddVector(CRCD(0x4b491900, "pos3D"), GetObject()->m_pos[X], GetObject()->m_pos[Y] + offset, GetObject()->m_pos[Z]);
153 
154  Front::CScreenElement *p_name_elem = Front::CScreenElementManager::Instance()->GetElement(m_screen_element_id);
155  if (p_name_elem)
156  {
157  p_name_elem->SetProperties(pParams);
158  }
159  else
160  {
161  Script::RunScript(CRCD(0x6a060cf0, "create_object_label"), pParams);
162  }
163 
164  delete pParams;
165 }
