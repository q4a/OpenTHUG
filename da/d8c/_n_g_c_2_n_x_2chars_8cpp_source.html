80 SFont *LoadFont( const char *Filename )
81 {
82  SFont *pFont;
83  SChar *pChar;
84  uint8 *pData;
85 // int i,Len,NumChars,Width,Height,Depth,NumBytes;
86  int i,NumChars,Width,Height,Depth,NumBytes;
87 
88  char * p8 = (char *)Filename;
89  while ( *p8 != '\0' )
90  {
91  if ( ( *p8 >= 'A' ) && ( *p8 <= 'Z' ) )
92  {
93  *p8 = *p8 - ( 'A' - 'a' );
94  }
95  p8++;
96  }
97  unsigned char * p_fnt = (unsigned char *)get_font_file_address( Filename );
98  Dbg_MsgAssert( p_fnt, ( "Font file '%s' not found in permanent data.", Filename ));
99 
100 #define get_bytes(p,n) { memcpy( p, p_fnt, n ); p_fnt += n; }
101 
102  // Allocate memory for the font structure.
103  pFont = new SFont();
104 
105  // Allocate a temporary buffer.
106  uint8 FontBuf[2048];
107 
108  // Load file header.
109  get_bytes( FontBuf, 16 );
110 // Len = File::Read( FontBuf, 16, 1, p_FH );
111 // Dbg_MsgAssert( Len==16, ( "couldn't read file header from font file %s", Filename ));
112  NumChars = ((uint32 *)FontBuf)[1];
113  pFont->DefaultHeight = ((uint32 *)FontBuf)[2];
114  pFont->DefaultBase = ((uint32 *)FontBuf)[3];
115 
116 #ifdef __PLAT_NGC__
117  // PJR - This needs to disappear once we start converting .ska to .ska.ngc etc.
118  NumChars = nReverse32( NumChars );
119  pFont->DefaultHeight = nReverse32( pFont->DefaultHeight );
120  pFont->DefaultBase = nReverse32( pFont->DefaultBase );
121 #endif // __PLAT_NGC__
122 
123  // Clear character map to zero.
124  memset( pFont->Map, 0, 256 );
125  memset(pFont->SpecialMap, 0, 32);
126 
127  // Allocate memory for character table.
128  pFont->pChars = new SChar[NumChars];
129 
130  // Load character map and character table.
131  get_bytes( FontBuf, NumChars << 2 );
132 // Len = File::Read( FontBuf, NumChars << 2, 1, p_FH );
133 // Dbg_MsgAssert( Len == ( NumChars << 2 ), ( "couldn't read character table in font file %s", Filename ));
134 
135  for( i = 0, pChar = pFont->pChars, pData = FontBuf; i < NumChars; i++,pChar++,pData += 4 )
136  {
137 #ifdef __PLAT_NGC__
138  // PJR - This needs to disappear once we start converting .ska to .ska.ngc etc.
139  pChar->Baseline = nReverse32( pChar->Baseline );
140  pChar->Baseline = nReverse16( ((uint16 *)pData)[0] );
141  sint16 ascii_val = nReverse16(((sint16*) pData)[1]);
142  if (ascii_val >= 0)
143  pFont->Map[(uint8) ascii_val] = i;
144  else
145  {
146  Dbg_Assert(ascii_val >= -32)
147  pFont->SpecialMap[(uint8) (-ascii_val - 1)] = i;
148  }
149 #else
150  pChar->Baseline = ((uint16 *)pData)[0];
151  pFont->Map[(uint8)((uint16*)pData)[1]] = i;
152 #endif // __PLAT_NGC__
153  }
154 
155  // now, if there is a null character in the font, make characters that could not be found
156  // in the font display that instead of 'A'
157  if (pFont->SpecialMap[31] != 0)
158  {
159  for (i = 0; i < 256; i++)
160  {
161  if (pFont->Map[i] == 0 && i != 'A' && i != 'a') pFont->Map[i] = pFont->SpecialMap[31];
162  if (i < 31 && pFont->SpecialMap[i] == 0) pFont->SpecialMap[i] = pFont->SpecialMap[31];
163  }
164  }
165 
166  // Load texture header.
167  get_bytes( FontBuf, 16 );
168 // Len = File::Read( FontBuf, 16, 1, p_FH );
169 // Dbg_MsgAssert( Len == 16, ( "couldn't read texture header from font file %s", Filename ));
170  Width = ((uint16 *)FontBuf)[2];
171  Height = ((uint16 *)FontBuf)[3];
172  Depth = ((uint16 *)FontBuf)[4];
173 
174 #ifdef __PLAT_NGC__
175  // PJR - This needs to disappear once we start converting .ska to .ska.ngc etc.
176  Width = nReverse16( Width );
177  Height = nReverse16( Height );
178  Depth = nReverse16( Depth );
179 #endif // __PLAT_NGC__
180 
181  // Create texture.
182  Dbg_Assert( Depth == 8 );
183 
184  NumBytes = ( Width * Height + 3 ) & 0xFFFFFFFC;
185 
186  // Align texture to 32 bytes & point up.
187  int size = ( Width * Height ) + ( 256 * 2 );
188  unsigned char * p_align = (unsigned char*)OSRoundUp32B( p_fnt );
189  for ( int lp = ( size - 1 ); lp >= 0; lp-- ) p_align[lp] = p_fnt[lp];
190  DCFlushRange( p_align, size );
191  pFont->mp_texture = p_align;
192  pFont->mp_palette = &p_align[( Width * Height )];
193  p_fnt += ( Width * Height ); // Texture
194  p_fnt += ( 256 * 2 ); // Palette
195  p_fnt += 32; // Padding
196 
197 // pFont->mp_texture = new char[ ( Width * Height ) ];
198 // Len = File::Read( pFont->mp_texture, ( Width * Height ), 1, p_FH );
199 // Dbg_MsgAssert( Len == ( Width * Height ), ( "couldn't read texture data from font file %s", Filename ));
200 //
201 // pFont->mp_palette = new char[ ( 256 * 4 ) ];
202 // Len = File::Read( pFont->mp_palette, ( 256 * 4 ), 1, p_FH );
203 // Dbg_MsgAssert( Len == ( 256 * 4 ), ( "couldn't read palette data from font file %s", Filename ));
204 
205  pFont->m_width = Width;
206  pFont->m_height = Height;
207 
208  // Skip numsubtextures, and load subtextures.
209  get_bytes( FontBuf, ( NumChars << 3 ) + 4 );
210 // Len = File::Read( FontBuf, ( NumChars << 3 ) + 4, 1, p_FH );
211 // Dbg_MsgAssert( Len == ( NumChars << 3 ) + 4, ( "couldn't read subtexture table from font file %s", Filename ));
212  for( i = 0, pChar = pFont->pChars, pData = FontBuf + 4; i < NumChars; i++, pChar++, pData += 8 )
213  {
214  uint16 x = ((uint16 *)pData )[0];
215  uint16 y = ((uint16 *)pData )[1];
216  uint16 w = ((uint16 *)pData )[2];
217  uint16 h = ((uint16 *)pData )[3];
218 
219 #ifdef __PLAT_NGC__
220  x = nReverse16( x );
221  y = nReverse16( y );
222  w = nReverse16( w );
223  h = nReverse16( h );
224 #endif // __PLAT_NGC__
225 
226  pChar->w = w;
227  pChar->h = h;
228  pChar->u0 = ( ((float)x + 0.5f) / (float)Width );
229  pChar->v0 = ( ((float)y + 0.5f) / (float)Height );
230  pChar->u1 = pChar->u0 + ((float)w / (float)Width );
231  pChar->v1 = pChar->v0 + ((float)h / (float)Height );
232  }
233 
234  // Add font to font list.
235  pFont->pNext = pFontList;
236  pFontList = pFont;
237 
238  // We're done with the font file now.
239 // File::Close( p_FH );
240 
241  // this will serve as the default spacing
242  pFont->mSpaceSpacing = pFont->pChars[pFont->Map['I']].w;
243 
244  return pFont;

255  if (pFontList == pFont)
