39 CDetailedShadow::CDetailedShadow(Nx::CModel *p_model, Mth::Vector camera_direction, float camera_distance)
40 {
41  m_type=vDETAILED_SHADOW;
42  mp_texture = Nx::CEngine::sCreateRenderTargetTexture( 256, 256, 16, 16 ); // maybe platform dependent?
43 
44  // Note, engine will figure out scene for itself
45  Nx::CEngine::sProjectTextureIntoScene( mp_texture, p_model );
46 
47  m_distance = camera_distance;
48  m_direction = camera_direction;
49  m_direction.Normalize();
50  mp_camera = new Camera();
51  mp_camera->SetPos(Mth::Vector(0,0,0));
52  // construct an orientation matrix for the camera
53  // so it is looking in the required direction
54  Mth::Matrix orient;
55  orient.Ident(); // initialize (since it aint)
56  orient[Z] = m_direction;
57  orient[Z].Normalize();
58 
59  // X and Y are at right angles to Z
60  orient[X][X] = orient[Z][Z];
61  orient[X][Y] = -orient[Z][X];
62  orient[X][Z] = -orient[Z][Y];
63 
64  orient[Y][X] = orient[Z][Y];
65  orient[Y][Y] = -orient[Z][Z];
66  orient[Y][Z] = -orient[Z][X];
67  mp_camera->SetMatrix(orient);
68 
69  // this loops through all the existing shadows,
70  // and sets their shadow flags... the model
71  // also remembers its shadow state, for any
72  // subsequent geoms that are added
73  p_model->EnableShadow( true );
74 }

99 // set the direction in which the projection camera points
100 void CDetailedShadow::UpdateDirection( const Mth::Vector& dir )
101 {
102  // Store camera position.
103  Mth::Vector cam_pos = mp_camera->GetPos();
104 
105  // Set new direction and normalize.
106  m_direction = dir;
107  m_direction.Normalize();
108 
109  // Zero out camera position.
110  mp_camera->SetPos( Mth::Vector( 0, 0, 0 ));
111 
112  // Construct an orientation matrix for the camera so it is looking in the required direction.
113  Mth::Matrix orient;
114  orient.Identity();
115  orient[Z] = m_direction;
116 
117  // X and Y are at right angles to Z.
118  orient[X][X] = orient[Z][Z];
119  orient[X][Y] = -orient[Z][X];
120  orient[X][Z] = -orient[Z][Y];
121  orient[Y][X] = orient[Z][Y];
122  orient[Y][Y] = -orient[Z][Z];
123  orient[Y][Z] = -orient[Z][X];
124  mp_camera->SetMatrix( orient );
125 
126  // Restore camera position.
127  mp_camera->SetPos( cam_pos );
128 
129  // Pass camera details on to engine.
130  Nx::CEngine::sSetProjectionTextureCamera( mp_texture, mp_camera );
131 }

178 void CSimpleShadow::UpdatePosition(Mth::Vector& parentPos, Mth::Matrix& parentMatrix, Mth::Vector normal)
179 {
180  /*
181  Dbg_MsgAssert(mp_poly,("NULL mp_poly ?"));
182  float w=Script::GetFloat("shadowwidth");
183  mp_poly->SetPos(pos,w,w,normal);
184  */

186  Mth::Matrix display_matrix;
187  display_matrix[Y]=normal;
188  display_matrix[Z]=parentMatrix[Z];
189  display_matrix[X]=Mth::CrossProduct(display_matrix[Y],display_matrix[Z]);
190 
191  Mth::Vector scale(m_scale,m_scale,m_scale,1.0f);
192  display_matrix.Scale(scale);
193 
194  // Change requested by LF/Andre: use the passed normal
195  display_matrix.SetPos(parentPos+normal*m_offset); //WAS: parentPos+parentMatrix[Y]*m_offset
196 
197  //Dbg_MsgAssert(mp_model,("NULL mp_model"));
198  if (mp_model)
199  {
200  mp_model->Render(&display_matrix,true,NULL);
