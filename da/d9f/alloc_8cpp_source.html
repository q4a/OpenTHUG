349 // find the block that contains this pointer
350 // return NULL if not found
351 // also includes the header area of the block
352 void * Allocator::find_block(void *p)
353 {
354 
355  BlockHeader* p_header = mp_context->mp_used_list;
356 
357  while ( p_header )
358  {
359 
360  void * p_start = (void*)((uint)p_header + BlockHeader::sSize);
361  int size = p_header->mSize;
362  void * p_end = (void*)((int)p_start + size);
363 
364  if (p >= p_start && p <p_end)
365  {
366  return (void*)p_header;
367  }
368  p_header = p_header->mp_next_used;
369  }
370  return NULL;
371 }
