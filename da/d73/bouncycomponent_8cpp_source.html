82 CBaseComponent::EMemberFunctionResult CBouncyComponent::CallMemberFunction( uint32 Checksum, Script::CScriptStructure *pParams, Script::CScript *pScript )
83 {
84 
85  switch ( Checksum )
86  {
87 
88  // @script | BouncyObj_PlayerCollisionOn | turn player collision on
89  case ( 0xf9055d81 ): // "BouncyObj_PlayerCollisionOn"
90  m_bouncyobj_flags &= ~BOUNCYOBJ_FLAG_PLAYER_COLLISION_OFF;
91  return MF_TRUE;
92 
93 
94  // @script | BouncyObj_PlayerCollisionOff | turn player collision off
95  case ( 0xf64ef88e ): // "BouncyObj_PlayerCollisionOff"
96  m_bouncyobj_flags |= BOUNCYOBJ_FLAG_PLAYER_COLLISION_OFF;
97  return MF_TRUE;
98 
99  // @script | BouncyObj_Go | The vector can be anything you want...
100  // the initial velocity. It's optional, without it the object will
101  // fall from it's current location. If it's already on the ground
102  // probably nothing will happen
103  // @uparmopt (0, 0, 0) | vector
104  case ( 0xfea0e952 ): // "BouncyObj_Go"
105  {
106  Mth::Vector vel;
107  vel.Set( );
108  pParams->GetVector( NONAME, &vel );
109  if ( m_state == BOUNCYOBJ_STATE_BOUNCING )
110  {
111  Dbg_Message( "\n%s\nWarning: Calling BouncyObj_Go when object is already bouncing.", pScript->GetScriptInfo( ) );
112  }
113  if ( pParams->ContainsFlag( 0x26af9dc9 ) ) // "FLAG_MAX_COORDS"
114  {
115  vel.ConvertToMaxCoords( );
116  }
117  bounce( vel );
118  return MF_TRUE;
119  }
120  }
121  return MF_NOT_EXECUTED;
122 }

342 float get_closest_allowable_angle( float current, float allowableAngle )
343 {
344 
345  float ang;
346  float halfAllowable = allowableAngle / 2.0f;
347  ang = DTR( -180.0f );
348  ang += halfAllowable;
349  while ( ang < DTR( 180.0f ) )
350  {
351  if ( current < ang )
352  {
353  return ( ang - ( halfAllowable ) );
354  }
355  ang += allowableAngle;
356  }
357  return ( ang - halfAllowable );
358 }

469 void CBouncyComponent::do_bounce( void )
470 {
471  m_old_pos = m_pos;
472 
473  int temp;
474  switch ( m_substate )
475  {
476  case ( 0 ):
477 
478  // generate a new constant rotation:
479  temp = Mth::Rnd( 8 );
480  m_rotation[ X ] = DEGREES_TO_RADIANS( ( float ) ( ( m_rot_per_axis >> 2 ) + ( Mth::Rnd( ( m_rot_per_axis >> 2 ) * 3 ) ) ) );
481  if ( temp & 1 )
482  {
483  m_rotation[ X ] *= -1.0f;
484  }
485  m_rotation[ Y ] = DEGREES_TO_RADIANS( ( float ) ( ( m_rot_per_axis >> 2 ) + ( Mth::Rnd( ( m_rot_per_axis >> 2 ) * 3 ) ) ) );
486  if ( temp & 2 )
487  {
488  m_rotation[ Y ] *= -1.0f;
489  }
490  m_rotation[ Z ] = DEGREES_TO_RADIANS( ( float ) ( ( m_rot_per_axis >> 2 ) + ( Mth::Rnd( ( m_rot_per_axis >> 2 ) * 3 ) ) ) );
491  if ( temp & 4 )
492  {
493  m_rotation[ Z ] *= -1.0f;
494  }
495  m_rotation[ W ] = 1.0f;
496  m_quat_initialized = false;
497  m_substate++;
498  // intentional fall through:
499  case ( 1 ):
500  {
501  Mth::Vector rot;
502  m_pos += m_vel * m_time;
503  if ( m_quat_initialized )
504  {
505  rotate_to_quat( );
506  }
507  else
508  {
509  rot = m_rotation;
510  rot *= m_time;
511  m_matrix.RotateLocal( rot );
512  }
513 
514 
515  // add the radius of the object so we don't stick through the ground:
516  Mth::Vector velNormalized = m_vel;
517  velNormalized.Normalize( );
518  Mth::Vector colPoint = m_pos + ( velNormalized * m_bounce_collision_radius );
519  m_vel[ Y ] -= m_gravity * m_time;
520  CFeeler feeler;
521  feeler.SetLine(m_old_pos, colPoint);
522  if ( feeler.GetCollision())
523  {
524  #if 0
525  // TODO - make it work, or remove if not used
526  ((CMovingObject*)(GetObject()))->RunNodeScript(0x2d075f90/*BounceScript*/,0x346b0c03/*BounceScriptParams*/);
527  #else
528  // This is also rather messy. Seems like it would be better
529  // to have this handled by an exception
530  if (m_bounce_script)
531  {
532  Script::CScript *p_script=Script::SpawnScript(m_bounce_script,mp_bounce_script_params);
533  Dbg_MsgAssert(p_script,("NULL p_script"));
534  #ifdef __NOPT_ASSERT__
535  p_script->SetCommentString("CBouncyComponent bounce script");
536  #endif
537  p_script->mpObject=GetObject();;
538  p_script->Update();
539  }
540  #endif
541 
542  m_bounce_count++;
543  Mth::Vector v;
544  Mth::Vector r;
545  float dP;
546  v = m_pos - m_old_pos;
547  Mth::Vector n;
548  n = feeler.GetNormal();
549  dP = Mth::DotProduct( v, n );
550  r = ( dP * 2.0f ) * n;
551  r -= v;
552  float origVel = m_vel.Length( );
553  m_vel = -r;
554  m_vel.Normalize( origVel * m_bounce_mult );
555  m_vel.RotateY( DTR( ( float ) ( Mth::Rnd( 2 * m_random_dir ) - m_random_dir ) ) );
556  m_pos = feeler.GetPoint() - ( velNormalized * m_bounce_collision_radius );
557  if ( fabsf( m_vel[ Y ] ) < ( m_min_bounce_vel + ( 2.0f *( m_gravity * m_time ) ) ) )
558  {
559  if ( m_rest_orientation_type )
560  {
561  if ( !m_quat_initialized )
562  {
563  set_up_quats( );
564  }
565  m_substate++;
566  break;
567  }
568 
569 
570  #if 0
571  // TODO - make it work, or remove if not used
572  // This "HigerLevel" is probably redundant for bouncy obs
573  ((CMovingObject*)(GetObject()))->HigherLevel( false );
574  #endif
575 
576  SWITCH_STATE( BOUNCYOBJ_STATE_IDLE );
577  if ( m_destroy_when_done )
578  {
579  GetObject()->MarkAsDead( );
580  }
581  else
582  {
583  #if 0
584  // TODO - get exceptions working
585  GetExceptionComponentFromObject( GetObject() )->FlagException( 0x61682835 ); // DoneBouncing
586  #endif
587  }
588  return;
589  }
590 
591  if ( origVel > m_min_bounce_vel * 2.0f )
592  {
593  if ( m_bounce_sound )
594  {
595  // adjust volume depending on magnitude of current velocity:
596  #if 0
597  // TODO - Get Bouncy object sounds working with components
598  float vol;
599  vol = 100.0f * ( origVel / ( MAX_SKATER_VEL * m_bounciness ) );
600  float pitch = 80.0f + ( float ) Mth::Rnd( 40 );
601  ((CMovingObject*)(GetObject()))->PlaySound_VolumeAndPan( m_bounce_sound, vol, pitch );
602  #endif
603  }
604 
605  // bounce rot:
606  if ( m_bounce_rot )
607  {
608  rot[ X ] = DEGREES_TO_RADIANS( ( float )( Mth::Rnd( m_bounce_rot * 2 ) - m_bounce_rot ) );
609  rot[ Y ] = DEGREES_TO_RADIANS( ( float )( Mth::Rnd( m_bounce_rot * 2 ) - m_bounce_rot ) );
610  rot[ Z ] = DEGREES_TO_RADIANS( ( float )( Mth::Rnd( m_bounce_rot * 2 ) - m_bounce_rot ) );
611  m_matrix.RotateLocal( rot );
612  }
613  m_substate = 0;
614  }
615  else if ( m_rest_orientation_type )
616  {
617  if ( !m_quat_initialized )
618  {
619  // rotate to the resting orientation:
620  set_up_quats( );
621  }
622  }
623  }
624  break;
625  }
626 
627  case ( 2 ):
628  if ( rotate_to_quat( ) )
629  {
630  #if 0
631  ((CMovingObject*)(GetObject()))->HigherLevel( false );
632  #endif
633 
634  SWITCH_STATE( BOUNCYOBJ_STATE_IDLE );
635  if ( m_destroy_when_done )
636  {
637  GetObject()->MarkAsDead( );
638  }
639  }
640  break;
641 
642  default:
643  Dbg_Message( 0, "Woah! Fire matt immediately." );
644  break;
645  }
646 
647  #ifdef __DEBUG_BOUNCY__
648 // printf ("%d: %s: (%.2f,%.2f,%.2f)\n",__LINE__,__PRETTY_FUNCTION__,m_pos[X],m_pos[Y],m_pos[Z]);
649  #endif
650
