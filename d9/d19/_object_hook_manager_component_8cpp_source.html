149 {
150  switch( Checksum )

153  // @script | DoSomething | does some functionality
154  case 0xbb4ad101: // DoSomething

156  break;

162  }
163  break;
164 */
165 
166  default:
167  return CBaseComponent::MF_NOT_EXECUTED;
168  }
169 
170  // the "default" case of the switch statement handles
171  // unrecognized functions; if we make it down here,
172  // that means that the component already handled it
173  // somehow
174  return CBaseComponent::MF_TRUE;
