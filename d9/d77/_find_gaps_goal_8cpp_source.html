76 
77  Mdl::Score* pScore = pSkater->GetScoreObject();
78  int numTricks = pScore->GetCurrentTrickCount();
79  bool found_gap = false;
80  for ( int i = 0; i < numTricks; i++ )
81  {
82  if ( pScore->TrickIsGap( i ) )
83  {
84  uint32 gap_name = pScore->GetTrickId( i );
85  if ( !pCompletedGaps->ContainsComponentNamed( gap_name ) )
86  {
87  pCompletedGaps->AddInteger( gap_name, 1 );
88  num_gaps_found++;
89  found_gap = true;
90  }
91  }
92  }
93  mp_params->AddInteger( CRCD(0xf1e59083,"num_gaps_found"), num_gaps_found );
94  mp_params->AddInteger( CRCD(0x129daa10,"number_collected"), num_gaps_found );
95 
96  if ( found_gap )
97  {
98  Script::CStruct* pScriptParams = new Script::CStruct();
99  pScriptParams->AddChecksum( CRCD(0x9982e501,"goal_id"), GetGoalId() );
100  Script::RunScript( CRCD(0x851aa6b8,"goal_find_gaps_found_gap"), pScriptParams );
101  delete pScriptParams;
102  }
