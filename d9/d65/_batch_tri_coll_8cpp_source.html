68 bool vu0RayTriangleCollision(const Mth::Vector *rayStart, const Mth::Vector *rayDir,
69  Mth::Vector *v0, Mth::Vector *v1, Mth::Vector *v2, float *t)
70 {
71  register bool result;
72 
73  asm __volatile__(
74  "
75  .set noreorder
76  lqc2 vf06,0x0(%4) # v0
77  lqc2 vf08,0x0(%6) # v2
78  lqc2 vf07,0x0(%5) # v1
79  lqc2 vf04,0x0(%2) # rayStart
80  lqc2 vf05,0x0(%3) # rayDir
81 
82  vcallms RayTriangleCollision # call microsubroutine
83  vnop # interlocking instruction, waits for vu0 completion
84 
85  cfc2 %0,$vi02 # get boolean result from vu0
86  #li %0,1 # store 1 for return value
87 
88  beq %0, $0, vu0RayTriDone # skip copy of t if not needed
89  nop
90 
91  qmfc2 $8, $vf17 # move t to $8
92  sw $8, 0(%1) # and write out
93 
94 vu0RayTriDone:
95 
96  .set reorder
97  ": "=r" (result), "+r" (t) : "r" (rayStart), "r" (rayDir), "r" (v0), "r" (v1) , "r" (v2) : "$8", "$9" );
98 
99  return result;
100 }
101 #endif // __PLAT_NGPS__

108 bool CBatchTriCollMan::sInit(CollData *p_coll_data)
109 {
110  // Check for nesting; abort if true
111  if (s_processing)
112  {
113  s_nested++;
114  return false;
115  }
116 
117  Dbg_Assert(s_processing == false);
118 
119  s_array_size = 0;
120  s_next_idx_to_batch = 0;
121  s_found_collision = false;
122 
123  CBatchTriColl::sp_coll_data = p_coll_data;
124 
125 #ifdef __PLAT_NGPS__
126  // install interrupt handler for render completion
127  if (s_collision_handler_id == -1)
128  {
129  s_collision_handler_id = AddIntcHandler(INTC_VU0, s_collision_done, 0);
130  EnableIntc(INTC_VU0);
131  sceDevVu0PutTBit(1); // Use T-bit interrupt
132  }
133 #endif // __PLAT_NGPS__
134 
135  return true;

201 {
202  Dbg_Assert(s_nested == 0);
203 
204  sWaitTriCollisions(); // make sure were done working
205  Dbg_Assert(!s_processing);
206 
207  // Check if we have anything to do
208  if (s_next_idx_to_batch >= s_array_size)
209  {
210  return;
211  }
212 
213  s_processing = true;
214 
215 #ifdef __PLAT_NGPS__
216  if(s_use_vu0_micro)
217  {
218  uint128 *p_vu0_mem = (uint128*)0x11004000; // VU0 data memory
219 
220  // Write out sizes first
221  uint32 *p_vu0_word_mem = (uint32 *) (p_vu0_mem++);
222  *(p_vu0_word_mem++) = s_next_idx_to_batch;
223  *(p_vu0_word_mem++) = s_array_size;
224 
225  for (int i = s_next_idx_to_batch; i < s_array_size; i++)
226  {
227  const CBatchTriColl &tri_coll = s_tri_collision_array[i][s_current_array];
228 
229  Mth::Vector *v0, *v1, *v2;
230  if (tri_coll.mp_collision_obj->mp_coll_tri_data->m_use_face_small)
231  {
232  CCollObjTriData::SFaceSmall *face = &(tri_coll.mp_collision_obj->mp_coll_tri_data->mp_face_small[tri_coll.m_face_index]);
233 
234  v0 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[0]];
235  v1 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[1]];
236  v2 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[2]];
237  } else {
238  CCollObjTriData::SFace *face = &(tri_coll.mp_collision_obj->mp_coll_tri_data->mp_faces[tri_coll.m_face_index]);
239 
240  v0 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[0]];
241  v1 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[1]];
242  v2 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[2]];
243  }
244 
245  *(p_vu0_mem++) = *((uint128*) &(tri_coll.m_test_line.m_start));
246  *(p_vu0_mem++) = *((uint128*) &(tri_coll.m_test_line_dir));
247  *(p_vu0_mem++) = *((uint128*) v0);
248  *(p_vu0_mem++) = *((uint128*) v1);
249  *(p_vu0_mem++) = *((uint128*) v2);
250  }
251 
252  //FlushCache(WRITEBACK_DCACHE);
253 
254  //sceDevVu0PutTBit(1); // Use T-bit interrupt
255 
256  // Switch the buffers before we make the call (in case it finishes immediately)
257  s_switch_buffers();
258 
259  asm __volatile__(
260  "
261  .set noreorder
262  sync.l
263  vcallms BatchRayTriangleCollision # call microsubroutine
264 
265  .set reorder
266  ");
267  }
268  else
269 #endif // __PLAT_NGPS__
270  {
271  // Just do it manually now
272  for (int i = s_next_idx_to_batch; i < s_array_size; i++)
273  {
274  const CBatchTriColl &tri_coll = s_tri_collision_array[i][s_current_array];
275 
276  Mth::Vector *v0, *v1, *v2;
277  if (tri_coll.mp_collision_obj->mp_coll_tri_data->m_use_face_small)
278  {
279  CCollObjTriData::SFaceSmall *face = &(tri_coll.mp_collision_obj->mp_coll_tri_data->mp_face_small[tri_coll.m_face_index]);
280 
281  v0 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[0]];
282  v1 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[1]];
283  v2 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[2]];
284  } else {
285  CCollObjTriData::SFace *face = &(tri_coll.mp_collision_obj->mp_coll_tri_data->mp_faces[tri_coll.m_face_index]);
286 
287  v0 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[0]];
288  v1 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[1]];
289  v2 = &tri_coll.mp_collision_obj->mp_coll_tri_data->mp_vert_pos[face->m_vertex_index[2]];
290  }
291 
292  float distance;
293 #ifdef __PLAT_NGPS__
294  if (vu0RayTriangleCollision(&tri_coll.m_test_line.m_start, &tri_coll.m_test_line_dir, v0, v1, v2, &distance))
295 #else
296  if (CCollObj::sRayTriangleCollision(&tri_coll.m_test_line.m_start, &tri_coll.m_test_line_dir,
297  v0, v1, v2, &distance))
298 #endif
299  {
300  SCollSurface collisionSurface;
301 
302  /* We've got one */
303  collisionSurface.point = (*v0);
304  collisionSurface.index = tri_coll.m_face_index;
305 
306  // Find normal
307  Mth::Vector vTmp1(*v1 - *v0);
308  Mth::Vector vTmp2(*v2 - *v0);
309  collisionSurface.normal = Mth::CrossProduct(vTmp1, vTmp2);
310  collisionSurface.normal.Normalize();
311 
312  if (CCollObj::s_found_collision(&tri_coll.m_test_line, tri_coll.mp_collision_obj, &collisionSurface, distance, CBatchTriColl::sp_coll_data))
313  {
314  s_found_collision = true;
315  }
316  }
317  }
318  s_processing = false; // Just for manual version
319 
320  s_switch_buffers();
321  }
322 }
