149 Manager::Manager( void )
150 {
151 
152 
153  m_current_id = 0;
154  mp_process_man = NULL;
155 
156 # if defined ( __PLAT_XBOX__ )
157  // Just grab 33mb of main memory.
158  _mem_start = (char*)XPhysicalAlloc( 33 * 1024 * 1024, // size of region
159  MAXULONG_PTR, // base physical address
160  0, // memory alignment
161  PAGE_READWRITE ); // memory protection and type
162 
163  _mem_end = _mem_start + ( 33 * 1024 * 1024 );
164  _std_mem_end = _mem_end;
165 # elif defined ( __PLAT_WN32__ )
166  // Nasty hack for WN32 for now - just grab 38mb of main memory via a malloc.
167  _mem_start = (char *)malloc( 38 * 1024 * 1024 );
168  _mem_end = _mem_start + ( 38 * 1024 * 1024 );
169  _std_mem_end = _mem_end;
170 
171 # endif
172 # if defined ( __PLAT_NGC__ )
173  // Fill in what we know.
174  _code_start = (char *)0xfadebabe; // Junk addresses.
175  _code_end = (char *)0xfacced00;
176  _data_end = (char *)0xcaccfacc;
177  _mem_start = &((char *)OSGetArenaLo())[8192]; // Real, actual useful addresses.
178  _mem_end = (char *)OSGetArenaHi();
179  _std_mem_end = (char *)OSGetArenaHi();
180 # endif
181 
182  // create root memory region
183  mp_region = new ((void*)s_region_buffer) Region( nAlignUp( _mem_start ), nAlignDown( _std_mem_end ) );
184 
185  // create default heaps
186  mp_top_heap = new ((void*)s_top_heap_buffer) Heap( mp_region, Heap::vTOP_DOWN, "Top Down" );
187  mp_bot_heap = new ((void*)s_bot_heap_buffer) Heap( mp_region, Heap::vBOTTOM_UP, "BottomUp" );
188 
189  m_heap_list[0] = mp_top_heap;
190  m_heap_list[1] = mp_bot_heap;
191 
192  m_num_heaps = 2;
193 
194 # if !defined( __PLAT_NGC__ ) || ( defined( __PLAT_NGC__ ) && !defined( __NOPT_FINAL__ ) )
195  uint codesize = ((uint)(_code_end) - (uint)(_code_start))/1024;
196  uint datasize = ((uint)(_data_end) - (uint)(_code_end))/1024;
197  printf ( "code [%p - %p] (%dK) + data [%p - %p] (%dK) = %dK \n",
198  _code_start, _code_end, codesize,
199  _code_end, _data_end, datasize,
200  codesize + datasize );
201 #endif
202 
203  m_pushed_context_count = 0;
204  mp_internet_region = NULL;
205  mp_net_misc_region = NULL;
206 
207  mp_cutscene_region = NULL;
208  mp_cutscene_bottom_heap = NULL;
209  mp_cutscene_top_heap = NULL;
210 
211 #ifdef __PLAT_NGPS__
212  // Create a semaphore to prevent threads from re-entering push/pop memory contexts
213  struct SemaParam params;
214 
215  params.initCount = 1;
216  params.maxCount = 10;
217 
218  s_context_semaphore = CreateSema( &params );
219 #endif

723 void Manager::RemoveHeap(Mem::Heap *pHeap)
724 {
725 
726 #ifndef __PLAT_WN32__
727  #ifdef __NOPT_ASSERT__
728  // Check first to see if there is something on the heap before it is deleted
729  // Deleting a heap with stuff on it might indicate an error
730  if (pHeap->mUsedBlocks.m_count)
731  {
732  printf ("Deleting a heap %s with %d used blocks still on it\n",pHeap->mp_name,pHeap->mUsedBlocks.m_count);
733  #ifndef __PLAT_NGC__
734  printf ("\n\nDumping Heap\n");
735  MemView_DumpHeap(pHeap);
736  printf ("\n\nAnalyzing Heap\n");
737  MemView_AnalyzeHeap(pHeap);
738  Dbg_MsgAssert(0, ("Deleting heap <%s> with %d used blocks still on it\n",pHeap->mp_name,pHeap->mUsedBlocks.m_count));
739 
740  #endif // __PLAT_NGC__
741  }
742  #endif
743 #endif
744 
745  // remove from list of heaps
746  for (int i=0;i<m_num_heaps;i++)
747  {
748  if (m_heap_list[i] == pHeap)
749  {
750  m_num_heaps--; // chop off the last heaps
751  m_heap_list[i] = m_heap_list[m_num_heaps]; // and insert if in the hole
752  break;
753  }
754  }
755 
756  delete pHeap;
757 
758 }
