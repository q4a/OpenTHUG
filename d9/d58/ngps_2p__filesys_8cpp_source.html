612 static size_t cdFread(void *addr, size_t size, size_t count, void *fptr)
613 {
614 
615  skyFile *fp = (skyFile *)fptr;
616  size_t numBytesToRead = size * count;
617  int bytesRead, bytesRead2;
618 
619  bytesRead = 0;
620 
621  StopStreaming( );
622 
623 // char * after = (char*)addr + numBytesToRead;
624 
625 
626 
627  /* Trim number of bytes for the size of the file */
628  if ((fp->POS + (int32)numBytesToRead) > fp->SOF)
629  {
630  numBytesToRead = fp->SOF - fp->POS;
631  }
632 
633 
634  /* First try and use the buffer */
635  if ((fp->bufferPos < READBUFFERSIZE) &&
636  (bytesRead < (int32)numBytesToRead))
637  {
638  /* Pull from the buffer */
639  if (numBytesToRead < (READBUFFERSIZE-fp->bufferPos))
640  {
641  /* Can satisfy entirely from buffer */
642  bytesRead = numBytesToRead;
643  }
644  else
645  {
646  /* Pull as much as possible from the buffer */
647  bytesRead = READBUFFERSIZE-fp->bufferPos;
648  }
649 
650  /* Copy it */
651  memcpy(addr, &fp->readBuffer[fp->bufferPos], bytesRead);
652 
653  /* Update target address and source address */
654  addr = (void *)((uint8 *)addr + bytesRead);
655  fp->bufferPos += bytesRead;
656  fp->POS += bytesRead;
657  }
658 
659  /* If next bit is bigger than a buffer, read it directly and ignore the
660  * buffer.
661  */
662  if ((numBytesToRead-bytesRead) > 0)
663  {
664  if ((numBytesToRead-bytesRead) >= READBUFFERSIZE)

669  //Dbg_Message("Seeking to LSN %d", fp->WadOffset+fp->POS);

710 static int cdFseek(void *fptr, long offset, int origin)
711 {
712 
713 
714  skyFile *fp = (skyFile *)fptr;
715  int32 oldFPos, bufStart;
716  bool noBuffer = FALSE;
717 
718  Dbg_AssertPtr( fptr );
719 
720  oldFPos = fp->POS;
721  bufStart = oldFPos - fp->bufferPos;
722  if (fp->bufferPos == READBUFFERSIZE) noBuffer = TRUE;
723  fp->bufferPos = READBUFFERSIZE;
724 
725  switch (origin)
726  {
727  case SEEK_CUR:
728  {
729  /* Does the seek stay in the buffer */
730  if ((oldFPos + offset >= bufStart) &&
731  (oldFPos + offset < bufStart+READBUFFERSIZE))
732  {
733  fp->bufferPos = (oldFPos + offset) - bufStart;
734  fp->POS += offset;
735  }
736  else
737  {
738  fp->POS+=offset;
739  if (fp->POS<0 || fp->POS>fp->SOF)
740  {
741  fp->POS=-1;
742  Dbg_MsgAssert(0,("Bad offset sent to cdFseek (SEEK_CUR) offset=%ld pos=%d size=%d",offset,fp->POS-offset,fp->SOF));
743  }
744  }
745  break;
746  }
747  case SEEK_END:
748  {
749  fp->POS=fp->SOF-offset;
750  if (fp->POS<0 || fp->POS>fp->SOF)
751  {
752  fp->POS=-1;
753  Dbg_MsgAssert(0,("Bad offset sent to cdFseek (SEEK_END)"));
754  }
755  break;
756  }
757  case SEEK_SET:
758  {
759  fp->POS=offset;
760  if (fp->POS<0 || fp->POS>fp->SOF)
761  {
762  fp->POS=-1;
763  Dbg_MsgAssert(0,("Bad offset sent to cdFseek (SEEK_SET)"));
764  }
