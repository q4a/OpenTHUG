64 void CXboxModelLights::plat_update_engine( Mth::Vector & pos, bool add_scene_light )
65 {
66  if( m_flags & mUSE_MODEL_AMBIENT )
67  {
68  // Use the local ambient color, modulate it with the local ambient brightness.
69  NxXbox::EngineGlobals.ambient_light_color[0] = m_ambient_color.r * ( 1.0f / 128.0f ) * m_ambient_brightness;
70  NxXbox::EngineGlobals.ambient_light_color[1] = m_ambient_color.g * ( 1.0f / 128.0f ) * m_ambient_brightness;
71  NxXbox::EngineGlobals.ambient_light_color[2] = m_ambient_color.b * ( 1.0f / 128.0f ) * m_ambient_brightness;
72  }
73  else
74  {
75  // Use the default ambient color, but modulate it with the local ambient brightness.
76  Image::RGBA amb = CLightManager::sGetLightAmbientColor();
77  NxXbox::EngineGlobals.ambient_light_color[0] = amb.r * ( 1.0f / 128.0f ) * m_ambient_brightness;
78  NxXbox::EngineGlobals.ambient_light_color[1] = amb.g * ( 1.0f / 128.0f ) * m_ambient_brightness;
79  NxXbox::EngineGlobals.ambient_light_color[2] = amb.b * ( 1.0f / 128.0f ) * m_ambient_brightness;
80  }
81 
82  for( int i = 0; i < 2; ++i )
83  {
84  if( m_flags & (( i == 0 ) ? mUSE_MODEL_DIFFUSE_0 : mUSE_MODEL_DIFFUSE_1 ))
85  {
86  // Use the local directional color, modulate it with the local directional brightness.
87  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 4] = m_diffuse_color[i].r * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
88  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 5] = m_diffuse_color[i].g * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
89  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 6] = m_diffuse_color[i].b * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
90 
91  // Use the local direction.
92  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 0] = -m_diffuse_direction[i][X];
93  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 1] = -m_diffuse_direction[i][Y];
94  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 2] = -m_diffuse_direction[i][Z];
95  }
96  else
97  {
98  // Use the default directional color, but modulate it with the local directional brightness.
99  Image::RGBA dif = CLightManager::sGetLightDiffuseColor( i );
100  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 4] = dif.r * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
101  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 5] = dif.g * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
102  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 6] = dif.b * ( 1.0f / 128.0f ) * m_diffuse_brightness[i];
103 
104  // Use the default direction.
105  Mth::Vector dir = CLightManager::sGetLightDirection( i );
106  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 0] = -dir[X];
107  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 1] = -dir[Y];
108  NxXbox::EngineGlobals.directional_light_color[( i * 8 ) + 2] = -dir[Z];
109  }
110  }
111 
112  // Figure Scene Lighting if required.
113  if( add_scene_light )
114  {
115  Nx::CSceneLight *p_scene_light = CLightManager::sGetOptimumSceneLight( pos );
116  if( p_scene_light )
117  {
118  Mth::Vector light_pos = p_scene_light->GetLightPosition();
119 
120  float dist = Mth::Distance( pos, light_pos );
121  float ratio = dist * p_scene_light->GetLightReciprocalRadius();
122 
123  light_pos = ( pos - light_pos ).Normalize();
124 
125  // Figure the direction...
126  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 0] = light_pos[X];
127  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 1] = light_pos[Y];
128  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 2] = light_pos[Z];
129 
130  // ...and the color.
131  ratio = sqrtf( 1.0f - ratio ) * ( 1.0f / 255.0f ) * p_scene_light->GetLightIntensity();
132  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 4] = ratio * p_scene_light->GetLightColor().r;
133  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 5] = ratio * p_scene_light->GetLightColor().g;
134  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 6] = ratio * p_scene_light->GetLightColor().b;
135  }
136  else
137  {
138  // Disbale this light by setting zero color.
139  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 4] = 0.0f;
140  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 5] = 0.0f;
141  NxXbox::EngineGlobals.directional_light_color[( 2 * 8 ) + 6] = 0.0f;
142  }
143  }
144 }
