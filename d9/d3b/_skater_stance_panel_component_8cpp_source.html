93 void CSkaterStancePanelComponent::Update()
94 {
95  Front::CSpriteElement* p_stance_icon = static_cast< Front::CSpriteElement* >(Front::CScreenElementManager::Instance()->GetElement(
96  CRCD(0x968ae5dd, "the_boardstance_sprite") + static_cast< CSkater* >(GetObject())->GetHeapIndex()
97  ).Convert());
98 
99  Dbg_Assert(p_stance_icon);
100 
101  int stance = determine_stance();
102 
103  if (stance == m_last_stance) return;
104  m_last_stance = stance;
105 
106  if (stance == 5)
107  {
108  p_stance_icon->SetAlpha(0.0f, Front::CScreenElement::FORCE_INSTANT);
109  }
110  else
111  {
112  const static uint32 sp_texture_checksums[] =
113  {
114  CRCD(0x33a15296, "nollie_icon"),
115  CRCD(0xbe91f3b6, "fakie_icon"),
116  CRCD(0x9f9d3907, "switch_icon"),
117  CRCD(0xb793ef40, "sw_pressure_icon"),
118  CRCD(0x4e304fa1, "pressure_icon")
119  };
120 
121  p_stance_icon->SetTexture(sp_texture_checksums[stance]);
122  p_stance_icon->SetAlpha(1.0f, Front::CScreenElement::FORCE_INSTANT);
123  }
124 }

131 CBaseComponent::EMemberFunctionResult CSkaterStancePanelComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
132 {
133  switch ( Checksum )
134  {
135  // @script | InNollie | true if in nollie
136  case CRCC(0x1eb61dce, "InNollie"):
137  return m_in_nollie ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
138  break;
139 
140  // @script | NollieOn | sets nollie to on (check with InNollie)
141  case CRCC(0x41eb710b, "NollieOn"):
142  if (m_in_nollie == false)
143  {
144  m_in_nollie = true;
145  GetObject()->BroadcastEvent(CRCD(0x8157ab31, "SkaterEnterNollie"));
146  }
147  break;
148 
149  // @script | NollieOff | sets nollie to off (check with InNollie)
150  case CRCC(0xfb9b7c9c, "NollieOff"):
151  if (m_in_nollie == true)
152  {
153  m_in_nollie = false;
154  GetObject()->BroadcastEvent(CRCD(0x3f70881a, "SkaterExitNollie"));
155  }
156  break;
157 
158  // @script | InPressure | true if in pressure
159  case CRCC(0x9fab9d0b,"InPressure"):
160  return m_in_pressure ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
161  break;
162 
163  // @script | PressureOn | sets pressure to on (check with InPressure)
164  case CRCC(0xa23d710,"PressureOn"):
165  if (m_in_pressure == false)
166  {
167  m_in_pressure = true;
168  GetObject()->BroadcastEvent(CRCD(0x2e8de921,"SkaterEnterPressure"));
169  }
170  break;
171 
172  // @script | PressureOff | sets pressure to off (check with InPressure)
173  case CRCC(0x71b57dd6,"PressureOff"):
174  if (m_in_pressure == true)
175  {
176  m_in_pressure = false;
177  GetObject()->BroadcastEvent(CRCD(0xfa9adb1d,"SkaterExitPressure"));
178  }
179  break;
180 
181  default:
182  return CBaseComponent::MF_NOT_EXECUTED;
183  }
184 
185  return CBaseComponent::MF_TRUE;
186 }
