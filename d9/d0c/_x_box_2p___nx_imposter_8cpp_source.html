22 static void frustum_project_box( Mth::CBBox &bbox, XGMATRIX *p_view_matrix, Mth::Vector *p_max_x, Mth::Vector *p_max_y, Mth::Vector *p_mid )
23 {
24  float max_projected_xx, max_projected_xz; // The camera space position of the point with the greatest x axis value when projected to z = mid_z.
25  float max_projected_x_mid_z; // The projected x axis value when this point is projected to z = mid_z;
26 
27  float max_projected_yy, max_projected_yz; // The camera space position of the point furthest along the y axis when projected to z = mid_z.
28  float max_projected_y_mid_z; // The projected y axis value when this point is projected to z = mid_z;
29 
30  float min_x = bbox.GetMin().GetX();
31  float min_y = bbox.GetMin().GetY();
32  float min_z = bbox.GetMin().GetZ();
33  float max_x = bbox.GetMax().GetX();
34  float max_y = bbox.GetMax().GetY();
35  float max_z = bbox.GetMax().GetZ();
36 
37  // Project the midpoint of the box, since this is the point through which the imposter polygon will pass.
38  XGVECTOR3 mid_in( min_x + ( 0.5f * ( max_x - min_x )), min_y + ( 0.5f * ( max_y - min_y )), min_z + ( 0.5f * ( max_z - min_z )));
39  XGVECTOR4 mid_out;
40  XGVec3Transform( &mid_out, &mid_in, p_view_matrix );
41 
42  for( uint32 v = 0; v < 8; ++v )
43  {
44  XGVECTOR3 in;
45  XGVECTOR4 out;
46 
47  in.x = ( v & 0x04 ) ? max_x : min_x;
48  in.y = ( v & 0x02 ) ? max_y : min_y;
49  in.z = ( v & 0x01 ) ? max_z : min_z;
50 
51  XGVec3Transform( &out, &in, p_view_matrix );
52 
53  out.x = fabsf( out.x );
54  out.y = fabsf( out.y );
55 
56  float projected_x_mid_z = out.x * ( mid_out.z / out.z );
57  float projected_y_mid_z = out.y * ( mid_out.z / out.z );
58 
59  if( v == 0 )
60  {
61  max_projected_x_mid_z = projected_x_mid_z;
62  max_projected_xx = out.x;
63  max_projected_xz = out.z;
64 
65  max_projected_y_mid_z = projected_y_mid_z;
66  max_projected_yy = out.y;
67  max_projected_yz = out.z;
68  }
69  else
70  {
71  if( projected_x_mid_z > max_projected_x_mid_z )
72  {
73  max_projected_xx = out.x;
74  max_projected_xz = out.z;
75  max_projected_x_mid_z = projected_x_mid_z;
76  }
77 
78  if( projected_y_mid_z > max_projected_y_mid_z )
79  {
80  max_projected_yy = out.y;
81  max_projected_yz = out.z;
82  max_projected_y_mid_z = projected_y_mid_z;
83  }
84  }
85  }
86 
87  p_max_x->Set( max_projected_xx, 0.0f, max_projected_xz, 0.0f );
88  p_max_y->Set( 0.0f, max_projected_yy, max_projected_yz, 0.0f );
89  p_mid->Set( mid_out.x, mid_out.y, mid_out.z );
90 }

234 void CXboxImposterGroup::plat_create_imposter_polygon( void )
235 {
236  Dbg_Assert( !m_imposter_polygon_exists );
237 
238  // Generate a camera matrix that will point the camera directly at the midpoint of the bounding box.
239  XGMATRIX cam_mat;
240  XGVECTOR3 look_at( m_composite_bbox_mid[X], m_composite_bbox_mid[Y], m_composite_bbox_mid[Z] );
241  XGMatrixLookAtRH( &cam_mat, &NxXbox::EngineGlobals.cam_position, &look_at, &NxXbox::EngineGlobals.cam_up );
242 
243  // Using this camera matrix, project all eight corners of the bounding box, in order to determine the best size for the texture.
244  Mth::Vector proj_max_x, proj_max_y, proj_mid;
245  frustum_project_box( m_composite_bbox, &cam_mat, &proj_max_x, &proj_max_y, &proj_mid );
246 
247  // Now project the minimum and maximum x and y values onto the projection plane - the plane through the midpoint of the box.
248  float max_projected_x = proj_max_x[X] * ( proj_mid[Z] / proj_max_x[Z] );
249  float max_projected_y = proj_max_y[Y] * ( proj_mid[Z] / proj_max_y[Z] );
250 
251  // Calculate the maximum width and height at the near plane.
252  float wnp = 2.0f * proj_max_x[X] * ( NxXbox::EngineGlobals.near_plane / fabsf( proj_max_x[Z] ));
253  float hnp = 2.0f * proj_max_y[Y] * ( NxXbox::EngineGlobals.near_plane / fabsf( proj_max_y[Z] ));
254 
255  m_tex_width = 16 + Ftoi_ASM(( 640.0f * wnp ) / NxXbox::EngineGlobals.near_plane_width );
256  m_tex_height = 16 + Ftoi_ASM(( 480.0f * hnp ) / NxXbox::EngineGlobals.near_plane_height );
257 
258  // Round texture to the nearest 16 pixel limit.
259  m_tex_width = (( m_tex_width + 0x0F ) & ~0x0F );
260  m_tex_height = (( m_tex_height + 0x0F ) & ~0x0F );
261 
262  // Clamp texture to maximum allowed size.
263  m_tex_width = ( m_tex_width > XBOX_IMPOSTER_MAX_U_TEXTURE_SIZE ) ? XBOX_IMPOSTER_MAX_U_TEXTURE_SIZE : m_tex_width;
264  m_tex_height = ( m_tex_height > XBOX_IMPOSTER_MAX_V_TEXTURE_SIZE ) ? XBOX_IMPOSTER_MAX_V_TEXTURE_SIZE : m_tex_height;
265 
266  // Calculate the corresponding projection matrix.
267  XGMATRIX proj_mat;
268  XGMatrixPerspectiveRH( &proj_mat, wnp, hnp, NxXbox::EngineGlobals.near_plane, NxXbox::EngineGlobals.far_plane );
269 
270  // Set the calculated view and projection matrices.
271  D3DDevice_SetTransform( D3DTS_VIEW, &cam_mat );
272  D3DDevice_SetTransform( D3DTS_PROJECTION, &proj_mat );
273 
274  // Create a render target texture into which the object will be drawn.
275  HRESULT hr;
276  if( mp_texture == NULL )
277  {
278  hr = D3DDevice_CreateTexture( m_tex_width, m_tex_height, 1, 0, D3DFMT_LIN_A8R8G8B8, 0, &mp_texture );
279  Dbg_Assert( hr == D3D_OK );
280  }
281 
282  // Create a corresponding depth texture (we need this only for the render - then it can be removed).
283  IDirect3DTexture8* p_depth_buffer;
284  hr = D3DDevice_CreateTexture( m_tex_width, m_tex_height, 1, 0, D3DFMT_LIN_D24S8, 0, &p_depth_buffer );
285  Dbg_Assert( hr == D3D_OK );
286 
287  // This call will increase the reference count of the IDirect3DTexture8 object.
288  LPDIRECT3DSURFACE8 p_surface, p_depth_surface;
289  mp_texture->GetSurfaceLevel( 0, &p_surface );
290  p_depth_buffer->GetSurfaceLevel( 0, &p_depth_surface );
291 
292  // This call will increase the reference count of the IDirect3DSurface8 object.
293  D3DDevice_SetRenderTarget( p_surface, p_depth_surface );
294 
295  // Clear the render target.
296  D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, 0x00000000, 1.0f, 0 );
297 // D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, 0x80800000, 1.0f, 0 );
298 
299  // The imposter polygon has now been created.
300  m_imposter_polygon_exists = true;
301 
302  // Set the camera position at the time of creation.
303  m_cam_pos.Set( NxXbox::EngineGlobals.cam_position.x, NxXbox::EngineGlobals.cam_position.y, NxXbox::EngineGlobals.cam_position.z );
304 
305  // Build a list of meshes, so we can sort them dynamically into draw order.
306  NxXbox::sMesh* mesh_list[256];
307  uint32 num_meshes = 0;
308 
309  Lst::Node<Nx::CGeom> *node, *next;
310  for( node = mp_geom_list->GetNext(); node; node = next )
311  {
312  next = node->GetNext();
313  Nx::CXboxGeom *p_xbox_geom = static_cast<Nx::CXboxGeom*>( node->GetData());
314 
315  for( uint32 m = 0; m < p_xbox_geom->m_num_mesh; ++m )
316  {
317  Dbg_Assert( num_meshes < 256 );
318  NxXbox::sMesh* p_mesh = p_xbox_geom->m_mesh_array[m];
319  mesh_list[num_meshes++] = p_mesh;
320  }
321  }
322 
323  if( num_meshes > 0 )
324  {
325  // Sort the array of pointers into draw order.
326  qsort( mesh_list, num_meshes, sizeof( NxXbox::sMesh* ), NxXbox::sort_by_material_draw_order );
327  }
328 
329  // Render each mesh into the render target.
330  NxXbox::EngineGlobals.blend_mode_override = NxXbox::vBLEND_MODE_ONE_INV_SRC_ALPHA;
331  NxXbox::sMaterial* p_material = NULL;
332  for( uint32 m = 0; m < num_meshes; ++m )
333  {
334  NxXbox::sMesh* p_mesh = mesh_list[m];
335  if( p_mesh->mp_material != p_material )
336  {
337  p_material = p_mesh->mp_material;
338  p_material->Submit();
339  }
340  p_mesh->Submit();
341  }
342  NxXbox::EngineGlobals.blend_mode_override = 0;
343 
344  // Can now set the meshes in the geom inactive.
345  for( node = mp_geom_list->GetNext(); node; node = next )
346  {
347  next = node->GetNext();
348  Nx::CXboxGeom *p_xbox_geom = static_cast<Nx::CXboxGeom*>( node->GetData());
349  p_xbox_geom->SetActive( false );
350  }
351 
352  // Remove references to surfaces.
353  p_surface->Release();
354  p_depth_surface->Release();
355  p_depth_buffer->Release();
356 
357  // Restore the default render target.
358  D3DDevice_SetRenderTarget( NxXbox::EngineGlobals.p_RenderSurface, NxXbox::EngineGlobals.p_ZStencilSurface );
359 
360  // Restore the view and projection transforms.
361  D3DDevice_SetTransform( D3DTS_VIEW, &NxXbox::EngineGlobals.view_matrix );
362  D3DDevice_SetTransform( D3DTS_PROJECTION, &NxXbox::EngineGlobals.projection_matrix );
363 
364  // Now figure the vertex positions for the polygon.
365  Mth::Vector at = m_composite_bbox_mid - m_cam_pos;
366  at.Normalize();
367  Mth::Vector right = Mth::CrossProduct( Mth::Vector( NxXbox::EngineGlobals.cam_up.x, NxXbox::EngineGlobals.cam_up.y, NxXbox::EngineGlobals.cam_up.z ), at );
368  right.Normalize();
369  Mth::Vector up = Mth::CrossProduct( at, right );
370 
371  Mth::Vector verts[4];
372  verts[0] = m_composite_bbox_mid - ( max_projected_x * right ) + ( max_projected_y * up );
373  verts[1] = m_composite_bbox_mid + ( max_projected_x * right ) + ( max_projected_y * up );
374  verts[2] = m_composite_bbox_mid + ( max_projected_x * right ) - ( max_projected_y * up );
375  verts[3] = m_composite_bbox_mid - ( max_projected_x * right ) - ( max_projected_y * up );
376 
377  for( int v = 0; v < 4; ++v )
378  {
379  m_vertex_buffer[v].x = verts[v][X];
380  m_vertex_buffer[v].y = verts[v][Y];
381  m_vertex_buffer[v].z = verts[v][Z];
382  }
383 
384  // The texture is a linear format, so the uv's are in texel space.
385  m_vertex_buffer[0].u = (float)m_tex_width;
386  m_vertex_buffer[0].v = 0.0f;
387  m_vertex_buffer[1].u = 0.0f;
388  m_vertex_buffer[1].v = 0.0f;
389  m_vertex_buffer[2].u = 0.0f;
390  m_vertex_buffer[2].v = (float)m_tex_height;
391  m_vertex_buffer[3].u = (float)m_tex_width;
392  m_vertex_buffer[3].v = (float)m_tex_height;
393 }

426 
427  // Can now set the meshes in the geom active.
428  p_xbox_geom->SetActive( true );
429  }
430  }

439 bool CXboxImposterGroup::plat_update_imposter_polygon( void )
440 {
441  // Calculate the new vector from bounding box midpoint to camera.
442  Mth::Vector new_vec( NxXbox::EngineGlobals.cam_position.x - m_composite_bbox_mid[X],
443  NxXbox::EngineGlobals.cam_position.y - m_composite_bbox_mid[Y],
444  NxXbox::EngineGlobals.cam_position.z - m_composite_bbox_mid[Z] );
445  new_vec.Normalize();
446 
447  // Calculate the old vector from bounding box midpoint to camera when the imposter was created.
448  Mth::Vector old_vec = m_cam_pos - m_composite_bbox_mid;
449  old_vec.Normalize();
450 
451  float angle_change = acosf( Mth::DotProduct( new_vec, old_vec ));
452 
453  // Rebuild the imposter polygon if the angle change is beyond some limit.
454  if( angle_change > Mth::DegToRad( 5.0f ))
455  {
456  RemoveImposterPolygon();
457  CreateImposterPolygon();
458  return true;
459  }
460 
461  // Check intermittently to see if the projected screen area of the imposter has changed sufficienty
462  // to warrant regenerating a new texture.
463  if( m_update_count >= XBOX_IMPOSTER_UPDATE_LIMIT )
464  {
465  m_update_count = 0;
466 
467  // Generate a camera matrix that will point the camera directly at the midpoint of the bounding box.
468  XGMATRIX cam_mat;
469  XGVECTOR3 look_at( m_composite_bbox_mid[X], m_composite_bbox_mid[Y], m_composite_bbox_mid[Z] );
470  XGMatrixLookAtRH( &cam_mat, &NxXbox::EngineGlobals.cam_position, &look_at, &NxXbox::EngineGlobals.cam_up );
471 
472  // Using this camera matrix, project all eight corners of the bounding box, in order to determine the best size for the texture.
473  Mth::Vector proj_max_x, proj_max_y, proj_mid;
474  frustum_project_box( m_composite_bbox, &cam_mat, &proj_max_x, &proj_max_y, &proj_mid );
475 
476  // Calculate the maximum width and height at the near plane.
477  float wnp = 2.0f * proj_max_x[X] * ( NxXbox::EngineGlobals.near_plane / fabsf( proj_max_x[Z] ));
478  float hnp = 2.0f * proj_max_y[Y] * ( NxXbox::EngineGlobals.near_plane / fabsf( proj_max_y[Z] ));
479 
480  // Round texture to the nearest 16 pixel limit.
481  int tex_width = 16 + Ftoi_ASM(( 640.0f * wnp ) / NxXbox::EngineGlobals.near_plane_width );
482  int tex_height = 16 + Ftoi_ASM(( 480.0f * hnp ) / NxXbox::EngineGlobals.near_plane_height );
483  tex_width = ( tex_width + 0x0F ) & ~0x0F;
484  tex_height = ( tex_height + 0x0F ) & ~0x0F;
485 
486  // Clamp texture to maximum allowed size.
487  tex_width = ( tex_width > XBOX_IMPOSTER_MAX_U_TEXTURE_SIZE ) ? XBOX_IMPOSTER_MAX_U_TEXTURE_SIZE : tex_width;
488  tex_height = ( tex_height > XBOX_IMPOSTER_MAX_V_TEXTURE_SIZE ) ? XBOX_IMPOSTER_MAX_V_TEXTURE_SIZE : tex_height;
489 
490  if(( tex_width != m_tex_width ) || ( tex_height != m_tex_height ))
491  {
492  RemoveImposterPolygon();
493  CreateImposterPolygon();
494  return true;
495  }
496  }
497 
498  return false;
499 }

522 float CXboxImposterGroup::plat_check_distance( void )
523 {
524  // First check the visibility, using the bounding sphere.
525  bool visible = NxXbox::frustum_check_sphere((D3DXVECTOR3*)&( m_composite_bbox_mid[X] ), m_composite_bsphere_radius );
526 
527  if( !visible )
528  return 0.0f;
529 
530  float min_x = m_composite_bbox.GetMin().GetX();
531  float min_y = m_composite_bbox.GetMin().GetY();
