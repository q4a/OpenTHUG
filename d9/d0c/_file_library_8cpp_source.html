148 bool CFileLibrary::PostLoad( bool assertOnFail, int file_size )
149 {
150  // Handle end of async, if that was used
151  if ( mp_fileHandle )
152  {
153  File::CAsyncFileLoader::sClose( mp_fileHandle );
154  mp_fileHandle = NULL;
155  }
156 
157  Dbg_MsgAssert( (file_size & 0x3) == 0, ( "Size of file is not multiple of 4 (%d bytes)", file_size ) );
158 
159 #ifdef __PLAT_NGC__
160 #define XFER_SIZE ( sizeof( SLibHeader ) + ( vMAX_LIB_FILES * sizeof( SFileInfo ) ) )
161  char buffer[ XFER_SIZE ];
162 
163  NsDMA::toMRAM( buffer, m_aramOffset, XFER_SIZE );
164 
165  uint8 *pFileData = (uint8*)buffer;
166 #else
167  uint8 *pFileData = (uint8*)mp_fileBuffer;
168 #endif // __PLAT_NGC__
169 
170  SLibHeader* pHeader = (SLibHeader*)pFileData;
171 
172  pFileData += sizeof( SLibHeader );
173 #ifndef __PLAT_NGC__
174  if ( pFileData > ( mp_fileBuffer + file_size ) )
175  {
176  // out of bounds
177  goto load_fail;
178  }
179 #endif // __PLAT_NGC__
180 
181  m_numFiles = pHeader->numFiles;
182 
183 #ifndef __PLAT_NGC__
184  Dbg_MsgAssert( m_numFiles > 0, ( "No files found in lib mp_fileBuffer = %p", mp_fileBuffer ) );
185  Dbg_MsgAssert( m_numFiles < vMAX_LIB_FILES, ( "Too many subfiles found in lib (%d files, max=%d)", m_numFiles, vMAX_LIB_FILES ) );
186 #endif // __PLAT_NGC__
187 
188  // read in table of contents information
189  memcpy( &m_fileInfo, pFileData, (sizeof(SFileInfo) * m_numFiles) );
190  pFileData += (sizeof(SFileInfo) * m_numFiles);
191 #ifndef __PLAT_NGC__
192  if ( pFileData > ( mp_fileBuffer + file_size ) )
193  {
194  // out of bounds
195  goto load_fail;
196  }
197 #endif // __PLAT_NGC__
198 
199  for ( int i = 0; i < m_numFiles; i++ )
200  {
201 #ifdef __PLAT_NGC__
202  m_fileOffsets[i] = (uint32)(m_aramOffset + m_fileInfo[i].fileOffset);
203  mp_filePointers[i] = NULL;
204 #else
205  mp_filePointers[i] = (uint32*)(mp_fileBuffer + m_fileInfo[i].fileOffset);
206 #endif // __PLAT_NGC__
207  Dbg_Message( "File %d @ %08x %s [%d bytes]", i, m_fileInfo[i].fileNameChecksum, Script::FindChecksumName(m_fileInfo[i].fileExtensionChecksum), m_fileInfo[i].fileSize );
208  }
209 
210 #ifndef __PLAT_NGC__
211  // to reduce the amount of temp memory needed, we first load the LIB file
212  // on the bottom up heap, and then copy sub-file individually onto the top
213  // down heap (as we reallocate shrink the original LIB file on the bottom up heap)
214  for ( int i = m_numFiles - 1; i >= 0; i-- )
215  {
216  int file_size = m_fileInfo[i].fileSize;
217 
218  uint32* pTempFilePointer = mp_filePointers[i];
219 
220  // GJ: When I first wrote this class, I wanted to make it generic,
221  // but it gradually became more cutscene-specific to fix
222  // memory issues. For example, the following uses the
223  // "cutscene heap" to reduce temp memory overhead. Perhaps
224  // later on, I can just store the appropriate top/bottomheap pointers
225  // during initialization
226  Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneTopDownHeap(), ( "No cutscene heap?" ) );
227  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneTopDownHeap());
228  mp_filePointers[i] = (uint32*)Mem::Malloc( file_size );
229  memcpy( mp_filePointers[i], pTempFilePointer, file_size );
230  Mem::Manager::sHandle().PopContext();
231 
232 #ifdef __NOPT_ASSERT__
233  uint8* pOldBuffer = mp_baseBuffer;
234 #endif
235 
236  Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneBottomUpHeap(), ( "No cutscene heap?" ) );
237  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneBottomUpHeap());
238  mp_baseBuffer = (uint8*)Mem::ReallocateShrink( Mem::GetAllocSize(mp_baseBuffer) - file_size, mp_baseBuffer );
239  Dbg_MsgAssert( mp_baseBuffer == pOldBuffer, ( "Pointer was not supposed to change except for shrinking" ) );
240  Mem::Manager::sHandle().PopContext();
241  }
242 
243  // by this point, all that should be left in the original block
244  // is the header, which we don't need any more
245  Mem::Free( mp_baseBuffer );
246  mp_baseBuffer = NULL;
247  mp_fileBuffer = NULL;
248 #endif // __PLAT_NGC__
249 
250  // if the data is ready to be accessed
251  m_dataLoaded = true;
252 
253  return true;
254 
255 #ifndef __PLAT_NGC__
256 load_fail:
257  Dbg_MsgAssert( 0, ( "Parsing of library failed" ) );
258  return false;
259 #endif // __PLAT_NGC__
260 }

267 bool CFileLibrary::Load( const char* p_fileName, bool assertOnFail, bool async_load )
268 {
269 #ifndef __PLAT_NGC__
270  // See Garrett, Garrett, or Mick for explanation.
271 
272  Dbg_MsgAssert( !async_load, ( "Cutscenes aren't supposed to be async loaded anymore" ) );
273  int file_size;
274 
275  Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneBottomUpHeap(), ( "No cutscene heap?" ) );
276  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneBottomUpHeap());
277 
278  mp_baseBuffer = (uint8*)LoadAlloc( p_fileName, &file_size );
279  mp_fileBuffer = mp_baseBuffer;
280 
281  Mem::Manager::sHandle().PopContext();
282 
283  return PostLoad(assertOnFail, file_size);
284 #endif
285 
286 #ifdef __PLAT_NGC__
287  // Make sure music is off here as we really need the memory.
288 // Pcm::StopMusic();
289 
290  async_load = false;
291 #else
292  // Turn off async for platforms that don't support it
293  if (!File::CAsyncFileLoader::sAsyncSupported())
294  {
295  async_load = false;
296  }
297 
298  if ( async_load )
299  {
300  // Pre-load
301  Dbg_Assert( !mp_fileHandle );
302 
303  mp_fileHandle = File::CAsyncFileLoader::sOpen( p_fileName, !async_load );
304 
305  // make sure the file is valid
306  if ( !mp_fileHandle )
307  {
308  Dbg_MsgAssert( assertOnFail, ( "Load of %s failed - file not found?", p_fileName ) );
309  return false;
310  }
311 
312  int file_size = mp_fileHandle->GetFileSize();
313 
314  if ( file_size == 0 )
315  {
316  Dbg_MsgAssert( 0, ("Library file %s size is 0 (possibly not found?)", p_fileName) );
317  File::CAsyncFileLoader::sClose( mp_fileHandle );
318  mp_fileHandle = NULL;
319  return false;
320  }
321 
322  Dbg_Assert( !mp_fileBuffer );
323  Dbg_Assert( !mp_baseBuffer );
324 
325  // to reduce the amount of temp memory needed, we first load the LIB file
326  // on the bottom up heap, and then copy sub-file individually onto the top
327  // down heap (as we reallocate shrink the original LIB file on the bottom up heap)
328  Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneBottomUpHeap(), ( "No cutscene heap?" ) );
329  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneBottomUpHeap());
330  mp_baseBuffer = (uint8*) Mem::Malloc( file_size + 64 );
331  mp_fileBuffer = (uint8*)(((uint32)mp_baseBuffer+63)&~63);
332  mp_fileBuffer = mp_baseBuffer; // Garrett: De-aligning pointer since async filesystem should take care of it.
333  Mem::Manager::sHandle().PopContext();
334 
335  // Set the callback
336  mp_fileHandle->SetCallback( filesync_async_callback, (unsigned int)this, (unsigned int)assertOnFail );
337 
338  // read the file in
339  mp_fileHandle->Read( mp_fileBuffer, 1, file_size );
340 
341  return true;
342  }
343  else
344 #endif // __PLAT_NGC__
345  {
346  // open the file as a stream
347  void* pStream = File::Open( p_fileName, "rb" );
348 
349  // make sure the file is valid
350  if ( !pStream )
351  {
352  Dbg_MsgAssert( assertOnFail, ("Load of %s failed - file not found?", p_fileName) );
353  return false;
354  }
355 
356  int file_size = File::GetFileSize( pStream );
357  Dbg_MsgAssert( file_size, ( "Library file size is 0" ) );
358 
359 #ifndef __PLAT_NGC__
360  Dbg_Assert( !mp_fileBuffer );
361  Dbg_Assert( !mp_baseBuffer );
362 #endif // __PLAT_NGC__
363 
364 #ifdef __PLAT_NGC__
365  // Allocate ARAM
366  m_aramOffset = NsARAM::alloc( file_size, NsARAM::TOPDOWN );
367 
368 #define BUFFER_SIZE ( 16 * 1024 )
369 
370  // read the file in blocks & transfer to ARAM
371  char buffer[BUFFER_SIZE];
372  int size = file_size;
373  uint32 offset = m_aramOffset;
374 
375  while ( size > BUFFER_SIZE )
376  {
377  File::Read( buffer, 1, BUFFER_SIZE, pStream );
378  NsDMA::toARAM( offset, buffer, BUFFER_SIZE );
379 
380  size -= BUFFER_SIZE;
381  offset += BUFFER_SIZE;
382  }
383 
384  // Read & transfer last bit.
385  if ( size )
386  {
387  File::Read( buffer, 1, size, pStream );
388  NsDMA::toARAM( offset, buffer, size );
389  }
390 // mp_baseBuffer = (uint8*)Mem::Malloc( file_size );
391 // mp_fileBuffer = mp_baseBuffer;
392 #else
393  // to reduce the amount of temp memory needed, we first load the LIB file
394  // on the bottom up heap, and then copy sub-file individually onto the top
395  // down heap (as we reallocate shrink the original LIB file on the bottom up heap)
396  Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneBottomUpHeap(), ( "No cutscene heap?" ) );
397  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneBottomUpHeap());
398  mp_baseBuffer = (uint8*)Mem::Malloc( file_size );
399  mp_fileBuffer = mp_baseBuffer;
400  Mem::Manager::sHandle().PopContext();
401 
402  // read the file in
403  if ( !File::Read( mp_fileBuffer, 1, file_size, pStream ) )
404  {
405  Mem::Free( mp_fileBuffer );
406  mp_baseBuffer = NULL;
407  mp_fileBuffer = NULL;
408  File::Close( pStream );
409 
410  Dbg_MsgAssert( assertOnFail, ( "Load of %s failed - bad format?", p_fileName ) );
411  return false;
412  }
413 #endif // __PLAT_NGC__
414 
415  File::Close(pStream);
416 
417  return PostLoad(assertOnFail, file_size);
418  }
419 }

436 uint32* CFileLibrary::GetFileData( int index )
437 {
438  Dbg_MsgAssert( index >= 0 && index < m_numFiles, ( "Out of range file index %d", index ) );
439 
440 #ifdef __PLAT_NGC__
441  if ( !mp_filePointers[index] )
442  {
443  // Allocate memory & copy over.
444 // Dbg_MsgAssert( Mem::Manager::sHandle().CutsceneTopDownHeap(), ( "No cutscene heap?" ) );
445  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().CutsceneTopDownHeap());
446 
447  int mem_available;
448  bool need_to_pop = false;
449  int size = m_fileInfo[index].fileSize;
450 
451  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().AudioHeap() );
452  mem_available = Mem::Manager::sHandle().Available();
453  if ( size < ( mem_available - ( 15 * 1024 ) ) )
454  {
455  need_to_pop = true;
456  }
457  else
458  {
459  Mem::Manager::sHandle().PopContext();
460  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().FrontEndHeap() );
461  mem_available = Mem::Manager::sHandle().Available();
462  if ( size < ( mem_available - ( 40 * 1024 ) ) )
463  {
464  need_to_pop = true;
465  }
466  else
467  {
468  Mem::Manager::sHandle().PopContext();
469  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().ThemeHeap() );
470  mem_available = Mem::Manager::sHandle().Available();
471  if ( size < ( mem_available - ( 5 * 1024 ) ) )
472  {
473  need_to_pop = true;
474  }
475  else
476  {
477  Mem::Manager::sHandle().PopContext();
478  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().ScriptHeap() );
479  mem_available = Mem::Manager::sHandle().Available();
480  if ( size < ( mem_available - ( 40 * 1024 ) ) )
481  {
482  need_to_pop = true;
483  }
484  else
485  {
486  Mem::Manager::sHandle().PopContext();
487  }
488  }
489  }
490  }
491 
492  mp_filePointers[index] = (uint32*)Mem::Malloc( m_fileInfo[index].fileSize );
493  NsDMA::toMRAM( mp_filePointers[index], m_fileOffsets[index], m_fileInfo[index].fileSize );
494 
495  if ( need_to_pop )
496  {
497  Mem::Manager::sHandle().PopContext();
498  }
499 
500  Mem::Manager::sHandle().PopContext();
501  }
502 #endif // __PLAT_NGC__
503 
504  return mp_filePointers[index];
505 }
