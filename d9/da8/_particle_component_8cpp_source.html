151 
152  orientation[Mth::AT] = vel;

162  }

339 // The component's Update() function is called from the CCompositeObject's
340 // Update() function. That is called every game frame by the CCompositeObjectManager
341 // from the s_logic_code function that the CCompositeObjectManager registers
342 // with the task manger.
343 void CParticleComponent::Update()
344 {
345  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
346 
347  // If it's a limited-time particle system in net games (like fireballs) never suspend the
348  // logic
349  if (!mp_suspend_component || !mp_suspend_component->SkipLogic() ||
350  ( gamenet_man->InNetGame() && ( m_system_lifetime > 0 )))
351  {
352  // ** You would put in here the stuff that you would want to get run every frame
353  // ** for example, a physics type component might update the position and orientation
354  // ** and update the internal physics state
355  if( m_update_script )
356  {
357  Script::RunScript( m_update_script, NULL, GetObject() );
358  }
359 
360  if( m_system_lifetime > 0 )
361  {
362  if(( Tmr::GetTime() - m_birth_time ) > m_system_lifetime )
363  {
364  GetObject()->MarkAsDead();
365  return;
366  }
367  }
368  // If we are using local coordinates, update all the world parameters in the
369  // particle instance.
370  if (mp_particle->GetParameters()->m_LocalCoord)
371  {
372  for (int i = 0; i < Nx::vNUM_BOXES; i++)
373  {
374  Mth::Vector world_pos(mp_particle->GetParameters()->m_LocalBoxPos[i] + GetObject()->m_pos );
375  //Mth::Vector world_pos(mp_particle->GetParameters()->m_LocalBoxPos[i] + Mth::Vector(100.0f, 100.0f, 0.0f));
376 
377  mp_particle->SetBoxPos(i, &world_pos);
378  }
379  }
380  }
381 }
