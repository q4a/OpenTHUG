69 // structure used to record the collision of a single triangle with a 2D object; a collision test for a 2D object will generate a set of these
70 struct S2DCollSurface
71 {
72  struct S2DCollEndpoint
73  {
74  Mth::Vector point; // endpoint position
75  Mth::Vector tangent; // normalized tangent vector to the endpoint
76  char tangent_exists; // if a tangent of the endpoint is defined
77  };
78 
79  S2DCollEndpoint ends[2]; // endpoints of the line cut through the triangle by the collision
80 
81  Mth::Vector normal; // normal of the collided triangle
82 
83  uint32 node_name;
84  char trigger;
85 };

87 struct CollData
88 {
89  SCollSurface surface;
90  float dist;
91  uint16 ignore_1; // bitmask, ignore poly if flags matching tis mask are 1
92  uint16 ignore_0; // ditto, but for flags that are 0
93  uint16 flags;
94  ETerrainType terrain;
95  CCollObj *p_coll_object; // the sector (object) with which we collided
96  uint32 node_name; // checsum of name of node associated with this object
97  uint32 script; // set to the script checksum of the object (or NULL if none)
98  void (*callback)( CollData* p_col_data); // a callback called every frame
99  void *p_callback_data; // pointer to some data the callback can use
100 
101  Obj::CCompositeObject *mp_callback_object; // set only during callbacks
102 
103  char coll_found;
104  char trigger; // true if it was a trigger (if mFD_TRIGGER flag is set on the colliding face)
105 };
