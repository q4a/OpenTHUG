195 bool CModelLights::SetBrightness(float brightness)
196 {
197  m_brightness_target = brightness;
198 
199  // We've found the target brightness, so move actual brightness towards this to avoid sudden unnatural transitions.
200  const float BRIGHTNESS_SPEED = 0.02f;
201  if( m_brightness < m_brightness_target )
202  {
203  m_brightness += BRIGHTNESS_SPEED;
204  if( m_brightness > m_brightness_target )
205  {
206  m_brightness = m_brightness_target;
207  }
208  }
209  else if( m_brightness > m_brightness_target )
210  {
211  m_brightness -= BRIGHTNESS_SPEED;
212  if( m_brightness < m_brightness_target )
213  {
214  m_brightness = m_brightness_target;
215  }
216  }
217 
218  m_ambient_brightness = 1.0f + ((2.0f * m_brightness) - 1.0f) * CLightManager::s_world_lights.m_ambient_light_modulation_factor;
219  if (m_ambient_brightness < 0.0f)
220  {
221  m_ambient_brightness = 0.0f;
222  }
223 
224  for (int i = 0; i < CLightManager::MAX_LIGHTS; i++)
225  {
226  m_diffuse_brightness[i] = 1.0f + ((2.0f * m_brightness) - 1.0f) * CLightManager::s_world_lights.m_diffuse_light_modulation_factor[i];
227  if (m_diffuse_brightness[i] < 0.0f)
228  {
229  m_diffuse_brightness[i] = 0.0f;
230  }
231  }
232 
233  // need to call this whenever the color/brightness changes
234  plat_update_brightness();
235 
236  return true;
237 }

244 void CModelLights::UpdateBrightness()
245 {
246  m_ambient_brightness = 1.0f + ((2.0f * m_brightness) - 1.0f) * CLightManager::s_world_lights.m_ambient_light_modulation_factor;
247  if (m_ambient_brightness < 0.0f)
248  {
249  m_ambient_brightness = 0.0f;
250  }
251 
252  for (int i = 0; i < CLightManager::MAX_LIGHTS; i++)
253  {
254  m_diffuse_brightness[i] = 1.0f + ((2.0f * m_brightness) - 1.0f) * CLightManager::s_world_lights.m_diffuse_light_modulation_factor[i];
255  if (m_diffuse_brightness[i] < 0.0f)
256  {
257  m_diffuse_brightness[i] = 0.0f;
258  }
259  }
260 
261  // GJ: the normal PS2 lighting code assumes that the collision
262  // code will be run once a frame to copy over the m_ambient_base_color
263  // into the m_ambient_mod_color. unfortunately, the cutscene objects'
264  // ref objects don't go through this code, so I had to add this function
265  // to explicitly copy over the data... there's probably a cleaner way
266  // to do it, but I didn't want to break the existing code.
267 
268  // need to call this whenever the color/brightness changes
269  plat_update_brightness();
270 }
