31 // NOTES:
32 
33 // A CModelAppearance contains one script structure, whose
34 // contents look something like this:
35 // {
36 // head = { desc_id=#"Andrew Reynolds" h = 0 s = 50 v = 100 use_default_hsv = 1 }
37 // torso = { desc_id=#"Long Sleeve - Collar" h = 0 s = 50 v = 100 use_default_hsv = 1 }
38 // legs = { desc_id=#"Reynolds's Pants" h = 0 s = 50 v = 100 use_default_hsv = 1 }
39 // shoes = { desc_id=#"Reynolds's" h = 0 s = 50 v = 100 use_default_hsv = 1 }
40 // boardup = { desc_id=#"Solid" }
41 // boarddown = { desc_id=#"Green Brand Logo" h = 0 s = 50 v = 100 use_default_hsv = 1 }
42 // scale = 1.02
43 // weight_scale = 0.97
44 // }
45 
46 // Basically, it's a bunch of "virtual structures", each
47 // assigned to a "part checksum" (head, torso, etc.).
48 // Each part checksum corresponds to the name of a global
49 // array containing a list of read-only "actual structures".
50 // Given a part checksum and a desc_id in the CModelAppearance,
51 // we can lookup the appropriate global array and find the
52 // appropriate actual structure.
53 
54 // CModelAppearance is really just a wrapper around the
55 // CStruct class, but after some consideration,
56 // I decided not to subclass from it. This is so that
57 // I'd be able to replace the underlying implementation
58 // while keeping the same interface.
59 
60 // There shouldn't be anything skater- (or even player-)
61 // specific hardcoded here. The intent is that we use
62 // this same class for any kind of customizable model
63 // (such as Create-a-Peds, Create-a-Cars, or even
64 // Create-a-Spaceships for future games).
65 
66 // The CModelAppearance should have no knowledge of the full
67 // list of possible part checksums; this list should be
68 // completely in script. This implies that the code should
69 // never have to iterate through a list of part checksums,
70 // (unlike THPS3, which had many explicit references to
71 // "editable_cas_options", Cas::GetBodyPartCount(), and
72 // Cas::GetBodyPartName()).
73 
74 // One of the reset types used to be "randomized", but
75 // that is something that is more appropriate at the skater
76 // profile level (or, even better, in script), which has
77 // a better understanding of what parts should be disqualified
78 // from working with a particular skater instance. If
79 // randomization were to be implemented at this level, it should
80 // be a purely naive randomization (no weighting, no part
81 // disqualification).

131 void CModelAppearance::resolve_randomized_desc_ids()
132 {
133  // for each structure component, see whether
134  // it's got any randomized desc ids. if so,
135  // select one to be used every time this
136  // appearance instance is used
137 
138  // sometimes the skin tone must be consistent
139  // among all the body parts... if any structure
140  // contains a "random_set" checksum, then
141  // make sure all future random parts get
142  // this selected as well
143  uint32 random_set = 0;
144 
145  Script::CComponent* p_comp = m_appearance.GetNextComponent( NULL );
146  while ( p_comp )
147  {
148  Script::CComponent* p_next = m_appearance.GetNextComponent( p_comp );
149 
150  if ( p_comp->mType == ESYMBOLTYPE_STRUCTURE )
151  {
152  uint32 partChecksum = p_comp->mNameChecksum;
153  int randomIndex;
154 
155  if ( p_comp->mpStructure->ContainsFlag( CRCD(0xd81c03b0,"randomized_desc_id") ) )
156  {
157  Script::CStruct* pActualStruct = Cas::GetRandomOptionStructure( partChecksum, random_set );
158  Dbg_MsgAssert( pActualStruct, ( "Unrecognized part checksum to randomize %s", Script::FindChecksumName(partChecksum) ) );
159 
160  // remember the random_set, if any
161  pActualStruct->GetChecksum( CRCD(0x0d7260fd,"random_set"), &random_set, Script::NO_ASSERT );
162 
163  p_comp->mpStructure->Clear();
164  p_comp->mpStructure->AddChecksum( CRCD(0x4bb2084e,"desc_id"), get_desc_id_from_structure(pActualStruct) );
165  }
166  else if ( p_comp->mpStructure->GetInteger( CRCD(0x4b833e64,"random_index"), &randomIndex, Script::NO_ASSERT ) )
167  {
168  Script::CStruct* pActualStruct = Cas::GetRandomOptionStructureByIndex( partChecksum, randomIndex, random_set );
169  Dbg_MsgAssert( pActualStruct, ( "Unrecognized part checksum to randomize %s", Script::FindChecksumName(partChecksum) ) );
170 
171  // remember the random_set, if any
172  pActualStruct->GetChecksum( CRCD(0x0d7260fd,"random_set"), &random_set, Script::NO_ASSERT );
173 
174  p_comp->mpStructure->Clear();
175  p_comp->mpStructure->AddChecksum( CRCD(0x4bb2084e,"desc_id"), get_desc_id_from_structure(pActualStruct) );
176  }
177  }
178 
179  p_comp = p_next;
180  }
181 }

198 void CModelAppearance::set_part(uint32 partChecksum, uint32 descID, Script::CStruct* pParams)
199 {
200  Script::CStruct* pStruct;
201 
202  // if the structure doesn't already exist, then add it...
203  if ( !m_appearance.GetStructure(partChecksum, &pStruct) )
204  {
205  pStruct = new Script::CStruct;
206  m_appearance.AddComponent(partChecksum, ESYMBOLTYPE_STRUCTUREPOINTER, (int)pStruct);
207  }
208 
209  pStruct->Clear();
210 
211  // check for extra color parameters
212  int use_default_hsv = 1;
213  pParams->GetInteger( CRCD(0x97dbdde6,"use_default_hsv"), &use_default_hsv, Script::NO_ASSERT );
214 
215  if ( !use_default_hsv )
216  {
217  int h, s, v;
218  pParams->GetInteger( CRCD(0x6e94f918,"h"), &h, Script::ASSERT );
219  pParams->GetInteger( CRCD(0xe4f130f4,"s"), &s, Script::ASSERT );
220  pParams->GetInteger( CRCD(0x949bc47b,"v"), &v, Script::ASSERT );
221 
222  pStruct->AddInteger( CRCD(0x97dbdde6,"use_default_hsv"), 0 );
223  pStruct->AddInteger( CRCD(0x6e94f918,"h"), h );
224  pStruct->AddInteger( CRCD(0xe4f130f4,"s"), s );
225  pStruct->AddInteger( CRCD(0x949bc47b,"v"), v );
226  }
227 
228  pStruct->AddChecksum( CRCD(0x4bb2084e,"desc_id"), descID );
229 }

355 bool CModelAppearance::Load( Script::CStruct* pStructure, bool resolve_randoms )
356 {
357  Dbg_Assert( pStructure );
358 
359  // In THPS3, we didn't clear the m_appearance first when
360  // loading from a memory card, so that if the saved data is
361  // missing anything, it won't matter because the default
362  // will already be in m_appearance. (This prevents asserts
363  // on autoloading when a new component has been added to
364  // m_appearance which is not present on the memory card)
365  // However, for THPS4, the appearances should be a lot more
366  // flexible, and should no longer fail when there are missing items.
367  Init();
368 
369 #if 1
370  // in case there are any global structure names,
371  // resolve them
372  pStructure->ExpandInto( &m_appearance, 0 );
373 #else
374  // add the new desired data
375  m_appearance.AppendStructure( pStructure );
376 #endif
377 
378  // at this point, all the randomized_desc_ids
379  // should be resolved
380  if ( resolve_randoms )
381  {
382  resolve_randomized_desc_ids();
383  }
384 
385 #ifdef __NOPT_ASSERT__
386  // Make sure that the m_appearance does not contain any flag members,
387  // which would potentially cause leaks as we append any structures.
388  uint32 dummy = 0;
389  Dbg_MsgAssert( !m_appearance.GetChecksum( NONAME, &dummy ), ( "m_appearance contains a flag '%s'", Script::FindChecksumName(dummy) ) );
390 #endif
391 
392  return true;
393 }

443 uint32 CModelAppearance::WriteToBuffer(uint8 *pBuffer, uint32 BufferSize, bool ignoreFaceData )
444 {
445  // for network message building
446  // (note that the face texture is not included in this buffer
447  // because the buffer is too small... face textures will
448  // be sent in a separate net packet)
449  Script::CStruct* pTempStructure = new Script::CStruct;
450  pTempStructure->AppendStructure( &m_appearance );
451  uint32 size = Script::WriteToBuffer(pTempStructure, pBuffer, BufferSize);
452 
453 #ifndef __PLAT_NGC__
454  // GJ: Need to compress the model appearances for Xbox and PS2
455  // versions, or else the worst-case model appearance will crash
456  // the server if he quits his own game (because it overflows the
457  // max net packet size). MAX_MODEL_APPEARANCE_SIZE is a guess
458  // based on the existing model appearance size, but it really
459  // depends on how big the info/tricks get... 700 bytes seems safe
460  const uint32 MAX_MODEL_APPEARANCE_SIZE = 700;
461  if ( size > MAX_MODEL_APPEARANCE_SIZE )
462  {
463  // if the model appearance is too big, then strip out
464  // some of the cosmetic stuff...
465  compress_model_appearance( pTempStructure );
466  size = Script::WriteToBuffer(pTempStructure, pBuffer, BufferSize);
467  }
468 #endif
469 
470  delete pTempStructure;
471 
472  // skip to the next chunk
473  pBuffer += size;
474  BufferSize -= size;
475 
476  // we do, however, still need to send a flag that says it
477  // will eventually have a face texture... so that the model
478  // builder knows to use the face-mapped head rather than
479  // the non-facemapped head
480  if( ignoreFaceData )
481  {
482 
483  *pBuffer = 0;
484  }
485  else
486  {
487  *pBuffer = ( ( GetFaceTexture() && GetFaceTexture()->IsValid() ) || m_willEventuallyHaveFaceTexture ) ? 1 : 0;
488  }
489 
490  pBuffer += 1;
491  BufferSize -= 1;
492  size += 1;
493 
494  return size;
495 }

658 bool CModelAppearance::CallMemberFunction( uint32 checksum, Script::CStruct* pParams, Script::CScript* pScript )
659 {
660  switch ( checksum )
661  {
662  case 0x0a23400c: // ClearPart
663  {
664  Dbg_MsgAssert(pParams, ("No params supplied to model appearance"));
665 
666  uint32 partChecksum;
667  pParams->GetChecksum( CRCD(0xb6f08f39,"part"), &partChecksum, Script::ASSERT );
668  clear_part(partChecksum);
669  }
670  return true;
671 
672  case 0x83339d0b: // SetPart
673  {
674  Dbg_MsgAssert(pParams, ("No params supplied to model appearance"));
675 
676  uint32 partChecksum;
677  pParams->GetChecksum( CRCD(0xb6f08f39,"part"), &partChecksum, Script::ASSERT );
678 
679  uint32 descChecksum;
680  pParams->GetChecksum( CRCD(0x4bb2084e,"desc_id"), &descChecksum, Script::ASSERT );
681 
682  set_part(partChecksum, descChecksum, pParams);
683  }
684  return true;
685 
686  case 0x10a225d6: // GetPart
687  {
688  Dbg_MsgAssert(pParams, ("No params supplied to model appearance"));
689 
690  uint32 partChecksum;
691  pParams->GetChecksum( CRCD(0xb6f08f39,"part"), &partChecksum, Script::ASSERT );
692 
693  Script::CStruct* pVirtualDescStructure = GetVirtualDescStructure( partChecksum );
694 
695  if ( pVirtualDescStructure )
696  {
697  // return all the parameters, including desc_id, use_default_hsv, h, s, v
698  pScript->GetParams()->AppendStructure( pVirtualDescStructure );
699  return true;
700  }
701 
702  return false;
703  }
704  return true;
705 
706  case 0xd27427ff: // SetChecksum
707  {
708  Dbg_MsgAssert(pParams, ("No params supplied to model appearance"));
709 
710  uint32 fieldChecksum;
