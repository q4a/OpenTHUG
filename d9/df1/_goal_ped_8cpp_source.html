47 inline void parse_first_name( char* pFirstName, const char** ppDisplayName, int buffer_size )
48 {
49  if ( !strstr( *ppDisplayName, " " ) )
50  {
51  Dbg_MsgAssert( strlen( *ppDisplayName ) < (uint32)buffer_size, ( "buffer overflow" ) );
52  strcpy( pFirstName, *ppDisplayName );
53  // printf("first name has no spaces, using %s\n", pFirstName);
54  }
55  else
56  {
57  char *p_in = pFirstName;
58  int length = 0;
59  while ( **ppDisplayName != ' ' )
60  {
61  *p_in++ = **ppDisplayName;
62  (*ppDisplayName)++;
63  length++;
64  }
65  *p_in++ = '\0';
66  length++;
67  Dbg_MsgAssert( length < buffer_size, ( "buffer overflow" ) );
68  }
69 }

76 const char* CGoalPed::GetProFirstName()
77 {
78  // get params
79  Script::CStruct* pParams = GetGoalParams();
80 
81  // grab the pro
82  const char* p_default_pro_name;
83  const char* p_alternate_pro_name;
84  if ( !pParams->GetText( CRCD(0x944b2900,"pro"), &p_default_pro_name ) )
85  {
86  // printf("no pro associated with this goal\n");
87  return 0;
88  }
89 
90  // check what pro the player is using
91  Mdl::Skate * pSkate = Mdl::Skate::Instance();
92  Obj::CSkaterProfile* pSkaterProfile = pSkate->GetCurrentProfile();
93  const char* p_current_pro_name;
94  p_current_pro_name = pSkaterProfile->GetDisplayName();
95  // printf("the current display name is %s\n", p_current_pro_name);
96 
97  // only check the display name if it's a pro
98  bool use_alternate = false;
99  if ( pSkaterProfile->IsPro() )
100  {
101  // chop off the first name
102  char current_pro_first_name[128];
103  parse_first_name( current_pro_first_name, &p_current_pro_name, 128 );
104 
105  Dbg_MsgAssert( strlen( current_pro_first_name ) > 0, ( "No first name found for the current pro" ) );
106  // printf("the player's first name is %s\n", current_pro_first_name);
107 
108  // see if we should use the alternate pro
109  if ( pSkaterProfile->IsPro() && ( Script::GenerateCRC( current_pro_first_name ) == Script::GenerateCRC( p_default_pro_name ) ) )
110  {
111  if ( pParams->GetText( CRCD(0x878ccc12,"alternate_pro"), &p_alternate_pro_name ) )
112  {
113  use_alternate = true;
114  // printf("found an alternate name of %s\n", p_alternate_pro_name);
115  }
116  else
117  {
118  // printf("no alternate pro found for this goal\n");
119  return 0;
120  }
121  }
122  }
123 
124  // build the stream name
125  if ( use_alternate )
126  return p_alternate_pro_name;
127  else
128  return p_default_pro_name;
129 }

209  char p_diff_stream_name[MAX_STREAM_NAME_LENGTH];

260  sprintf( p_diff_cam_speaker_stream_name, "%s_f", p_diff_cam_speaker_stream_name );

262  possibly_add_success_suffix( success, p_diff_cam_speaker_stream_name, MAX_STREAM_NAME_LENGTH );
263  }
264  }
265 
266  char p_diff_speaker_stream_name[MAX_STREAM_NAME_LENGTH];
267  strcpy( p_diff_speaker_stream_name, "" );
268  char p_speaker_stream_name[MAX_STREAM_NAME_LENGTH];
269  strcpy( p_speaker_stream_name, "" );
270  if ( p_speaker_name )
271  {
272  #ifdef __NOPT_ASSERT__
273  int speaker_name_length = strlen( p_speaker_name );
274  // diff level and speaker name
275  Dbg_MsgAssert( diff_stream_name_length + speaker_name_length + 1 < MAX_STREAM_NAME_LENGTH, ( "buffer overflow" ) );
276  #endif
277  sprintf( p_diff_speaker_stream_name, "%s_%s", p_diff_stream_name, p_speaker_name );
278  if ( is_female_skater )
279  {
280  Dbg_MsgAssert( strlen( p_diff_speaker_stream_name ) + 2 < MAX_STREAM_NAME_LENGTH, ( "buffer overflow" ) );
281  sprintf( p_diff_speaker_stream_name, "%s_f", p_diff_speaker_stream_name );
282  }
283  possibly_add_success_suffix( success, p_diff_speaker_stream_name, MAX_STREAM_NAME_LENGTH );
284 
285  // speaker name
286  Dbg_MsgAssert( stream_name_length + speaker_name_length + 1 < MAX_STREAM_NAME_LENGTH, ( "buffer overflow" ) );
287  sprintf( p_speaker_stream_name, "%s_%s", p_stream_name, p_speaker_name );
288  if ( is_female_skater )
289  {
290  Dbg_MsgAssert( strlen( p_speaker_stream_name ) + 2 < MAX_STREAM_NAME_LENGTH, ( "buffer overflow" ) );
291  sprintf( p_speaker_stream_name, "%s_f", p_speaker_stream_name );
292  }
293  possibly_add_success_suffix( success, p_speaker_stream_name, MAX_STREAM_NAME_LENGTH );
294  }
295 
296  // now that we're done using the diff stream and stream name
297  // as pieces, add the success suffix
298  possibly_add_success_suffix( success, p_diff_stream_name, MAX_STREAM_NAME_LENGTH );
299  possibly_add_success_suffix( success, p_stream_name, MAX_STREAM_NAME_LENGTH );
300 
301  // array of pointers to my streams to make the code more readable
302  // 0 - p_stream_name
303  // 1 - p_diff_stream_name
304  // 2 - p_diff_cam_stream_name
305  // 3 - p_cam_stream_name
306  // 4 - p_cam_speaker_stream_name
307  // 5 - p_diff_cam_speaker_stream_name
308  // 6 - p_diff_speaker_stream_name
309  // 7 - p_speaker_stream_name
310  char* pp_stream_names[8];
311  pp_stream_names[0] = p_stream_name;
312  pp_stream_names[1] = p_diff_stream_name;
313  pp_stream_names[2] = p_speaker_stream_name;
314  pp_stream_names[3] = p_diff_speaker_stream_name;
315  pp_stream_names[4] = p_diff_cam_stream_name;
316  pp_stream_names[5] = p_cam_stream_name;
317  pp_stream_names[6] = p_cam_speaker_stream_name;
318  pp_stream_names[7] = p_diff_cam_speaker_stream_name;
319 
320  // search!
321  // if there's a cam anim index defined, don't search for
322  // ones without cam anim index
323  int start_index = 0;
324  if ( cam_anim_index >= 0 )
325  {
326  start_index = 4;
327  }
328 
329  // search for regular or lip version of stream
330  uint32 existing_stream = 0;
331  bool found_lip = false;
332  for ( int i = start_index; i < 8; i++ )
333  {
334  // make sure there's something to check
335  int stream_name_length = strlen( pp_stream_names[i] );
336  if ( stream_name_length == 0 )
337  {
338  continue;
339  }
340 
341  // test lip version first - it's more likely to happen
342  char p_lip_stream[MAX_STREAM_NAME_LENGTH];
343  Dbg_MsgAssert( (uint32)stream_name_length + 4 < MAX_STREAM_NAME_LENGTH, ( "buffer overflow" ) );
344  sprintf( p_lip_stream, "%s_lip", pp_stream_names[i] );
345  // printf("looking for %s\n", p_lip_stream );
346  uint32 lip_stream_checksum = Script::GenerateCRC( p_lip_stream );
347  if ( Pcm::StreamExists( lip_stream_checksum ) )
348  {
349  // printf("Existing stream: %s\n", p_lip_stream);
350  existing_stream = lip_stream_checksum;
351  found_lip = true;
352  break;
353  }
354  else
355  {
356  uint32 stream_checksum = Script::GenerateCRC( pp_stream_names[i] );
357  // printf("looking for %s\n", pp_stream_names[i]);
358  if ( Pcm::StreamExists( stream_checksum ) )
359  {
360  // printf( "Existing stream: %s\n", pp_stream_names[i] );
361  existing_stream = stream_checksum;
362  break;
363  }
364  }
365  }
366  *p_streamChecksum = existing_stream;
367  return found_lip;
368 }

844 {

847 
848  if ( m_goalLogicSuspended )
849  return;
850 
851  // stop any current vo's

853 
854  // get the first name of the speaker and the player

869  // printf("got a first name of %s\n", p_player_name);
870

913  PlayGoalStream( Script::GenerateCRC( stream_name ), use_pos_info );
