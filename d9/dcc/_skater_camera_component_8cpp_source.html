281 void CSkaterCameraComponent::Update( void )
282 {
283  if( mpSkater == NULL )
284  {
285  return;
286  }
287 
288  // optimization KLUDGE
289  if (mpSkaterPhysicsControlComponent && mpSkaterPhysicsControlComponent->IsDriving())
290  {
291  GetObject()->Pause(true);
292  return;
293  }
294 
295  // We need a CCameraComponent attached in order to get camera details.
296  CCameraComponent *p_cam_comp = GetCameraComponentFromObject( GetObject());
297  Dbg_MsgAssert( p_cam_comp, ( "CSkaterCameraComponent needs CCameraComponent attached to parent" ));
298 
299  // This used to be passed in as a param for the old CSkaterCam::Update().
300  // Now we get it from a flag in the skater
301  bool instantly = false;
302 
303  Dbg_Assert( mpSkater );
304  if (mpSkater->HasTeleported())
305  {
306  instantly = true;
307  }
308 
309  if( instantly )
310  {
311  mInstantCount = 3;
312  }
313 
314  if( mInstantCount > 0 )
315  {
316  --mInstantCount;
317  instantly = true;
318  }
319 
320 
321 // if (instantly)
322 // {
323 // printf ("%d: Instantly!\n",(int)Tmr::GetRenderFrame());
324 // }
325 
326  // Length of last frame in seconds.
327  float delta_time = Tmr::FrameLength();
328 
329  // Update any values we are currently interpolating.
330  UpdateInterpolators();
331 
332  // Would have used a SetPos() function, but the camera update in rwviewer.cpp is spread out all over the place,
333  // so it's not easy to do. We'll just store the old position before updating anything...
334 // Gfx::Camera::StoreOldPos();
335  p_cam_comp->StoreOldPosition();
336 
337 // Mth::Matrix& frame_matrix = Gfx::Camera::GetMatrix();
338  Mth::Matrix& frame_matrix = p_cam_comp->GetMatrix();
339 
340  frame_matrix[X] = mLastActualRight;
341  frame_matrix[Y] = mLastActualUp;
342  frame_matrix[Z] = mLastActualAt;
343 
344  Mth::Vector current_at_vector( frame_matrix[Z][X], frame_matrix[Z][Y], frame_matrix[Z][Z] );
345 
346  // Obtain skater state.
347  EStateType state = mpSkaterStateComponent->GetState();
348  Mth::Matrix target;
349 
350  if( state == WALL )
351  {
352 # ifdef DEBUG_CAMERA
353  printf ("SkaterCam Using mLastPreWallSkaterMatrix, as wallriding\n");
354 # endif
355  target = mLastPreWallSkaterMatrix;
356  }
357  else
358  {
359 # ifdef DEBUG_CAMERA
360  printf ("SkaterCam Using GetCameraDisplayMatrix\n");
361 # endif
362  // target = static_cast< CCompositeObject* >(mpSkater)->GetDisplayMatrix();
363  if (mpSkater->IsLocalClient())
364  {
365  target = mpSkater->GetDisplayMatrix();
366  }
367  else
368  {
369  target = static_cast< CCompositeObject* >(mpSkater)->GetDisplayMatrix();
370  }
371  mLastPreWallSkaterMatrix = target;
372  }
373 
374  // Lerp towards the required lean.
375  float cam_lean = 0.0f;
376  if( state == RAIL && mpSkater->IsLocalClient())
377  {
378  // Need to translate the lean value [-4096, 4096] into a reasonable camera lean angle.
379  cam_lean = GetSkaterBalanceTrickComponentFromObject(mpSkater)->mGrind.mManualLean * SKATERCAMERACOMPONENT_LEAN_TO_SKATERCAM_LEAN;
380  }
381 
382  if( cam_lean != mLean )
383  {
384  if( instantly )
385  {
386  mLean = cam_lean;
387  }
388  else
389  {
390  mLean += (( cam_lean - mLean ) * mGrindLerp );
391  }
392  }
393 
394  Mth::Vector skater_up;
395 
396  if (state != GROUND)
397  {
398  skater_up = target[Y];
399  }
400  else
401  {
402  skater_up = mpSkaterStateComponent->GetCameraDisplayNormal();
403  }
404 
405 // Gfx::AddDebugLine( p_skater->m_pos, ( target[X] * 72.0f ) + p_skater->m_pos, MAKE_RGB( 128, 0, 0 ), MAKE_RGB( 128, 0, 0 ), 4 );
406 // Gfx::AddDebugLine( p_skater->m_pos, ( target[Y] * 72.0f ) + p_skater->m_pos, MAKE_RGB( 0, 128, 0 ), MAKE_RGB( 0, 128, 0 ), 4 );
407 // Gfx::AddDebugLine( p_skater->m_pos, ( target[Z] * 72.0f ) + p_skater->m_pos, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 4 );
408 
409  // Use vel for forward facing, if moving in xz plane.
410  Mth::Vector vel_xz( mpSkater->GetVel().GetX(), 0.0f, mpSkater->GetVel().GetZ());
411 
412  bool use_vert_cam = UseVertCam();
413 
414  // use velocity if physics not paused, and if we are moving fast enough, or if we are doing vert cam
415  // or if we are doing spine physics
416  if(
417  !(mpSkaterPhysicsControlComponent && mpSkaterPhysicsControlComponent->IsPhysicsSuspended()) && (vel_xz.Length() > 0.1f || use_vert_cam || mpSkaterStateComponent->GetFlag( SPINE_PHYSICS ))
418  )
419  {
420  if( use_vert_cam )
421  {
422  // If it's vert, then set target direction to near straight down.
423  target[Z].Set( 0.0f, -1.0f, 0.0f );
424 
425  // Zero lookaround when in vert air (and not in lookaround locked or override mode, or doing a lip trick).
426  if( !mp_lookaround_component->mLookaroundLock && !mp_lookaround_component->mLookaroundOverride && ( state != LIP ))
427  {
428  mp_lookaround_component->mLookaroundTilt = 0.0f;
429  mp_lookaround_component->mLookaroundHeading = 0.0f;
430  }
431  }
432  else
433  {
434  target[Z] = mpSkater->GetVel();
435 
436 # if 1 // for use with non-straight spine transfer physics
437  if( mpSkaterStateComponent->GetFlag( SPINE_PHYSICS ))
438  {
439  // Just use the up velocity, plus the velocity over the spine will be straight down when coming down the other side.
440  target[Z][X] = 0.0f;
441  target[Z][Z] = 0.0f;
442  target[Z] += mpSkaterStateComponent->GetSpineVel();
443  }
444 # endif
445 
446  target[Z].Normalize();
447 
448 // printf ("Using Velocity (%f, %f, %f)\n", target[Z][X],target[Z][Y], target[Z][Z]);
449 
450 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Z] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 128, 128, 128 ), MAKE_RGB( 128, 128, 128 ), 4 );
451 
452  // Flatten out the velocity component perpendicular to the ground. However, if the skater is in the air, the
453  // m_last_normal will contain the last ground normal, which we don't want, so set it to be 'up'.
454  float dot;
455  Mth::Vector normal;
456 
457  if( state == AIR )
458  {
459  // Patch for spine physics
460  // only reduce component perpendicular to plane by 60%
461  // so we still get some up/down camera movmeent
462  // only do this when moving down though, otherwise it looks poo
463  if (mpSkaterStateComponent->GetFlag(SPINE_PHYSICS) && target[Z][Y] < 0.0f)
464  {
465  normal.Set( 0.0f, 1.0f, 0.0f );
466  dot = 0.7f * Mth::DotProduct(target[Z],normal); // change this to 0.8 to look down when coming down other side of a spine
467  }
468  else
469  {
470  normal.Set( 0.0f, 1.0f, 0.0f );
471  dot = target[Z].GetY();
472  }
473  // Set world up as the target vector.
474  target[Y].Set( 0.0f, 1.0f, 0.0f );
475  }
476  else if( state == WALL )
477  {
478  normal.Set( 0.0f, 1.0f, 0.0f );
479  dot = target[Z].GetY();
480  }
481  else
482  {
483  normal = mpSkaterStateComponent->GetCameraCurrentNormal();
484  dot = 0.8f * Mth::DotProduct(target[Z],normal);
485  }
486 
487  target[Z] -= normal * dot;
488  target[Z].Normalize();
489 
490 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Z] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 256, 256, 256 ), MAKE_RGB( 256, 256, 256 ), 4 );
491  }
492 
493  // We need to orthonormalize in a specific order, as when the velocity is going backwards, then
494  // the X orient will be wrong, and so plonk the skater upside down.
495  target[X] = Mth::CrossProduct( target[Y], target[Z] );
496  target[X].Normalize();
497 
498  target[Y] = Mth::CrossProduct( target[Z], target[X] );
499  target[Y].Normalize();
500 
501 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[X] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 128, 0, 0 ), MAKE_RGB( 128, 0, 0 ), 4 );
502 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Y] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 0, 128, 0 ), MAKE_RGB( 0, 128, 0 ), 4 );
503 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Z] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 4 );
504  }
505 
506  if (mpSkater->IsLocalClient() && mpSkaterStateComponent->GetFlag(IN_ACID_DROP))
507  {
508  CSkaterCorePhysicsComponent* pSkaterPhysicsComponent = GetSkaterCorePhysicsComponentFromObject(mpSkater);
509  Dbg_Assert(pSkaterPhysicsComponent);
510 
511  target = pSkaterPhysicsComponent->m_acid_drop_camera_matrix;
512  }
513 
514  // Tilt further if in regular air (going off a big jump, for example), or doing a lip trick.
515  if( state == AIR )
516  {
517  mTiltAddition += TILT_INCREMENT * Tmr::FrameRatio();
518  if( mTiltAddition > TILT_MAX )
519  {
520  mTiltAddition = TILT_MAX;
521  }
522  }
523  else if( mTiltAddition > 0.0f )
524  {
525  mTiltAddition -= TILT_RESTORE * Tmr::FrameRatio();
526  if( mTiltAddition < 0.0f )
527  {
528  mTiltAddition = 0.0f;
529  }
530  }
531  else if( mTiltAddition < 0.0f )
532  {
533  mTiltAddition += TILT_RESTORE * Tmr::FrameRatio();
534  if( mTiltAddition > 0.0f )
535  {
536  mTiltAddition = 0.0f;
537  }
538  }
539 
540  // PJR: Needed to do this to fix a compiler crash on NGC. Hopefully SN will
541  // fix this & I can revert this hack...
542  target.RotateYLocal( mp_lookaround_component->mLookaroundHeading + mp_lookaround_component->mLookaroundHeadingStartingPoint );
543 
544  // Now tilt the matrix down a bit.
545  if( state == LIP )
546  {
547 // target.RotateYLocal( 0.8f );
548 // target.RotateXLocal( mLipTrickTilt + mTiltAddition );
549  _xrotlocal ( target, mLipTrickTilt + mTiltAddition );
550  }
551  else if( !(mpSkaterStateComponent->GetFlag( VERT_AIR ) && !mpSkaterStateComponent->GetFlag( CAN_BREAK_VERT ) && !mpSkaterStateComponent->GetFlag(SPINE_PHYSICS) ) /*|| !mpSkaterStateComponent->GetFlag(IN_ACID_DROP)*/)
552  {
553 // target.RotateXLocal( mTilt + mTiltAddition );
554  _xrotlocal ( target, mTilt + mTiltAddition );
555  }
556 
557  // Now adjust for 'lookaround'.
558 // target.RotateXLocal( mLookaroundTilt );
559  _xrotlocal( target, mp_lookaround_component->mLookaroundTilt );
560 
561  static float lip_rotate = 0.0f;
562 
563  if( state == LIP )
564  {
565  lip_rotate = Mth::PI / 2.0f;
566 
567  if( mLipSideDecided == 0 )
568  {
569  target.RotateY( lip_rotate );
570 
571  // Do collision check.
572  CFeeler feeler;
573 
574  Mth::Vector horiz_z( target[Z][X], 0.0f, target[Z][Z] );
575  horiz_z.Normalize();
576  Mth::Vector a = mpSkater->m_pos - ( target[Z] * 3.0f );
577  Mth::Vector b = a - ( horiz_z * 72.0f );
578 
579  feeler.SetIgnore(IGNORE_FACE_FLAGS_1, IGNORE_FACE_FLAGS_0);
580  bool collision = feeler.GetCollision(a, b, true);
581  if( !collision )
582  {
583  // Side 1 is fine.
584  mLipSideDecided = 1;
585  }
586  else
587  {
588  float dist = feeler.GetDist();
589  target.RotateY( -lip_rotate * 2.0f );
590  horiz_z.Set( target[Z][X], 0.0f, target[Z][Z] );
591  horiz_z.Normalize();
592  a = mpSkater->m_pos - ( target[Z] * 3.0f );
593  b = a - ( horiz_z * 72.0f );
594 
595  collision = feeler.GetCollision(a, b, true);
596  if( !collision )
597  {
598  // Side 2 is fine.
599  mLipSideDecided = 2;
600  }
601  else
602  {
603  if( feeler.GetDist() < dist )
604  {
605  // Side 2 is better than side 1.
606  mLipSideDecided = 2;
607  }
608  else
609  {
610  // Side 1 is better than side 2.
611  target.RotateY( lip_rotate * 2.0f );
612  mLipSideDecided = 1;
613  }
614  }
615  }
616  }
617  else
618  {
619  if( mLipSideDecided == 1 )
620  {
621  target.RotateY( lip_rotate );
622  }
623  else
624  {
625  target.RotateY( -lip_rotate );
626  }
627  }
628  }
629  else
630  {
631  mLipSideDecided = 0;
632  }
633 
634  // Set skater flag to indicate when the liptrick camera has spun to the opposite side.
635  if( mpSkater && ( mLipSideDecided == 2 ))
636  {
637  mpSkater->mScriptFlags |= ( 1 << Script::GetInteger( 0x16b8e4cb /* "FLAG_SKATER_LIPTRICK_CAM_REVERSED" */ ));
638  }
639  else
640  {
641  mpSkater->mScriptFlags &= ~( 1 << Script::GetInteger( 0x16b8e4cb /* "FLAG_SKATER_LIPTRICK_CAM_REVERSED" */ ));
642  }
643 
644 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[X] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 128, 0, 0 ), MAKE_RGB( 128, 0, 0 ), 4 );
645 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Y] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 0, 128, 0 ), MAKE_RGB( 0, 128, 0 ), 4 );
646 // Gfx::AddDebugLine( mpSkater->m_pos, ( target[Z] * 72.0f ) + mpSkater->m_pos, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 4 );
647 
648  // Why doing this, when setting below?
649  Mth::Matrix t_matrix(0.0f, 0.0f, 0.0f);
650 
651  // Since camera points in -Z, but player in +Z, we must negate the X and Z axes
652  t_matrix[X][X] = -target[0][0];
653  t_matrix[X][Y] = -target[0][1];
654  t_matrix[X][Z] = -target[0][2];
655  t_matrix[X][W] = 0.0f;
656 
657  t_matrix[Y][X] = target[1][0];
658  t_matrix[Y][Y] = target[1][1];
659  t_matrix[Y][Z] = target[1][2];
660  t_matrix[Y][W] = 0.0f;
661 
662  t_matrix[Z][X] = -target[2][0];
663  t_matrix[Z][Y] = -target[2][1];
664  t_matrix[Z][Z] = -target[2][2];
665  t_matrix[Z][W] = 0.0f;
666 
667  t_matrix[W][X] = 0.0f;
668  t_matrix[W][Y] = 0.0f;
669  t_matrix[W][Z] = 0.0f;
670  t_matrix[W][W] = 1.0f;
671 
672  frame_matrix[W][X] = 0.0f;
673  frame_matrix[W][Y] = 0.0f;
674  frame_matrix[W][Z] = 0.0f;
675  frame_matrix[W][W] = 1.0f;
676 
677  // if we want an instant update, then just move it
678  if (instantly)
679  {
680  frame_matrix = t_matrix;
681  }
682 
683  float dotx = Mth::DotProduct( t_matrix[X], frame_matrix[X] );
684  float doty = Mth::DotProduct( t_matrix[Y], frame_matrix[Y] );
685  float dotz = Mth::DotProduct( t_matrix[Z], frame_matrix[Z] );
686 
687  if( dotx > CAMERA_SLERP_STOP && doty > CAMERA_SLERP_STOP && dotz > CAMERA_SLERP_STOP )
688  {
689  // Do nothing, camera has reached the target so we just leave the frame matrix exactly where it is.
690 // frame_matrix = t_matrix;
691  }
692  else
693  {
694  // Initialise the SLERP interpolator, with the current matrix as the start, and the ideal
695  // matrix as the end (target).
696 
697  // GARY: is this right? i assume p_matrix = &t_matrix
698  mpSlerp->setMatrices( &frame_matrix, &t_matrix );
699 
700  // Copy flags and shit (why copying all this when most of the values will be overwritten?)
701  Mth::Matrix stored_frame;
702  stored_frame = frame_matrix;
703  frame_matrix = t_matrix;
704 
705  // Continue to slerp towards the target, updating frame_matrix with the slerped values.
706  if( !(mpSkaterStateComponent->GetFlag( VERT_AIR ) && !mpSkaterStateComponent->GetFlag( CAN_BREAK_VERT ) && !mpSkaterStateComponent->GetFlag(SPINE_PHYSICS)))
707  {
708  // If the skater has just landed from big air, use the big air land slerp value.
709  if(( mpSkaterStateComponent->GetState() == GROUND ) && ( mVertAirLandedTimer >= delta_time ))
710  {
711  mVertAirLandedTimer -= delta_time;
712  mpSlerp->getMatrix( &frame_matrix, GetTimeAdjustedSlerp( mVertAirLandedSlerp, delta_time ));
713  }
714  else
715  {
716  // Clear the vert air landed timer.
717  mVertAirLandedTimer = 0.0f;
718 
719  mpSlerp->getMatrix( &frame_matrix, GetTimeAdjustedSlerp( mSlerp, delta_time ));
720 
721  // At this point we can check to see the angle that the camera's 'at' vector moved through this frame.
722  // If this angle is too large, we need to limit it, regardless of the slerp. (This is primarily to avoid
723  // the nasty jerk when transitioning rails etc.
724  float this_dot = ( current_at_vector.GetX() * frame_matrix[Mth::AT][X] )
725  + ( current_at_vector.GetY() * frame_matrix[Mth::AT][Y] )
726  + ( current_at_vector.GetZ() * frame_matrix[Mth::AT][Z] );
727 
728  // Dot values will be in the range [0,1], with 1 being no change.
729  if( this_dot < ( mLastDot * 0.9998f ))
730  {
731  // The angular change this frame is too big. Need to recalculate with an adjusted slerp value.
732  float new_slerp = mSlerp * ( this_dot / ( mLastDot * 0.9998f ));
733  mpSlerp->setMatrices( &stored_frame, &t_matrix );
734  mpSlerp->getMatrix( &frame_matrix, GetTimeAdjustedSlerp( new_slerp, delta_time ));
735 
736  this_dot = mLastDot * 0.9998f;
737  }
738  mLastDot = this_dot;
739  }
740  }
741  else
742  {
743  mpSlerp->getMatrix( &frame_matrix, GetTimeAdjustedSlerp( mVertAirSlerp, delta_time ));
744 
745  // Set the vert air landed timer to the max, since the skater is in vert air.
746  mVertAirLandedTimer = VERT_AIR_LANDED_TIME;
747  }
748  }
749 
750  // Mick: This is a patch to allow the walk camera to override the updating of the camera frame
751  // At this point, frame_matrix is valid to store.
752  mLastActualRight = frame_matrix[X];
753  mLastActualUp = frame_matrix[Y];
754  mLastActualAt = frame_matrix[Z];
755 
756  // Set camera position to be the same as the skater.
757  Mth::Vector cam_pos = GetTripodPos( instantly );
758 
759  // If in the air doing a trick, we slowly reduce the behind value to 'zoom' in on the skater.
760  float above, behind;
761  if( instantly )
762  {
763  float temp = mZoomLerp;
764  mZoomLerp = 1.0f; // fully lerp instantly
765  CalculateZoom( &above, &behind );
766  mZoomLerp = temp;
767  }
768  else
769  {
770  CalculateZoom( &above, &behind );
771  }
772 
773  // Note we use the camera's at vector, as this will keep the skater in the middle of the frame.
774  //cam_pos -= frame_matrix[Z] * behind;
775  cam_pos += frame_matrix[Z] * behind;
776 
777  // Move camera along the Skater's up vector.
778  cam_pos += skater_up * above;
779 
780  Mth::Vector actual_focus_pos = mpSkater->m_pos + ( skater_up * above );
781 
782 // Gfx::AddDebugLine( mpSkater->m_pos, actual_focus_pos, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 4 );
783 
784 // Dbg_Message("Matrix Z (%5.4f,%5.4f,%5.4f)", frame_matrix[Z][X], frame_matrix[Z][Y], frame_matrix[Z][Z]);
785 // Dbg_Message("Matrix UP (%5.4f,%5.4f,%5.4f)", skater_up[X], skater_up[Y], skater_up[Z]);
786 // Dbg_Message("Above %f Behind %f", above, behind);
787 // Dbg_Message("focus pos (%5.1f,%5.1f,%5.1f)", actual_focus_pos[X], actual_focus_pos[Y], actual_focus_pos[Z]);
788 // Dbg_Message("Camera after: pos (%5.1f,%5.1f,%5.1f)", cam_pos[X], cam_pos[Y], cam_pos[Z]);
789 
790  // Reorient the camera to look directly at the skater. We don't want to look at the interpolated position,
791  // since this can lead to nasty jerkiness, especially on rails etc.
792  target[Z].Set( actual_focus_pos.GetX() - cam_pos.GetX(), actual_focus_pos.GetY() - cam_pos.GetY(), actual_focus_pos.GetZ() - cam_pos.GetZ());
793  target[Z].Normalize();
794 
795  // Read back the Y from the current matrix.
796  target[Y][0] = frame_matrix[Y][X];
797  target[Y][1] = frame_matrix[Y][Y];
798  target[Y][2] = frame_matrix[Y][Z];
799 
800  // Generate new orthonormal X and Y axes.
801  target[X] = Mth::CrossProduct( target[Y], target[Z] );
802  target[X].Normalize();
803  target[Y] = Mth::CrossProduct( target[Z], target[X] );
804  target[Y].Normalize();
805 
806  // Here is where lean may safely be applied without moving the focus position on screen.
807  if( mLean != 0.0f )
808  {
809  target[Y].Rotate( target[Z], mLean );
810  target[X].Rotate( target[Z], mLean );
811  }
812 
813  // Write back into camera matrix.
814  // Since camera points in -Z, but player in +Z, we must negate the X and Z axes
815  frame_matrix[X][X] = -target[0][0];
816  frame_matrix[X][Y] = -target[0][1];
817  frame_matrix[X][Z] = -target[0][2];
818  frame_matrix[X][W] = 0.0f;
819 
820  frame_matrix[Y][X] = target[1][0];
821  frame_matrix[Y][Y] = target[1][1];
822  frame_matrix[Y][Z] = target[1][2];
823  frame_matrix[Y][W] = 0.0f;
824 
825  frame_matrix[Z][X] = -target[2][0];
826  frame_matrix[Z][Y] = -target[2][1];
827  frame_matrix[Z][Z] = -target[2][2];
828  frame_matrix[Z][W] = 0.0f;
829 
830  // Update the position if there is a shake active.
831  if( mShakeDuration > 0.0f )
832  {
833  addShake( cam_pos, frame_matrix );
834  }
835 
836  // Now do collision detection.
837 
838  bool no_camera_collision;
839  if (mpSkater->IsLocalClient())
840  {
841  CSkaterCorePhysicsComponent* pSkaterPhysicsComponent = GetSkaterCorePhysicsComponentFromObject(mpSkater);
842  Dbg_Assert(pSkaterPhysicsComponent);
843 
844  no_camera_collision = pSkaterPhysicsComponent->mp_movable_contact_component->GetContact() && state == LIP;
845  }
846  else
847  {
848  no_camera_collision = false;
849  }
850 
851  if (no_camera_collision)
852  {
853  // no camera collision if doing a lip trick on a moving object
854  //printf ("Lip trick on moving object\n");
855  }
856  else
857  {
858  ApplyCameraCollisionDetection(cam_pos, frame_matrix, cam_pos - ((Mth::Vector)( frame_matrix[Z] )) * behind, actual_focus_pos);
859  }
860 
861  cam_pos[W] = 1.0f;
862 // Gfx::Camera::SetPos( cam_pos );
863  p_cam_comp->SetPosition( cam_pos );
864 
865  // reset old position if in instant update
866  if (instantly)
867  {
868  p_cam_comp->StoreOldPosition();
869  }
870 
871 // Dbg_Message("Camera Final: pos (%5.1f,%5.1f,%5.1f)", cam_pos[X], cam_pos[Y], cam_pos[Z]);
872 
873  // Store the position of the camera in CSkaterCam also (used for proximity nodes).
874 
875  // Dave - there is no m_pos anymore...
876 // m_pos = cam_pos;
877 
878  // Dave - this used to return the value below when it was CSkaterCam::Update()
879 // Mth::Vector to_target = at_pos - cam_pos;
880 // return to_target.Length();
881 }
