95 void CVibrationComponent::Update()
96 {
97  Dbg_Assert(mp_input_device);
98 
99  if (!m_active_state)
100  {
101  Suspend(true);
102  return;
103  }
104 
105  // check the duration of active actuators
106  for (int i = vVB_NUM_ACTUATORS; i--; )
107  {
108  // actuator inactive or has no associated duration
109  if (!mp_vibration_timers[i].active) continue;
110 
111  // duration not yet up
112  if (Tmr::ElapsedTime(mp_vibration_timers[i].start_time) < mp_vibration_timers[i].duration) continue;
113 
114  // NOTE: ignoring all replay code in VibrationComponent for now
115  // Replay::WritePadVibration(i, 0);
116 
117  // turn off actuator
118  mp_input_device->ActivateActuator(i, 0);
119 
120  mp_vibration_timers[i].active = false;
121  }
122 }

129 CBaseComponent::EMemberFunctionResult CVibrationComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
130 {
131  switch ( Checksum )
132  {
133  // @script | Vibrate |
134  // @parm int | Actuator | actuator num
135  // @parmopt float | Percent | 0 |
136  // @flag Off | turn off both actuators
137  // @parmopt float | Duration | 0 | time to vibrate
138  case CRCC(0xdf8a5f0a, "Vibrate"):
139  {
140  // NOTE: ignoring pause state in VibrationComponent for now
141  // Don't let it be switched on if skater is paused
142  if (!pParams->ContainsFlag(CRCD(0xd443a2bc, "Off")) && (Mdl::FrontEnd::Instance()->GamePaused() || GetObject()->IsPaused())) break;
143 
144  Dbg_MsgAssert(mp_input_device, ("No associated input device"));
145 
146  if (!m_active_state) break;
147 
148  int actuator_idx = 0;
149  pParams->GetInteger(CRCD(0x114fc131, "actuator"), &actuator_idx);
150  Dbg_MsgAssert(actuator_idx >= 0 && actuator_idx < vVB_NUM_ACTUATORS, ("%s\nActuator must be between 0 and %d", pScript->GetScriptInfo(), vVB_NUM_ACTUATORS));
151 
152  float percent = 0;
153  pParams->GetFloat(CRCD(0x9e497fc6, "percent"), &percent);
154  Dbg_MsgAssert(percent >= 0 && percent <= 100, ("\n%s\nPercent must be between 0 and 100", pScript->GetScriptInfo()));
155 
156  if (pParams->ContainsFlag("off"))
157  {
158  // This will switch off vibration, including if the game is paused.
159  // Ie, unpausing won't restore the old vibration. Hooray!
160  mp_input_device->StopAllVibrationIncludingSaved();
161 
162  for (int i = vVB_NUM_ACTUATORS; i--; )
163  {
164  mp_vibration_timers[i].active = false;
165  }
166 
167  // NOTE: ignoring all replay code in VibrationComponent for now
168  // for (int i = vVB_NUM_ACTUATORS; i--; )
169  // {
170  // Replay::WritePadVibration(i, 0);
171  // }
172  }
173  else
174  {
175  // NOTE: ignoring all replay code in VibrationComponent for now
176  // Replay::WritePadVibration(Actuator, (int)Percent);
177  mp_input_device->ActivateActuator(actuator_idx, static_cast<int>(percent));
178  }
179 
180  // record duration so that we can later turn off the actuator; a duration of 0 gives an unbounded duration
181  float duration = 0;
182  if (pParams->GetFloat(CRCD(0x79a07f3f, "duration"), &duration))
183  {
184  mp_vibration_timers[actuator_idx].active = true;
185  mp_vibration_timers[actuator_idx].duration = static_cast<Tmr::Time>(1000.0f * duration);
186  mp_vibration_timers[actuator_idx].start_time = Tmr::GetTime();
187  }
188  else
189  {
190  mp_vibration_timers[actuator_idx].active = false;
191  }
192 
193  break;
194  }
195 
196  default:
197  return CBaseComponent::MF_NOT_EXECUTED;
198  }
199 
200  return CBaseComponent::MF_TRUE;
201 }
