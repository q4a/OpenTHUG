84 struct SLogBufferInfo
85 {
86  // Pointer to the start of the big log buffer, which will be in the debug heap somewhere.
87  SLogEntry *mpBuffer;
88  // The total number of entries in the buffer.
89  int mTotalEntries;
90  // The number of entries written to so far. This starts at 0 when the game starts,
91  // and increases till it hits mTotalEntries, then stays at that value.
92  int mNumEntries;
93 
94  // Points to just after the last entry that got added to the buffer.
95  // So it may not point to a valid SLogEntry, for example when the buffer is filling
96  // up, or when pTop points to the end of pBuffer.
97  SLogEntry *mpTop;
98 };

121 void Init()
122 {
123 
124  if (Config::GotExtraMemory())
125  {
126  Dbg_MsgAssert(!sInitialised,("Tried to call Log::Init twice"));
127 
128  Dbg_MsgAssert((uint32)(_log_info_end-_log_info_start) >= sizeof(SLogBufferInfo),("log_info section too small, got=%d, required=%d",(uint32)(_log_info_end-_log_info_start),sizeof(SLogBufferInfo)));
129  spLogInfo=(SLogBufferInfo*)_log_info_start;
130 
131  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().DebugHeap());
132 
133  spLogInfo->mNumEntries=0;
134  spLogInfo->mTotalEntries=MAX_LOG_ENTRIES;
135  spLogInfo->mpBuffer=(SLogEntry*)Mem::Malloc(MAX_LOG_ENTRIES*sizeof(SLogEntry));
136  spNotALeak=spLogInfo->mpBuffer;
137  spLogInfo->mpTop=spLogInfo->mpBuffer;
138 
139  uint32 *p_long=(uint32*)spLogInfo->mpBuffer;
140  Dbg_MsgAssert((sizeof(SLogEntry)&3)==0,("sizeof(SLogEntry) not a multiple of 4"));
141  for (uint32 i=0; i<MAX_LOG_ENTRIES*sizeof(SLogEntry)/4; ++i)
142  {
143  *p_long++=0;
144  }
145 
146  Mem::Manager::sHandle().PopContext();
147 
148  sInitialised=true;
149  }
150 }

152 void AddEntry(char *p_fileName, int lineNumber, char *p_functionName, char *p_message)
153 {
154  if (Config::GotExtraMemory())
155  {
156  Dbg_MsgAssert(sInitialised,("Log::AddEntry called before Log::Init called"));
157 
158  SLogEntry *p_new=NULL;
159  if (spLogInfo->mNumEntries < MAX_LOG_ENTRIES)
160  {
161  p_new=spLogInfo->mpTop++;
162  ++spLogInfo->mNumEntries;
163  }
164  else
165  {
166  if (spLogInfo->mpTop >= spLogInfo->mpBuffer+MAX_LOG_ENTRIES)
167  {
168  spLogInfo->mpTop=spLogInfo->mpBuffer;
169  }
170  p_new=spLogInfo->mpTop++;
171  }
172 
173  p_new->mCPUTime=Tmr::GetTimeInCPUCycles();
174  p_new->mLineNumber=lineNumber;
175  p_new->mpFunctionName=p_functionName;
176  p_new->mpSourceFileName=p_fileName;
177  if (p_message)
178  {
179  Dbg_MsgAssert(strlen(p_message)<=MAX_MESSAGE_LENGTH,("Log message '%s' too long",p_message));
180  strcpy(p_new->mpMessage,p_message);
181  }
182  else
183  {
184  p_new->mpMessage[0]=0;
185  }
186  }
187 }
