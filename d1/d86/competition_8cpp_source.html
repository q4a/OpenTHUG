99 int CLeaderBoardEntry::GetTotalScore(int round)
100 {
101  int score1 = 0;
102  int score2 = 0;
103 
104  for (int i=0;i<round+1;i++)
105  {
106  if (m_score[i] > score1)
107  {
108  score2 = score1;
109  score1 = m_score[i];
110  }
111  else
112  if (m_score[i] > score2)
113  {
114  score2 = m_score[i];
115  }
116  }
117  return score1 + score2;
118 }

123 void CCompetition::StartCompetition(float bronze, float silver, float gold, float bronze_score, float silver_score, float gold_score, float bail)
124 {
125  // store the defining parameters
126 
127  m_bronze = bronze;
128  m_silver = silver;
129  m_gold = gold;
130  m_bronze_score = bronze_score;
131  m_silver_score = silver_score;
132  m_gold_score = gold_score;
133  m_bail = bail;
134 
135 
136  m_end_competition = false;
137 
138  // Get the current profile
139  Mdl::Skate * pSkate = Mdl::Skate::Instance();
140  Obj::CPlayerProfileManager* pPlayerProfileManager=pSkate->GetPlayerProfileManager();
141  Obj::CSkaterProfile* pSkaterProfile=pPlayerProfileManager->GetCurrentProfile();
142 
143  // need to use GetDisplayName() wrapper function instead of GetUIString(), bec. ui string can now theoretically be empty
144  m_leader_board[0].m_name = pSkaterProfile->GetDisplayName();
145 
146  // Let's count how many pros there are, so we can pcik from them at random
147 
148  int possible_pros = pPlayerProfileManager->GetNumProfileTemplates();
149 
150 // get an array of pros, except the one that has the same name as me
151  int pro_order[100];
152  int num_pros = 0;
153  for (int i = 0;i<possible_pros;i++)
154  {
155 
156  Obj::CSkaterProfile* pProfile = pPlayerProfileManager->GetProfileTemplateByIndex(i);
157 
158 // printf ("Pro=%d, Secret = %d, name = %s",pProfile->IsPro(), pProfile->IsSecret(), pProfile->GetDisplayName());
159 
160  if (!pProfile->IsPro() || pProfile->IsSecret() || Script::GenerateCRC(m_leader_board[0].m_name.getString()) == Script::GenerateCRC(pProfile->GetDisplayName()))
161  {
162  // skip over this one
163  }
164  else
165  {
166  pro_order[num_pros++] = i;
167  }
168  }
169 
170  // shuffle them
171  for (int i = 0;i<num_pros;i++)
172  {
173  int other = Mth::Rnd2(num_pros);
174  int t = pro_order[i];
175  pro_order[i] = pro_order[other];
176  pro_order[other] = t;
177  }
178 
179  // we initially populate the table with entries for each skater
180  // other than skater 0, with random scores for 1st, 2nd and 3rd
181  // that are close to the requied scores
182  // and then populate the "round" scores with scores
183  // that will eventually add up to the final score
184 
185  // first determine a random order for the 7 other skaters
186 
187  int order[vMAX_PLAYERS-1];
188 
189  // set up an initial linear order
190  for (int i = 0;i<vMAX_PLAYERS-1;i++)
191  {
192  order[i] = i+1;
193  }
194 
195  // shuffle the list of players
196  for (int i = 0;i<vMAX_PLAYERS-1;i++)
197  {
198  int swap = Mth::Rnd2(vMAX_PLAYERS-1);
199  int t = order[i];
200  order[i] = order[swap];
201  order[swap] = t;
202  }
203 
204  // zero out the initial player
205 // m_leader_board[0].mp_name = "PLAYER 1";
206  m_leader_board[0].m_player_number = 0;
207  for (int round = 0; round < vNUM_ROUNDS; round++)
208  {
209  m_leader_board[0].m_score[round] = 0;
210  }
211 
212  // add in the other players in the shuffled order
213  // for the first three, we need to calculate the score they are going to get
214 
215  bronze += Mth::PlusOrMinus(2.0f);
216  silver += Mth::PlusOrMinus(2.0f);
217  gold += Mth::PlusOrMinus(2.0f);
218 
219  if (silver >= gold)
220  {
221  silver = gold - 0.1f;
222  }
223  if (bronze >= silver)
224  {
225  bronze = silver - 0.1f;
226  }
227 
228 
229  // go through all remaining players, and fill in the blanks
230  for (int i = 0;i<vMAX_PLAYERS-1;i++)
231  {
232  int player = i+1;
233  m_leader_board[player].m_player_number = order[i]; // say which player is in this position
234  // ... note, need to get the name in here.
235  // ...
236 
237  Script::CArray* pLeaderBoardNames;
238  if ( mp_params->GetArray( CRCD(0x1cc6f48d,"leader_board_names"), &pLeaderBoardNames, Script::NO_ASSERT ) )
239  {
240  Dbg_MsgAssert( pLeaderBoardNames->GetSize() == vMAX_PLAYERS - 1, ( "leader_board_names has wrong size. Needs to be %i", vMAX_PLAYERS - 1 ) );
241  m_leader_board[player].m_name = pLeaderBoardNames->GetString( i );
242  }
243  else
244  m_leader_board[player].m_name = pPlayerProfileManager->GetProfileTemplate(pPlayerProfileManager->GetProfileTemplateChecksum(pro_order[i]))->GetDisplayName();
245 
246 
247  // now calculate the score
248  float score = 0.0f;
249  if (i == 0) score = gold;
250  else if (i == 1) score = silver;
251  else if (i == 2) score = bronze;
252  else
253  {
254  // scores for the player after bronze are essentially any random number less than bronze
255  score = bronze - (bronze/8) + Mth::PlusOrMinus(bronze/8);
256  if (score >= bronze)
257  {
258  score = bronze - 0.1f;
259  }
260  }
261  // now we have the total score, we calculate the random components that
262  // will add up to that score
263  // the final sore is calculated as the average of the best of two scores
264  // So... calculate two scores that add up to twice the total score
265  // then add a third that is less than both of these
266  // and put these in random order
267  //
268  float score1 = score + Mth::PlusOrMinus(2.0f);
269  float score2 = score*2.0f - score1;
270  float lower = score1;
271  if (score2 <lower)
272  {
273  lower = score2;
274  }
275  int pos1 = Mth::Rnd2(vNUM_ROUNDS);
276  int pos2 = pos1;
277  while (pos2 == pos1)
278  {
279  pos2 = Mth::Rnd2(vNUM_ROUNDS);
280  }
281 
282  for (int round = 0;round <vNUM_ROUNDS;round++)
283  {
284  // all other scores default to numbers just below the lower of the two scores
285  // m_leader_board[player].m_score[round] = (int)(10.0f*(lower - 2.0f + Mth::PlusOrMinus(1.9f)));
286  m_leader_board[player].m_score[round] = (int)Mth::Round((lower - 2.0f + Mth::PlusOrMinus(1.9f)));
287  }
288  // finally fill in the two higher scores
289  // into the randomly chosen poistions
290  // m_leader_board[player].m_score[pos1] = (int)(score1*10.0f);
291  // m_leader_board[player].m_score[pos2] = (int)(score2*10.0f);
292  m_leader_board[player].m_score[pos1] = (int)Mth::Round(score1);
293  m_leader_board[player].m_score[pos2] = (int)Mth::Round(score2);
294  }
295 
296  printf ("\n\n");
297  // let up now print out the fruits of our efforts
298  for (int i = 0;i<vMAX_PLAYERS;i++)
299  {
300  printf ("%d: pos %d, name %12s, %3d, %3d, %3d, %3d\n",
301  i,
302  m_leader_board[i].m_player_number,
303  m_leader_board[i].m_name.getString(),
304  m_leader_board[i].m_score[0],
305  m_leader_board[i].m_score[1],
306  m_leader_board[i].m_score[2],
307  m_leader_board[i].GetTotalScore(3));
308 
309  }
310  Sort();
311 
312  const char* p_first_place_name;
313  if ( mp_params->GetString( CRCD(0xfaf19012,"first_place_name"), &p_first_place_name, Script::NO_ASSERT ) )
314  {
315  m_leader_board[0].m_name = p_first_place_name;
316  }
317 
318  printf ("\nSorted\n");
319  // let up now print out the fruits of our efforts
320  for (int i = 0;i<vMAX_PLAYERS;i++)
321  {
322  printf ("%d: pos %d, name %12s, %3d, %3d, %3d, %3d\n",
323  i,
324  m_leader_board[i].m_player_number,
325  m_leader_board[i].m_name.getString(),
326  m_leader_board[i].m_score[0],
327  m_leader_board[i].m_score[1],
328  m_leader_board[i].m_score[2],
329  m_leader_board[i].GetTotalScore(3));
330 
331  }
332 
333 // Default names for judges (dunno if we'll ever change it though...).
334 #if (ENGLISH == 0)
335  m_judge_name[0] = (char*)Script::GetLocalString( "comp_str_Judge1");
336  m_judge_name[1] = (char*)Script::GetLocalString( "comp_str_Judge2");
337  m_judge_name[2] = (char*)Script::GetLocalString( "comp_str_Judge3");
338  m_judge_name[3] = (char*)Script::GetLocalString( "comp_str_Judge4");
339  m_judge_name[4] = (char*)Script::GetLocalString( "comp_str_Judge5");
340 #else
341  m_judge_name[0] = "Judge 1";
342  m_judge_name[1] = "Judge 2";
343  m_judge_name[2] = "Judge 3";
344  m_judge_name[3] = "Judge 4";
345  m_judge_name[4] = "Judge 5";
346 #endif
347  m_current_round = -1;
348 }

357 void CCompetition::EndRun(int score)
358 {
359  // don't end a run if we're not in a comp
360  if ( m_end_competition )
361  return;
362 
363  int i;
364 
365  printf ("Ending Competition with score of %d\n",score);
366  printf ("Bronze %f, silver %f, gold %f\n",m_bronze_score,m_silver_score,m_gold_score);
367 
368  float points = 0.0f;
369  if (score < m_bronze_score)
370  {
371  points = m_bronze * score / m_bronze_score;
372  }
373  else if (score < m_silver_score)
374  {
375  points = m_bronze + (m_silver - m_bronze) * (score - m_bronze_score) / (m_silver_score - m_bronze_score);
376  }
377  else if (score < m_gold_score)
378  {
379  points = m_silver + (m_gold - m_silver) * (score - m_silver_score) / (m_gold_score - m_silver_score);
380  }
381  else
382  {
383  // we have scored above the gold score
384  // so interpolate up to perfect (100 for gold * 2)
385  points = m_gold + (100.0f - m_gold) * (score - m_gold_score) / ( m_gold_score*2 - m_gold_score);
386  }
387 
388  printf ("points calculated at %f\n",points);
389 
390  // Now we have the score, so decrement it by the bail points
391  points -= m_bail_points;
392 
393  printf ("points decremented by %f bail points to %f\n",m_bail_points,points);
394 
395 
396  // we've got the total score now (in points, for this round)
397  // so store it
398  m_total_score = points;
399 
400  if (m_total_score > 99.9f)
401  m_total_score = 99.9f;
402 
403  // we calculate a "varience", which is the maximum amount
404  // a score can be randomly varied
405  // this ranges from 10 to 0 as the actual score goes from 0 to 99.9
406 
407  float variance = ( 99.9f - m_total_score ) / 10.0f;
408 
409  // perform some initial varying by half the variance
410  // to make the score seen more random
411  m_total_score += Mth::PlusOrMinus(variance/2);
412 
413  // clamp it, just to be safe
414  if (m_total_score > 99.9f)
415  m_total_score = 99.9f;
416  if (m_total_score < 0.0f)
417  m_total_score = 0.0f;
418 
419  // and find the n Judges scores
420  // here the final score is the average of the top three scores.
421  // so we need to generate three scores that add up to the
422  // total score
423 
424 
425  m_score[0] = m_total_score + Mth::PlusOrMinus(variance);
426  if (m_score[0] > 99.85f)
427  m_score[0] = 99.9f;
428  if (m_score[0] < 0.0f)
429  m_score[0] = 0.0f;
430  m_score[1] = m_total_score + Mth::PlusOrMinus(variance);
431  if (m_score[1] > 99.85f)
432  m_score[1] = 99.9f;
433  if (m_score[1] < 0.0f)
434  m_score[1] = 0.0f;
435  m_score[2] = (m_total_score * 3.0f) - (m_score[0] + m_score[1]);
436  if (m_score[2] < 0.0f)
437  m_score[2] = 0.0f;
438  if (m_score[2] > 99.85f)
439  m_score[2] = 99.9f;
440 
441 
442  m_score[0] = (float)( (int) (m_score[0] * 10.0f)) /10.0f + 0.0001f;
443  m_score[1] = (float)( (int) (m_score[1] * 10.0f)) /10.0f + 0.0001f;
444  m_score[2] = (float)( (int) (m_score[2] * 10.0f)) /10.0f + 0.0001f;
445 
446  m_score[0] = (int)Mth::Round(m_score[0]);
447  m_score[1] = (int)Mth::Round(m_score[1]);
448  m_score[2] = (int)Mth::Round(m_score[2]);
449 
450  if ( m_score[0] > 99 )
451  m_score[0] = 99;
452  if ( m_score[1] > 99 )
453  m_score[1] = 99;
454  if ( m_score[2] > 99 )
455  m_score[2] = 99;
456 
457  // recalculate the total score
458  m_total_score = ((float)(m_score[0]) + (float)(m_score[1]) + (float)(m_score[2])) / 3;
459  m_total_score = (int)Mth::Round( m_total_score );
460  if ( m_total_score > 99 )
461  m_total_score = 99;
462  // m_judge_score[m_current_round] = (int)(10.0f * m_total_score + 0.5f);
463  m_judge_score[m_current_round] = (int)m_total_score;
464 
465  printf ("3 scores are %f,%f,%f, total is %f\n",m_score[0],m_score[1],m_score[2],m_total_score);
466 
467  // Stick the total score in the correct place in the leaderboard
468  for (i = 0;i<vMAX_PLAYERS;i++)
469  {
470  if (m_leader_board[i].m_player_number == 0)
471  {
472  m_leader_board[i].m_score[m_current_round] = m_judge_score[m_current_round];
473  }
474  }
475 
476  // and re-sort it
477  Sort();
478 
479  // re-calculate our placing
480 
481  for (i = 0;i<vMAX_PLAYERS;i++)
482  {
483  if (m_leader_board[i].m_player_number == 0)
484  {
485  m_place = i+1;
486  }
487  }
488 
489  // then fill in the remaining judge scores with scores that are lower than the
490  // lowest of these three scores
491 
492  // calculate lowest score
493  float lowest = m_score[0];
494  if (m_score[1] < lowest)
495  lowest = m_score[1];
496  if (m_score[2] < lowest)
497  lowest = m_score[2];
498 
499  printf("Lowest = %f\n",lowest);
500 
501  // add in remaining scores, in a range of (lowest - 2* variance) to (lowest)
502  for (i = 3; i< vNUM_JUDGES; i++)
503  {
504  m_score[i] = lowest - variance + Mth::PlusOrMinus(variance);
505  printf ("Score %d is %f\n",i,m_score[i]);
506  }
507 
508  // set any negative scores to 0.0f
509  // (this will not effect the three highest scores
510  for (i = 0; i< vNUM_JUDGES; i++)
511  {
512  if (m_score[i] < 0.0f)
513  {
514  m_score[i] = 0.0f;
515  }
516  }
517 
518  // Finally, shuffle the scores into random order
519  // shuffle the list of players
520  for (i = 0;i<vNUM_JUDGES-1;i++)
521  {
522  int swap = Mth::Rnd2(vNUM_JUDGES);
523  float t = m_score[i];
524  m_score[i] = m_score[swap];
525  m_score[swap] = t;
526  }
527 
528  Game::CGoalManager* pGoalManager = Game::GetGoalManager();
529  Dbg_Assert( pGoalManager );
530  if ( pGoalManager->IsInCompetition() )
531  {
532  // and let us have a look at what we come up with;
533  Script::CStruct* p_run_scores = new Script::CStruct();
534  // pack the scores in a carray to make them easy to deal with in script
535  Script::CArray* p_run_scores_array = new Script::CArray();
536  p_run_scores_array->SetSizeAndType( vNUM_JUDGES, ESYMBOLTYPE_STRUCTURE );
537  printf( "\nJudgement Day\n\n" );
538  for ( i = 0; i < vNUM_JUDGES; i++ )
539  {
540  // add an integer version of this score to the params, so it's
541  // easy to deal with in script
542  Script::CStruct* p_temp = new Script::CStruct();
543  p_temp->AddInteger( "score", (int)Mth::Round( m_score[i] ) );
544 
545  // add a flag to the top judges
546  if ( IsTopJudge( i ) )
547  {
548  p_temp->AddChecksum( NONAME, CRCD( 0x7b1b959f, "top_judge" ) );
549  }
550 
551  p_run_scores_array->SetStructure( i, p_temp );
552  printf ("%d: %f Top=%d\n",i,m_score[i],IsTopJudge(i));
553  }
554 
555  p_run_scores->AddArray( "scores", p_run_scores_array );
556 
557  // add the total score
558  // p_run_scores->AddInteger( "total_score", (int)Mth::Round( ( m_total_score * 10 ) ) );
559  p_run_scores->AddInteger( "total_score", (int)Mth::Round( ( m_total_score ) ) );
560 
561  Script::CleanUpArray( p_run_scores_array );
562  delete p_run_scores_array;
563  #ifdef __NOPT_ASSERT__
564  Script::CScript *p_script=Script::SpawnScript( "goal_comp_show_run_scores", p_run_scores );
565  p_script->SetCommentString("Spawned from CCompetition::EndRun(...)");
566  #else
567  Script::SpawnScript( "goal_comp_show_run_scores", p_run_scores );
568  #endif
569  delete p_run_scores;
570  }

579 void CCompetition::Sort()
580 {
581  // sort the list, based on total score
582 
583  int round = m_current_round;
584  // if the competition has ended (maybe early), then total up for all the rounds
585  if (CompetitionEnded())
586  {
587  round = 2;
588  }
589 
590  for (int i=0;i<vMAX_PLAYERS-1;i++)
591  {
592  int high = i;
593  for (int j=i+1; j<vMAX_PLAYERS;j++)
594  {
595  if (m_leader_board[j].GetTotalScore(round) > m_leader_board[high].GetTotalScore(round))
596  {
597  high = j;
598  }
599  // give the player the higher position if there is a tie
600  else if ( m_leader_board[j].m_player_number == 0 && m_leader_board[j].GetTotalScore(round) == m_leader_board[high].GetTotalScore(round) )
601  {
602  high = j;
603  }
604  }
605  if (high != i)
606  {
607  CLeaderBoardEntry t = m_leader_board[high];
608  m_leader_board[high] = m_leader_board[i];
609  m_leader_board[i] = t;
610  }
611  }

779 bool CCompetition::IsTopJudge(int judge)
780 {
781  int judge_order[vNUM_JUDGES];
782 
783  for (int i=0;i<vNUM_JUDGES;i++)
784  {
785  judge_order[i] = i;
786  }
787  for (int i=0;i<vNUM_JUDGES-1;i++)
788  {
789  for (int j = i;j<vNUM_JUDGES;j++)
790  {
791  if (GetJudgeScore(judge_order[i]) < GetJudgeScore(judge_order[j]))
792  {
793  int t=judge_order[i];
794  judge_order[i]=judge_order[j];
795  judge_order[j]=t;
796  }
797  }
798  }
799 
800  // if it is in the top three, then return true
801  if (judge_order[0] == judge || judge_order[1] == judge || judge_order[2] == judge)
802  {
803  return true;
804  }
805
