87 void CreatePed(CGeneralManager* p_obj_man, Script::CStruct* pNodeData)
88 {
89  CMovingObject* pPed = new CMovingObject;
90 
91  pPed->MovingObjectCreateComponents();
92 
93  pPed->SetType( SKATE_TYPE_PED );
94 
95  Dbg_MsgAssert(pPed, ("Failed to create pedestrian."));
96  p_obj_man->RegisterObject(*pPed);
97 
98  // GJ: This should come before the model gets initialized
99  // get position, from the node that created us:
100  SkateScript::GetPosition( pNodeData, &pPed->m_pos );
101 
102  pPed->MovingObjectInit( pNodeData, p_obj_man );
103 
104  Script::RunScript( CRCD(0xf3eec45e,"ped_add_components"), pNodeData, pPed );
105  // Finalize the components, after everything has been added
106  pPed->Finalize();
107  Script::RunScript( CRCD(0x955a2767,"ped_init_model"), pNodeData, pPed );
108 
109  // designer controlled variables:
110  // set defaults, to be overridden by script values if they exist:
111  Obj::CMotionComponent* pMotionComponent = GetMotionComponentFromObject( pPed );
112 
113  pMotionComponent->m_max_vel = ( MPH_TO_INCHES_PER_SECOND ( DEFAULT_PED_MAX_VEL ) );
114  pMotionComponent->m_acceleration = FEET_TO_INCHES( DEFAULT_PED_ACCELERATION );
115  pMotionComponent->m_deceleration = FEET_TO_INCHES( DEFAULT_PED_DECELERATION );
116  pMotionComponent->m_stop_dist = FEET_TO_INCHES( DEFAULT_PED_STOP_DIST );
117  pMotionComponent->EnsurePathobExists( pPed );
118  pMotionComponent->GetPathOb()->m_enter_turn_dist = FEET_TO_INCHES( DEFAULT_PED_TURN_DIST );
119 
120  // don't let his pitch get messed up when going up hills...
121  pMotionComponent->m_movingobj_flags |= MOVINGOBJ_FLAG_NO_PITCH_PLEASE;
122 
123  // need to synchronize rendered model's position to initial world position
124  GetModelComponentFromObject( pPed )->FinalizeModelInitialization();
125 
126  // now that the skeleton has been updated at least once,
127  // we can turn off the extra bones that don't animate:
128  Script::RunScript( CRCD(0x48ca2d26,"ped_disable_bones"), NULL, pPed );
129  pPed->SetProfileColor(0xc000c0); // magenta = ped
130 
131 
132 }
