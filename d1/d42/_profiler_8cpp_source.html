180 void Profiler::start_new_frame( void )
181 {
182  mp_current_context_list = m_context_list[ s_flip ];
183  mpp_current_context_stack = m_context_stack[ s_flip ];
184  mp_current_context = NULL; // nothing has been pushed yet
185  mp_current_context_count = &m_context_count[ s_flip ];
186  m_context_stack_depth = 0; // nothing on stack
187  m_next_context = 0; // start of list (index into mp_current_context_list)
188  m_acc = false;
189  mp_acc_context = NULL;
190 }

194 void Profiler::render()
195 {
196  if ( s_active )
197  {
198  /* We draw from the buffer that's not being updated this frame */
199  int drawFlip = s_flip ^ 1;
200 
201  SProfileContext *p_context = m_context_list[ drawFlip ];
202  Tmr::MicroSeconds base_time = s_base_time[ drawFlip ];
203  int n = m_context_count[ drawFlip ];
204  while ( n--)
205  {
206  float startx = ( (uint32)(p_context->m_start_time - base_time) ) * s_scale;
207  float endx = ( (uint32)(p_context->m_end_time - base_time) ) * s_scale;
208  float top_y = m_top_y +( p_context->m_depth)* (s_height+2);
209  box((s_left_x+startx),(top_y), (s_left_x+endx),( top_y + s_height) , p_context->m_color);
210  p_context++;
211  }
212  }
213 }

219 void Profiler::PushContext( uint32 color, bool acc )
220 {
221 
222  if( s_active )
223  {
224  if ( acc && mp_acc_context )
225  {
226  // Want to accumulate to last context.
227  m_acc_time = Tmr::GetTimeInUSeconds();
228  m_acc_first = false;
229  }
230  else
231  {
232  // if we have a current context, then push it
233  if (mp_current_context)
234  {
235  Dbg_MsgAssert(m_context_stack_depth < PROF_MAX_LEVELS,("Context stack overflow"));
236  mpp_current_context_stack[m_context_stack_depth++] = mp_current_context;
237  }
238 
239  // get a new context from the list
240  Dbg_MsgAssert(m_next_context < PROF_MAX_CONTEXTS,("Context heap overflow"));
241  mp_current_context = &mp_current_context_list[m_next_context++];
242  // and fill it in with the start time, depth and color
243  // (the end time is filled in by the Pop)
244  mp_current_context->m_start_time = Tmr::GetTimeInUSeconds();
245  mp_current_context->m_depth = m_context_stack_depth;
246  mp_current_context->m_color = color;
247  if ( acc )
248  {
249  mp_acc_context = mp_current_context;
250  m_acc_first = true;
251  }
252  }
253  m_acc = acc;
254  }
255 }

265 
266 void Profiler::PopContext( void )
267 {
268  if( s_active)
269  {
270  if ( m_acc && !m_acc_first )
271  {
272  mp_acc_context->m_end_time += Tmr::GetTimeInUSeconds() - m_acc_time;
273  }
274  else
275  {
276  // set time on the current context
277  Dbg_MsgAssert(mp_current_context,("Popped one more profiler context than we pushed"));
278 
279  mp_current_context->m_end_time = Tmr::GetTimeInUSeconds();
280 
281  if ( ! m_context_stack_depth)
282  {
283  // nothing left on stack, so set current context to NULL
284  mp_current_context = NULL;
285  // update the count of valid contexts
286  *mp_current_context_count = m_next_context;
287  }
288  else
289  {
290  // get the last context pushed on the stack
291  mp_current_context = mpp_current_context_stack[--m_context_stack_depth];
292  }
293  }
294  }
