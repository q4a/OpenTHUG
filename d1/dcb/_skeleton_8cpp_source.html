691 static void sQuatVecToMatrix( Mth::Quat* pQ, Mth::Vector* pT, Mth::Matrix* pMatrix, bool rotate, bool flip )
692 {
693  Dbg_Assert( pQ );
694  Dbg_Assert( pT );
695  Dbg_Assert( pMatrix );
696 
697 #ifdef __PLAT_NGPS__
698  Dbg_MsgAssert( ( (uint32)&(pQ->GetVector()) - (uint32)pQ )==0, ("Quat::quat is no longer offset 0 bytes from the start of the Quat, need to modify the VU code in sQuatVecToMatrix"));
699 
700  asm __volatile__("
701 
702  lw $8,(%3) # Put 'rotate' flag in t0
703  lw $9,(%4) # Put 'flip' flag in t1
704 
705  lqc2 vf4,0x00(%0) # Load pQ into vf4
706  lqc2 vf7,0x00(%1) # Load pT into vf7
707 
708  # square[X] = (*pQ)[X] * (*pQ)[X];
709  # square[Y] = (*pQ)[Y] * (*pQ)[Y];
710  # square[Z] = (*pQ)[Z] * (*pQ)[Z];
711  vmul.xyz vf5,vf4,vf4 # vf5=square
712 
713  # cross[X] = (*pQ)[Y] * (*pQ)[Z];
714  # cross[Y] = (*pQ)[Z] * (*pQ)[X];
715  # cross[Z] = (*pQ)[X] * (*pQ)[Y];
716  vopmula.xyz ACC,vf4,vf4 # ACC=cross
717 
718  # wimag[X] = -(*pQ)[W] * (*pQ)[X];
719  # wimag[Y] = -(*pQ)[W] * (*pQ)[Y];
720  # wimag[Z] = -(*pQ)[W] * (*pQ)[Z];
721  vmsubw.xyzw vf8,vf4,vf4w # vf8=cross+wimag
722  vmaddw.xyzw vf9,vf4,vf4w # vf9=cross-wimag
723 
724 
725  vmulax.w ACC,vf0,vf5x # ACCw=pQx squared
726  vmadday.w ACC,vf0,vf5y # ACCw=pQx squared + pQy squared
727  vmaddz.w vf1,vf0,vf5z # vf1w=pQx squared + pQy squared + pQz squared
728 
729  # 3 vnops here ... use 2 of them with the following.
730  vadd.xyz vf8,vf8,vf8 # vf8=2*(cross+wimag)
731  vadd.xyz vf9,vf9,vf9 # vf9=2*(cross-wimag)
732 
733  vsubw.xyz vf1,vf5,vf1w # vf1=(-y2-z2,-x2-z2,-x2-y2)
