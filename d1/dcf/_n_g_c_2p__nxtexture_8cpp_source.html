63 bool CNgcTexture::plat_replace_texture( CTexture *p_texture )
64 {
65  Mem::Manager::sHandle().BottomUpHeap()->PushAlign( 32 );
66 
67  Dbg_Assert(p_texture);
68 
69  CNgcTexture *p_Ngc_texture = static_cast<CNgcTexture *>( p_texture );
70 
71  // Go through and copy the texture.
72  NxNgc::sTexture *p_src = p_Ngc_texture->GetEngineTexture();
73  NxNgc::sTexture *p_dst = GetEngineTexture();
74 
75  // Couple of problem cases.
76 // Dbg_MsgAssert( p_src->format == p_dst->format, ( "Cannot replace textures of different formats.\n" ));
77 // if ( p_src->pAlphaData && !p_dst->pAlphaData )
78 // {
79 // Dbg_MsgAssert( false, ( "Cannot assign a texture with alpha to a texture without alpha.\n" ));
80 // }
81 // if ( !p_src->pAlphaData && p_dst->pAlphaData )
82 // {
83 // Dbg_MsgAssert( false, ( "Cannot assign a texture without alpha to a texture with alpha.\n" ));
84 // }
85 
86  // Delete & re-allocate alpha space if new image is bigger, or not single owner alpha, or not channel 0.
87  if ( p_src->flags & NxNgc::sTexture::TEXTURE_FLAG_HAS_ALPHA )
88  {
89  // See if this texture needs to be stored in pOldAlpha.
90  if ( !( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_SINGLE_OWNER ) &&
91  ( ( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_MASK ) == NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_GREEN ) )
92  {
93  p_dst->pOldAlphaData = p_dst->pAlphaData;
94  p_dst->flags |= NxNgc::sTexture::TEXTURE_FLAG_OLD_DATA;
95  }
96 
97  if ( ( p_dst->byte_size < p_src->byte_size ) ||
98  !( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_SINGLE_OWNER ) ||
99  ( ( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_MASK ) != NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_GREEN ) )
100  {
101  if( p_dst->pAlphaData &&
102  ( p_dst->byte_size < p_src->byte_size ) &&
103  ( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_SINGLE_OWNER ) &&
104  ( ( p_dst->flags & NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_MASK ) == NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_GREEN ) )
105  {
106  delete [] p_dst->pAlphaData;
107  }
108  p_dst->pAlphaData = new uint8[p_src->byte_size];
109  }
110  }
111 
112  // Delete & re-allocate texture space if new image is bigger.
113  if ( p_dst->byte_size < p_src->byte_size )
114  {
115  if( p_dst->pTexelData )
116  {
117  delete [] p_dst->pTexelData;
118  }
119  p_dst->pTexelData = new uint8[p_src->byte_size];
120  p_dst->byte_size = p_src->byte_size;
121  }
122 
123  p_dst->BaseWidth = p_src->BaseWidth;
124  p_dst->BaseHeight = p_src->BaseHeight;
125  p_dst->ActualWidth = p_src->ActualWidth;
126  p_dst->ActualHeight = p_src->ActualHeight;
127 
128  p_dst->flags = ( p_dst->flags & ~NxNgc::sTexture::TEXTURE_FLAG_HAS_HOLES ) | ( p_src->flags & NxNgc::sTexture::TEXTURE_FLAG_HAS_HOLES );
129 
130  // Copy the pixel data.
131  memcpy( p_dst->pTexelData, p_src->pTexelData, p_src->byte_size );
132  DCFlushRange ( p_dst->pTexelData, p_src->byte_size );
133 
134  // Copy the alpha data if necessary.
135  if ( p_src->pAlphaData && p_dst->pAlphaData )
136  {
137  memcpy( p_dst->pAlphaData, p_src->pAlphaData, p_src->byte_size );
138  DCFlushRange ( p_dst->pAlphaData, p_src->byte_size );
139  p_dst->flags = ( p_dst->flags & ~NxNgc::sTexture::TEXTURE_FLAG_HAS_ALPHA ) | ( p_src->flags & NxNgc::sTexture::TEXTURE_FLAG_HAS_ALPHA );
140  p_dst->flags = ( p_dst->flags & ~NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_MASK ) | NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_GREEN;
141  p_dst->flags = p_dst->flags | NxNgc::sTexture::TEXTURE_FLAG_SINGLE_OWNER;
142  }
143 
144  // If the replacement texture doesn't have alpha, but the original does, twiddle the flag.
145  if ( p_dst->pAlphaData && !p_src->pAlphaData )
146  {
147  p_dst->flags = ( p_dst->flags & ~NxNgc::sTexture::TEXTURE_FLAG_HAS_ALPHA );
148  }
149 
150  // Replacement texture does not mip-map.
151  p_dst->Levels = 0;
152 
153  // So that if the regular texture is 32-bit, it can have a CMPR texture copied over it.
154  p_dst->format = p_src->format;
155 
156  // Flag this texture as having been replaced.
157  p_dst->flags |= NxNgc::sTexture::TEXTURE_FLAG_REPLACED;
158 
159  Mem::Manager::sHandle().BottomUpHeap()->PopAlign();
160 
161  return true;
162 }

410 {
411  uint8 *p_data = (uint8*)( *pp_mem );
412 
413  uint32 temp;
414 
415  // Open the texture file.
416 // void *p_FH = File::Open( Filename, "rb" );
417 // if( !p_FH )
418 // {
419 // Dbg_Message( "Couldn't open texture file %s\n", Filename );
421 // return;
422 // }
423 
424  // Read the texture file version.
425  int version;
426  MemoryRead( &version, sizeof( int ), 1, p_data );
427 
428  // Read the number of textures.
429  int num_textures;
430  MemoryRead( &num_textures, sizeof( int ), 1, p_data );
431 
432 // NumTextures += num_textures;
433 
434  int bytes;
435  bool need_to_pop;
436 
437  NxNgc::sTexture ** p_tex = new (Mem::Manager::sHandle().TopDownHeap()) NxNgc::sTexture *[num_textures];
438  unsigned short * p_resolvem = new (Mem::Manager::sHandle().TopDownHeap()) unsigned short[num_textures];
439  unsigned short * p_resolvec = new (Mem::Manager::sHandle().TopDownHeap()) unsigned short[num_textures];
440  int num_resolve = 0;
441 
442  int mem_available;
443  for( int t = 0; t < num_textures; ++t )
444  {
445  need_to_pop = false;
446  bytes = sizeof( NxNgc::sTexture );
447  if ( g_in_cutscene )
448  {
449  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().FrontEndHeap() );
450  mem_available = Mem::Manager::sHandle().Available();
451  if ( bytes < ( mem_available - ( 40 * 1024 ) ) )
452  {
453  need_to_pop = true;
454  }
455  else
456  {
457  Mem::Manager::sHandle().PopContext();
458  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().ThemeHeap() );
459  mem_available = Mem::Manager::sHandle().Available();
460  if ( bytes < ( mem_available - ( 5 * 1024 ) ) )
461  {
462  need_to_pop = true;
463  }
464  else
465  {
466  Mem::Manager::sHandle().PopContext();
467  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().ScriptHeap() );
468  mem_available = Mem::Manager::sHandle().Available();
469  if ( bytes < ( mem_available - ( 40 * 1024 ) ) )
470  {
471  need_to_pop = true;
472  }
473  else
474  {
475  Mem::Manager::sHandle().PopContext();
476  }
477  }
478  }
479  }
480 
481  NxNgc::sTexture *p_texture = new NxNgc::sTexture;
482 
483  if ( need_to_pop )
484  {
485  Mem::Manager::sHandle().PopContext();
486  }
487 
488  p_tex[t] = p_texture;
489 
490  MemoryRead( &p_texture->Checksum, sizeof( uint32 ), 1, p_data );
491  MemoryRead( &temp, sizeof( uint32 ), 1, p_data );
492  p_texture->BaseWidth = (uint16)temp;
493  MemoryRead( &temp, sizeof( uint32 ), 1, p_data );
494  p_texture->BaseHeight = (uint16)temp;
495  MemoryRead( &temp, sizeof( uint32 ), 1, p_data );
496  p_texture->Levels = (uint16)temp;
497 
498  p_texture->ActualWidth = ( p_texture->BaseWidth + 3 ) & ~3;
499  p_texture->ActualHeight = ( p_texture->BaseHeight + 3 ) & ~3;
500 
501  int tex_format;
502  int channel;
503  int index;
504  MemoryRead( &tex_format, sizeof( uint32 ), 1, p_data );
505  channel = ( tex_format >> 8 ) & 0xff;
506  index = ( tex_format >> 16 ) & 0xffff;
507  tex_format = tex_format & 0xff;
508 
509  switch ( tex_format ) {
510  case 0:
511  p_texture->format = GX_TF_CMPR;
512  break;
513  case 1:
514  p_texture->format = GX_TF_CMPR;
515  break;
516  case 2:
517  p_texture->format = GX_TF_RGBA8;
518  break;
519  default:
520  Dbg_MsgAssert( false, ("Illegal texture format: %d\n", tex_format ));
521  break;
522  }
523 
524  p_texture->flags = 0;
525 
526  int has_holes;
527  MemoryRead( &has_holes, sizeof( uint32 ), 1, p_data );
528  p_texture->flags |= has_holes ? NxNgc::sTexture::TEXTURE_FLAG_HAS_HOLES : 0;
529

532

534  bytes = 0;

642  mem_available = Mem::Manager::sHandle().Available();
643  if ( bytes < ( mem_available - ( 40 * 1024 ) ) )
644  {
645  need_to_pop = true;
646  }
647  else

649  Mem::Manager::sHandle().PopContext();
650  }
651  }
652  }
653  }
654 
655  Mem::Manager::sHandle().BottomUpHeap()->PushAlign( 32 );
656  p_texture->pAlphaData = new uint8[bytes];
657  Mem::Manager::sHandle().BottomUpHeap()->PopAlign();
658 
659  if ( need_to_pop )
660  {
661  Mem::Manager::sHandle().PopContext();
662  }
663 
664  uint8 * pTex = p_texture->pAlphaData;
665  for( uint32 mip_level = 0; mip_level < p_texture->Levels; ++mip_level )
666  {
667  memcpy( pTex, p8[mip_level], mipbytes[mip_level] );
668  delete p8[mip_level];
669  pTex += mipbytes[mip_level];
670  }

672  }
673  else
674  {
675  p_texture->flags |= NxNgc::sTexture::TEXTURE_FLAG_HAS_ALPHA;
676 
677  p_resolvem[num_resolve] = t;
678  p_resolvec[num_resolve] = index;
679  num_resolve++;
680  }
681  switch ( channel )
682  {
683  case 0:
684  default:
685  p_texture->flags |= NxNgc::sTexture::TEXTURE_FLAG_CHANNEL_GREEN;
686  break;
687  case 1:

709  if ( g_in_cutscene )
710  {

712  mem_available = Mem::Manager::sHandle().Available();

721  mem_available = Mem::Manager::sHandle().Available();
