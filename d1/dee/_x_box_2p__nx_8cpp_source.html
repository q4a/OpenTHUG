174 void CEngine::s_plat_post_render( void )
175 {
176  // No rendering should take place whilst the loading screen is visible.
177  if( !NxXbox::EngineGlobals.loadingscreen_visible )
178  {
179  D3DDevice_Swap( D3DSWAP_DEFAULT );
180 
181  if( NxXbox::EngineGlobals.screenshot_name[0] != 0 )
182  {
183  Spt::SingletonPtr< Gfx::Manager > gfx_manager;
184  gfx_manager->ScreenShot( NxXbox::EngineGlobals.screenshot_name );
185  NxXbox::EngineGlobals.screenshot_name[0] = 0;
186  }
187 
188  // Now that the swap instruction has been pushed, clear the buffer for next frame.
189  if( NxXbox::EngineGlobals.screen_blur > 0 )
190  {
191  D3DDevice_Clear( 0, NULL, D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, NxXbox::EngineGlobals.clear_color, 1.0f, 0 );
192  ++NxXbox::EngineGlobals.screen_blur_duration;
193  }
194  else
195  {
196  if( NxXbox::EngineGlobals.letterbox_active )
197  {
198  D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, 0x00000000UL, 1.0f, 0 );
199  }
200  else if( NxXbox::EngineGlobals.clear_color_buffer )
201  {
202  D3DDevice_Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, NxXbox::EngineGlobals.clear_color, 1.0f, 0 );
203  }
204  else
205  {
206  D3DDevice_Clear( 0, NULL, D3DCLEAR_ZBUFFER | D3DCLEAR_STENCIL, NxXbox::EngineGlobals.clear_color, 1.0f, 0 );
207  }
208  NxXbox::EngineGlobals.screen_blur_duration = 0;
209  }
210  }

219 void CEngine::s_plat_render_world( void )
220 {
221  // Remove the loading bar.
222  if( NxXbox::EngineGlobals.loadingbar_timer_event != 0 )
223  {
224  timeKillEvent( NxXbox::EngineGlobals.loadingbar_timer_event );
225  NxXbox::EngineGlobals.loadingbar_timer_event = 0;
226  }
227 
228  // No rendering should take place whilst the loading screen is visible.
229  if( !NxXbox::EngineGlobals.loadingscreen_visible )
230  {
231  // Store time at start of render, used as reference throughout.
232  NxXbox::EngineGlobals.render_start_time = (int)Tmr::GetTime();
233 
234  // Render objects of interest for the render target (shadow objects).
235  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
236  NxXbox::set_render_state( RS_ZTESTENABLE, 1 );
237 
238  NxXbox::render_shadow_targets();
239 
240  CEngine::sGetImposterManager()->ProcessImposters();
241 
242  // Start up the screen blur if we're using it.
243  NxXbox::start_screen_blur();
244 
245  int num_viewports = CViewportManager::sGetNumActiveViewports();
246  for( int v = 0; v < num_viewports; ++v )
247  {
248  CViewport *p_cur_viewport = CViewportManager::sGetActiveViewport( v );
249  Gfx::Camera *p_cur_camera = p_cur_viewport->GetCamera();
250 
251  // Check for a valid camera.
252  if( p_cur_camera == NULL )
253  {
254  continue;
255  }
256 
257  NxXbox::EngineGlobals.viewport.X = (DWORD)( p_cur_viewport->GetOriginX() * NxXbox::EngineGlobals.backbuffer_width );
258  NxXbox::EngineGlobals.viewport.Y = (DWORD)( p_cur_viewport->GetOriginY() * NxXbox::EngineGlobals.backbuffer_height );
259  NxXbox::EngineGlobals.viewport.Width = (DWORD)( p_cur_viewport->GetWidth() * NxXbox::EngineGlobals.backbuffer_width );
260  NxXbox::EngineGlobals.viewport.Height = (DWORD)( p_cur_viewport->GetHeight() * NxXbox::EngineGlobals.backbuffer_height );
261  NxXbox::EngineGlobals.viewport.MinZ = 0.0f;
262  NxXbox::EngineGlobals.viewport.MaxZ = 1.0f;
263 
264  if( NxXbox::EngineGlobals.letterbox_active )
265  {
266  NxXbox::EngineGlobals.viewport.Y += (DWORD)( NxXbox::EngineGlobals.backbuffer_height / 8 );
267  NxXbox::EngineGlobals.viewport.Height -= (DWORD)( NxXbox::EngineGlobals.backbuffer_height / 4 );
268  }
269 
270  D3DDevice_SetViewport( &NxXbox::EngineGlobals.viewport );
271 
272  // There is no bounding box transform for rendering the world.
273  NxXbox::set_frustum_bbox_transform( NULL );
274 
275  // Set up the camera..
276  float aspect_ratio = p_cur_viewport->GetAspectRatio();
277 
278  NxXbox::set_camera( &( p_cur_camera->GetMatrix()), &( p_cur_camera->GetPos()), p_cur_camera->GetAdjustedHFOV(), aspect_ratio );
279 
280  // Render the non-sky world scenes.
281  for( int i = 0; i < MAX_LOADED_SCENES; i++ )
282  {
283  if( sp_loaded_scenes[i] )
284  {
285  CXboxScene *pXboxScene = static_cast<CXboxScene *>( sp_loaded_scenes[i] );
286 
287  if( !pXboxScene->IsSky())
288  {
289  // Build relevant occlusion poly list, now that the camera is set.
290  NxXbox::BuildOccluders( &( p_cur_camera->GetPos()), v );
291 
292  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
293  NxXbox::set_render_state( RS_ZTESTENABLE, 1 );
294 
295  // Flag this scene as receiving shadows.
296  pXboxScene->GetEngineScene()->m_flags |= SCENE_FLAG_RECEIVE_SHADOWS;
297 
298  NxXbox::render_scene( pXboxScene->GetEngineScene(), NxXbox::vRENDER_OPAQUE |
299  NxXbox::vRENDER_OCCLUDED |
300  NxXbox::vRENDER_SORT_FRONT_TO_BACK |
301  NxXbox::vRENDER_BILLBOARDS, v );
302  }
303  }
304  }
305 
306  NxXbox::set_render_state( RS_ZWRITEENABLE, 0 );
307  Nx::TextureSplatRender();
308  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
309 
310  CEngine::sGetImposterManager()->DrawImposters();
311 
312  // Render all opaque instances.
313  NxXbox::render_instances( NxXbox::vRENDER_OPAQUE );
314 
315  // Now that opaque geometry is drawn, do the render tests for the light glows.
316  NxXbox::render_light_glows( true );
317 
318  // Render the sky, followed by all the non-sky semitransparent scene geometry. There is no bounding box transform for rendering the world.
319  NxXbox::set_frustum_bbox_transform( NULL );
320 
321  // Set up the sky camera.
322  Mth::Vector centre_pos( 0.0f, 0.0f, 0.0f );
323  NxXbox::set_camera( &( p_cur_camera->GetMatrix()), &centre_pos, p_cur_camera->GetAdjustedHFOV(), aspect_ratio, true );
324 
325  // Render the sky. We have to fudge the fog here to ensure the sky is fully fogged, since it is rendered with a non-standard projection
326  // matrix to ensure a constant z=1, but which breaks the fog interpolation value.
327  float fog_start = NxXbox::EngineGlobals.fog_start;
328  float fog_end = NxXbox::EngineGlobals.fog_end;
329  NxXbox::EngineGlobals.fog_start = -20.0f;
330  NxXbox::EngineGlobals.fog_end = -21.0f;
331  D3DDevice_SetRenderState( D3DRS_FOGSTART, *((DWORD*)( &NxXbox::EngineGlobals.fog_start )));
332  D3DDevice_SetRenderState( D3DRS_FOGEND, *((DWORD*)( &NxXbox::EngineGlobals.fog_end )));
333 
334  NxXbox::set_render_state( RS_ZWRITEENABLE, 0 );
335  NxXbox::set_render_state( RS_ZTESTENABLE, 1 );
336  for( int i = 0; i < MAX_LOADED_SCENES; i++ )
337  {
338  if( sp_loaded_scenes[i] )
339  {
340  CXboxScene *pXboxScene = static_cast<CXboxScene *>( sp_loaded_scenes[i] );
341  if( pXboxScene->IsSky())
342  {
343  // No anisotropic filtering for the sky.
344  DWORD stage_zero_minfilter, stage_zero_mipfilter;
345  D3DDevice_GetTextureStageState( 0, D3DTSS_MINFILTER, &stage_zero_minfilter );
346  D3DDevice_GetTextureStageState( 0, D3DTSS_MIPFILTER, &stage_zero_mipfilter );
347  D3DDevice_SetTextureStageState( 0, D3DTSS_MINFILTER, D3DTEXF_LINEAR );
348  D3DDevice_SetTextureStageState( 0, D3DTSS_MIPFILTER, D3DTEXF_NONE );
349 
350  NxXbox::render_scene( pXboxScene->GetEngineScene(), NxXbox::vRENDER_OPAQUE | NxXbox::vRENDER_SEMITRANSPARENT, v );
351 
352  D3DDevice_SetTextureStageState( 0, D3DTSS_MIPFILTER, stage_zero_mipfilter );
353  D3DDevice_SetTextureStageState( 0, D3DTSS_MINFILTER, stage_zero_minfilter );
354  }
355  }
356  }
357  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
358 
359  // Restore fog values.
360  NxXbox::EngineGlobals.fog_start = fog_start;
361  NxXbox::EngineGlobals.fog_end = fog_end;
362  D3DDevice_SetRenderState( D3DRS_FOGSTART, *((DWORD*)( &NxXbox::EngineGlobals.fog_start )));
363  D3DDevice_SetRenderState( D3DRS_FOGEND, *((DWORD*)( &NxXbox::EngineGlobals.fog_end )));
364 
365  // Revert to the regular camera.
366  NxXbox::set_camera( &( p_cur_camera->GetMatrix()), &( p_cur_camera->GetPos()), p_cur_camera->GetAdjustedHFOV(), aspect_ratio );
367 
368  // Render all semitransparent instances.
369  NxXbox::render_instances( NxXbox::vRENDER_SEMITRANSPARENT | NxXbox::vRENDER_INSTANCE_PRE_WORLD_SEMITRANSPARENT );
370 
371  // Render the non-sky semitransparent scene geometry.
372  // Setting the depth clip control to clamp here means that semitransparent periphary objects that would usually cull out
373  // are now drawn correctly (since they will clamp at 1.0, and the z test is <=).
374  D3DDevice_SetRenderState( D3DRS_DEPTHCLIPCONTROL, D3DDCC_CLAMP );
375  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
376  NxXbox::set_render_state( RS_ZTESTENABLE, 1 );
377  for( int i = 0; i < MAX_LOADED_SCENES; i++ )
378  {
379  if( sp_loaded_scenes[i] )
380  {
381  CXboxScene *pXboxScene = static_cast<CXboxScene *>( sp_loaded_scenes[i] );
382  if( !pXboxScene->IsSky())
383  {
384  // Build relevant occlusion poly list, now that the camera is set.
385  NxXbox::render_scene( pXboxScene->GetEngineScene(), NxXbox::vRENDER_SEMITRANSPARENT |
386  NxXbox::vRENDER_OCCLUDED |
387  NxXbox::vRENDER_BILLBOARDS, v );
388  }
389  }
390  }
391  D3DDevice_SetRenderState( D3DRS_DEPTHCLIPCONTROL, D3DDCC_CULLPRIMITIVE );
392 
393  // Render the particles.
394  NxXbox::set_render_state( RS_ZWRITEENABLE, 0 );
395  render_particles();
396 
397  // New style particles. Update should probably be somewhere else.
398  mp_particle_manager->UpdateParticles();
399  mp_particle_manager->RenderParticles();
400 
401  // Render weather effects.
402  mp_weather->Process( Tmr::FrameLength());
403  mp_weather->Render();
404 
405  NxXbox::set_render_state( RS_ZWRITEENABLE, 1 );
406 
407  // We want shatter objects to be rendered after the sky since they will typically be semitransparent.
408  Nx::ShatterRender();
409 
410  // Render all semitransparent instances.
411  NxXbox::render_instances( NxXbox::vRENDER_SEMITRANSPARENT | NxXbox::vRENDER_INSTANCE_POST_WORLD_SEMITRANSPARENT );
412 
413  // Render the light glows (the visibility tests were performed earlier).
414  NxXbox::render_light_glows( false );
415 
416  // Render the shadow volumes.
417 // for( int i = 0; i < MAX_LOADED_SCENES; i++ )
418 // {
419 // if( sp_loaded_scenes[i] )
420 // {
421 // CXboxScene *pXboxScene = static_cast<CXboxScene *>( sp_loaded_scenes[i] );
422 // if( !pXboxScene->IsSky())
423 // {
424 // NxXbox::render_scene( pXboxScene->GetEngineScene(), NxXbox::vRENDER_SHADOW_VOLUMES, v );
425 // }
426 // }
427 // }
428 
429  Nx::ScreenFlashRender( v, 0 );
430  }
431 
432  // This x86 instruction writes back and invalidates the cache. This should hopefully mean that all updated vertex buffer
433  // data (vc wibble) are flushed out by the time the GPU hits them.
434  _asm
435  {
436  wbinvd;
437  }
438 
439  // Draw debug lines.
440 # ifdef __NOPT_ASSERT__
441  Gfx::DebugGfx_Draw();
442 # endif
443 
444  // Reset viewport so text appears on full screen.
445  NxXbox::EngineGlobals.viewport.X = 0;
446  NxXbox::EngineGlobals.viewport.Y = 0;
447  NxXbox::EngineGlobals.viewport.Width = NxXbox::EngineGlobals.backbuffer_width;
448  NxXbox::EngineGlobals.viewport.Height = NxXbox::EngineGlobals.backbuffer_height;
449  NxXbox::EngineGlobals.viewport.MinZ = 0.0f;
450  NxXbox::EngineGlobals.viewport.MaxZ = 1.0f;
451 
452  if( NxXbox::EngineGlobals.letterbox_active )
453  {
454  NxXbox::EngineGlobals.viewport.Y += (DWORD)( NxXbox::EngineGlobals.backbuffer_height / 8 );
455  NxXbox::EngineGlobals.viewport.Height -= (DWORD)( NxXbox::EngineGlobals.backbuffer_height / 4 );
456  }
457 
458  D3DDevice_SetViewport( &NxXbox::EngineGlobals.viewport );
459 
460 // Obj::CSkater *pSkater = Mdl::Skate::Instance()->GetLocalSkater();
461 // Mth::Vector pos = pSkater->GetPos();
462 // NxXbox::set_focus_blur_focus( pos, 0.0f, 10.0f * 12.0f, 100.0f * 12.0f );
463 // NxXbox::finish_focus_blur();
464 
465  // Horrible hack - this should be somewhere else ASAP.
466  NxXbox::SDraw2D::DrawAll();
467 
468  // Finish the screen blur if we're using it.
469  NxXbox::finish_screen_blur();
470  }

472  // Reset viewport so screen clear will work.
473  NxXbox::EngineGlobals.viewport.X = 0;

475  NxXbox::EngineGlobals.viewport.Width = NxXbox::EngineGlobals.backbuffer_width;

480 }

512 CScene *CEngine::s_plat_load_scene_from_memory( void *p_mem, CTexDict *p_tex_dict, bool add_super_sectors, bool is_sky, bool is_dictionary )
513 {
514  uint8 *p_data = (uint8*)p_mem;
515  CSector *pSector;
516  CXboxSector *pXboxSector;
517 
518  // Create a new sScene so the engine can track assets for this scene.
519  NxXbox::sScene *p_engine_scene = new NxXbox::sScene();
520 
521  // Set the dictionary flag.
522  p_engine_scene->m_is_dictionary = is_dictionary;
523 
524  // Version numbers.
525  uint32 mat_version, mesh_version, vert_version;
526  MemoryRead( &mat_version, sizeof( uint32 ), 1, p_data );
527  MemoryRead( &mesh_version, sizeof( uint32 ), 1, p_data );
528  MemoryRead( &vert_version, sizeof( uint32 ), 1, p_data );
529 
530  // Import materials (they will now be associated at the engine-level with this scene).
531  p_engine_scene->pMaterialTable = NxXbox::LoadMaterialsFromMemory( (void**)&p_data, p_tex_dict->GetTexLookup());
532 
533  // Read number of sectors.
534  int num_sectors;
535  MemoryRead( &num_sectors, sizeof( int ), 1, p_data );
536 
537  // Figure optimum hash table lookup size.
538  uint32 optimal_table_size = num_sectors * 2;
539  uint32 test = 2;
540  uint32 size = 1;
541  for( ;; test <<= 1, ++size )
542  {
543  // Check if this iteration of table size is sufficient, or if we have hit the maximum size.
544  if(( optimal_table_size <= test ) || ( size >= 12 ))
545  {
546  break;
547  }
548  }
549 
550  // Create scene class instance, using optimum size sector table.
551  CScene* new_scene = new CXboxScene( size );
552  new_scene->SetInSuperSectors( add_super_sectors );
553  new_scene->SetIsSky( is_sky );
554 
555  // Get a scene id from the engine.
556  CXboxScene *p_new_xbox_scene = static_cast<CXboxScene *>( new_scene );
557  p_new_xbox_scene->SetEngineScene( p_engine_scene );
558 
559  for( int s = 0; s < num_sectors; ++s )
560  {
561  // Create a new sector to hold the incoming details.
