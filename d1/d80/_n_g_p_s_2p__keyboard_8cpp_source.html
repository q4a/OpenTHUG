114 void kb_read_logic_code ( const Tsk::Task< int >& task )
115 {
116  int i, result;
117 
118  if (!s_active || !s_enabled)
119  {
120  return;
121  }
122 
123  s_num_makes = 0;
124 
125  result = sceUsbKbGetInfo( &s_info );
126  if( result != USBKB_OK )
127  {
128  Dbg_Printf( "Error%d : sceUsbKbGetInfo\n", result );
129  return;
130  }
131  sceUsbKbSync( USBKB_WAIT, &result );
132  if( result != USBKB_OK )
133  {
134  Dbg_Printf( "Error%d : sceUsbKbSync\n", result );
135  return;
136  }
137 
138  // reading keyboard
139  for( i = 0; i < s_max_connect; i++)
140  {
141  if( s_info.status[i] == 0 )
142  {
143  continue;
144  } // not connected
145 
146  if( s_old_status[i] == 0 )
147  {
148  Dbg_Printf( "New keyboard %d is connected\n", i );
149  s_init_newkeyboard(i);
150  }
151 
152  result = sceUsbKbRead( i, &s_kdata );
153  if( result != USBKB_OK )
154  {
155  Dbg_Printf( "Error%d : sceUsbKbRead\n", result );
156  continue;
157  }
158  sceUsbKbSync( USBKB_WAIT, &result );
159  if( result != USBKB_OK )
160  {
161  Dbg_Printf( "Error%d : sceUsbKbSync\n",result);
162  continue;
163  }
164 
165  if( s_kdata.len == 0 )
166  {
167  continue;
168  }
169 
170  result = sceUsbKbGetLocation( i, s_location );
171  if( result != USBKB_OK )
172  {
173  Dbg_Printf( "Error%d : sceUsbKbGetLocation\n", result );
174  continue;
175  }
176 
177  sceUsbKbSync( USBKB_WAIT, &result );
178  if( result != USBKB_OK )
179  {
180  Dbg_Printf( "Error%d : sceUsbKbSync\n", result );
181  continue;
182  }
183 
184  s_num_makes = s_capture_input( i, s_makes );
185  }
186 
187  for( i = 0; i < s_max_connect; i++ )
188  {
189  s_old_status[i] = s_info.status[i];
190  }
191 }

202 int KeyboardInit(void)
203 {
204 
205 
206  int result;
207  char *option = USBKEYBD_ARG;
208  int i;
209  Mlp::Manager * mlp_man = Mlp::Manager::Instance();
210 
211 
212  //sceSifInitRpc(0);
213 
214  if (Config::CD())
215  {
216  char path[128];
217  sprintf(path,"cdrom0:\\%sIOP\\USBKB.IRX",Config::GetDirectory());
218 
219  result = sceSifLoadModule(
220  path,
221  sizeof( USBKEYBD_ARG ) + 1, option );
222  }
223  else
224  {
225  result = sceSifLoadModule(
226  "host0:IOPModules/usbkb.irx",
227  sizeof( USBKEYBD_ARG ) + 1, option );
228  }
229 
230  if( result < 0 )
231  {
232  Dbg_MsgAssert( 0,( "EE:Can't load module usbkb.irx\n" ));
233  return -1;
234  }
235 
236  result = sceUsbKbInit( &s_max_connect );
237  if( result == USBKB_NG )
238  {
239  Dbg_Printf( "Initialize error\n" );
240  return -1;
241  }
242 
243  s_old_status = new unsigned char[ s_max_connect ];
244  for( i = 0; i < s_max_connect; i++ )
245  {
246  s_old_status[i] = 0;
247  }
248 
249  int arg;
250  // Just send an int because Tsk::Task is a template and needs some sort of
251  // specific data type. So I just chose int and pass an int even though I don't use it
252  s_kb_read_logic_task = new Tsk::Task< int > ( kb_read_logic_code, arg );
253  mlp_man->AddSystemTask( *s_kb_read_logic_task );
254  return 0;
255 }

330

344 
345  sceUsbKbSetRepeat( i, 30, 2 );

355 static int s_capture_input( int i, char* makes )
356 {
357  u_short kcode;
358  int j, index;
359 
360  //Dbg_Printf( "usbkeybd%d : ", i );
361 
362  //Dbg_Printf( "port=" );
363  //for( k = 0; k < 7 && s_location[k] != 0; k++ )
364  //{
365  //Dbg_Printf( "%s%d", ((k)? ",": "" ), s_location[k] );
366  //}
367 
368 
369  //Dbg_Printf( " : LED[%02X] ", s_kdata.led );
370 
371  //Dbg_Printf( "MKEY[%02X] ", s_kdata.mkey );
372 
373  //Dbg_Printf( "KEY[" );
374 
375  index = 0;
376  for( j=0; j < s_kdata.len; j++ )
377  {
378  kcode = s_kdata.keycode[j];
379  if( kcode & USBKB_RAWDAT )
380  {
381  //Dbg_Printf( "%04X ", kcode );
382  switch( kcode )
383  {
384  case vKEYCODE_F1:
385  makes[index++] = vKB_F1;
386  break;
387  case vKEYCODE_F2:
388  makes[index++] = vKB_F2;
389  break;
390  case vKEYCODE_F3:
391  makes[index++] = vKB_F3;
392  break;
393  case vKEYCODE_F4:
394  makes[index++] = vKB_F4;
395  break;
396  case vKEYCODE_RIGHT:
397  makes[index++] = vKB_RIGHT;
398  break;
399  case vKEYCODE_LEFT:
400  makes[index++] = vKB_LEFT;
401  break;
402  case vKEYCODE_DOWN:
403  makes[index++] = vKB_DOWN;
404  break;
405  case vKEYCODE_UP:
406  makes[index++] = vKB_UP;
407  break;
408  case vKEYCODE_ESC:
409  makes[index++] = vKB_ESCAPE;
410  break;
411  }
412  continue;
413  }
414  if( kcode & USBKB_KEYPAD )
415  {
416  // 10 key
417  if((kcode & 0x00ff) == '\n')
418  {
419  //Dbg_Printf( "(\\n)" );
420  makes[index++] = vKB_ENTER;
421  continue;
422  }
423  //Dbg_Printf( "(%c) ", kcode & ~USBKB_KEYPAD );
424  makes[index++] = kcode & ~USBKB_KEYPAD;
425  continue;
426  }
427  // Normal key
428  if((kcode & 0x00ff) == '\0' )
429  {
430  //Dbg_Printf( "\'\\0\'" );
431  continue;
432  }
433  if((kcode & 0x00ff) == '\n' )
434  {
435  //Dbg_Printf( "\'\\n\'" );
436  makes[index++] = vKB_ENTER;
437  continue;
438  }
439  if((kcode & 0x00ff) == '\t' )
440  {
441  //Dbg_Printf( "\'\\t\'" );
442  continue;
443  }
444  if((kcode & 0x00ff) == '\b' )
445  {
446  //Dbg_Printf( "\'\\b\'" );
447  makes[index++] = vKB_BACKSPACE;
448  continue;
449  }
450  //Dbg_Printf( "\'%c\' ", kcode );
451  makes[index++] = kcode;
452  }
453  //Dbg_Printf( "]\n" );
454 
455  return index;
456 }
