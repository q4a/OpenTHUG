32 static void get_checksums( const char * p_name, uint32 * p_dir, uint32 * p_file )
33 {
34  char name[128];
35  char dir[128];
36  char file[128];
37  uint lp;
38 
39  // Convert / to \\.
40  for ( lp = 0; lp <= strlen( p_name ); lp++ )
41  {
42  name[lp] = p_name[lp];
43  if ( name[lp] == '/' ) name[lp] = '\\';
44  }
45 
46  // Find the directory name and the filename.
47  for ( lp = strlen( name ) - 1; lp >= 0; lp-- )
48  {
49  if ( name[lp] == '\\' )
50  {
51  // Found the directory end.
52  strcpy( file, &name[lp+1] );
53  name[lp] = '\0';
54  strcpy( dir, name );
55  break;
56  }
57  }
58 
59  // Chop off the extension (if it exists).
60  if ( file[strlen(file)-4] == '.' )
61  {
62  file[strlen(file)-4] = '\0';
63  }
64 
65  // Create checksums.
66  if ( p_dir ) *p_dir = Script::GenerateCRC( dir );
67  if ( p_file ) *p_file = Script::GenerateCRC( file );
68 }

74 SHedFile *FindFileInHed(const char *pFilename, SHed *pHed)
75 {
76  if ( sLoadingHed ) return NULL;
77 
78  Dbg_MsgAssert(pFilename,("NULL pFilename"));
79  Dbg_MsgAssert(pHed,("NULL pHed"));
80 
81  uint32 check_dir;
82  uint32 check_file;
83 
84  get_checksums( pFilename, &check_dir, &check_file );
85 
86  SHed *pHd=pHed;
87  while ( pHd->numFiles != 0xffffffff )
88  {
89  if ( pHd->Checksum == check_dir )
90  {
91  SHedFile * pF = pHd->p_fileList;
92  for ( uint lp2 = 0; lp2 < pHd->numFiles; lp2++ )
93  {
94  if ( pF->Checksum == check_file )
95  {
96  return pF;
97  }
98  pF++;
99  }
100  }
101  pHd++;
102  }
103 
104  return NULL;
105 }

188 SHed *LoadHed( const char *filename )
189 {
190  sLoadingHed = true;
191 
192  int lp;
193  char tempFileName[255];
194  sprintf( tempFileName, "%sNGC.HED", filename );
195 
196  void *fp = File::Open( tempFileName, "rb" );
197 
198  int fileSize = File::GetFileSize( fp );
199 
200  uint32 *pHed = (uint32 *)new (Mem::Manager::sHandle().TopDownHeap()) uint8[fileSize];
201  Dbg_MsgAssert( pHed,( "Failed to allocate memory for hed %s", tempFileName ));
202 
203  File::Read( pHed, 1, fileSize, fp );
204  File::Close( fp );
205 
206  // Convert to more compact list.
207  int files = 0;
208  int dirs = 0;
209 
210  // First, count how many files we have.
211  uint32 * p32 = pHed;
212  while ( p32[0] != 0xffffffff )
213  {
214  files++;
215 
216  // Byte order is reversed on NGC.
217  p32[0] = ( ( p32[0] >> 24 ) & 0xff ) | ( ( p32[0] >> 8 ) & 0xff00 ) | ( ( p32[0] << 8 ) & 0xff0000 ) | ( ( p32[0] << 24 ) & 0xff000000 );
218  p32[1] = ( ( p32[1] >> 24 ) & 0xff ) | ( ( p32[1] >> 8 ) & 0xff00 ) | ( ( p32[1] << 8 ) & 0xff0000 ) | ( ( p32[1] << 24 ) & 0xff000000 );
219 
220  int len = strlen( (char *)&p32[2] ) + 1;
221  len = ( len + 3 ) & ~3;
222  p32 = (uint32*)(((int)&p32[2]) + len);
223  }
224 
225  // Build list of checksums.
226  uint32 *p_offset = new (Mem::Manager::sHandle().TopDownHeap()) uint32[files];
227  uint32 *p_size = new (Mem::Manager::sHandle().TopDownHeap()) uint32[files];
228  uint32 *dir_check = new (Mem::Manager::sHandle().TopDownHeap()) uint32[files];
229  uint32 *file_check = new (Mem::Manager::sHandle().TopDownHeap()) uint32[files];
230  int count = 0;
231  p32 = pHed;
232  while ( p32[0] != 0xffffffff )
233  {
234  p_offset[count] = p32[0];
235  p_size[count] = p32[1];
236  get_checksums( (char *)&p32[2], &dir_check[count], &file_check[count] );
237  count++;
238 
239  int len = strlen( (char *)&p32[2] ) + 1;
240  len = ( len + 3 ) & ~3;
241  p32 = (uint32*)(((int)&p32[2]) + len);
242  }
243 
244  // Second, count how many directories we have.
245  for ( lp = 0; lp < files; lp++ )
246  {
247  bool unique = true;
248  for ( int lp2 = 0; lp2 < lp; lp2++ )
249  {
250  if ( dir_check[lp] == dir_check[lp2] )
251  {
252  unique = false;
253  break;
254  }
255  }
256  if ( unique ) dirs++;
257  }
258 
259  // Now, allocate final piece of memory.
260  int size = ( sizeof( SHed ) * ( dirs + 1 ) ) + ( sizeof( SHedFile ) * files );
261  SHed * p_hed = (SHed *)new uint8[size];
262 
263  // Fill in directory entries.
264  int dirs_added = 0;
265  for ( lp = 0; lp < files; lp++ )
266  {
267  bool unique = true;
268  for ( int lp2 = 0; lp2 < dirs_added; lp2++ )
269  {
270  if ( p_hed[lp2].Checksum == dir_check[lp] )
271  {
272  unique = false;
273  break;
274  }
275  }
276  if ( unique )
277  {
278  p_hed[dirs_added].Checksum = dir_check[lp];
279  dirs_added++;
280  }
281  }
282 
283  // Fill in file entries.
284  SHedFile * p_hed_file = (SHedFile *)&p_hed[(dirs+1)];
285  for ( lp = 0; lp < dirs; lp++ )
286  {
287  p_hed[lp].p_fileList = p_hed_file;
288  int files_this_dir = 0;
289  for ( int lp2 = 0; lp2 < files; lp2++ )

292  {
293  p_hed_file->Offset = p_offset[lp2];
294  p_hed_file->FileSize = p_size[lp2];
295  p_hed_file->Checksum = file_check[lp2];

301  }

303 
304  // Get rid of temp data.
305  delete file_check;
306  delete dir_check;
307  delete p_size;
308  delete p_offset;
309  delete pHed;
310 
311  sLoadingHed = false;
312 
313  return p_hed;
