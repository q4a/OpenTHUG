115 void sOccluder::add_to_stack( sOcclusionPoly *p_poly )
116 {
117  if( NumOccluders < MAX_OCCLUDERS )
118  {
119  Dbg_Assert( p_poly->available );
120 
121  Occluders[NumOccluders].p_poly = p_poly;
122  p_poly->in_use = true;
123 
124  // Reset scores for all views.
125  memset( Occluders[NumOccluders].score, 0, sizeof( int ) * MAX_VIEWS_PER_OCCLUDER );
126 
127  ++NumOccluders;
128  }
129 }

137 static int cmp( const void *p1, const void *p2 )
138 {
139  // Sort based on the sum total of scores in all views.
140  int score1 = 0;
141  int score2 = 0;
142  for( uint v = 0; v < MAX_VIEWS_PER_OCCLUDER; ++v )
143  {
144  // Zero the score for any occlusion poly that is no longer available. This will force it out of the stack.
145  if(((sOccluder*)p1)->p_poly->available == false )
146  ((sOccluder*)p1)->score[v] = 0;
147 
148  if(((sOccluder*)p2)->p_poly->available == false )
149  ((sOccluder*)p2)->score[v] = 0;
150 
151  score1 += ((sOccluder*)p1)->score[v];
152  score2 += ((sOccluder*)p2)->score[v];
153  }
154 
155  return score1 < score2 ? 1 : score1 > score2 ? -1 : 0;
156 }

171 /******************************************************************/
172 /* */
173 /* */
174 /******************************************************************/
175 void sOccluder::tidy_stack( void )
176 {
177  if( NumOccluders > 0 )
178  {
179  // Sort in descending score order.
180  sort_stack();
181 
182  // Count backwards so we know we get all the bad occluders.
183  for( int i = NumOccluders - 1; i >= 0; --i )
184  {
185  // If we have hit an occluder with zero meshes culled, cut off the stack at this point.
186  int total_score = 0;
187  for( uint v = 0; v < MAX_VIEWS_PER_OCCLUDER; ++v )
188  {
189  total_score += Occluders[i].score[v];
190  }
191 
192  if( total_score == 0 )
193  {
194  // No longer using this poly.
195  Occluders[i].p_poly->in_use = false;
196 
197  // One less occluder to worry about.
198  --NumOccluders;
199  }
200  else
201  {
202  // Reset the good occluders.
203  Occluders[i].score[CurrentView] = 0;
204  }
205  }
206  }

223 void AddOcclusionPoly( Mth::Vector &v0, Mth::Vector &v1, Mth::Vector &v2, Mth::Vector &v3, uint32 checksum )
224 {
225  Dbg_Assert( NumOcclusionPolys < MAX_OCCLUSION_POLYS );
226 
227  OcclusionPolys[NumOcclusionPolys].in_use = false;
228  OcclusionPolys[NumOcclusionPolys].available = true;
229  OcclusionPolys[NumOcclusionPolys].checksum = checksum;
230  OcclusionPolys[NumOcclusionPolys].verts[0] = v0;
231  OcclusionPolys[NumOcclusionPolys].verts[1] = v1;
232  OcclusionPolys[NumOcclusionPolys].verts[2] = v2;
233  OcclusionPolys[NumOcclusionPolys].verts[3] = v3;
234  OcclusionPolys[NumOcclusionPolys].normal = Mth::CrossProduct( v1 - v0, v3 - v0 );
235  OcclusionPolys[NumOcclusionPolys].normal.Normalize();
236 
237  ++NumOcclusionPolys;
238 }

278 /* */

280 void CheckForOptimalOccluders( Mth::Vector &cam_pos, Mth::Vector &view_direction )
281 {
282  if( NumOcclusionPolys > 0 )
283  {
284  uint32 added = 0;
285  uint32 checked = 0;
286 
287  while( added < MAX_NEW_OCCLUSION_POLYS_TO_CHECK_PER_FRAME )
288  {
289  // Given the current position of the camera, check through the unused, available occlusion polys to see if one scores higher
290  // than the lowest scoring occlusion poly in use.
291  sOcclusionPoly *poly_to_check = &OcclusionPolys[NextOcclusionPolyToCheck++];
292  if(( !poly_to_check->in_use ) && ( poly_to_check->available ))
293  {
294  sOccluder::add_to_stack( poly_to_check );
295  ++added;
296  }
297  ++checked;
298 
299  // Ensure we are always checking within bounds.
300  if( NextOcclusionPolyToCheck >= NumOcclusionPolys )
301  {
302  NextOcclusionPolyToCheck = 0;
303  }
304 
305  // Quit out if we have less available occluders than spaces to fill.
306  if( checked >= NumOcclusionPolys )
307  {
308  break;
309  }
310  }
311  }
312 }

333  Dbg_Assert( (uint32)view < MAX_VIEWS_PER_OCCLUDER );
334  CurrentView = view;
335 
336  // Tidy up from last frame.
337  sOccluder::tidy_stack();
338 
339  // Cyclically add more occluders for checking.
340  CheckForOptimalOccluders( *p_cam_pos, *p_cam_pos );
341 
342  // Build all 5 planes for each occluder.
343  Mth::Vector u0, u1, p;
344 
345  // The order in which the verts are used to build tha planes depends upon where the camera is in relation to the occlusion
346  // poly. We use the default order when the viewpoint is on the side of the poly on which the default poly normal faces.
347  for( uint32 i = 0; i < sOccluder::NumOccluders; ++i )
348  {
349  sOcclusionPoly *p_poly = sOccluder::Occluders[i].p_poly;
350 
351 // Gfx::AddDebugLine( p_poly->verts[0], p_poly->verts[1], 0, 0, 1 );
352 // Gfx::AddDebugLine( p_poly->verts[1], p_poly->verts[2], 0, 0, 1 );

354 // Gfx::AddDebugLine( p_poly->verts[3], p_poly->verts[0], 0, 0, 1 );
355 
356  if( Mth::DotProduct( *p_cam_pos - p_poly->verts[0], p_poly->normal ) >= 0.0f )
357  {
358  // Start with the front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
359  // are considered occluded. (1->3->4)...
360  u0 = p_poly->verts[2] - p_poly->verts[0];
361  u1 = p_poly->verts[3] - p_poly->verts[0];
362  p = Mth::CrossProduct( u0, u1 );
363  p.Normalize();
364  p[W] = Mth::DotProduct( p, p_poly->verts[0] );
365  sOccluder::Occluders[i].planes[0] = p;
366 
367  // ...then left (0->1->2)...
368  u0 = p_poly->verts[0] - *p_cam_pos;
369  u1 = p_poly->verts[1] - *p_cam_pos;
370  p = Mth::CrossProduct( u0, u1 );
371  p.Normalize();
372  p[W] = Mth::DotProduct( p, *p_cam_pos );
373  sOccluder::Occluders[i].planes[1] = p;
374 
375  // ...then right (0->3->4)...
376  u0 = p_poly->verts[2] - *p_cam_pos;
377  u1 = p_poly->verts[3] - *p_cam_pos;
378  p = Mth::CrossProduct( u0, u1 );
379  p.Normalize();
380  p[W] = Mth::DotProduct( p, *p_cam_pos );
381  sOccluder::Occluders[i].planes[2] = p;
382 
383  // ...then top (0->2->3)...
384  u0 = p_poly->verts[1] - *p_cam_pos;
385  u1 = p_poly->verts[2] - *p_cam_pos;
386  p = Mth::CrossProduct( u0, u1 );
387  p.Normalize();
388  p[W] = Mth::DotProduct( p, *p_cam_pos );
389  sOccluder::Occluders[i].planes[3] = p;
390 
391  // ...then bottom (0->4->1)...
392  u0 = p_poly->verts[3] - *p_cam_pos;
393  u1 = p_poly->verts[0] - *p_cam_pos;
394  p = Mth::CrossProduct( u0, u1 );
395  p.Normalize();
396  p[W] = Mth::DotProduct( p, *p_cam_pos );
397  sOccluder::Occluders[i].planes[4] = p;
398  }
399  else
400  {
401  // Start with the front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
402  // are considered occluded. (1->4->3)...
403  u0 = p_poly->verts[3] - p_poly->verts[0];
404  u1 = p_poly->verts[2] - p_poly->verts[0];
405  p = Mth::CrossProduct( u0, u1 );
406  p.Normalize();
407  p[W] = Mth::DotProduct( p, p_poly->verts[0] );
408  sOccluder::Occluders[i].planes[0] = p;
409 
410  // ...then left (0->2->1)...
411  u0 = p_poly->verts[1] - *p_cam_pos;
412  u1 = p_poly->verts[0] - *p_cam_pos;
413  p = Mth::CrossProduct( u0, u1 );
414  p.Normalize();
415  p[W] = Mth::DotProduct( p, *p_cam_pos );
416  sOccluder::Occluders[i].planes[1] = p;
417 
418  // ...then right (0->4->3)...
419  u0 = p_poly->verts[3] - *p_cam_pos;
420  u1 = p_poly->verts[2] - *p_cam_pos;
421  p = Mth::CrossProduct( u0, u1 );
422  p.Normalize();
423  p[W] = Mth::DotProduct( p, *p_cam_pos );
424  sOccluder::Occluders[i].planes[2] = p;
425

450 bool TestSphereAgainstOccluders( Mth::Vector *p_sphere, uint32 meshes )
451 {
452  // Test against each occluder.
453  for( uint32 o = 0; o < sOccluder::NumOccluders; ++o )
454  {
455  bool occluded = true;
456 
457  // Test against each plane in the occluder.
458  for( uint32 p = 0; p < 5; ++p )
459  {
460  float result = ( sOccluder::Occluders[o].planes[p][X] * p_sphere->GetX() ) +
461  ( sOccluder::Occluders[o].planes[p][Y] * p_sphere->GetY() ) +
462  ( sOccluder::Occluders[o].planes[p][Z] * p_sphere->GetZ() ) -
463  ( sOccluder::Occluders[o].planes[p][W] );
464  if( result >= -p_sphere->GetW() )
465  {
466  // Outside of this plane, therefore not occluded by this occluder.
467  occluded = false;
468  break;
469  }
470  }
471 
472  if( occluded )
473  {
474  // Inside all planes, therefore occluded. Increase score for this occluder.
475  sOccluder::Occluders[o].score[CurrentView] += meshes;
476  return true;
477  }
478  }
479  return false;
480 }
