153 int wait_for_async_call_to_finish( Card* card )
154 {
155  int ret_val;
156 
157 
158 
159  sceMcSync( 0, NULL, &ret_val );
160  if( ret_val < 0 )
161  {
162  switch( ret_val )
163  {
164  case -1:
165  Dbg_Printf( "Switched to formated card\n" );
166  card->SetError( Card::vERROR_FORMATTED );
167  break;
168  case -2:
169  Dbg_Printf( "Error: Unformatted Card\n" );
170  card->SetError( Card::vERROR_UNFORMATTED );
171  break;
172  case -3:
173  Dbg_Printf( "Error: Insufficient Space\n" );
174  card->SetError( Card::vINSUFFICIENT_SPACE );
175  break;
176  case -4:
177  Dbg_Printf( "Error: Invalid Path\n" );
178  card->SetError( Card::vINVALID_PATH );
179  break;
180  case -5:
181  Dbg_Printf( "Error: Access Error\n" );
182  card->SetError( Card::vACCESS_ERROR );
183  break;
184  case -6:
185  Dbg_Printf( "Error: Subdir not empty\n" );
186  card->SetError( Card::vSUBDIR_NOT_EMPTY );
187  break;
188  case -7:
189  Dbg_Printf( "Error: Too many open files\n" );
190  card->SetError( Card::vTOO_MANY_OPEN_FILES );
191  break;
192  case -10:
193  Dbg_Printf( "Error: No card\n" );
194  card->SetError( Card::vNO_CARD );
195  break;
196  default:
197  card->SetError( Card::vUNKNOWN );
198  break;
199  }
200  }
201 
202  return ret_val;
203 }

520 // Ken: This used to be a local variable in Card::GetFileList, but sceMcGetDir requires that
521 // this buffer be 64 byte aligned, and occasionally it would not be, causing the contents of
522 // file_table to be undefined after the call to sceMcGetDir.
523 // Unfortunately, Sony did not make sceMcGetDir return an error code in this case, the blighters.
524 //
525 static sceMcTblGetDir file_table[Card::vMAX_FILES_PER_DIRECTORY];
526 
527 bool Card::GetFileList( const char* mask, Lst::Head< File > &file_list )
528 {
529 
530  int result, i;
531 
532  Dbg_MsgAssert( mask != NULL,( "Mask must be a valid search string. You may use wildcards\n" ));
533 
534  Dbg_MsgAssert( (((uint32)file_table)&63)==0, ("file_table not 64 byte aligned"));
535 
536 
537  int CallNumber=0;
538  while (true)
539  {
540  result = sceMcGetDir( m_port, m_slot, mask, CallNumber, vMAX_FILES_PER_DIRECTORY, file_table );
541  if( result != 0 )
542  {
543  SetError( vUNKNOWN );
544  return 0;
545  }
546 
547  int NumFilesReturned = wait_for_async_call_to_finish( this );
548  for( i = 0; i < NumFilesReturned; i++ )
549  {
550  File *new_file;
551 
552  new_file = new File( 0, this );
553 
554  strcpy( new_file->m_Filename, (char *) file_table[i].EntryName );
555  new_file->m_Size = file_table[i].FileSizeByte;
556 
557  new_file->m_Attribs = 0;
558  if( file_table[i].AttrFile & sceMcFileAttrReadable )
559  {
560  new_file->m_Attribs |= File::mATTRIB_READABLE;
561  }
562  if( file_table[i].AttrFile & sceMcFileAttrWriteable )
563  {
564  new_file->m_Attribs |= File::mATTRIB_WRITEABLE;
565  }
566  if( file_table[i].AttrFile & sceMcFileAttrExecutable )
567  {
568  new_file->m_Attribs |= File::mATTRIB_EXECUTABLE;
569  }
570  if( file_table[i].AttrFile & sceMcFileAttrDupProhibit )
571  {
572  new_file->m_Attribs |= File::mATTRIB_DUP_PROHIBITED;
573  }
574  if( file_table[i].AttrFile & sceMcFileAttrSubdir )
575  {
576  new_file->m_Attribs |= File::mATTRIB_DIRECTORY;
577  }
578  if( file_table[i].AttrFile & sceMcFileAttrClosed )
579  {
580  new_file->m_Attribs |= File::mATTRIB_CLOSED;
581  }
582  if( file_table[i].AttrFile & sceMcFileAttrPDAExec )
583  {
584  new_file->m_Attribs |= File::mATTRIB_PDA_APP;
585  }
586  if( file_table[i].AttrFile & sceMcFileAttrPS1 )
587  {
588  new_file->m_Attribs |= File::mATTRIB_PS1_FILE;
589  }
590 
591  sceCdCLOCK myClock;
592  myClock.day=itob(file_table[i]._Modify.Day);
593  myClock.hour=itob(file_table[i]._Modify.Hour);
594  myClock.minute=itob(file_table[i]._Modify.Min);
595  myClock.month=itob(file_table[i]._Modify.Month);
596  myClock.second=itob(file_table[i]._Modify.Sec);
597  myClock.year=itob(file_table[i]._Modify.Year%100);
598  sceScfGetLocalTimefromRTC(&myClock);
599  new_file->m_Modified.m_Seconds = btoi(myClock.second);
600  new_file->m_Modified.m_Minutes = btoi(myClock.minute);
601  new_file->m_Modified.m_Hour = btoi(myClock.hour);
602  new_file->m_Modified.m_Day = btoi(myClock.day);
603  new_file->m_Modified.m_Month = btoi(myClock.month);
604  new_file->m_Modified.m_Year = btoi(myClock.year)+2000;
605 
606  myClock.day=itob(file_table[i]._Create.Day);
607  myClock.hour=itob(file_table[i]._Create.Hour);
608  myClock.minute=itob(file_table[i]._Create.Min);
609  myClock.month=itob(file_table[i]._Create.Month);
610  myClock.second=itob(file_table[i]._Create.Sec);
611  myClock.year=itob(file_table[i]._Create.Year%100);
612  sceScfGetLocalTimefromRTC(&myClock);
613  new_file->m_Created.m_Seconds = btoi(myClock.second);
614  new_file->m_Created.m_Minutes = btoi(myClock.minute);
615  new_file->m_Created.m_Hour = btoi(myClock.hour);
616  new_file->m_Created.m_Day = btoi(myClock.day);
617  new_file->m_Created.m_Month = btoi(myClock.month);
618  new_file->m_Created.m_Year = btoi(myClock.year)+2000;
619 
620  file_list.AddToTail( new_file );
621  }
622 
623  if (NumFilesReturned<vMAX_FILES_PER_DIRECTORY)
624  {
625  break;
626  }
627  ++CallNumber;
628  }
629 
630  return true;
631 }
