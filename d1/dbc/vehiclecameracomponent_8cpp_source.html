133 void CVehicleCameraComponent::Update()
134 {
135  // NOTE: plenty of room for optimiziation
136 
137  GetCameraComponentFromObject(GetObject())->StoreOldPosition();
138 
139  calculate_attractor_direction();
140 
141  // Due to rounding errors this can sometimes be > |1|, which hoses acosf(), so limit here.
142  float angular_distance = acosf(Mth::Clamp(Mth::DotProduct(m_direction, m_attractor_direction), -1.0f, 1.0f));
143  if (angular_distance > Mth::PI / 2.0f)
144  {
145  angular_distance = Mth::PI - angular_distance;
146  }
147 
148  bool sign = Mth::CrossProduct(m_direction, m_attractor_direction)[Y] > 0.0f;
149 
150  float step = m_alignment_rate * angular_distance * Tmr::FrameLength();
151 
152  if (step > angular_distance)
153  {
154  step = angular_distance;
155  }
156 
157  m_direction.RotateY((sign ? 1.0f : -1.0f) * step);
158  m_direction.Normalize();
159 
160  calculate_dependent_variables();
161 
162  ApplyCameraCollisionDetection(
163  m_pos,
164  m_orientation_matrix,
165  m_pos - m_offset_distance * m_orientation_matrix[Z],
166  m_pos - m_offset_distance * m_orientation_matrix[Z],
167  false,
168  false
169  );
170 
171  apply_state();
172 }

242 void CVehicleCameraComponent::calculate_attractor_direction ( )
243 {
244  Mth::Vector vel_direction = -mp_subject_vehicle_component->GetVel();
245  vel_direction[Y] = 0.0f;
246  float vel = vel_direction.Length();
247  vel_direction.Normalize();
248 
249  if (mp_subject_vehicle_component->GetNumWheelsInContact() < 2)
250  {
251  // if vel under certain threshold, we lock the attractor
252  if (vel < vLOCK_ATTRACTOR_VELOCITY_THRESHOLD)
253  {
254  return;
255  }
256 
257  m_attractor_direction = vel_direction;
258  }
259  else
260  {
261  float vel_weight = Mth::ClampMax(vel / vVELOCITY_WEIGHT_DROP_THRESHOLD, 1.0f) * Mth::DotProduct(vel_direction, -mp_subject->GetMatrix()[Z]);
262 
263  m_attractor_direction = -mp_subject->GetMatrix()[Z];
264  m_attractor_direction[Y] = 0.0f;
265  m_attractor_direction.Normalize();
266 
267  if (vel_weight > 0.0f)
268  {
269  m_attractor_direction += vel_weight * vel_direction;
270  m_attractor_direction.Normalize();
271  }
272 
273  // NOTE: potential bug if car is pointing straight down
274  }
275 }

282 void CVehicleCameraComponent::calculate_dependent_variables ( )
283 {
284  Mth::Vector frame_direction = m_direction;
285  frame_direction.RotateY(mp_camera_lookaround_component->mLookaroundHeading);
286 
287  m_orientation_matrix[X].Set(frame_direction[Z], 0.0f, -frame_direction[X]);
288  m_orientation_matrix[Y].Set(0.0f, 1.0f, 0.0f);
289  m_orientation_matrix[Z] = frame_direction;
290  m_orientation_matrix[W].Set();
291 
292  m_pos = mp_subject->GetPos() + Mth::Vector(m_offset_distance * frame_direction[X], m_offset_height, m_offset_distance * frame_direction[Z]);
293 
294  m_orientation_matrix.RotateZLocal(DEGREES_TO_RADIANS(m_angle));
295 }
