246 static void s_remove_asset( const char* pFileName )
247 {
248  // if the file is in the asset manager, unload it so
249  // that the artists can display the latest version
250  // ("cas_artist" must be set to 1, or else it'll just
251  // reload it from the PRE file)
252 
253  if ( !Script::GetInt( "cas_artist", false ) )
254  {
255  Dbg_Message( "Warning: Can't reload asset %s from disk (need to define cas_artist=1)", pFileName );
256  return;
257  }
258 
259  // get the assman
260  Ass::CAssMan * ass_man = Ass::CAssMan::Instance();
261 
262  // kills the original asset
263  Ass::CAsset* pAsset = ass_man->GetAssetNode( Script::GenerateCRC( pFileName ), false );
264 
265  if ( pAsset )
266  {
267  Dbg_Message( "Unloading asset %s", pFileName );
268  ass_man->DestroyReferences( pAsset );
269  ass_man->UnloadAsset( pAsset );
270  }
271  else
272  {
273  Dbg_Message( "Couldn't find asset %s to unload", pFileName );
274  }
275 }

282 void CViewerObject::LoadModel( Script::CStruct* pNodeData )
283 {
284  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
285  skate_mod->GetObjectManager()->RegisterObject( *this );
286  MovingObjectCreateComponents();
287  MovingObjectInit( pNodeData, skate_mod->GetObjectManager() );
288 
289  Nx::CEngine::sFinishRendering();
290 
291  if (Config::GotExtraMemory())
292  {
293  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().DebugHeap());
294  }
295  else
296  {
297  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().BottomUpHeap());
298  }
299 
300  Dbg_Assert( pNodeData );
301 
302  // get rid of old model, if
303  // it was stored in the assman
304  int skaterProfileNumber;
305  const char* pModelName;
306  uint32 profileName;
307 
308  for ( int i = 0; i < Obj::CModelComponent::vNUM_LODS; i++ )
309  {
310  char paramName[256];
311  sprintf( paramName, "model%d", i );
312 
313  if ( pNodeData->GetText( paramName, &pModelName, false ) )
314  {
315  Dbg_Message( "Removing model LOD data from asset manager, in order to replace it with new data" );
316 
317  Str::String fullModelName;
318  fullModelName = Gfx::GetModelFileName(pModelName, ".skin");
319  s_remove_asset( fullModelName.getString() );
320 
321  fullModelName = Gfx::GetModelFileName(pModelName, ".mdl");
322  s_remove_asset( fullModelName.getString() );
323  }
324  }
325 
326  if ( pNodeData->GetText( "modelName", &pModelName, false )
327  || pNodeData->GetText( "model", &pModelName, false ) )
328  {
329  Dbg_Message( "Removing model data from asset manager, in order to replace it with new data" );
330 
331  Str::String fullModelName;
332  fullModelName = Gfx::GetModelFileName(pModelName, ".skin");
333  s_remove_asset( fullModelName.getString() );
334 
335  fullModelName = Gfx::GetModelFileName(pModelName, ".mdl");
336  s_remove_asset( fullModelName.getString() );
337  }
338  else if ( pNodeData->GetChecksum( "profile", &profileName, false ) )
339  {
340  Dbg_Message( "Warning: can't remove old model data from asset manager... asset not reloaded from disk." );
341  }
342  else if ( pNodeData->GetInteger( "skater_profile_index", &skaterProfileNumber, false ) )
343  {
344  Dbg_Message( "Warning: can't remove old model data from asset manager... asset not reloaded from disk." );
345 
346  Obj::CSkaterProfile* pSkaterProfile = Mdl::Skate::Instance()->GetProfile( skaterProfileNumber );
347  Script::CStruct* pAppearanceStruct = pSkaterProfile->GetAppearance()->GetStructure();
348  Script::CStruct* pNewStruct = new Script::CStruct;
349  pNewStruct->AppendStructure( pAppearanceStruct );
350  pNodeData->AddStructurePointer( "profile", pNewStruct );
351  // don't need to delete the structure,
352  // because it gets added permanently to the struct
353  }
354  else
355  {
356  Dbg_MsgAssert( 0, ( "Unrecognized parameters for model loading" ) );
357  }
358 
359  m_pos = vPOS_VIEWEROBJECT;
360  m_matrix.Ident();
361 
362  Script::RunScript( CRCD(0x24c71bc7,"viewerobj_add_components"), pNodeData, this );
363 
364  Finalize();
365 
366  Script::RunScript( CRCD(0xbbab79de,"viewerobj_init_model"), pNodeData, this );
367 
368  // loop the idle anim, if it exists
369  CViewerObject::SetAnim( CRCD(0x23db4aea,"idle") );
370 
371  Mem::Manager::sHandle().PopContext();
372 }

474 void CViewerObject::IncrementFrame( bool forwards )
475 {
476  if ( GetAnimationComponent() )
477  {
478  if ( forwards )
479  {
480  // frame-rate independent
481  GetAnimationComponent()->AddTime( 1.0f / 60.0f );
482  GetAnimationComponent()->PrintStatus();
483 
484  Script::CStruct* pTempStruct = new Script::CStruct;
485  pTempStruct->AddFloat( CRCD(0x06c9f278,"incVal"), 1.0f / 60.0f );
486  GetAnimationComponent()->CallMemberFunction( CRCD(0x6aaeb76f,"IncrementPartialAnimTime"), pTempStruct, NULL );
487  delete pTempStruct;
488  }
489  else
490  {
491  // reverse direction so that it loops properly
492  GetAnimationComponent()->ReverseDirection( false );
493 
494  // frame-rate independent
495  GetAnimationComponent()->AddTime( -1.0f / 60.0f );
496  GetAnimationComponent()->PrintStatus();
497 
498  GetAnimationComponent()->ReverseDirection( false );
499 
500  GetAnimationComponent()->CallMemberFunction( CRCD(0xf5e2b871,"ReversePartialAnimDirection"), NULL, NULL );
501 
502  Script::CStruct* pTempStruct = new Script::CStruct;
503  pTempStruct->AddFloat( CRCD(0x06c9f278,"incVal"), -1.0f / 60.0f );
504  GetAnimationComponent()->CallMemberFunction( CRCD(0x6aaeb76f,"IncrementPartialAnimTime"), pTempStruct, NULL );
505  delete pTempStruct;
506 
507  GetAnimationComponent()->CallMemberFunction( CRCD(0xf5e2b871,"ReversePartialAnimDirection"), NULL, NULL );
508  }
509 
510  // don't want to call the animation component's
511  // Update() function, because that changes the
512  // time... instead, call UpdateSkeleton()
513  // which syncs the skeleton to the current time
514  GetAnimationComponent()->UpdateSkeleton();
515 
516  // reset the paused state...
517  GetAnimationComponent()->Suspend( true );
518  }
519 
520  m_paused = true;
521 }

564 void CViewerObject::UpdateWheels()
565 {
566  Nx::CHierarchyObject* pHierarchyObjects = GetModel()->GetHierarchy();
567  Dbg_Assert( pHierarchyObjects );
568 
569  Gfx::CSkeleton* pCarSkeleton = GetSkeletonComponent()->GetSkeleton();
570  Dbg_Assert( pCarSkeleton );
571 
572  Dbg_MsgAssert( GetModel()->GetNumObjectsInHierarchy()==pCarSkeleton->GetNumBones(),
573  ( "Skeleton and hierarchy doesn't match (%d hierarchy objs vs %d bones)",
574  GetModel()->GetNumObjectsInHierarchy(), pCarSkeleton->GetNumBones() ) );
575 
576  static float s_wheelRotationX = 0.0f;
577 
578  // not frame rate independent (doesn't have valid m_time)
579  s_wheelRotationX += ( Script::GetFloat("max_wheel_speed", Script::ASSERT) * m_animSpeed );
580 
581  // make sure it's within bounds
582  while ( s_wheelRotationX > 360.0f )
583  {
584  s_wheelRotationX -= 360.0f;
585  }
586 
587  while ( s_wheelRotationX < 0.0f )
588  {
589  s_wheelRotationX += 360.0f;
590  }
591 
592  if ( GetModel()->GetNumObjectsInHierarchy() > 5 )
593  {
594  Mth::Matrix* pMatrices = pCarSkeleton->GetMatrices();
595  Dbg_Assert( pMatrices );
596 
597  // initialize the setup matrix once per frame
598  for ( int i = 0; i < GetModel()->GetNumObjectsInHierarchy(); i++ )
599  {
600  *(pMatrices + i) = ( pHierarchyObjects + i )->GetSetupMatrix();
601  }
602 
603  // apply the appropriate rotations to the wheels
604  for ( int i = 2; i < 6; i++ )
605  {
606  (pMatrices + i)->RotateXLocal( s_wheelRotationX * 2.0f * Mth::PI / 360.0f );
607  }
608 
609  // get children into object space
610  for ( int i = 1; i < 6; i++ )
611  {
612  *(pMatrices + i) *= *(pMatrices + 0);
613  }
614  }
615 }

678 void CViewerObject::s_input_logic_code ( const Inp::Handler < CViewerObject >& handler )
679 {
680  CViewerObject& obj = handler.GetData();

682  // this is the minimum amount of time
683  // you need to hold down the left/right keys
684  // before it will auto repeat
685  int vAUTO_REPEAT_TIME = Script::GetInteger( CRCD(0xe60442d5,"ViewerAutoRepeatTime"), Script::NO_ASSERT );
686 
687  // this is how fast it auto repeats
688  int vAUTO_REPEAT_SPEED = Script::GetInteger( CRCD(0x2da212e5,"ViewerAutoRepeatSpeed"), Script::NO_ASSERT );
689
