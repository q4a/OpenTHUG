68 void* Pile::allocate( size_t size, bool assert_on_fail )
69 {
70 
71 
72 #ifdef __PLAT_NGC__
73  size = (uint)nAlignUpBy( size, 5 );
74 #else
75  size = (uint)nAlignUpBy( size, 4 );
76 #endif
77 
78  BlockHeader* p_header = (BlockHeader*)mp_region->Allocate( this, size + BlockHeader::sSize, assert_on_fail );
79 
80  if ( p_header )
81  {
82  new ((void*)p_header) BlockHeader( this, size );
83 
84  Dbg_Code
85  (
86  m_alloc_count++;
87  )
88  }
89  else if ( assert_on_fail )
90  {
91  Dbg_MsgAssert ( false,( "Failed to allocate %d bytes", size ));
92  }
93 
94  MemDbg_AllocateBlock ( p_header );
95 
96  return (void*)((uint)p_header + BlockHeader::sSize);
97 
98 }
