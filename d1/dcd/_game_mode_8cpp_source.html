92 void CGameMode::clear_victorycondition_list()
93 {
94 
95 
96  Lst::Node<CVictoryCondition> *pNode = m_VictoryConditionList.GetNext();
97  while(pNode)
98  {
99  Lst::Node<CVictoryCondition> *pNext = pNode->GetNext();
100  pNode->Remove();
101  delete pNode->GetData();
102  delete pNode;
103  pNode = pNext;
104  }
105 
106  Dbg_Assert( m_VictoryConditionList.IsEmpty() );
107 }

114 void CGameMode::build_victorycondition_list()
115 {
116 
117 
118  clear_victorycondition_list();
119 
120  Script::CArray* pArray;
121  if ( !m_ModeDefinition.GetArray( CRCD(0x94a24c5e,"victory_conditions"), &pArray ) )
122  {
123  // no victory conditions found
124  return;
125  }
126 
127  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().SkaterInfoHeap());
128 
129  // loop through the victory conditions
130  for ( uint32 i = 0; i < pArray->GetSize(); i++ )
131  {
132  Script::CScriptStructure* pStructure = pArray->GetStructure( i );
133  Dbg_Assert( pStructure );
134 
135  CVictoryCondition* pVictoryCondition = CreateVictoryConditionInstance( pStructure );
136  Dbg_Assert( pVictoryCondition );
137  m_VictoryConditionList.AddToTail( new Lst::Node<CVictoryCondition>( pVictoryCondition ) );
138  }
139 
140  Mem::Manager::sHandle().PopContext();
141  // TODO: Scan for combinations that don't make sense
142  // (i.e. highest score AND best combo)
143 }

232 bool
233 CGameMode::EndConditionsMet()
234 {
235  // this checks whether time has run out
236  if ( ShouldUseClock() )
237  {
238  CGoalManager* pGoalManager = Game::GetGoalManager();
239 
240  if( pGoalManager->GetNumActiveGoals( true ) > 0 && pGoalManager->GetGoalTime() <= 0 )
241  {
242  return true;
243  }
244  }
245 
246  uint32 i;
247  int num_terminals = 0;
248  int num_completed_terminals = 0;
249 
250  // TODO: Allow boolean conditions (&'s and |'s)
251 
252  // Look through the victory conditions also
253  for ( i = 0; i < m_VictoryConditionList.CountItems(); i++ )
254  {
255  Lst::Node<CVictoryCondition> *pNode = m_VictoryConditionList.GetItem( i );
256 
257  if (pNode->GetData()->IsTerminal())
258  {
259  num_terminals++;
260 
261  if ( pNode->GetData()->ConditionComplete() )
262  {
263  num_completed_terminals++;
264  }
265  }
266  }
267 
268  // If all of the terminal ones have been met,
269  // then we've also met our end conditions
270  if ( num_terminals && ( num_completed_terminals == num_terminals ) )
271  {
272  return true;
273  }
274 
275  return false;
276 }

283 bool CGameMode::ShouldUseClock()
284 {
285  int time_limit;
286 
287  if( ( GetNameChecksum() == CRCD(0x5d32129c,"king")) ||
288  ( GetNameChecksum() == CRCD(0x6ef8fda0,"netking")) ||
289  ( GetNameChecksum() == CRCD(0xf135ecb6,"scorechallenge")) ||
290  ( GetNameChecksum() == CRCD(0x1498240a,"netscorechallenge")) ||
291  // K: Added check for creategoals to fix bug where end of run triggered straight away on
292  // entering a level in creategoals mode.
293  ( GetNameChecksum() == CRCD(0xe45f0ca2,"creategoals")) ||
294  ( GetNameChecksum() == CRCD(0x3d6d444f,"firefight")) ||
295  ( GetNameChecksum() == CRCD(0xbff33600,"netfirefight")))
296  {
297  return false;
298  }
299 
300  m_ModeDefinition.GetInteger( CRCD(0x8d38a280,"default_time_limit"), &time_limit, true );
301  return ( time_limit != 0 );
302 }

615 Nx::ScreenMode CGameMode::GetScreenMode()
616 {
617 
618 
619  uint32 checksum;
620 
621  m_ModeDefinition.GetChecksum( CRCD(0x9bab2eae,"screenmode"), &checksum, true );
622  switch (checksum)
623  {
624  case 0x3558d8e6: // single
625  return Nx::vONE_CAM;
626  break;
627  case 0x9d65d0e7: // horse
628  return Nx::vHORSE1;
629  break;
630  case 0x06ab02f2: // splitscreen
631  {
632  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
633  Prefs::Preferences* pPreferences;
634 
635  pPreferences = skate_mod->GetSplitScreenPreferences();
636  Dbg_Assert(pPreferences);
637 
638  Script::CScriptStructure* pStructure = pPreferences->GetPreference( CRCD(0x6f82c71e,"viewport_type") );
639  Dbg_Assert(pStructure);
640 
641  uint32 checksum;
642  pStructure->GetChecksum( CRCD(0x21902065,"checksum"), &checksum, true );
643  if ( checksum == CRCD(0x92f5b4cb,"viewport_type_vertical") )
644  {
645  return Nx::vSPLIT_V;
646  }
647  else if ( checksum == CRCD(0x7c442511,"viewport_type_horizontal") )
648  {
649  return Nx::vSPLIT_H;
650  }
651  else
652  {
653  Dbg_Assert( 0 );
654  return Nx::vSPLIT_V;
655  }
656  }
657  break;
658  default:
659  Dbg_Assert( 0 );
660  return (Nx::ScreenMode) 0;
661  }
662 }

756 
757 // if the checksum is the name of a net game, then return the name of the regular equivalent
758 // used by the time functions, wehre we need to know what type of game it is
759 // and not if it's net or not
760 uint32 CGameMode::ConvertNetNameChecksum(uint32 nameChecksum)
761 {
762 
763  uint32 checksum = nameChecksum;
764 
765  if( nameChecksum == CRCD(0x30c2ffa3,"nettrickattack"))
766  {
767  checksum = CRCD( 0xf9283ee7,"trickattack" );
768  }
769  else if( nameChecksum == CRCD( 0x5e2ea50c, "netgraffiti" ))
770  {
771  checksum = CRCD(0xc8a82b5a, "graffiti");
772  }
773  else if( nameChecksum == CRCD( 0xc50affd0, "netcombomambo" ))
774  {
775  checksum = CRCD(0x23eb3da3, "combomambo");
776  }
777  else if ( nameChecksum == CRCD( 0xf9d5d933, "netslap" ))
778  {
779  checksum = CRCD( 0xca1f360f, "slap" );
780  }
781  else if ( nameChecksum == CRCD(0x6ef8fda0, "netking" ) )
782  {
783  checksum = CRCD( 0x5d32129c,"king" );
784  }
785  else if( nameChecksum == CRCD(0xbff33600,"netfirefight") )
786  {
787  checksum = CRCD(0x3d6d444f,"firefight");
788  }
789  else if( nameChecksum == CRCD(0x6c5ff266,"netctf") )

802 int compFunc( const void *arg1, const void *arg2 )
803 {

808  int score2 = pRanking2->pScore->GetTotalScore();

811  {
812  return 0;
813  }
814  else if ( score1 < score2 )
815  {
816  return 1;
817  }
