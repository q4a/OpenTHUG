250  if ( pDetails )

254  aborted = true;
255  }
256  }
257  else
258  {
259  // delete as many skippable movies as there are...
260  CMovieDetails* pDetails = get_movie_details( 0 );
261  while ( pDetails && pDetails->IsSkippable() )
262  {
263  CleanupMovie( pDetails );
264 
265  pDetails = get_movie_details( 0 );
266  aborted = true;
267  }
268  }
269 
270  // now that a new movie is about to begin,
271  // reset its custom keys
272  CMovieDetails* pDetails = get_movie_details( 0 );
273  if ( pDetails )
274  {
275  pDetails->ResetCustomKeys();
276  }
277 
278  // restore skater cam
279  // Nx::CViewportManager::sSetCamera( 0, Mdl::Skate::Instance()->GetSkater(0)->GetCamera() );
280 
281  // if there are no more movies, then restore the camera
282 // if (aborted) <-- causes one frame glitch with queued up cameras
283  if ( !aborted && (get_movie_count() == 0) )
284  {
285 // Script::RunScript(CRCD(0x15674315, "Restore_skater_camera")); // Temp pathc to get the skater camera back at end of cutscenes
286  }
287 
288  return true;

461 bool CMovieManager::AddMovieToQueue( uint32 name, Script::CStruct* pParams, uint32 movieType )
462 {
463  CMovieDetails* pDetails = NULL;
464 
465  bool old_last_movie_camera = m_use_last_movie_camera;
466 
467  // GJ: a new camera has been queued up, so don't need
468  // to do anything special to fix the 1-frame glitch
469  // (otherwise, some of the cutscenes will have the
470  // wrong camera for the first frame...) must be done before
471  // the cutscenedetails constructor, which updates the model component,
472  // which contains a check for "IsRolling"
473  m_use_last_movie_camera = false;
474 
475  switch ( movieType )
476  {
477  case 0xf5012f52: // cutscene
478 
479  // check for multiple movies in the queue
480  if ( get_movie_count() != 0 )
481  {
482  for ( int i = 0; i < get_movie_count(); i++ )
483  {
484  CMovieDetails* pDetails = get_movie_details( i );
485  if ( pDetails->GetName() == CRCD(0xf5012f52,"cutscene") )
486  {
487  // GJ: theoretically, we should be able to just clear the movie queue
488  // to get rid of the extra cutscene, but i suspect that there will be
489  // problems with cleanup, and that it would be too risky to do
490  // that so close to the end of the project...
491  Dbg_MsgAssert( 0, ( "Cutscenes should not be queued (only 1 cutscene at a time, please)" ) );
492  }
493  }
494  // if we've currently got skatercamanims in the queue,
495  // then trash them... (due to the convoluted level
496  // loading process, it's possible for a skatercamanim
497  // to be launched on the same frame as a cutscene;
498  // in this case, we want the cutscene to take
499  // precedence)
500  ClearMovieQueue();
501 
502  // we should set this to false, since
503  // we know we're about to add a new movie
504  // in a moment anyway. (otherwise, modelcomponent
505  // will assert because it doesn't expect the
506  // cutscene camera to be active...
507  m_use_last_movie_camera = false;
508  }
509 
510  {
511  // fix 1-frame glitch with a loading screen here...
512  char* pFileName = NULL;
513  bool freeze = true;
514  bool blank = false;
515  Nx::CLoadScreen::sDisplay( pFileName, freeze, blank );
516  }
517 
518  pDetails = new CCutsceneDetails;
519  Dbg_Message( "Trying to create new cutscene" );
520  break;
521  case 0x12743edb: // objectanim
522  pDetails = new CObjectAnimDetails;
523  Dbg_Message( "Trying to create new object anim" );
524  break;
525  case 0x66b7dd11: // skatercamanim
526  pDetails = new CSkaterCamDetails;
527  Dbg_Message( "Trying to create new skater cam anim" );
528  break;
529  default:
530  Dbg_MsgAssert( 0, ( "Unrecognized movie type %s", Script::FindChecksumName(movieType) ) );
531  break;
532  }
533 
534  Dbg_Assert( pDetails );
535 
536  pDetails->SetName( name );
537 
538  if ( !pDetails->InitFromStructure( pParams ) )
539  {
540  // failed
541  delete pDetails;
542 
543  // restore the old "last movie camera" flag
544  m_use_last_movie_camera = old_last_movie_camera;
545 
546  return false;
547  }
548 
549  m_movieDetailsList.AddToTail( pDetails );
550  //Dbg_Message("Adding a movie on frame %ld. Count now %d\n", Tmr::GetRenderFrame(), get_movie_count());
551 
552  // GJ: if it's the first movie in the queue,
553  // then set the correct camera (to prevent
554  // one frame glitch between movies)
555  if ( get_movie_count() == 1 )
556  {
557  Nx::CViewportManager::sSetCamera( 0, this->GetActiveCamera() );
558  }
559 
560  // TODO: Support the following parameters
561 // pDetails->SetTarget( Obj::MovingObject* pObject );
562 // pDetails->SetFocusSkater( pParams->ContainsFlag("focus_skater") );
563 // pDetails->SetFocusTarget( pParams->ContainsFlag("focus_target") );
564 
565  return true;
566 }

583 bool CMovieManager::Update()
584 {
585  // Clear out old camera if we advanced a frame
586  if (m_use_last_movie_camera && (m_last_movie_frame <= Tmr::GetRenderFrame()))
587  {
588  //Dbg_Message("Turning off camera on frame %d", Tmr::GetRenderFrame());
589  m_use_last_movie_camera = false;
590 
591  if ( get_movie_count() == 0 )
592  {
593 // Script::RunScript(CRCD(0x15674315, "Restore_skater_camera")); // Temp pathc to get the skater camera back at end of cutscenes
594  }
595  }
596 
597  CMovieDetails* pDetails = get_movie_details( 0 );
598 
599  // there are no movies in the queue,
600  // so return "no movies" code
601  // (the camera manager should then immediately
602  // pass control back to the skater cam)
603  if ( !pDetails )
604  {
605  return false;
606  }
607 
608 #if 0
609  printf( "Movie manager update %d:\n", get_movie_count() );
610  for ( int i = 0; i < get_movie_count(); i++ )
611  {
612  printf( "\tmovie name = %s\n", Script::FindChecksumName(get_movie_details(i)->GetName()) );
613  }
614 #endif
615 
616  if ( Mdl::FrontEnd::Instance()->GamePaused() && ( Mdl::CViewer::sGetViewMode() == 0 ) )
617  {
618  // printf("game is paused in normal view mode\n");
619  if ( !pDetails->ShouldPauseMovie() )
620  {
621  pDetails->Update();
622  }
623  }
624  else
625  {
626  // printf("view mode is not normal\n");
627  pDetails->Update();
628  }
629 
630  // if the animation is done,
631  // the remove it from the list
632  if ( pDetails->IsComplete() )
633  {
634  // TODO: run callback function if necessary?
635 
636  AbortCurrentMovie( false );
637  }
638 
639  return ( get_movie_count() != 0 );
640 }
