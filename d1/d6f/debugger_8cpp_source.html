564 void CScriptDebugger::set_object_update_mode(SObjectUpdateMode *pObjectUpdateMode)
565 {
566  // Make sure the game is not paused, which it will be if in single step mode.
567  Dbg::CScriptDebugger::Instance()->ScriptDebuggerUnpause();
568 
569  switch (pObjectUpdateMode->mMode)
570  {
571  case Dbg::EVERY_FRAME:
572  {
573  Obj::CCompositeObject* p_obj=(Obj::CCompositeObject*)Obj::CCompositeObjectManager::Instance()->GetObjectByID(pObjectUpdateMode->mObjectId);
574  if (p_obj)
575  {
576  p_obj->SetDebuggerEveryFrameFlag(pObjectUpdateMode->mOn);
577  }
578  break;
579  }
580  case Dbg::SINGLE_STEP:
581  if (pObjectUpdateMode->mOn)

631 void CScriptDebugger::send_script_global_info(uint32 id)
632 {
633  // Look up the symbol to see if it is the name of some global.
634  Script::CSymbolTableEntry *p_entry=Script::LookUpSymbol(id);
635  if (p_entry)
636  {
637  int message_size=0;
638  uint8 *p_buf=(uint8*)gpDebugInfoBuffer;
639 
640  // Write in the name, type, and source file name.
641  Dbg_MsgAssert(SCRIPT_DEBUGGER_BUFFER_SIZE > 8,("SCRIPT_DEBUGGER_BUFFER_SIZE is like way too small dude"));
642  *(uint32*)p_buf=id;
643  p_buf+=4;
644  *(uint32*)p_buf=p_entry->mType;
645  p_buf+=4;
646  message_size+=8;
647 
648  const char *p_source_filename=Script::FindChecksumName(p_entry->mSourceFileNameChecksum);
649  int len=strlen(p_source_filename)+1;
650  Dbg_MsgAssert(SCRIPT_DEBUGGER_BUFFER_SIZE-message_size > len,("SCRIPT_DEBUGGER_BUFFER_SIZE is too small"));
651 
652  strcpy((char*)p_buf,p_source_filename);
653  p_buf+=len;
654  message_size+=len;
655 
656  // For pure data types, write in a structure containing the value.
657  switch (p_entry->mType)
658  {
659  case ESYMBOLTYPE_INTEGER:
660  case ESYMBOLTYPE_FLOAT:
661  case ESYMBOLTYPE_STRING:
662  case ESYMBOLTYPE_LOCALSTRING:
663  case ESYMBOLTYPE_PAIR:
664  case ESYMBOLTYPE_VECTOR:
665  case ESYMBOLTYPE_STRUCTURE:
666  case ESYMBOLTYPE_ARRAY:
667  case ESYMBOLTYPE_NAME:
668  {
669  // Copy the name and value into a structure for sending to the
670  // script debugger for display.
671  Script::CStruct *p_struct=new Script::CStruct;
672 
673  // Create a structure component and fill it in with the value of the global.
674  Script::CComponent *p_source=new Script::CComponent;
675  p_source->mType=p_entry->mType;
676  p_source->mUnion=p_entry->mUnion; // What mUnion represents depends on mType
677

679  // be a pointer to some entity, and we don't want that to be deleted when p_struct
680  // gets deleted.
681  Script::CComponent *p_new=new Script::CComponent;
682  Script::CopyComponent(p_new,p_source);
683  // Delete p_source now that we've got a copy.
684  p_source->mUnion=0;
685  delete p_source;
686

688  // Ie, if Foo is a global with value 6, then it will appear as Foo=6 in the structure.
689  p_new->mNameChecksum=id;
690  p_struct->AddComponent(p_new);
691 
692 
693  int structure_bytes_written=Script::WriteToBuffer(p_struct,p_buf,SCRIPT_DEBUGGER_BUFFER_SIZE-message_size);
694  delete p_struct;
695 
696  p_buf+=structure_bytes_written;
697  message_size+=structure_bytes_written;
698  break;
699  }
700  default:
701  break;
702  }
703 
704  // Send the info to the debugger.
705  Net::MsgDesc msg;
706  msg.m_Data = gpDebugInfoBuffer;
707  msg.m_Length = message_size;
708  msg.m_Id = GameNet::vMSG_ID_DBG_SCRIPT_GLOBAL_INFO;

1157 // This can be used in place of EnqueueMessage or StreamMessage when the message
1158 // data is large, ie bigger than the maximum packet size that EnqueueMessage will allow,
1159 // which is about 1300 bytes.
1160 // The advantage of EnqueueMessage is that the messages arrive very quickly at the other end.
1161 // The disadvantage is that it asserts for packet sizes > 1300 bytes or so.
1162 // The advantage of StreamMessage is that it does allow for large packets.
1163 // The disadvantage is that the packets can take a long time to arrive, since it guarantees that they
1164 // arrive in the correct order, a feature that the script debugger does not always need.
1165 // This means that composite object info cannot be sent every frame, because the packets start piling
1166 // up at the PS2 side and it eventually asserts.
1167 
1168 // SplitAndEnqueueMessage tries to get the advantages of both Enqueue and Stream.
1169 // It allows for large packets because it splits them up into smaller packets and sends each of them
1170 // using EnqueueMessage.
1171 void CScriptDebugger::SplitAndEnqueueMessage(Net::MsgDesc *p_message)
1172 {
1173  if (!mp_server)
1174  {
1175  return;
1176  }
1177 
1178  printf("Splitting message of size %d\n",p_message->m_Length);
1179 
1180  int num_packets=p_message->m_Length / SPLIT_MESSAGE_MAX_SIZE;
1181  int total_num_packets=num_packets;
1182  int remainder=p_message->m_Length % SPLIT_MESSAGE_MAX_SIZE;
1183  if (remainder)
1184  {
1185  ++total_num_packets;
1186  }
1187 
1188  char *p_source_data=(char*)p_message->m_Data;
1189  for (int i=0; i<num_packets; ++i)
1190  {
1191  send_sub_packet(p_message,SPLIT_MESSAGE_MAX_SIZE,p_source_data,i,total_num_packets);
1192  p_source_data+=SPLIT_MESSAGE_MAX_SIZE;
1193  }
1194
