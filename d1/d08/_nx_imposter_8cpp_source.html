84 void CImposterManager::AddGeomToImposter( uint32 group_checksum, Nx::CGeom *p_geom )
85 {
86  // See if this imposter group exists already, create it if not.
87  CImposterGroup *p_group = mp_group_table->GetItem( group_checksum );
88  if( p_group == NULL )
89  {
90  p_group = plat_create_imposter_group();
91  mp_group_table->PutItem( group_checksum, p_group );
92 
93  // Set the bounding box to be that of the 'founding' CGeom.
94  p_group->SetCompositeBoundingBox( p_geom->GetBoundingBox());
95  }
96  else
97  {
98  // Add the bounding box details of this CGeom.
99  p_group->AddBoundingBox( p_geom->GetBoundingBox());
100  }
101 
102  // Add the specified CGeom to the CImposterGroup's list.
103  p_group->AddGeom( p_geom );
104 }

126 /* */
127 /* */
128 /******************************************************************/
129 void CImposterManager::ProcessImposters( void )
130 {
131  int imposters_drawn = 0;
132 
133  mp_group_table->IterateStart();
134  CImposterGroup *p_imposter_group = mp_group_table->IterateNext();
135  while( p_imposter_group )
136  {
137  // Once per frame process function, for platform-specific tasks.
138  p_imposter_group->Process();
139 
140  // Determine whether this imposter is no longer visible, or is outside the switch distance.
141  float d = p_imposter_group->CheckDistance();
142 
143  if( d <= p_imposter_group->GetSwitchDistance())
144  {
145  // This imposter is no longer outside the switch distance, or is not visible. Remove the imposter polygon if it exists.
146  p_imposter_group->RemoveImposterPolygon();
147  }
148  else
149  {
150  // This imposter is outside the switch distance. Create the imposter polygon if it doesn't exist...
151  if( p_imposter_group->ImposterPolygonExists() == false )
152  {
153  if( imposters_drawn < m_max_imposters_to_redraw_per_frame )
154  {
155  p_imposter_group->CreateImposterPolygon();
156  ++imposters_drawn;
157  }
158  }
159  else
160  {
161  // ...Otherwise determine whether the imposter polygon requires updating.
162  if( imposters_drawn < m_max_imposters_to_redraw_per_frame )
163  {
164  bool drawn = p_imposter_group->UpdateImposterPolygon();
165  if( drawn )
166  ++imposters_drawn;
167  }
168  }
169  }
170  p_imposter_group = mp_group_table->IterateNext();
171  }
172 }
