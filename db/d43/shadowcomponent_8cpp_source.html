103 void CShadowComponent::InitFromStructure( Script::CStruct* pParams )
104 {
105  m_shadowType = CRCD(0x3e84c2fd,"simple");
106  pParams->GetChecksum( CRCD(0x9ac24b18,"ShadowType"), &m_shadowType, Script::NO_ASSERT );
107 
108  if ( mp_shadow )
109  {
110  // get rid of existing shadow, in case it was the wrong type...
111  SwitchOffShadow();
112  }
113 
114  switch ( m_shadowType )
115  {
116  case 0x3e84c2fd: // simple
117  {
118  if ( !mp_shadow )
119  {
120  SwitchOnShadow( Gfx::vSIMPLE_SHADOW );
121  }
122 
123  float scale = 1.0f;
124  pParams->GetFloat( CRCD(0x6f8cd62f,"ShadowScale"), &scale, Script::NO_ASSERT );
125 
126  const char *p_shadow_model_name = "Ped_Shadow";
127  pParams->GetString( CRCD(0x545f8172,"ShadowModel"), &p_shadow_model_name, Script::NO_ASSERT );
128 
129  Dbg_MsgAssert( mp_shadow, ("NULL mp_shadow") );
130  Gfx::CSimpleShadow* pSimpleShadow = (Gfx::CSimpleShadow*)mp_shadow;
131  pSimpleShadow->SetScale( scale );
132  pSimpleShadow->SetModel( p_shadow_model_name );
133 
134  // GJ: need to immediately change the shadow's position if Obj_ShadowOn gets called
135  // this is because sometimes the shadow component will be suspended, and so
136  // update_shadow() won't get called (fixes shadow appearing at the origin in SC2)
137  pSimpleShadow->UpdatePosition( m_shadowPos, GetObject()->m_matrix, m_shadowNormal );
138  }
139  break;
140 
141  case 0x76a54cd1: // detailed
142  {
143  if ( !mp_shadow )
144  {
145  SwitchOnShadow( Gfx::vDETAILED_SHADOW );
146  }
147  }
148  break;
149 
150  case 0x806fff30: // none
151  {
152  if ( mp_shadow )
153  {
154  SwitchOffShadow();
155  }
156  }
157  break;
158 
159  default:
160  Dbg_MsgAssert( 0, ( "Unrecognized shadow type %s", Script::FindChecksumName(m_shadowType) ) );
161  }
162 }

169 void CShadowComponent::update_shadow()
170 {
171  if ( mp_shadow )
172  {
173 // Dbg_MsgAssert(mp_shadow->GetShadowType()!=Gfx::vDETAILED_SHADOW,("Tried to update a detailed shadow in MovingObj_Update, will need to change code to send a higher-up position"));
174 
175 
176  Nx::CModel* pModel = NULL;
177  if ( mp_model_component )
178  {
179  pModel = mp_model_component->GetModel();
180  }
181 
182  // if we have a model and the model is not active,then don't need to update the shadwo
183  if (pModel)
184  {
185  if (pModel->GetActive())
186  {
187  switch( mp_shadow->GetShadowType() )
188  {
189  case Gfx::vDETAILED_SHADOW:
190  {
191  // Mth::Vector shadow_target_pos = pos + ( matrix.GetUp() * 36.0f );
192  Mth::Vector shadow_target_pos = GetObject()->m_pos + ( GetObject()->m_matrix.GetUp() * 36.0f );
193 
194  #ifdef __PLAT_XBOX__
195  // K: Moved this in here cos it was giving an unused-variable compile error on PS2
196  Mth::Vector ground_dir( 0.2f, -0.8f, 0.3f );
197 
198  // If lights are active, set the ground direction to be that of the primary light.
199  if( pModel )
200  {
201  Nx::CModelLights* p_lights = pModel->GetModelLights();
202  if( p_lights )
203  {
204  ground_dir = p_lights->GetLightDirection( 0 ) * -1.0f;
205  }
206  }
207  // mp_shadow->UpdateDirection( ground_dir );
208  #endif
209  mp_shadow->UpdatePosition( shadow_target_pos );
210  }
211  break;
212 
213  case Gfx::vSIMPLE_SHADOW:
214  {
215  if ( pModel )
216  {
217  ((Gfx::CSimpleShadow*)mp_shadow)->SetScale( pModel->GetScale().GetX() );
218  }
219 
220  /*
221  TODO: Commented this section out, because it
222  references m_jump_start_pos, which isn't accessible
223  from the shadow code yet...
224 
225  if ( GetMotionComponentFromObject( GetObject() )->m_movingobj_status & MOVINGOBJ_STATUS_JUMPING )
226  {
227  // If jumping, use the jump's start-y so that the shadow stays on the ground.
228  Mth::Vector p=GetObject()->m_pos;
229  p[Y]=m_jump_start_pos[Y];
230  mp_shadow->UpdatePosition(p,GetObject()->m_matrix,m_shadowNormal);
231  }
232  else
233  */
234  {
235  mp_shadow->UpdatePosition( m_shadowPos, GetObject()->m_matrix, m_shadowNormal );
236  }
237  }
238  break;
239 
240  default:
241  Dbg_MsgAssert(0,("Bad shadow type: %d",mp_shadow->GetShadowType()));
242  break;
243  }
244 
245  mp_shadow->UnHide();
246  }
247  else
248  {
249  // model is not active, so we probably don't want the shadow's model to be active either
250  mp_shadow->Hide();
251  }
252  }
253  }

309 CBaseComponent::EMemberFunctionResult CShadowComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
310 {
311  switch ( Checksum )
312  {
313  // @script | Obj_ShadowOff | turn off shadow
314  case ( 0xaaf6e513 ): // Obj_ShadowOff
315  SwitchOffShadow();
316  break;
317 
318  // @script | Obj_ShadowOn | turn shadow on
319  // @parmopt float | Offset | 0.2 | The y offset of the shadow, > 0 = up
320  // @parmopt float | ShadowScale | 1.0 | The scale of the shadow.
321  // @parmopt string | ShadowModel | "Ped_Shadow" | The mdl file to use
322  // @parmopt name | shadow_type | Either "detailed" or "simple"
323  case ( 0xf272c43a ): // Obj_ShadowOn
324  {
325  // add a default shadow_type
326  uint32 shadowType;
327  if ( !pParams->GetChecksum( "shadowType", &shadowType, Script::NO_ASSERT ) )
328  {
329  pParams->AddChecksum( "shadowType", CRCD(0x3e84c2fd,"simple") );
330  }
331 
332  this->InitFromStructure( pParams );
333 
334  if ( mp_shadow )
335  {
336  if( mp_shadow->GetShadowType() == Gfx::vSIMPLE_SHADOW )
337  {
338  float offset=0.2f;
339  pParams->GetFloat(CRCD(0xa6f5352f,"Offset"),&offset);
340  ((Gfx::CSimpleShadow*)mp_shadow)->SetOffset(offset);
341  }
342  }
343  }
344  break;
345 
346  default:
347  return CBaseComponent::MF_NOT_EXECUTED;
348  }
349 
350  // the "default" case of the switch statement handles
351  // unrecognized functions; if we make it down here,
352  // that means that the component already handled it
353  // somehow
354  return CBaseComponent::MF_TRUE;
355 }

362 void CShadowComponent::SwitchOnShadow( Gfx::EShadowType mode )
363 {
364 
365 
366  if ( mp_shadow )
367  {
368  // it's already got a shadow
369  return;
370  }
371 
372 // Note we can't use mp_model_component here, as SwitchOnShaodow is called
373 // from InitFromStructure, which is obviously called before Finalize();
374  CModelComponent* pModelComponent = GetModelComponentFromObject( GetObject() );
375  Nx::CModel* pModel = NULL;
376  if ( pModelComponent )
377  {
378  pModel = pModelComponent->GetModel();
379  }
380 
381 
382  switch (mode)
383  {
384  case Gfx::vDETAILED_SHADOW:
385  {
386  Dbg_MsgAssert( mp_shadow == NULL, ("mp_shadow not NULL?") );
387  Dbg_MsgAssert( pModel, ("adding detailed shadow to something with no model?") );
388  mp_shadow = new Gfx::CDetailedShadow( pModel );
389  }
390  break;
391 
392  case Gfx::vSIMPLE_SHADOW:
393  {
394  Dbg_MsgAssert( mp_shadow == NULL, ("mp_shadow not NULL?") );
395  Dbg_MsgAssert( pModel, ("adding simple shadow to something with no model?") );
396 
397  Gfx::CSimpleShadow* pSimpleShadow = new Gfx::CSimpleShadow;
398  pSimpleShadow->SetScale( pModel->GetScale().GetX() );
399  pSimpleShadow->SetModel( "Ped_Shadow" );
400 
401  mp_shadow = pSimpleShadow;
402  mp_shadow->UpdatePosition( m_shadowPos, GetObject()->m_matrix, m_shadowNormal );
403  }
404  break;
405 
406  default:
407  Dbg_Message( "Unrecognized shadow mode %d", mode );
408  break;
409  }
410 }

504 void CShadowComponent::SwitchOnSkaterShadow()
505 {
506  // NOTE: skater specific and should not be here; migrate to somewhere like Mdl::Skate
507 
508  // only call on shadows attached to skaters
509  Dbg_MsgAssert(GetObject()->GetType() == SKATE_TYPE_SKATER, ("CShadowComponent::SwitchOnSkaterShadow called on shadow not attached to skater object"));
510 
511  CSkater* pSkater = static_cast< CSkater* >(GetObject());
512 
513  Gfx::EShadowType mode = Gfx::vDETAILED_SHADOW;
514 
515  // put it on the bottom up heap, because we don't want to fragment the skater geom heap...
516  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().SkaterHeap(pSkater->GetHeapIndex()));
517 
518  // In Splitscreen games, shadow type for local player is optional
519  if (CFuncs::ScriptInSplitScreenGame(NULL, NULL))
520  {
521  mode = Mdl::Skate::Instance()->GetShadowMode();
522  }
523 
524  if (!pSkater->IsLocalClient())
525  {
526  Dbg_Printf( "************************ SWITCHING ON SKATER SHADOW FOR CLIENT\n" );
527  mode = Gfx::vSIMPLE_SHADOW;
528  }
529 
530  Script::CStruct* pTempParams = new Script::CStruct;
531  pTempParams->AddFloat(CRCD(0x6f8cd62f, "ShadowScale"), 1.0f);
532  pTempParams->AddString(CRCD(0x545f8172, "ShadowModel"), "Ped_Shadow");
533  InitFromStructure(pTempParams);
534  delete pTempParams;
535 
536  SwitchOffShadow();
537  SwitchOnShadow(mode);
538 
539  Mem::Manager::sHandle().PopContext();
540 }
