513 /******************************************************************/
514 /* */
515 /* */
516 /******************************************************************/
517 void PCMAudio_Init( void )
518 {
519  // Zero the music and filestream info arrays.
520  ZeroMemory( &gMusicInfo, sizeof( sFileStreamInfo ));
521  ZeroMemory( &gStreamInfo[0], sizeof( sFileStreamInfo ) * NUM_STREAMS );
522 
523  // Enumerate user soundtracks.
524  GetNumSoundtracks();
525 
526  // Figure out the language, and open appropriate file.
527  Config::ELanguage lang = Config::GetLanguage();
528 
529  // Assume English.
530 # ifdef STREAMS_ARE_PCM
531  ghWADFile = CreateFile( "d:\\data\\streams\\pcm\\pcm.wad",
532 # else
533  ghWADFile = CreateFile( "d:\\data\\streams\\wma\\wma.wad",
534 # endif
535  GENERIC_READ,
536  FILE_SHARE_READ, // Share mode.
537  NULL, // Ignored (security attributes).
538  OPEN_EXISTING, // File has to exist already.
539  FILE_FLAG_OVERLAPPED | FILE_FLAG_NO_BUFFERING, // Xbox has no asynchronous i/o buffering.
540  NULL ); // Ignored (template file).
541 # ifdef STREAMS_ARE_PCM
542  ghMusicWADFile = CreateFile( "d:\\data\\streams\\pcm\\music_pcm.wad",
543 # else
544  ghMusicWADFile = CreateFile( "d:\\data\\streams\\wma\\music_wma.wad",
545 # endif
546  GENERIC_READ,
547  FILE_SHARE_READ, // Share mode.
548  NULL, // Ignored (security attributes).
549  OPEN_EXISTING, // File has to exist already.
550  FILE_FLAG_OVERLAPPED | FILE_FLAG_NO_BUFFERING, // Xbox has no asynchronous i/o buffering.
551  NULL ); // Ignored (template file).
552 
553  // Now read in the data files used for indexing into the WAD files.
554 # ifdef STREAMS_ARE_PCM
555  HANDLE wad_data = CreateFile( "d:\\data\\streams\\pcm\\pcm.dat",
556 # else
557  HANDLE wad_data = CreateFile( "d:\\data\\streams\\wma\\wma.dat",
558 # endif
559  GENERIC_READ,
560  FILE_SHARE_READ, // Share mode.
561  NULL, // Ignored (security attributes).
562  OPEN_EXISTING, // File has to exist already.
563  FILE_FLAG_SEQUENTIAL_SCAN,
564  NULL ); // Ignored (template file).
565 
566  if( wad_data != INVALID_HANDLE_VALUE )
567  {
568  uint32 bytes_read;
569  ReadFile( wad_data, &numWADFileEntries, sizeof( uint32 ), &bytes_read, NULL );
570  pWADData = new uint32[numWADFileEntries * 3];
571  ReadFile( wad_data, pWADData, sizeof( uint32 ) * numWADFileEntries * 3, &bytes_read, NULL );
572  CloseHandle( wad_data );
573  }
574 
575  // Sort the wad file entries into increasing checksum order, so that we can use a binary search algorithm to
576  // find the checksum quickly.
577  for( uint32 i = 0; i < numWADFileEntries; ++i )
578  {
579  for( uint32 j = i + 1; j < numWADFileEntries; ++j )
580  {
581  if( pWADData[i * 3] > pWADData[j * 3] )
582  {
583  uint32 temp[3];
584  temp[0] = pWADData[i * 3];
585  temp[1] = pWADData[i * 3 + 1];
586  temp[2] = pWADData[i * 3 + 2];
587  pWADData[i * 3] = pWADData[j * 3];
588  pWADData[i * 3 + 1] = pWADData[j * 3 + 1];
589  pWADData[i * 3 + 2] = pWADData[j * 3 + 2];
590  pWADData[j * 3] = temp[0];
591  pWADData[j * 3 + 1] = temp[1];
592  pWADData[j * 3 + 2] = temp[2];
593  }
594  }
595  }
596 
597 # ifdef STREAMS_ARE_PCM
598  wad_data = CreateFile( "d:\\data\\streams\\pcm\\music_pcm.dat",
599 # else
600  wad_data = CreateFile( "d:\\data\\streams\\wma\\music_wma.dat",
601 # endif
602  GENERIC_READ,
603  FILE_SHARE_READ, // Share mode.
604  NULL, // Ignored (security attributes).
605  OPEN_EXISTING, // File has to exist already.
606  FILE_FLAG_SEQUENTIAL_SCAN,
607  NULL ); // Ignored (template file).
608 
609  if( wad_data != INVALID_HANDLE_VALUE )
610  {
611  uint32 bytes_read;
612  ReadFile( wad_data, &numMusicWADFileEntries, sizeof( uint32 ), &bytes_read, NULL );
613  pMusicWADData = new uint32[numMusicWADFileEntries * 3];
614  ReadFile( wad_data, pMusicWADData, sizeof( uint32 ) * numMusicWADFileEntries * 3, &bytes_read, NULL );
615  CloseHandle( wad_data );
616  }
617 }

621 /******************************************************************/
622 /* */
623 /* Call every frame to make sure music and stream buffers are */
624 /* loaded and current status is checked each frame... */
625 /* */
626 /******************************************************************/
627 int PCMAudio_Update( void )
628 {
629  if( gMusicInfo.HasFileStream())
630  {
631  HRESULT hr = gMusicInfo.Process();
632 
633  if( gMusicInfo.IsCompleted())
634  {
635  gMusicInfo.DestroyFileStream();
636  }
637  else if( gMusicInfo.IsAwaitingDeletion() && gMusicInfo.IsSafeToDelete())
638  {
639  gMusicInfo.DestroyFileStream();
640  }
641  }
642 
643  for( int i = 0; i < NUM_STREAMS; ++i )
644  {
645  if( gStreamInfo[i].HasFileStream())
646  {
647  HRESULT hr = gStreamInfo[i].Process();
648 
649  if( gStreamInfo[i].IsCompleted())
650  {
651  gStreamInfo[i].DestroyFileStream();
652  }
653  else if( gStreamInfo[i].IsAwaitingDeletion() && gStreamInfo[i].IsSafeToDelete())
654  {
655  gStreamInfo[i].DestroyFileStream();
656  }
657  }
658  }
659 
660  // A non-zero return value singnals an error condition.
661  return 0;
662 }

991 /* */
992 /* Return (any) position if t in sorted x[0..n-1] or -1 if t is */
993 /* not present. */
994 /* */
995 /******************************************************************/
996 static int binarySearch( uint32 checksum )
997 {
998  int l = 0;
999  int u = numWADFileEntries - 1;
1000  while( l <= u )
1001  {
1002  int m = ( l + u ) / 2;
1003  if( pWADData[m * 3] < checksum )
1004  l = m + 1;
1005  else if ( pWADData[m * 3] == checksum )

1009  }
1010  return -1;
