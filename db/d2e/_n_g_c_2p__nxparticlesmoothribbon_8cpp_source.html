172 void CNgcParticleSmoothRibbon::plat_render( void )
173 {
174  NxNgc::sMaterialHeader mat;
175  NxNgc::sMaterialPassHeader pass;
176 
177  // Header.
178  mat.m_checksum = 0xa9db601e; // particle
179  mat.m_passes = 1;
180  mat.m_alpha_cutoff = 1;
181  mat.m_flags = (1<<1); // 2 sided.
182 // mat.m_shininess = 0.0f;
183 
184  // Pass 0.
185  pass.m_texture.p_data = mp_engine_texture;
186  pass.m_flags = ( mp_engine_texture ? (1<<0) : 0 ) | (1<<5) | (1<<6); // textured, clamped.
187  pass.m_filter = 0;
188  pass.m_blend_mode = (unsigned char)m_blend;
189  pass.m_alpha_fix = (unsigned char)m_fix;
190  pass.m_k = 0;
191  pass.m_color.r = 128;
192  pass.m_color.g = 128;
193  pass.m_color.b = 128;
194  pass.m_color.a = 255;
195 
196  NxNgc::multi_mesh( &mat, &pass, true, true );
197 
198  GX::SetTexCoordGen( GX_TEXCOORD0, GX_TG_MTX2x4, GX_TG_TEX0, GX_FALSE, GX_PTIDENTITY );
199  GX::SetCurrMtxPosTex03( GX_PNMTX0, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY );
200 
201  // Draw the particles.
202  int lp;
203  CParticleEntry *p_particle;
204  float *p_v0;
205  float *p_v1;
206 
207  GXColor color[4];
208  GXColor *p_col0;
209  GXColor *p_col1;
210 
211  NsMatrix* p_matrix = &NxNgc::EngineGlobals.camera;
212  Mth::Vector at( p_matrix->getAtX(), p_matrix->getAtY(), p_matrix->getAtZ(), 0.0f );
213 
214  GX::SetVtxDesc( 2, GX_VA_POS, GX_DIRECT, GX_VA_CLR0, GX_DIRECT );
215 
216  for ( lp = 0, p_particle = mp_particle_array, p_v0 = mp_vertices[0], p_v1 = mp_vertices[(m_num_vertex_buffers-1)]; lp < m_num_particles; lp++, p_particle++, p_v0 += 3, p_v1 += 3 )
217  {
218  float terp = p_particle->m_time / p_particle->m_life;
219 
220  Mth::Vector pos0( p_v0[0] + m_pos[X], p_v0[1] + m_pos[Y], p_v0[2] + m_pos[Z] );
221  Mth::Vector pos1( p_v1[0] + m_pos[X], p_v1[1] + m_pos[Y], p_v1[2] + m_pos[Z] );
222 
223  Mth::Vector part_vec = pos1 - pos0;
224  Mth::Vector perp_vec = Mth::CrossProduct( part_vec, at );
225  perp_vec.Normalize();
226 
227  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
228 
229  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
230  Mth::Vector tmp[4];
231 
232  tmp[0] = pos0 + ( perp_vec * w );
233  tmp[1] = pos0 - ( perp_vec * w );
234  tmp[2] = pos1 - ( perp_vec * w );
235  tmp[3] = pos1 + ( perp_vec * w );
236 
237  if ( m_mid_time >= 0.0f )
238  {
239  if ( terp < m_mid_time )
240  {
241  p_col0 = m_start_color;
242  p_col1 = m_mid_color;
243  // Adjust interpolation for this half of the color blend.
244  terp = terp / m_mid_time;
245  }
246  else
247  {
248  p_col0 = m_mid_color;
249  p_col1 = m_end_color;
250  // Adjust interpolation for this half of the color blend.
251  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
252  }
253  }
254  else
255  {
256  // No mid color specified.
257  p_col0 = m_start_color;
258  p_col1 = m_end_color;
259  }
260 
261  for ( int c = 0; c < 4; c++ )
262  {
263  GXColor start = *p_col0++;
264  GXColor end = *p_col1++;
265 
266  color[c].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
267  color[c].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
268  color[c].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
269  color[c].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
270  }
271 
272  GX::Begin( GX_TRIANGLESTRIP, GX_VTXFMT0, 6 );
273 
274  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
275  GX::Color1u32( *((uint32*)&color[1]) );
276 
277  GX::Position3f32( tmp[3][X], tmp[3][Y], tmp[3][Z] );
278  GX::Color1u32( *((uint32*)&color[2]) );
279 
280  GX::Position3f32( pos0[X], pos0[Y], pos0[Z] );
281  GX::Color1u32( *((uint32*)&color[0]) );
282 
283  GX::Position3f32( pos1[X], pos1[Y], pos1[Z] );
284  GX::Color1u32( *((uint32*)&color[3]) );
285 
286  GX::Position3f32( tmp[1][X], tmp[1][Y], tmp[1][Z] );
287  GX::Color1u32( *((uint32*)&color[1]) );
288 
289  GX::Position3f32( tmp[2][X], tmp[2][Y], tmp[2][Z] );
290  GX::Color1u32( *((uint32*)&color[2]) );
291 
292  GX::End();
293  }
294 }
