56 void SetupBMPHeader(uint8 *bmp, int width, int height)
57 {
58  // Initialize the BMP header structure.
59  // Bitmapfileheader
60  *bmp = 'B'; // Type
61  *( bmp + 1 ) = 'M';
62  SET_UINT32( bmp + 2, BITMAPSIZE ); // Size
63  *( uint16 * )( bmp + 6 ) = 0; // Reserved
64  SET_UINT32( bmp + 10, HEADSIZE ); // Offset
65 
66  // Bitmapinfoheader
67  SET_UINT32( bmp + 14, 40 ); // Bitmapinfoheader size
68  SET_UINT32( bmp + 18, width ); // Width
69  SET_UINT32( bmp + 22, height ); // Height
70  *( uint16 * )( bmp + 26 ) = 1; // Planes
71  *( uint16 * )( bmp + 28 ) = 24; // Bitcount
72  SET_UINT32( bmp + 30, 0 ); // Compression
73  SET_UINT32( bmp + 34, IMAGESIZE ); // Image size in bytes
74  SET_UINT32( bmp + 38, 4740 ); // X Pels per metre
75  SET_UINT32( bmp + 42, 4740 ); // Y Pels per metre
76  SET_UINT32( bmp + 46, 0 ); // ClrUsed
77  SET_UINT32( bmp + 50, 0 ); // ClrImportant

339 //--------------------------------------------------------------------------------------------
340 // Function: StoreTextureVIF1()
341 // Description: Stores texture data from GS to main memory or scratchpad memory. See notes on
342 // data flow
343 // Paramaters: u_long128* base_addr: base address of stored texture data in MAIN/SPR memory
344 // short start_addr: start address in GS memory to transfer from
345 // short pixel_mode: pixel mode of stored data
346 // short x: x location in GS memory
347 // short y: y location in GS memory
348 // short w: width of stored image
349 // short h: height of stored image
350 // short frame_width: width of frame buffer
351 // Returns: N/A
352 // Notes: - data flow of store image
353 // - wait for all 3 PATHS to GIF to be complete (FLUSHA vif code)
354 // - disable PATH3 transfer using MSKPATH3 vifcode (mask)
355 // - set BITBLTBUF register (source parameters)
356 // - set TRXPOS register (x,y postion)
357 // - set TRXREG register (width and height)
358 // - set FINISH register (set event)
359 // - set TRXDIR register (LOCAL->HOST)
360 // - get previous Interrupt Mask Control (IMR)
361 // - enable the FINISH event
362 // - send GIF packet to GS
363 // - wait for DMA to be complete
364 // - wait for FINISH event to be generated (all drawing is complete)
365 // - change direction of the GS bus
366 // - change direction of VIF1-FIFO (VIF1_STAT.FDR)
367 // - DMA GS data to main/spr memory
368 // - wait for DMA to be complete
369 // - restore direction of the GS bus
370 // - restore direction of VIF1-FIFO
371 // - restore previous Interrupt Mask Control (IMR)
372 // - enable the FINISH event
373 // - enable PATH3 transfer using MSKPATH3 vifcode
374 //--------------------------------------------------------------------------------------------
375 
376 void StoreTextureVIF1( u_long128 * base_addr, short start_addr, short pixel_mode, short x,
377  short y, short w, short h, short frame_width )
378 {
379 
380  int texture_qwc;
381  sceVif1Packet vif1_pkt;
382  u_long128 settup_base[10];
383  int buff_width;
384  static u_int enable_path3[4] __attribute__ ( ( aligned( 16 ) ) ) = {
385  0x06000000,
386  0x00000000,
387  0x00000000,
388  0x00000000,
389  };
390 
391  // get quad word count for image
392  if ( pixel_mode == SCE_GS_PSMCT32 )
393  texture_qwc = ( w * h * 32 ) >> 7;
394  else if ( pixel_mode == SCE_GS_PSMCT24 )
395  texture_qwc = ( w * h * 24 ) >> 7;
396  else if ( pixel_mode == SCE_GS_PSMCT16 )
397  texture_qwc = ( w * h * 16 ) >> 7;
398  else if ( pixel_mode == SCE_GS_PSMT8 )
399  texture_qwc = ( w * h * 8 ) >> 7;
400  else
401  texture_qwc = ( w * h * 4 ) >> 7;
402 
403  if ( texture_qwc > GIFTAG_MAX_NLOOP )
404  {
405  printf( "ERROR: Texture QWC is greater then GIFTAG_NLOOP! line:(%d), file:(%s)\n", __LINE__,
406  __FILE__ );
407  exit( 0 );
408  }
409 
410  buff_width = frame_width >> 6;
411 
412  if ( buff_width <= 0 )
413  buff_width = 1;
414 
415  // set base address of GIF packet
416  sceVif1PkInit( &vif1_pkt, &settup_base[0] );
417  sceVif1PkReset( &vif1_pkt );
418 
419  // will start transfer with VIF code and GS data will follow
420  sceVif1PkAddCode( &vif1_pkt, SCE_VIF1_SET_NOP( 0 ) );
421  // disable PATH 3 transfer
422  sceVif1PkAddCode( &vif1_pkt, SCE_VIF1_SET_MSKPATH3( 0x8000, 0 ) );
423  // wait for all 3 PATHS to GS to be complete
424  sceVif1PkAddCode( &vif1_pkt, SCE_VIF1_SET_FLUSHA( 0 ) );
425  // transfer 6 QW's to GS
426  sceVif1PkAddCode( &vif1_pkt, SCE_VIF1_SET_DIRECT( 6, 0 ) );
427 
428  // GIF tag for texture settings
429  sceVif1PkAddGsData( &vif1_pkt, SCE_GIF_SET_TAG( 5, GIFTAG_EOP, NULL, NULL, SCE_GIF_PACKED, 1 ) );
430  sceVif1PkAddGsData( &vif1_pkt, 0xEL );
431 
432  // set transmission between buffers
433  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_SET_BITBLTBUF( start_addr, buff_width, pixel_mode, // SRC
434  NULL, NULL, NULL ) ); // DEST
435  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_BITBLTBUF );
436 
437  // set transmission area between buffers ( source x,y dest x,y and direction )
438  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_SET_TRXPOS( x, y, 0, 0, 0 ) );
439  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_TRXPOS );
440 
441  // set size of transmission area
442  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_SET_TRXREG( w, h ) );
443  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_TRXREG );
444 
445  // set FINISH event occurrence request
446  sceVif1PkAddGsData( &vif1_pkt, ( u_long ) ( 0x0 ) );
447  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_FINISH );
448 
449  // set transmission direction ( LOCAL -> HOST Transmission )
450  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_SET_TRXDIR( 1 ) );
451  sceVif1PkAddGsData( &vif1_pkt, SCE_GS_TRXDIR );
452 
453  // get packet size in quad words
454  sceVif1PkTerminate( &vif1_pkt );
455 
456 
457 
458 
459  // get current IMR status
460 // u_long prev_imr = 0;
461 // prev_imr = sceGsPutIMR( sceGsGetIMR( ) | 0x0200 ); // <<<<< Mick, removed.
462 
463 
464  // set the FINISH event
465  DPUT_GS_CSR( GS_CSR_FINISH_M );
466 
467 
468 
469  // DMA from memory and start DMA transfer
470  FlushCache( WRITEBACK_DCACHE );
471 
472  DPUT_D1_QWC( 0x7 );
473 
474  DPUT_D1_MADR( ( u_int ) vif1_pkt.pBase & DMA_MEM );
475 
476 
477  DPUT_D1_CHCR( 1 | ( 1 << 8 ) );
478 
479  asm __volatile__( " sync.l " );
480 
481 
482  // check if DMA is complete (STR=0)
483  while ( DGET_D1_CHCR( ) & 0x0100 );
484 // printf( " 5 GS registers set\n" );
485 
486  // check if FINISH event occured
487  while ( ( DGET_GS_CSR( ) & GS_CSR_FINISH_M ) == 0 );
488 // printf( " Finish event complete\n" );
489 
490 
491 
492  // change VIF1-FIFO transfer direction (VIF1 -> MAIN MEM or SPR)
493  *VIF1_STAT = 0x00800000;
494 
495 
496  // change GS bus direction (LOCAL->HOST)
497  DPUT_GS_BUSDIR( ( u_long ) 0x00000001 );
498 
499 
500 // printf( " Changed VIF1 and GS direction complete\n" );
501 
502  // DMA to memory and start DMA transfer
503  FlushCache( WRITEBACK_DCACHE );
504 
505  DPUT_D1_QWC( texture_qwc );
506 
507  DPUT_D1_MADR( ( u_int ) base_addr & DMA_MEM );
508 
509 
510  DPUT_D1_CHCR( 0 | ( 1 << 8 ) );
511 
512  asm __volatile__( " sync.l " );
513 
514  // check if DMA is complete (STR=0)
515  while ( DGET_D1_CHCR( ) & 0x0100 );
516 // printf( " Transferred:(%d) QW from GS -> MEM complete\n", texture_qwc );
517 
518 
519 
520  // change VIF1-FIFO transfer direction (MAIN MEM or SPR -> VIF1)
521  *VIF1_STAT = 0;
522 
523  // change GS bus direction (HOST->LOCAL)
524  DPUT_GS_BUSDIR( ( u_long ) 0 );
525 
526 
527  // restore previous IMR status
528 // sceGsPutIMR( prev_imr ); // <<<<< Mick, removed.
529 
530 
531 
532  // set the FINISH event
533  DPUT_GS_CSR( GS_CSR_FINISH_M );
534 // printf( " Restore VIF1 and GS direction complete\n" );
535 
536  // MSKPATH3 is now enabled to allow transfer via PATH3
537  DPUT_VIF1_FIFO( *( u_long128 * ) enable_path3 );
538 
539 // printf( " Restore PATH3 direction complete\n" );
540 
541 
542 } // end StoreTextureVIF1
