165 bool Skate::should_send_update( GameNet::PlayerInfo* send_player, Obj::CObject* object )
166 {
167  Obj::SPosEvent* latest_state;
168  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
169  GameNet::PlayerInfo* skater_player;
170  Obj::CSkater* skater_obj;
171  int id, index;
172 
173  // Dont send a client updates on his own skater. He is up to date.
174  skater_player = gamenet_man->GetPlayerByObjectID( object->GetID() );
175  if( ( skater_player == NULL ) ||
176  ( skater_player == send_player ))
177  {
178  return false;
179  }
180 
181  skater_obj = skater_player->m_Skater;
182  // Check if we have something to report
183  if( skater_obj->GetStateHistory()->GetNumPosUpdates() == 0 )
184  {
185  //Dbg_Printf( "Not enough pos updates\n" );
186  return false;
187  }
188 
189  id = skater_obj->GetID();
190  index = ( skater_obj->GetStateHistory()->GetNumPosUpdates() - 1 ) % Obj::CSkaterStateHistoryComponent::vNUM_POS_HISTORY_ELEMENTS;
191  latest_state = &skater_obj->GetStateHistory()->GetPosHistory()[index];
192 
193  // Only send an update if there's something new to tell
194  if( latest_state->LoTime == send_player->m_LastSentProps.m_LastSkaterUpdateTime[id] )
195  {
196  //Dbg_Printf( "Nothing new to tell\n" );
197  return false;
198  }
199 
200  return true;
201 }

208 static int s_get_update_flags( GameNet::PlayerInfo* player,
209  Obj::SPosEvent* latest_state,
210  int skater_id )
211 {
212  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
213  int i, update_flags;
214  short pos[3];
215  short rot[3];
216  Flags< int > skater_flags;
217  Flags< int > end_run_flags;
218  Mth::Vector eulers;
219  char state, doing_trick, terrain, walking, driving;
220  bool forced_send;
221  unsigned char id;
222  sint16 rail_node;
223 
224  update_flags = 0;
225 
226  // Force an object update every N frames
227  forced_send = !( gamenet_man->GetServer()->m_FrameCounter % vFORCE_UPDATE_INTERVAL );
228  id = skater_id;
229 
230  for( i = 0; i < 3; i++ )
231  {
232  eulers[i] = latest_state->Eulers[i];
233  }
234 
235  // Write out the object's position as three shorts (fixed-point)
236  for( i = 0; i < 3; i++ )
237  {
238  if( i == Y )
239  {
240  pos[i] = (short) ( latest_state->Position[i] * 4.0f );
241  //pos[i] = latest_state->Position[i];
242  }
243  else
244  {
245  pos[i] = (short) ( latest_state->Position[i] * 2.0f );
246  //pos[i] = latest_state->Position[i];
247  }
248  if(( pos[i] != player->m_LastSentProps.m_LastSkaterPosUpdate[id][i] ) || forced_send )
249  {
250  if( i == X )
251  {
252  update_flags |= GameNet::mUPDATE_FIELD_POS_X;
253  }
254  else if( i == Y )
255  {
256  update_flags |= GameNet::mUPDATE_FIELD_POS_Y;
257  }
258  else if( i == Z )
259  {
260  update_flags |= GameNet::mUPDATE_FIELD_POS_Z;
261  }
262  }
263  }
264 
265  // Write out the object's orientation as three short euler angles (fixed-point)
266  for( i = 0; i < 3; i++ )
267  {
268  rot[i] = (short) ( eulers[i] * 4096.0f );
269  if(( rot[i] != player->m_LastSentProps.m_LastSkaterRotUpdate[id][i] ) || forced_send )
270  {
271  if( i == X )
272  {
273  update_flags |= GameNet::mUPDATE_FIELD_ROT_X;
274  }
275  else if( i == Y )
276  {
277  update_flags |= GameNet::mUPDATE_FIELD_ROT_Y;
278  }
279  else if( i == Z )
280  {
281  update_flags |= GameNet::mUPDATE_FIELD_ROT_Z;
282  }
283  }
284  }
285 
286  state = (char) latest_state->State;
287  doing_trick = (char) latest_state->DoingTrick;
288  terrain = (char) latest_state->Terrain;
289  walking = (char) latest_state->Walking;
290  driving = (char) latest_state->Driving;
291 
292  if(( state != player->m_LastSentProps.m_LastSkaterStateUpdate[id] ) ||
293  ( doing_trick != player->m_LastSentProps.m_LastDoingTrickUpdate[id] ) ||
294  ( terrain != player->m_LastSentProps.m_LastSkaterTerrain[id] ) ||
295  ( walking != player->m_LastSentProps.m_LastSkaterWalking[id] ) ||
296  ( driving != player->m_LastSentProps.m_LastSkaterDriving[id] ) ||
297  ( forced_send == true ))
298  {
299  update_flags |= GameNet::mUPDATE_FIELD_STATE;
300  }
301 
302  skater_flags = latest_state->SkaterFlags;
303  end_run_flags = latest_state->EndRunFlags;
304 
305  if(( skater_flags != player->m_LastSentProps.m_LastSkaterFlagsUpdate[id] ) ||
306  ( end_run_flags != player->m_LastSentProps.m_LastEndRunFlagsUpdate[id] ) ||
307  ( forced_send == true ))
308  {
309  update_flags |= GameNet::mUPDATE_FIELD_FLAGS;
310  }
311 
312  rail_node = latest_state->RailNode;
313  if(( rail_node != player->m_LastSentProps.m_LastRailNodeUpdate[id] ) ||
314  ( forced_send == true ))
315  {
316  update_flags |= GameNet::mUPDATE_FIELD_RAIL_NODE;
317  }
318 
319  return update_flags;
320 }

327 void Skate::s_object_update( Obj::CObject* object, void* data )
328 {
329  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
330  GameNet::PlayerInfo* player, *skater_player;
331  ObjUpdateInfo* update_info;
332  Obj::CSkater* skater_obj;
333  Mth::Vector eulers;
334  short pos[3];
335  short rot[3];
336  unsigned char id;
337  char state, doing_trick, terrain, walking, driving;
338  Flags< int > skater_flags;
339  Flags< int > end_run_flags;
340  int update_flags;
341  Obj::SPosEvent* latest_state;
342  int i, index;
343  unsigned short time;
344 
345  Dbg_Assert( data );
346 
347  update_info = (ObjUpdateInfo *) data;
348 
349  skater_player = gamenet_man->GetPlayerByObjectID( object->GetID() );
350  skater_obj = skater_player->m_Skater;
351  id = (char) skater_obj->GetID();
352  player = update_info->m_Player;
353  index = ( skater_obj->GetStateHistory()->GetNumPosUpdates() - 1 ) % Obj::CSkaterStateHistoryComponent::vNUM_POS_HISTORY_ELEMENTS;
354  latest_state = &skater_obj->GetStateHistory()->GetPosHistory()[index];
355  player->m_LastSentProps.m_LastSkaterUpdateTime[id] = latest_state->GetTime();
356 
357  for( i = 0; i < 3; i++ )
358  {
359  eulers[i] = latest_state->Eulers[i];
360  }
361 
362  time = latest_state->LoTime;
363  update_info->m_BitStream.WriteValue( time, sizeof( uint16 ) * 8 );
364 
365  update_flags = s_get_update_flags( player, latest_state, id );
366  update_info->m_BitStream.WriteValue( update_flags, 9 );
367 
368  //Dbg_Printf( "*** Sending Pos[%.2f %.2f %.2f]\n", latest_state->Position[X], latest_state->Position[Y], latest_state->Position[Z] );
369  // Write out the object's position as three shorts (fixed-point)
370  for( i = 0; i < 3; i++ )
371  {
372  if( i == Y )
373  {
374  pos[i] = (short) ( latest_state->Position[i] * 4.0f );
375  //pos[i] = latest_state->Position[i];
376  }
377  else
378  {
379  pos[i] = (short) ( latest_state->Position[i] * 2.0f );
380  //pos[i] = latest_state->Position[i];
381  }
382 
383  if( i == X )
384  {
385  if( update_flags & GameNet::mUPDATE_FIELD_POS_X )
386  {
387  update_info->m_BitStream.WriteValue( pos[i], sizeof( short ) * 8 );
388  //update_info->m_BitStream.WriteFloatValue( pos[i] );
389  }
390  }
391  else if( i == Y )
392  {
393  if( update_flags & GameNet::mUPDATE_FIELD_POS_Y )
394  {
395  update_info->m_BitStream.WriteValue( pos[i], sizeof( short ) * 8 );
396  //update_info->m_BitStream.WriteFloatValue( pos[i] );
397  }
398  }
399  else if( i == Z )
400  {
401  if( update_flags & GameNet::mUPDATE_FIELD_POS_Z )
402  {
403  update_info->m_BitStream.WriteValue( pos[i], sizeof( short ) * 8 );
404  //update_info->m_BitStream.WriteFloatValue( pos[i] );
405  }
406  }
407 
408  player->m_LastSentProps.m_LastSkaterPosUpdate[id][i] = pos[i];
409  }
410 
411  // Write out the object's orientation as three short euler angles (fixed-point)
412  for( i = 0; i < 3; i++ )
413  {
414  rot[i] = (short) ( eulers[i] * 4096.0f );
415  if( i == X )
416  {
417  if( update_flags & GameNet::mUPDATE_FIELD_ROT_X )
418  {
419  update_info->m_BitStream.WriteValue( rot[i], sizeof( short ) * 8 );
420  }
421  }
422  else if( i == Y )
423  {
424  if( update_flags & GameNet::mUPDATE_FIELD_ROT_Y )
425  {
426  update_info->m_BitStream.WriteValue( rot[i], sizeof( short ) * 8 );
427  }
428  }
429  else if( i == Z )
430  {
431  if( update_flags & GameNet::mUPDATE_FIELD_ROT_Z )
432  {
433  update_info->m_BitStream.WriteValue( rot[i], sizeof( short ) * 8 );
434  }
435  }
436 
437  player->m_LastSentProps.m_LastSkaterRotUpdate[id][i] = rot[i];
438  }
439 
440  state = (char) latest_state->State;
441  doing_trick = (char) latest_state->DoingTrick;
442  terrain = (char) latest_state->Terrain;
443  walking = (char) latest_state->Walking;
444  driving = (char) latest_state->Driving;
445 
446  // Write out the skaters' state
447  // Write out the skater's "doing trick" flag
448  // Write out the skater's terrain type
449  if( update_flags & GameNet::mUPDATE_FIELD_STATE )
450  {
451  char mask;
452 
453  mask = state;
454  if( doing_trick )
455  {
456  mask |= GameNet::mDOING_TRICK_MASK;
457  }
458 
459  update_info->m_BitStream.WriteValue( mask, 4 );
460  update_info->m_BitStream.WriteValue( terrain, 6 );
461  update_info->m_BitStream.WriteValue( walking, 1 );
462  update_info->m_BitStream.WriteValue( driving, 1 );
463 
464  player->m_LastSentProps.m_LastSkaterStateUpdate[id] = state;
465  player->m_LastSentProps.m_LastDoingTrickUpdate[id] = doing_trick;
466  player->m_LastSentProps.m_LastSkaterTerrain[id] = terrain;
467  player->m_LastSentProps.m_LastSkaterWalking[id] = walking;
468  player->m_LastSentProps.m_LastSkaterDriving[id] = driving;
469  }
470 
471  // Write out the skaters' flags
472  skater_flags = latest_state->SkaterFlags;
473  end_run_flags = latest_state->EndRunFlags;
474 
475  if( update_flags & GameNet::mUPDATE_FIELD_FLAGS )
476  {
477  player->m_LastSentProps.m_LastSkaterFlagsUpdate[id] = skater_flags;
478  player->m_LastSentProps.m_LastEndRunFlagsUpdate[id] = end_run_flags;
479  update_info->m_BitStream.WriteValue( skater_flags, 5 );
480  update_info->m_BitStream.WriteValue( end_run_flags, 3 );
481  }
482 
483  // Write out the skaters' rail node
484  if( update_flags & GameNet::mUPDATE_FIELD_RAIL_NODE )
485  {
486  player->m_LastSentProps.m_LastRailNodeUpdate[id] = latest_state->RailNode;
487  update_info->m_BitStream.WriteValue( latest_state->RailNode, sizeof( sint16 ) * 8 );
488  }
489 
490 }

493 /* Update Clients with objects' positional/rotational/anim data */

497 void Skate::s_object_update_code ( const Tsk::Task< Skate >& task )
498 {
499  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
500  GameNet::PlayerInfo* player;
501  Lst::Search< GameNet::PlayerInfo > sh;
502  Net::Server* server;
503 
504  Dbg_AssertType ( &task, Tsk::Task< Skate > );
505  Skate& mdl = task.GetData();
506 
507  server = gamenet_man->GetServer();
508 
509  for( player = gamenet_man->FirstPlayerInfo( sh, true ); player; player =
510  gamenet_man->NextPlayerInfo( sh, true ))
511  {
512  Net::MsgMax update_msg;
513  unsigned int length;
514  ObjUpdateInfo update_info;
515  GameNet::PlayerInfo* update_player;
516  int i, start_id, obj_id, num_updates;
517 
518  // Don't send updates to the local client. His stuff is already up-to-date
519  if( player->m_Conn->IsRemote())
520  {
521  int obj_id_mask;
522 
523  update_info.m_BitStream.SetOutputData( update_msg.m_Data, 4096 ); // skip Obj ID Mask
524  update_info.m_Handle = player->m_Conn->GetHandle();
525  update_info.m_Player = player;
526 
527  obj_id_mask = 0;
528 
529  // First decide which objects' updates we will send, then append them to the stream
530  // in increasing id order
531  num_updates = 0;
532  start_id = ( player->GetLastObjectUpdateID() + 1 ) % vMAX_SKATERS;
533  for( i = 0; i < vMAX_SKATERS; i++ )
534  {
535  obj_id = ( start_id + i ) % vMAX_SKATERS;
536  update_player = gamenet_man->GetPlayerByObjectID( obj_id );
537  if( update_player )
538  {
539  if( mdl.should_send_update( player, update_player->m_Skater ))
540  {
541  obj_id_mask |= ( 1 << obj_id );
542  if( ++num_updates >= player->GetMaxObjectUpdates())
543  {
544  break;
545  }
546  }
547  }
548  }
549 
550  if( num_updates == 0 )
551  {
552  continue;
553  }
554 
555  update_info.m_BitStream.WriteValue( obj_id_mask, sizeof( char ) * 8 );
556  for( i = 0; i < vMAX_SKATERS; i++ )
557  {
558  if( obj_id_mask & ( 1 << i ))
559  {
560  update_player = gamenet_man->GetPlayerByObjectID( i );
561 
562  s_object_update( update_player->m_Skater, &update_info );
563  player->SetLastObjectUpdateID( i );
564  }
565  }
566 
567  update_info.m_BitStream.Flush();
568  length = update_info.m_BitStream.GetByteLength();
569 
570  if( length > 0 )
571  {
572  Net::MsgDesc msg_desc;
573 
574  msg_desc.m_Data = &update_msg;
575  msg_desc.m_Length = length;
576  msg_desc.m_Id = GameNet::MSG_ID_OBJ_UPDATE_STREAM;
577  server->EnqueueMessage( player->GetConnHandle(), &msg_desc );
578  }
579  }
580  }
581 }

588 void Skate::SendScoreUpdates( bool final )
589 {
590  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
591  GameNet::PlayerInfo* player, *score_player;
592  Lst::Search< GameNet::PlayerInfo > sh, score_sh;
593 
594  for( player = gamenet_man->FirstPlayerInfo( sh, true ); player; player =
595  gamenet_man->NextPlayerInfo( sh, true ))
596  {
597  GameNet::MsgScoreUpdate score_msg;
598  char num_scores;
599  int length;
600  char* data;
601 
602  score_msg.m_Final = (char) final;
603  if( final )
604  {
605  score_msg.m_TimeLeft = 0;
606  }
607  else
608  {
609  score_msg.m_TimeLeft = Tmr::InSeconds( GetGameMode()->GetTimeLeft());
610  }
611  num_scores = 0;
612  data = score_msg.m_ScoreData;
613  for( score_player = gamenet_man->FirstPlayerInfo( score_sh ); score_player; score_player =
614  gamenet_man->NextPlayerInfo( score_sh ))
615  {
616  bool send_all_scores;
617 
618  send_all_scores = final ||
619  ( GetGameMode()->IsTrue( "should_modulate_color" )) ||
620  ( GetGameMode()->ShouldTrackTrickScore() == false );
621  if( ( score_player != player ) || ( send_all_scores ) )
622  {
623  char id;
624  int score;
625  bool in_goal_attack;
626 
627  id = (char) score_player->m_Skater->GetID();
628  in_goal_attack = GetGameMode()->GetNameChecksum() == Script::GenerateCRC( "netgoalattack" );
629 
630  if( in_goal_attack )
631  {
632  Game::CGoalManager* pGoalManager;
633 
634  pGoalManager = Game::GetGoalManager();
635  score = pGoalManager->NumGoalsBeatenBy( score_player->m_Skater->GetID());
636  }
637  else
638  {
639  score = gamenet_man->GetPlayerScore( score_player->m_Skater->GetID());
640  }
641 
642  *data = id;
643  data++;
644  memcpy( data, &score, sizeof( int ));
645  data += sizeof( int );
646 
647  num_scores++;
648  }
649  }
650 
651  if( num_scores > 0 )
652  {
653  Net::MsgDesc msg_desc;
654 
655  score_msg.m_Cheats = GetCheatChecksum();
656  score_msg.m_NumScores = num_scores;
657  length = sizeof( int ) + // m_TimeLeft
658  sizeof( int ) + // m_Cheats
659  sizeof( char ) + // m_Final
660  sizeof( char ) + // m_NumScores
661  (( sizeof( char ) + sizeof( int )) * num_scores );
662 
663  msg_desc.m_Data = &score_msg;
664  msg_desc.m_Length = length;
665  msg_desc.m_Id = GameNet::MSG_ID_SCORE_UPDATE;
666  // If it's the final tally or if it's reporting cheats, it HAS to get there
667  if( final || ( score_msg.m_Cheats != 0 ))
668  {
669  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
670  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
671  }
672 
673  gamenet_man->GetServer()->EnqueueMessage( player->GetConnHandle(), &msg_desc );
674 
675  if( !player->IsLocalPlayer() && player->IsFullyIn())
676  {
677  GameNet::MsgCheatChecksum cheat_msg;
678 
679  cheat_msg.m_ServerChecksum = GetCheatChecksum();
