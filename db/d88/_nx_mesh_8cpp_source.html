75 bool CMesh::LoadCollision(const char *p_name)
76 {
77 
78  // for now collision is kind of assumed to be platform independent
79  strcpy(m_coll_filename, p_name);
80  char *p_ext = strstr(m_coll_filename, ".");
81  if (p_ext)
82  {
83  strcpy(p_ext, ".col.");
84  } else {
85  strcat(m_coll_filename, ".col.");
86  }
87  strcat(m_coll_filename, CEngine::sGetPlatformExtension());
88 
89 // Dbg_Message ( "Loading collision %s....", m_coll_filename );
90 
91  Mem::PushMemProfile((char*)m_coll_filename);
92 
93  uint8 *p_base_addr = (uint8 *) Pip::Load(m_coll_filename);
94  if (p_base_addr)
95  {
96  Nx::CCollObjTriData::SReadHeader *p_header = (Nx::CCollObjTriData::SReadHeader *) p_base_addr;
97  p_base_addr += sizeof(Nx::CCollObjTriData::SReadHeader);
98 
99 // Dbg_Message ( "Version # %d header sizeof %d", p_header->m_version, sizeof(Nx::CCollObjTriData));
100 #ifdef __PLAT_NGC__
101  Dbg_Message ( "Number of objects: %d verts: %d faces: %d", p_header->m_num_objects, p_header->m_total_num_verts, p_header->m_total_num_faces );
102 #else
103 // Dbg_Message ( "Number of objects: %d verts: %d faces: %d", p_header->m_num_objects, p_header->m_total_num_verts, p_header->m_total_num_faces_large + p_header->m_total_num_faces_small);
104 // Dbg_Message ( "Small (%d) verts: %d Large (%d) verts: %d", Nx::CCollObjTriData::GetVertSmallElemSize(), p_header->m_total_num_verts_small, Nx::CCollObjTriData::GetVertElemSize(), p_header->m_total_num_verts_large);
105 #endif // __PLAT_NGC__
106  Dbg_MsgAssert(p_header->m_version >= 9, ("Collision version must be at least 9."));
107 
108  // reserve space for objects
109  m_num_coll_objects = p_header->m_num_objects;
110  mp_coll_objects = (Nx::CCollObjTriData *) p_base_addr;
111 
112  // Calculate base addresses for vert and face arrays
113  uint8 *p_base_vert_addr = (uint8 *) (mp_coll_objects + m_num_coll_objects);
114 #ifndef __PLAT_NGC__
115  p_base_vert_addr = (uint8 *)(((uint)(p_base_vert_addr+15)) & 0xFFFFFFF0); // Align to 128 bit boundary
116 #ifdef FIXED_POINT_VERTICES
117  uint8 *p_base_intensity_addr = p_base_vert_addr + (p_header->m_total_num_verts_large * Nx::CCollObjTriData::GetVertElemSize() +
118  p_header->m_total_num_verts_small * Nx::CCollObjTriData::GetVertSmallElemSize());
119  uint8 *p_base_face_addr = p_base_intensity_addr + p_header->m_total_num_verts;
120  p_base_face_addr = (uint8 *)(((uint)(p_base_face_addr+3)) & 0xFFFFFFFC); // Align to 32 bit boundary
121 #else
122  uint8 *p_base_intensity_addr = NULL;
123  uint8 *p_base_face_addr = p_base_vert_addr + (p_header->m_total_num_verts * Nx::CCollObjTriData::GetVertElemSize());
124  p_base_face_addr = (uint8 *)(((uint)(p_base_face_addr+15)) & 0xFFFFFFF0); // Align to 128 bit boundary
125 #endif // FIXED_POINT_VERTICES
126 #else
127  uint8 *p_base_face_addr = p_base_vert_addr + (p_header->m_total_num_faces * Nx::CCollObjTriData::GetVertElemSize());
128  p_base_face_addr = (uint8 *)(((uint)(p_base_face_addr+3)) & 0xFFFFFFFC); // Align to 32 bit boundary
129 #endif // __PLAT_NGC__
130 
131  // Calculate addresses for BSP arrays
132 #ifndef __PLAT_NGC__
133  uint8 *p_node_array_size = p_base_face_addr + (p_header->m_total_num_faces_large * Nx::CCollObjTriData::GetFaceElemSize() +
134  p_header->m_total_num_faces_small * Nx::CCollObjTriData::GetFaceSmallElemSize());
135  p_node_array_size += ( p_header->m_total_num_faces_large & 1 ) ? 2 : 0;
136 #else
137  uint8 *p_node_array_size = p_base_face_addr + ( p_header->m_total_num_faces * Nx::CCollObjTriData::GetFaceElemSize() );
138  p_node_array_size += ( p_header->m_total_num_faces & 1 ) ? 2 : 0;
139 #endif // __PLAT_NGC__
140  uint8 *p_base_node_addr = p_node_array_size + 4;
141  uint8 *p_base_face_idx_addr = p_base_node_addr + *((int *) p_node_array_size);
142 
143  // Read objects
144  for (int oidx = 0; oidx < p_header->m_num_objects; oidx++)
145  {
146 #ifdef __PLAT_NGC__
147  CScene * p_scene = static_cast<CScene*>( (static_cast<CNgcMesh*>( this ))->GetScene() );
148  mp_coll_objects[oidx].InitCollObjTriData(p_scene, p_base_vert_addr, NULL, p_base_face_addr, p_base_node_addr, p_base_face_idx_addr);
149 #else
150  mp_coll_objects[oidx].InitCollObjTriData(NULL, p_base_vert_addr, p_base_intensity_addr, p_base_face_addr,
151  p_base_node_addr, p_base_face_idx_addr);
152 #endif // __PLAT_NGC__
153  mp_coll_objects[oidx].InitBSPTree();
154 
155  // Add to mesh bbox
156  m_collision_bbox.AddPoint(mp_coll_objects[oidx].GetBBox().GetMin());
157  m_collision_bbox.AddPoint(mp_coll_objects[oidx].GetBBox().GetMax());
158  }
159 
160 // Dbg_Message("Mesh bounding box: min (%f, %f, %f) max (%f, %f, %f)",
161 // m_collision_bbox.GetMin()[X], m_collision_bbox.GetMin()[Y], m_collision_bbox.GetMin()[Z],
162 // m_collision_bbox.GetMax()[X], m_collision_bbox.GetMax()[Y], m_collision_bbox.GetMax()[Z]);
163 
164  } else {
165  Dbg_Error ( "Could not open collision file\n" );
166  return false;
167  }
168 
169 // Dbg_Message ( "successfully loaded collision" );
170 
171  if (m_num_coll_objects > 0)
172  {
173 #if 0
174  // Add to CSectors
175  for (int i = 0; i < m_num_coll_objects; i++)
176  {
177  CSector *p_sector = GetSector(mp_coll_objects[i].GetChecksum());
178  if (p_sector) // Don't assert now since there may not be renderable data
179  {
180  Dbg_MsgAssert(p_sector, ("LoadCollision: Can't find CSector with checksum %x", mp_coll_objects[i].GetChecksum()));
181  p_sector->AddCollSector(&(mp_coll_objects[i]));
182  }
183  }
184 #endif
185  }
186 
187  Mem::PopMemProfile(/*(char*)m_coll_filename*/);
188 
189  return true;
190 }
