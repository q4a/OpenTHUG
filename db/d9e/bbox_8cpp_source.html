66 // Checks whether pStart-pEnd collides with the axis-aligned bounding box defined by
67 // pMax and pMin.
68 bool LineCollidesWithBox( const Mth::Vector &pStart, const Mth::Vector &pEnd, const Mth::Vector &pMax, const Mth::Vector &pMin )
69 {
70 
71 
72  // if either point is in the box... that counts!
73  if ( PointInsideBox( pStart, pMax, pMin ) ||
74  PointInsideBox( pEnd, pMax, pMin ) )
75  {
76  return ( true );
77  }
78 
79  // Trivial rejection.
80  if((pStart[ Y ]>pMax[ Y ])&&( pEnd[ Y ]>pMax[ Y ])) return false;
81  if((pStart[ Y ]<pMin[ Y ])&&( pEnd[ Y ]<pMin[ Y ])) return false;
82  if((pStart[ X ]>pMax[ X ])&&( pEnd[ X ]>pMax[ X ])) return false;
83  if((pStart[ X ]<pMin[ X ])&&( pEnd[ X ]<pMin[ X ])) return false;
84  if((pStart[ Z ]>pMax[ Z ])&&( pEnd[ Z ]>pMax[ Z ])) return false;
85  if((pStart[ Z ]<pMin[ Z ])&&( pEnd[ Z ]<pMin[ Z ])) return false;
86 
87  float dx=pEnd[ X ]-pStart[ X ];
88  float dy=pEnd[ Y ]-pStart[ Y ];
89  float dz=pEnd[ Z ]-pStart[ Z ];
90 
91  // avoid divide by zeros.
92  if ( !dx )
93  {
94  dx = ( 0.000001 );
95  }
96  if ( !dy )
97  {
98  dy = ( 0.000001 );
99  }
100  if ( !dz )
101  {
102  dz = ( 0.000001 );
103  }
104 
105  // Check the max-x face.
106  if (pStart[ X ]>pMax[ X ] && pEnd[ X ]<pMax[ X ])
107  {
108  // It crosses the plane of the face, so calculate the y & z coords
109  // of the intersection and see if they are in the face,
110  float d=pMax[ X ]-pStart[ X ];
111  float y=d*dy/dx+pStart[ Y ];
112  float z=d*dz/dx+pStart[ Z ];
113  if (y<pMax[ Y ] && y>pMin[ Y ] && z<pMax[ Z ] && z>pMin[ Z ])
114  {
115  // It does collide!
116  return true;
117  }
118  }
119 
120  // Check the min-x face.
121  if (pStart[ X ]<pMin[ X ] && pEnd[ X ]>pMin[ X ])
122  {
123  // It crosses the plane of the face, so calculate the y & z coords
124  // of the intersection and see if they are in the face,
125  float d=pMin[ X ]-pStart[ X ];
126  float y=d*dy/dx+pStart[ Y ];
127  float z=d*dz/dx+pStart[ Z ];
128  if (y<pMax[ Y ] && y>pMin[ Y ] && z<pMax[ Z ] && z>pMin[ Z ])
129  {
130  // It does collide!
131  return true;
132  }
133  }
134 
135  // Check the max-y face.
136  if (pStart[ Y ]>pMax[ Y ] && pEnd[ Y ]<pMax[ Y ])
137  {
138  // It crosses the plane of the face, so calculate the x & z coords
139  // of the intersection and see if they are in the face,
140  float d=pMax[ Y ]-pStart[ Y ];
141  float x=d*dx/dy+pStart[ X ];
142  float z=d*dz/dy+pStart[ Z ];
143  if (x<pMax[ X ] && x>pMin[ X ] && z<pMax[ Z ] && z>pMin[ Z ])
144  {
145  // It does collide!
146  return true;
147  }
148  }
149 
150  // Check the min-y face.
151  if (pStart[ Y ]<pMin[ Y ] && pEnd[ Y ]>pMin[ Y ])
152  {
153  // It crosses the plane of the face, so calculate the x & z coords
154  // of the intersection and see if they are in the face,
155  float d=pMin[ Y ]-pStart[ Y ];
156  float x=d*dx/dy+pStart[ X ];
157  float z=d*dz/dy+pStart[ Z ];
158  if (x<pMax[ X ] && x>pMin[ X ] && z<pMax[ Z ] && z>pMin[ Z ])
159  {
160  // It does collide!
161  return true;
162  }
163  }
164 
165  // Check the max-z face.
166  if (pStart[ Z ]>pMax[ Z ] && pEnd[ Z ]<pMax[ Z ])
167  {
168  // It crosses the plane of the face, so calculate the x & y coords
169  // of the intersection and see if they are in the face,
170  float d=pMax[ Z ]-pStart[ Z ];
171  float x=d*dx/dz+pStart[ X ];
172  float y=d*dy/dz+pStart[ Y ];
173  if (x<pMax[ X ] && x>pMin[ X ] && y<pMax[ Y ] && y>pMin[ Y ])
174  {
175  // It does collide!

186  float d=pMin[ Z ]-pStart[ Z ];
187  float x=d*dx/dz+pStart[ X ];
188  float y=d*dy/dz+pStart[ Y ];
