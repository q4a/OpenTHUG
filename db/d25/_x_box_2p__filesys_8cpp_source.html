83 /* prefopen */
84 /* Same as fopen() except it will prepend the data root directory */
85 /* For Xbox, path is always relative to location of .xbe image */
86 /* and default path from that location is d:\ */
87 /* So incoming files of the form: */
88 /* c:\skate3\data\foo... become d:\data\foo... */
89 /* foo... become d:\data\foo... */
90 /* */
91 /******************************************************************/
92 static void* prefopen( const char *filename, const char *mode )
93 {
94  // Used for prepending the root data directory on filesystem calls.
95  char nameConversionBuffer[256] = "d:\\data\\";
96  int index = PREPEND_START_POS;
97  const char* p_skip;
98 
99  if(( filename[0] == 'c' ) && ( filename[1] == ':' ))
100  {
101  // Filename is of the form c:\skate4\data\foo...
102  p_skip = filename + 15;
103  }
104  else
105  {
106  // Filename is of the form foo...
107  p_skip = filename;
108  }
109 
110  while( nameConversionBuffer[index] = *p_skip )
111  {
112  // Switch forward slash directory separators to the supported backslash.
113  if( nameConversionBuffer[index] == '/' )
114  {
115  nameConversionBuffer[index] = '\\';
116  }
117  ++index;
118  ++p_skip;
119  }
120  nameConversionBuffer[index] = 0;
121 
122  // If this is a .tex file, switch to a .txx file.
123  if((( nameConversionBuffer[index - 1] ) == 'x' ) &&
124  (( nameConversionBuffer[index - 2] ) == 'e' ) &&
125  (( nameConversionBuffer[index - 3] ) == 't' ))
126  {
127  nameConversionBuffer[index - 2] = 'x';
128  }
129 
130  // If this is a .pre file, switch to a .prx file.
131  if(((( nameConversionBuffer[index - 1] ) == 'e' ) || (( nameConversionBuffer[index - 1] ) == 'x' )) &&
132  (( nameConversionBuffer[index - 2] ) == 'r' ) &&
133  (( nameConversionBuffer[index - 3] ) == 'p' ))
134  {
135 # ifdef __PAL_BUILD__
136  // Switch to a .prf, .prg or .prx file, depending on language.
137  switch( Config::GetLanguage())
138  {
139  case Config::LANGUAGE_FRENCH:
140  {
141  nameConversionBuffer[index - 1] = 'f';
142  break;
143  }
144  case Config::LANGUAGE_GERMAN:
145  {
146  nameConversionBuffer[index - 1] = 'g';
147  break;
148  }
149  default:
150  {
151  nameConversionBuffer[index - 1] = 'x';
152  break;
153  }
154  }
155 # else
156  nameConversionBuffer[index - 1] = 'x';
157 # endif // __PLAT_BUILD__
158  }
159 
160  // First we try reading the file from the utility partition (z:\) on the HD, rather than the DVD.
161  HANDLE h_file = INVALID_HANDLE_VALUE;
162  if( OkayToUseUtilityDrive )
163  {
164  nameConversionBuffer[0] = 'Z';
165  h_file = CreateFile( nameConversionBuffer, GENERIC_READ, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL );
166  }
167 
168  if( h_file == INVALID_HANDLE_VALUE )
169  {
170  // Not on the utility partition, so load it from the DVD.
171  nameConversionBuffer[0] = 'D';
172  h_file = CreateFile( nameConversionBuffer, GENERIC_READ, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL );
173 
174  // Deal with various error returns.
175  if( h_file == INVALID_HANDLE_VALUE )
176  {
177  DWORD error = GetLastError();
178 
179  // Need to exclude this error from the test, since screenshot and other code sometimes check to see if a file exists, and it
180  // is valid to just return the error code if it doesn't.
181  if( error != ERROR_FILE_NOT_FOUND )
182  {
183  // Catch-all error indicating a fatal problem. Can't continue at this point.
184  // The ideal solution would be a catch/throw exception mechanism, but we don't include exception handling at the moment.
185  // For now just call this NxXbox function, which is slightly messy since it means we have to include a gfx\ file.
186  printf( "FatalFileError: %x %s\n", error, nameConversionBuffer );
187  NxXbox::FatalFileError((uint32)INVALID_HANDLE_VALUE );
188  }
189  return NULL;
190  }
191  else
192  {
193  // All is well.
194  return h_file;
195  }
196  }
197  else
198  {
199  return h_file;
200  }
201 }

436 size_t Read( void *addr, size_t size, size_t count, void *pFP )
437 {
438  if (PreMgr::sPreEnabled())
439  {
440  size_t retval = PreMgr::pre_fread( addr, size, count, (PreFile::FileHandle*)pFP );
441  if( PreMgr::sPreExecuteSuccess())
442  return retval;
443  }
444 
445  DWORD bytes_read;
446  if( ReadFile((HANDLE)pFP, addr, size * count, &bytes_read, NULL ))
447  {
448  // All is well.
449  return bytes_read;
450  }
451  else
452  {
453  // Read error.
454  DWORD last_error = GetLastError();
455 
456  if( last_error == ERROR_HANDLE_EOF )
457  {
458  // Continue in this case.
459  return bytes_read;
460  }
461 
462  NxXbox::FatalFileError( last_error );
463  return bytes_read;
464  }
465 }
