692 CSector * CSector::clone(bool instance, bool add_to_super_sectors, CScene *p_dest_scene)
693 {
694  // Create new instance and clone stuff below p-line
695  CSector *p_new_sector = plat_clone(instance, p_dest_scene);
696 
697  if (p_new_sector)
698  {
699  // Get new checksum
700  p_new_sector->m_checksum = CEngine::sGetUnusedSectorChecksum();
701 
702  // Now clone stuff above p-line
703  p_new_sector->m_flags = m_flags | mCLONE; // mark as clone
704  p_new_sector->m_rot_y = m_rot_y;
705 
706  if (mp_geom)
707  {
708  p_new_sector->mp_geom = mp_geom->Clone(instance, p_dest_scene);
709  Dbg_Assert(p_new_sector->mp_geom);
710  }
711 
712  if (mp_coll_sector)
713  {
714 
715  // Note, static_cast seems to be essentially a macro
716  // so you must pass it a pointer, and not the result of an expression
717  // otherwise it will continually re-evaluate that expression
718  // which here would casue memory leaks.
719  CCollObj *p_coll_obj = mp_coll_sector->Clone(false /*instance*/); // Garrett: We're not even going to try to instamce it
720  if (p_coll_obj)
721  {
722  p_new_sector->mp_coll_sector = static_cast<CCollStatic *>(p_coll_obj);
723 
724  p_new_sector->mp_coll_sector->SetChecksum(p_new_sector->m_checksum);
725 
726  p_new_sector->m_flags &= ~mIN_SUPER_SECTORS; // Not in there yet
727 
728  if (add_to_super_sectors)
729  {
730  p_new_sector->m_flags |= mADD_TO_SUPER_SECTORS;
731  }
732 
733  Dbg_Assert(p_new_sector->mp_coll_sector);
734 
735  // For now, just copy the CCollObjTriData pointer over
736  if (p_new_sector->mp_geom)
737  {
738  // Garrett: Commented assert out because the pointer is now copied in CGeom
739  //Dbg_Assert(p_new_sector->mp_geom->GetCollTriData() == NULL);
740  p_new_sector->mp_geom->SetCollTriData(p_coll_obj->GetGeometry());
741  }
742 
743  Dbg_Assert(p_coll_obj->GetGeometry() != mp_coll_sector->GetGeometry());
744  }
745  else
746  {
747  p_new_sector->mp_coll_sector = NULL;
748  p_new_sector->m_flags &= ~mIN_SUPER_SECTORS; // Not in there at all
749  if (p_new_sector->mp_geom)
750  {
751  p_new_sector->mp_geom->SetCollTriData(NULL);
752  }
753  }
754  }
755 
756  return p_new_sector;
757  } else {
758  return NULL;
759  }
760 }
