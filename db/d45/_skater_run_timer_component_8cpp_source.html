85 void CSkaterRunTimerComponent::Update()
86 {
87  if (m_state == ACTIVE_RUNNING)
88  {
89  switch (mp_walk_component->GetState())
90  {
91  case CWalkComponent::WALKING_HANG:
92  case CWalkComponent::WALKING_LADDER:
93  case CWalkComponent::WALKING_ANIMWAIT:
94  m_timer -= Script::GetFloat(CRCD(0x92fb4d09, "Hang_Run_Timer_Speed_Adjustment")) * Tmr::FrameLength();
95  break;
96 
97  default:
98  m_timer -= Tmr::FrameLength();
99  break;
100  }
101 
102  if (m_timer < 0.0f)
103  {
104  set_state(ACTIVE_TIME_UP);
105  GetObject()->SelfEvent(CRCD(0x9e22a8e9, "RunTimerUp"));
106  }
107  }
108  else if (m_state == ACTIVE_TIME_UP)
109  {
110  GetObject()->SelfEvent(CRCD(0x9e22a8e9, "RunTimerUp"));
111  }
112 }

119 CBaseComponent::EMemberFunctionResult CSkaterRunTimerComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
120 {
121  switch ( Checksum )
122  {
123  // @script | RunTimer_Pause | pauses the run timer
124  case CRCC(0x82445966, "RunTimer_Pause"):
125  pause();
126  break;
127 
128  // @script | RunTimer_UnPause | unpauses the run timer
129  case CRCC(0x956236b1, "RunTimer_UnPause"):
130  unpause();
131  break;
132 
133  // @script | RunTimer_GetFactorComplete | returns the time left of the run timer as a float between zero and one in FactorComplete
134  case CRCC(0xab04cbf9, "RunTimer_GetFactorComplete"):
135  pScript->GetParams()->AddFloat(CRCD(0x41f56511, "FactorComplete"), m_timer / GetSkater()->GetScriptedStat(CRCD(0xb84f532, "Physics_RunTimer_Duration")));
136  break;
137 
138  // @script | RunTimer_GetRunTimerControllerId | returns the id of the display controller script for the run timer in RunTimerControllerId
139  case CRCC(0x96d81d8e, "RunTimer_GetRunTimerControllerId"):
140  pScript->GetParams()->AddChecksum(CRCD(0x39778122, "RunTimerControllerId"), get_run_timer_controller_id());
141  break;
142 
143  // @script | RunTimer_GetRunTimerId | returns the id of the display controller script for the run timer in RunTimerId
144  case CRCC(0x81aaa6dd, "RunTimer_GetRunTimerId"):
145  pScript->GetParams()->AddChecksum(CRCD(0x628f21c9, "RunTimerId"), CRCD(0x7d42821c, "the_run_timer") + GetSkater()->GetSkaterNumber());
146  break;
147 
148  default:
149  return CBaseComponent::MF_NOT_EXECUTED;
150  }
151 
152  return CBaseComponent::MF_TRUE;
153 }

198 void CSkaterRunTimerComponent::unpause ( )
199 {
200  if (m_state == INACTIVE) return;
201 
202  set_state(m_timer > 0.0f ? ACTIVE_RUNNING : ACTIVE_TIME_UP);
203 
204  if (m_unpause_count < vRT_NUM_TIME_CHUNKS)
205  {
206  float max_timer = (vRT_NUM_TIME_CHUNKS - m_unpause_count) * GetSkater()->GetScriptedStat(CRCD(0xb84f532, "Physics_RunTimer_Duration")) / vRT_NUM_TIME_CHUNKS;
207  m_timer = Mth::Min(m_timer, max_timer);
208  }
209 
210  m_unpause_count++;
211 
212  CTracker::Instance()->LaunchEvent(CRCD(0x47eec244, "ShowRunTimer"), get_run_timer_controller_id(), GetObject()->GetID());
213 }
