174 int Score::get_packed_score( int start, int end )
175 {
176  int packed_score = 0;
177  int current_spin_mult = 2; // n over 2
178  for (int i = start; i < end; i++)
179  {
180  // if a blocking trick, use default spin multiplier
181  if (m_infoTab[i].flags & vBLOCKING)
182  {
183  current_spin_mult = 2;
184  }
185  // if not a blocking trick, but the last trick was, then use the spin multiplier here
186  // to apply to subsequent tricks up to next blocking one
187  else if (i == 0 || (m_infoTab[i-1].flags & vBLOCKING))
188  {
189  current_spin_mult = spinMult(m_infoTab[i].spin_mult_index);
190  }
191 
192  int deprec_mult = deprecMult(m_infoTab[i].mult_index);
193 
194 // if (m_debug)
195 // printf(" base score %d, reduction mult %.2f, spin mult %.1f\n",
196 // m_infoTab[i].score, (float) deprec_mult / 100, (float) current_spin_mult / 2);
197 
198  int score = (m_infoTab[i].score * deprec_mult * current_spin_mult) / 200;
199  if (m_infoTab[i].switch_mode)
200  {
201  score = score * 120 / 100;
202  }
203 
204  packed_score += score;
205  }
206  return packed_score;
207 }

209 void Score::pack_trick_info_table()
210 {
211  TrickInfo theTrickInfo;
212  strcpy( theTrickInfo.trickNameText, "... + " );
213  strcpy( theTrickInfo.trickTextFormatted, "...\\_+\\_" );
214  theTrickInfo.score = 0;
215  theTrickInfo.id = 0;
216  theTrickInfo.switch_mode = 0;
217  theTrickInfo.flags = 0;
218  theTrickInfo.mult_index = 0;
219  theTrickInfo.spin_mult_index = 0;
220  theTrickInfo.spin_degrees = 0;
221 
222  int firstBlockingTrick = m_infoTab.GetSize();
223 
224  for ( int i = 0; i < m_infoTab.GetSize(); i++ )
225  {
226 // if ( ( i != 0 ) && ( m_infoTab[i].flags & vBLOCKING ) )
227  if ( ( i != 0 ) && ( m_infoTab[i].trickNameText[0] == 0 ) )
228  {
229  firstBlockingTrick = i;
230  break;
231  }
232  }
233 
234  // GJ: see note below
235  if ( ( firstBlockingTrick > m_currentSpinTrick )
236  || ( firstBlockingTrick > m_currentBlockingTrick ) )
237  {
238  return;
239  }
240 
241  theTrickInfo.score = get_packed_score( 0, firstBlockingTrick );
242 
243  // shift them all...
244  for ( int i = 0; i < firstBlockingTrick; i++ )
245  {
246  m_infoTab.Remove( 1 );
247 
248  m_currentTrick--;
249  m_currentSpinTrick--;
250  m_currentBlockingTrick--;
251 
252  // GJ: these asserts will fire off if the trick list is filled
253  // with more non-blocking tricks than spin tricks... it rarely
254  // happens, except for those glitches that lets the player do
255  // infinite tricks. The if-test above will abort early,
256  // allowing the calling function ( Trigger() ) to gracefully
257  // handle it
258  Dbg_MsgAssert( m_currentTrick > -1, ( "m_currentTrick is negative (%d)", m_currentTrick ) );
259  Dbg_MsgAssert( m_currentSpinTrick > -1, ( "m_currentSpinTrick is negative (%d)", m_currentSpinTrick ) );
260  Dbg_MsgAssert( m_currentBlockingTrick > -1, ( "m_currentBlockingTrick is negative (%d)", m_currentBlockingTrick ) );
261  }
262 
263  m_infoTab[0] = theTrickInfo;
264 
265  // so that the packed score will be correct next time
266  m_infoTab[0].flags |= vBLOCKING;
267 
268 #ifdef __USER_GARY__
269  Dbg_Message( "Info table has %d elements.", m_infoTab.GetSize() );
270 #endif

499 void Score::Trigger(char *trick_name, int base_score, Flags flags)
500 {
501  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
502 
503  bool null_trick=false;
504  if (!trick_name[0])
505  {
506  null_trick=true;
507  }
508 
509  // once some trick/score has appeared on-screen
510  // we know that the trick has started
511  Obj::CSkater* pSkater = skate_mod->GetSkaterById( m_skaterId );
512  Dbg_Assert( pSkater );
513  Obj::CTrickComponent* pTrickComponent = GetTrickComponentFromObject(pSkater);
514  Dbg_Assert( pTrickComponent );
515  Obj::CStatsManagerComponent* pStatsManagerComponent = GetStatsManagerComponentFromObject(pSkater);
516  Dbg_Assert( pStatsManagerComponent );
517 
518  // Only flag a trick as having started if a real trick is being triggered
519  if( trick_name && trick_name[0] )
520  {
521  pTrickComponent->SetFirstTrickStarted( true );
522  }
523 
524  bool is_switch = false; // need to make this grab the switch flag eventually!!!
525  pStatsManagerComponent->SetTrick( trick_name, base_score, is_switch );
526 
527  // We place a high limit on tricks, so if they
528  // cheat with perfect baclance, and keep tricking
529  // then they will not run out of memory and crash
530  if (m_currentTrick > TRICK_LIMIT-1)
531  {
532  pack_trick_info_table();
533 
534  if (m_currentTrick > TRICK_LIMIT-1)
535  {
536  // GJ: You should never be able to hit the trick limit any more
537 
538  if ( !null_trick )
539  {
540  printf ("Trick Limit (%d) Reached\n",TRICK_LIMIT);
541  //tweak_last_valid_trick(base_score);
542  m_infoTab.Last().score += base_score; // give them the score, just stop recording tricks
543  m_currentMult++; // but still keep up the multipler
544  captureScore();
545  }
546  return;
547  }
548  }
549 
550  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().FrontEndHeap());
551 
552  // trick type is based on id + switch mode
553  uint32 id = Script::GenerateCRC(trick_name);
554  uint32 switch_mode = (flags & (vSWITCH | vNOLLIE)) >> 1;
555 
556  if (m_infoTab.GetSize() < TRICK_LIMIT)
557  {
558  m_infoTab.Add(new TrickInfo);
559  m_infoTab.Last().score = 0;
560  }
561  // if we have filled up the trick limit, then we just use the last one we added
562 
563  m_infoTab.Last().score += base_score;
564  //printf("triggering trick with id %s\n",trick_name);
565  m_infoTab.Last().id = id;
566  m_infoTab.Last().switch_mode = switch_mode;
567  m_infoTab.Last().flags = flags;
568  m_infoTab.Last().spin_mult_index = 0;
569  m_infoTab.Last().trickNameText[0] = 0;
570  m_infoTab.Last().trickTextFormatted[0] = 0;
571  m_infoTab.Last().spin_degrees = 0;
572 
573  if (!null_trick)
574  {
575  copy_trick_name_with_special_spaces(m_infoTab.Last().trickNameText, trick_name);
576 
577  if ( flags & Score::vCAT )
578  {
579  sprintf(m_infoTab.Last().trickTextFormatted, "\\c3%s\\c0 ", m_infoTab.Last().trickNameText);
580  }
581  else if ( flags & Score::vSPECIAL )
582  {
583  // K: If the 'special' flag is set, use color 2
584  sprintf(m_infoTab.Last().trickTextFormatted, "\\c2%s\\c0 ", m_infoTab.Last().trickNameText);
585  }
586  else
587  {
588  sprintf(m_infoTab.Last().trickTextFormatted, "%s ", m_infoTab.Last().trickNameText);
589  }
590 
591  // toss the "+" onto the end of the previous trick
592 
593  // K: This bit of code used to do a sprintf, but I changed it to use a strcat instead
594  // to preserve any color formatting added previously, such as the special-trick color
595  // added above.
596 
597  // This is a loop so that null tricks (tricks with no name) get skipped over.
598  int index=m_infoTab.GetSize() - 2;
599  while (true)
600  {
601  if (index<0)
602  {
603  break;
604  }
605 
606  char *p_trick_text_formatted=m_infoTab[index].trickTextFormatted;
607  Dbg_MsgAssert(p_trick_text_formatted,("NULL p_trick_text_formatted"));
608 
609  int len=strlen(p_trick_text_formatted);
610  if (*p_trick_text_formatted == '.')
611  {
612  // skip items that begin with "..."
613  // (or else you get things like
614  // "... + + blah blah")
615  }
616  else if (len)
617  {
618  // Remove any trailing space character
619  if (p_trick_text_formatted[len-1]==' ')
620  {
621  p_trick_text_formatted[len-1]=0;
622  }
623 
624  // Wack on the +
625  strcat(p_trick_text_formatted,"\\_+ ");
626  break;
627  }
628  // Keep searching backwards until a named trick is found.
629  --index;
630  }
