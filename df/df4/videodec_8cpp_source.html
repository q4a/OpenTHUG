407 static int cpy2area(u_char *pd0, int d0, u_char *pd1, int d1,
408  u_char *ps0, int s0, u_char *ps1, int s1)
409 {
410  if (d0 + d1 < s0 + s1) {
411  return 0;
412  }
413 
414  if (s0 >= d0) {
415  memcpy(pd0, ps0, d0);
416  memcpy(pd1, ps0 + d0, s0 - d0);
417  memcpy(pd1 + s0 - d0, ps1, s1);
418  } else { // s0 < d0
419  if (s1 >= d0 - s0) {
420  memcpy(pd0, ps0, s0);
421  memcpy(pd0 + s0, ps1, d0 - s0);
422  memcpy(pd1, ps1 + d0 - s0, s1 - (d0 - s0));
423  } else { // s1 < d0 - s0
424  memcpy(pd0, ps0, s0);
425  memcpy(pd0 + s0, ps1, s1);
426  }
427  }
428  return s0 + s1;
429 }
