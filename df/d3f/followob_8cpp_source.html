28 // Adds a new position taken from the leader object, if that position is suitably interesting.
29 // (Ie, it is not too close to the last position)
30 // FOLLOW_RESOLUTION should be kept as big as possible. That way, the array of positions does
31 // not need to be so big, and calculating the position will be faster because it won't have to
32 // step over so many segments.
33 // If it is too big though the object won't follow the leader's path so accurately and
34 // will start cutting corners noticeably.
35 #define FOLLOW_RESOLUTION 40.0f
36 void CFollowOb::GetNewPathPointFromObjectBeingFollowed()
37 {
38  //CMovingObject *p_obj=CMovingObject::m_hash_table.GetItem(m_leader_name);
39  CMovingObject *p_obj=(CMovingObject*)Obj::ResolveToObject(m_leader_name);
40 
41  if (!p_obj)
42  {
43  // Can't find the object, oh well.
44  return;
45  }
46 
47  m_leader_pos=p_obj->m_pos;
48 
49  if (m_num_path_points==0)
50  {
51  // Insert the first point.
52  m_most_recent_path_point=0;
53  mp_path_points[0]=m_leader_pos;
54  ++m_num_path_points;
55  return;
56  }
57 
58  if (Mth::DistanceSqr(mp_path_points[m_most_recent_path_point],m_leader_pos) >
59  FOLLOW_RESOLUTION*FOLLOW_RESOLUTION)
60  {
61  // The new point is sufficiently far from the last, so insert it into the cyclic buffer.
62  ++m_most_recent_path_point;
63  if (m_most_recent_path_point>=MAX_FOLLOWOB_PATH_POINTS)
64  {
65  m_most_recent_path_point=0;
66  }
67 
68  mp_path_points[m_most_recent_path_point]=m_leader_pos;
69 
70  // m_num_path_points increases till it hits MAX_FOLLOWOB_PATH_POINTS then stays there.
71  if (m_num_path_points<MAX_FOLLOWOB_PATH_POINTS)
72  {
73  ++m_num_path_points;
74  }
75  }

78 void CFollowOb::CalculatePositionAndOrientation(Mth::Vector *p_pos, Mth::Matrix *p_orientation)
79 {
80  Dbg_MsgAssert(p_pos,("NULL p_pos"));
81  Dbg_MsgAssert(p_orientation,("NULL p_orientation"));
82 
83  if (m_num_path_points<1)
84  {
85  // Need at least 1 path point before the interpolated position can be calculated.
86  return;
87  }
88 
89  Mth::Vector seg_start;
90  Mth::Vector seg_end;
91  Mth::Vector seg_vec;
92  float seg_length;
93 
94  int num_points_left=m_num_path_points;
95  float dist_remaining=m_distance;
96 
97  seg_start=m_leader_pos;
98  int i=m_most_recent_path_point;
99  seg_end=mp_path_points[i];
100  num_points_left-=1;
101 
102 
103  while (true)
104  {
105  seg_vec=seg_end-seg_start;
106  seg_length=seg_vec.Length();
107  if (dist_remaining > seg_length)
108  {
109  dist_remaining-=seg_length;
110  seg_start=seg_end;
111  if (!num_points_left)
112  {
113  break;
114  }
115  --i;
116  if (i < 0)
117  {
118  i=MAX_FOLLOWOB_PATH_POINTS-1;
119  }
120  seg_end=mp_path_points[i];
121  --num_points_left;
122  }
123  else
124  {
125  break;
126  }
127  }
128 
129  if (seg_length>0.0f)
130  {
131  Mth::Vector seg_dir=seg_vec/-seg_length;
132 
133  *p_pos=seg_start-seg_dir*dist_remaining;
134 
135  if (m_orient_y)
136  {
137  // Instead of setting z to seg_dir directly, use some of the previous value
138  // so that the orientation gradually moves towards the new dir rather than snapping.
139  (*p_orientation)[Z]=10.0f*(*p_orientation)[Z]+seg_dir;
140  (*p_orientation)[Z].Normalize();
141 
142  Mth::Vector Up(0.0f,1.0f,0.0f);
143  (*p_orientation)[X]=Mth::CrossProduct(Up,(*p_orientation)[Z]);
144  (*p_orientation)[X].Normalize();
145  (*p_orientation)[Y]=Mth::CrossProduct((*p_orientation)[Z],(*p_orientation)[X]);
146  }
147  else
148  {
149  (*p_orientation)[Z]=10.0f*(*p_orientation)[Z]+seg_dir;
150  (*p_orientation)[Z][Y]=0;
151  (*p_orientation)[Z].Normalize();
152  (*p_orientation)[Y].Set(0.0f,1.0f,0.0f);
153  (*p_orientation)[X]=Mth::CrossProduct((*p_orientation)[Y],(*p_orientation)[Z]);
154  }
155  }
156 }
