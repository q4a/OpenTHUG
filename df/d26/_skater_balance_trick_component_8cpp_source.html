105 CBaseComponent::EMemberFunctionResult CSkaterBalanceTrickComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
106 {
107  switch ( Checksum )
108  {
109  // @script | SetWobbleDetails | gets called from the animation
110  // component's PlayAnim handler (needs to come after the
111  // animation has launched...)
112  case CRCC(0xea6d0efd,"SetWobbleDetails"):
113  {
114  // Intercepts the animation component's PlayAnim member function and handle's skater specific logic
115 
116  // If this is a wobbling balance anim, then allow the manual or grind to programmatically wobble it.
117  if (pParams->ContainsFlag(CRCD(0x6d941203, "Wobble")))
118  {
119  mManual.EnableWobble();
120  mGrind.EnableWobble();
121  mLip.EnableWobble();
122  mSkitch.EnableWobble();
123  }
124 
125  Script::CStruct* p_wobble_params = Script::GetStructure(CRCD(0x5adb96d4, "DefaultWobbleParams"), Script::ASSERT);
126  pParams->GetStructure(CRCD(0xd095379c, "WobbleParams"), &p_wobble_params);
127  set_wobble_params(p_wobble_params);
128 
129  return CBaseComponent::MF_TRUE;
130  }
131 
132  // @script | DoBalanceTrick |
133  // @parmopt int | Tweak | 0 |
134  // @parm name | ButtonA |
135  // @parm name | ButtonB |
136  // @parmopt name | Type | 0 | balance trick type (NoseManual,
137  // Manual, Grind, Slide, Lip)
138  // @flag DoFlipCheck | If set, the range anim will be played backwards if the skater is flipped.
139  // @flag PlayRangeAnimBackwards | If set, the range anim will be played backwards. Can be
140  // used in conjunction with DoFlipCheck. Ie, if both flags are set, then the anim will be
141  // played forwards if the skater is flipped.
142  // @flag ClearCheese | If set this will reset any previous cheese.
143  // @parmopt structure | BalanceParams | | If not specified then the balance trick will use
144  // the params defined by the script globals LipParams, or ManualParams etc depending on the
145  // Type value.
146  case CRCC(0x7e92670c, "DoBalanceTrick"):
147  {
148  if (mpBalanceParams)
149  {
150  delete mpBalanceParams;
151  mpBalanceParams = NULL;
152  }
153  Script::CStruct* p_balance_params = NULL;
154  if (pParams->GetStructure(CRCD(0x647facc2, "BalanceParams"), &p_balance_params))
155  {
156  set_balance_trick_params(p_balance_params);
157  }
158 
159  int Tweak = 0;
160  pParams->GetInteger(CRCD(0x2bfd3d35, "Tweak"), &Tweak);
161 
162  uint32 ButtonA = 0;
163  uint32 ButtonB = 0;
164  pParams->GetChecksum(CRCD(0x4f1a0b01, "ButtonA"), &ButtonA, Script::ASSERT);
165  pParams->GetChecksum(CRCD(0xd6135abb, "ButtonB"), &ButtonB, Script::ASSERT);
166 
167  uint32 NewBalanceTrickType = 0;
168  pParams->GetChecksum(CRCD(0x7321a8d6, "Type"), &NewBalanceTrickType);
169 
170  // Do nothing if already doing the same type of balance trick. This is so that the meter does not glitch when switching to
171  // a balance trick from another balance trick where there is no StopBalanceTrick script command in between.
172  if (mBalanceTrickType == NewBalanceTrickType) break;
173  mBalanceTrickType = NewBalanceTrickType;
174 
175  bool DoFlipCheck = pParams->ContainsFlag(CRCD(0x1304e677, "DoFlipCheck"));
176  bool PlayRangeAnimBackwards = pParams->ContainsFlag(CRCD(0x8fe31ed, "PlayRangeAnimBackwards"));
177 
178  // This needed because for the extra grind tricks (eg left triangle triangle) the
179  // DoBalanceTrick command gets called twice in a row, once for the first time the grind is detected
180  // by the first triangle, then again once the extra trick is detected.
181  // In that case the cheese needs to be reset otherwise it thinks you've jumped and re-railed & you
182  // fall off straight away.
183  bool clear_cheese=pParams->ContainsFlag(CRCD(0xbbdb0f78,"ClearCheese"));
184 
185  switch (mBalanceTrickType)
186  {
187  case 0:
188  Dbg_MsgAssert(false, ("\n%s\nMust specify a type in DoBalanceTrick, eg Type=NoseManual", pScript->GetScriptInfo()));
189  break;
190 
191  case CRCC(0xac90769, "NoseManual"):
192  case CRCC(0xef24413b, "Manual"):
193  if (clear_cheese)
194  {
195  mManual.Reset();
196  }
197  mManual.UpdateRecord();
198  mManual.SetUp(ButtonA, ButtonB, Tweak, DoFlipCheck, PlayRangeAnimBackwards);
199  break;
200 
201  case CRCC(0x255ed86f, "Grind"):
202  case CRCC(0x8d10119d, "Slide"):
203  if (clear_cheese)
204  {
205  mGrind.Reset();
206  }
207  mGrind.UpdateRecord();
208  mGrind.SetUp(ButtonA, ButtonB, Tweak, DoFlipCheck, PlayRangeAnimBackwards);
209  break;
210 
211  case CRCC(0xa549b57b, "Lip"):
212  if (clear_cheese)
213  {
214  mLip.Reset();
215  }
216  mLip.UpdateRecord();
217  mLip.SetUp(ButtonA, ButtonB, Tweak, DoFlipCheck, PlayRangeAnimBackwards);
218  break;
219 
220  case CRCC(0x3506ce64, "Skitch"):
221  if (clear_cheese)
222  {
223  mSkitch.Reset();
224  }
225  mSkitch.UpdateRecord();
226  mSkitch.SetUp(ButtonA, ButtonB, Tweak, DoFlipCheck, PlayRangeAnimBackwards);
227  break;
228 
229  default:
230  Dbg_Assert(false);
231  break;
232  }
233  break;
234  }
235 
236  // @script | AdjustBalance | Make minor adjustments, nudges, to the current balance trick, if any
237  // @parmopt float | TimeAdd | 0.0 |
238  // @parmopt float | LeanAdd | 0.0 |
239  // @parmopt float | SpeedAdd | 0.0 |
240  // @parmopt float | TimeMult | 1.0 |
241  // @parmopt float | LeanMult | 1.0 |
242  // @parmopt float | SpeedMult | 1.0 |
243  case CRCC(0x7a0dcd4b, "AdjustBalance"):
244  {
245  if (mBalanceTrickType)
246  {
247  float TimeMult = 1.0f;
248  float LeanMult = 1.0f;
249  float SpeedMult = 1.0f;
250  float TimeAdd = 0.0f;
251  float LeanAdd = 0.0f;
252  float SpeedAdd = 0.0f;
253  pParams->GetFloat(CRCD(0x3a4e9413, "TimeAdd"), &TimeAdd);
254  pParams->GetFloat(CRCD(0xb4cecc1, "LeanAdd"), &LeanAdd);
255  pParams->GetFloat(CRCD(0xeb9ecabb, "SpeedAdd"), &SpeedAdd);
256  pParams->GetFloat(CRCD(0x4c074698, "TimeMult"), &TimeMult);
257  pParams->GetFloat(CRCD(0x24ebf718, "LeanMult"), &LeanMult);
258  pParams->GetFloat(CRCD(0x94dbbd1c, "SpeedMult"), &SpeedMult);
259 
260  // Get whichever balance trick we are doing
261  CManual* pBalance;
262  switch (mBalanceTrickType)
263  {
264  case CRCC(0x255ed86f, "Grind"):
265  case CRCC(0x8d10119d, "Slide"):
266  pBalance = &mGrind;
267  break;
268 
269  case CRCC(0xa549b57b, "Lip"):
270  pBalance = &mLip;
271  break;
272 
273  case CRCC(0x3506ce64, "Skitch"):
274  pBalance = &mSkitch;
275  break;
276 
277  case CRCC(0xac90769, "NoseManual"):
278  case CRCC(0xef24413b, "Manual"):
279  pBalance = &mManual;
280  break;
281 
282  default:
283  pBalance = NULL;
284  Dbg_Assert(false);
285  }
286 
287  pBalance->mManualTime *= TimeMult;
288  pBalance->mManualTime += TimeAdd;
289  if (pBalance->mManualTime < 0.0f)
290  {
291  pBalance->mManualTime = 0.0f;
292  }
293  pBalance->mManualLean *= LeanMult;
294  pBalance->mManualLean += LeanAdd * Mth::Sgn(pBalance->mManualLean);
295  pBalance->mManualLeanDir *= SpeedMult;
296  pBalance->mManualLeanDir += SpeedAdd * Mth::Sgn(pBalance->mManualLeanDir);
297  }
298  break;
299  }
300 
301  // @script | StopBalanceTrick |
302  case CRCC(0xe553a5b8, "StopBalanceTrick"):
303  stop_balance_trick();
304  break;
305 
306  // @script | StartBalanceTrick |
307  case CRCC(0x6cc475b7, "StartBalanceTrick"):
308  mDoingBalanceTrick = true;
309  break;
310 
311  case CRCC(0x78e669ab, "SwitchOffBalanceMeter"):
312  {
313  CSkaterScoreComponent* p_score_component = GetSkaterScoreComponentFromObject(GetObject());
314  Dbg_Assert(p_score_component);
315 
316  p_score_component->GetScore()->SetBalanceMeter(false);
317  p_score_component->GetScore()->SetManualMeter(false);
318 
319  mManual.SwitchOffMeters();
320  mGrind.SwitchOffMeters();
321  mLip.SwitchOffMeters();
322  mSkitch.SwitchOffMeters();
323  break;
324  }
325 
326  // @script | SwitchOnBalanceMeter |
327  case CRCC(0x73727701, "SwitchOnBalanceMeter"):
328  mManual.SwitchOnMeters();
329  mGrind.SwitchOnMeters();
330  mLip.SwitchOnMeters();
331  mSkitch.SwitchOnMeters();
332  break;
333 
334  default:
335  return CBaseComponent::MF_NOT_EXECUTED;
336  }
337  return CBaseComponent::MF_TRUE;
338 }

404 void CSkaterBalanceTrickComponent::ExcludeBalanceButtons ( int& numButtonsToIgnore, uint32 pButtonsToIgnore[] )
405 {
406  // If a balance trick is being done, then this will stop the buttons being used to control the balance from
407  // generating the events used to trigger tricks.
408  // Otherwise, ChrisR can't do kickflips by very quickly jumping out of a grind & kickflipping by rolling
409  // from X to Square.
410 
411  // Note, in case of problems later:
412  // Now, potentially a button could be pressed just before the balance trick,
413  // then released during the balance trick, and the release would not get recorded as an event.
414  // So if the event buffer were analysed, it would appear that the button was still pressed.
415  // However, this does not seem to cause any problems, because generally if we want to know if a button
416  // is pressed right now we just read the pad. Just noting this if any problems occur later.
417  switch (mBalanceTrickType)
418  {
419  case CRCC(0xac90769, "NoseManual"):
420  case CRCC(0xef24413b, "Manual"):
421  numButtonsToIgnore = 2;
422  pButtonsToIgnore[0] = mManual.mButtonAChecksum;
423  pButtonsToIgnore[1] = mManual.mButtonBChecksum;
424  break;
425 
426  case CRCC(0xa549b57b, "Lip"):
427  numButtonsToIgnore = 2;
428  pButtonsToIgnore[0] = mLip.mButtonAChecksum;
429  pButtonsToIgnore[1] = mLip.mButtonBChecksum;
430  break;
431 
432  case CRCC(0x255ed86f, "Grind"):
433  case CRCC(0x8d10119d, "Slide"):
434  numButtonsToIgnore = 2;
435  pButtonsToIgnore[0] = mGrind.mButtonAChecksum;
436  pButtonsToIgnore[1] = mGrind.mButtonBChecksum;
437  break;
438 
439  case CRCC(0x3506ce64, "Skitch"):
440  numButtonsToIgnore = 2;
441  pButtonsToIgnore[0] = mSkitch.mButtonAChecksum;
442  pButtonsToIgnore[1] = mSkitch.mButtonBChecksum;
443  break;
444 
445  default:
446  numButtonsToIgnore = 0;
447  break;
448  }
449 }
