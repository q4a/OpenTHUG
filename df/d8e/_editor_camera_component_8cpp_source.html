48 static void s_check_if_kill_poly ( CFeeler* p_feeler )
49 {
50  if (!p_feeler->GetTrigger() || !p_feeler->GetNodeChecksum())
51  {
52  #ifdef DEBUG_COLLISION
53  //printf("Not got trigger or not got node checksum\n");
54  #endif
55  return;
56  }
57 
58  int node = SkateScript::FindNamedNode(p_feeler->GetNodeChecksum(), false);// false means don't assert
59  if (node < 0)
60  {
61  // Got a trigger, but no node, so assume it is a kill poly. This case arises in the park editor
62  // where a kill piece such as water or lava has been placed but the node array has not been generated yet.
63  //sHitKillPoly=true;
64  // Note: Commented out, to fix TT8832, where the cursor could get stuck on cars.
65  // The PE now allows the cursor to go anywhere, even onto kill polys.
66  return;
67  }
68 
69  Script::CArray *p_node_array = Script::GetArray(CRCD(0xc472ecc5, "NodeArray"));
70  Script::CStruct *p_node = p_node_array->GetStructure(node);
71 
72  uint32 trigger_script=0;
73  p_node->GetChecksum(CRCD(0x2ca8a299,"TriggerScript"),&trigger_script);
74 
75  if (trigger_script)
76  {
77  //printf("trigger_script=%s\n",Script::FindChecksumName(trigger_script));
78  if (Script::ScriptContainsAnyOfTheNames(trigger_script,spKillScripts,sizeof(spKillScripts)/sizeof(uint32)))
79  {
80  sHitKillPoly=true;
81  }
82  }
83 }

85 // Checks whether the vector posA -> posB goes through any Kill polys.
86 // Does not check collidable polys.
87 static bool s_goes_through_kill_poly(const Mth::Vector &posA, const Mth::Vector &posB)
88 {
89  sHitKillPoly=false;
90 
91  CFeeler feeler;
92 
93  feeler.m_start = posA;
94  feeler.m_end = posB;
95 
96  feeler.SetIgnore(0, mFD_NON_COLLIDABLE | mFD_TRIGGER);
97  feeler.SetCallback(s_check_if_kill_poly);
98  //feeler.DebugLine(0,255,0);
99  feeler.GetCollision();
100 
101  return sHitKillPoly;
102 }

104 bool CEditorCameraComponent::find_y(Mth::Vector start, Mth::Vector end, float *p_y)
105 {
106  #ifdef DEBUG_COLLISION
107  //printf("s_find_y: ");
108  #endif
109 
110  CFeeler feeler;
111  feeler.SetStart(start);
112  feeler.SetEnd(end);
113  // Ignore invisible polys, otherwise one can get stuck on top of the hotel in Hawaii,
114  // because it is surrounded by horizontal invisible polys.
115  //feeler.SetIgnore(mFD_INVISIBLE, 0);
116 
117  if (feeler.GetCollision())
118  {
119  if (m_simple_collision)
120  {
121  #ifdef DEBUG_COLLISION
122  printf("Simple drop down\n");
123  #endif
124 
125  sHitKillPoly=false;
126  Mth::Vector point=feeler.GetPoint();
127  *p_y=point[Y];
128  return true;
129  }
130 
131  sHitKillPoly=false;
132  s_check_if_kill_poly(&feeler);
133  if (sHitKillPoly)
134  {
135  // Can't go onto collidable Kill polys
136  #ifdef DEBUG_COLLISION
137  printf("Kill poly!\n");
138  #endif
139  return false;
140  }
141 
142  // Check the steepness
143  Mth::Vector normal=feeler.GetNormal();
144  // .087=max steepness of 85 degrees
145  if (normal[Y] < 0.087f/*sinf(Mth::DegToRad(90.0f-Script::GetFloat(CRCD(0x87b926b4,"MaxSteepness"))))*/)
146  {
147  // The ground is too steep!
148  // Need this check otherwise the skater can be placed halfway down the almost vertical
149  // riverbank in NJ
150  #ifdef DEBUG_COLLISION
151  printf("Too steep!\n");
152  #endif
153  return false;
154  }
155 
156  #ifdef DEBUG_COLLISION
157  //printf("Found y=%f normal[Y]=%f\n",feeler.GetPoint()[Y],normal[Y]);
158  #endif
159 
160  Mth::Vector point=feeler.GetPoint();
161  *p_y=point[Y];
162 
163  // But, before returning true, check whether there are any kill polys between the start and the
164  // collision point.
165  if (s_goes_through_kill_poly(start, point))
166  {
167  #ifdef DEBUG_COLLISION
168  printf("Kill poly above found collision!\n");
169  #endif
170  return false;
171  }
172 
173  return true;
174  }
175 
176  #ifdef DEBUG_COLLISION
177  printf("No collision\n");
178  #endif
179 
180  return false;
181 }

183 bool CEditorCameraComponent::find_collision_at_offset(Mth::Vector oldCursorPos, float upOffset)
184 {
185  Mth::Vector start=m_cursor_pos;
186  start[Y]+=upOffset;
187  Mth::Vector end=m_cursor_pos;
188  end[Y]+=Script::GetFloat(CRCD(0xf230d521,"EditorCam_CursorCollisionDownOffset"));
189 
190  bool new_position_is_ok=false;
191 
192  float y;
193  if (find_y(start,end,&y))
194  {
195  new_position_is_ok=true;
196  }
197 
198  if (new_position_is_ok)
199  {
200  Mth::Vector from = oldCursorPos;
201  Mth::Vector to = m_cursor_pos;
202  to[Y]=y;
203  to[Y]+=72.0f; // 6 feet above the ground
204  // Make it really be horizontal to avoid steep collision check weirdness
205  from[Y]=to[Y];
206 
207  if (!m_simple_collision)
208  {
209  // Do a horizontal collision check to see if the cursor
210  // is going to move through a kill poly
211  if (s_goes_through_kill_poly(from, to))
212  {
213  #ifdef DEBUG_COLLISION
214  printf("Goes through kill poly\n");
215  #endif
216  new_position_is_ok=false;
217  }
218  }
219 
220  if (!m_allow_movement_through_walls)
221  {
222  // Also don't allow moving through walls
223  CFeeler feeler;
224  feeler.SetIgnore(mFD_NON_COLLIDABLE, 0);
225  feeler.SetStart(from);
226  feeler.SetEnd(to);
227  if (feeler.GetCollision())
228  {
229  #ifdef DEBUG_COLLISION
230  printf("oof\n");
231  #endif
232  new_position_is_ok=false;
233  }
234  }
235  }
236 
237  if (new_position_is_ok)
238  {
239  m_cursor_pos[Y]=y;
240  return true;
241  }
242  return false;
243 }

426 // The component's Update() function is called from the CCompositeObject's
427 // Update() function. That is called every game frame by the CCompositeObjectManager
428 // from the s_logic_code function that the CCompositeObjectManager registers
429 // with the task manger.
430 void CEditorCameraComponent::Update()
431 {
432  Dbg_MsgAssert(mp_input_component,("NULL mp_input_component"));
433 
434  // TODO: When in park editor, use different values here, cos they'll need to be
435  // tweaked according to the shell geometry.
436  m_tilt_angle_min=Script::GetFloat(CRCD(0x7d16d113,"EditorCam_TiltMin"));
437  m_tilt_angle_max=Script::GetFloat(CRCD(0x411bee4a,"EditorCam_TiltMax"));
438 
439  CControlPad& control_pad = mp_input_component->GetControlPad();
440 
441 
442  // Update m_angle
443  float angle_vel=Script::GetFloat(CRCD(0x8242d878,"EditorCam_TurnSpeed"))*control_pad.m_scaled_rightX;
444  m_angle+=angle_vel;
445  while (m_angle > 2*3.141592654f)
446  {
447  m_angle -= 2*3.141592654f;
448  }
449  while (m_angle < 0.0f)
450  {
451  m_angle += 2*3.141592654f;
452  }
453 
454  // Update m_tilt_angle
455  angle_vel=-Script::GetFloat(CRCD(0x3cfa3a72,"EditorCam_TiltSpeed"))*control_pad.m_scaled_rightY;
456  m_tilt_angle+=angle_vel;
457  if (m_tilt_angle > m_tilt_angle_max)
458  {
459  m_tilt_angle = m_tilt_angle_max;
460  }
461  if (m_tilt_angle < m_tilt_angle_min)
462  {
463  m_tilt_angle = m_tilt_angle_min;
464  }
465 
466  // Update m_radius
467  bool zoom_in_button_pressed=false;
468  bool zoom_out_button_pressed=false;
469  bool raise_pressed=false;
470  bool lower_pressed=false;
471  switch (Config::GetHardware())
472  {
473  case Config::HARDWARE_XBOX:
474  {
475  uint32 input_mask = mp_input_component->GetInputMask();
476  if (input_mask & Inp::Data::mA_BLACK)
477  {
478  zoom_out_button_pressed=true;
479  }
480  if (input_mask & Inp::Data::mA_WHITE)
481  {
482  zoom_in_button_pressed=true;
483  }
484  // Note: See p_siodev.cpp for the low level code that does the weird XBox buttons mappings
485  if (Ed::CParkEditor::Instance()->EditingCustomPark())
486  {
487  // When editing a park, XBox L gives L2, XBox R gives L1
488  // We want XBox L to lower, XBox R to raise
489  if (input_mask & Inp::Data::mA_L1)
490  {
491  raise_pressed=true;
492  }
493  if (input_mask & Inp::Data::mA_L2)
494  {
495  lower_pressed=true;
496  }
497  }
498  else
499  {
500  // When not editing a park, XBox L gives L1, XBox R gives R1
501  // We want XBox L to lower, XBox R to raise
502  if (input_mask & Inp::Data::mA_R1)
503  {
504  raise_pressed=true;
505  }
506  if (input_mask & Inp::Data::mA_L1)
507  {
508  lower_pressed=true;
509  }
510  }
511  break;
512  }
513  case Config::HARDWARE_NGC:
514  {
515  uint32 input_mask = mp_input_component->GetInputMask();
516  if (input_mask & Inp::Data::mA_Z)
517  {
518  if (input_mask & Inp::Data::mA_R1)
519  {
520  zoom_out_button_pressed=true;
521  }
522  if (input_mask & Inp::Data::mA_L1)
523  {
524  zoom_in_button_pressed=true;
525  }
526  }
527  else
528  {
529  if (input_mask & Inp::Data::mA_R1)
530  {
531  lower_pressed=true;
532  }
533  if (input_mask & Inp::Data::mA_L1)
534  {
535  raise_pressed=true;
536  }
537  }
538  break;
539  }
540  default:
541  {
542  zoom_out_button_pressed=control_pad.m_R1.GetPressed();
543  zoom_in_button_pressed=control_pad.m_R2.GetPressed();
544  raise_pressed=control_pad.m_L1.GetPressed();
545  lower_pressed=control_pad.m_L2.GetPressed();
546  break;
547  }
548  }
549 
550  float radius_vel=Script::GetFloat(CRCD(0x4e58f829,"EditorCam_InOutSpeed"));
551  if (zoom_out_button_pressed)
552  {
553  }
554  else if (zoom_in_button_pressed)
555  {
556  radius_vel=-radius_vel;
557  }
558  else
559  {
560  radius_vel=0.0f;
561  }
562 
563  // Make the speed proportional to the current radius so that the camera can be zoomed
564  // out to the max nice and quick.
565  radius_vel*=m_radius;
566 
567  m_radius+=radius_vel;
568  if (m_radius < m_radius_min)
569  {
570  m_radius = m_radius_min;
571  }
572  if (m_radius > m_radius_max)
573  {
574  m_radius = m_radius_max;
575  }
576  float zoom_factor=(m_radius-m_radius_min)/(m_radius_max-m_radius_min);
577  // zoom_factor is between 0 and 1 and indicates how far out the camera is zoomed.
578  // 0 = closest to cursor, 1 = furthest away.
579 
580 
581  // Update the height
582  float height_vel_min=Script::GetFloat(CRCD(0x604961f4,"EditorCam_UpDownSpeedMin"));
583  float height_vel_max=Script::GetFloat(CRCD(0x5c445ead,"EditorCam_UpDownSpeedMax"));
584  float height_vel=height_vel_min + zoom_factor * (height_vel_max-height_vel_min);
585 
586  // If very close in allow very fine movements, needed for rail placement in rail editor.
587  // TODO: Fix this so that the change in speed is not so sudden, make it linear like above.
588  if (m_radius < 100)
589  {
590  height_vel=0.2f;
591  }
592 
593  if (raise_pressed)
594  {
595  }
596  else if (lower_pressed)
597  {
598  height_vel=-height_vel;
599  }
600  else
601  {
602  height_vel=0.0f;
603  }
604 
605  m_cursor_height+=height_vel;
606  float max_height=Script::GetFloat(CRCD(0x61d56fa0,"EditorCam_MaxHeight"));
607  if (m_cursor_height > max_height)
608  {
609  m_cursor_height=max_height;
610  }
611 
612  // Switch off collision on any created-rail sectors whilst doing collision checks.
613  // This is because the rail sectors have a vertical collidable poly, which occasionally does cause
614  // collisions, making the cursor ping up into the air. Must be some sort of floating point innaccuracy
615  // (or too much accuracy), probably because the collision check vector is straight down the vertical poly.
616  Mdl::Skate * p_skate_mod = Mdl::Skate::Instance();
617  if (p_skate_mod->m_cur_level == CRCD(0xe8b4b836,"load_sk5ed"))
618  {
619  Obj::GetRailEditor()->SetSectorActiveStatus(false);
620  }
621 
622  // Make sure that the cursor cannot be raised through a collidable poly, otherwise it
623  // can be raised through the stadium roof in slam city jam (TT3142)
624  CFeeler feeler;
625  feeler.SetStart(m_cursor_pos);
626  feeler.SetEnd(m_cursor_pos + Mth::Vector(0.0f, m_cursor_height+10.0f, 0.0f));
627  if (feeler.GetCollision())
628  {
629  m_cursor_height=feeler.GetPoint()[Y]-m_cursor_pos[Y]-10.0f;
630  }
631 
632  // This allows the height to be less than m_min_height, but as soon as it goes
633  // over m_min_height it won't be able to go below it again.
634  // This is for the rail editor, where the min height needs to be limited whilst
635  // still allowing snapping to a vertex.
636  if (height_vel < 0.0f && m_cursor_height-height_vel+0.01 >= m_min_height)
637  {
638  if (m_cursor_height < m_min_height)
639  {
640  m_cursor_height=m_min_height;
641  }
642  }
643  if (m_cursor_height < 0.0f)
644  {
645  m_cursor_height=0.0f;
646  }
647 
648  // Update m_cursor_pos ...
649 
650  // First move it horizontally according to the left analogue stick
651  m_cursor_u.Set(cosf(m_angle),0.0f,-sinf(m_angle));
652  m_cursor_v.Set(sinf(m_angle),0.0f,cosf(m_angle));
653 
654  float speed_min=Script::GetFloat(CRCD(0x83bde786,"EditorCam_MoveSpeedMin"));
655  float speed_max=Script::GetFloat(CRCD(0xbfb0d8df,"EditorCam_MoveSpeedMax"));
656  // Make the speed proportional to the zoom factor so that fine movements can be done
657  // when zoomed in, whilst fast movements can be made when zoomed out.
658  float speed = speed_min + zoom_factor * (speed_max-speed_min);
659 
660  // If very close in allow very fine movements, needed for rail placement in rail editor.
661  // TODO: Fix this so that the change in speed is not so sudden, make it linear like above.
662  if (m_radius < 100)
663  {
664  speed=1.0f;
665  }
666 
667  float horiz=control_pad.m_leftX / 128.0f;
668  float vert=control_pad.m_leftY / 128.0f;
669  if (fabs(horiz) < 0.01f)
670  {
671  if (control_pad.m_left.GetPressed())
672  {
673  horiz=-0.5f;
674  }
675  if (control_pad.m_right.GetPressed())
676  {
677  horiz=0.5f;
678  }
679  }
680 
681  if (fabs(vert) < 0.01f)
682  {
683  if (control_pad.m_up.GetPressed())
684  {
685  vert=-0.5f;
686  }
687  if (control_pad.m_down.GetPressed())
688  {
689  vert=0.5f;
690  }
691  }
692 
693  Mth::Vector vel = m_cursor_u * horiz*speed+

841  Mth::Vector pos;

843  m_cursor_pos=pos;
844  m_cursor_height=0.0f;
845 
846  // Call update to do the ground collision check.
847  Update();
848  // That might have changed m_cursor_pos[Y], so adjust the height so that the cursor does

852  // be possible to snap to rail vertices, because rails are uncollidable, so on the next frame the cursor

854  break;

859  }

862  // unrecognized functions; if we make it down here,
863  // that means that the component already handled it
