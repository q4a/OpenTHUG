76  * Description: *

80 void NsPrim::point ( float x, float y, float z, GXColor color )
81 {
82 // assert ( inDrawingContext );
83 
84  // Set current vertex descriptor to enable position and color0.
85  // Both use 8b index to access their data arrays.
86  GX::SetVtxDesc( 1, GX_VA_POS, GX_DIRECT );
87 
88  // Set material color.
89  GX::SetChanMatColor( GX_COLOR0A0, color );
90  GX::SetChanAmbColor( GX_COLOR0A0, (GXColor){128,128,128,255} );
91 
92  GX::SetChanCtrl( GX_COLOR0A0, GX_ENABLE, GX_SRC_REG, GX_SRC_REG, GX_LIGHT_NULL, GX_DF_NONE, GX_AF_NONE );
93 
94  // Send coordinates.
95  GX::Begin( GX_POINTS, GX_VTXFMT0, 1 );
96  GX::Position3f32(x, y, z);
97  GX::End();
98 }
