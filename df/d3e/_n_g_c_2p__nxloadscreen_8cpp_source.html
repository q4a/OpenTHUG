250 void CLoadScreen::s_plat_display(const char* filename, bool just_freeze, bool blank)
251 {
252  // See if we need to just flag that a loading screen is active to disable draws.
253  if ( just_freeze )
254  {
255  gLoadingScreenActive = true;
256  return;
257  }
258 
259  // See if we need to clear the screen to black.
260  if ( blank )
261  {
262  for ( int lp = 0; lp < 2; lp++ )
263  {
264  NsDisplay::begin();
265  NsRender::begin();
266  NsRender::end();
267  NsDisplay::end();
268  }
269 
270  gLoadingScreenActive = true;
271  return;
272  }
273 
274  NxNgc::EngineGlobals.gpuBusy = true;
275 
276  char name[256];
277  char pre[256];
278 
279  File::PreMgr* pre_mgr = File::PreMgr::Instance();
280 
281  // Hmm seems we need to prepend 'images\\' to the start of this string...
282  strcpy( name, "images\\" );
283  switch( Config::GetLanguage())
284  {
285  case Config::LANGUAGE_GERMAN:
286  strcpy( pre, filename );
287  strcat( pre, ".prd" );
288  pre_mgr->LoadPre( pre, false );
289  strcat( name, "grimages_ngc\\" );
290  break;
291  case Config::LANGUAGE_FRENCH:
292  strcpy( pre, filename );
293  strcat( pre, ".prf" );
294  pre_mgr->LoadPre( pre, false );
295  strcat( name, "frimages_ngc\\" );
296  break;
297  default:
298  break;
299  }
300  strcat( name, filename );
301 
302  // ...and append '.img.ngc' to the end.
303  strcat( name, ".img.ngc" );
304 
305  // Load the texture a few lines at a time & render to the display.
306  gLoadingLoadScreen = true;
307  void *p_FH = File::Open( name, "rb" );
308 
309  if( p_FH )
310  {
311  uint32 version;
312  uint32 checksum;
313  uint32 width;
314  uint32 height;
315  uint32 depth;
316  uint32 levels;
317  uint32 rwidth;
318  uint32 rheight;
319  uint16 alphamap;
320  uint16 hasholes;
321 
322  // Load the actual texture
323 
324  File::Read( &version, 4, 1, p_FH );
325  File::Read( &checksum, 4, 1, p_FH );
326  File::Read( &width, 4, 1, p_FH );
327  File::Read( &height, 4, 1, p_FH );
328  File::Read( &depth, 4, 1, p_FH );
329  File::Read( &levels, 4, 1, p_FH );
330  File::Read( &rwidth, 4, 1, p_FH );
331  File::Read( &rheight, 4, 1, p_FH );
332  File::Read( &alphamap, 2, 1, p_FH );
333  File::Read( &hasholes, 2, 1, p_FH );
334 
335 // Dbg_MsgAssert( version <= TEXTURE_VERSION, ("Illegal texture version number: %d (The newest version we deal with is: %d)\n", version, TEXTURE_VERSION ));
336  Dbg_MsgAssert( depth == 32, ("We only deal with 32-bit textures.\n"));
337 
338 #define LINES 16
339 
340  // Load in 8 line chunks.
341 // uint16 temp_buffer[width*LINES*2];
342 // uint32 tex_buffer[width*LINES];
343 
344  uint16 * temp_buffer = (uint16 *)g_p_buffer;
345  uint32 * tex_buffer = (uint32 *)&temp_buffer[width*LINES*2];
346 
347  NsDisplay::begin();
348  NsRender::begin();
349 
350  GX::PokeColorUpdate( GX_TRUE );
351  GX::PokeAlphaUpdate( GX_FALSE );
352  uint yy = 0;
353  while ( yy < height )
354  {
355  int lines = ( height - yy ) > LINES ? LINES : ( height - yy );
356 
357  File::Read( temp_buffer, width * 4 * lines, 1, p_FH );
358 
359  if ( gReload )
360  {
361  gReload = false;
362  yy = 0;
363  File::Close( p_FH );
364  void *p_FH = File::Open( name, "rb" );
365  File::Read( &version, 4, 1, p_FH );
366  File::Read( &checksum, 4, 1, p_FH );
367  File::Read( &width, 4, 1, p_FH );
368  File::Read( &height, 4, 1, p_FH );
369  File::Read( &depth, 4, 1, p_FH );
370  File::Read( &levels, 4, 1, p_FH );
371  File::Read( &rwidth, 4, 1, p_FH );
372  File::Read( &rheight, 4, 1, p_FH );
373  File::Read( &alphamap, 2, 1, p_FH );
374  File::Read( &hasholes, 2, 1, p_FH );
375  NsDisplay::begin();
376  NsRender::begin();
377  continue;
378  }
379 
380  // Need to swizzle data we just read.
381  for ( int lp = 0; lp < lines; lp += 4 )
382  {
383  int xx = lp * width;
384  for ( uint lp2 = 0; lp2 < width; lp2 += 4 )
385  {
386  for ( int lp3 = 0; lp3 < 16; lp3++ )
387  {
388  uint16 p0 = temp_buffer[lp3+(lp2*8)+(xx*2)];
389  uint16 p1 = temp_buffer[16+lp3+(lp2*8)+(xx*2)];
390  tex_buffer[(lp3&3)+(width*(lp3>>2))+lp2+xx] = ( ( p0 << 16 ) & 0xff000000 ) |
391  ( ( p0 << 16 ) & 0x00ff0000 ) |
392  ( ( p1 << 0 ) & 0x0000ff00 ) |
393  ( ( p1 << 0 ) & 0x000000ff );
394  }
395  }
396  }
397 
398  for ( int yp = 0; yp < lines; yp++ )
399  {
400  uint y_pos = ( ( height - 1 ) - ( yy + yp ) );
401  for ( uint xp = 0; xp < width; xp++ )
402  {
403  GX::PokeARGB( xp, y_pos, tex_buffer[(width*yp)+xp] );
404  }
405  }
406 
407  yy += lines;
408  }
409  NsRender::end();
410  NsDisplay::end(false);
411 
412  File::Close( p_FH );
413  }
414  gLoadingLoadScreen = false;
415 
416  switch( Config::GetLanguage())
417  {
418  case Config::LANGUAGE_GERMAN:
419  case Config::LANGUAGE_FRENCH:
420  pre_mgr->UnloadPre( pre );
421  break;
422  default:
423  break;
424  }
425 // NsDisplay::flush();
426 
427  NxNgc::EngineGlobals.gpuBusy = false;
428 
429  gLoadingScreenActive = true;
430 }
