64 CSuspendComponent::CSuspendComponent() : CBaseComponent()
65 {
66  SetType( CRC_SUSPEND );
67 
68  m_suspend_distance=Script::GetFloat(CRCD(0xad501995,"DefaultMovingObjectSuspendDistance")) * 12.0f;
69  m_suspend_distance_squared=m_suspend_distance*m_suspend_distance;
70  m_never_suspend=true;
71  m_no_suspend_count = 3; // default to not allowing suspensions in the first 3 frames
72 
73  // Mick: We don't want to display things before they have done at least one frame of
74  // animation, because they start out in their default position (the "Blair Witch" position)
75  // so, we set a default number of animations that we MUST do before we are allowed to
76  // not animate.
77  // This also overrides the SkipLogic test, so we do a frame of logic
78  // which fixes problems where something is moving to the first node of a path
79 
80  // Since the first couple of frames of the game are hidden behind the loading screen
81  // we don't need to worry about framerate glitched due to the large
82  // number of objects being created at start.
83 
84  // I Initially had this at 1, which seemed fine
85  // however, when you change levels, it needs to be at 3
86  // I'm not sure why - it might be good to look into this later.
87  m_initial_animations = 3;

104 void CSuspendComponent::InitFromStructure( Script::CStruct* pParams )
105 {
106  // anything that has a suspend distance will allow suspension
107  if (pParams->GetFloat(CRCD(0xe4279ba2,"SuspendDistance"),&m_suspend_distance))
108  {
109  if (m_suspend_distance == 0.0f)
110  {
111  // if zero suspend dist, then try to get it from lod_dist1
112  if (!pParams->GetFloat(CRCD(0x4eedfae7,"Lod_dist1"),&m_suspend_distance))
113  {
114  // else get the default distance
115  m_suspend_distance=Script::GetFloat(CRCD(0xad501995,"DefaultMovingObjectSuspendDistance"));
116  }
117  }
118 
119  // if we've got a suspend distance, then convert it into inches
120  m_suspend_distance *= 12.0f;
121  m_suspend_distance_squared=m_suspend_distance*m_suspend_distance;
122  m_never_suspend=false;
123  }
124 
125  // suspension can be overridden by the "NeverSuspend" flag
126  if (pParams->ContainsFlag(CRCD(0xcb839dc1,"NeverSuspend")))
127  {
128  m_never_suspend=true;
129  }
130 
131  // Get the LOD distances from the node
132  if (Script::GetInt(CRCD(0x95edc6f6,"NoLOD"),false))
133  {
134  m_lod_dist[0] = 1000000;
135  m_lod_dist[1] = 1000001;
136  m_lod_dist[2] = 1000002; // Note, this is used as a magic number to turn off the high level culling
137  m_lod_dist[3] = 1000003;
138  }
139  else
140  {
141  Dbg_Assert( pParams );
142  m_lod_dist[0] = 50;
143  m_lod_dist[2] = 1000001;
144  m_lod_dist[3] = 1000002;
145  pParams->GetFloat(CRCD(0x4eedfae7,"lod_dist1"),&m_lod_dist[0]);
146  m_lod_dist[1] = m_lod_dist[0] + 75; // default for LOD dist 2 is 75 feet beyond 1
147  pParams->GetFloat(CRCD(0xd7e4ab5d,"lod_dist2"),&m_lod_dist[1]);
148  pParams->GetFloat(CRCD(0xa0e39bcb,"lod_dist3"),&m_lod_dist[2]);
149  pParams->GetFloat(CRCD(0x3e870e68,"lod_dist4"),&m_lod_dist[3]);
150  // LOD dist is in feet
151  m_lod_dist[0] *= 12.0f;
152  m_lod_dist[1] *= 12.0f;
153  m_lod_dist[2] *= 12.0f;
154  m_lod_dist[3] *= 12.0f;
155  if ( m_lod_dist[0]>=m_lod_dist[1] )
156  {
157  Script::PrintContents( pParams );
158  Dbg_MsgAssert(0,("lod_dist1 (%f) >= lod_dist2 (%f) in node",m_lod_dist[0],m_lod_dist[1]));
159  }
160  if ( m_lod_dist[1]>=m_lod_dist[2] )
161  {
162  Script::PrintContents( pParams );
163  Dbg_MsgAssert(0,("lod_dist2 (%f) >= lod_dist3 (%f) in node",m_lod_dist[1],m_lod_dist[2]));
164  }
165  if ( m_lod_dist[2]>=m_lod_dist[3] )
166  {
167  Script::PrintContents( pParams );
168  Dbg_MsgAssert(0,("lod_dist3 (%f) >= lod_dist4 (%f) in node",m_lod_dist[2],m_lod_dist[3]));
169  }
170 
171 #if defined( __PLAT_XBOX__ ) || defined( __PLAT_NGC__ )
172  // Some platform specific changes to the LOD stuff
173  m_lod_dist[0] *= Script::GetFloat( 0xd3d072f9 /* LOD0DistanceMultiplier */ );
174  m_lod_dist[1] *= Script::GetFloat( 0x0432f2a1 /* LOD1DistanceMultiplier */ );
175 #endif
176 
177  }

292 bool CSuspendComponent::should_animate( float *p_dist )
293 {
294  bool should_animate = true;
295 
296  // Mick: don't update the skeleton if it's a long way from the camera
297 // Gfx::Camera * p_camera = Nx::CViewportManager::sGetActiveCamera( 0 );
298 // if (p_camera)
299  {
300 // Mth::Vector cam_pos = p_camera->GetPos();
301 // cam_pos = cam_pos - GetObject()->GetPos();
302 // float dist = cam_pos.LengthSqr();
303 
304  float dist = m_camera_distance_squared;
305 
306  // Save this dist if a valid pointer was passed.
307  if( p_dist )
308  {
309  *p_dist = dist;
310  }
311 
312  if (dist > m_lod_dist[0] * m_lod_dist[0]) // dist at what we stop animating
313  {
314  should_animate = false;
315  }
316 
317  // update animation intermittently
318  // more so as distance from camera increases
319 
320  else
321  {
322 #ifndef TESTING_GUNSLINGER
323 
324 #if defined( __PLAT_XBOX__ ) || defined( __PLAT_NGC__ )
325  float interleave2 = Script::GetFloat( 0xdb120fb5 /* AnimLODInterleave2 */ );
326  float interleave4 = Script::GetFloat( 0x3271aa80 /* AnimLODInterleave4 */ );
327  interleave2 *= interleave2;
328  interleave4 *= interleave4;
329 #else
330  float interleave2 = 500.0f * 500.0f;
331  float interleave4 = 800.0f * 800.0f;
332 #endif
333 
334  m_interleave++;
335  should_animate = false;
336  if (dist > interleave4)
337  {
338  // 1 in 4 for distance 500-1000
339  if (!(m_interleave&3))
340  {
341  should_animate = true;
342  }
343  }
344  else if (dist > interleave2)
345  {
346  // 1 in 2 for 200 to 500
347  if ((m_interleave&1))
348  {
349  should_animate = true;
350  }
351  }
352  else
353  {
354  // always animate if < 200
355  should_animate = true;
356  }
357 #endif
358  }
359 
360  }
361 
362  if (m_initial_animations)
363  {
364  should_animate = true;
365  }
366 
367  return should_animate;
368 }

375 // Set Model's Active/Inactive flag based on other flags
376 // and if the model is actually visible or not (with culling and occlusion)
377 // and upon the load_dist2 distance
378 void CSuspendComponent::CheckModelActive()
379 {
380  if ( mp_model_component && mp_model_component->GetModel() )
381  {
382  bool active = false;
383  if ( GetObject()->GetFlags() & CObject::vINVISIBLE )
384  {
385  // just leave it inactive
386  }
387  else
388  {
389  // quick check for if the NoLOD flag is set
390  // Note use of magic number (but this is only for debugging/screenshots)
391  if ( m_lod_dist[2] == 1000002 )
392  {
393  active = true ;
394  }
395  else
396  {
397  // PATCH for multiple viewports
398  // always return true, until we sort out the per-viewport visibility (which might not be needed)
399  if ( Nx::CViewportManager::sGetNumActiveViewports() > 1 )
400  {
401  active = true;
402  }
403  else if ( mp_model_component->HasRefObject() )
404  {
405  // GJ: a ref model doesn't actually have a valid bounding sphere
406  // so don't let Nx::CEngine::sIsVisible() be called...
407  // (alternatively, we could fake a large bounding sphere
408  // but that seems kludgy)
409  active = true;
410  }
411  else
412  {
413  float dist = m_camera_distance_squared;
414 
415  if ( dist < m_lod_dist[1] * m_lod_dist[1] )
416  {
417  Mth::Vector sphere = mp_model_component->GetModel()->GetBoundingSphere();
418  Mth::Vector position = sphere;
419  position[3] = 1.0f;
420  position *= GetObject()->GetMatrix();
421  position += GetObject()->GetPos();
422  active = ( Nx::CEngine::sIsVisible( position, sphere[3]) );
423  }
424  else
425  {
426  active = false;
427  }
428  }
429  }
430  }
431  bool was_active = mp_model_component->GetModel()->GetActive();
432  mp_model_component->GetModel()->SetActive( active);
433  // If we've just turned the model component back on, then run the update on it
434  // so we set it to the right place
435  // otherwise it will be in the old place for a frame
436  // TODO: This all might be better IN the modelcomponent
437  if (active && !was_active)
438  {
439  mp_model_component->Update();
440  }
441  }
442 }

470 // The following is the SuspendOb manager, which iterates over the suspendob components
471 // to suspend them to whatever they are suspended to
472 
473 namespace SuspendManager
474 {
475 // The purpose of this manager is just to update all the suspended objects, by calling
476 // CMovingObject::SuspendToObject on all objects that have had Obj_SuspendToObject run on them.
477 // It is a manager so that it can be given a task with a priority such that it is called
478 // after all the objects are updated.
479 // The reason the suspend logic cannot be done in the object update is because sometimes the parent's
480 // update might happen after the child's, causing a frame lag. This was making the gorilla appear
481 // behind the tram seat in the zoo when the tram was being skitched. (TT2324)
482

514 void Manager::s_code( const Tsk::Task< Manager >& task )
515 {
516 
517  Mdl::SPreCalculatedObjectUpdateInfo* p_info = Mdl::Skate::Instance()->GetPreCalculatedObjectUpdateInfo();
518 
519  Obj::CSuspendComponent *p_suspend_component = static_cast<Obj::CSuspendComponent*>( Obj::CCompositeObjectManager::Instance()->GetFirstComponentByType( CRC_SUSPEND ));
520  while( p_suspend_component )
521  {
522 
523  // Set camera dist to zero for the purposes of LOD
524  p_suspend_component->m_camera_distance_squared = 0;
525  p_suspend_component->m_skip_logic = false;
526 
527  if (p_suspend_component->m_no_suspend_count)
528  {
529  p_suspend_component->m_no_suspend_count--;
530  continue;
531  }
532 
533  if ( p_info->mDoNotSuspendAnything )
534  {
535  }
536  else if ( p_suspend_component->m_never_suspend )
537  {
538  }
539  else if ( p_suspend_component->m_initial_animations )
540  {
541  p_suspend_component->m_initial_animations--;
542  }
543  else
544  {
545  p_suspend_component->m_camera_distance_squared = Mth::DistanceSqr(p_suspend_component->GetObject()->GetPos(),p_info->mActiveCameraPosition);
546 
547  if ( p_suspend_component->m_camera_distance_squared > p_suspend_component->m_suspend_distance_squared )
548  {
549  p_suspend_component->m_skip_logic = true;
550  }
551  else
552  {
553  p_suspend_component->m_skip_logic = false;
554  }
555  }
556  // suspend or unsuspend the object and its components
557  // based on the new state of m_skip_logic
558  if (p_suspend_component->m_skip_logic)
559  {
560  if (!p_suspend_component->GetObject()->IsSuspended())
561  {
562  p_suspend_component->GetObject()->Suspend(true);
563  }
564  }
565  else
566  {
567  if (p_suspend_component->GetObject()->IsSuspended())
568  {
569  p_suspend_component->GetObject()->Suspend(false);
570  }
571  }
572 
573  // If it has a model, then check for LOD to nothing
574  if (p_suspend_component->mp_model_component)
575  {
576  if (p_suspend_component->m_initial_animations)
577  {
578  if (p_suspend_component->mp_model_component->GetModel())
579  {
580  p_suspend_component->mp_model_component->GetModel()->SetActive(true);
581  }
582  }
583  else
584  {
585  p_suspend_component->CheckModelActive();
586  }
587  }
588 
589  p_suspend_component = static_cast<Obj::CSuspendComponent*>( p_suspend_component->GetNextSameType());
590  }
591 }
