89 void CSkaterMatrixQueriesComponent::Update()
90 {
91  // Store the matrix from the last frame. Used by script functions for measuring angles & stuff. Has to use the last frame's matrix
92  // because the landing physics snaps the orientation to the ground, yet the land script (which executes just after) needs to measure
93  // the angles on impact to maybe trigger bails and such.
94  m_latest_matrix = mp_core_physics_component->m_lerping_display_matrix;
95 }

102 CBaseComponent::EMemberFunctionResult CSkaterMatrixQueriesComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
103 {
104  switch ( Checksum )
105  {
106  // @script | YawBetween | yaw between the two specified angles
107  // @uparm (45, 135) | angle range values
108  case CRCC(0xb992f3cc, "YawBetween"):
109  {
110  // if (CHEAT_SNOWBOARD)
111  // {
112  // return false;
113  // }
114 
115  Script::CPair Pair;
116  if (!pParams->GetPair(NO_NAME, &Pair, Script::ASSERT))
117  Dbg_MsgAssert(Pair.mX < Pair.mY,("\n%s\n1st angle must be less than the 2nd angle", pScript->GetScriptInfo()));
118 
119  Mth::Vector a = GetSkater()->m_vel;
120  a.RotateToPlane(m_latest_matrix[Y]);
121  return (Mth::AngleBetweenGreaterThan(a, m_latest_matrix[Z], Pair.mX)
122  && !Mth::AngleBetweenGreaterThan(a, m_latest_matrix[Z], Pair.mY))
123  ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
124  }
125 
126  // @script | YawingLeft | true if currently yawing left
127  case CRCC(0xa745c080, "YawingLeft"):
128  {
129  Mth::Vector a = GetSkater()->m_vel;
130  a.RotateToPlane(m_latest_matrix[Y]);
131  return Mth::CrossProduct(a, m_latest_matrix[Z])[Y] > 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
132  }
133 
134  // @script | YawingRight | true if currently yawing right
135  case CRCC(0xc8c4d2f4, "YawingRight"):
136  {
137  Mth::Vector a = GetSkater()->m_vel;
138  a.RotateToPlane(m_latest_matrix[Y]);
139  return Mth::CrossProduct(a, m_latest_matrix[Z])[Y] < 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
140  }
141 
142  // @script | PitchGreaterThan | true if the pitch is greater
143  // than the specified value
144  // @uparm 0.0 | test angle
145  case CRCC(0xa0551543, "PitchGreaterThan"):
146  {
147  float TestAngle = 0.0f;
148  pParams->GetFloat(NO_NAME, &TestAngle, Script::ASSERT);
149 
150  return Mth::DotProduct(m_latest_matrix[Y], mp_core_physics_component->m_last_display_normal) < cosf(Mth::DegToRad(TestAngle))
151  ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
152  }
153 
154  case CRCC(0x5e269b2b, "AbsolutePitchGreaterThan"):
155  {
156  float TestAngle = 0.0f;
157  pParams->GetFloat(NO_NAME, &TestAngle, Script::ASSERT);
158 
159  return m_latest_matrix[Y][Y] < cosf(Mth::DegToRad(TestAngle)) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
160  }
161 
162  /*
163  // @script | PitchingForward | true if pitching forward
164  case CRCC(0xdaeda59c, "PitchingForward"):
165  {
166  Mth::Vector b = Mth::CrossProduct(m_latest_matrix[Y], mp_core_physics_component->m_last_display_normal);
167  return Mth::DotProduct(b, m_latest_matrix[X]) < 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
168  }
169 
170  // @script | PitchingBackward | true if pitching backward
171  case CRCC(0x7dd9e92c, "PitchingBackward"):
172  {
173  Mth::Vector b = Mth::CrossProduct(m_latest_matrix[Y], mp_core_physics_component->m_last_display_normal);
174  return Mth::DotProduct(b, m_latest_matrix[X]) > 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
175  }
176  */
177 
178  // @script | RollGreaterThan | true if the roll is greater than
179  // the specified angle value
180  // @uparm 0.0 | angle value
181  case CRCC(0xd3313e92, "RollGreaterThan"):
182  {
183  float TestAngle = 0.0f;
184  pParams->GetFloat(NO_NAME, &TestAngle, Script::ASSERT);
185 
186  Mth::Vector a = m_latest_matrix[X];
187  a.RotateToPlane(mp_core_physics_component->m_last_display_normal);
188  return Mth::AngleBetweenGreaterThan(a, m_latest_matrix[X], TestAngle) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
189  }
190 
191  /*
192  // @script | RollingLeft | true if rolling left
193  case CRCC(0x7328c9ad, "RollingLeft"):
194  {
195  Mth::Vector b = Mth::CrossProduct(m_latest_matrix[Y], mp_core_physics_component->m_last_display_normal);
196  return Mth::DotProduct(b, m_latest_matrix[Z]) > 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
197  }
198 
199  // @script | RollingRight | true if rolling right
200  case CRCC(0x8dcfe388, "RollingRight"):
201  {
202  Mth::Vector b = Mth::CrossProduct(m_latest_matrix[Y], mp_core_physics_component->m_last_display_normal);
203  return Mth::DotProduct(b, m_latest_matrix[Z]) < 0.0f ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
204  }
205  */
206 
207  // @script | GetSlope | Puts the angle of the slope into a param called Slope.
208  // Units are degrees. Zero is horizontal, positive is up, negative is down.
209  // The change in slope since the last call to GetSlope is put in a parameter
210  // called ChangeInSlope
211  case CRCC(0x97201739, "GetSlope"):
212  {
213  Mth::Vector v = GetObject()->m_matrix[Z];
214  v[Y] = 0.0f;
215  float slope = Mth::GetAngle(GetObject()->m_matrix[Z], v);
216  if (GetObject()->m_matrix[Z][Y] < 0.0f)
217  {
218  slope = -slope;
219  }
220  pScript->GetParams()->AddFloat(CRCD(0xa733ba7a, "Slope"), slope);
221  pScript->GetParams()->AddFloat(CRCD(0x21afff16, "ChangeInSlope"), slope - m_last_slope);
222  m_last_slope = slope;
223  break;
224  }
225 
226  case CRCC(0x8e7833be, "GetHeading"):
227  {
228  float heading = Mth::RadToDeg(cosf(GetObject()->m_matrix[Z][X]));
229  if (GetObject()->m_matrix[Z][Z] < 0.0f)
230  {
231  heading = 360.0f - heading;
232  }
233  pScript->GetParams()->AddFloat(CRCD(0xfd4bc03e, "heading"), heading);
234  pScript->GetParams()->AddFloat(CRCD(0x2315ef17, "cosine"), GetObject()->m_matrix[Z][X]);
235  pScript->GetParams()->AddFloat(CRCD(0x26910cc0, "sine"), GetObject()->m_matrix[Z][Z]);
236  break;
237  }
238 
239 
240  default:
241  return CBaseComponent::MF_NOT_EXECUTED;
242  }
243  return CBaseComponent::MF_TRUE;
244 }
