33 // The number of bits is set to be high to speed up AddChecksumName, so that
34 // LoadQB does not take ages to execute.
35 #define CHECKSUM_LOOKUP_HASH_BITS 16
36 // This probably does not need to be so big, but we've got lots of memory in debug so it's OK.
37 #define CHECKSUM_ENTRIES_PER_SLOT 10
38 
39 // There are (1<<CHECKSUM_LOOKUP_HASH_BITS) * CHECKSUM_ENTRIES_PER_SLOT of these,
40 // which uses about 5.2Meg.
41 struct SChecksumName
42 {
43  uint32 mChecksum;
44  const char *mpName;
45 };
46 
47 // The space available to hold all the checksum names. The names are stored contiguously
48 // with no wasted space in between. They stay in memory forever.
49 #define CHECKSUM_NAME_SPACE 5000000
50 
51 static SChecksumName *sp_checksum_name_hash_table=NULL;
52 static char *sp_checksum_names=NULL;
53 static char *sp_end_of_checksum_names=NULL;

55 void AllocateChecksumNameLookupTables()
56 {
57  if (!Config::GotExtraMemory())
58  {
59  return;
60  }
61 
62  Dbg_MsgAssert(sp_checksum_name_hash_table==NULL,("sp_checksum_name_hash_table not NULL ?"));
63  Dbg_MsgAssert(sp_checksum_names==NULL,("sp_checksum_names not NULL ?"));
64  Dbg_MsgAssert(sp_end_of_checksum_names==NULL,("sp_end_of_checksum_names not NULL ?"));
65 
66  #ifdef __PLAT_NGPS__
67 
68  #ifdef __NOPT_ASSERT__
69  // Use the chunk of memory that was set aside for script debug stuff in NGPS.lk
70  uint32 script_debug_mem_size=(uint32)(_script_end-_script_start);
71 
72  // Check there is enough space.
73  uint32 hash_table_size=(1<<CHECKSUM_LOOKUP_HASH_BITS) * CHECKSUM_ENTRIES_PER_SLOT * sizeof(SChecksumName);
74  Dbg_MsgAssert(script_debug_mem_size>=hash_table_size+CHECKSUM_NAME_SPACE,(
75  "Chunk of debug memory reserved for script stuff is not big enough.\nRequired=%d, got=%d",
76  hash_table_size+CHECKSUM_NAME_SPACE, script_debug_mem_size));
77  #endif
78 
79  sp_checksum_name_hash_table=(SChecksumName*)_script_start;
80  sp_checksum_names=(char*)(sp_checksum_name_hash_table+((1<<CHECKSUM_LOOKUP_HASH_BITS) * CHECKSUM_ENTRIES_PER_SLOT));
81  sp_end_of_checksum_names=sp_checksum_names;
82 
83  // Set up pointers to the locations of sp_checksum_names and sp_end_of_checksum_names
84  // so that later the script debugger can find the list of names using the target manager API
85  // and register their checksums too.
86  Dbg_MsgAssert((uint32)_script_debugger_start==Dbg::SCRIPT_DEBUGGER_MEMORY_START,("Need to update SCRIPT_DEBUGGER_MEMORY_START in debugger.h to match the location of the script_debugger group in ngps.lk"));
87  Dbg::SChecksumNameInfo *p_checksum_name_info=(Dbg::SChecksumNameInfo*)_script_debugger_start;
88  p_checksum_name_info->mppStart=&sp_checksum_names;
89  p_checksum_name_info->mppEnd=&sp_end_of_checksum_names;
90 
91  #else
92 
93  // Just allocate the arrays off the heap.
94 
95  // Checksum-name lookup stuff.
96  sp_checksum_name_hash_table=(SChecksumName*)Mem::Malloc((1<<CHECKSUM_LOOKUP_HASH_BITS) * CHECKSUM_ENTRIES_PER_SLOT * sizeof(SChecksumName));
97  sp_checksum_names=(char*)Mem::Malloc(CHECKSUM_NAME_SPACE);
98  sp_end_of_checksum_names=sp_checksum_names;
99 
100  #endif // __PLAT_NGPS__
101 
102  // Initialise the hash table.
103  for (int i=0; i<(1<<CHECKSUM_LOOKUP_HASH_BITS) * CHECKSUM_ENTRIES_PER_SLOT; ++i)
104  {
105  sp_checksum_name_hash_table[i].mChecksum=0;
106  sp_checksum_name_hash_table[i].mpName=NULL;
107  }
108  // Initialise the pile of strings.
109  *sp_checksum_names=0;
110 }

152 void AddChecksumName(uint32 checksum, const char *p_name)
153 {
154  if (!Config::GotExtraMemory())
155  {
156  return;
157  }
158 
159  Dbg_MsgAssert(checksum,("Zero checksum sent to AddChecksumName"));
160  Dbg_MsgAssert(p_name,("NULL p_name sent to AddChecksumName"));
161  Dbg_MsgAssert(sp_checksum_name_hash_table,("NULL sp_checksum_name_hash_table"));
162 
163  SChecksumName *p_first=&sp_checksum_name_hash_table[(checksum&((1<<CHECKSUM_LOOKUP_HASH_BITS)-1))*CHECKSUM_ENTRIES_PER_SLOT];
164  int c=0;
165  while (p_first->mpName)
166  {
167  // Already a name here.
168  // See if it has the same checksum.
169  if (p_first->mChecksum==checksum)
170  {
171  // It does! Check whether it is the same name.
172  if (stricmp(p_first->mpName,p_name)==0)
173  {
174  // Phew, the name matches. No need to do anything.
175  return;
176  }
177  else
178  {
179  // Oh bugger, a checksum clash.
180  #ifdef __PLAT_WN32__
181  #ifdef __QDEBUG__
182  char p_foo[1024];
183  sprintf(p_foo,"Checksum clash between %s and %s, both have checksum 0x%08x",p_name,p_first->mpName,checksum);
184  MessageBox(NULL,p_foo,"Warning",MB_OK);
185  #endif
186  // Carry on anyway, won't cause a crash.
187  #else
188  Dbg_MsgAssert(0,("Checksum clash between %s and %s, both have checksum 0x%08x",p_name,p_first->mpName,checksum));
189  #endif
190  return;
191  }
192  }
193 
194  // Not the same checksum. Onto the next entry.
195  ++c;
196  Dbg_MsgAssert(c<CHECKSUM_ENTRIES_PER_SLOT,("Need to increase CHECKSUM_ENTRIES_PER_SLOT in checksum.cpp"));
197  ++p_first;
198  }
199 
200  //++sNumChecksums;
201  //printf("sNumChecksums=%d\n",sNumChecksums);
202 
203  p_first->mChecksum=checksum;
204  p_first->mpName=sp_end_of_checksum_names;
205 
206  int space_left=CHECKSUM_NAME_SPACE-(sp_end_of_checksum_names-sp_checksum_names);
207  while (*p_name)
208  {
209  Dbg_MsgAssert(space_left>0,("Need to increase CHECKSUM_NAME_SPACE in checksum.cpp"));
210  *sp_end_of_checksum_names++=*p_name++;
211  --space_left;
212  }
213  Dbg_MsgAssert(space_left>0,("Need to increase CHECKSUM_NAME_SPACE in checksum.cpp"));
214  *sp_end_of_checksum_names++=0;
215 }

219 // Returns NULL if not found.
220 // This version is not safe to use in a printf because it may return NULL, but can be handy for
221 // when the calling code needs to do something special if the name is not found.
222 const char *FindChecksumNameNULL(uint32 checksum)
223 {
224  if (!Config::GotExtraMemory())
225  {
226  return NULL;
227  }
228 
229  Dbg_MsgAssert(sp_checksum_name_hash_table,("NULL sp_checksum_name_hash_table"));
230  SChecksumName *p_first=&sp_checksum_name_hash_table[(checksum&((1<<CHECKSUM_LOOKUP_HASH_BITS)-1))*CHECKSUM_ENTRIES_PER_SLOT];
231  int c=CHECKSUM_ENTRIES_PER_SLOT;
232  while (p_first->mpName)
233  {
234  if (p_first->mChecksum==checksum)
235  {
236  return p_first->mpName;
237  }
238  --c;
239  if (c==0)
240  {
241  // Run out of slots.
242  break;
243  }
244  ++p_first;
245  }
246 
247  return NULL;
248 }

250 // Returns "Unknown" if not found.
251 // This version is always safe to use in a printf.
252 const char *FindChecksumName(uint32 checksum)
253 {
254  const char *p_name=FindChecksumNameNULL(checksum);
255  if (p_name)
256  {
257  return p_name;
258  }
259  else
260  {
261  #ifdef __PLAT_WN32__
262  // The monitor.exe application requires that the unknown checksum be
263  // displayed like this.
264  sprintf(sp_unknown_checksum_buf,"0x%08x",checksum);
265  #else
266  sprintf(sp_unknown_checksum_buf,"Unknown(0x%08x)",checksum);
267  #endif
268  return sp_unknown_checksum_buf;
269  }
270 }
