178 void CNgcParticleRibbonTrail::plat_render( void )
179 {
180  NxNgc::sMaterialHeader mat;
181  NxNgc::sMaterialPassHeader pass;
182 
183  // Header.
184  mat.m_checksum = 0xa9db601e; // particle
185  mat.m_passes = 1;
186  mat.m_alpha_cutoff = 1;
187  mat.m_flags = (1<<1); // 2 sided.
188 // mat.m_shininess = 0.0f;
189 
190  // Pass 0.
191  pass.m_texture.p_data = mp_engine_texture;
192  pass.m_flags = ( mp_engine_texture ? (1<<0) : 0 ) | (1<<5) | (1<<6); // textured, clamped.
193  pass.m_filter = 0;
194  pass.m_blend_mode = (unsigned char)m_blend;
195  pass.m_alpha_fix = (unsigned char)m_fix;
196  pass.m_k = 0;
197  pass.m_color.r = 128;
198  pass.m_color.g = 128;
199  pass.m_color.b = 128;
200  pass.m_color.a = 255;
201 
202  NxNgc::multi_mesh( &mat, &pass, true, true );
203 
204  GX::SetTexCoordGen( GX_TEXCOORD0, GX_TG_MTX2x4, GX_TG_TEX0, GX_FALSE, GX_PTIDENTITY );
205  GX::SetCurrMtxPosTex03( GX_PNMTX0, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY );
206 
207  // Draw the particles.
208  int lp;
209  CParticleEntry *p_particle;
210  float *p_v;
211 
212  GXColor color[2];
213  GXColor *p_col0;
214  GXColor *p_col1;
215 
216  NsMatrix* p_matrix = &NxNgc::EngineGlobals.camera;
217  Mth::Vector at( p_matrix->getAtX(), p_matrix->getAtY(), p_matrix->getAtZ(), 0.0f );
218 
219  GX::SetVtxDesc( 2, GX_VA_POS, GX_DIRECT, GX_VA_CLR0, GX_DIRECT );
220 
221  for ( lp = 0, p_particle = mp_particle_array; lp < m_num_particles; lp++, p_particle++ )
222  {
223  float terp = p_particle->m_time / p_particle->m_life;
224 
225  Mth::Vector pos[2];
226  p_v = &mp_vertices[0][lp*3];
227  pos[0].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
228  p_v = &mp_vertices[1][lp*3];
229  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
230 
231  Mth::Vector part_vec = pos[1] - pos[0];
232  Mth::Vector perp_vec = Mth::CrossProduct( part_vec, at );
233  perp_vec.Normalize();
234 
235  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
236 
237  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
238  Mth::Vector tmp[4];
239 
240  if ( m_mid_time >= 0.0f )
241  {
242  if ( terp < m_mid_time )
243  {
244  p_col0 = m_start_color;
245  p_col1 = m_mid_color;
246  // Adjust interpolation for this half of the color blend.
247  terp = terp / m_mid_time;
248  }
249  else
250  {
251  p_col0 = m_mid_color;
252  p_col1 = m_end_color;
253  // Adjust interpolation for this half of the color blend.
254  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
255  }
256  }
257  else
258  {
259  // No mid color specified.
260  p_col0 = m_start_color;
261  p_col1 = m_end_color;
262  }
263 
264  GXColor start = *p_col0++;
265  GXColor end = *p_col1++;
266 
267  color[0].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
268  color[0].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
269  color[0].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
270  color[0].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
271 
272  tmp[0] = pos[0] + ( perp_vec * w );
273  tmp[1] = pos[0] - ( perp_vec * w );
274 
275  GX::Begin( GX_QUADS, GX_VTXFMT0, 4 * ( m_num_vertex_buffers - 1 ) );
276 
277  for ( int c = 1; c < m_num_vertex_buffers; c++ )
278  {
279  start = *p_col0++;
280  end = *p_col1++;
281 
282  color[1].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
283  color[1].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
284  color[1].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
285  color[1].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
286 
287  if ( c > 1 )
288  {
289  p_v = &mp_vertices[c][lp*3];
290  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
291  part_vec = pos[1] - pos[0];
292  perp_vec = Mth::CrossProduct( part_vec, at );
293  perp_vec.Normalize();
294  }
295 
296  tmp[2] = pos[1] - ( perp_vec * w );
297  tmp[3] = pos[1] + ( perp_vec * w );
298 
299  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
300  GX::Color1u32( *((uint32*)&color[0]) );
301 
302  GX::Position3f32( tmp[1][X], tmp[1][Y], tmp[1][Z] );
303  GX::Color1u32( *((uint32*)&color[0]) );
304 
305  GX::Position3f32( tmp[2][X], tmp[2][Y], tmp[2][Z] );
306  GX::Color1u32( *((uint32*)&color[1]) );
307 
308  GX::Position3f32( tmp[3][X], tmp[3][Y], tmp[3][Z] );
309  GX::Color1u32( *((uint32*)&color[1]) );
310 
311  color[0] = color[1];
312  pos[0] = pos[1];
313  tmp[0] = tmp[3];
314  tmp[1] = tmp[2];
315  }
316  GX::End();
317  }
318 }
