22 // Returns true if p_string starts with p_prefix. Non-case-sensitive.
23 // (don't want to use strstr cos it is case sensitive)
24 static bool sIsPrefixedWith(const char *p_string, const char *p_prefix)
25 {
26  if (!p_string || !p_prefix) return false;
27 
28  int string_len=strlen(p_string);
29  int prefix_len=strlen(p_prefix);
30  if (prefix_len>string_len)
31  {
32  // Can't be prefixed if shorter than the prefix.
33  return false;
34  }
35 
36  char p_buf[100];
37  Dbg_MsgAssert(string_len<100,("String too long for buffer"));
38  strcpy(p_buf,p_string);
39  p_buf[prefix_len]=0;
40  if (stricmp(p_buf,p_prefix)==0)
41  {
42  return true;
43  }
44  return false;
45 }

47 // If p_name is found to be followed by an equals in the command line, then this function
48 // will return the word following the equals.
49 // If p_name is not found, or equals blank, it will return NULL.
50 const char *GetCommandLineParam(const char *p_name, sint argc, char** argv)
51 {
52  const char *p_found=NULL;
53 
54  for (int i=0; i<argc; ++i)
55  {
56  if (sIsPrefixedWith(argv[i],p_name))
57  {
58  // p_name matches the start of argv[i]
59  // Now see if it is followed by an equals.
60  int len=strlen(p_name);
61  char *p_next_char=argv[i]+len;
62  if (*p_next_char=='=')
63  {
64  // It is followed by an equals.
65  ++p_next_char;
66  if (*p_next_char)
67  {
68  // There is something after the equals, so use that.
69  p_found=argv[i]+len+1;
70  }
71  else
72  {
73  // Use the next word, if there is one.
74  if (i<argc-1)
75  {
76  p_found=argv[i+1];
77  }
78  }
79  }
80  else if (*p_next_char==0)
81  {
82  // p_name matches argv[i] exactly.
83  // So see if argv[i+1] starts with an equals.
84  if (i<argc-1 && argv[i+1][0]=='=')
85  {
86  if (argv[i+1][1]==0)
87  {
88  // argv[i+1] is just '=', so use the next string.
89  if (i<argc-2)
90  {
91  p_found=argv[i+2];
92  }
93  }
94  else
95  {
96  // There is a word following the initial '=' in argv[i+1],
97  // so use that word.
98  p_found=argv[i+1]+1;
99  }
100  }
101  }
102  }
103  }
104  return p_found;
105 }

122 // This function is the first thing called from main()
123 // It fills in all the globals in the Config namespace.
124 // It calls the platform specific Plat_Init() first.
125 // The idea is that the Plat_Init() function will do its best to fill in as many of the globals
126 // as it can.
127 // Then, the rest of the Init() function will fill in any that are missing, or override some
128 // based on the command line parameters.
129 
130 // For example, the PS2 Plat_Init will look at the executable name to derive the territory.
131 // However, if the executable name is just skate5.elf it contains no territory info.
132 // So the Init() function, on seeing that the territory is undefined, will derive the territory
133 // from the language and whether it is PAL.
134 void Init(sint argc, char** argv)
135 {
136  Plat_Init(argc,argv);
137 
138  // Override the gCD flag if CD or NotCD are present on the command line.
139  if (CommandLineContainsFlag("CD",argc,argv))
140  {
141  gCD=true;
142  }
143  if (CommandLineContainsFlag("NotCD",argc,argv))
144  {
145  gCD=false;
146  }
147 
148  if (CommandLineContainsFlag("Bootstrap",argc,argv))
149  {
150  gBootstrap=true;
151  if (argc > 2)
152  {
153  strcpy(gReturnTo,argv[2]);
154  if (argc > 3)
155  {
156  strcpy(gReturnString,argv[3]);
157  }
158  else
159  {
160  sprintf(gReturnString,"THPS4");
161  }
162  printf ("This bootstrap demo will return to %s passing back: %s\n",gReturnTo,gReturnString);
163  }
164  }
165  if (CommandLineContainsFlag("NotBootstrap",argc,argv))
166  {
167  gBootstrap=false;
168  }
169 
170  if (CommandLineContainsFlag("Pal",argc,argv))
171  {
172  gDisplayType=DISPLAY_TYPE_PAL;
173  }
174  if (CommandLineContainsFlag("NTSC",argc,argv))
175  {
176  gDisplayType=DISPLAY_TYPE_NTSC;
177  }
178 
179  const char *p_frame_rate=GetCommandLineParam("FrameRate",argc,argv);
180  if (p_frame_rate)
181  {
182  if (strcmp(p_frame_rate,"50")==0)
183  {
184  gFPS=50;
185  }
186  if (strcmp(p_frame_rate,"60")==0)
187  {
188  gFPS=60;
189  }
190  }
191 }
