69 struct sDSPADPCM
70 {
71  // for WAV header generation during decode
72  u32 num_samples; // total number of RAW samples
73  u32 num_adpcm_nibbles; // number of ADPCM nibbles (including frame headers)
74  u32 sample_rate; // Sample rate, in Hz
75 
76  // DSP addressing and decode context
77  u16 loop_flag; // 1=LOOPED, 0=NOT LOOPED
78  u16 format; // Always 0x0000, for ADPCM
79  u32 sa; // Start offset address for looped samples (zero for non-looped)
80  u32 ea; // End offset address for looped samples
81  u32 ca; // always zero
82  u16 coef[16]; // decode coefficients (eight pairs of 16-bit words)
83 
84  // DSP decoder initial state
85  u16 gain; // always zero for ADPCM
86  u16 ps; // predictor/scale
87  u16 yn1; // sample history
88  u16 yn2; // sample history
89 
90  // DSP decoder loop context
91  u16 lps; // predictor/scale for loop context
92  u16 lyn1; // sample history (n-1) for loop context
93  u16 lyn2; // sample history (n-2) for loop context
94  u16 pad[10]; // reserved
95 };

211 static bool loadGamecubeDSP( PlatformWaveInfo *pInfo, const char *filename, bool loadPerm, float *samplePitch )
212 {
213  int vagAddr = 0; // VAG address in SPU2 RAM, returned to calling function
214  bool rv = false;
215 
216  void* p_file = File::Open( filename, "rb" );
217  if( p_file )
218  {
219  int size = File::GetFileSize( p_file );
220  if ( size )
221  {
222  int align_size = ( size + 31 ) & ~31;
223  int data_size = align_size - 96;
224 
225  // Grab enough memory to ensure a multiple of 32 bytes data length. To avoid fragmentation, we grab
226  // memory for the header copy first.
227  char* p_header_copy = (char*)Mem::Malloc( 96 );
228 // File::Read( p_header_copy, 1, 96, p_file );
229  Mem::Manager::sHandle().BottomUpHeap()->PushAlign( 32 );
230  char* p_data = (char*)Mem::Malloc( align_size );
231  Mem::Manager::sHandle().BottomUpHeap()->PopAlign();
232 // #ifdef __NOPT_ASSERT__
233  int bytes_read = File::Read( p_data, 1, size, p_file );
234 // #else
235 // File::Read( p_data, 1, size, p_file );
236  DCFlushRange ( p_data, size );
237 // #endif // __NOPT_ASSERT__
238 // Dbg_Assert( bytes_read == size )
239  if ( bytes_read != size )
240  {
241  File::Close( p_file );
242  delete p_data;
243  return false;
244  }
245 // file.close();
246 
247  // Copy the header to the reserved area.
248  memcpy( p_header_copy, p_data, 96 );
249 
250  // Obtain information from the file header.
251  sDSPADPCM* p_header = (sDSPADPCM*)p_header_copy;
252  // OSReport( "num samples: %d\n", p_header->num_samples );
253  // OSReport( "sample rate: %d\n", p_header->sample_rate );
254  // OSReport( "loop flag: %d\n", p_header->loop_flag );
255 
256  if( loadPerm )
257  {
258  // Stack the permanent sounds up from the bottom.
259  SpuRAMUsedPerm += data_size;
260  if(( GAMECUBE_END_DSP_DATA_ADDR - SpuRAMUsedPerm ) & 0x0F )
261  {
262  #ifdef __NOPT_ASSERT__
263  int oldSpuRAM = SpuRAMUsedPerm;
264  #endif // __NOPT_ASSERT__
265  SpuRAMUsedPerm += (( END_WAVE_DATA_ADDR - SpuRAMUsedPerm ) & 0x0F );
266  Dbg_MsgAssert( SpuRAMUsedPerm > oldSpuRAM, ( "Duh asshole." ));
267  }
268  vagAddr = GAMECUBE_END_DSP_DATA_ADDR - SpuRAMUsedPerm;
269  Dbg_MsgAssert( !( vagAddr & 0x0F ), ( "Fire Matt please." ));
270  }
271  else
272  {
273  // If a temporary sound, add the details of the header to the cleanup array.
274  tempSoundHeaders[numTempSoundHeaders++] = p_header_copy;
275 
276  if(( SpuRAMUsedTemp + GAMECUBE_START_DSP_DATA_ADDR ) & 0x0F )
277  {
278  SpuRAMUsedTemp += 0x10 - ( ( SpuRAMUsedTemp + BASE_WAVE_DATA_ADDR ) & 0x0F );
279  }
280  vagAddr = GAMECUBE_START_DSP_DATA_ADDR + SpuRAMUsedTemp;
281  SpuRAMUsedTemp += data_size;
282  }
283 
284  // Start DMA transfer buffer to sound RAM...
285  Dbg_Assert(((unsigned int)p_data & 31 ) == 0 );
286  Dbg_Assert(((unsigned int)vagAddr & 31 ) == 0 );
287 
288  // Read data in chunks & DMA.
289 // int toread = size - 96;
290 // int offset = 0;
291 // while ( toread )
292 // {
293 //#define BUFFER_SIZE 16384
294 // char readBuffer[BUFFER_SIZE];
295 // if ( toread > BUFFER_SIZE )
296 // {
297 // // Read a full buffer.
298 // File::Read( readBuffer, 1, BUFFER_SIZE, p_file );
299 // toread -= BUFFER_SIZE;
300 // offset += BUFFER_SIZE;
301 // NsDMA::toARAM( vagAddr + offset, readBuffer, BUFFER_SIZE );
302 // }
303 // else
304 // {
305 // // Last chunk
306 // File::Read( readBuffer, 1, toread, p_file );
307 // NsDMA::toARAM( vagAddr + offset, readBuffer, ( toread + 31 ) & ~31 );
308 // toread = 0;
309 // }
310 // }
311 //
312 
313  NsDMA::toARAM( vagAddr, p_data + 96, data_size );
314 
315  pInfo->headerAddress = p_header_copy;
316  pInfo->vagAddress = vagAddr;
317  pInfo->looping = ( p_header->loop_flag > 0 );
318 
319  // Figure pitch adjustment coefficient.
320  if( samplePitch )
321  {
322  *samplePitch = ( (float)p_header->sample_rate * 100.0f ) / 32000.0f;
323  }
324 
325  // Release main RAM version of data.
326  Mem::Free( p_data );
327 
328  rv = true;
329  }
330  File::Close( p_file );
331  }
332  else
333  {
334 // OSReport( 0, "************ failed to load sound: %s\n", filename );
335 // while ( 1== 1 );
336  //Dbg_MsgAssert( 0, ( "failed to load sound: %s\n", filename ));
337  }
338  return rv;
339 }

389 // AXInit();

391  // Set the callback responsible for freeing up voices.
392  AXRegisterCallback( AXUserCBack );

395 # if aram_zero_size != 1024

535 /******************************************************************/
536 /* */
537 /* */
538 /******************************************************************/
539 //int PlaySoundPlease( PlatformWaveInfo *pInfo, float volL, float volR, float pitch )
540 int PlaySoundPlease( PlatformWaveInfo *pInfo, sVolume *p_vol, float pitch )
541 {
542  //Dbg_Assert( pInfo->headerAddress != NULL );
543 
544 // Spt::SingletonPtr< Sfx::CSfxManager > sfx_manager;
545 // if ( sfx_manager->GetMainVolume() < 1.0f ) return -1;
546 
547  int voice = getFreeVoice();
548  if( voice == -1 )
549  {
550  // No voice in our array available.
551  return -1;
552  }
553 
554  // Attempt to grab a voice. Set the priority high if it is a looping sound.
555  sDSPADPCM* p_header = (sDSPADPCM*)pInfo->headerAddress;
556  // HACK!!!
557  if( !p_header )
558  {
559  // No voice available.
560  return -1;
561  }
562  AXVPB* p_axvpb = AXAcquireVoice( ( p_header->loop_flag ) ? AX_PRIORITY_NODROP : AX_PRIORITY_LOWEST, ax_voice_reacquisition_callback, 0 );
563  if( p_axvpb == NULL )
564  {
565  // No voice available.
566  return -1;
567  }
568 
569  AXPBADDR addr;
570  AXPBSRC src;
571  AXPBADPCM adpcm;
572  AXPBADPCMLOOP loop;
573 
574  addr.loopFlag = p_header->loop_flag ? AXPBADDR_LOOP_ON : AXPBADDR_LOOP_OFF;
575  addr.format = AX_PB_FORMAT_ADPCM;
576 
577  // Convert from byte to nibble address, and split into low and high words.
578  u32 start_address = ( pInfo->vagAddress * 2 ) + 2;
579  u32 end_address = ( pInfo->vagAddress * 2 ) + ( p_header->num_adpcm_nibbles - 1 );
580 
581  addr.currentAddressHi = (u16)( start_address >> 16 );
582  addr.currentAddressLo = (u16)( start_address & 0xFFFFU );
583  addr.endAddressHi = (u16)( end_address >> 16 );
584  addr.endAddressLo = (u16)( end_address & 0xFFFFU );
585 
586  if( p_header->loop_flag )
587  {
588  addr.loopAddressHi = (u16)( start_address >> 16 );
589  addr.loopAddressLo = (u16)( start_address & 0xFFFFU );
590  }
591  else
592  {
593  u32 zero_buffer_address = NxNgc::EngineGlobals.aram_zero * 2;
594  addr.loopAddressHi = (u16)( zero_buffer_address >> 16 );
595  addr.loopAddressLo = (u16)( zero_buffer_address & 0xFFFFU );
596  }
597 
598  loop.loop_pred_scale = p_header->lps;
599  loop.loop_yn1 = p_header->lyn1;
600  loop.loop_yn2 = p_header->lyn2;
601 
602 // uint32 srcBits = (uint32)(0x00010000 * ((f32)p_header->sample_rate / AX_IN_SAMPLES_PER_SEC));
603 // src.ratioHi = (u16)(srcBits >> 16);
604 // src.ratioLo = (u16)(srcBits & 0xFFFF);
605  src.ratioHi = 1;
606  src.ratioLo = 0;
607  src.currentAddressFrac = 0;
608  src.last_samples[0] = 0;
609  src.last_samples[1] = 0;
610  src.last_samples[2] = 0;
611  src.last_samples[3] = 0;
612 
613  memcpy( &adpcm.a[0][0], p_header->coef, sizeof( u16 ) * 16 );
614  adpcm.gain = p_header->gain;
615  adpcm.pred_scale = p_header->ps;
616  adpcm.yn1 = p_header->yn1;
617  adpcm.yn2 = p_header->yn2;
618 
619  AXPBVE ve;
620  ve.currentVolume = 0x7FFF; // This is max volume.
621  ve.currentDelta = 0x0000;
622 
623  AXSetVoiceType( p_axvpb, AX_PB_TYPE_NORMAL );
624 
625  AXSetVoiceAddr( p_axvpb, &addr ); // input addressing
626  AXSetVoiceAdpcm( p_axvpb, &adpcm ); // ADPCM coefficients
627  AXSetVoiceAdpcmLoop( p_axvpb, &loop ); // ADPCM loop context
628 
629  AXSetVoiceVe( p_axvpb, &ve );
630 
631  AXSetVoiceSrcType( p_axvpb, AX_SRC_TYPE_LINEAR ); // SRC type
632  AXSetVoiceSrc( p_axvpb, &src ); // initial SRC settings
633 
634  // Set the voice to run.
635  AXSetVoiceState( p_axvpb, AX_PB_STATE_RUN );
636 
637  // Fill in voice details.
638  VoiceSimulator[voice].p_axvpb = p_axvpb;
639  VoiceSimulator[voice].p_dspadpcm = p_header;
640 // OSReport( "voice %d started %d\n", voice, OSGetTick() );
641 
642  // Adjust pitch to account for lower sampling rates:
643  pitch = PERCENT( pitch, pInfo->pitchAdjustmentForSamplingRate );
644 
645  // This takes the place of AxSetVoiceMix().
646  SetVoiceParameters( voice, p_vol, pitch );
647 
648  return voice;
649 }
