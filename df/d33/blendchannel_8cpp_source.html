71 CBaseAnimController* CBlendChannel::AddController( Script::CStruct* pParams )
72 {
73  uint32 type;
74  pParams->GetChecksum( CRCD(0x7321a8d6,"type"), &type, Script::ASSERT );
75 
76  CBaseAnimController* pController = NULL;
77 
78  int priority = 500;
79 
80  switch ( type )
81  {
82  case 0xcb4533ff: // bonedanim
83  {
84  pController = new CBonedAnimController( this );
85  }
86  break;
87 
88  case 0xd079853e: // ik
89  {
90  pController = new CIKController( this );
91  }
92  break;
93 
94  case 0x6d941203: // wobble
95  {
96  pController = new CWobbleController( this );
97  }
98  break;
99 
100  case 0xcfc5b380: // pose
101  {
102  pController = new CPoseController( this );
103  }
104  break;
105 
106  case 0xfdf0436c: // proceduralanim
107  {
108  pController = new CProceduralAnimController( this );
109  }
110  break;
111 
112  case 0xdf5c091a: // fliprotate
113  {
114  pController = new CFlipRotateController( this );
115 
116  // make it be processed later
117  // (so that partial anims can take effect of flipping)
118  priority = 100;
119  }
120  break;
121 
122  case 0x26205db6: // lookat
123  {
124  pController = new CLookAtController( this );
125  }
126  break;
127 
128  case 0x659bf355: // partialanim
129  {
130  pController = new CPartialAnimController( this );
131  }
132  break;
133 
134  default:
135  {
136  Dbg_MsgAssert( 0, ( "Unrecognized controller %s", Script::FindChecksumName(type) ) );
137  }
138  }
139 
140  // override priority, if necessary
141  pParams->GetInteger( CRCD(0x9d5923d8,"priority"), &priority, Script::NO_ASSERT );
142 
143  Dbg_MsgAssert( pController, ( "No controller" ) );
144  pController->InitFromStructure( pParams );
145  pController->SetPriority( priority );
146  add_controller( pController, priority );
147 
148  return pController;
149 }

156 void CBlendChannel::add_controller( CBaseAnimController* pAnimController, int priority )
157 {
158  Dbg_MsgAssert( m_numControllers < vMAX_CONTROLLERS, ( "Too many controllers for this channel on %s", Script::FindChecksumName( GetObject()->GetID() ) ) );
159  Dbg_MsgAssert( pAnimController, ( "No controller" ) );
160 
161  int i = 0;
162  for ( i = 0; i < m_numControllers; i++ )
163  {
164  if ( priority > mp_controllers[i]->GetPriority() )
165  {
166  // shift all of them
167  for ( int j = m_numControllers; j > i; j-- )
168  {
169  mp_controllers[j] = mp_controllers[j - 1];
170  }
171 
172  break;
173  }
174  }
175 
176  mp_controllers[i] = pAnimController;
177  m_numControllers++;
178 }

249 void CBlendChannel::Update()
250 {
251  // reset the anim complete field for this frame
252  m_animComplete = false;
253 
254  if ( GetStatus() == ANIM_STATUS_DEGENERATING )
255  {
256  m_degenerationTime -= ( m_blendSpeed * Tmr::FrameRatio() );
257  if ( m_degenerationTime <= 0.0f )
258  {
259  // This animation is no longer active.
260  SetStatus( ANIM_STATUS_INACTIVE );
261  }
262  else if ( ( m_degenerationTime * m_degenerationTimeToBlendMultiplier ) < 0.0f )
263  {
264  // Check for being too small to care about.
265  SetStatus( ANIM_STATUS_INACTIVE );
266  }
267  }
268  else
269  {
270  // don't really need to do this any more, because
271  // the update happens in the CBonedAnimController...
272  // CAnimChannel::Update();
273  }
274 
275  if ( GetStatus() == ANIM_STATUS_ACTIVE )
276  {
277  for ( int i = 0; i < m_numControllers; i++ )
278  {
279  // one of these update functions should set the correct m_time for this frame
280  mp_controllers[i]->Update();
281  }
282  }

557 void CBlendChannel::add_custom_keys( uint32 animName )
558 {
559  Obj::CAnimationComponent* pAnimationComponent = GetAnimationComponentFromObject( GetObject() );
560  if ( !pAnimationComponent )
561  {
562  return;
563  }
564 
565  uint32 animEventTableName = pAnimationComponent->GetAnimEventTableName();
566  if ( !animEventTableName )
567  {
568  return;
569  }
570 
571  Script::CStruct* pAnimEventTable = Script::GetStructure( animEventTableName, Script::ASSERT );
572  if ( pAnimEventTable )
573  {
574  Script::CArray* pSubArray;
575  if ( pAnimEventTable->GetArray( animName, &pSubArray, Script::NO_ASSERT ) )
576  {
577  for ( uint32 i = 0; i < pSubArray->GetSize(); i++ )
578  {
579  Script::CStruct* pSubStruct;
580  pSubStruct = pSubArray->GetStructure( i );
581 
582  int frame = 0;
583  // if the frame is not specified, then look for a time value
584  if ( !pSubStruct->GetInteger( CRCD(0x4a07c332,"frame"), &frame, Script::NO_ASSERT ) )
585  {
586  float keyTime = 0.0f;
587  pSubStruct->GetFloat( CRCD(0x906b67ba,"time"), &keyTime, Script::ASSERT );
588  frame = (int)(keyTime * 60.0f);
589  }
590 
591  uint32 eventType;
592  pSubStruct->GetChecksum( CRCD(0xc451f558,"event"), &eventType, Script::ASSERT );
593 
594  Script::CStruct* pEventParams = NULL;
595  pSubStruct->GetStructure( CRCD(0x7031f10c,"params"), &pEventParams, Script::NO_ASSERT );
596 
597  Gfx::CCustomAnimKey* pKey = new Gfx::CEventKey( frame, eventType, pEventParams );
598  pKey->SetActive( true );
599  m_customAnimKeyList.AddToTail( pKey );
600  }
601  }
602  }

625 void CBlendChannel::ProcessCustomKeys( float startTime, float endTime )
626 {
627  // Only fire off events on the primary anim
628  if ( GetStatus() == ANIM_STATUS_ACTIVE )
629  {
630  startTime *= 60.0f;
631  endTime *= 60.0f;
632 
633  // we've just looped, so reset the keys
634  if ( m_direction == ANIM_DIR_FORWARDS )
635  {
636  if ( startTime > endTime )
637  {
638  ResetCustomKeys();
639  startTime = m_startTime;
640  }
641  }
642  else
643  {
644  if ( endTime > startTime )
645  {
646  ResetCustomKeys();
647  startTime = m_startTime;
648  }
649  }
650 
651  int customKeyCount = m_customAnimKeyList.CountItems();
652  for ( int i = 0; i < customKeyCount; i++ )
653  {
654  Gfx::CCustomAnimKey* pKey = (Gfx::CCustomAnimKey*)m_customAnimKeyList.GetItem(i);
655  if ( pKey->WithinRange( startTime, endTime ) )
656  {
657 // printf( "Processing key at %f (%f %f)\n", 0.0f, startTime, endTime );
658  pKey->ProcessKey( GetObject() );
659  }
660  else
661  {
662 // printf( "Not processing key at %f (%f %f)\n", custom_key_time, startTime, endTime );
663  }
664  }
665  }
666 
667  // TODO: reset the custom keys when it loops?
668 }

675 EAnimFunctionResult CBlendChannel::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
676 {
677  switch ( Checksum )
678  {
679  case 0x83654874: // AddAnimController
680  {
681  AddController( pParams );
682  return AF_TRUE;
683  }
684  break;
685 
686  case 0x986d274e: // RemoveAnimController
687  {
688  uint32 id;
689  pParams->GetChecksum( CRCD(0x40c698af,"id"), &id, Script::ASSERT );
690  CBaseAnimController* pAnimController = get_controller_by_id( id );
691  if ( pAnimController )
692  {
693  remove_controller( pAnimController );
694  }
695  }
696  break;
697  }
698 
699  for ( int i = 0; i < m_numControllers; i++ )
700  {
701  EAnimFunctionResult theResult = mp_controllers[i]->CallMemberFunction( Checksum, pParams, pScript );
702 
703  if ( theResult != AF_NOT_EXECUTED )
704  {
705  return theResult;
706  }
707  }
708 
709  return AF_NOT_EXECUTED;
710 }
