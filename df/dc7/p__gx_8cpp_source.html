1045 /********************************************************************************
1046  * *
1047  ********************************************************************************/
1048 void SetFog( GXFogType type,
1049  f32 startz,
1050  f32 endz,
1051  f32 nearz,
1052  f32 farz,
1053  GXColor color )
1054 {
1055 #ifndef __PLAT_WN32__
1056  f32 A, B, B_mant, C, A_f;
1057  u32 b_expn, b_m, a_hex, c_hex;
1058  u32 fsel, proj;
1059 
1060  ASSERTMSG(farz >= 0, "GDSetFog: The farz should be positive value");
1061  ASSERTMSG(farz >= nearz, "GDSetFog: The farz should be larger than nearz");
1062 
1063  fsel = (u32)(type & 0x07);
1064  proj = (u32)((type >> 3) & 0x01);
1065 
1066  if ( proj ) // ORTHOGRAPHIC
1067  {
1068  // Calculate constants a and c (TEV HW requirements).
1069  if ((farz == nearz) || (endz == startz))
1070  {
1071  // take care of the odd-ball case.
1072  A_f = 0.0f;
1073  C = 0.0f;
1074  }
1075  else
1076  {
1077  A = 1.0F / (endz - startz);
1078  A_f = (farz - nearz) * A;
1079  C = (startz - nearz) * A;
1080  }
1081 
1082  b_expn = 0;
1083  b_m = 0;
1084  }
1085  else // PERSPECTIVE
1086  {
1087  // Calculate constants a, b, and c (TEV HW requirements).
1088  if ((farz == nearz) || (endz == startz))
1089  {
1090  // take care of the odd-ball case.
1091  A = 0.0f;
1092  B = 0.5f;
1093  C = 0.0f;
1094  }
1095  else
1096  {
1097  A = (farz * nearz) / ((farz-nearz) * (endz-startz));
1098  B = farz / (farz-nearz);
1099  C = startz / (endz-startz);
1100  }
1101 
1102  B_mant = B;
1103  b_expn = 1;
1104  while (B_mant > 1.0)
1105  {
1106  B_mant /= 2;
1107  b_expn++;
1108  }
1109  while ((B_mant > 0) && (B_mant < 0.5))
1110  {
1111  B_mant *= 2;
1112  b_expn--;
1113  }
1114 
1115  A_f = A / (1 << (b_expn));
1116  b_m = (u32)(B_mant * 8388638);
1117  }
1118 
1119  a_hex = (* (u32 *) &A_f);
1120  c_hex = (* (u32 *) &C);
1121 
1122  // Write out register values.
1123  _BPCmd( TEV_FOG_PARAM_0_PS( (a_hex >> 12), TEV_FOG_PARAM_0_ID ));
1124 
1125  _BPCmd( TEV_FOG_PARAM_1( b_m, TEV_FOG_PARAM_1_ID ));
1126  _BPCmd( TEV_FOG_PARAM_2( b_expn, TEV_FOG_PARAM_2_ID ));
1127 
1128  _BPCmd( TEV_FOG_PARAM_3_PS( (c_hex >> 12), proj, fsel,
1129  TEV_FOG_PARAM_3_ID ));
1130 
1131  _BPCmd( TEV_FOG_COLOR( color.b, color.g, color.r, TEV_FOG_COLOR_ID ));

1501 /********************************************************************************
1502  * *
1503  ********************************************************************************/
1504 void SetVtxDesc( int items, ... )
1505 {
1506  u32 nnorms = 0;
1507  u32 ncols = 0;
1508  u32 ntexs = 0;
1509 
1510  u32 pnMtxIdx = GX_NONE;
1511  u32 txMtxIdxMask = 0;
1512  u32 posn = GX_DIRECT;
1513  u32 norm = GX_NONE;
1514  u32 col0 = GX_NONE;
1515  u32 col1 = GX_NONE;
1516  u32 tex0 = GX_NONE;
1517  u32 tex1 = GX_NONE;
1518  u32 tex2 = GX_NONE;
1519  u32 tex3 = GX_NONE;
1520  u32 tex4 = GX_NONE;
1521  u32 tex5 = GX_NONE;
1522  u32 tex6 = GX_NONE;
1523  u32 tex7 = GX_NONE;
1524 
1525  // Start the arguments.
1526  va_list vl;
1527  va_start( vl, items );
1528 
1529  for ( int item = 0; item < items; item++ )
1530  {
1531  // Get this item.
1532  GXAttr attribute = va_arg( vl, GXAttr );
1533  GXAttrType type = va_arg( vl, GXAttrType );
1534 
1535 #ifndef __PLAT_WN32__
1536  ASSERTMSG(((attribute >= GX_VA_PNMTXIDX) &&
1537  (attribute <= GX_VA_MAX_ATTR)),
1538  "GDSetVtxDescv: invalid attribute");
1539 
1540  ASSERTMSG(((type >= GX_NONE) &&
1541  (type <= GX_INDEX16)),
1542  "GDSetVtxDescv: invalid type");
1543 
1544  ASSERTMSG(((attribute >= GX_VA_PNMTXIDX) &&
1545  (attribute <= GX_VA_TEX7MTXIDX)) ?
1546  ((type == GX_NONE) ||
1547  (type == GX_DIRECT)) : 1,
1548  "GDSetVtxDescv: invalid type for given attribute");
1549 #endif // __PLAT_WN32__
1550 
1551  switch (attribute) {
1552 
1553  case GX_VA_PNMTXIDX: pnMtxIdx = type; break;
1554 
1555  case GX_VA_TEX0MTXIDX:
1556  txMtxIdxMask = (txMtxIdxMask & ~1) | (type << 0); break;
1557  case GX_VA_TEX1MTXIDX:
1558  txMtxIdxMask = (txMtxIdxMask & ~2) | (type << 1); break;
1559  case GX_VA_TEX2MTXIDX:
1560  txMtxIdxMask = (txMtxIdxMask & ~4) | (type << 2); break;
1561  case GX_VA_TEX3MTXIDX:
1562  txMtxIdxMask = (txMtxIdxMask & ~8) | (type << 3); break;
1563  case GX_VA_TEX4MTXIDX:
1564  txMtxIdxMask = (txMtxIdxMask & ~16) | (type << 4); break;
1565  case GX_VA_TEX5MTXIDX:
1566  txMtxIdxMask = (txMtxIdxMask & ~32) | (type << 5); break;
1567  case GX_VA_TEX6MTXIDX:
1568  txMtxIdxMask = (txMtxIdxMask & ~64) | (type << 6); break;
1569  case GX_VA_TEX7MTXIDX:
1570  txMtxIdxMask = (txMtxIdxMask & ~128) | (type << 7); break;
1571 
1572  case GX_VA_POS: posn = type; break;
1573 
1574  case GX_VA_NRM:
1575  if (type != GX_NONE)
1576  { norm = type; nnorms = 1; }
1577  break;
1578  case GX_VA_NBT:
1579  if (type != GX_NONE)
1580  { norm = type; nnorms = 2; }
1581  break;
1582 
1583  case GX_VA_CLR0: col0=type; ncols+=(col0 != GX_NONE); break;
1584  case GX_VA_CLR1: col1=type; ncols+=(col1 != GX_NONE); break;
1585 
1586  case GX_VA_TEX0: tex0=type; ntexs+=(tex0 != GX_NONE); break;
1587  case GX_VA_TEX1: tex1=type; ntexs+=(tex1 != GX_NONE); break;
1588  case GX_VA_TEX2: tex2=type; ntexs+=(tex2 != GX_NONE); break;
1589  case GX_VA_TEX3: tex3=type; ntexs+=(tex3 != GX_NONE); break;
1590  case GX_VA_TEX4: tex4=type; ntexs+=(tex4 != GX_NONE); break;
1591  case GX_VA_TEX5: tex5=type; ntexs+=(tex5 != GX_NONE); break;
1592  case GX_VA_TEX6: tex6=type; ntexs+=(tex6 != GX_NONE); break;
1593  case GX_VA_TEX7: tex7=type; ntexs+=(tex7 != GX_NONE); break;
1594  default: break;
1595  }
1596  }
1597 
1598  va_end( vl );
1599 
1600  // Write out the attributes.
1601  _CPCmd( CP_VCD_LO_ID, CP_VCD_REG_LO_PS( pnMtxIdx, txMtxIdxMask, posn, norm, col0, col1 ));
1602 
1603  _CPCmd( CP_VCD_HI_ID, CP_VCD_REG_HI( tex0, tex1, tex2, tex3, tex4, tex5, tex6, tex7 ));
1604 
1605  _XFCmd( XF_INVTXSPEC_ID, XF_INVTXSPEC( ncols, nnorms, ntexs ));
1606 }
