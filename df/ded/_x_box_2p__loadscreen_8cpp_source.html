172 void Display( char* filename, bool just_freeze, bool blank )
173 {
174  // Redundancy Check
175  if( s_loading_screen_on )
176  {
177  return;
178  }
179 
180  // Set filename up for foreign languages.
181  char name[256];
182 
183  // Seems that filename may sometimes be NULL.
184  if( filename )
185  {
186  char * p8 = &filename[strlen(filename)];
187 
188  if( IsEnglish())
189  {
190  strcpy ( name, filename );
191  }
192  else
193  {
194  switch ( XGetLanguage() ) {
195  case XC_LANGUAGE_FRENCH:
196  while ( ( *p8 != '\\' ) && ( *p8 != '/' ) ) p8--;
197  sprintf( name, "images\\frimages\\%s", &p8[1] );
198  break;
199  case XC_LANGUAGE_GERMAN:
200  while ( ( *p8 != '\\' ) && ( *p8 != '/' ) ) p8--;
201  sprintf( name, "images\\grimages\\%s", &p8[1] );
202  break;
203  default:
204  strcpy ( name, filename );
205  break;
206  }
207  }
208  }
209 
210 // Tmr::Time start_time = Tmr::GetTime();
211 
212 // Skate3 code, disabled for now.
213 # if 0

215  {
216  RwImageSetPath( "" );

218 
219  if( raster == NULL )
220  {
221  // Try the english version.
222  raster = RwRasterRead( filename );
223  if( raster == NULL )
224  {
225  Dbg_Printf( "Could not load %s\n", filename );
226  return;
227  }
228  }
229 
230  // Wait a few VBlanks for RW stuff to disappear.
