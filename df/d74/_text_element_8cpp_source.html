170 void CTextElement::SetText(const char *pText)
171 {
172  int new_length = strlen(pText) + 1;
173  Dbg_MsgAssert(new_length < vMAX_TEXT_LENGTH, ("string too long %d", new_length));
174 
175  // if the old string is not big enough, then we need to allocate a new one
176  // otherwize, overwrite it, as allocations are slow
177  if (!mp_text || (int)(strlen(mp_text)) < new_length-1) // note, does not account for actual allocated size, but will work is string are same size
178  {
179  if (mp_text)
180  delete mp_text;
181  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().FrontEndHeap());
182  mp_text = new char[new_length];
183  Mem::Manager::sHandle().PopContext();
184  }
185  // copy pText to mp_text, replacing '\m' tags with '\b' tags
186  const char *p_in = pText;
187  char *p_out = mp_text;
188  while(*p_in)
189  {
190  if (*p_in == '\\' && *(p_in+1) == '\\')
191  {
192  // output "\\" tag unchanged
193  *p_out++ = *p_in++;
194  *p_out++ = *p_in++;
195  }
196  // ignore sticky space
197  else if (*p_in == '\\' && *(p_in+1) == '_')
198  {
199  p_in++;
200  p_in++;
201  *p_out++ = ' ';
202  }
203  else if (*p_in == '\\' && *(p_in+1) == 'm')
204  {
205  *p_out++ = *p_in++;
206  *p_out++ = 'b';
207  p_in++;
208  *p_out++ = Nx::CFontManager::sMapMetaCharacterToButton(p_in++);
209  }
210  else
211  {
212  *p_out++ = *p_in++;
213  }
214  }
215  *p_out = '\0';
216 
217  if( mp_font )
218  {
219  float w, h;
220  mp_font->QueryString(mp_text, w, h); // width and height in pixels
221 
222  if (!(m_object_flags & vFORCED_DIMS))
223  {
224  SetDims(w, h);
225  }
226  }
227 
228  m_object_flags |= CScreenElement::vCHANGED_STATIC_PROPS;
229 }

1232  m_out_line = 0;
1233  m_out_char = 0;
1234  m_current_line_width = 0.0f;
1235  }
1236
