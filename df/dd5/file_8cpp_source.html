26 void restart_dirty_scripts()
27 {
28  // For each symbol, check if it got reloaded by the above load, and if it did and it is a script
29  // then run through all the existing CScript's restarting those that referred to it.
30  // This is essential, because otherwise those CScripts' program counter pointers will now be invalid.
31  CSymbolTableEntry *p_sym=GetNextSymbolTableEntry();
32  while (p_sym)
33  {
34  if (p_sym->mGotReloaded && p_sym->mType==ESYMBOLTYPE_QSCRIPT)
35  {
36  CScript *p_script=GetNextScript();
37  while (p_script)
38  {
39  if (p_script->RefersToScript(p_sym->mNameChecksum))
40  {
41  p_script->Restart();
42  }
43 
44  p_script=GetNextScript(p_script);
45  }
46 
47  // Reset the reloaded flag, otherwise it will stick on forever.
48  p_sym->mGotReloaded=false;
49  }
50  p_sym=GetNextSymbolTableEntry(p_sym);
51  }
52 }

54 // Loads a QB file.
55 // It will open the file, load it into memory and parse it, creating all the
56 // symbols (scripts, arrays, integers etc) defined within in.
57 void LoadQB(const char *p_fileName, EBoolAssertIfDuplicateSymbols assertIfDuplicateSymbols)
58 {
59  Dbg_MsgAssert(p_fileName,("NULL p_fileName"));
60 #ifndef __PLAT_NGC__
61  Dbg_MsgAssert(strcmp(p_fileName+strlen(p_fileName)-3,".qb")==0,("File does not have extension .qb. File %s",p_fileName));
62 #endif __PLAT_NGC__
63 
64  // Mick - Pip::Load is not going to load it from a Pip::Pre, just a regular pre
65  // so I'm sticking it on the top-down heap to avoid fragmentation
66  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
67  uint8 *p_qb=(uint8*)Pip::Load(p_fileName);
68  Mem::Manager::sHandle().PopContext();
69 
70  // Parse the QB, which creates all the symbols defined within it.
71  ParseQB(p_fileName,p_qb,assertIfDuplicateSymbols);
72 
73  Pip::Unload(p_fileName);
74 
75  restart_dirty_scripts();
76 }

105 // Deletes all the symbols that were defined in the passed QB file.
106 void UnloadQB(uint32 fileNameChecksum)
107 {
108  // Do nothing if the checksum is zero. This is essential, because symbols
109  // that have a mSourceFileNameChecksum of 0 are those that did not come
110  // from a qb file, ie C-functions. These must never be deleted.
111  if (!fileNameChecksum)
112  {
113  return;
114  }
115 
116  // Scan through all the symbols in the symbol table.
117  CSymbolTableEntry *p_sym=GetNextSymbolTableEntry();
118  while (p_sym)
119  {
120  if (p_sym->mSourceFileNameChecksum==fileNameChecksum)
121  {
122  // This symbol was defined in the passed qb file, so remove it.
123  CleanUpAndRemoveSymbol(p_sym);
124  // Need to start checking from the start of the list again rather than storing
125  // a p_next pointer before calling CleanUpAndRemoveSymbol.
126  // This is because CleanUpAndRemoveSymbol may cause the symbol above to move down,
127  // invalidating p_next.
128  // SPEEDOPT Starting again from the start of the list is inefficient though, because
129  // it may have to scan through thousands of symbols just to delete one, and may have to
130  // delete say a hundred. Optimize later. (Seems fast enough at the moment)
131  p_sym=NULL;
132  }
133  p_sym=GetNextSymbolTableEntry(p_sym);
134  }
135 
136  // Scan through all the existing CScripts stopping any that are referring to a script
137  // that got deleted above.
138  CScript *p_script=GetNextScript();
139  while (p_script)
140  {
141  if (!p_script->AllScriptsInCallstackStillExist())
142  {
143  p_script->ClearScript();
144  p_script->ClearEventHandlerTable();
145  }
146 
147  p_script=GetNextScript(p_script);
148  }
149 }
