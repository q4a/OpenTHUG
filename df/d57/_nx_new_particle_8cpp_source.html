68 static Mth::Vector extrema( Mth::Vector p0, Mth::Vector p1, Mth::Vector p2, float t1, float t2 )
69 {
70  Mth::Vector e1, e2, u, a, p;
71  float q;
72 
73  e1 = t1 * ( p2 - p0 );
74  e2 = t2 * ( p1 - p0 );
75  q = 1.0f / ( t1 * t2 * ( t2 - t1 ));
76  u = ( t2 * e2 - t1 * e1 ) * q; // Intitial velocity.
77  a = ( e1 - e2 ) * q; // Twice initial acceleration.
78 
79  for( int i = 0; i < 3; i++ )
80  {
81  float t = -0.5f * u[i] / a[i]; // Time of extremum for given coordinate.
82  p[i] = p0[i] + u[i] * t + a[i] * t * t; // Value of coordinate at time t.
83  }
84  p[3] = 0.0f;
85 
86  return p;
87 }

93 static Mth::Vector componentwise_min( Mth::Vector& p0, Mth::Vector& p1 )
94 {
95  Mth::Vector r = p0;
96 
97  if( p1[X] < p0[X] )
98  r[X] = p1[X];
99  if( p1[Y] < p0[Y] )
100  r[Y] = p1[Y];
101  if( p1[Z] < p0[Z] )
102  r[Z] = p1[Z];
103  if( p1[W] < p0[W] )
104  r[W] = p1[W];
105 
106  return r;
107 }

209 void CNewParticle::CalculateBoundingVolumes( void )
210 {
211  // Calculate the bounding box.
212  Mth::Vector p0, p1, p2; // box centres
213  Mth::Vector s0, s1, s2; // box dimensions
214  Mth::Vector p0_lo, p1_lo, p2_lo; // corresponding minimum possible coords
215  Mth::Vector p0_hi, p1_hi, p2_hi; // corresponding maximum possible coords
216  Mth::Vector AABB_min, AABB_max; // AABB corners
217  Mth::Vector one( 1.0f, 1.0f, 1.0f, 1.0f );
218 
219  p0 = m_params.m_BoxPos[0];
220  p1 = m_params.m_BoxPos[1];
221  p2 = m_params.m_BoxPos[2];
222 
223  p0[W] = m_params.m_Radius[0];
224  p1[W] = m_params.m_Radius[1];
225  p2[W] = m_params.m_Radius[2];
226 
227  s0 = 0.5f * m_params.m_BoxDims[0];
228  s1 = 0.5f * m_params.m_BoxDims[1];
229  s2 = 0.5f * m_params.m_BoxDims[2];
230 
231  s0[W] = 0.5f * m_params.m_RadiusSpread[0];
232  s1[W] = 0.5f * m_params.m_RadiusSpread[1];
233  s2[W] = 0.5f * m_params.m_RadiusSpread[2];
234 
235  p0_lo = p0 - s0 - p0[W]*one - s0[W]*one;
236  p1_lo = p1 - s1 - p1[W]*one - s1[W]*one;
237  p2_lo = p2 - s2 - p2[W]*one - s2[W]*one;
238 
239  p0_hi = p0 + s0 + p0[W]*one + s0[W]*one;
240  p1_hi = p1 + s1 + p1[W]*one + s1[W]*one;
241  p2_hi = p2 + s2 + p2[W]*one + s2[W]*one;
242 
243  AABB_min = componentwise_min( p0_lo, p2_lo );
244  AABB_max = componentwise_max( p0_hi, p2_hi );
245 
246  if( m_params.m_UseMidpoint )
247  {
248  Mth::Vector p;
249 
250  p = extrema( p0_lo, p1_lo, p2_lo,
251  m_params.m_Lifetime * m_params.m_MidpointPct * 0.01f,
252  m_params.m_Lifetime );
253 
254  AABB_min = componentwise_min( AABB_min, p );
255 
256  p = extrema( p0_hi, p1_hi, p2_hi,
257  m_params.m_Lifetime * m_params.m_MidpointPct * 0.01f,
258  m_params.m_Lifetime );
259 
260  AABB_max = componentwise_max( AABB_max, p );
261  }
262 
263  // Set the bounding box.
264  m_bbox.Set( AABB_min, AABB_max );
265 
266  // Now calculate the bounding sphere.
267  Mth::Vector diag = ( AABB_max - AABB_min ) * 0.5f;
268  diag[W] = 0.0f;
269  m_bsphere = AABB_min + diag;
270  m_bsphere[W] = diag.Length();
271 }
