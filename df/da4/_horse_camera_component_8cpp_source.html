146 void CHorseCameraComponent::Update()
147 {
148  // Decide whether we want to be active right now.
149  if( mp_subject_vehicle_component )
150  {
151  if( !mp_subject_vehicle_component->ShouldUpdateCamera())
152  {
153  return;
154  }
155  }
156 
157  m_frame_length = Tmr::FrameLength();
158 
159  get_controller_input();
160 
161  calculate_attractor_direction();
162 
163  // Due to rounding errors this can sometimes be > |1|, which hoses acosf(), so limit here.
164  float angular_distance = acosf( Mth::Clamp( Mth::DotProduct( m_direction, m_attractor_direction ), -1.0f, 1.0f ));
165  if( angular_distance > Mth::PI / 2.0f )
166  {
167  angular_distance = Mth::PI - angular_distance;
168  }
169 
170  bool sign = Mth::CrossProduct( m_direction, m_attractor_direction )[Y] > 0.0f;
171  float step = m_alignment_rate * angular_distance * Tmr::FrameLength();
172  if( step > angular_distance )
173  {
174  step = angular_distance;
175  }
176 
177  m_direction.RotateY(( sign ? 1.0f : -1.0f ) * step );
178 
179  m_direction.Normalize();
180  calculate_orientation_matrix();
181 
182  // Save off the orientation matrix at this point.
183  Mth::Matrix saved_mat = m_orientation_matrix;
184 
185  m_orientation_matrix.RotateYLocal( m_lookaround_heading );
186  m_orientation_matrix.RotateXLocal( m_lookaround_tilt - m_offset_tilt );
187 
188  // Test the weapon component to generate the 'sticky' targetting behavior.
189  if( mp_best_target )
190  {
191  // Need to get the rider so we can get the base object and then the weapon component (messy).
192  CCompositeObject* p_obj = mp_subject_vehicle_component->GetRider();
193  if( p_obj )
194  {
195  CWeaponComponent* p_weapon = GetWeaponComponentFromObject( p_obj );
196  p_weapon->SetCurrentTarget( mp_best_target );
197  Mth::Vector sight_pos = mp_subject->GetPos() + Mth::Vector( 0.0f, m_offset_height, 0.0f, 0.0f );
198  p_weapon->SetSightPos( sight_pos );
199  p_weapon->SetSightMatrix( m_orientation_matrix );
200 
201  float extra_heading_change, extra_tilt_change;
202  p_weapon->ProcessStickyTarget( m_lookaround_heading_delta, m_lookaround_tilt_delta, &extra_heading_change, &extra_tilt_change );
203 
204  if(( extra_heading_change != 0.0f ) || ( extra_tilt_change != 0.0f ))
205  {
206  // Reset the matrix to what it was prior to the heading and tilt adjustments.
207  m_orientation_matrix = saved_mat;
208 
209  m_lookaround_heading += extra_heading_change;
210  m_orientation_matrix.RotateYLocal( m_lookaround_heading );
211 
212  m_lookaround_tilt += extra_tilt_change;
213  m_orientation_matrix.RotateXLocal( m_lookaround_tilt - m_offset_tilt );
214  }
215  }
216  }
217 
218  calculate_position();
219 
220  ApplyCameraCollisionDetection(
221  m_pos,
222  m_orientation_matrix,
223  m_pos - m_offset_distance * m_orientation_matrix[Z],
224  m_pos - m_offset_distance * m_orientation_matrix[Z],
225  false,
226  false
227  );
228 
229  m_pos[W] = 1.0f;
230  m_orientation_matrix[X][W] = 0.0f;
231  m_orientation_matrix[Y][W] = 0.0f;
232  m_orientation_matrix[Z][W] = 0.0f;
233 
234  do_reticle();
235 
236  GetObject()->SetPos( m_pos );
237  GetObject()->SetMatrix( m_orientation_matrix );
238  GetObject()->SetDisplayMatrix( m_orientation_matrix );
239 }

280 void CHorseCameraComponent::get_controller_input( void )
281 {
282  CControlPad& control_pad = mp_input_component->GetControlPad();
283 
284  // Switch the aiming state if triggered.
285  if( control_pad.m_R3.GetTriggered())
286  {
287  control_pad.m_R3.ClearTrigger();
288 
289  // Restore screen angle.
290  screen_angle = 72.0f;
291  Nx::CViewportManager::sSetScreenAngle( screen_angle );
292 
293  if( m_state == STATE_NORMAL )
294  m_state = STATE_AIMING;
295  else if( m_state == STATE_AIMING )
296  m_state = STATE_NORMAL;
297  }
298 
299  // Fire if triggered.
300  gun_fired = false;
301  if( control_pad.m_R1.GetTriggered())
302  {
303  gun_fired = true;
304  control_pad.m_R1.ClearTrigger();
305  }
306 
307  // Switch sniper mode if triggered and in aiming mode.
308  if( control_pad.m_up.GetTriggered())
309  {
310  control_pad.m_up.ClearTrigger();
311 
312  // Unfortunately, pressing up on the left control stick will also cause m_up to be triggered, so check this
313  // is really the d-pad press.
314  if( control_pad.GetScaledLeftAnalogStickMagnitude() == 0.0f )
315  {
316  if( m_state == STATE_AIMING )
317  {
318  // This would be a good place to put the field of view setting code for the sniper rifle.
319  if( screen_angle == 72.0f )
320  {
321  screen_angle = 25.0f;
322  }
323  else if( screen_angle == 25.0f )
324  {
325  screen_angle = 7.5f;
326  }
327  else if( screen_angle == 7.5f )
328  {
329  screen_angle = 72.0f;
330  }
331  Nx::CViewportManager::sSetScreenAngle( screen_angle );
332  }
333  }
334  }
335 
336  // Deal with adjusting aim direction.
337  float tilt_target = control_pad.m_scaled_rightY;
338  float heading_target = control_pad.m_scaled_rightX;
339 
340  if( m_state == STATE_NORMAL )
341  {
342  // Aiming input should be ignored.
343  tilt_target = 0.0f;
344  heading_target = 0.0f;
345  }
346 
347  // Modulate with the variable used to damp cursor movement when aiming at a heading_target.
348  heading_target = heading_target * m_spin_modulator;
349  tilt_target = tilt_target * m_tilt_modulator;
350 
351  // Get script values.
352  float heading_ka = Script::GetFloat( CRCD( 0xe31bc279, "GunslingerHorseLookaroundHeadingKa" ), Script::ASSERT );
353  float heading_ea = Script::GetFloat( CRCD( 0x7d98eff7, "GunslingerHorseLookaroundHeadingEa" ), Script::ASSERT );
354  float heading_ks = Script::GetFloat( CRCD( 0x10a2b331, "GunslingerHorseLookaroundHeadingKs" ), Script::ASSERT );
355  float heading_es = Script::GetFloat( CRCD( 0x8e219ebf, "GunslingerHorseLookaroundHeadingEs" ), Script::ASSERT );
356 
357  // Calculate acceleration.
358  float a = heading_ka * powf( Mth::Abs( heading_target ), heading_ea ) * (( heading_target > 0.0f ) ? 1.0f : ( heading_target < 0.0f ) ? -1.0f : 0.0f );
359 
360  // Calculate max speed.
361  float s = heading_ks * powf( Mth::Abs( heading_target ), heading_es ) * (( heading_target > 0.0f ) ? 1.0f : ( heading_target < 0.0f ) ? -1.0f : 0.0f );
362 
363  m_lookaround_heading_angular_speed += a;
364 
365  if( s == 0.0f )
366  {
367  m_lookaround_heading_angular_speed = 0.0f;
368  }
369  else if((( s > 0.0f ) && ( m_lookaround_heading_angular_speed > s )) || (( s < 0.0f ) && ( m_lookaround_heading_angular_speed < s )))
370  {
371  m_lookaround_heading_angular_speed = s;
372  }
373 
374  float lookaround_heading_speed = Script::GetFloat( CRCD( 0x8cf5f0cd, "GunslingerHorseLookaroundHeadingSpeed" ), Script::ASSERT );
375 
376  // Control stick left - reticle should move left.
377  m_lookaround_heading_delta = -m_lookaround_heading_angular_speed * lookaround_heading_speed * m_frame_length;
378  m_lookaround_heading += m_lookaround_heading_delta;
379 
380  if( m_lookaround_heading > Mth::PI )
381  m_lookaround_heading = m_lookaround_heading - ( 2 * Mth::PI );
382  else if( m_lookaround_heading <= -Mth::PI )
383  m_lookaround_heading = ( 2 * Mth::PI ) + m_lookaround_heading;
384 
385  if( Script::GetInteger( CRCD( 0x9edfc7af, "GunslingerInvertAiming" )) == 0 )
386  {
387  // Negate value if vertical aiming invert is not enabled.
388  tilt_target = -tilt_target;
389  }
390 
391  // Get script values.
392  float tilt_ka = Script::GetFloat( CRCD( 0x3e8e974f, "GunslingerHorseLookaroundTiltKa" ), Script::ASSERT );
393  float tilt_ea = Script::GetFloat( CRCD( 0xa00dbac1, "GunslingerHorseLookaroundTiltEa" ), Script::ASSERT );
394  float tilt_ks = Script::GetFloat( CRCD( 0xcd37e607, "GunslingerHorseLookaroundTiltKs" ), Script::ASSERT );
395  float tilt_es = Script::GetFloat( CRCD( 0x53b4cb89, "GunslingerHorseLookaroundTiltEs" ), Script::ASSERT );
396 
397  // Calculate acceleration.
398  a = tilt_ka * powf( Mth::Abs( tilt_target ), tilt_ea ) * (( tilt_target > 0.0f ) ? 1.0f : ( tilt_target < 0.0f ) ? -1.0f : 0.0f );
399 
400  // Calculate max speed.
401  s = tilt_ks * powf( Mth::Abs( tilt_target ), tilt_es ) * (( tilt_target > 0.0f ) ? 1.0f : ( tilt_target < 0.0f ) ? -1.0f : 0.0f );
402 
403  m_lookaround_tilt_angular_speed += a;
404 
405  if( s == 0.0f )
406  {
407  m_lookaround_tilt_angular_speed = 0.0f;
408  }
409  else if((( s > 0.0f ) && ( m_lookaround_tilt_angular_speed > s )) || (( s < 0.0f ) && ( m_lookaround_tilt_angular_speed < s )))
410  {
411  m_lookaround_tilt_angular_speed = s;
412  }
413 
414  float lookaround_tilt_speed = Script::GetFloat( CRCD( 0xebbaa7e8, "GunslingerHorseLookaroundTiltSpeed" ), Script::ASSERT );
415  m_lookaround_tilt_delta = m_lookaround_tilt_angular_speed * lookaround_tilt_speed * m_frame_length;
416  m_lookaround_tilt += m_lookaround_tilt_delta;
417 
418  if( m_lookaround_tilt > 1.1f )
419  m_lookaround_tilt = 1.1f;
420  else if( m_lookaround_tilt < -0.85f )
421  m_lookaround_tilt = -0.85f;
422 
423  if( m_state == STATE_NORMAL )
424  {
425  // Smoothly move the lookaround values back to zero.
426 
427  m_lookaround_tilt -= ( m_lookaround_tilt * 0.05f );
428  m_lookaround_heading -= ( m_lookaround_heading * 0.05f );
429  }

464 void CHorseCameraComponent::calculate_attractor_direction( void )
465 {
466  if( m_state == STATE_NORMAL )
467  {
468  Mth::Vector vel_direction = -mp_subject_vehicle_component->GetVel();
469  vel_direction[Y] = 0.0f;
470  float vel = vel_direction.Length();
471  vel_direction.Normalize();
472 
473  float vel_weight = Mth::ClampMax(vel / vVELOCITY_WEIGHT_DROP_THRESHOLD, 1.0f) * Mth::DotProduct(vel_direction, -mp_subject->GetMatrix()[Z]);
474 
475  m_attractor_direction = -mp_subject->GetMatrix()[Z];
476  m_attractor_direction[Y] = 0.0f;
477  m_attractor_direction.Normalize();
478 
479  if (vel_weight > 0.0f)
480  {
481  m_attractor_direction += vel_weight * vel_direction;
482  m_attractor_direction.Normalize();
483  }
484  }
485  else if( m_state == STATE_AIMING )
486  {
487  m_attractor_direction = m_direction;
488  }
489 }
