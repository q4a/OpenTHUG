31 static int cmp( const void *p1, const void *p2 )
32 {
33  NxXbox::sCASData *p_casdata1 = (NxXbox::sCASData*)p1;
34  NxXbox::sCASData *p_casdata2 = (NxXbox::sCASData*)p2;
35 
36  uint32 mesh1 = p_casdata1->data0 >> 16;
37  uint32 mesh2 = p_casdata2->data0 >> 16;
38 
39  if( mesh1 > mesh2 )
40  {
41  return 1;
42  }
43  else if( mesh1 < mesh2 )
44  {
45  return -1;
46  }
47  else
48  {
49  uint32 indexzero1 = p_casdata1->data0 & 0xFFFF;
50  uint32 indexzero2 = p_casdata2->data0 & 0xFFFF;
51  if( indexzero1 > indexzero2 )
52  {
53  return 1;
54  }
55  else if( indexzero1 < indexzero2 )
56  {
57  return -1;
58  }
59  }
60  return 0;
61 }

69 bool CXboxMesh::build_casdata_table( const char* pFileName )
70 {
71  void *pFile = File::Open( pFileName, "rb" );
72  Dbg_MsgAssert( pFile, ( "Couldn't open CAS data file %s\n", pFileName ));
73 
74  uint32 version;
75  File::Read( &version, sizeof( uint32 ), 1, pFile );
76 
77  if( version >= 2 )
78  {
79  File::Read( &m_CASRemovalMask, sizeof( uint32 ), 1, pFile );
80  }
81 
82  File::Read( &m_numCASData, sizeof( int ), 1, pFile );
83 
84  if( m_numCASData > 0 )
85  {
86  // CAS flags.
87  mp_CASData = new NxXbox::sCASData[m_numCASData];
88 
89  for( uint32 i = 0; i < m_numCASData; ++i )
90  {
91  File::Read( &mp_CASData[i].mask, sizeof( uint32 ) * 3, 1, pFile );
92  }
93 
94  // Sort the CAS data based first on mesh, then on the first tri index, lowest to highest. This allows some efficient early-out checking
95  // during the poly removal.
96  qsort( mp_CASData, m_numCASData, sizeof( NxXbox::sCASData ), cmp );
97  }
98 
99  File::Close( pFile );
100 
101  return ( m_numCASData > 0 );
102 }

224 void CXboxMesh::SetScene( CXboxScene *p_scene )
225 {
226  mp_scene = p_scene;
227 
228  // Copy the hierarchy info from the scene so that above the p-line stuff can access it.
229  mp_hierarchyObjects = mp_scene->GetEngineScene()->mp_hierarchyObjects;
230  m_numHierarchyObjects = mp_scene->GetEngineScene()->m_numHierarchyObjects;
231 
232  // Now that we have a scene attached, resolve any cas flag data into specific indexes in the mesh data.
233  if( mp_CASData )
234  {
235  NxXbox::sCASData *p_cas_entry = mp_CASData;
236  for( uint32 entry = 0; entry < m_numCASData; ++entry, ++p_cas_entry )
237  {
238  // Get the mesh this entry references.
239  uint32 load_order = p_cas_entry->data0 >> 16;
240  NxXbox::sMesh *p_mesh = p_scene->GetEngineScene()->GetMeshByLoadOrder( load_order );
241  if( p_mesh )
242  {
243  // Get the indices of the poly referenced.
244  uint32 i0 = p_cas_entry->data0 & 0xFFFF;
245  uint32 i1 = p_cas_entry->data1 >> 16;
246  uint32 i2 = p_cas_entry->data1 & 0xFFFF;
247 
248  // For every vertex index in this mesh...
249  uint16 *p_indices = p_mesh->mp_index_buffer[0];
250  uint32 index0 = p_indices[0] & 0x7FFF;
251  uint32 index1 = p_indices[1] & 0x7FFF;
252  for( uint32 i = 2; i < p_mesh->m_num_indices[0]; ++i )
253  {
254  uint32 index2 = p_indices[i] & 0x7FFF;
255  if(( index0 == i0 ) && ( index1 == i1 ) && ( index2 == i2 ))
256  {
257  p_cas_entry->start_index = i - 2;
258  break;
259  }
260  index0 = index1;
261  index1 = index2;
262  }
263  }
264  }
265  }
266 }
