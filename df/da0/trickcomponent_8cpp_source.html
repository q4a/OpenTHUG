545 void CTrickComponent::ButtonRecord(uint Button, bool Pressed)
546 {
547  Dbg_MsgAssert(Button<PAD_NUMBUTTONS,("Bad Button"));
548 
549  if (mButtonDebounceTime[Button])
550  {
551  if ((!Pressed) || (Tmr::GetTime() > mButtonDebounceTime[Button]))
552  {
553  mButtonDebounceTime[Button] = 0;
554  }
555  else
556  {
557  return;
558  }
559  }
560 
561  if (mButtonState[Button]==Pressed)
562  {
563  return;
564  }
565 
566  mButtonState[Button]=Pressed;
567 
568 
569  uint32 ButtonChecksum=Inp::GetButtonChecksum( Button );
570 
571  // Perhaps ignore the button.
572  // This feature is used by the skater to ignore the button events used to control the balance
573  // when doing a manual, otherwise ChrisR can't do kickflips by very quickly jumping out of a
574  // grind & kickflipping by rolling from X to Square.
575  for (int i=0; i<mNumButtonsToIgnore; ++i)
576  {
577  if (ButtonChecksum==mpButtonsToIgnore[i])
578  {
579  return;
580  }
581  }
582 
583  // Generate an event.
584  if (Pressed)
585  {
586  SButtonEvent *pEvent=AddEvent(EVENT_BUTTON_PRESSED);
587  Dbg_MsgAssert(pEvent,("NULL pEvent ?"));
588  pEvent->ButtonNameChecksum=ButtonChecksum;
589  }
590  else
591  {
592  SButtonEvent *pEvent=AddEvent(EVENT_BUTTON_RELEASED);
593  Dbg_MsgAssert(pEvent,("NULL pEvent ?"));
594  pEvent->ButtonNameChecksum=ButtonChecksum;

603 // Note: This probably needs a lot of optimization, hmmm.
604 bool CTrickComponent::QueryEvents(Script::CStruct *pQuery, uint32 UsedMask, uint32 IgnoreMask)
605 {
606  if (!pQuery)
607  {
608  return false;
609  }
610 
611  if (mp_input_component->IsInputDisabled())
612  {
613  return false;
614  }
615 
616  //Script::PrintContents(pQuery);
617  Script::CComponent *pComp=pQuery->GetNextComponent(NULL);
618  if (!pComp)
619  {
620  return false;
621  }
622 
623  Dbg_MsgAssert(pComp->mType==ESYMBOLTYPE_NAME,("Bad component type, expected name"));
624  uint32 NameChecksum=pComp->mChecksum;
625  pComp=pComp->mpNext;
626 
627  uint32 p_button_names[MAX_TRICK_BUTTONS];
628  for (int b=0; b<MAX_TRICK_BUTTONS; ++b)
629  {
630  p_button_names[b]=0;
631  }
632 
633  switch (NameChecksum)
634  {
635  case 0x14fe9b7d: // AirTrickLogic
636  {
637  // Get two button names followed by a duration value.
638  uint32 Duration=get_buttons_and_duration(pComp,2,p_button_names);
639 
640  if (GetButtonState(p_button_names[1]))
641  {
642  // Search for the event saying that button 2 got pressed, and flag it as used.
643  // If this was not done, we'd be able to get two kickflips by doing left-square,
644  // then letting go of left and pressing square again.
645  int Index=mLastEvent;
646  for (int i=0; i<mNumEvents; ++i)
647  {
648  if (mpButtonEvents[Index].EventType==EVENT_BUTTON_PRESSED &&
649  mpButtonEvents[Index].ButtonNameChecksum==p_button_names[1])
650  {
651  mpButtonEvents[Index].MaybeUsed=true;
652  break;
653  }
654 
655  --Index;
656  if (Index<0)
657  {
658  Index+=MAX_EVENTS;
659  }
660  }
661 
662  if (TriggeredInLastNMilliseconds(p_button_names[0],Duration,IgnoreMask))
663  {
664  ConvertMaybeUsedToUsed(UsedMask);
665  record_last_tricks_buttons(p_button_names);
666  return true;
667  }
668  }
669  else
670  {
671  if (BothTriggeredNothingInBetween(p_button_names[0],p_button_names[1],Duration,IgnoreMask))
672  {
673  ConvertMaybeUsedToUsed(UsedMask);
674  record_last_tricks_buttons(p_button_names);
675  return true;
676  }
677  }
678  break;
679  }
680 
681  case 0xffb718bd: // PressAndRelease
682  {
683  // Get two button names followed by a duration value.
684  uint32 Duration=get_buttons_and_duration(pComp,2,p_button_names);
685 
686  int Index=mLastEvent;
687  int Count=0;
688  for (int i=0; i<mNumEvents; ++i)
689  {
690  if (Tmr::ElapsedTime(mpButtonEvents[Index].Time)>=Duration)
691  {
692  return false;
693  }
694 
695  switch (Count)
696  {
697  case 0:
698  if (!(mpButtonEvents[Index].Used & IgnoreMask) &&
699  mpButtonEvents[Index].EventType==EVENT_BUTTON_RELEASED &&
700  mpButtonEvents[Index].ButtonNameChecksum==p_button_names[1])
701  {
702  mpButtonEvents[Index].MaybeUsed=true;
703  Count=1;
704  }
705  break;
706  case 1:
707  if (!(mpButtonEvents[Index].Used & IgnoreMask) &&
708  mpButtonEvents[Index].EventType==EVENT_BUTTON_PRESSED &&
709  mpButtonEvents[Index].ButtonNameChecksum==p_button_names[0])
710  {
711  mpButtonEvents[Index].MaybeUsed=true;
712  ConvertMaybeUsedToUsed(UsedMask);
713  record_last_tricks_buttons(p_button_names);
714  return true;
715  }
716  break;
717  default:
718  break;
