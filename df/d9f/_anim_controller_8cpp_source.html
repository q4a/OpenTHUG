64 float CAnimChannel::get_new_anim_time( void )
65 {
66 #ifdef __NOPT_ASSERT__
67  float old_time = m_currentTime;
68 #endif
69 
70  float new_time=0;
71  float new_hold_time=0;
72  float new_loop_time=0;
73 
74  m_animComplete = false;
75 
76  if ( m_loopingType == LOOPING_WOBBLE )
77  {
78  Dbg_MsgAssert( 0, ( "Was not expecting a wobble looping type in this controller" ) );
79  return 0.0f;
80  }
81 
82  if ( m_direction == ANIM_DIR_FORWARDS )
83  {
84  new_time = m_currentTime + ( m_animSpeed * Tmr::FrameRatio() );
85  m_animComplete = new_time > m_endTime;
86 
87  if (m_startTime == m_endTime)
88  {
89  new_loop_time = m_endTime;
90  }
91  else
92  {
93  new_loop_time = m_startTime + ( new_time - m_endTime );
94  while ( new_loop_time > m_endTime )
95  {
96  // when the FrameRatio() > 1.0f, it's possible that
97  // we need to subtract more than once, esp. with short anims
98  new_loop_time = m_startTime + ( new_loop_time - m_endTime );
99  }
100  }
101  new_hold_time = m_endTime;
102  }
103  else if ( m_direction == ANIM_DIR_BACKWARDS )
104  {
105  new_time = m_currentTime - ( m_animSpeed * Tmr::FrameRatio() );
106  m_animComplete = new_time < m_startTime;
107  if (m_startTime == m_endTime)
108  {
109  new_loop_time = m_endTime;
110  }
111  else
112  {
113  new_loop_time = m_endTime + ( new_time - m_startTime );
114  while ( new_loop_time < m_startTime )
115  {
116  // when the FrameRatio() > 1.0f, it's possible that
117  // we need to subtract more than once, esp. with short anims
118  new_loop_time = m_endTime + ( new_loop_time - m_startTime );
119  }
120  }
121  new_hold_time = m_startTime;
122  }
123  else
124  {
125  Dbg_Assert( 0 );
126  }
127 
128  if ( m_animComplete )
129  {
130  if ( m_loopingType == LOOPING_HOLD )
131  {
132  new_time = new_hold_time;
133  }
134  else if ( m_loopingType == LOOPING_PINGPONG )
135  {
136  m_direction = ( m_direction == ANIM_DIR_FORWARDS ) ? ANIM_DIR_BACKWARDS : ANIM_DIR_FORWARDS;
137  new_time = new_hold_time;
138  }
139  else if ( m_loopingType == LOOPING_CYCLE )
140  {
141  new_time = new_loop_time;
142  }
143  }
144 
145  // Ken: This is temporary, to prevent the following assert going off when running a replay.
146  // Eventually the replays will be fixed to always recreate the correct new_time
147  if ( new_time < m_startTime )
148  {
149  new_time = m_startTime;
150  }
151  if ( new_time > m_endTime )
152  {
153  new_time = m_endTime;
154  }
155 
156 #ifdef __NOPT_ASSERT__
157  if ( new_time < m_startTime || new_time > m_endTime )
158  {
159  Dbg_Message( "old_time = %f", old_time );
160  Dbg_Message( "new_time = %f", new_time );
161  Dbg_Message( "new_loop_time = %f", new_loop_time );
162  Dbg_Message( "new_hold_time = %f", new_hold_time );
163  Dbg_Message( "m_startTime = %f", m_startTime );
164  Dbg_Message( "m_endTime = %f", m_endTime );
165  Dbg_Message( "m_loopingType = %d", m_loopingType );
166  Dbg_Message( "m_direction = %d", m_direction );
167  Dbg_Message( "m_animComplete = %d", m_animComplete );
168  Dbg_Message( "m_animName = %s", Script::FindChecksumName( m_animName ) );
169  Dbg_Assert( 0 );
170  }
171 #endif
172 
173  return new_time;
174 }

204 void CAnimChannel::Reset()
205 {
206  m_startTime = 0.0f;
207  m_currentTime = 0.0f;
208  m_endTime = 0.0f;
209  m_animSpeed = 0.0f;
210  m_direction = ANIM_DIR_FORWARDS;
211  m_loopingType = LOOPING_CYCLE;
212  m_animComplete = false;
213 
214  // Ken: Changed this to be a hard wired checksum rather than calling Script::GenerateCRC
215  // The reason for this is that CAnimChannel::Reset() gets called from
216  // CSkaterTrackingInfo::Reset() in replay.cpp, which gets called from the CSkaterTrackingInfo
217  // constructor. There is a static instance of a CSkaterTrackingInfo in replay.cpp, so the
218  // constructor will get called at some point before main(). Since Script::GenerateCRC accesses
219  // an array of data, there is a chance that it might get called before the array is initialised,
220  // since the order of execution is not well defined.
221  // It probably would be OK, I just want to be safe.
222  m_animName = 0xf60c9090; // unassigned <-- just giving it a non-zero value, so that
223  // we can be sure it's getting initialized correctly...
224 }

418 void CAnimChannel::GetDebugInfo( Script::CStruct* p_info )
419 {
420 #ifdef __DEBUG_CODE__
421  Dbg_MsgAssert( p_info, ( "NULL p_info sent to CAnimChannel::GetDebugInfo" ) );
422 
423  // put the name first
424  p_info->AddChecksum( "m_animName", m_animName );
425 
426  uint32 looping_type_checksum = 0;
427  switch ( m_loopingType )
428  {
429  case 0:
430  looping_type_checksum = CRCD( 0x3f00df04, "LOOPING_HOLD" );
431  break;
432  case 1:
433  looping_type_checksum = CRCD( 0xfbada576, "LOOPING_CYCLE" );
434  break;
435  case 2:
436  looping_type_checksum = CRCD( 0x792b924b, "LOOPING_PINGPONG" );
437  break;
438  case 3:
439  looping_type_checksum = CRCD( 0x90423ff2, "LOOPING_WOBBLE" );
440  break;
441  default:
442  Dbg_MsgAssert( 0, ( "Unknown looping type found in CAnimChannel::GetDebugInfo. Was the enum changed?" ) );
443  break;
444  }
445  if ( looping_type_checksum )
446  p_info->AddChecksum( CRCD(0x84b06b3,"m_loopingType"), looping_type_checksum );
447 
448  switch ( m_direction )
449  {
450  case 0:
451  p_info->AddChecksum( CRCD(0x9f648df4,"m_direction"), CRCD( 0xdf8b2509, "ANIM_DIR_FORWARDS" ) );
452  break;
453  case -1:
454  p_info->AddChecksum( CRCD(0x9f648df4,"m_direction"), CRCD( 0xfdf0c4a4, "ANIM_DIR_BACKWARDS" ) );
455  break;
456  default:
457  Dbg_MsgAssert( 0, ( "Unknown direction found in CAnimChannel::GetDebugInfo. Was the enum changed?" ) );
458  break;
459  }
460 
461  p_info->AddFloat( CRCD(0x8fbac25e,"m_startTime"), m_startTime );
462  p_info->AddFloat( CRCD(0xb73f0945,"m_currentTime"), m_currentTime );
463  p_info->AddFloat( CRCD(0xf1cef774,"m_endTime"), m_endTime );
464  p_info->AddFloat( CRCD(0xaa1e73de,"m_animSpeed"), m_animSpeed );
465  p_info->AddInteger( CRCD(0x2d574054,"m_animComplete"), m_animComplete );
466 #endif
467 }
