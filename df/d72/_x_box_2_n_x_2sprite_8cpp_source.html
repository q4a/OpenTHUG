177 void SDraw2D::InsertDrawList( void )
178 {
179  if( !sp_2D_draw_list || ( m_pri <= sp_2D_draw_list->m_pri ))
180  {
181  // Empty or start of list.
182  mp_next = sp_2D_draw_list;
183  sp_2D_draw_list = this;
184  }
185  else
186  {
187  SDraw2D *p_cur = sp_2D_draw_list;
188 
189  // Find where to insert.
190  while( p_cur->mp_next )
191  {
192  if( m_pri <= p_cur->mp_next->m_pri )
193  break;
194 
195  p_cur = p_cur->mp_next;
196  }
197 
198  // Insert at this point.
199  mp_next = p_cur->mp_next;
200  p_cur->mp_next = this;
201  }
202 }

294 {
295  // Sprites are based on .img files, which in turn are converted from .png files, which are upside down,
296  // so reverse the v components of the texture coordinates.
297  float u0 = 0.0f;
298  float v0 = 1.0f;
299  float u1, v1;
300 
301  if( mp_texture )
302  {
303  u1 = (float)mp_texture->ActualWidth / (float)mp_texture->BaseWidth;
304  v1 = 1.0f - ((float)mp_texture->ActualHeight / (float)mp_texture->BaseHeight );
305  }
306  else
307  {
308  u1 = 1.0f;
309  v1 = 0.0f;
310  }
311 
312  // Check for flip.
313  float abs_scale_x = m_scale_x;
314  float abs_scale_y = m_scale_y;
315  if( abs_scale_x < 0.0f )
316  {
317  float temp = u0;
318  u0 = u1;
319  u1 = temp;
320  abs_scale_x = -abs_scale_x;
321  }
322  if( abs_scale_y < 0.0f )
323  {
324  float temp = v0;
325  v0 = v1;
326  v1 = temp;
327  abs_scale_y = -abs_scale_y;
328  }
329 
330  float x0 = -( m_xhot * abs_scale_x );
331  float y0 = -( m_yhot * abs_scale_y );
332  float x1 = x0 + ( m_width * abs_scale_x );
333  float y1 = y0 + ( m_height * abs_scale_y );
334 
335  DWORD current_color = ( m_rgba & 0xFF00FF00 ) | (( m_rgba & 0xFF ) << 16 ) | (( m_rgba & 0xFF0000 ) >> 16 );
336  DWORD* p_push;
337 
338  if( m_rot != 0.0f )
339  {
340  Mth::Vector p0( x0, y0, 0.0f, 0.0f );
341  Mth::Vector p1( x1, y0, 0.0f, 0.0f );
342  Mth::Vector p2( x0, y1, 0.0f, 0.0f );
343  Mth::Vector p3( x1, y1, 0.0f, 0.0f );
344

346  p1.RotateZ( m_rot );
347  p2.RotateZ( m_rot );
348  p3.RotateZ( m_rot );

357  p3[Y] = SCREEN_CONV_Y( p3[Y] + m_ypos );
358

361  p_push[0] = D3DPUSH_ENCODE( D3DPUSH_SET_BEGIN_END, 1 );
362  p_push[1] = D3DPT_QUADLIST;
