308 void setup_weighted_mesh_vertex_shader( void *p_root_matrix, void *p_bone_matrices, int num_bone_matrices )
309 {
310  XGMATRIX dest_matrix;
311  XGMATRIX inverse_view_matrix;
312  XGMATRIX temp_matrix;
313  XGMATRIX projMatrix;
314  XGMATRIX viewMatrix;
315  XGMATRIX worldMatrix;
316 
317  // Projection matrix.
318  XGMatrixTranspose( &projMatrix, &EngineGlobals.projection_matrix );
319 
320  // View matrix.
321  XGMatrixTranspose( &viewMatrix, &EngineGlobals.view_matrix );
322  viewMatrix.m[3][0] = 0.0f;
323  viewMatrix.m[3][1] = 0.0f;
324  viewMatrix.m[3][2] = 0.0f;
325  viewMatrix.m[3][3] = 1.0f;
326 
327  // World space transformation matrix. (3x3 rotation plus 3 element translation component).
328  worldMatrix.m[0][0] = ((float*)p_root_matrix )[0];
329  worldMatrix.m[0][1] = ((float*)p_root_matrix )[1];
330  worldMatrix.m[0][2] = ((float*)p_root_matrix )[2];
331  worldMatrix.m[0][3] = ((float*)p_root_matrix )[3];
332  worldMatrix.m[1][0] = ((float*)p_root_matrix )[4];
333  worldMatrix.m[1][1] = ((float*)p_root_matrix )[5];
334  worldMatrix.m[1][2] = ((float*)p_root_matrix )[6];
335  worldMatrix.m[1][3] = ((float*)p_root_matrix )[7];
336  worldMatrix.m[2][0] = ((float*)p_root_matrix )[8];
337  worldMatrix.m[2][1] = ((float*)p_root_matrix )[9];
338  worldMatrix.m[2][2] = ((float*)p_root_matrix )[10];
339  worldMatrix.m[2][3] = ((float*)p_root_matrix )[11];
340  worldMatrix.m[3][0] = 0.0f;
341  worldMatrix.m[3][1] = 0.0f;
342  worldMatrix.m[3][2] = 0.0f;
343  worldMatrix.m[3][3] = 1.0f;
344 
345  // Calculate composite world->view->projection matrix.
346  XGMatrixMultiply( &temp_matrix, &viewMatrix, &worldMatrix );
347  XGMatrixMultiply( &dest_matrix, &projMatrix, &temp_matrix );
348 
349  // Switch to 192 constant mode, removing the lock on the reserved constants c-38 and c-37.
350 // D3DDevice_SetShaderConstantMode( D3DSCM_192CONSTANTS | D3DSCM_NORESERVEDCONSTANTS );
351 
352  // Load up the combined world, camera & projection matrix.
353  D3DDevice_SetVertexShaderConstantFast( VSCONST_REG_TRANSFORM_OFFSET, (void*)&dest_matrix, VSCONST_REG_TRANSFORM_SIZE );
354  D3DDevice_SetVertexShaderConstantFast( VSCONST_REG_WORLD_TRANSFORM_OFFSET, (void*)&worldMatrix, VSCONST_REG_WORLD_TRANSFORM_SIZE );
355 
356  // We want to transform the light directions by the inverse of the world transform - this means we don't have to transform
357  // the normal by the world transform for every vertex in the vertex shader. However, the function D3DXVec3TransformNormal
358  // (used below) does the inverse transform for us, so need to actually figure the inverse...
359 // XGMATRIX inverse_world_transform = worldMatrix;
360 // D3DXMatrixInverse( &inverse_world_transform, NULL, &worldMatrix );
361 
362  float directional_light_color[24];
363  CopyMemory( directional_light_color, EngineGlobals.directional_light_color, sizeof( float ) * 24 );
364 
365  XGVec3TransformNormal((XGVECTOR3*)&directional_light_color[0], (XGVECTOR3*)&EngineGlobals.directional_light_color[0], &worldMatrix );
366  XGVec3TransformNormal((XGVECTOR3*)&directional_light_color[8], (XGVECTOR3*)&EngineGlobals.directional_light_color[8], &worldMatrix );
367  XGVec3TransformNormal((XGVECTOR3*)&directional_light_color[16], (XGVECTOR3*)&EngineGlobals.directional_light_color[16], &worldMatrix );
368 
369  XGVec3Normalize((XGVECTOR3*)&directional_light_color[0], (XGVECTOR3*)&directional_light_color[0] );
370  XGVec3Normalize((XGVECTOR3*)&directional_light_color[8], (XGVECTOR3*)&directional_light_color[8] );
371  XGVec3Normalize((XGVECTOR3*)&directional_light_color[16], (XGVECTOR3*)&directional_light_color[16] );
372 
373  // Load up the directional light data.
374  D3DDevice_SetVertexShaderConstantFast( VSCONST_REG_DIR_LIGHT_OFFSET, (void*)directional_light_color, 6 );
375 
376  // Load up the ambient light color.
377  D3DDevice_SetVertexShaderConstantFast( VSCONST_REG_AMB_LIGHT_OFFSET, (void*)EngineGlobals.ambient_light_color, 1 );
378 
379  // Calculate and load up the model-relative camera position.
380  EngineGlobals.model_relative_cam_position = XGVECTOR3( EngineGlobals.cam_position.x - worldMatrix.m[0][3],
381  EngineGlobals.cam_position.y - worldMatrix.m[1][3],
382  EngineGlobals.cam_position.z - worldMatrix.m[2][3] );
383  XGVec3TransformNormal( &EngineGlobals.model_relative_cam_position, &EngineGlobals.model_relative_cam_position, &worldMatrix );
384 
385  float specular_attribs[4] = { EngineGlobals.model_relative_cam_position.x,
386  EngineGlobals.model_relative_cam_position.y,
387  EngineGlobals.model_relative_cam_position.z,
388  0.0f };
389  D3DDevice_SetVertexShaderConstantFast( VSCONST_REG_CAM_POS_OFFSET, (void*)&specular_attribs, 1 );
390 
391  // Safety check here to limit number of bones to that available.
392  if( num_bone_matrices > 55 )
393  {
394  num_bone_matrices = 55;
395  }
396 
397  DWORD* p_bone_element = (DWORD*)p_bone_matrices;
398 
399  // Begin state block to set vertex shader constants for bone transforms.
400  DWORD *p_push;
401  EngineGlobals.p_Device->BeginState(( num_bone_matrices * ( 12 + 1 )) + 3, &p_push );
402 
403  // 1 here isn't the parameter for SET_TRANSFORM_CONSTANT_LOAD; rather, it's the number of dwords written to that register.
404  p_push[0] = D3DPUSH_ENCODE( D3DPUSH_SET_TRANSFORM_CONSTANT_LOAD, 1 );
405 
406  // Here is the actual parameter for SET_TRANSFORM_CONSTANT_LOAD. Always add 96 to the constant register.
407  p_push[1] = VSCONST_REG_MATRIX_OFFSET + 96;
408 
409  p_push += 2;
410 
411  while( num_bone_matrices > 0 )
412  {
413  // A 3x4 matrix is 12 dwords. You can encode a maximum of 32 dwords to D3DPUSH_SET_TRANSFORM_CONSTANT before needing another D3DPUSH_ENCODE.
414  p_push[0] = D3DPUSH_ENCODE( D3DPUSH_SET_TRANSFORM_CONSTANT, 12 );
415 
416  p_push[1] = p_bone_element[0];
417  p_push[2] = p_bone_element[4];
418  p_push[3] = p_bone_element[8];
419  p_push[4] = p_bone_element[12];
420 
421  p_push[5] = p_bone_element[1];
422  p_push[6] = p_bone_element[5];
423  p_push[7] = p_bone_element[9];
424  p_push[8] = p_bone_element[13];
425 
426  p_push[9] = p_bone_element[2];
427  p_push[10] = p_bone_element[6];
428  p_push[11] = p_bone_element[10];
429  p_push[12] = p_bone_element[14];
430 
431  --num_bone_matrices;
432 
433  p_bone_element += 16;
434  p_push += 13;
435  }
436 
437  EngineGlobals.p_Device->EndState( p_push );
438 
439  // Load up the replacement registers for c-38 and c-37
440  // The z value is 2^24 - to take 1.0f to the max z buffer value for a 24bit z buffer.
441  static float homogenous_to_screen_reg[8] = { 320.0f, -240.0f, 1.6777215e7f, 0.0f, 320.53125f, 240.53125f, 0.0f, 0.0f };
442 
443  if( EngineGlobals.is_orthographic )
444  {
445  homogenous_to_screen_reg[0] = 128.0f;
446  homogenous_to_screen_reg[1] = -128.0f;
447 
448  homogenous_to_screen_reg[2] = 1.6777215e7f;
449 
450  homogenous_to_screen_reg[4] = 128.53125f;
451  homogenous_to_screen_reg[5] = 128.53125f;
452  }
453  else
454  {
455  homogenous_to_screen_reg[0] = (float)EngineGlobals.viewport.Width * 0.5f;
456  homogenous_to_screen_reg[1] = (float)EngineGlobals.viewport.Height * -0.5f;
457 
458  homogenous_to_screen_reg[2] = ( EngineGlobals.zstencil_depth == 16 ) ? 65535.0f : 1.6777215e7f;
459 
460  homogenous_to_screen_reg[4] = (float)NxXbox::EngineGlobals.viewport.X + ((float)NxXbox::EngineGlobals.viewport.Width * 0.5f ) + 0.53125f;
461  homogenous_to_screen_reg[5] = (float)NxXbox::EngineGlobals.viewport.Y + ((float)NxXbox::EngineGlobals.viewport.Height * 0.5f ) + 0.53125f;
462  }
463 
464  D3DDevice_SetVertexShaderConstantFast( 94, (void*)homogenous_to_screen_reg, 2 );
465 }
