126 void CMotionComponent::ProcessWait( Script::CScript * pScript )
127 {
128  switch (pScript->GetWaitType())
129  {
130  case Script::WAIT_TYPE_OBJECT_MOVE:
131  if ( !IsMoving() )
132  {
133  pScript->ClearWait();
134  }
135  break;
136 
137  case Script::WAIT_TYPE_OBJECT_JUMP_FINISHED:
138  if ( !(m_movingobj_status & MOVINGOBJ_STATUS_JUMPING) )
139  {
140  pScript->ClearWait();
141  }
142  break;
143 
144  case Script::WAIT_TYPE_OBJECT_ROTATE:
145  if ( !IsRotating() )
146  {
147  pScript->ClearWait();
148  }
149  break;
150  case Script::WAIT_TYPE_OBJECT_STOP_FINISHED:
151  if ( !( m_vel_z > 0.0f ) )
152  {
153  pScript->ClearWait();
154  }
155  break;
156  default:
157  Dbg_MsgAssert(0,("\n%s\nWait type of %d not supported by CMotionComponent",pScript->GetScriptInfo(),pScript->GetWaitType()));
158  break;
159  }
160 }

167 void CMotionComponent::Update()
168 {
169 
170  m_time = Tmr::FrameLength();
171 
172  if ( m_movingobj_status & MOVINGOBJ_STATUS_HOVERING )
173  {
174  // Having these functions next to each other here does mean that objects cannot be
175  // moving and hovering at the same time, but currently nothing requires that.
176  UndoHover();
177  ApplyHover();
178  }
179 
180  if ( m_movingobj_status & MOVINGOBJ_STATUS_ROTXYZ )
181  {
182  RotUpdate( );
183  }
184 
185  if ( m_movingobj_status & MOVINGOBJ_STATUS_QUAT_ROT )
186  {
187  QuatRot( );
188  }
189 
190  if ( m_movingobj_status & MOVINGOBJ_STATUS_MOVETO )
191  {
192  if ( Move( ) )
193  {
194  m_movingobj_status &= ~MOVINGOBJ_STATUS_MOVETO;
195  }
196  }
197  else if ( m_movingobj_status & MOVINGOBJ_STATUS_ON_PATH )
198  {
199  FollowPath( );
200  }
201  else if ( m_movingobj_status & MOVINGOBJ_STATUS_FOLLOWING_LEADER )
202  {
203  Dbg_MsgAssert( !(m_movingobj_status & MOVINGOBJ_STATUS_QUAT_ROT), ("Cannot have object rotating when it is following a leader"));
204  Dbg_MsgAssert( !(m_movingobj_status & MOVINGOBJ_STATUS_ROTXYZ), ("Cannot have object rotating when it is following a leader"));
205  FollowLeader( );
206  }
207 
208  if ( m_movingobj_status & MOVINGOBJ_STATUS_JUMPING )
209  {
210  DoJump();
211  GetObject()->m_pos = m_jump_pos;
212  // m_pos[Y]=m_jump_y;
213  }
214 
215 
216  // TODO TODO TODO TODO TODO TODO TODO TODO TODO TODO TODO TODO
217  // BIT OF A PATCH, as the skater has a motion component
218  // but we don't want to set his display matrix directly here.
219  if (GetObject()->GetID() > 15)
220  {
221  GetObject()->SetDisplayMatrix(GetObject()->GetMatrix());
222  }
223 
224 }

261  {
262  // miles per hour then...
263  *pVel = MPH_TO_IPS( *pVel );

308 {
309  uint32 t=Tmr::ElapsedTime(0)%m_hover_period;
310  m_hover_offset=m_hover_amp*sinf(t*2*3.141592653f/m_hover_period);

325  // If it is doing a Quat rot, then pretend it is not hovering.
326  // This is so that the rocking buoy in the shipyard is not indicated as
