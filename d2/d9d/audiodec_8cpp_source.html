315 int audioDecSendToIOP(AudioDec *ad)
316 {
317  int pd0, pd1, d0, d1;
318  u_char *ps0, *ps1;
319  int s0, s1;
320  int count_sent = 0;
321  int countAdj;
322  int pos = 0;
323 
324  switch (ad->state) {
325  case AU_STATE_INIT:
326  return 0;
327  break;
328 
329  case AU_STATE_PRESET:
330  pd0 = ad->iopBuff + (ad->totalBytesSent) % ad->iopBuffSize;
331  d0 = ad->iopBuffSize - ad->totalBytesSent;
332  pd1 = 0;
333  d1 = 0;
334  break;
335 
336  case AU_STATE_PLAY:
337  pos = ((sceSdRemote(1, rSdBlockTransStatus, AUTODMA_CH)
338  & 0x00FFFFFF) - ad->iopBuff);
339  iopGetArea(&pd0, &d0, &pd1, &d1, ad, pos);
340  break;
341 
342  case AU_STATE_PAUSE:
343  return 0;
344  break;
345  }
346 
347  ps0 = ad->data + (ad->put - ad->count + ad->size) % ad->size;
348  ps1 = ad->data;
349 
350  // adjust to UNIT_SIZE boundary
351  countAdj = (ad->count / UNIT_SIZE) * UNIT_SIZE;
352 
353  s0 = min(ad->data + ad->size - ps0, countAdj);
354  s1 = countAdj - s0;
355 
356  if (d0 + d1 >= UNIT_SIZE && s0 + s1 >= UNIT_SIZE) {
357  count_sent = sendToIOP2area(pd0, d0, pd1, d1, ps0, s0, ps1, s1);
358  }
359 
360  ad->count -= count_sent;
361 
362  ad->totalBytesSent += count_sent;
363  ad->iopLastPos = (ad->iopLastPos + count_sent) % ad->iopBuffSize;
364 
365  return count_sent;
366 }

399 static int sendToIOP2area(int pd0, int d0, int pd1, int d1,
400  u_char *ps0, int s0, u_char *ps1, int s1)
401 {
402  if (d0 + d1 < s0 + s1) {
403  int diff = (s0 + s1) - (d0 + d1);
404  if (diff >= s1) {
405  s0 -= (diff - s1);
406  s1 = 0;
407  } else {
408  s1 -= diff;
409  }
410  }
411 
412  //
413  // (d0 + d1 >= s0 + s1)
414  //
415  if (s0 >= d0) {
416  sendToIOP(pd0, ps0, d0);
417  sendToIOP(pd1, ps0 + d0, s0 - d0);
418  sendToIOP(pd1 + s0 - d0, ps1, s1);
419  } else { // s0 < d0
420  if (s1 >= d0 - s0) {
421  sendToIOP(pd0, ps0, s0);
422  sendToIOP(pd0 + s0, ps1, d0 - s0);
423  sendToIOP(pd1, ps1 + d0 - s0, s1 - (d0 - s0));
424  } else { // s1 < d0 - s0
425  sendToIOP(pd0, ps0, s0);
426  sendToIOP(pd0 + s0, ps1, s1);
427  }
428  }
429  return s0 + s1;
430 }
