144 // Decode an LZSS encoded stream
145 // Runs at approx 12MB/s on PS2 without scratchpad (which slows it down in C)
146 // a 32x CD would run at 4.8MB/sec, although we seem to get a lot less than this

153 static int aram_len = 0;
154 static int buffer_bytes = 0;
155 static int buffer_offset = 0;
156 static unsigned char get_byte( unsigned char * p_buffer )
157 {
158  if ( buffer_bytes == 0 )
159  {
160  int toread;
161  if ( aram_len >= INPUT_BUFFER_SIZE )
162  {
163  toread = INPUT_BUFFER_SIZE;
164  }
165  else
166  {
167  toread = aram_len;
168  }

170  NsDMA::toMRAM( p_buffer, p_aram_in, toread );
171  p_aram_in += INPUT_BUFFER_SIZE;
172  aram_len -= INPUT_BUFFER_SIZE;
173  buffer_bytes = toread;
174  buffer_offset = 0;
175  }
176  unsigned char bb = p_buffer[buffer_offset];
177  buffer_bytes--;
178  buffer_offset++;
179  return bb;

183 void DecodeLZSS(unsigned char *pIn, unsigned char *pOut, int Len) /* Just the reverse of Encode(). */
184 {
185  int i, j, k, r, c;
186 // uint64 LongWord;
187 // int bytes = 0;
188 // unsigned char *pScratch;
189  unsigned int flags;
190 
191  // Ensure we have decent values for the decode length
192  Dbg_Assert(( Len >= 0 ) && ( Len < 0x2000000 ));
193 
194  if(( Len < 0 ) || ( Len >= 0x2000000 ))
195  {
196  while( 1 );
197  }
198 #ifdef __PRE_ARAM__
199  unsigned char p_buffer[INPUT_BUFFER_SIZE];
200  p_aram_in = (uint32)pIn;
201  aram_len = Len;
202 #endif // __PRE_ARAM__

206 // int len = Len;
207 
208 // int OutBytes = 4;
209 // int OutWord = 0;
210 
211  #if USE_BUFFER
212  for (i = 0; i < RINGBUFFERSIZE - MATCHLIMIT; i++)
213  text_buf[i] = ' ';
214  r = RINGBUFFERSIZE - MATCHLIMIT;
215  #else
216  r = RINGBUFFERSIZE - MATCHLIMIT;
217  #endif
218  flags = 0;
219  for ( ; ; )
220  {
221  if (((flags >>= 1) & 256) == 0)
222  {
223  ReadInto(c);
224  flags = c | 0xff00; /* uses higher byte cleverly */
225  } /* to count eight */
226  if (flags & 1)
227  {
228  ReadInto(c);
229  // putc(c, outfile);
230  WriteOut(c);
231  #if USE_BUFFER
232  text_buf[r++] = c;
233  r &= (RINGBUFFERSIZE - 1);
234  #else
235  r++;
236 // r &= (RINGBUFFERSIZE - 1); // don't need to wrap r until it is used
237  #endif
238  }
239  else

241  ReadInto(i);

251  if (i>=r) // if offset > r, then

253  pStream += i; // add in the offset to the base
254  r+=j; // add size to r

261  {

836  // music and streams.

838 // scePrintf("Loading PRE file %s...\n", pFilename);
839 # endif
840 
841  char fullname[256];
842  sprintf(fullname, "pre\\%s", pFilename);
843 
844 # ifdef __PLAT_XBOX__
845  // Replace the .pre extension (if one exists) with .prx for Xbox PRE file.
846  if( strrchr( pFilename, '.' ) && ( strlen( fullname ) > 4 ))
847  {
848  fullname[strlen( fullname ) - 1] = 'x';
849  }
850 # endif
851 
852 # if !defined( __PLAT_NGC__ ) || ( defined( __PLAT_NGC__ ) && !defined( __NOPT_FINAL__ ) )
853  Tmr::Time basetime = Tmr::ElapsedTime(0);
854 #endif

858 
859  // Try loading asynchronously
860  if (async)

863

865  if (p_fileHandle)
866  {
867  Dbg_MsgAssert(strlen(pFilename) < MAX_COMPACT_FILE_NAME, ("Pre file name %s is greater than %d bytes", pFilename, MAX_COMPACT_FILE_NAME - 1));

871  m_pending_pre_files[m_num_pending_pre_files++].mp_file_handle = p_fileHandle;

877  if (useBottomUpHeap)
878  {
879  pFile = new (Mem::Manager::sHandle().BottomUpHeap()) uint8[file_size];
880  }
881  else

883  pFile = new (Mem::Manager::sHandle().TopDownHeap()) uint8[file_size];
884  }

887  // Set the callback
888  p_fileHandle->SetCallback(async_callback, (unsigned int) this, (unsigned int) pFile);
889 
890  // read the file in
891  p_fileHandle->Read( pFile, 1, file_size );
892  return;
893  }
894  }
895 
896  // If we got here, we didn't do an async load
897  file_size = CanFileBeLoadedQuickly( fullname );
898  if ( file_size )
899  {
900  Mem::PushMemProfile((char*)fullname);
901  if (useBottomUpHeap)
902  {
903  pFile = new (Mem::Manager::sHandle().BottomUpHeap()) uint8[file_size];
904  }
905  else
906  {
907  pFile = new (Mem::Manager::sHandle().TopDownHeap()) uint8[file_size];
908  }
909  Mem::PopMemProfile();
910  bool fileLoaded= LoadFileQuicklyPlease( fullname, pFile );
911  if ( !fileLoaded )
912  {
913  printf( "pre file %s failed to load quickly.\n", fullname );
914  Dbg_MsgAssert( 0,( "Fire Matt - pre file didn't load quickly." ));
915  }
916  }
917  else
918  {
919  void *fp = File::Open(fullname, "rb");
920  if (!fp)
921  {
922  // always run the code below if CD build
923  if (dont_assert || Config::CD())
924  {
925  printf("couldn't open %s\n", fullname);
926  return;
927  }
928  Dbg_MsgAssert(0,( "couldn't open %s\n", fullname));
929  }
930 
931 
932 #ifdef __PRE_ARAM__
933  File::Read(&file_size, 4, 1, fp);
934  Dbg_MsgAssert(file_size > 0,( "%s has incorrect file size\n", fullname));
935  if (Config::CD())
936  {
937  if (file_size <= 0) printf("%s has incorrect file size\n", fullname);
938  }
939 
940  // Stream the file into ARAM.
941  #define PRE_BUFFER_SIZE (16 * 1024)
942  char p_buffer[PRE_BUFFER_SIZE];
943 
944  uint32 p_aram;
945  if (useBottomUpHeap)
946  {
947  p_aram = NsARAM::alloc( file_size, NsARAM::BOTTOMUP );
948  }
949  else
950  {
951  p_aram = NsARAM::alloc( file_size, NsARAM::TOPDOWN );
952  }
953  if ( p_aram )
954  {
955  int adjust = 4;
956  uint32 toread = file_size;
957  uint32 current_aram_offset = p_aram;
958  while ( toread )
959  {
960  uint32 thistime = PRE_BUFFER_SIZE;
961  if ( toread < PRE_BUFFER_SIZE ) thistime = toread;
962 
963  File::Read(&p_buffer[adjust], 1, thistime - adjust, fp);
964  DCFlushRange ( p_buffer, thistime );
965  NsDMA::toARAM( current_aram_offset, p_buffer, thistime );
966  toread -= thistime;
967  current_aram_offset += thistime;
968  adjust = 0;
969  }
970  File::Close(fp);
971  pFile = (uint8 *)p_aram;
972  }
973 #else
974  file_size = File::GetFileSize(fp);
975  // Now allocates the .PRE file from the top of the heap, to avoid fragmentation.
976  Mem::PushMemProfile((char*)fullname);
977  if (useBottomUpHeap)
978  {
979  pFile = new (Mem::Manager::sHandle().BottomUpHeap()) uint8[file_size];
980  }
981  else
982  {
983  pFile = new (Mem::Manager::sHandle().TopDownHeap()) uint8[file_size];
984  }
985  Mem::PopMemProfile();
986  //uint8 *pFile = new uint8[file_size];
987  File::Read(pFile, 1, file_size, fp);
988 
989  int read_file_size = *((int *) pFile);
990  Dbg_MsgAssert(file_size == read_file_size,( "%s has incorrect file size: %d vs. expected %d\n", fullname, file_size, read_file_size));
991  if (Config::CD())
992  {
993  if (file_size != read_file_size) printf("%s has incorrect file size\n", fullname);
994  }
995 
996  File::Close(fp);
997 #endif // __PRE_ARAM__
998  }
999 
1000 # if !defined( __PLAT_NGC__ ) || ( defined( __PLAT_NGC__ ) && !defined( __NOPT_FINAL__ ) )
1001  printf("load time for file %s size %d is %d ms\n", pFilename, file_size, (int) Tmr::ElapsedTime(basetime));
1002 #endif
1003 
1004  // the PRE file object winds up at the top of the heap, too. This is fine because
1005  // it will be unloaded at the same time as the big file buffer
1006  if (useBottomUpHeap)
1007  {
1008  if (!mp_table->PutItem(pFilename, new (Mem::Manager::sHandle().BottomUpHeap()) PreFile(pFile,useBottomUpHeap)))
1009  Dbg_MsgAssert(0,( "PRE %s loaded twice", pFilename));
1010  }
1011  else
1012  {
1013  if (!mp_table->PutItem(pFilename, new (Mem::Manager::sHandle().TopDownHeap()) PreFile(pFile)))
1014  Dbg_MsgAssert(0,( "PRE %s loaded twice", pFilename));
1015  }
