82 int handle_ack( MsgHandlerContext *context )
83 {
84  MsgImpLink *msg_imp_link, *next_imp;
85  MsgSeqLink *msg_seq_link, *next_seq;
86  Lst::Search< MsgImpLink > imp_sh;
87  Lst::Search< MsgSeqLink > seq_sh;
88  MsgAck *ack_msg;
89  QueuedMsg* msg;
90  QueuedMsgSeq* msg_seq;
91  unsigned int send_time;
92 
93 
94 
95  Dbg_Assert( context );
96 
97  ack_msg = (MsgAck*) context->m_Msg;
98  send_time = 0;
99 
100 #ifdef DEBUG_MESSAGES
101  if( context->m_Conn->IsRemote())
102  {
103  Dbg_Printf( "(%d) Conn %d: (%d) Got Ack for Packetstamp %d\n",
104  context->m_App->m_FrameCounter,
105  context->m_Conn->GetHandle(),
106  context->m_App->m_Timestamp,
107  ack_msg->m_Packetstamp );
108  }
109 #endif
110 
111 #ifdef __PLAT_NGPS__
112  /*if( context->m_Conn->GetHandle() != ack_msg->m_Handle )
113  {
114  Lst::Search< Conn > sh;
115  Conn* conn;
116 
117  for( conn = context->m_App->FirstConnection( &sh ); conn; conn = context->m_App->NextConnection( &sh ))
118  {
119  Dbg_Printf( "*** Conn %d : %p. Ip: 0x%x Port: %d\n", conn->GetHandle(), conn, conn->GetIP(), conn->GetPort());
120  }
121  }
122 
123  Dbg_MsgAssert( context->m_Conn->GetHandle() == ack_msg->m_Handle, ("*** Different handles! %d %d\n", context->m_Conn->GetHandle(), ack_msg->m_Handle ));*/
124 #endif
125 
126  // Tell the dispatcher the size of this message because it does not know (optimization)
127  context->m_MsgLength = sizeof( MsgAck );
128 
129  // Don't ack foreign acks. We only even handle them because we need to tell the dispatcher
130  // the length of the message
131  if( context->m_PacketFlags & mHANDLE_FOREIGN )
132  {
133  return HANDLER_MSG_DONE;
134  }
135 
136  for( msg_imp_link = imp_sh.FirstItem( context->m_Conn->m_important_msg_list );
137  msg_imp_link; msg_imp_link = next_imp )
138  {
139  next_imp = imp_sh.NextItem();
140  if( ( ack_msg->m_Packetstamp == (unsigned short) msg_imp_link->m_Packetstamp ) &&
141  ( msg_imp_link->m_Timestamp != 0 )) // guard against acking packet stamp of zero where zero is the initialized value
142  {
143  msg = msg_imp_link->m_QMsg;
144  msg->m_Ref.Release();
145  if( msg->m_Ref.InUse() == false )
146  {
147  delete msg;
148  }
149  send_time = msg_imp_link->m_Timestamp;
150  delete msg_imp_link;
151  }
152  }
153 
154  for( msg_seq_link = seq_sh.FirstItem( context->m_Conn->m_sequenced_msg_list );
155  msg_seq_link; msg_seq_link = next_seq )
156  {
157  next_seq = seq_sh.NextItem();
158  // ACK the message
159  if( ( ack_msg->m_Packetstamp == (unsigned short) msg_seq_link->m_Packetstamp ) &&
160  ( msg_seq_link->m_Timestamp != 0 )) // guard against acking packet stamp of zero where zero is the initialized value
161  {
162  msg_seq = msg_seq_link->m_QMsg;
163  msg_seq->m_Ref.Release();
164  // If this message is no longer being referenced, free it
165  if( msg_seq->m_Ref.InUse() == false )
166  {
167  delete msg_seq;
168  }
169  send_time = msg_seq_link->m_Timestamp;
170  delete msg_seq_link;
171  }
172  }
173 
174  // If we can detect that the connection is starting to take longer to ack
175  // our messages then affect his calculated latency so that we don't expect
176  // acks back so quickly (which prompts resends )
177  if( ( context->m_Conn->IsRemote()) &&
178  ( context->m_Conn->TestStatus( Conn::mSTATUS_READY )) &&
179  ( send_time > 0 ))
180  {
181  unsigned int latency_value;
182 
183  latency_value = context->m_App->m_Timestamp - send_time;
184  if( latency_value > (unsigned int) context->m_Conn->GetAveLatency())
185  {
186  if( latency_value > App::MAX_LATENCY )
187  {
188  latency_value = App::MAX_LATENCY;
189  }
190 
191  context->m_Conn->m_latency_test.InputLatencyValue( latency_value );
192  }
193  }
194 
195  return HANDLER_CONTINUE;
196 }

203 int App::handle_stream_messages( MsgHandlerContext *context )
204 {
205  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().NetworkHeap());
206 
207  //Dbg_Printf( "******** Got Stream Message\n" );
208 
209  switch( context->m_MsgId )
210  {
211  case MSG_ID_STREAM_START:
212  {
213  //Dbg_Printf( "******** Got Stream Start\n" );
214  MsgStreamStart* msg;
215  StreamLink* str_link;
216  StreamDesc* new_stream;
217  Lst::Search< StreamLink > str_sh;
218  int packet_len;
219 
220  msg = (MsgStreamStart*) context->m_Msg;
221 
222  for( str_link = str_sh.FirstItem( context->m_Conn->m_StreamInList ); str_link;
223  str_link = str_sh.NextItem())
224  {
225  StreamDesc* desc;
226 
227  desc = str_link->m_Desc;
228  if( desc->m_StreamId == msg->m_StreamId )
229  {
230  Mem::Manager::sHandle().PopContext();
231  return HANDLER_MSG_DONE;
232  }
233  }
234 
235  new_stream = new StreamDesc;
236  strcpy( new_stream->m_StreamDesc, msg->m_StreamDesc );
237  Dbg_Printf( "StreamDesc: %s\n", new_stream->m_StreamDesc );
238  new_stream->m_Size = msg->m_Size;
239  new_stream->m_StreamId = msg->m_StreamId;
240  new_stream->m_Checksum = msg->m_Checksum;
241  new_stream->m_MsgId = msg->m_MsgId;
242  new_stream->m_Data = new char[ msg->m_Size ];
243  new_stream->m_DataPtr = new_stream->m_Data;
244  packet_len = ( msg->m_Size < MAX_STREAM_CHUNK_LENGTH ) ? msg->m_Size : MAX_STREAM_CHUNK_LENGTH;
245  memcpy( new_stream->m_Data, msg->m_Data, packet_len );
246  new_stream->m_DataPtr += packet_len;
247  str_link = new StreamLink( new_stream );
248  context->m_Conn->m_StreamInList.AddToTail( str_link );
249 
250  if(((unsigned int) new_stream->m_DataPtr - (unsigned int) new_stream->m_Data ) >= (unsigned int) new_stream->m_Size )
251  {
252  MsgHandlerContext msg_context;
253  int result;
254 
255  // Shouldn't go over
256  Dbg_Assert( ((unsigned int) new_stream->m_DataPtr - (unsigned int) new_stream->m_Data ) == (unsigned int) new_stream->m_Size );
257 
258  msg_context.m_Conn = context->m_Conn;
259  msg_context.m_App = context->m_App;
260  msg_context.m_PacketFlags = 0;
261  msg_context.m_MsgId = new_stream->m_MsgId;
262  msg_context.m_MsgLength = new_stream->m_Size;
263  msg_context.m_Msg = new char[new_stream->m_Size];
264 
265  memcpy( msg_context.m_Msg, new_stream->m_Data, new_stream->m_Size );
266 
267  result = context->m_App->m_Dispatcher.DispatchMessage( &msg_context );
268  if( result != HANDLER_MSG_DESTROYED )
269  {
270  delete str_link->m_Desc;
271  delete str_link;
272  }
273 
274  delete [] msg_context.m_Msg;
275  Mem::Manager::sHandle().PopContext();
276  return result;
277  }
278  break;
279  }
280 
281  case MSG_ID_STREAM_DATA:
282  {
283  MsgStreamData* msg;
284  StreamLink* str_link;
285  Lst::Search< StreamLink > str_sh;
286 
287  msg = (MsgStreamData*) context->m_Msg;
288 
289  //Dbg_Printf( "******** Got Stream Data\n" );
290  for( str_link = str_sh.FirstItem( context->m_Conn->m_StreamInList ); str_link;
291  str_link = str_sh.NextItem())
292  {
293  StreamDesc* desc;
294  int packet_len;
295 
296  desc = str_link->m_Desc;
297  if( desc->m_StreamId == msg->m_StreamId )
298  {
299  packet_len = context->m_MsgLength - sizeof( int ); // subtract the size of the stream id
300  memcpy( desc->m_DataPtr, msg->m_Data, packet_len );
301  desc->m_DataPtr += packet_len;
302 
303  if(((unsigned int) desc->m_DataPtr - (unsigned int) desc->m_Data ) >= (unsigned int) desc->m_Size )
304  {
305  MsgHandlerContext msg_context;
306  int result;
307  uint32 actual_crc;
308 
309  // Shouldn't go over
310  Dbg_Assert( ((unsigned int) desc->m_DataPtr - (unsigned int) desc->m_Data ) == (unsigned int) desc->m_Size );
311 
312  msg_context.m_Conn = context->m_Conn;
313  msg_context.m_App = context->m_App;
314  msg_context.m_PacketFlags = 0;
315  msg_context.m_MsgId = desc->m_MsgId;
316  msg_context.m_MsgLength = desc->m_Size;
317  msg_context.m_Msg = new char[desc->m_Size];
318 
319  memcpy( msg_context.m_Msg, desc->m_Data, desc->m_Size );
320  actual_crc = Crc::GenerateCRCCaseSensitive( msg_context.m_Msg, desc->m_Size );
321  if( actual_crc != desc->m_Checksum )
322  {
323  msg_context.m_PacketFlags |= mHANDLE_CRC_MISMATCH;
324  Dbg_Printf( "CRC MISMATCH!!!\n" );
325  }
326 
327  Dbg_Printf( "handling stream: %s\n", desc->m_StreamDesc );
328  result = context->m_App->m_Dispatcher.DispatchMessage( &msg_context );
329  if( result != HANDLER_MSG_DESTROYED )
330  {
331  delete str_link->m_Desc;
332  delete str_link;
333  }
334 
335  delete [] msg_context.m_Msg;
336 
337  Mem::Manager::sHandle().PopContext();
338  return result;
339  }
340  }
341  }
342  break;
343  }
344  }
345 
346  Mem::Manager::sHandle().PopContext();
347  return HANDLER_CONTINUE;
348 }

355 int App::handle_sequenced_messages( MsgHandlerContext *context )
356 {
357  MsgSeqLink *msg_link;
358  char* data;
359  unsigned short msg_len;
360  QueuedMsgSeq* queued_msg;
361  unsigned char group_id, msg_id;
362  unsigned int seq_id;
363  Lst::Search< MsgHandler > sh;
364 
365 
366 
367  Dbg_Assert( context );
368 
369  data = context->m_Msg;
370  memcpy( &group_id, data, sizeof( char ));
371  data += sizeof( char );
372  memcpy( &seq_id, data, sizeof( unsigned int ));
373  data += sizeof (unsigned int );
374 
375  // don't accept old sequenced messages
376  if( seq_id < context->m_Conn->m_WaitingForSequenceId[ group_id ] )
377  {
378  return HANDLER_CONTINUE;
379  }
380 
381  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().NetworkHeap());
382 
383  memcpy( &msg_id, data, sizeof( unsigned char ));
384  data += sizeof( unsigned char );
385 
386  msg_len = context->m_MsgLength - Manager::vMSG_SEQ_HEADER_LENGTH;
387 
388  queued_msg = new QueuedMsgSeq( msg_id, msg_len, data, group_id );
389 
390  msg_link = new MsgSeqLink( queued_msg );
391  msg_link->SetPri( seq_id );
392  msg_link->m_Timestamp = 0;
393  msg_link->m_Packetstamp = 0;
394  msg_link->m_GroupId = group_id;
395  msg_link->m_SequenceId = seq_id;
396 
397  Mem::Manager::sHandle().PopContext();
398 
399  // This might be dangerous in some apps, but it makes sense right now. Don't add
400  // this message to the sequenced queue if we dont have a handler for it.
401  /*if( sh.FirstItem( context->m_App->m_Dispatcher.m_handler_list[msg_id] ) == NULL )
402  {
403  Dbg_MsgAssert( 0, ( "No handler for msg %d\n", msg_id ));
404  delete msg_link->m_QMsg;
405  delete msg_link;
406  return HANDLER_MSG_DONE;
407  }*/
408  // Don't re-add identical sequenced messages
409  if( context->m_Conn->m_SequencedBuffer[group_id].AddUniqueSequence( msg_link ) == false )
410  {
411  delete msg_link->m_QMsg;
412  delete msg_link;
413  return HANDLER_MSG_DONE;
414  }
415 
416  return HANDLER_CONTINUE;
417 }

490 /******************************************************************/
491 /* Handle the timestamp from the server. Check for lost & late */

494 
495 int Client::handle_timestamp( MsgHandlerContext *context )
496 {
497  MsgPacketStamp *time_msg;
498  MsgAck ack_msg;
499  MsgDesc msg_desc;
500 
501  Dbg_Assert( context );
502 
503  // Tell the dispatcher the size of this message because it does not know (optimization)
504  context->m_MsgLength = sizeof( MsgPacketStamp );
505 
506  // Don't ack foreign timestamps. We only even handle them because we need to tell the dispatcher
507  // the length of the message
508  if( context->m_PacketFlags & mHANDLE_FOREIGN )
509  {
510  return HANDLER_MSG_DONE;
511  }
512 
513  time_msg = (MsgPacketStamp *) context->m_Msg;
514  ack_msg.m_Packetstamp = time_msg->m_Packetstamp;
515  //ack_msg.m_Handle = time_msg->m_Handle;
516 
517 #ifdef DEBUG_MESSAGES
518  if( context->m_Conn->IsRemote())
519  {
520  Dbg_Printf( "(%d) Conn %d: (%d) Got Packetstamp %d\n",
521  context->m_App->m_FrameCounter,
522  context->m_Conn->GetHandle(),
523  context->m_App->m_Timestamp,
524  time_msg->m_Packetstamp );
525  }
526 #endif
527 
528  msg_desc.m_Data = &ack_msg;
529  msg_desc.m_Length = sizeof( MsgAck );
530  msg_desc.m_Id = MSG_ID_ACK;
531  context->m_App->EnqueueMessageToServer( &msg_desc );
532 
533  // Todo : Change this logic as packetstamps DO wraparound
534  // if == 0, we just connected : accept first timestamp unconditionally
535  if( context->m_Conn->m_latest_packet_stamp == 0 )
536  {
537  context->m_Conn->m_latest_packet_stamp = time_msg->m_Packetstamp;
538  }
539  else
540  {
541  // Check for late packets, but also make sure it's actually late, and not just wraparound
542  if( time_msg->m_Packetstamp < context->m_Conn->m_latest_packet_stamp )
543  {
544  if(( context->m_Conn->m_latest_packet_stamp - time_msg->m_Packetstamp ) > 5000 )
545  {
546  context->m_Conn->m_latest_packet_stamp = time_msg->m_Packetstamp;
547  }
548  else
549  {
550  context->m_App->m_LatePackets++;
551  context->m_App->m_LostPackets--;
552  context->m_PacketFlags |= mHANDLE_LATE;
553  }
554  }
555  else if( time_msg->m_Packetstamp == context->m_Conn->m_latest_packet_stamp )
556  {
557  context->m_App->m_DupePackets++;
558  Dbg_Printf( "**** Got Duplicate Packet: %d\n", context->m_App->m_DupePackets );
559  return HANDLER_ERROR; // don't process dupes
560  }
561  else if(( time_msg->m_Packetstamp - context->m_Conn->m_latest_packet_stamp ) > 1 )
562  {
563  // Check for lost packets and make sure it's not just wraparound
564  if(( time_msg->m_Packetstamp - context->m_Conn->m_latest_packet_stamp ) > 5000 )
565  {
566  context->m_App->m_LatePackets++;
567  context->m_App->m_LostPackets--;
568  context->m_PacketFlags |= mHANDLE_LATE;
569  }
570  else
571  {
572  context->m_App->m_LostPackets +=
573  (( time_msg->m_Packetstamp - context->m_Conn->m_latest_packet_stamp ) - 1 );
574  context->m_Conn->m_latest_packet_stamp = time_msg->m_Packetstamp;
575  }
576  }
577  else
578  {
579  context->m_Conn->m_latest_packet_stamp = time_msg->m_Packetstamp;
580  }
581  }
582 
583  return HANDLER_CONTINUE;
584 }

619 int handle_latency_response( MsgHandlerContext *context )
620 {

623  int cur_time;
624 
625  Dbg_Assert( context );
626 
627  time_msg = (MsgTimestamp *) context->m_Msg;

644  if( latency_value > App::MAX_LATENCY )
645  {
646  latency_value = App::MAX_LATENCY;
647  }
648  conn->m_latency_test.InputLatencyValue( latency_value );
649  }

651  return HANDLER_CONTINUE;
