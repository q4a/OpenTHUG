48 CTexDict * CTexDictManager::sCreateTextureDictionary(const char *p_tex_dict_name)
49 {
50  CTexDict *p_dict;
51  uint32 checksum = Crc::GenerateCRCFromString(p_tex_dict_name);
52 
53  p_dict = sGetTextureDictionary(checksum);
54 
55  // Assert for now unless we can think of a reason to use with ref counts
56  Dbg_MsgAssert((p_dict == NULL), ("Texture dictionary %s already exists", p_tex_dict_name));
57 
58  if (p_dict)
59  {
60  // If already loaded, then just link up with it
61  p_dict->IncRefCount();
62  } else {
63  p_dict = s_plat_create_texture_dictionary(checksum);
64  s_tex_dict_lookup.PutItem(checksum, p_dict);
65  }
66  return p_dict;

611 // @script | CombineTextures | Combine two textures
612 // @parm name | src | name of orig texture (and final texture if no new texture is supplied)

614 // @parmopt name | new | | name of new destination texture
615 // @parmopt flag | no_palette_gen | | use original palette (faster)
616 bool ScriptCombineTextures(Script::CScriptStructure *pParams, Script::CScript *pScript)
617 {
618  uint32 new_checksum = 0;
619  uint32 tex1_checksum;
620  uint32 tex2_checksum;
621  bool gen_palette = true;
622 
623  if (!pParams->GetChecksum(CRCD(0x9fbbdb72,"src"), &tex1_checksum))

626  }

635  }

638  if (p_texture1)

641  if (p_texture2)
642  {
643  if (new_checksum)
644  {
645  // Make new texture
646  CTexDictManager::sp_sprite_tex_dict->CombineTextures(new_checksum, p_texture1, p_texture2, gen_palette);
647  }
648  else
649  {

651  p_texture1->CombineTextures(p_texture2, gen_palette);
652  }
653  }
654  else
655  {
656  Dbg_MsgAssert(0, ("Can't find texture %s to combine", Script::FindChecksumName(tex2_checksum)));
657  }
658  }
659  else
660  {
661  Dbg_MsgAssert(0, ("Can't find texture %s to combine", Script::FindChecksumName(tex1_checksum)));
662  }
