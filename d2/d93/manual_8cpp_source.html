209 void CManual::SetUp(uint32 ButtonAChecksum, uint32 ButtonBChecksum, int Tweak, bool DoFlipCheck, bool PlayRangeAnimBackwards)
210 {
211 
212  mActualManualTime = 0.0f;
213  mManualReleased=false;
214  mButtonAChecksum=ButtonAChecksum;
215  mButtonBChecksum=ButtonBChecksum;
216  mTweak=Tweak;
217  mDoFlipCheck=DoFlipCheck;
218  mPlayRangeAnimBackwards=PlayRangeAnimBackwards;
219 
220  Dbg_MsgAssert(mpSkater,("NULL mpSkater"));
221 
222  // Check for whether the skater has re-railed onto the same rail
223  // or a different one.
224  if (mpSkaterBalanceTrickComponent->mBalanceTrickType==0x255ed86f || // Grind
225  mpSkaterBalanceTrickComponent->mBalanceTrickType==0x8d10119d) // Slide
226  {
227  const CRailNode* p_rail_node = GetSkaterCorePhysicsComponentFromObject(mpSkater)->mp_rail_node;
228 
229  Dbg_MsgAssert(p_rail_node,("What, NULL mp_rail_node ??"));
230 
231  if (mOldRailNode==-1)
232  {
233  // This is a fresh rail.
234  }
235  else
236  {
237 
238  if (p_rail_node->ProbablyOnSameRailAs(mOldRailNode))
239  {
240  // Jumped and landed on the same rail without ending
241  // the combo.
242  // Add some time to make this more difficult.
243  mManualTime+=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xfbec5b6c, "Same_Grind_Add_Time"));
244  #ifdef __DEBUG_CHEESE__
245  printf ("%d: %.2f: leaving cheese, probably on same rail\n",__LINE__,mManualCheese);
246  #endif
247  }
248  else
249  {
250  // It's a different rail from the last one, ie jumped from a rail to
251  // another rail without ending the combo.
252  // Reward the player by subtracting a bit of time to make it easier.
253  mManualTime-=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xc5e1c8eb,"New_Grind_Sub_Time"));
254  if (mManualTime<0.0f)
255  {
256  mManualTime=0.0f;
257  }
258  // And clear the cheese
259  // as it's not cheesey to change rails
260  // so we don't to penalize the player for doing so
261  #ifdef __DEBUG_CHEESE__
262  printf ("%d: Resetting cheese to 0 as it's a new grind\n",__LINE__);
263  #endif
264 
265  mManualCheese = 0.0f;
266  }
267  }
268  mOldRailNode=p_rail_node->GetNode();
269 
270  Mdl::Score *pScore=mpSkaterScoreComponent->GetScore();
271  float robot = pScore->GetRobotRailMult();
272  if ( robot < 0.5f) // <0.5 means any time you are on the same rail twice in a combo
273  {
274 // printf ("SHOULD ADJUST BALANCE HERE, AS ON A ROBOT LINE\n");
275  // Need to add something to the Manual Time
276  mManualTime += Script::GetFloat(CRCD(0xf54ca12f,"robot_rail_add_time") )* 0.5f / robot;
277 
278  // And give him percentage nudge away from 0
279  // (actually, an absolute value would be better, more consistent
280  // perhaps not even based on the amount, just trigger 20% whenever
281  // you land on something robotesque
282  mManualLean += Script::GetFloat(CRCD(0x30990680,"robot_rail_nudge")) * Mth::Sgn(mManualLeanDir) * 0.5f / robot;
283 // printf ("Time = %.2f, Lean = %.2f, dir = %.2f\n",mManualTime, mManualLean, mManualLeanDir);
284  }
285 
286 
287  }
288 
289  float repeat_min = mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x8f4db5d3,"Repeat_Min"));
290  if ( Mth::Abs(mManualLeanDir) != 0.0f)
291  {
292  // first multiplied by the reat multipler
293  mManualLeanDir = mManualLeanDir * mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x93dd6a10,"Repeat_Multiplier"));
294 
295  // use the sign of previous balance
296  if ( Mth::Abs(mManualLeanDir) < repeat_min)
297  {
298  // if too small, then set to minimum
299  mManualLeanDir = repeat_min * Mth::Sgn(mManualLeanDir);
300  }
301  else
302  {
303  }
304  }
305  else
306  {
307  // No previous balance
308  // so just set to default minimum
309  mManualLeanDir = repeat_min;
310  // Adjust the sign of the lean to point in the direction of a bail.
311  switch (mpSkaterBalanceTrickComponent->mBalanceTrickType)
312  {
313  case 0xef24413b: // Manual
314  // Manuals keep positive sign.
315  break;
316  case 0x0ac90769: // NoseManual
317  // Nosemanuals fall the other way.
318  mManualLeanDir=-mManualLeanDir;
319  break;
320  default:
321  // If in a grind, choose a random side to fall towards.
322  // Hmm, should choose differently for a lip?
323  if (Mth::Rnd(2))
324  {
325  mManualLeanDir=-mManualLeanDir;
326  }
327  break;
328  }
329 
330  }
331 
332 
333  // multiply this initial direction by some value based on the time
334 // mManualLean = Mth::Abs(mManualLean) * Mth::Sgn(mManualLeanDir); // possibly alter lean to face the bail direction
335 
336  mManualLean *= mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x6ce10ea4,"Lean_Repeat_Multiplier"));
337 
338  // add cheese after lean multiply (otherwise, it can be ineffective)
339  #ifdef __DEBUG_CHEESE__
340  printf ("%d: %.2f + %.2f = %.2f: Adding cheese to lean \n",__LINE__,mManualLean, mManualCheese * Mth::Sgn(mManualLean),mManualLean += mManualCheese * Mth::Sgn(mManualLean));
341  #endif
342 
343  mManualLean += mManualCheese * Mth::Sgn(mManualLean); // add in the cheese to the lean
344 
345  float BailAngle=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xf81ab0e9,"Lean_Bail_Angle"));
346  if (Mth::Abs(mManualLean) > BailAngle)
347  {
348  #ifdef __DEBUG_CHEESE__
349  printf ("%d: %.2f: Cheese caused an instant bail!!! \n",__LINE__,mManualCheese);
350  #endif
351  // The cheese has caused an instant bail, so in order that the meter does become
352  // visible, make the lean a wee bit less that the bail angle.
353  mManualLean=BailAngle*0.9f;
354  mManualLean = Mth::Abs(mManualLean) * Mth::Sgn(mManualLeanDir);
355  mManualLeanDir = Mth::Sgn(mManualLeanDir) * BailAngle; // ensure a bail next frame
356  }
357 
358 
359  // Reset the cheese.
360  mManualCheese=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x1231ae3c,"Cheese"));
361 
362  #ifdef __DEBUG_CHEESE__
363  printf ("%d: %.2f: Cheese reset to initial scripted value \n",__LINE__,mManualCheese);
364  #endif
365 
366 
367  // The programmatic wobble applied to the animation is off initially, so that it does not get applied
368  // to the init anim.
369  mWobbleEnabled=false;
370 
371  mStartTime=Tmr::ElapsedTime(0);
372 }

382 void CManual::DoManualPhysics()
383 {
384  float frame_length = Tmr::FrameLength();
385 
386  // Do nothing if no buttons are specified.
387  if (!mButtonAChecksum || !mButtonBChecksum)
388  {
389  return;
390  }
391 
392  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
393  Net::Server* server;
394  GameNet::PlayerInfo* player;
395  Net::Conn* skater_conn;
396 
397  Dbg_MsgAssert(mpSkater,("NULL mpSkater"));
398 
399  skater_conn = NULL;
400  if(( server = gamenet_man->GetServer()))
401  {
402  if(( player = gamenet_man->GetPlayerByObjectID( mpSkater->GetID() )))
403  {
404  skater_conn = player->m_Conn;
405  }
406  }
407 
408  // decrement the cheese value, which is added to the lean angle when we do the next manual
409  // this is to prevent "cheesing" the manuals (by rapidly doing several manuals that only
410  // last a frame or two, and don't give the balancing physics any time to take effect).
411 
412  if (mManualCheese>0.01f)
413  {
414  mManualCheese -= scale_with_frame_length(frame_length, mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x1231ae3c,"Cheese"))/mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xa4362662,"CheeseFrames")));
415  if (mManualCheese < 0.0f)
416  mManualCheese = 0.0f;
417  #ifdef __DEBUG_CHEESE__
418  printf ("%d: %.2f: Cheese decremented\n",__LINE__,mManualCheese);
419  #endif
420 
421  }
422 // printf(Cheese=%f\n",mManualCheese);
423 
424  // Add in the tweak to the score.
425  Mdl::Score *pScore=mpSkaterScoreComponent->GetScore();
426  //pScore->TweakTrickRequest(mTweak);
427  pScore->TweakTrick(mTweak);
428 
429  mActualManualTime+=frame_length;
430  mManualTime+=frame_length;
431 
432  Obj::CStatsManagerComponent* pStatsManagerComponent = GetStatsManagerComponentFromObject(mpSkater);
433  Dbg_Assert( pStatsManagerComponent );
434 
435  pStatsManagerComponent->CheckTimedRecord( mpSkaterBalanceTrickComponent->mBalanceTrickType, mActualManualTime, false );
436 
437  float ManualLeanGravity=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x6851748,"Lean_Gravity_Stat"));
438  float ManualInstableRate=mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xa9fef5ab,"Instable_Rate"));
439  float ManualBase = mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xb6a634f3,"Instable_base"));
440  float ManualInstability = ManualBase + (mManualTime * ManualInstableRate);
441 
442 
443  // the more you lean, the more you lean
444 // mManualLean += scale_with_frame_length(mManualLean/ManualLeanGravity * (0.5f + mManualTime/2)/ManualInstableRate);
445 // mManualLean += scale_with_frame_length(mManualLeanDir * (0.5f + mManualTime/2)/ManualInstableRate); // speed of turn
446 
447  mManualLean += scale_with_frame_length(frame_length, mManualLean * ManualLeanGravity * ManualInstability);
448  mManualLean += scale_with_frame_length(frame_length, mManualLeanDir * ManualInstability);
449 
450 
451  CSkaterButton *pButt=mpInputComponent->GetControlPad().GetButton(mButtonAChecksum);
452  bool APressed=pButt->GetPressed();
453 
454  pButt=mpInputComponent->GetControlPad().GetButton(mButtonBChecksum);
455  bool BPressed=pButt->GetPressed();
456 
457  // The released flag stays false until they let go of the buttons. This is so that if the player
458  // was holding one of the control buttons at the instant the balance trick started, as they probably
459  // will be if doing a manual, then the button won't have any effect on the balance, so as not to
460  // unbalance them straight away.
461  if (!APressed && !BPressed)
462  {
463  mManualReleased = true;
464  }
465 
466 
467  uint32 safe_period = (uint32)Script::GetInt(CRCD(0x8cc8f176,"BalanceSafeButtonPeriod"));
468  uint32 elapsed_time = Tmr::ElapsedTime(mStartTime);
469 
470  // If the buttons have not technically been released, but have been held for longer than a certain
471  // amount of time, then assume that the player does want the button to have an effect after all.
472  if (!mManualReleased && elapsed_time>(uint32)Script::GetInt(CRCD(0x7da1621,"BalanceIgnoreButtonPeriod")))
473  {
474  mManualReleased = true;
475  }
476 
477 
478  if (APressed && mManualReleased)
479  {
480  // this ramping up of the accleration for a few hundred miliseconds
481  // will prevent us from overcompensating
482  // due to the button being held down
483  // and will actually make it easier to "Stick" the mManualLeanDir
484  // the ramping is only in effect if you press in the direction opposite
485  // from what you need to get more upright
486  // so it prevents you from becoming more unbalanced
487  float mult = 1.0f;
488  if (mManualLean < 0.0f && elapsed_time < safe_period)
489  {
490  mult = elapsed_time / safe_period;
491  }
492 
493  mManualLeanDir -= scale_with_frame_length(frame_length, mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x61ad5596,"Lean_Acc"))) * mult;
494  }
495  else if (BPressed && mManualReleased)
496  {
497  float mult = 1.0f;
498  if (mManualLean > 0.0f && elapsed_time < safe_period)
499  {
500  mult = elapsed_time / safe_period;
501  }
502  mManualLeanDir += scale_with_frame_length(frame_length, mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x61ad5596,"Lean_Acc"))) * mult;
503  }
504  else
505  {
506  if (Mth::Abs(mManualLeanDir) < mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xef27e8ff,"Lean_Min_Speed")))
507  {
508  int r=(int)mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0x32eb3308,"Lean_Rnd_Speed"));
509  r|=1;
510 // mManualLeanDir = Mth::Rnd(r)-r/2;
511  mManualLeanDir = Mth::Rnd(r)*Mth::Sgn(mManualLeanDir);
512  }
513  else
514  {
515  // always add some randomness to it, to make sure we don't ever get stuck
516  mManualLeanDir += (Mth::Rnd(50))/100.0f * Mth::Sgn(mManualLeanDir);
517  }
518  }
519 
520 
521  /*
522  HUD::PanelMgr* panelMgr = HUD::PanelMgr::Instance();
523  HUD::Panel *pPanel=panelMgr->GetPanelBySkaterId(mpSkater->GetID());
524 
525  char pBuf[100];
526  sprintf(pBuf,"Record %.2f",mMaxTime);
527  panelMgr->SendConsoleMessage(pBuf, 2);
528  sprintf(pBuf,"Time %.2f",mActualManualTime);
529  panelMgr->SendConsoleMessage(pBuf, 3);
530  */
531 
532  if (!mpSkater->GetScript())
533  {
534  mpSkater->SetScript(new Script::CScript);
535  }
536 
537  if (Mth::Abs(mManualLean)>mpSkaterBalanceTrickComponent->GetBalanceStat(CRCD(0xf81ab0e9,"Lean_Bail_Angle")))
538  {
539  if (mManualLean>0)
540  {
541  mpSkater->SelfEvent(CRCD(0xd77b6ff9,"OffMeterTop"));
542  }
543  else
544  {
545  mpSkater->SelfEvent(CRCD(0x47d44b84,"OffMeterBottom"));
546  }
547 
548  mManualLean=0;
549  mpSkaterBalanceTrickComponent->mBalanceTrickType=0;
550  }
551 
552  if (mWobbleEnabled)
553  {
554  bool flip=mPlayRangeAnimBackwards;
555  if (mDoFlipCheck && GetSkaterCorePhysicsComponentFromObject(mpSkater)->GetFlag(FLIPPED))
556  {
557  flip=!flip;
558  }
559  // Update the anim frame.
560  Obj::CAnimationComponent* pAnimComponent = GetAnimationComponentFromObject(mpSkater);
561  if ( pAnimComponent )
562  {
563  if (flip)
564  {
565  pAnimComponent->SetWobbleTarget((-mManualLean+4096)/8192, true);
566  }
567  else
568  {
569  pAnimComponent->SetWobbleTarget((mManualLean+4096)/8192, true);
570  }
571  }
572  }
573 
574 
575  // Update the balance meter.
576  GameNet::MsgByteInfo msg;
577  float final_val = -mManualLean/4096;
578 
579  if( server && skater_conn )
580  {
581  msg.m_Data = (char) ( final_val * 127.0f );
582  /*server->EnqueueMessage( skater_conn->GetHandle(), GameNet::MSG_ID_SET_MANUAL_METER,
583  sizeof(GameNet::MsgByteInfo), &msg );*/
584  }
585 
586  // Need to also check we are still doing a balance trick, since exceptions
587  // that we fired above (offmetertop/bottom) might have canceled it
588  if (!mNeverShowMeters && mpSkaterBalanceTrickComponent->mBalanceTrickType)
589  {
590  if ((mButtonAChecksum==0xbc6b118f/*Up*/ || mButtonAChecksum==0xe3006fc4/*Down*/) &&
591  (mButtonBChecksum==0xbc6b118f/*Up*/ || mButtonBChecksum==0xe3006fc4/*Down*/))
592  {
593  if (mpSkaterScoreComponent->GetScore())
594  {
595  mpSkaterScoreComponent->GetScore()->SetManualMeter( true, final_val );
596  }
597  }
598  else
599  {
600  if (mpSkaterScoreComponent->GetScore())
601  {
602  mpSkaterScoreComponent->GetScore()->SetBalanceMeter( true, final_val );
603  }
604  }
605  }
606 }
