190 void CHorse::EndRun()
191 {
192  // calculate the score post-run
193  // update the horse data structure so that we know that
194  // the score to beat is
195 
196  sprintf( m_nextHorseMessage, "unknown" );
197 
198  // by default, don't change the skater
199  m_nextSkaterId = m_currentSkaterId;
200 
201  int score = get_score( m_currentSkaterId );
202 
203  printf( "End run = %d\n", score );
204 
205  if ( score > m_scoreToBeat )
206  {
207  // current player wins the run
208  m_scoreToBeat = score;
209 
210  sprintf( m_nextHorseMessage, "%s %s", Script::GetLocalString( "horse_str_setscoreis" ), Str::PrintThousands(m_scoreToBeat) );
211  m_status = s_horse_status[vBEATSCORE];
212  }
213  else if ( score < m_scoreToBeat )
214  {
215  // current player loses the run
216  m_scoreToBeat = 0;
217 
218  m_numLetters[ m_currentSkaterId ]++;
219 
220  // skip to next restart point
221  SetNextRestartPoint();
222 
223  if ( Ended() )
224  {
225  // GJ: this string is never used... it says "Player 1 is a horse" somewhere else
226  sprintf( m_nextHorseMessage, "You are a %s!", m_string );
227 
228  m_status = s_horse_status[vENDED];
229  }
230  else
231  {
232  char letter[2];
233  letter[0] = *(m_string + m_numLetters[m_currentSkaterId] - 1);
234  // make it uppercase
235  if ( letter[0] >= 'a' && letter[0] <= 'z' )
236  {
237  letter[0] -= 'a';
238  letter[0] += 'A';
239  }
240  letter[1] = 0;
241  sprintf( m_nextHorseMessage, "%s %s", Script::GetLocalString( "horse_str_getstheletter" ), letter );
242  m_status = s_horse_status[vGOTLETTER];
243  }
244  }
245  else
246  {
247  // current player draws
248  if ( score == 0 )
249  {
250  strcpy( m_nextHorseMessage, Script::GetLocalString( "horse_str_noscoreset" ));
251  m_status = s_horse_status[vNOSCORESET];
252  }
253  else
254  {
255  strcpy( m_nextHorseMessage, Script::GetLocalString( "horse_str_youtiedthetargetscore" ));
256  m_status = s_horse_status[vTIESCORE];
257  }
258  }
259 
260 #ifdef __USER_GARY__
261  printf( "Current Score: %s - %s\n",
262  GetWordForPlayer( 0 ).getString(),
263  GetWordForPlayer( 1 ).getString() );
264 #endif
265 
266  //SwitchPlayers();
267 }

431 Str::String CHorse::GetString( uint32 checksum )
432 {
433 
434 
435  char msg[256];
436  strcpy( msg, "Unknown" );
437 
438  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
439  Obj::CSkater *pSkater = skate_mod->GetSkaterById( m_currentSkaterId );
440  Dbg_Assert( pSkater );
441 
442  switch ( checksum )
443  {
444  case 0x9196d920: // playerName
445  {
446 // int currentPlayerNumber = pSkater->GetID() + 1;
447 // sprintf( msg, "Player %d", currentPlayerNumber );
448  sprintf( msg, "%s", skate_mod->GetSkaterById( pSkater->GetID() )->GetDisplayName() );
449  }
450  break;
451 
452  case 0x5078bc7c: // horsePreRun
453  {
454  if ( m_scoreToBeat > 0 )
455  {
456  sprintf( msg, "%s %s", Script::GetLocalString( "horse_str_scoretobeat" ), Str::PrintThousands(m_scoreToBeat) );
457  }
458  else
459  {
460  sprintf( msg, Script::GetLocalString( "horse_str_setscore" ) );
461  }
462  }
463  break;
464 
465  case 0x2a13f850: // horsePostRun
466  {
467  strcpy( msg, m_nextHorseMessage );
468  }
469  break;
470 
471  case 0xde802177: // panelString1
472  {
473  sprintf( msg, "%s: %s", skate_mod->GetSkaterById( 0 )->GetDisplayName(), GetWordForPlayer( 0 ).getString() );
474  }
475  break;
476 
477  case 0x478970cd: // panelString2
478  {
479  sprintf( msg, "%s: %s", skate_mod->GetSkaterById( 1 )->GetDisplayName(), GetWordForPlayer( 1 ).getString() );
480  }
481  break;
482 
483  case 0x2523853d: // youAreA
484  {
485  bool starts_with_vowel = false;
486  char first_letter = GetWord()[0];
487  if ( first_letter >= 'a' && first_letter <= 'z' )
488  {
489  // make them upper case
490  first_letter -= 'a';
491  first_letter += 'A';
492  }
493  if ( first_letter == 'A' || first_letter == 'E' || first_letter == 'I'
494  || first_letter == 'O' || first_letter == 'U' )
495  {
496  starts_with_vowel = true;
497  }
498 
499  sprintf( msg, "%s %s", GetString( Script::GenerateCRC("playerName") ).getString(), starts_with_vowel ? Script::GetLocalString( "horse_str_isan" ) : Script::GetLocalString( "horse_str_isa" ) );
500  }
501  break;
502 
503  case 0x5f7d73e9: // finalWord
504  {
505  sprintf( msg, "%s", GetWord() );
506  }
507  break;
508  }
509 
510  return msg;
