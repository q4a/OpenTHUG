132 // Given the "Checksum" of a script command, then possibly handle it
133 // if it's a command that this component will handle
134 CBaseComponent::EMemberFunctionResult CSpecialItemComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
135 {
136  switch ( Checksum )
137  {
138  // @script | CreateSpecialItem | attaches a new special item to the skater in the specified slot
139  // @parm int | index | Index of special item to create
140  // @parm structure | params | Special item creation data (should look like an entry in the node array)
141  // @parmopt name | bone | | The bone to which the special item is attached; otherwise, it will be attached to the root
142  // @parmopt vector | offset | (0, 0, 0) | An additional offset to the specified bone (or root) applied to the special item's position
143  case 0x827c6bd1: // CreateSpecialItem
144  {
145  int index;
146  uint32 bone = 0;
147  pParams->GetInteger( CRCD(0x7f8c98fe,"index"), &index, true );
148  pParams->GetChecksum( "bone", &bone );
149 
150  uint32 checksum;
151  pParams->GetChecksum( "params", &checksum, Script::ASSERT );
152  Script::CStruct* pSpecialItemParams;
153  pSpecialItemParams = Script::GetStructure( checksum, Script::ASSERT );
154 
155  CCompositeObject* pMovingObject = CreateSpecialItem( index, pSpecialItemParams );
156 
157  // special items need a lock object to work...
158  Script::CStruct* pLockParams = new Script::CStruct;
159  pLockParams->AppendStructure( pParams ); // pass along any bone or offset parameters...
160  pLockParams->AddChecksum( "id", GetObject()->GetID() );
161  Obj::CLockObjComponent* pLockObjComponent = GetLockObjComponentFromObject( pMovingObject );
162  Dbg_MsgAssert( pLockObjComponent, ( "No lock obj component" ) );
163  pLockObjComponent->InitFromStructure( pLockParams );
164  delete pLockParams;
165 
166  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
167  if( gamenet_man->InNetGame())
168  {
169  Net::Client* client;
170  GameNet::MsgSpecialItem msg;
171  Net::MsgDesc msg_desc;
172 
173  client = gamenet_man->GetClient( 0 );
174  Dbg_Assert( client );
175 
176  msg.m_ObjId = GetObject()->GetID();
177  msg.m_Params = checksum;
178  msg.m_Index = index;
179  msg.m_Bone = bone;
180 
181  msg_desc.m_Data = &msg;
182  msg_desc.m_Length = sizeof( GameNet::MsgSpecialItem );
183  msg_desc.m_Id = GameNet::MSG_ID_CREATE_SPECIAL_ITEM;
184  client->EnqueueMessageToServer( &msg_desc );
185  }
186  }
187  break;
188 
189  // @script | DestroySpecialItem | Destroys a special item attached to the skater, if it exists
190  // @parm int | index | Index of special item to destroy
191  case 0x742188fa: // DestroySpecialItem
192  {
193  int index;
194  pParams->GetInteger( CRCD(0x7f8c98fe,"index"), &index, true );
195 
196  this->DestroySpecialItem( index );
197 
198  GameNet::Manager * gamenet_man = GameNet::Manager::Instance();
199  if( gamenet_man->InNetGame())
200  {
201  Net::Client* client;
202  GameNet::MsgSpecialItem msg;
203  Net::MsgDesc msg_desc;
204 
205  client = gamenet_man->GetClient( 0 );
206  Dbg_Assert( client );
207 
208  msg.m_ObjId = GetObject()->GetID();
209  msg.m_Index = index;
210 
211  msg_desc.m_Data = &msg;
212  msg_desc.m_Length = sizeof( GameNet::MsgSpecialItem );
213  msg_desc.m_Id = GameNet::MSG_ID_DESTROY_SPECIAL_ITEM;
214  client->EnqueueMessageToServer( &msg_desc );
215  }
216  }
217  break;
218 
219  // @script | DestroyAllSpecialItems | Destroys all special items currently attached to the skater
220  case 0xa1baa25e: // DestroyAllSpecialItems
221  {
222  this->DestroyAllSpecialItems();
223  }
224  break;
225 
226  // @script | SpecialItemExists | Returns whether a special item exists
227  // @parm int | index | Index of special item to check
228  case 0xce887492: // SpecialItemExists
229  {
230  int index;
231  if ( pParams->GetInteger( CRCD(0x7f8c98fe,"index"), &index, false ) )
232  {
233  Dbg_MsgAssert( index >= 0 && index < vMAX_SPECIAL_ITEMS, ( "Special item index %d must be between 0 and %d", index, vMAX_SPECIAL_ITEMS ) );
234  return ( mp_special_items[index] ) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
235  }
236  else
237  {
238  for ( int i = 0; i < vMAX_SPECIAL_ITEMS; i++ )
239  {
240  if ( mp_special_items[i] )
241  {
242  return CBaseComponent::MF_TRUE;
243  }
244  }
245  return CBaseComponent::MF_FALSE;
246  }
247  }
248  break;
249 
250  default:
251  return CBaseComponent::MF_NOT_EXECUTED;
252  }
253 
254  // the "default" case of the switch statement handles
255  // unrecognized functions; if we make it down here,
256  // that means that the component already handled it
257  // somehow
258  return CBaseComponent::MF_TRUE;
259 }

290 Obj::CCompositeObject* CSpecialItemComponent::CreateSpecialItem( int index, Script::CStruct* pNodeData )
291 {
292  Dbg_MsgAssert( index >= 0 && index < vMAX_SPECIAL_ITEMS, ( "Special item index %d must be between 0 and %d", index, vMAX_SPECIAL_ITEMS ) );
293  Dbg_MsgAssert( pNodeData, ( "Missing parameter" ) );
294 
295  if ( mp_special_items[index] )
296  {
297  // TODO: Maybe this should just be a warning?
298 // Dbg_MsgAssert( 0, ( "Special item index %d not empty", index ) );
299  DestroySpecialItem( index );
300  }
301 
302 // Dbg_Message( "Adding special item to index %d", index );
303 
304  // eventually, we'll want to create a generic composite object
305  // (maybe by running a script)
306  Mem::PushMemProfile("Game Objects");
307  Obj::CCompositeObject* pGameObj = Obj::CreateGameObj( Mdl::Skate::Instance()->GetObjectManager(), pNodeData );
308  Dbg_Assert( pGameObj );
309  Mem::PopMemProfile();
310 
311  // GJ: This is a special-case function that's used for
312  // replays... I'm not bothering to move it over to the
313  // CCompositeObject stuff, because we've already made the
314  // decision not to make the replay backwards-compatible.
315 // pGameObj->FlagAsSpecialItem();
316 
317  mp_special_items[index] = pGameObj;
318 
319  // add the skater's id to the object's tags
320  // so that the object knows which skater it's tied to
321  Script::CStruct* pTempStructure;
322  pTempStructure = new Script::CStruct;
323  pTempStructure->AddChecksum( "parentId", GetObject()->GetID() );
324  uint32 cleanupScript;
325  if ( pNodeData->GetChecksum( CRCD(0x40764820,"CleanupScript"), &cleanupScript ) )
326  {
327  pTempStructure->AddChecksum( CRCD(0x40764820,"CleanupScript"), cleanupScript );
328  }
329  mp_special_items[index]->SetTagsFromScript( pTempStructure );
330  delete pTempStructure;
331 
332  // in case there's any scaling applied to the model,
333  // apply that to the special item's model as well...
334  Obj::CModelComponent* pModelComponent = GetModelComponentFromObject( GetObject() );
335  Nx::CModel* pModel = NULL;
336  if ( pModelComponent )
337  {
338  pModel = pModelComponent->GetModel();
339  }
340 
341  Obj::CModelComponent* pGameObjModelComponent = GetModelComponentFromObject( pGameObj );
342  Nx::CModel* pGameObjModel = NULL;
343  if ( pGameObjModelComponent )
344  {
345  pGameObjModel = pGameObjModelComponent->GetModel();
346  }
347 
348  // (Mick) ... but only if it has a model (might be just collision)
349  if ( pModel && pGameObjModel )
350  {
351  Mth::Vector scaleVec = pModel->GetScale();
352  pGameObjModel->SetScale( scaleVec ); // (Mick) - see check above
353  }
354 
355  return pGameObj;
356 }

363 void CSpecialItemComponent::DestroySpecialItem( int index )
364 {
365  Dbg_MsgAssert( index >= 0 && index < vMAX_SPECIAL_ITEMS, ( "Special item index %d must be between 0 and %d", index, vMAX_SPECIAL_ITEMS ) );
366 
367  if ( mp_special_items[index] )
368  {
369  // run the cleanup script if it exists...
370  Script::CStruct* pStruct = new Script::CStruct;
371  mp_special_items[index]->CopyTagsToScriptStruct(pStruct);
372  uint32 cleanupScript;
373  if ( pStruct->GetChecksum( CRCD(0x40764820,"CleanupScript"), &cleanupScript, Script::NO_ASSERT ) )
374  {
375  // run a cleanup script if it exists
376  Script::RunScript( cleanupScript, NULL, mp_special_items[index] );
377  }
378  delete pStruct;
379 
380  // don't actually delete it yet...
381  // instead, just mark it as dead, so
382  // that if any objects still refer to
383  // it on this frame, it won't crash...
384  mp_special_items[index]->MarkAsDead();
385  mp_special_items[index] = NULL;
386 
387  // THPS4: flushes dead objects so that we can
388  // recreate it on the same frame
389  // (theoretically, we should have been
390  // able to call "wait 1 gameframe",
391  // but it turns out that CScript::Update()
392  // can be called more than once per frame,
393  // if someone uses the MakeSkaterGoto
394  // command)
395  // THPS5: doesn't seem to be a problem any
396  // more, perhaps due to the subtle changes in
397  // order which object/component logic gets
398  // run. having this line in here does cause
399  // some issues now, in that it's possible
400  // for an object to indirectly delete itself from
401  // its update() function, which is bad. i've
402  // taken this line out and will fix any
403  // related bugs as they arise.
404 // Mdl::Skate::Instance()->GetObjectManager()->FlushDeadObjects();
405  }
406 }
