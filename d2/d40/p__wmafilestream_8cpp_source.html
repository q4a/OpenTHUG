52 // Define the source packet size:
53 // This value is hard-coded assuming a WMA file of stero, 16bit resolution. If
54 // this value can by dynamically set based on the wma format, keeping in mind
55 // that wma needs enough buffer for a minimum of 2048 samples worth of PCM data
56 #define WMASTRM_SOURCE_PACKET_BYTES ( 2048 * 2 * 2 )
57 
58 
59 
60 // This is a homegrown value which is used in conjunction with the existing
61 // DSound packet status values. This indicates a packet which has been written
62 // to, but which has not yet been submitted to the renderer. The low-order 24
63 // bits contain a timestamp. Packets should be submitted lowest-timestamp first.
64 #define XMEDIAPACKET_STATUS_AWAITING_RENDER 0x40000000UL

86 uint32 CALLBACK WMAXMediaObjectDataCallback( LPVOID pContext, uint32 offset, uint32 num_bytes, LPVOID *ppData )
87 {
88  Dbg_Assert( pContext != NULL );
89  Dbg_Assert( ppData != NULL );
90 
91  CWMAFileStream *p_this = (CWMAFileStream*)pContext;
92  Dbg_Assert(( p_this->m_DecoderCreation == 0 ) || ( p_this->m_DecoderCreation == 1 ));
93 
94  *ppData = (BYTE*)( p_this->m_pFileBuffer ) + ( offset % ( 8 * 8192 ));
95 
96  // Update current progress.
97  p_this->m_FileBytesProcessed = offset;
98 
99  return num_bytes;
100 }

173 //-----------------------------------------------------------------------------
174 // Name: AsyncRead()
175 // Desc: Called while the async read is in progress. If the read is still
176 // underway, simply returns. If the read has completed, sets up the
177 // next read. If the file has been completely read, closes the file.
178 //-----------------------------------------------------------------------------
179 void CWMAFileStream::AsyncRead( void )
180 {
181  Dbg_Assert( m_hFile != INVALID_HANDLE_VALUE );
182 
183  // If paused, do nothing.
184  if( m_Paused )
185  {
186  return;
187  }
188 
189  // See if the previous read is complete.
190  uint32 dwBytesTransferred;
191  bool bIsReadDone = GetOverlappedResult( m_hFile, m_pOverlapped, &dwBytesTransferred, false );
192  uint32 dwLastError = GetLastError();
193 
194  // If the read isn't complete, keep going.
195  if( !bIsReadDone )
196  {
197  Dbg_Assert( dwLastError == ERROR_IO_INCOMPLETE );
198  if( dwLastError != ERROR_IO_INCOMPLETE )
199  {
200  m_AwaitingDeletion = true;
201  }
202  return;
203  }
204 
205  // If we get here, the read is complete.
206  m_pOverlapped->Offset += dwBytesTransferred;
207  m_FileBytesRead += dwBytesTransferred;
208 
209  // If we just read to the first block in the buffer, copy to the extra block at the end, to ensure jitter-free playback.
210  if(( m_SuccessiveReads & 0x07 ) == 0 )
211  {
212  CopyMemory((BYTE*)m_pFileBuffer + ( 8 * 8192 ), (BYTE*)m_pFileBuffer, 8192 );
213  }
214 
215  ++m_SuccessiveReads;
216 
217 # define BYTES_PER_CALL 8192
218 
219  if( dwBytesTransferred < 8192 )
220  {
221  // We've reached the end of the file during the call to ReadFile.
222 
223  // Close the file (if not using the global WAD file).
224  if( !m_bUseWAD )
225  {
226  bool bSuccess = CloseHandle( m_hFile );
227  Dbg_Assert( bSuccess );
228  }
229 
230  m_hFile = INVALID_HANDLE_VALUE;
231 
232  // All done
233  m_ReadComplete = true;
234  }
235  else
236  {
237  if( m_bUseWAD && ( m_FileBytesRead >= (int)m_dwWADLength ))
238  {
239  m_hFile = INVALID_HANDLE_VALUE;
240 
241  // All done
242  m_ReadComplete = true;
243  return;
244  }
245 
246  // We still have more data to read. Start another asynchronous read from the file.
247  bool bComplete = ReadFile( m_hFile, (BYTE*)m_pFileBuffer + ( m_FileBytesRead % ( 8 * 8192 )), BYTES_PER_CALL, NULL, m_pOverlapped );
248  dwLastError = GetLastError();
249 
250  // Deal with hitting EOF (for files that are some exact multiple of 8192 bytes).
251  if( bComplete || ( !bComplete && ( dwLastError == ERROR_HANDLE_EOF )))
252  {
253  // Close the file
254  if( !m_bUseWAD )
255  {
256  bool bSuccess = CloseHandle( m_hFile );
257  Dbg_Assert( bSuccess );
258  }
259  m_hFile = INVALID_HANDLE_VALUE;
260 
261  // All done
262  m_ReadComplete = true;
263  }
264  else
265  {
266  Dbg_MsgAssert( bComplete || ( !bComplete && ( dwLastError == ERROR_IO_PENDING )), ( "ReadFile error: %x\n", dwLastError ));
267  if( !bComplete && ( dwLastError != ERROR_IO_PENDING ))
268  {
269  // There was a problem, so shut this stream down.
270  m_AwaitingDeletion = true;
271  }
272  }
273  }
274 }

280 // Desc: Initializes the wave file streaming subsystem.

282 HRESULT CWMAFileStream::Initialize( HANDLE h_file, unsigned int offset, unsigned int length, void* fileBuffer )
283 {
284  m_dwPercentCompleted = 0;
285 
286  // At this stage we don't want to create the decoder or the stream. We do want to just allocate the read
287  // buffer, and start pulling in the data. Once sufficient data has been grabbed, we can analyze the header
288  // and create the required objects for playback.
289  m_hFile = h_file;
290  m_bUseWAD = true;
291  m_dwWADOffset = offset;
292  m_dwWADLength = length;
293 
294  // Grab a 72k (9 * 8k packets) read buffer. Must be uint32 aligned.
295  // The last packet mirrors the first packet, for cases when the decoder reads past the end of the ring buffer.
296  m_pFileBuffer = fileBuffer;
297  Dbg_Assert( uint32( m_pFileBuffer ) % sizeof( uint32 ) == 0 );
298 
299  return PostInitialize();
300 }

334 //-----------------------------------------------------------------------------
335 // Name: PostInitialize()
336 // Desc: Initialisation stuff following the file creation
337 //-----------------------------------------------------------------------------
338 HRESULT CWMAFileStream::PostInitialize( void )
339 {
340  // Start the asynchronous read from the start of the file.
341  m_FirstRead = true;
342  m_ReadComplete = false;
343  m_FileBytesRead = 0;
344  m_FileBytesProcessed = 0;
345  m_SuccessiveReads = 0;
346 
347  if( m_bUseWAD )
348  {
349  m_pOverlapped->Offset = m_dwWADOffset;
350  }
351  else
352  {
353  m_pOverlapped->Offset = 0;
354  }
355  m_pOverlapped->OffsetHigh = 0;
356 
357  bool bComplete = ReadFile( m_hFile, m_pFileBuffer, BYTES_PER_CALL, NULL, m_pOverlapped );
358  if( !bComplete )
359  {
360  uint32 dwLastError = GetLastError();
361  Dbg_Assert( dwLastError == ERROR_IO_PENDING );
362  if( dwLastError != ERROR_IO_PENDING )
363  {
364  return HRESULT_FROM_WIN32( dwLastError );
365  }
366  }
367  else
368  {
369  m_ReadComplete = true;
370  }
371 
372  // That's it for now.
373  return S_OK;
374 }

412 //-----------------------------------------------------------------------------
413 // Name: Process()
414 // Desc: Performs any work necessary to keep the stream playing.
415 //-----------------------------------------------------------------------------
416 HRESULT CWMAFileStream::Process( void )
417 {
418  HRESULT hr;
419  DSSTREAMDESC dssd;
420  uint32 dwPacketIndex;
421 
422  // Do nothing if waiting to die.
423  if( m_AwaitingDeletion )
424  {
425  return S_OK;
426  }
427 
428  // Do we need to kick off another read of data? Don't read anymore if we have read ahead more than 32k.
429  if( !m_ReadComplete )
430  {
431  if( m_FileBytesRead > ( m_FileBytesProcessed + 32768 ))
432  {
433 // OutputDebugString( "waiting...\n" );
434  }
435  else
436  {
437  AsyncRead();
438 // OutputDebugString( "reading...\n" );
439  }
440  }
441  else
442  {
443 // OutputDebugString( "complete...\n" );
444  }
445 
446  // Has the first block of raw data been read? If so we need to instantiate the playback objects and data buffers.
447  if( m_FirstRead && ( m_FileBytesRead > 1024 ))
448  {
449  if( m_pSourceFilter == NULL )
450  {
451  // Create the thread which will create the in-memory decoder.
452  m_DecoderCreation = 0;
453 
454  HRESULT hr = WmaCreateInMemoryDecoder( WMAXMediaObjectDataCallback, // Callback pointer.
455  this, // Callback context pointer.
456  3, // Yield rate during decoding.
457  &m_wfxSourceFormat, // Source format description.
458  (LPXMEDIAOBJECT*)&m_pSourceFilter ); // Pointer to media object.
459  if( FAILED( hr ))
460  {
461  // Signal a failed creation.
462  m_DecoderCreation = 2;
463  m_pSourceFilter = NULL;
464  }
465  else
466  {
467  m_DecoderCreation = 1;
468  }
469 
470  // That's all we can do until the decoder has been instantiated.
471  }
472 
473  if( m_DecoderCreation == 0 )
474  {
475  // Still waiting to create decoder.
476  return S_OK;
477  }
478  else if( m_DecoderCreation == 2 )
479  {
480  // Failed to create decoder, just mark for deletion.
481  m_AwaitingDeletion = true;
482  return S_OK;
483  }
484  else if( m_DecoderCreation == 1 )
485  {
486  // Managed to create decoder.
487  Dbg_Assert( m_pSourceFilter != NULL );
488 
489  m_FirstRead = false;
490 
491  // Create the render (DirectSoundStream) filter.
492  DSMIXBINS dsmixbins;
493  DSMIXBINVOLUMEPAIR dsmbvp[7];
494  ZeroMemory( &dssd, sizeof( dssd ));
495 
496  dssd.dwFlags = 0;
497  dssd.dwMaxAttachedPackets = WMASTRM_PACKET_COUNT;
498  dssd.lpwfxFormat = &m_wfxSourceFormat;
499  dssd.lpMixBins = &dsmixbins;
500 
501  if( m_bUse3D )
502  {
503  // This is only designed to be fed a mono signal.
504  Dbg_Assert( m_wfxSourceFormat.nChannels == 1 );
505 
506  dsmixbins.dwMixBinCount = 6;
507  dsmixbins.lpMixBinVolumePairs = dsmbvp;
508  dsmbvp[0].dwMixBin = DSMIXBIN_3D_FRONT_LEFT;
509  dsmbvp[0].lVolume = DSBVOLUME_EFFECTIVE_MIN;
510  dsmbvp[1].dwMixBin = DSMIXBIN_3D_FRONT_RIGHT;
511  dsmbvp[1].lVolume = DSBVOLUME_EFFECTIVE_MIN;
512  dsmbvp[2].dwMixBin = DSMIXBIN_3D_BACK_LEFT;
513  dsmbvp[2].lVolume = DSBVOLUME_EFFECTIVE_MIN;
514  dsmbvp[3].dwMixBin = DSMIXBIN_3D_BACK_RIGHT;
515  dsmbvp[3].lVolume = DSBVOLUME_EFFECTIVE_MIN;
516  dsmbvp[4].dwMixBin = DSMIXBIN_FRONT_CENTER;
517  dsmbvp[4].lVolume = DSBVOLUME_EFFECTIVE_MIN;
518  dsmbvp[5].dwMixBin = DSMIXBIN_I3DL2;
519  dsmbvp[5].lVolume = DSBVOLUME_EFFECTIVE_MIN;
520 
521  m_Mixbins = ( 1 << DSMIXBIN_3D_FRONT_LEFT ) |
522  ( 1 << DSMIXBIN_3D_FRONT_RIGHT ) |
523  ( 1 << DSMIXBIN_3D_BACK_LEFT ) |
524  ( 1 << DSMIXBIN_3D_BACK_RIGHT ) |
525  ( 1 << DSMIXBIN_FRONT_CENTER ) |
526  ( 1 << DSMIXBIN_I3DL2 );
527  m_NumMixbins = dsmixbins.dwMixBinCount;
528  }
529  else
530  {
531  // If we are playing a music track, and if proper 5.1 output is selected, we want to feed the music to the left and
532  // right back speakers with a slight echo via mixbins 5 and 6.
533  if( XGetAudioFlags() & XC_AUDIO_FLAGS_ENABLE_AC3 )
534  {
535  // This is only designed to be fed a stereo signal.
536  Dbg_Assert( m_wfxSourceFormat.nChannels == 2 );
537 
538  dsmixbins.dwMixBinCount = 4;
539  dsmixbins.lpMixBinVolumePairs = dsmbvp;
540  dsmbvp[0].dwMixBin = DSMIXBIN_FRONT_LEFT;
541  dsmbvp[0].lVolume = DSBVOLUME_MIN;
542  dsmbvp[1].dwMixBin = DSMIXBIN_FRONT_RIGHT;
543  dsmbvp[1].lVolume = DSBVOLUME_MIN;
544  dsmbvp[2].dwMixBin = DSMIXBIN_FXSEND_5;
545  dsmbvp[2].lVolume = DSBVOLUME_MIN;
546  dsmbvp[3].dwMixBin = DSMIXBIN_FXSEND_6;
547  dsmbvp[3].lVolume = DSBVOLUME_MIN;
548 
549  m_Mixbins = ( 1 << DSMIXBIN_FRONT_LEFT ) |
550  ( 1 << DSMIXBIN_FRONT_RIGHT ) |
551  ( 1 << DSMIXBIN_FXSEND_5 ) |
552  ( 1 << DSMIXBIN_FXSEND_6 );
553  m_NumMixbins = dsmixbins.dwMixBinCount;
554  }
555  else
556  {
557  // This is only designed to be fed a stereo signal.
558  Dbg_Assert( m_wfxSourceFormat.nChannels == 2 );
559 
560  dsmixbins.dwMixBinCount = 4;
561  dsmixbins.lpMixBinVolumePairs = dsmbvp;
562  dsmbvp[0].dwMixBin = DSMIXBIN_FRONT_LEFT;
563  dsmbvp[0].lVolume = DSBVOLUME_MIN;
564  dsmbvp[1].dwMixBin = DSMIXBIN_FRONT_RIGHT;
565  dsmbvp[1].lVolume = DSBVOLUME_MIN;
566  dsmbvp[2].dwMixBin = DSMIXBIN_BACK_LEFT;
567  dsmbvp[2].lVolume = DSBVOLUME_MIN;
568  dsmbvp[3].dwMixBin = DSMIXBIN_BACK_RIGHT;
569  dsmbvp[3].lVolume = DSBVOLUME_MIN;
570 
571  m_Mixbins = ( 1 << DSMIXBIN_FRONT_LEFT ) |
572  ( 1 << DSMIXBIN_FRONT_RIGHT ) |
573  ( 1 << DSMIXBIN_BACK_LEFT ) |
574  ( 1 << DSMIXBIN_BACK_RIGHT );
575  m_NumMixbins = dsmixbins.dwMixBinCount;
576  }
577  }
578 
579  hr = DirectSoundCreateStream( &dssd, &m_pRenderFilter );
580  if( FAILED( hr ))
581  {
582  Dbg_Assert( 0 );
583  m_pRenderFilter = NULL;
584  m_AwaitingDeletion = true;
585  return S_OK;
586  }
587 
588  // Set deferred volume if present.
589  SetDeferredVolume();
590 
591  // Handle deferred pause.
592  if( m_Paused )
593  {
594  m_pRenderFilter->Pause( DSSTREAMPAUSE_PAUSE );
595  }
596  }
597  }
598 
599  // We only want to do processing when there is sufficient data available.
600  if( m_ReadComplete || ( m_FileBytesRead >= ( m_FileBytesProcessed + 8192 )))
601  {
602  if( !m_Completed && m_pSourceFilter && m_pRenderFilter )
603  {
604  // Find a free packet. If there's none free, we don't have anything to do.
605  if( FindFreePacket( &dwPacketIndex ))
606  {
607  // Read from the source filter.
608  hr = ProcessSource( dwPacketIndex );
609  if( FAILED( hr ) )
610  {
611  return hr;
612  }
613  }
614 
615  // Don't want to submit if we're paused, or we're flagged not to play. (Flagging is used to allow
616  // the stream to be spooled up ready to play instantly when triggered).
617  if(( !m_Paused ) && m_bOkayToPlay )
618  {
619  if( FindRenderablePacket( &dwPacketIndex ))
620  {
621  // Send the data to the renderer
622  hr = ProcessRenderer( dwPacketIndex );
623  if( FAILED( hr ))
624  {
625  return hr;
626  }
627  }
628  }
629  }
630  }
631  return S_OK;
632 }

637 //-----------------------------------------------------------------------------
638 // Name: FindFreePacket()
639 // Desc: Finds a render packet available for processing.
640 //-----------------------------------------------------------------------------
641 bool CWMAFileStream::FindFreePacket( uint32* pdwPacketIndex )
642 {
643  for( uint32 dwPacketIndex = 0; dwPacketIndex < WMASTRM_PACKET_COUNT; ++dwPacketIndex )
644  {
645  if(( m_adwPacketStatus[dwPacketIndex] != XMEDIAPACKET_STATUS_PENDING ) &&
646  (( m_adwPacketStatus[dwPacketIndex] & XMEDIAPACKET_STATUS_AWAITING_RENDER ) != XMEDIAPACKET_STATUS_AWAITING_RENDER ))
647  {
648  if( pdwPacketIndex )
649  {
650  *pdwPacketIndex = dwPacketIndex;
651 
652  // Mark this packet as awaiting rendering.
653  uint32 timestamp = (uint32)Tmr::GetVblanks();
654  m_adwPacketStatus[dwPacketIndex] = XMEDIAPACKET_STATUS_AWAITING_RENDER | ( timestamp & 0xFFFFFFUL );
655  }
656  return true;
657  }
658  }
659  return false;
660 }

664 //-----------------------------------------------------------------------------
665 // Name: FindFreePacket()
666 // Desc: Finds a render packet available for rendering.
667 //-----------------------------------------------------------------------------
668 bool CWMAFileStream::FindRenderablePacket( uint32* pdwPacketIndex )
669 {
670  // Found nothing yet.
671  uint32 lowest_timestamp = 0xFFFFFFUL;
672  uint32 lowest_index = WMASTRM_PACKET_COUNT;
673 
674  for( uint32 dwPacketIndex = 0; dwPacketIndex < WMASTRM_PACKET_COUNT; ++dwPacketIndex )
675  {
676  if(( m_adwPacketStatus[dwPacketIndex] & XMEDIAPACKET_STATUS_AWAITING_RENDER ) == XMEDIAPACKET_STATUS_AWAITING_RENDER )
677  {
678  uint32 timestamp = m_adwPacketStatus[dwPacketIndex] & 0xFFFFFFUL;
679  if( timestamp < lowest_timestamp )
680  {
681  lowest_timestamp = timestamp;
682  lowest_index = dwPacketIndex;
683  }
684  }
685  }
686 
687  // If we found a packet, return its index.
688  if( lowest_index < WMASTRM_PACKET_COUNT )
689  {
690  if( pdwPacketIndex )
691  {
692  *pdwPacketIndex = lowest_index;
693  }
694  return true;
695  }
696 
697  // No packets awaiting rendering.
698  return false;
699 }

703 //-----------------------------------------------------------------------------
704 // Name: ProcessSource()
705 // Desc: Reads data from the source filter.
706 //-----------------------------------------------------------------------------
707 HRESULT CWMAFileStream::ProcessSource( uint32 dwPacketIndex )
708 {
709  uint32 dwTotalSourceUsed = 0;
710  uint32 dwSourceUsed;
711  XMEDIAPACKET xmp;
712  HRESULT hr;
713 
714  if( m_pvSourceBuffer == NULL )
715  {
716  CreateSourceBuffer();
717  }
718 
719  // We're going to read a full packet's worth of data into the source buffer.
720  ZeroMemory( &xmp, sizeof( xmp ));
721  xmp.pvBuffer = (BYTE*)m_pvSourceBuffer + (dwPacketIndex * WMASTRM_SOURCE_PACKET_BYTES);
722  xmp.dwMaxSize = WMASTRM_SOURCE_PACKET_BYTES;

816 void CWMAFileStream::SetVolume( float volume )
817 {

821  int i_volume_rear = DSBVOLUME_EFFECTIVE_MIN;

823  {
824  // Figure base volume.
825  float attenuation = 20.0f * log10f( volume * 0.01f );
826  i_volume = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
827  if( i_volume < DSBVOLUME_EFFECTIVE_MIN )
828  i_volume = DSBVOLUME_EFFECTIVE_MIN;
829  else if( i_volume > DSBVOLUME_MAX )
830  i_volume = DSBVOLUME_MAX;
831 
832  // Also figure half volume, in case we are routing to the back speakers.
833  attenuation = 20.0f * log10f( volume * 0.5f * 0.01f );
834  i_volume_rear = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
835  if( i_volume_rear < DSBVOLUME_EFFECTIVE_MIN )
836  i_volume_rear = DSBVOLUME_EFFECTIVE_MIN;
837  else if( i_volume_rear > DSBVOLUME_MAX )
838  i_volume_rear = DSBVOLUME_MAX;
839  }
840 
841  // Set individual mixbins for panning.
842  DSMIXBINS dsmixbins;
843  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
844 
845  dsmixbins.dwMixBinCount = 0;
846  dsmixbins.lpMixBinVolumePairs = dsmbvp;
847 
848  if( i_volume > DSBVOLUME_EFFECTIVE_MIN )
849  {
850  // Set the volume up depending on how the initial mixbins were set up.
851  int mbbf = 0;
852  for( uint32 mb = 0; mb < m_NumMixbins; ++mb )

860

862  if(( mbbf == DSMIXBIN_FXSEND_5 ) || ( mbbf == DSMIXBIN_FXSEND_6 ) || ( mbbf == DSMIXBIN_BACK_LEFT ) || ( mbbf == DSMIXBIN_BACK_RIGHT ))
863  {
864  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volume_rear;
865  }
866  else

868  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volume;
869  }

871  ++mbbf;
872  }
873  }
874 
875  // Set all speaker volumes.
876  m_pRenderFilter->SetMixBinVolumes( &dsmixbins );
877 
878  // Set overall buffer volume.
879  if( dsmixbins.dwMixBinCount > 0 )
880  {
881  m_pRenderFilter->SetVolume( DSBVOLUME_MAX );
882  }
883  else
884  {
885  m_pRenderFilter->SetVolume( DSBVOLUME_MIN );
886  }
887  }
888  else

890  m_SetDeferredVolume = true;
891  m_DeferredVolume[0] = volume;

900 //-----------------------------------------------------------------------------
901 void CWMAFileStream::SetVolume( float volL, float volR )
902 {
903  if( m_pRenderFilter )
904  {
905  // This array will hold individual volumes for the five speakers.
906  // In order, they are: front left, center, front right, rear right, rear left.
907  float volumes[5];
908  int i_volumes[5], max_i_volume;
909  memset( volumes, 0, sizeof( float ) * 5 );
910 
911  if(( volL == 0.0f ) && ( volR == 0.0f ))
912  {
913  // Pointless doing any more work.
914  }
915  else
916  {
917  // Get the length of the vector here which will be used to multiply out the normalised speaker volumes.
918  Mth::Vector test( fabsf( volL ), fabsf( volR ), 0.0f, 0.0f );
919  float amplitude = test.Length();
920 
921  // Look just at the normalized right component to figure the sound angle from Matt's calculations.
922  test.Normalize();
923 
924  float angle;
925  angle = asinf( test[Y] );
926  angle = ( angle * 2.0f ) - ( Mth::PI * 0.5f );
927  angle = ( volL < 0.0f ) ? ( Mth::PI - angle ) : angle;
928 
929  // Now figure volumes based on speaker coverage.
930  angle = Mth::RadToDeg( angle );
931 
932  Spt::SingletonPtr< Sfx::CSfxManager > sfx_manager;
933  sfx_manager->Get5ChannelMultipliers( angle, &volumes[0] );
934 
935  // Now readjust the relative values...
936  for( int v = 0; v < 5; ++v )
937  {
938  // Scale back up to original amplitude.
939  volumes[v] *= amplitude;
940 
941  if( volumes[v] > 100.0f )
942  volumes[v] = 100.0f;
943  }
944  }
945 
946  // Now figure the attenuation of the sound. To convert to a decibel value, figure the ratio of requested
947  // volume versus max volume, then calculate the log10 and multiply by (10 * 2). (The 2 is because sound
948  // power varies as square of pressure, and squaring doubles the log value).
949  max_i_volume = DSBVOLUME_EFFECTIVE_MIN;
950  for( int v = 0; v < 5; ++v )
951  {
952  if( volumes[v] > 0.0f )
953  {
954  float attenuation = 20.0f * log10f( volumes[v] * 0.01f );
955  i_volumes[v] = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
956  if( i_volumes[v] < DSBVOLUME_EFFECTIVE_MIN )
957  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
958  else if( i_volumes[v] > DSBVOLUME_MAX )
959  i_volumes[v] = DSBVOLUME_MAX;
960 
961  if( i_volumes[v] > max_i_volume )
962  max_i_volume = i_volumes[v];
963  }
964  else
965  {
966  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
967  }
968  }
969 
970  // Set individual mixbins for panning.
971  DSMIXBINS dsmixbins;
972  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
973 
974  dsmixbins.dwMixBinCount = 0;
975  dsmixbins.lpMixBinVolumePairs = dsmbvp;
976 
977  if( i_volumes[0] > DSBVOLUME_EFFECTIVE_MIN )
978  {
979  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_LEFT;
980  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[0];
981  dsmixbins.dwMixBinCount++;
982  }
983  if( i_volumes[1] > DSBVOLUME_EFFECTIVE_MIN )
984  {
985  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_RIGHT;
986  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[1];
987  dsmixbins.dwMixBinCount++;
988  }
989  if( i_volumes[2] > DSBVOLUME_EFFECTIVE_MIN )
990  {
991  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_LEFT;
992  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[2];
993  dsmixbins.dwMixBinCount++;
994  }
995  if( i_volumes[3] > DSBVOLUME_EFFECTIVE_MIN )
996  {
997  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_RIGHT;
998  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[3];
999  dsmixbins.dwMixBinCount++;
1000  }
1001  if( i_volumes[4] > DSBVOLUME_EFFECTIVE_MIN )
1002  {
1003  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_FRONT_CENTER;
1004  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[4];
1005  dsmixbins.dwMixBinCount++;
1006  }
1007  if( dsmixbins.dwMixBinCount > 0 )
1008  {
1009  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_I3DL2;
1010  dsmbvp[dsmixbins.dwMixBinCount].lVolume = DSBVOLUME_MAX;
1011  dsmixbins.dwMixBinCount++;
1012  }
1013 
1014  // Set all speaker volumes.
1015  m_pRenderFilter->SetMixBinVolumes( &dsmixbins );
1016 
1017  // Set overall buffer volume.
1018  if( dsmixbins.dwMixBinCount > 0 )
1019  {
1020  m_pRenderFilter->SetVolume( DSBVOLUME_MAX );
1021  }
1022  else
1023  {
1024  m_pRenderFilter->SetVolume( DSBVOLUME_MIN );
1025  }
1026  }
1027  else
1028  {
1029  m_SetDeferredVolumeLR = true;
1030  m_DeferredVolume[0] = volL;
1031  m_DeferredVolume[1] = volR;
1032  }
