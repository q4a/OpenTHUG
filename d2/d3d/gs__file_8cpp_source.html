48 void LoadAllStartupQBFiles()
49 {
50  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
51 #ifdef __PLAT_NGC__
52  switch ( Config::GetLanguage() )
53  {
54  case Config::LANGUAGE_FRENCH:
55  Pip::LoadPre("qb.prf");
56  break;
57  case Config::LANGUAGE_GERMAN:
58  Pip::LoadPre("qb.prd");
59  break;
60  default:
61  Pip::LoadPre("qb.pre");
62  break;
63  }
64 #else
65  Pip::LoadPre("qb.pre");
66 #endif // __PLAT_NGC__
67  Mem::Manager::sHandle().PopContext();
68 
69  // Load the qdir.txt file
70  const char *p_qdir=(const char *)Pip::Load(QDIR_FILE);
71  uint32 file_size=Pip::GetFileSize(QDIR_FILE);
72 
73  UsePermanentStringHeap();
74 
75  // Load each of the files listed.
76  #define MAX_FILENAME_CHARS 200
77  char p_file_name[MAX_FILENAME_CHARS+1];
78 
79  const char *p_scan=p_qdir;
80  while (p_scan < p_qdir+file_size)
81  {
82  int c=0;
83  while (p_scan < p_qdir+file_size && *p_scan!=0x0d && *p_scan!=0x0a)
84  {
85  Dbg_MsgAssert(c<MAX_FILENAME_CHARS,("File name too long"));
86  p_file_name[c++]=*p_scan++;
87  }
88  p_file_name[c]=0;
89  // If the above loop broke out because *p_scan was 0x0d or 0x0a then
90  // this will skip over it. If p_scan was >= p_qdir+file_size then it
91  // still will be after incrementing p_scan so that's OK too.
92  ++p_scan;
93 
94  // Skip any empty strings, which will happen when encountering 0x0d,0x0a pairs.
95  if (*p_file_name)
96  {
97  // Note: p_file_name will contain the complete path, eg c:\skate5\data\scripts\blaa.qb
98 
99  // Strip off any preceding data path.
100  char *p_data_backslash=strstr(p_file_name,"data\\");
101  if (p_data_backslash)
102  {
103  strcpy( p_file_name, p_data_backslash+5); // Safe cos it will copy backwards
104  }
105 
106  // Make sure this script isn't already loaded.
107  SkateScript::UnloadQB( Crc::GenerateCRCFromString(p_file_name) );
108 
109  SkateScript::LoadQB(p_file_name,
110  // We do want assertions if duplicate symbols when loading all
111  // the qb's on startup. (Just not when reloading a qb)
112  ASSERT_IF_DUPLICATE_SYMBOLS);
113  }
114 #ifdef __PLAT_NGC__
115  NsDisplay::doReset();
116 #endif // __PLAT_NGC__
117 
118  }
119 
120  UseRegularStringHeap();
121 
122  Pip::Unload(QDIR_FILE);
123 #ifdef __PLAT_NGC__
124  switch ( Config::GetLanguage() )
125  {
126  case Config::LANGUAGE_FRENCH:
127  Pip::UnloadPre("qb.prf");
128  break;
129  case Config::LANGUAGE_GERMAN:
130  Pip::UnloadPre("qb.prd");
131  break;
132  default:
133  Pip::UnloadPre("qb.pre");
134  break;
135  }
136 #else
137  Pip::UnloadPre("qb.pre");
138 #endif // __PLAT_NGC__
139 
140  // A very small memory optimization ...
141  // Remove the debugger_names array since it only exists to get its checksum names registered
142  // for the script debugger. They will have got registered by now, so the array is no longer
143  // needed. Saves about 3K.
144  CSymbolTableEntry *p_debugger_names=LookUpSymbol(CRCD(0xc2f97ba7,"debugger_names"));
145  if (p_debugger_names)
146  {
147  CleanUpAndRemoveSymbol(p_debugger_names);
148  }
149 }

151 // Calls the Script::LoadQB, and then do any game-specific stuff that needs to be done when a qb is reloaded, such as
152 // generating the node name hash table & prefix info
153 void LoadQB(const char *p_fileName, EBoolAssertIfDuplicateSymbols assertIfDuplicateSymbols)
154 {
155  Dbg_MsgAssert(p_fileName,("NULL p_fileName"));
156  //if (strcmp("..\\runnow.qb",p_fileName) != 0)
157  //{
158  // printf("Loading %s\n",p_fileName); // REMOVE
159  //}
160  // Call the non-game-specific LoadQB, which open the qb and create all the symbols defined within.
161  Script::LoadQB(p_fileName, assertIfDuplicateSymbols);
162 
163  CSymbolTableEntry *p_sym=GetNextSymbolTableEntry();
164  while (p_sym)
165  {
166  if (p_sym->mGotReloaded &&
167  p_sym->mType==ESYMBOLTYPE_ARRAY &&
168  p_sym->mNameChecksum==0xc472ecc5/*NodeArray*/)
169  {
170  // This will resize the NodeArray, adding the extra nodes required by the goal editor.
171  // Note that p_sym->mpArray is still valid after doing this, because the resizing only
172  // causes a reallocation of a buffer inside the CArray, the CArray itself is not reallocated.
173  Obj::InsertGoalEditorNodes();
174 
175  CreateNodeNameHashTable();
176  GeneratePrefixInfo();
177 
178  // Do a bit of node array processing.
179  int size=p_sym->mpArray->GetSize();
180  for (int i=0; i<size; ++i)
181  {
182  CStruct *p_node=p_sym->mpArray->GetStructure(i);
183 
184  // K: This swapping of Position for Pos used to be at a lower level in parse.cpp,
185  // but that had the problem that it was doing the swap on all structures, including those
186  // passed to commands in scripts. So I moved it here so as to only affect those structures
187  // in the node array.
188  Mth::Vector p(0.0f,0.0f,0.0f);
189  if (p_node->GetVector(CRCD(0xb9d31b0a,"Position"),&p))
190  {
191  // GJ: The "position" currently written out
192  // by the exporter has an incorrect Z, and there
193  // are a bunch of places in the game code which
194  // negate the Z value in order to correct it.
195  // We're currently in the process of phasing this
196  // "Position" out, in favor of the "Pos" vector,
197  // which will have the correct Z. For backwards
198  // compatibility, the code here removes the old
199  // "Position" vector if it exists and replaces
200  // it with the new "Pos" vector. Eventually,
201  // once all the levels have been re-exported,
202  // we can remove this kludge.
203  p_node->RemoveComponent(CRCD(0xb9d31b0a,"Position"));
204  p_node->AddVector(CRCD(0x7f261953,"Pos"),p[X],p[Y],-p[Z]);
205  }
206  }
207 
208  // Do some path preprocessing. This adds PathNum parameters to all nodes that
209  // are on a path so that one can quickly find out what path a particular car is
210  // on without having to scan through a bunch of nodes.
211  Obj::CPathMan::Instance()->AddPathInfoToNodeArray();
212 
213  // Reset the reloaded flag, otherwise it will stick on forever.
214  p_sym->mGotReloaded=false;
215 
216  // No need to keep checking for more NodeArray's, there can be only one.
217  break;
218  }
219  p_sym=GetNextSymbolTableEntry(p_sym);
220  }
221 
222  // Make sure checksum name lookup table is removed since it is only needed for the node array.
223  RemoveChecksumNameLookupHashTable();
224 }

226 // Calls the Script::UnloadQB and then does any game-specific stuff that needs to be done
227 // when a qb is unloaded, such as resetting the node name hash table & prefix info.
228 void UnloadQB(uint32 fileNameChecksum)
229 {
230  // Call the non-game-specific UnloadQB, which will delete all the symbols that came from this qb.
231  Script::UnloadQB(fileNameChecksum);
232 
233  if (!GetArray(0xc472ecc5/*NodeArray*/))
234  {
235  // If there is no node array any more then make sure the prefix info and node name lookup
236  // info is cleaned up.
237  DeletePrefixInfo();
238  ClearNodeNameHashTable();
239  }
240 }

242 uint32 GenerateFileNameChecksum(const char *p_fileName)
243 {
244  Dbg_MsgAssert(p_fileName,("NULL p_fileName"));
245 
246  // TODO May need to process the file name a bit first, eg prefix it with the
247  // full path if it is not there already, before calculating its checksum, in order to ensure a match.
248  // We may want UnloadQB to be able to be passed a filename without the full path.
249 
250  // For the moment though, just calculate the checksum for the string as it is.
251  return Crc::GenerateCRCFromString(p_fileName);
252 }
