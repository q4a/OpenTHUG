55 uint32 switch_set_callback(void *p_nothing)
56 {
57  int core;
58 
59  for (core = 0; core < NUM_CORES; core++)
60  {
61  // First, get the status
62  s_spu_reg_endx[core] = sceSdGetSwitch( core | SD_S_ENDX );
63 
64  if (s_spu_kon_changed[core])
65  {
66  sceSdSetSwitch( core | SD_S_KON, s_spu_reg_kon[core] );
67  s_spu_kon_changed[core] = FALSE;
68  s_spu_reg_endx[core] &= ~s_spu_reg_kon[core]; // Update the status since SPU won't do it for another tick
69  s_spu_reg_kon[core] = 0;
70  }
71 
72  if (s_spu_koff_changed[core])
73  {
74  sceSdSetSwitch( core | SD_S_KOFF, s_spu_reg_koff[core] );
75  s_spu_koff_changed[core] = FALSE;
76  s_spu_reg_endx[core] |= s_spu_reg_koff[core]; // Update the status since SPU won't do it for another tick
77  s_spu_reg_koff[core] = 0;
78  }
79  }
80 
81  s_switch_alarm_set = FALSE;
82  s_send_spu_status = TRUE;
83 
84  // And wake up the dispatch thread
85  iSignalSema(gSndSem);
86 
87  return 0;
88 }

92 void queue_set_switch(unsigned short entry, unsigned int value)
93 {
94  int core = entry & 0x1;
95  int switch_reg = entry & 0xFF00;
96  int oldStat;
97 
98  // Make sure alarm callback doesn't run here
99  CpuSuspendIntr(&oldStat);
100 
101  switch (switch_reg)
102  {
103  case SD_S_KON:
104  if(s_spu_reg_koff[core] & value)
105  {
106  //Dbg_MsgAssert(0, ("Turning on a voice that was turned off this frame."));
107  Kprintf("Turning on a voice that was turned off this SPU tick.\n");
108  s_spu_reg_koff[core] &= ~(s_spu_reg_koff[core] & value); // turn off those bits
109  }
110 
111  s_spu_reg_kon[core] |= value;
112  s_spu_kon_changed[core] = TRUE;
113 
114  break;
115 
116  case SD_S_KOFF:
117  if(s_spu_reg_kon[core] & value)
118  {
119  //Dbg_MsgAssert(0, ("Turning off a voice that was turned on this frame."));
120  Kprintf("Turning off a voice that was turned on this SPU tick.\n");
121  s_spu_reg_kon[core] &= ~(s_spu_reg_kon[core] & value); // turn off those bits
122  }
123 
124  s_spu_reg_koff[core] |= value;
125  s_spu_koff_changed[core] = TRUE;
126 
127  break;
128 
129  default:
130  // Just set SPU and exit (so we don't set alarm)
131  sceSdSetSwitch( entry, value );
132  CpuResumeIntr(oldStat);
133  return;
134  }
135 
136  // Turn on alarm if it isn't already on
137  if (!s_switch_alarm_set)
138  {
139  struct SysClock clock;
140  USec2SysClock(SET_PARAM_ALARM_TIME, &clock);
141 
142  SetAlarm(&clock, switch_set_callback, NULL);
143 
144  s_switch_alarm_set = TRUE;
145  }
146 
147  CpuResumeIntr(oldStat);
148 }

298 int sce_sound_loop( void )
299 {
300  int oldStat;
301  int last_cmd_id = -1;
302 
303  // Create semaphore to signal command came in
304  struct SemaParam sem;
305  sem.initCount = 0;
306  sem.maxCount = 1;
307  sem.attr = AT_THFIFO;
308  gSndSem = CreateSema (&sem);
309 
310  // Init the stream queue
311  FirstSoundRequest = 0;
312  FreeSoundRequest = 0;
313 
314  // set local buffer & functions
315  CpuSuspendIntr(&oldStat);
316 
317  // SIFCMD
318  // No longer need to call sceSifSetCmdBuffer() since we share it with fileio.irx
319  sceSifAddCmdHandler(SOUND_REQUEST_COMMAND, (void *) sound_request_callback, NULL );
320 
321  CpuResumeIntr(oldStat);
322 
323  // The loop
324  while (1) {
325  //printf("waiting for sound command semaphore\n");
326  WaitSema(gSndSem); // Get signal from callback
327  //printf("got sound command semaphore\n");
328 
329  // Note that FreeSoundRequest can change in the interrupt at any time
330  // Also, FirstSoundRequest is examined in the interrupt, but just to make sure we don't overflow the buffer
331  //Dbg_Assert(FreeSoundRequest != FirstSoundRequest);
332  while (FreeSoundRequest != FirstSoundRequest)
333  {
334  int ret;
335  ESendSoundResult send_result;
336 
337  volatile SSoundRequest *p_request = &(SoundRequestArray[FirstSoundRequest]);
338  //Kprintf("Sound: %d executing command %x with entry %x and value %x\n", FirstSoundRequest, p_request->m_function, p_request->m_entry, p_request->m_value);
339  ret = execute_sound_request(p_request->m_function, p_request->m_entry, p_request->m_value,
340  p_request->m_arg_0, p_request->m_arg_1, p_request->m_arg_2, &send_result);
341 
342  if ((send_result != SEND_SOUND_RESULT_FALSE) || global_sound_panic)
343  {
344  //Kprintf("Sending sound result from command %x with entry %x and result %x\n", p_request->m_function, p_request->m_entry, ret);
345 
346  // Send the result back
347  if (last_cmd_id >= 0) // Wait for previous send to complete (it should already be done, though)
348  {
349  while(sceSifDmaStat(last_cmd_id) >= 0)
350  ;
351  }
352 
353  // Gotta copy the function and entry
354  SoundResult.m_function = p_request->m_function;
355  SoundResult.m_entry = p_request->m_entry;
356  SoundResult.m_result = ret;
357  SoundResult.m_panic = (send_result == SEND_SOUND_RESULT_PANIC) || global_sound_panic;
358  last_cmd_id = sceSifSendCmd(SOUND_RESULT_COMMAND, &SoundResult, sizeof(SoundResult), 0, 0, 0);
359  }
360 
361  // Increment request index; Note that interrupt can look at this
362  FirstSoundRequest = (FirstSoundRequest + 1) % NUM_SOUND_REQUESTS;
363  }
364 
365  // Send the status back, if it has been updated
366  if (s_send_spu_status)
367  {
368  int core;
369 
370  //Kprintf("Sending sound status back\n");
371 
372  for (core = 0; core < NUM_CORES; core ++)
373  {
374  if (last_cmd_id >= 0) // Wait for previous send to complete
375  {
376  while(sceSifDmaStat(last_cmd_id) >= 0)
377  ;
378  }
379 
380  // Gotta copy the function and entry
381  SoundResult.m_function = rSdGetSwitch;
382  SoundResult.m_entry = SD_S_ENDX | core;
383  SoundResult.m_result = s_spu_reg_endx[core];
384  SoundResult.m_panic = FALSE;
385  last_cmd_id = sceSifSendCmd(SOUND_RESULT_COMMAND, &SoundResult, sizeof(SoundResult), 0, 0, 0);
386  }
387  s_send_spu_status = FALSE;
388  }
389  }
390 
391  return 0;
392 }

394 void sound_request_callback(void *p, void *q)
395 {
396  SSifCmdSoundRequestPacket *h = (SSifCmdSoundRequestPacket *) p;
397 
398  // Check to make sure we can add
399  int nextFreeReq = (FreeSoundRequest + 1) % NUM_SOUND_REQUESTS;
400  if (nextFreeReq == FirstSoundRequest)
401  {
402  // We can't allow a request to be ignored. We must abort
403  Kprintf("Fuck, we've overrun the sound request buffer\n");
404  global_sound_panic = TRUE;
405  }
406 
407 #if PRINT_DROPPED_REQUESTS
408  if ((++s_received_requests) != h->m_request.m_pad[0])
409  {
410  Kprintf("ERROR: Missed sound request. Expected %d but only got up to %d\n", h->m_request.m_pad[0], s_received_requests);
411  s_received_requests = h->m_request.m_pad[0];
412  }
413 #endif PRINT_DROPPED_REQUESTS
414 
415  // Copy the request into the reuest queue
416  SoundRequestArray[FreeSoundRequest] = h->m_request;
417  FreeSoundRequest = nextFreeReq;
418 
419  //Kprintf("Sound: %d got command %x with entry %x and value %x\n", FreeSoundRequest, h->m_request.m_function, h->m_request.m_entry, h->m_request.m_value);
420 
421  // And wake up the dispatch thread
422  iSignalSema(gSndSem);
423 
424  return;
425 }
