120 void CNgcWeather::plat_update_grid( void )
121 {
122  m_system_active = false;
123  // Get super sector manager.
124  SSec::Manager *ss_man;
125  Mth::Line line;
126  ss_man = Nx::CEngine::sGetNearestSuperSectorManager( line ); // Line is ignored, 1st manager is returned.
127  if ( !ss_man ) return;
128 
129  // Calculate the size of the world in cels.
130  int min_x = ( ((int)ss_man->GetWorldBBox()->GetMin()[X]) / WEATHER_CEL_SIZE ) - 1;
131  int max_x = ( ((int)ss_man->GetWorldBBox()->GetMax()[X]) / WEATHER_CEL_SIZE ) + 1;
132  int min_z = ( ((int)ss_man->GetWorldBBox()->GetMin()[Z]) / WEATHER_CEL_SIZE ) - 1;
133  int max_z = ( ((int)ss_man->GetWorldBBox()->GetMax()[Z]) / WEATHER_CEL_SIZE ) + 1;
134 
135  // Define a maximum...
136  if ( ( max_x - min_x ) > 350 )
137  {
138  int wid = ( max_x - min_x );
139  int excess = ( wid - 350 );
140  min_x += ( excess / 2 );
141  max_x -= ( excess / 2 );
142  }
143 
144  if ( ( max_z - min_z ) > 300 )
145  {
146  int wid = ( max_z - min_z );
147  int excess = ( wid - 300 );
148  min_z += ( excess / 2 );
149  max_z -= ( excess / 2 );
150  }
151 
152  // This is the actual width;
153  m_width = ( max_x - min_x ) + 1;
154  m_height = ( max_z - min_z ) + 1;
155 
156  // Allocate a new piece of memory for the grid.
157 // if ( mp_roof_height_index ) delete mp_roof_height_index;
158 // mp_roof_height_index = new unsigned char[m_width * m_height];
159 
160 // mp_roof_height_index = grid_bytes;
161 
162  m_min_x = ( min_x * WEATHER_CEL_SIZE );
163  m_min_z = ( min_z * WEATHER_CEL_SIZE );
164 
165  // Temporary buffer for the raw array.
166  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().TopDownHeap());
167  unsigned char * p8 = new unsigned char[m_width*m_height];
168  Mem::Manager::sHandle().PopContext();
169 
170  // Go through and get the height for each cel corner.
171  CFeeler feeler;
172  int num_heights = 0;
173  int xx, zz;
174  for ( zz = min_z; zz <= max_z; zz++ )
175  {
176  for ( xx = min_x; xx <= max_x; xx++ )
177  {
178  // The cel to fill.
179  int cel = ( ( zz - min_z ) * m_width ) + ( xx - min_x );
180 
181  // The position to check.
182  Mth::Vector pos;
183  pos[X] = (float)( xx * WEATHER_CEL_SIZE );
184  pos[Y] = FEELER_START;
185  pos[Z] = (float)( zz * WEATHER_CEL_SIZE );
186  pos[W] = 1.0f;
187 
188  feeler.SetStart( pos );
189  pos[Y] = -FEELER_START;
190  feeler.SetEnd( pos );
191 
192  // Get the y position.
193  float y;
194  if ( feeler.GetCollision( false, false ) ) // No movables, nearest collision.
195  {
196  y = feeler.GetPoint()[Y];
197  }
198  else
199  {
200  y = FEELER_START;
201  }
202 
203  // See if a close enough y pos already exists.
204  int found = -1;
205  for ( int lp = 0; lp < num_heights; lp++ )
206  {
207  if ( fabsf( ( m_roof_height[lp] - y ) ) < HEIGHT_TOLERANCE )
208  {
209  found = lp;
210  break;
211  }
212  }
213 
214  // Fill in the cel.
215  if ( found != -1 )
216  {
217  // Existing height.
218  p8[cel] = found;
219  }
220  else
221  {
222  // New height.
223  p8[cel] = num_heights;
224  m_roof_height[num_heights] = y;
225  num_heights++;
226  }
227  }
228  }
229 
230  // Work out highest height for each cel.
231  for ( zz = min_z; zz <= max_z; zz++ )
232  {
233  for ( xx = min_x; xx <= max_x; xx++ )
234  {
235  // The cel to fill.
236  int cel = ( ( zz - min_z ) * m_width ) + ( xx - min_x );
237  int celx = ( ( zz - min_z ) * m_width ) + ( ( xx + 1 ) - min_x );
238  int celz = ( ( ( zz + 1 ) - min_z ) * m_width ) + ( xx - min_x );
239  int celxz = ( ( ( zz + 1 ) - min_z ) * m_width ) + ( ( xx + 1 ) - min_x );
240 
241  if ( m_roof_height[p8[celx]] > m_roof_height[p8[cel]] ) p8[cel] = p8[celx];
242  if ( m_roof_height[p8[celz]] > m_roof_height[p8[cel]] ) p8[cel] = p8[celz];
243  if ( m_roof_height[p8[celxz]] > m_roof_height[p8[cel]] ) p8[cel] = p8[celxz];
244  }
245  }
246 
247  // Create a sparse array.
248  mp_roof_row = (sRowEntry *)grid_bytes;
249  mp_roof_height_index = (unsigned char *)&mp_roof_row[m_height];
250 
251  // 0 = offset
252  // 1 = width
253  // 2 = index
254 
255  unsigned short index = 0;
256  for ( zz = 0; zz <= m_height; zz++ )
257  {
258  unsigned short start = 0;
259  unsigned short end = m_width - 1;
260 
261  // Scan to find the start.
262  bool start_set = false;
263  for ( xx = 0; xx < m_width; xx++ )
264  {
265  int cel = ( zz * m_width ) + xx;
266 
267  if ( m_roof_height[p8[cel]] != FEELER_START )
268  {
269  if ( !start_set )
270  {
271  // Set start value.
272  start = xx;
273  start_set = true;
274  }
275  else
276  {
277  // Set end value.
278  end = xx;
279  }
280 
281  }
282  }
283 
284  // Copy data & set row entry.
285  if ( start < end )
286  {
287  mp_roof_row[zz].start = start;
288  mp_roof_row[zz].end = end;
289  mp_roof_row[zz].index = index;
290  for ( xx = start; xx <= end ; xx++ )
291  {
292  int cel = ( zz * m_width ) + xx;
293 
294  mp_roof_height_index[index] = p8[cel];
295  index++;
296  }
297  }
298  else
299  {
300  // Row doesn't exist.
301  mp_roof_row[zz].start = 16384;
302  mp_roof_row[zz].end = 0;
303  mp_roof_row[zz].index = 0;
304  }
305  }
306 
307  delete p8;
308 
309 #ifndef __NOPT_FINAL__
310  int new_size = ( m_height * 6 ) + index;
311 
312  printf( "Grid Size: Old: %d New: %d Num Heights: %d\n", ( m_width * m_height ), new_size, num_heights );
313 #endif // __NOPT_FINAL__
314 // printf( "Num Weather Grid Heights: %d (%d x %d)\n", num_heights, m_width, m_height );
315 
316  // Set all drip time counters to 255.
317  // 255 means the drop won't be rendered.
318  // Setting to anything other than 255 will mean that it will increment to 255 and stop.
319  for ( int lp = 0; lp < ( DROP_SIZE * DROP_SIZE * DROP_LAYERS ); lp++ )
320  {
321  m_drop_time[lp] = 255;
322  m_x_offset[lp] = Mth::Rnd( 255 );
323  m_z_offset[lp] = Mth::Rnd( 255 );
324  }
325  m_active_drops = 0;
326 
327  m_seq = SEQ_START;
328 
329  Nx::CTexture *p_texture;
330  Nx::CNgcTexture *p_Ngc_texture;
331 
332  // Set Rain Texture.
333  p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( CRCD(0x45c7eb0f,"splash") );
334  p_Ngc_texture = static_cast<Nx::CNgcTexture*>( p_texture );
335  if ( p_Ngc_texture )
336  {
337  mp_rain_texture = p_Ngc_texture->GetEngineTexture();
338  }
339 
340  // Set Snow Texture.
341  p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( CRCD(0xc97c5a4c,"snow") );
342  p_Ngc_texture = static_cast<Nx::CNgcTexture*>( p_texture );
343  if ( p_Ngc_texture )
344  {
345  mp_snow_texture = p_Ngc_texture->GetEngineTexture();
346  }
347 
348  m_system_active = true;
349 
350  // Zero out the splashes.
351  for ( int sp = 0; sp < NUM_SPLASH_ACTIVE; sp++ )
352  {
353  m_splash_current_life[sp] = 0;
354  }
355  m_current_splash = 0;
356 }

363 void CNgcWeather::plat_process( float delta_time )
364 {
365  if ( !m_system_active ) return;
366 
367  if ( m_raining )
368  {
369  // It's raining.
370  float rate = m_rain_drops_per_frame;
371  if ( rate > (float)( ( DROP_SIZE * DROP_SIZE * DROP_LAYERS ) / m_rain_frames ) ) rate = (float)( ( DROP_SIZE * DROP_SIZE * DROP_LAYERS ) / m_rain_frames );
372 
373  float last = m_rain_rate;
374  m_rain_rate += rate;
375  int new_drops = (int)m_rain_rate - (int)last;
376 
377  for ( int lp = 0; lp < new_drops; lp++ )
378  {
379  // If this cel is not inactive, we caught up with ourselves.
380  if ( m_drop_time[m_seq] != 255 ) break;
381 
382  // Setup the drop.
383  m_drop_time[m_seq] = ( 254 - m_rain_frames );
384  m_x_offset[m_seq] = Mth::Rnd( 255 );
385  m_z_offset[m_seq] = Mth::Rnd( 255 );
386  m_active_drops++;
387 
388  // Calculate next sequence value. Notice hack to get entry 0 into the sequence.
389  switch ( m_seq )
390  {
391  case SEQ_START:
392  m_seq = 0;
393  break;
394  case 0:
395  m_seq = SEQ_START;
396  // Fall through to default case...
397  default:
398  if ( m_seq & 1 )
399  {
400  m_seq = ( m_seq >> 1 ) ^ SEQ_MASK;
401  }
402  else
403  {
404  m_seq = ( m_seq >> 1 );
405  }
406  break;
407  }
408  }
409  }
410  else
411  {
412  // It's snowing.
413  float rate = m_snow_flakes_per_frame;
414  if ( rate > (float)( ( DROP_SIZE * DROP_SIZE * DROP_LAYERS ) / m_snow_frames ) ) rate = (float)( ( DROP_SIZE * DROP_SIZE * DROP_LAYERS ) / m_snow_frames );
415 
416  float last = m_snow_rate;
417  m_snow_rate += rate;
418  int new_drops = (int)m_snow_rate - (int)last;
419 
420  for ( int lp = 0; lp < new_drops; lp++ )
421  {
422  // If this cel is not inactive, we caught up with ourselves.
423  if ( m_drop_time[m_seq] != 255 ) break;
424 
425  // Setup the drop.
426  m_drop_time[m_seq] = ( 254 - m_snow_frames );
427  m_x_offset[m_seq] = Mth::Rnd( 255 );
428  m_z_offset[m_seq] = Mth::Rnd( 255 );
429  m_active_drops++;
430 
431  // Calculate next sequence value. Notice hack to get entry 0 into the sequence.
432  switch ( m_seq )
433  {
434  case SEQ_START:
435  m_seq = 0;
436  break;
437  case 0:
438  m_seq = SEQ_START;
439  // Fall through to default case...
440  default:
441  if ( m_seq & 1 )
442  {
443  m_seq = ( m_seq >> 1 ) ^ SEQ_MASK;
444  }
445  else
446  {
447  m_seq = ( m_seq >> 1 );
448  }
449  break;
450  }
451  }
452  }
