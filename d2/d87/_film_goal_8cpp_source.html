41 bool CFilmGoal::Activate()
42 {
43  m_timeOnCamera = 0;
44  m_timeRequired = m_totalTime = 0;
45  m_numShotsRequired = 0;
46  m_filming = false;
47 
48  // script param for displaying time on camera...
49  mp_params->AddInteger( CRCD( 0x9278ca8d, "time_on_camera" ), 0 );
50 
51  int time_required;
52  if ( mp_params->GetInteger( CRCD(0x184e30fd,"total_time_required"), &time_required, Script::NO_ASSERT ) )
53  {
54  m_timeRequired = (Tmr::Time)( time_required * 1000 );
55  int total_time;
56  mp_params->GetInteger( CRCD(0x906b67ba,"time"), &total_time, Script::ASSERT );
57  m_totalTime *= (Tmr::Time)( total_time * 1000 );
58  }
59  else if ( !mp_params->GetInteger( CRCD(0xc808f9fb,"total_shots_required"), &m_numShotsRequired, Script::NO_ASSERT ) )
60  {
61  Dbg_MsgAssert( 0, ( "Film goal %s requires either total_time_required or total_shots_required", Script::FindChecksumName( GetGoalId() ) ) );
62  }
63  return CGoal::Activate();
64 }

71 bool CFilmGoal::Update()
72 {
73  mp_params->AddInteger( CRCD(0xfc7fc309,"last_time_on_camera"), (int)m_timeOnCamera );
74 
75  if ( IsActive() && !IsPaused() )
76  {
77  // check if they're still looking at the target object
78  bool should_win = false;
79  if ( m_filming )
80  {
81  if ( target_object_visible() )
82  {
83  m_timeOnCamera += (Tmr::Time)( Tmr::FrameLength() * 1000 );
84 
85  // script param for displaying time on camera...
86  mp_params->AddInteger( CRCD( 0x9278ca8d, "time_on_camera" ), (int)m_timeOnCamera );
87  }
88  // printf( "on camera: %i, required: %i\n", (int)m_timeOnCamera, (int)m_timeRequired );
89 
90  if ( m_timeOnCamera >= m_timeRequired )
91  {
92  should_win = true;
93  }
94  else if ( m_timeRequired - m_timeOnCamera > m_timeLeft )
95  {
96  // can't possibly win!
97  CGoalManager* pGoalManager = GetGoalManager();
98  Dbg_Assert( pGoalManager );
99  return pGoalManager->LoseGoal( GetGoalId() );
100  }
101  }
102  else if ( m_numShotsRequired && m_numShotsAchieved >= m_numShotsRequired )
103  {
104  should_win = true;
105  }
106 
107  if ( should_win )
108  {
109  CGoalManager* pGoalManager = GetGoalManager();
110  Dbg_Assert( pGoalManager );
111  return pGoalManager->WinGoal( GetGoalId() );
112  }
113  else
114  return CGoal::Update();
115  }
116  else
117  return CGoal::Update();
118 }

153 // does a visibility check as well as a distance check - you can't just
154 // look at him from a mile away
155 bool CFilmGoal::target_object_visible()
156 {
157  uint32 target_obj_id;
158  mp_params->GetChecksum( CRCD( 0xd262ed09, "film_target" ), &target_obj_id, Script::ASSERT );
159  Obj::CCompositeObject* pTargetObj = (Obj::CCompositeObject*)Obj::ResolveToObject( target_obj_id );
160  Dbg_Assert( pTargetObj );
161 
162  bool visible = false;
163  if ( pTargetObj )
164  {
165  Obj::CModelComponent* pTargetObjModelComponent = GetModelComponentFromObject( pTargetObj );
166  Mth::Vector sphere = pTargetObjModelComponent->GetModel()->GetBoundingSphere();
167  Mth::Vector position = sphere;
168  position[3] = 1.0f;
169  position *= pTargetObj->GetMatrix();
170  position += pTargetObj->GetPos();
171  //if ( ( Nx::CEngine::sIsVisible( position, sphere[3] ) ) )
172  if ( ( Nx::CEngine::sIsVisible( position, 6.0f ) ) ) // TT#2555 (Mick), use a 6 inch radius sphere te ensure we can really see him
173  {
174  // get the cam
175  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
176  Obj::CSkater* pSkater = skate_mod->GetLocalSkater();
177  Dbg_Assert( pSkater );
178  Gfx::Camera* pCurrentCamera = pSkater->GetActiveCamera();
179 
180  float distance_to_target = Mth::Distance( pCurrentCamera->GetPos(), pTargetObj->GetPos() );
181  int max_distance;
182  mp_params->GetInteger( CRCD(0x856fab92,"max_distance_to_target"), &max_distance, Script::ASSERT );
183  if ( distance_to_target < (float)max_distance )
184  {
185  visible = true;
186  }
187  }
188  }
189  return visible;
190 }
