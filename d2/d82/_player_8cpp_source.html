103 int Manager::GetTeamScore( int team_id )
104 {
105  PlayerInfo* player, *local_player;
106  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
107  Lst::Search< PlayerInfo > sh;
108  Mdl::Score* score_obj;
109  int score;
110  bool in_goal_attack;
111 
112  score = 0;
113 
114  local_player = GetLocalPlayer();
115  in_goal_attack = skate_mod->GetGameMode()->GetNameChecksum() == Script::GenerateCRC( "netgoalattack" );
116  // Next create a sorted list of the players (sorted in order of highest to lowest score)
117  for( player = FirstPlayerInfo( sh ); player; player = NextPlayerInfo( sh ))
118  {
119  // Only show the names/scores of players that are fully in the game
120  if((player->IsFullyIn() == false ) || ( player->m_Team != team_id ))
121  {
122  continue;
123  }
124 
125  if( in_goal_attack && !local_player->IsObserving())
126  {
127  Game::CGoalManager* pGoalManager;
128 
129  pGoalManager = Game::GetGoalManager();
130  return pGoalManager->NumGoalsBeatenByTeam( team_id );
131  }
132  else
133  {
134  Obj::CSkaterScoreComponent* p_skater_score_component = GetSkaterScoreComponentFromObject(player->m_Skater);
135  Dbg_Assert(p_skater_score_component);
136 
137  score_obj = p_skater_score_component->GetScore();
138  score += score_obj->GetTotalScore();
139  }
140  }
141 
142  return score;
143 }

176 void Manager::s_render_scores_code( const Tsk::Task< Manager >& task )
177 {
178  Front::CScreenElementManager* p_screen_elem_man = Front::CScreenElementManager::Instance();
179  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
180  Script::CScriptStructure *pParams;
181  Game::CGameMode* pGameMode;
182  PlayerInfo* player;
183  Lst::Search< PlayerInfo > sh;
184  static Tmr::Time last_update = 0;
185  Tmr::Time cur_time;
186  Lst::Head< ScoreRank > rank_list;
187  Lst::Search< ScoreRank > rank_sh;
188  ScoreRank* rank, *next;
189  bool show_score, king_mode;
190 
191  Manager& man = task.GetData();
192  // Only draw the scores in modes in which the score accumulates
193  pGameMode = skate_mod->GetGameMode();
194  show_score = skate_mod->GetGameMode()->GetNameChecksum() != CRCD( 0x1c471c60, "netlobby" );
195 
196  // in King of the hill games, "score" is actually milliseconds with the crown
197  king_mode = ( skate_mod->GetGameMode()->GetNameChecksum() == CRCD( 0x6ef8fda0, "netking" )) ||
198  ( skate_mod->GetGameMode()->GetNameChecksum() == CRCD( 0x5d32129c, "king" ));
199 
200  cur_time = Tmr::GetTime();
201  if(( cur_time - last_update ) >= Tmr::Seconds( 1 ))
202  {
203  int i, j;
204  uint32 id;
205 
206  last_update = cur_time;
207 
208  // First clear all previous entries
209  Script::RunScript( "clear_scores" );
210 
211  if( show_score && man.ShouldDisplayTeamScores())
212  {
213  ScoreRank* ranks[vMAX_TEAMS];
214 
215  for( j = 0; j < skate_mod->GetGameMode()->NumTeams(); j++ )
216  {
217  if( man.NumTeamMembers( j ) > 0 )
218  {
219  uint32 team_name_checksum;
220 
221  switch( j )
222  {
223  case 0:
224  team_name_checksum = CRCD( 0x3224348d, "team_1_name" );
225  break;
226  case 1:
227  team_name_checksum = CRCD( 0xb4b04623, "team_2_name" );
228  break;
229  case 2:
230  team_name_checksum = CRCD( 0x7fec9586, "team_3_name" );
231  break;
232  case 3:
233  team_name_checksum = CRCD( 0x62e9a53e, "team_4_name" );
234  break;
235  default:
236  Dbg_Assert( 0 );
237  team_name_checksum = 0;
238  break;
239  }
240 
241  ranks[j] = new ScoreRank;
242  strcpy( ranks[j]->m_Name, Script::GetString( team_name_checksum ));
243  ranks[j]->m_IsKing = false;
244  ranks[j]->m_HasFlag = false;
245  ranks[j]->m_WhichFlags = 0;
246  ranks[j]->m_ColorIndex = j + 2;
247  ranks[j]->m_TotalScore = man.GetTeamScore( j );
248  }
249  }
250 
251  // Next create a sorted list of the players (sorted in order of highest to lowest score)
252  for( player = man.FirstPlayerInfo( sh ); player; player = man.NextPlayerInfo( sh ))
253  {
254  // Only show the names/scores of players that are fully in the game
255  if( !( player->IsFullyIn()))
256  {
257  continue;
258  }
259  if( player->IsLocalPlayer())
260  {
261  char new_name[64];
262 
263  // This is my team. Make that obvious to the players
264  sprintf( new_name, "> %s", ranks[player->m_Team]->m_Name );
265  strcpy( ranks[player->m_Team]->m_Name, new_name );
266  }
267 
268  if( player->IsKing())
269  {
270  ranks[player->m_Team]->m_IsKing = true;
271  }
272  if( player->HasCTFFlag())
273  {
274  ranks[player->m_Team]->m_HasFlag = true;
275  ranks[player->m_Team]->m_WhichFlags |= ( 1 << player->HasWhichFlag());
276  }
277  }
278 
279  for( j = 0; j < skate_mod->GetGameMode()->NumTeams(); j++ )
280  {
281  if( man.NumTeamMembers( j ) > 0 )
282  {
283  // Lists sort based on node's priority so set the node's priority to that of
284  // the team's score and add it to the list
285  ranks[j]->SetPri( ranks[j]->m_TotalScore );
286  rank_list.AddNode( ranks[j] );
287  }
288  }
289  }
290  else
291  {
292  // Next create a sorted list of the players (sorted in order of highest to lowest score)
293  for( player = man.FirstPlayerInfo( sh ); player; player = man.NextPlayerInfo( sh ))
294  {
295  // Only show the names/scores of players that are fully in the game
296  if( !( player->IsFullyIn()))
297  {
298  continue;
299  }
300 
301  rank = new ScoreRank;
302  if( player->IsLocalPlayer())
303  {
304  // This is me. Make that obvious.
305  sprintf( rank->m_Name, "> %s", player->m_Name );
306  }
307  else
308  {
309  strcpy( rank->m_Name, player->m_Name );
310  }
311 
312  rank->m_IsKing = player->IsKing();
313  if( skate_mod->GetGameMode()->IsTeamGame())
314  {
315  rank->m_ColorIndex = player->m_Team + 2;
316  }
317  else
318  {
319  rank->m_ColorIndex = player->m_Skater->GetID() + 2;
320  }
321  if( player->HasCTFFlag())
322  {
323  rank->m_HasFlag = true;
324  rank->m_WhichFlags |= ( 1 << player->HasWhichFlag());
325  }
326 
327  if( show_score )
328  {
329  // Lists sort based on node's priority so set the node's priority to that of
330  // the player's score and add him to the list
331  rank->SetPri( man.GetPlayerScore( player->m_Skater->GetID()));
332  rank_list.AddNode( rank );
333  }
334  else
335  {
336  // Just add the player to the list so that the order is the same regardless of score
337  rank_list.AddToTail( rank );
338  }
339  }
340  }
341 
342 
343  // Now loop through the sorted list in order of highest to lowest score and print them out
344  i = 0;
345  for( rank = rank_sh.FirstItem( rank_list ); rank; rank = next )
346  {
347  char score_str[64];
348  char title[16];
349 
350  title[0]='\0';
351  if( rank->m_IsKing )
352  {
353 #if ENGLISH == 0
354  sprintf( title, Script::GetLocalString( "player_str_king_abbreviation" ));
355 #else
356  sprintf( title, "K" );
357 #endif
358  }
359  else
360  {
361  if( rank->m_HasFlag )
362  {
363  if( rank->m_WhichFlags & 1 )
364  {
365  strcat( title, "\\s0" );
366  }
367  if( rank->m_WhichFlags & 2 )
368  {
369  strcat( title, "\\s1" );
370  }
371  if( rank->m_WhichFlags & 4 )
372  {
373  strcat( title, "\\s2" );
374  }
375  if( rank->m_WhichFlags & 8 )
376  {
377  strcat( title, "\\s3" );
378  }
379  }
380  else
381  {
382  title[0] = '\0';
383  }
384  }
385 
386  next = rank_sh.NextItem();
387  if( show_score )
388  {
389  if( king_mode )
390  {
391  int seconds;
392 
393  seconds = Tmr::InSeconds( rank->GetPri() );
394  if( seconds >= 3600 )
395  {
396  sprintf( score_str, "\\c4%s\\c%d %s \\c%d %d:%.2d:%.2d",
397  title,
398  rank->m_ColorIndex,
399  rank->m_Name,
400  rank->m_ColorIndex,
401  seconds / 3600,
402  ( seconds % 3600 ) / 60,
403  seconds % 60 );
404  }
405  else
406  {
407  sprintf( score_str, "\\c4%s\\c%d %s \\c%d %d:%.2d",
408  title,
409  rank->m_ColorIndex,
410  rank->m_Name,
411  rank->m_ColorIndex,
412  seconds / 60,
413  seconds % 60 );
414  }
415 
416  }
417  else
418  {
419  sprintf( score_str, "%s\\c%d %s \\c%d %d", title,
420  rank->m_ColorIndex,
421  rank->m_Name,
422  rank->m_ColorIndex,
423  rank->GetPri());
424  }
425 
426  }
427  else
428  {
429  sprintf( score_str, "\\c%d%s", rank->m_ColorIndex, rank->m_Name );
430  }
431 
432  switch( i )
433  {
434  case 0:
435  id = CRCD( 0x2b083809, "net_score_1" );
436  break;
437  case 1:
438  id = CRCD( 0xb20169b3, "net_score_2" );
439  break;
440  case 2:
441  id = CRCD( 0xc5065925, "net_score_3" );
442  break;
443  case 3:
444  id = CRCD( 0x5b62cc86, "net_score_4" );
445  break;
446  case 4:
447  id = CRCD( 0x2c65fc10, "net_score_5" );
448  break;
449  case 5:
450  id = CRCD( 0xb56cadaa, "net_score_6" );
451  break;
452  case 6:
453  id = CRCD( 0xc26b9d3c, "net_score_7" );
454  break;
455  case 7:
456  id = CRCD( 0x52d480ad, "net_score_8" );
457  break;
458  default:
459  Dbg_Assert( 0 );
460  id = 0;
461  break;
462  }
463 
464  pParams = new Script::CScriptStructure;
465  pParams->AddComponent( CRCD( 0xc4745838, "text" ), ESYMBOLTYPE_STRING, score_str );
466  pParams->AddComponent( CRCD( 0x40c698af, "id" ), ESYMBOLTYPE_NAME, id );
467 
468 
469  Front::CScreenElement* p_name_elem = p_screen_elem_man->GetElement( id, Front::CScreenElementManager::DONT_ASSERT );
470  // If any operable menus are up, ignore input
471  if( p_name_elem )
472  {
473  //make sure the scores are enabled
474  //if(!Mdl::Skate::Instance()->GetCareer()->GetGlobalFlag(Script::GetInteger(CRCD(0xfbb66146,"NO_DISPLAY_NET_SCORES")))))
475  p_name_elem->SetProperties( pParams );
476  }
477  Script::RunScript( "update_score", pParams );
478 
479  delete pParams;
480  delete rank;
481  i++;
482  }
483  }
484 }

713 bool PlayerInfo::IsNonCollidable( void )
714 {
715  Mdl::Skate * skate_mod = Mdl::Skate::Instance();
716 
717  if( !m_Conn->TestStatus( Net::Conn::mSTATUS_READY ))
718  {
719  return true;
720  }
721 
722  if( m_Skater->IsInWorld() == false )
723  {
724  return true;
725  }
726 
727  // Make skaters invulnerable for the first 10 frames of the game
728  if( m_Skater->GetStateHistory()->GetNumPosUpdates() < vINVULNERABILITY_PERIOD )
729  {
730  return true;
731  }
732 
733  if(( Tmr::GetTime() - m_last_bail_time ) < Tmr::Seconds( vINVULNERABILITY_BAIL_SECONDS ))
734  {
735  return true;
736  }
737 
738  // Don't collide with eliminated players in firefight
739  if( skate_mod->GetGameMode()->GetNameChecksum() == CRCD(0xbff33600,"netfirefight"))
740  {
741  Mdl::Score* score_obj;
742  Obj::CSkaterScoreComponent* p_score_component = GetSkaterScoreComponentFromObject(m_Skater);
743  Dbg_Assert(p_score_component);
744 
745  score_obj = p_score_component->GetScore();
746  if( score_obj->GetTotalScore() <= 0 )
747  {
748  return true;
749  }
750  }
751 
752  // If they've been in this state for more than 5 seconds, that means that
753  // for one reason or another, we didn't get their "NotifyBailDone" message.
754  // This will rectify things.
755  if(( Tmr::GetTime() - m_last_bail_time ) > Tmr::Seconds( 5 ))
756  {
757  ClearNonCollidable();
758  return false;
759  }
760 
761  if( m_flags.TestMask( mNON_COLLIDABLE ))
762  {
763  return true;
764  }
765 
766  return false;
767 }
