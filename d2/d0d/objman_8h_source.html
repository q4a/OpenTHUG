55 /*
56  This class manages usage of the bits of a uint32. Use of CBitManager insures that no two sections of the greater code are using the same bit. When
57  a bit is requested from a CBitManager, no other requests for bits will give that bit until access to the bit has been returned to the manager.
58  Bit requests and returns are done in the form of uint32 masks.
59 */

61 {
62 public:
63  CBitManager ( );
64 
65  uint32 RequestBit ( );
66  void ReturnBit ( uint32 mask );
67 
68 private:
69  uint32 m_used_bit_mask;
70 };

212 inline uint32 CBitManager::RequestBit ( )
213 {
214  Dbg_MsgAssert(m_used_bit_mask != 0xFFFFFFFF, ("Out of available bits in CBitManager::RequestBit"));
215 
216  uint32 ready_mask = nBit(0);
217  while (m_used_bit_mask & ready_mask)
218  {
219  ready_mask <<= 1;
220  }
221 
222  m_used_bit_mask |= ready_mask;
223 
224  return ready_mask;
225 }
226 
227 /******************************************************************/
228 /* */
229 /* */
230 /******************************************************************/
231 
232 inline void CBitManager::ReturnBit ( uint32 returning_mask )
233 {
234  Dbg_MsgAssert((returning_mask & (returning_mask - 1)) == 0, ("Calling CBitManager::ReturnBit with a mask contining more than a single set bit"));
235  Dbg_MsgAssert(returning_mask & m_used_bit_mask, ("Returning an unused bit to CBitManager::ReturnBit"));
236 
237  m_used_bit_mask &= ~returning_mask;
238 }
