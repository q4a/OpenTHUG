56 void CreateNodeNameHashTable()
57 {
58  for ( int lp = 0; lp < HASH_SIZE; lp++ )
59  {
60  node_hash[lp] = -1;
61  }
62  // Get the NodeArray
63  CArray *p_node_array=GetArray(0xc472ecc5/*NodeArray*/);
64  Dbg_MsgAssert(p_node_array,("CreateNodeNameHashTable could not find NodeArray"));
65 
66  // Scan through each node, getting it's name, and if it has a name add it in to the hash table.
67  for (uint32 i=0; i<p_node_array->GetSize(); ++i)
68  {
69  CStruct *p_node=p_node_array->GetStructure(i);
70 
71  uint32 name_checksum=0;
72  if (p_node->GetChecksum("Name",&name_checksum))
73  {
74  uint16 hash = (uint16)(name_checksum & HASH_MASK );
75 
76  while ( node_hash[hash] != -1 )
77  {
78  hash++;
79  hash &= HASH_MASK;
80  }
81  node_hash[hash] = i;
82  }

231 // Given that a new NodeArray symbol has got loaded due to the loading of a qb containing it,
232 // this will generate the node name hash table.
233 void CreateNodeNameHashTable()
234 {
235  Dbg_MsgAssert(s_node_name_hash_table_initialised,("Node name hash table not initialised"));
236 
237  // Make sure the old one is cleared.
238  ClearNodeNameHashTable();
239 
240  // Get the NodeArray
241  CArray *p_node_array=GetArray(0xc472ecc5/*NodeArray*/);
242  Dbg_MsgAssert(p_node_array,("CreateNodeNameHashTable could not find NodeArray"));
243 
244  // Make sure we're using the script heap, cos the hash table is going to hang around
245  // for the duration of the level.
246  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
247 
248  // Scan through each node, getting it's name, and if it has a name add it in to the hash table.
249  for (uint32 i=0; i<p_node_array->GetSize(); ++i)
250  {
251  CStruct *p_node=p_node_array->GetStructure(i);
252  Dbg_MsgAssert(p_node,("NULL p_node"));
253 
254  uint32 name_checksum=0;
255  if (p_node->GetChecksum("Name",&name_checksum))
256  {
257  // Node i has name name_checksum, so add it in to the hash table.
258 
259  // A checksum of zero is used to indicate an empty hash table entry,
260  // so check that the name checksum is not zero by chance.
261  Dbg_MsgAssert(name_checksum,("Node has zero name checksum ?"));
262 
263  SNodeNameHashEntry *p_entry=&sp_node_name_hash_table[name_checksum&((1<<NODE_NAME_HASHBITS)-1)];
264  // p_entry is now the entry in the hash table array indexed by the lower bits of the checksum.
265 
266  #if __OLD_HASH_TABLE__
267  if (p_entry->mNameChecksum)
268  {
269  // p_entry is already occupied, so skip to the end of the list that starts at p_entry.
270  while (p_entry->mpNext)
271  {
272  p_entry=p_entry->mpNext;
273  }
274 
275  // p_entry is now the last entry in the list.
276 
277  // Create a new entry and fill it in.
278  SNodeNameHashEntry *p_new=(SNodeNameHashEntry*)Mem::Malloc(sizeof(SNodeNameHashEntry));
279  p_new->mNameChecksum=name_checksum;
280  p_new->mNodeIndex=i;
281  p_new->mpNext=NULL;
282 
283  // Tag it onto the end of the list.
284  p_entry->mpNext=p_new;
285 
286 
287  }
288  else
289  {
290  // p_entry is free, so stick the info in there.
291  p_entry->mNameChecksum=name_checksum;
292  p_entry->mNodeIndex=i;
293  // Quick check for leaks.
294  Dbg_MsgAssert(p_entry->mpNext==NULL,("p_entry->mpNext not NULL ??"));
295  p_entry->mpNext=NULL;
296  }
297  #else
298  // just skip forward until we find an empty entry
299  while (p_entry->mNameChecksum)
300  {
301  p_entry++;
302  // note in the following assertion we leave one entry to guarentee we have
303  // a zero entry roadblock to stop searches
304  // otherwise we have a slight possibility of very obscure bugs
305  Dbg_MsgAssert(p_entry < &sp_node_name_hash_table[NODE_NAME_HASH_SIZE-1],("Hash table overflow"));
306  }
307  p_entry->mNameChecksum=name_checksum;
308  p_entry->mNodeIndex=i;
309  #endif
310 
311  }
312  }
313  Mem::Manager::sHandle().PopContext();
314 }

402 // Linked list stuff, used when generating the node lists.

404 struct STempNode
405 {
406  int mNodeIndex;
407  STempNode *mpNext;

410 // MEMOPT 800K TEMP Size of temporary buffer used when generating prefix info.
411 #define NUM_TEMP_NODES 120000
412 static STempNode *sp_temp_nodes=NULL;
413 static STempNode *sp_free_temp_nodes=NULL;
414 
415 // The lookup table, which exists in memory for the duration of the level.
416 struct SPrefixLookup
417 {
418  // Checksum of the prefix string
419  uint32 mChecksum;
420 
421  // Pointer to the list of node indices of the nodes that have this prefix.
422  union
423  {
424  // Points to somewhere inside sp_node_list_buffer
425  // pNodeList[0] contains the number of nodes.
426  uint16 *mpNodeList;
427 
428  // Points to a temporary linked list of nodes, which is used for speed when generating the prefix info.
429  // (Uses up too much memory to keep in memory all the time)
430  STempNode *mpTempNodesHeadPointer;
431  };
432 };
433 
434 // MEMOPT 80K PERM An array of SPrefixLookup's, one for each possible prefix.
435 // This array is kept in order of checksum, smallest checksum first, to enable quick searching
436 // using a binary search.
437 #define MAX_PREFIX_LOOKUPS 13000
438 static SPrefixLookup sp_prefix_lookups[MAX_PREFIX_LOOKUPS];
439 static uint32 s_num_prefix_lookups=0;
440 
441 // MEMOPT 200K PERM Big array of uint16's for holding the node lists.
442 #define NODE_LIST_BUFFER_SIZE 121000
443 static uint32 s_node_list_buffer_used=0;
444 static uint16 sp_node_list_buffer[NODE_LIST_BUFFER_SIZE];
445 
446 // Does a binary search of the lookup table, and returns the index of the entry
447 // that has the passed checksum.
448 // If there is no matching entry, it will return the index of the entry with the next smallest
449 // checksum.
450 // If there is no matching entry, and the passed checksum is smaller than the smallest checksum in the
451 // table, then it will return 0.
452 static uint32 sFindPrefixLookup(uint32 checksum)
453 {
454  Dbg_MsgAssert(s_num_prefix_lookups,("Zero s_num_prefix_lookups"));
455  uint32 bottom=0;
456  uint32 top=s_num_prefix_lookups-1;
457  uint32 middle=(bottom+top)>>1;
458 
459  while (bottom!=middle)
460  {
461  uint32 ch=sp_prefix_lookups[middle].mChecksum;
462 
463  if (ch==checksum)
464  {
465  return middle;
466  }
467  else if (checksum<ch)
468  {
469  top=middle;
470  }
471  else
472  {
473  bottom=middle;
474  }
475  middle=(bottom+top)>>1;
476  }
477  if (sp_prefix_lookups[top].mChecksum > checksum)
478  {
479  return bottom;
480  }
481  return top;
482 }

484 // Creates the big temporary array of nodes.
485 static void sCreateTempNodes()
486 {
487  Dbg_MsgAssert(sp_temp_nodes==NULL,("sp_temp_nodes not NULL ?"));
488  sp_temp_nodes=(STempNode*)Mem::Malloc(NUM_TEMP_NODES*sizeof(STempNode));
489  Dbg_MsgAssert(sp_temp_nodes,("Could not allocate sp_temp_nodes"));

495  }

498  sp_free_temp_nodes=sp_temp_nodes;

512 // Grabs a new node out of the array.
513 static STempNode *sNewTempNode()

521 // Converts the linked lists of nodes into simple arrays, kept in the sp_node_list_buffer static array.
522 // Uses up less memory. Also means the rest of the game code that calls GetPrefixInfo does not have
523 // to be changed.
524 static void sConvertTempNodes()
525 {
526  Dbg_MsgAssert(s_node_list_buffer_used==0,("Expected s_node_list_buffer_used to be 0"));
527  uint16 *p_buf=sp_node_list_buffer;
528 
529  // For each prefix in the lookup table ...
530  for (uint32 i=0; i<s_num_prefix_lookups; ++i)
531  {
532  // Get the linked list.
533  STempNode *p_first=sp_prefix_lookups[i].mpTempNodesHeadPointer;
534  Dbg_MsgAssert(p_first,("NULL p_first"));
535 
536  // Count how many nodes are in it.
537  int count=0;
538  STempNode *p_node=p_first;
539  while (p_node)
540  {
541  ++count;
542  p_node=p_node->mpNext;
543  }
544 
545  // Check there is enough space to copy the node list into.
546  Dbg_MsgAssert(s_node_list_buffer_used+1+count<=NODE_LIST_BUFFER_SIZE,("Node list buffer overflow"));
547 
548  // Make the prefix entry now point to the simple array.
549  sp_prefix_lookups[i].mpNodeList=p_buf;
550 
551  // First entry is the count.
552  *p_buf++=count;
553  // Copy in all the node indices.
554  p_node=p_first;
555  while (p_node)
556  {
557  *p_buf++=p_node->mNodeIndex;
558  p_node=p_node->mpNext;
559  }
560 
561  // Update the space used. The +1 is for the count.
562  s_node_list_buffer_used+=1+count;
563  }
564 // Dbg_MsgAssert(0,("buffer = %d\n",s_node_list_buffer_used));
565 }

567 // Adds a new node index to a particular prefix checksum's entry in the lookup table.
568 static void sAddNewPrefix(uint32 checksum, int nodeIndex)
569 {
570  // sFindPrefixLookup will assert if s_num_prefix_lookups is zero, so do this
571  // as a special case for the very first one added.
572  if (s_num_prefix_lookups==0)
573  {
574  STempNode *p_new=sNewTempNode();
575  p_new->mNodeIndex=nodeIndex;
576  p_new->mpNext=NULL;
577 
578  // Stick it in at index 0. The array will be maintained as a sorted
579  // array from now on.
580  sp_prefix_lookups[0].mChecksum=checksum;
581  sp_prefix_lookups[0].mpTempNodesHeadPointer=p_new;
582 
583  ++s_num_prefix_lookups;
584  // All done.
585  return;
586  }
587 
588  // Look up the checksum, which may not be there.
589  int index=sFindPrefixLookup(checksum);
590  uint32 ch=sp_prefix_lookups[index].mChecksum;
591  // ch may or may not be checksum. If it isn't, it will either be the next smallest one in the
592  // array, or it will be the checksum at index 0 in the case of checksum being smaller than all of them.
593 
594  if (ch==checksum)
595  {
596  // This prefix is already in the array, so just have to add the new node index to
597  // its list.
598 
599  STempNode *p_new=sNewTempNode();
600  p_new->mNodeIndex=nodeIndex;
601  p_new->mpNext=sp_prefix_lookups[index].mpTempNodesHeadPointer;
602 
603  sp_prefix_lookups[index].mpTempNodesHeadPointer=p_new;
604  }
605  else
606  {
607  // The prefix is not in the table, so we have to add it.
608 
609  Dbg_MsgAssert(s_num_prefix_lookups<MAX_PREFIX_LOOKUPS,("Too many prefixes"));
610 
611  if (checksum>ch)
612  {
613  // The above if will mostly be true, because usually Ch will be the next smaller checksum
614  // in the table. Hence we increment index so that all the checksums above ch get shifted up,
615  // so that the new checksum can be inserted at index, maintaining the sort order.
616  ++index;
617  }
618  else
619  {
620  // The above if will be false in the case of checksum being smaller than all the current checksums
621  // in the table. In this case, all the checksums need to be shifted up and checksum inserted at the
622  // bottom, so index should be zero here.
623  Dbg_MsgAssert(index==0,("index not zero ?"));
624  }
625 
626  // Shift everything up one so that the new checksum can be inserted.
627  for (int i=s_num_prefix_lookups; i>index; --i)
628  {
629  sp_prefix_lookups[i]=sp_prefix_lookups[i-1];
630  }
631 
632  // Insert the new checksum, and make a new list for it with just one entry at the moment.
633  STempNode *p_new=sNewTempNode();
634  p_new->mNodeIndex=nodeIndex;
635  p_new->mpNext=NULL;
636 
637  sp_prefix_lookups[index].mChecksum=checksum;
638  sp_prefix_lookups[index].mpTempNodesHeadPointer=p_new;
639 
640  // Increment the count.
641  ++s_num_prefix_lookups;
642  }
643 }

654 // This is called from LoadQB if the QB contains a NodeArray.
655 void GeneratePrefixInfo()
656 {
657  DeletePrefixInfo();
658 
659  // Create the temporary array of nodes used to make linked lists in sAddNewPrefix, for speed.
660  sCreateTempNodes();
661 
662  // Scan through all the nodes, look up their names, and add all their possible
663  // prefixes to the lookup table.
664  CArray *p_node_array=GetArray(0xc472ecc5/*NodeArray*/);
665  Dbg_MsgAssert(p_node_array,("NodeArray not found"));
666  for (uint32 i=0; i<p_node_array->GetSize(); ++i)
667  {

690  sConvertTempNodes();
691  // Remove the temporary stuff.
692  sDeleteTempNodes();

696  if (s_num_prefix_lookups)
697  {
698  for (uint32 i=0; i<s_num_prefix_lookups-1; ++i)
699  {
700  Dbg_MsgAssert(sp_prefix_lookups[i].mChecksum<sp_prefix_lookups[i+1].mChecksum,("Out of order"));

707 //
708 // GetPrefixedNodes function.
709 // This will return an array of node indices, which are all the nodes
710 // whose names are prefixed with the passed string.
711 // For example, if passed "a", it will return all nodes whose names
712 // begin with 'a'.
713 // If passed "TRG_" it will return all nodes beginning with TRG_, etc.
714 // It is not case sensitive.
715 //
716 //
717 // It returns an array of uint16's, which are the node numbers of
718 // the matching nodes.
719 // It loads the size of the array into the passed p_numMatches.
720 //
721 // Note: It will never return NULL, but p_numMatches may contain zero.
722 //
724 const uint16 *GetPrefixedNodes(uint32 checksum, uint16 *p_numMatches)
725 {
726  static uint16 s_dummy=0;
727 
728  if (s_num_prefix_lookups)
729  {
730  // Look it up using a binary search.
731  uint32 index=sFindPrefixLookup(checksum);
732  if (sp_prefix_lookups[index].mChecksum==checksum)
733  {
734  // It matches, so return the pre-calculated node list for this prefix.
735  uint16 *p_node_list=sp_prefix_lookups[index].mpNodeList;
736  *p_numMatches=*p_node_list++;
737 
738  return p_node_list;
739  }
740  }
741 
742  // No matches, so return a node count of zero.
743  *p_numMatches=0;
744  return &s_dummy;
745 }

749 {
750  uint32 checksum=Crc::GenerateCRCFromString(p_prefix);
751  return GetPrefixedNodes(checksum,p_numMatches);

1044 //
1045 // Given a script, and the checksum of a function to look for,
1046 // this will search through the script for any calls to that function.
1047 // It will recursively search through any script calls.
1048 // The callback function is called for each occurrence of the function
1049 // call found, and the parameter list for that call is also passed.
1050 //
1051 // NOTE: Don't store the passed CStruct pointer!
1052 // It's a local variable in FindReferences.
1053 // To store the parameters create a new CStruct and use the
1054 // AppendStructure member function to copy in the contents of the
1055 // passed one.
1056 //
1058 
1059 // Note: p_args contains the set of parameters that were passed to scriptToScanThrough. It is passed so that
1060 // the <,> syntax can be evaluated. This is necessary because sometimes the EndGap command is not directly
1061 // passed the text and score, they might get passed via the <...> syntax, for example see the
1062 // Gap_Gen_End script in sk4_scripts.q
1063 static void FindReferences(uint32 scriptToScanThrough, uint32 functionToScanFor, void (*p_callback)(Script::CStruct *, const uint8 *), Script::CStruct *p_args, int Count)
1064 {
1065  // Don't recurse too deeply otherwise some levels take ages finding the gaps (eg, the secret level)
1066  if (Count > 5)
1067  {
1068  return;
1069  }
1070 
1071  Dbg_MsgAssert(p_callback,("NULL p_callback"));
1072 
1073  #ifdef __NOPT_ASSERT__
1074  // Look up the function being scanned for to see if it is a cfunction.
1075  bool (*p_cfunction)(CStruct *pParams, CScript *pCScript)=NULL;
1076  CSymbolTableEntry *p_function_entry=LookUpSymbol(functionToScanFor);
1077  if (p_function_entry && p_function_entry->mType==ESYMBOLTYPE_CFUNCTION)
1078  {
1079  p_cfunction=p_function_entry->mpCFunction;
1080  }
1081  Dbg_MsgAssert(p_cfunction==NULL,("Cannot use FindReferences to find cfunction calls yet ..."));
1082  #endif
1083 
1084  #ifdef NO_SCRIPT_CACHING
1085  // Look up the script
1086  CSymbolTableEntry *p_script_entry=LookUpSymbol(scriptToScanThrough);
1087  if (!p_script_entry)
1088  {
1089  Dbg_Warning("Script '%s' not found in call to FindReferences",FindChecksumName(scriptToScanThrough));
1090  return;
1091  }
1092  Dbg_MsgAssert(p_script_entry->mType==ESYMBOLTYPE_QSCRIPT,("'%s' is not a qscript",FindChecksumName(scriptToScanThrough)));
1093 
1094  // Get a pointer to it.
1095  uint8 *p_token=p_script_entry->mpScript;
1096  Dbg_MsgAssert(p_token,("NULL p_token ???"));
1097  // Skip over the 4-byte contents checksum.
1098  p_token+=4;
1099  #else
1100 
1101  Script::CScriptCache *p_script_cache=Script::CScriptCache::Instance();
1102  Dbg_MsgAssert(p_script_cache,("NULL p_script_cache"));
1103  uint8 *p_script=p_script_cache->GetScript(scriptToScanThrough);
1104  Dbg_MsgAssert(p_script,("Script %s not found in script cache",Script::FindChecksumName(scriptToScanThrough)));
1105  uint8 *p_token=p_script;
1106 
1107  #endif
1108 
1109 
1110  // Create a structure for holding parameters.
1111  CStruct *p_params=new CStruct;
1112 
1113  // Skip to the first line of the script.
1114  p_token=SkipToStartOfNextLine(p_token);
1115 
1116 
1117  // Now scan through each token of the script in turn, looking for any calls to functionToScanFor.
1118 
1119  bool finished=false;
1120  while (!finished)
1121  {
1122  switch (*p_token)
1123  {
1124  case ESCRIPTTOKEN_RUNTIME_CFUNCTION:
1125  {
1126  // Skip over CFunction calls for now.
1127  p_token+=5;
1128  p_token=SkipToStartOfNextLine(p_token);
1129  break;
1130  }
1131 
1132  case ESCRIPTTOKEN_NAME:
1133  case ESCRIPTTOKEN_RUNTIME_MEMBERFUNCTION:
1134  {
1135  // Remember the location for passing to the callback.
1136  const uint8 *p_location=p_token;
1137 
1138  ++p_token;
1139  uint32 name_checksum=Read4Bytes(p_token).mChecksum;
1140  p_token+=4;
1141 
1142  // Skip over lines of script that are setting parameters
1143  if (*p_token==ESCRIPTTOKEN_EQUALS)
1144  {
1145  break;
1146  }
1147 
1148  CSymbolTableEntry *p_entry=Resolve(name_checksum);
1149  if (p_entry)
1150  {
1151  switch (p_entry->mType)
1152  {
1153  case ESYMBOLTYPE_CFUNCTION:
1154  Dbg_MsgAssert(p_entry->mpCFunction,("\nLine %d of %s\nNULL pCFunction",GetLineNumber(p_token),FindChecksumName(p_entry->mSourceFileNameChecksum)));
1155 
1156  if (name_checksum==functionToScanFor)
1157  {
1158  // Found the required function call! Get the params and call the callback.
1159  p_params->Clear();
1160  // Note: Uses p_args to look up any params enclosed in <,> that are encountered.
1161  p_token=AddComponentsUntilEndOfLine(p_params,p_token,p_args);
1162 
1163  //printf("Callback: %s\n",FindChecksumName(scriptToScanThrough));
1164  (*p_callback)(p_params,p_location);
1165  }
1166  else
1167  {
1168  p_token=SkipToStartOfNextLine(p_token);
1169  }
1170  break;
1171 
1172  case ESYMBOLTYPE_MEMBERFUNCTION:
1173  if (name_checksum==functionToScanFor)
1174  {
1175  // Found the required function call! Get the params and call the callback.
1176  p_params->Clear();
1177  // Note: Uses p_args to look up any params enclosed in <,> that are encountered.
1178  p_token=AddComponentsUntilEndOfLine(p_params,p_token,p_args);
1179 
1180  //printf("Callback: %s\n",FindChecksumName(scriptToScanThrough));
1181  (*p_callback)(p_params,p_location);
1182 
1183  if (name_checksum==0xe5399fb2) // EndGap
1184  {
1185  uint32 GapScript=0;
1186  if (p_params->GetChecksum("GapScript",&GapScript))
1187  {
1188  // Passing NULL for p_args here because it is not clear what parameters are
1189  // going to be passed to the GapScript when it is run.
1190  FindReferences(GapScript,functionToScanFor,p_callback,NULL,Count+1);
1191  }
1192  }
1193  }
1194  else
1195  {
1196  p_token=SkipToStartOfNextLine(p_token);
1197  }
1198  break;
1199 
1200  case ESYMBOLTYPE_QSCRIPT:
1201  p_params->Clear();
1202  // Note: Uses p_args to look up any params enclosed in <,> that are encountered.
1203  p_token=AddComponentsUntilEndOfLine(p_params,p_token,p_args);
1204 
1205  if (name_checksum==functionToScanFor)
1206  {
1207  // Found the required function call! Get the params and call the callback.
1208  //printf("Callback: %s\n",FindChecksumName(scriptToScanThrough));
1209  (*p_callback)(p_params,p_location);
1210  }
1211  else
1212  {
1213  p_token=SkipToStartOfNextLine(p_token);
1214  }
1215 
1216  // It's a q-script, so recursively search that too.
1217  // Passing the just calculated p_params so that the <,> syntax within the script
1218  // can be evaluated.
1219  FindReferences(name_checksum,functionToScanFor,p_callback,p_params,Count+1);
1220  break;
1221 
1222  default:
1223  break;
1224  }
1225  }
1226  break;
1227  }
1228 
1229  case ESCRIPTTOKEN_KEYWORD_REPEAT:
1230  p_token=SkipToken(p_token);
1231  // Scan over any repeat parameters.
1232  p_token=SkipToStartOfNextLine(p_token);
1233  break;
1234 
1235  case ESCRIPTTOKEN_KEYWORD_IF:
1236  p_token=SkipToken(p_token);
1237 
1238  if (*p_token==ESCRIPTTOKEN_KEYWORD_NOT)
1239  {
1240  p_token=SkipToken(p_token);
1241  }
1242 
1243  // If the if keyword is followed by an open parenthesis, then it must be using an
1244  // expression. Often the expression may contain c-function calls, such as:
1245  // if ( (GotParam Foo) | (GotParam Blaa) )
1246  // This was causing a problem because this loop would skip over the open parenthesis,
1247  // get to the GotParam, and then recognizing that GotParam is a cfunction it would
1248  // try to skip to the next line, but then that would cause a parenthesis mismatch
1249  // assert in SkipToStartOfNextLine because the open parenth had been skipped over.
1250  // So to get around that, skip to the next line as soon as the open parenth is detected.
1251  if (*p_token==ESCRIPTTOKEN_OPENPARENTH)
1252  {
1253  p_token=SkipToStartOfNextLine(p_token);
1254  }
1255  break;
1256 
1257  case ESCRIPTTOKEN_KEYWORD_ENDSCRIPT:
1258  finished=true;
1259  break;
1260 
1261  default:
1262  p_token=SkipToken(p_token);
1263  break;
1264  }
1265  }
1266 
1267  #ifndef NO_SCRIPT_CACHING
1268  p_script_cache->DecrementScriptUsage(scriptToScanThrough);
1269  #endif
1270 
1271  // Delete the temporary p_params.
1272  delete p_params;
