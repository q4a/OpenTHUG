1169  if( mode != EngineGlobals.blend_mode_value )
1170  {
1171  // Low 24 bits contain the mode, high 8 bits contain the fixed alpha value.
1172  if(( mode & 0x00FFFFFFUL ) != ( EngineGlobals.blend_mode_value & 0x00FFFFFFUL ))
1173  {
1174  // For additive and subtractive, we set the fog color to black.
1175  if( EngineGlobals.fog_enabled )
1176  {
1177  if((( mode & 0x00FFFFFFUL ) >= vBLEND_MODE_ADD ) && (( mode & 0x00FFFFFFUL ) <= vBLEND_MODE_SUB_FIXED ))
1178  {
1179  D3DDevice_SetRenderState( D3DRS_FOGCOLOR, 0x00000000UL );
1180  }
1181  else
1182  {
1183  D3DDevice_SetRenderState( D3DRS_FOGCOLOR, NxXbox::EngineGlobals.fog_color );
1184  }
1185  }
1186 
1187  int blend_op, src_blend, dest_blend;
1188 
1189  switch( mode & 0x00FFFFFFUL )
1190  {
1191  case vBLEND_MODE_DIFFUSE: // ( 0 - 0 ) * 0 + Src
1192  {
1193  blend_op = D3DBLENDOP_ADD;
1194  src_blend = D3DBLEND_ONE;
1195  dest_blend = D3DBLEND_ZERO;
1196  break;
1197  }
1198  case vBLEND_MODE_ADD: // ( Src - 0 ) * Src + Dst
1199  {
1200  blend_op = D3DBLENDOP_ADD;
1201  src_blend = D3DBLEND_SRCALPHA;
1202  dest_blend = D3DBLEND_ONE;
1203  break;
1204  }
1205  case vBLEND_MODE_ADD_FIXED: // ( Src - 0 ) * Fixed + Dst
1206  {
1207  blend_op = D3DBLENDOP_ADD;
1208  src_blend = D3DBLEND_CONSTANTALPHA;
1209  dest_blend = D3DBLEND_ONE;
1210  break;
1211  }
1212  case vBLEND_MODE_SUBTRACT: // ( 0 - Src ) * Src + Dst
1213  {
1214  blend_op = D3DBLENDOP_REVSUBTRACT;
1215  src_blend = D3DBLEND_SRCALPHA;
1216  dest_blend = D3DBLEND_ONE;
1217  break;
1218  }
1219  case vBLEND_MODE_SUB_FIXED: // ( 0 - Src ) * Fixed + Dst
1220  {
1221  blend_op = D3DBLENDOP_REVSUBTRACT;
1222  src_blend = D3DBLEND_CONSTANTALPHA;
1223  dest_blend = D3DBLEND_ONE;
1224  break;
1225  }
1226  case vBLEND_MODE_BLEND: // ( Src - Dst ) * Src + Dst
1227  {
1228  blend_op = D3DBLENDOP_ADD;
1229  src_blend = D3DBLEND_SRCALPHA;
1230  dest_blend = D3DBLEND_INVSRCALPHA;
1231  break;
1232  }
1233  case vBLEND_MODE_BLEND_FIXED: // ( Src - Dst ) * Fixed + Dst
1234  {
1235  blend_op = D3DBLENDOP_ADD;
1236  src_blend = D3DBLEND_CONSTANTALPHA;
1237  dest_blend = D3DBLEND_INVCONSTANTALPHA;
1238  break;
1239  }
1240  case vBLEND_MODE_MODULATE: // ( Dst - 0 ) * Src + 0
1241  {
1242  blend_op = D3DBLENDOP_ADD;
1243  src_blend = D3DBLEND_ZERO;
1244  dest_blend = D3DBLEND_SRCALPHA;
1245  break;
1246  }
1247  case vBLEND_MODE_MODULATE_FIXED: // ( Dst - 0 ) * Fixed + 0
1248  {
1249  blend_op = D3DBLENDOP_ADD;
1250  src_blend = D3DBLEND_ZERO;
1251  dest_blend = D3DBLEND_CONSTANTALPHA;
1252  break;
1253  }
1254  case vBLEND_MODE_BRIGHTEN: // ( Dst - 0 ) * Src + Dst
1255  {
1256  blend_op = D3DBLENDOP_ADD;
1257  src_blend = D3DBLEND_DESTCOLOR;
1258  dest_blend = D3DBLEND_ONE;
1259  break;
1260  }
1261  case vBLEND_MODE_BRIGHTEN_FIXED: // ( Dst - 0 ) * Fixed + Dst
1262  {
1263  blend_op = D3DBLENDOP_ADD;
1264  src_blend = D3DBLEND_DESTCOLOR;
1265  dest_blend = D3DBLEND_CONSTANTALPHA;
1266  break;
1267  }
1268  case vBLEND_MODE_GLOSS_MAP:
1269  {
1270  // Treat as diffuse for now.
1271  blend_op = D3DBLENDOP_ADD;
1272  src_blend = D3DBLEND_ONE;
1273  dest_blend = D3DBLEND_ZERO;
1274  break;
1275  }
1276  case vBLEND_MODE_MODULATE_COLOR: // ( Dst - 0 ) * Src(col) + 0 - specially for the shadow.
1277  {
1278  blend_op = D3DBLENDOP_ADD;
1279  src_blend = D3DBLEND_ZERO;
1280  dest_blend = D3DBLEND_SRCCOLOR;
1281  break;
1282  }
1283  case vBLEND_MODE_BLEND_PREVIOUS_MASK:
1284  {
1285  // Meaningless unless destination alpha is enabled.
1286  blend_op = D3DBLENDOP_ADD;
1287  src_blend = D3DBLEND_DESTALPHA;
1288  dest_blend = D3DBLEND_INVDESTALPHA;
1289  break;
1290  }
1291  case vBLEND_MODE_BLEND_INVERSE_PREVIOUS_MASK:
1292  {
1293  // Meaningless unless destination alpha is enabled.
1294  blend_op = D3DBLENDOP_ADD;
1295  src_blend = D3DBLEND_INVDESTALPHA;
1296  dest_blend = D3DBLEND_DESTALPHA;
1297  break;
1298  }
1299  case vBLEND_MODE_ONE_INV_SRC_ALPHA:
1300  {
1301  blend_op = D3DBLENDOP_ADD;
1302  src_blend = D3DBLEND_ONE;
1303  dest_blend = D3DBLEND_INVSRCALPHA;
1304  break;
1305  }
1306  default:
1307  {
1308  Dbg_Assert( 0 );
1309  break;
1310  }
1311  }
1312 
1313  // Now set the values if they have changed.
1314  if( blend_op != EngineGlobals.blend_op )
1315  {
1316  D3DDevice_SetRenderState( D3DRS_BLENDOP, blend_op );
1317  EngineGlobals.blend_op = blend_op;
1318  }
1319  if( src_blend != EngineGlobals.src_blend )
1320  {
1321  D3DDevice_SetRenderState( D3DRS_SRCBLEND, src_blend );
1322  EngineGlobals.src_blend = src_blend;
1323  }
1324  if( dest_blend != EngineGlobals.dest_blend )
1325  {
1326  D3DDevice_SetRenderState( D3DRS_DESTBLEND, dest_blend );
1327  EngineGlobals.dest_blend = dest_blend;
1328  }
1329  }
1330 
1331  // Change the fixed alpha value if different.
1332  if(( mode & 0xFF000000UL ) != ( EngineGlobals.blend_mode_value & 0xFF000000UL ))
1333  {
1334  uint32 fixed_alpha = mode & 0xFF000000UL;
1335  fixed_alpha = fixed_alpha >= 0x80000000UL ? 0xFF000000UL : ( fixed_alpha << 1 );
1336  D3DDevice_SetRenderState( D3DRS_BLENDCOLOR, fixed_alpha );
1337  }
1338 
1339  // Set the new blend mode value.
1340  EngineGlobals.blend_mode_value = mode;
1341  }
1342 }
