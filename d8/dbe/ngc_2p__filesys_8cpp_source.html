174 // At first, I assumed that only one skyfile could be open at
175 // a time, but that's not true when testing streams from the
176 // PC build.
177 // pc builds need to have two (one for normal files, and one for streams)
178 // GJ: for THPS4, we increased this to 3, so that we can play multiple streams
179 const int MAXOPENFILES = 3;
180 
181 // Ken note: It used to be that MAXOPENFILES was defined to be 1 when __NOPT_CDROM__OLD
182 // was set. Now that __NOPT_CDROM__OLD is not used anymore and CD() is used at runtime
183 // instead, I made MAXOPENFILES always 2. Then, where MAXOPENFILES used to be used in the code
184 // it now uses a var set to 1 or 2 depending on CD()
185 static skyFile g_skyFile[MAXOPENFILES];
186 
187 skyFile* lock_skyfile( void )
188 {
189  // cd builds can only have one file open at any given time
190  int max_open_files=MAXOPENFILES;
191 
192  int i;
193  // this tries to find an unlocked skyfile
194  for ( i = 0; i < max_open_files; i++ )
195  {
196  if ( !g_skyFile[i].locked )
197  {
198  g_skyFile[i].locked = true;
199  g_skyFile[i].ngcFile.m_FileOpen = 0;
200  g_skyFile[i].ngcFile.m_sizeCached = 0;
201  g_skyFile[i].ngcFile.m_numCacheBytes = 0;
202  g_skyFile[i].ngcFile.m_seekOffset = 0;
203  g_skyFile[i].ngcFile.m_preHandle = NULL;
204  g_skyFile[i].ngcFile.m_unique_tag = 0x1234ABCD;
205  return &g_skyFile[i];
206  }
207  }
208 
209  Dbg_Message( "Here are the files currently open:" );
210  for ( i = 0; i < max_open_files; i++ )
211  {
212  Dbg_Message( "%s", g_skyFile[ i ].filename );
213  }
214 
215  // if we get here, that means that all the sky files were locked
216  Dbg_MsgAssert( 0, ( "Trying to open too many files simultaneously (max=%d)", max_open_files ) );
217  return NULL;
218 }
219 
220 void unlock_skyfile( skyFile* pSkyFile )
221 {
222  // cd builds can only have one file open at any given time
223  int max_open_files=MAXOPENFILES;
224 
225 
226  // this tries to find the sky file that the caller is referencing
227  for ( int i = 0; i < max_open_files; i++ )
228  {
229  if ( pSkyFile == &g_skyFile[i] )
230  {
231  Dbg_MsgAssert( g_skyFile[i].locked, ( "Trying to unlock a sky file too many times" ) );
232  g_skyFile[i].locked = false;
233  return;
234  }
235  }
236 
237  // if we get here, that means that the pointer didn't match one of the valid skyfiles
238  Dbg_MsgAssert( 0, ( "Unrecognized sky file" ) );
239 }

527  break;
