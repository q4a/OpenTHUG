256 #define MAX_IRX_NAME_CHARS 100
257 static char pIRXNameBuf[MAX_IRX_NAME_CHARS+1];
258 
259 // Loads an IRX file. The name must not have the .irx extension, it gets added by the function.
260 // Returns 0 if successful, -1 if error. (But it will assert in debug mode)
261 int LoadIRX(const char *pName, int num_args, char* args, bool assert_on_fail)
262 {
263 
264 
265  File::StopStreaming( );
266  if ( Pcm::UsingCD( ) )
267  {
268  Dbg_MsgAssert( 0,( "Using CD returned TRUE." ));
269  return ( -1 );
270  }
271 
272  // Start the name with the appropriate thingy.
273  if (Config::CD())
274  {
275  Dbg_MsgAssert(strlen("cdrom0:\\xxxxxxxxxxxxxxxxxIOP\\")+strlen(pName)+strlen(".IRX")<=MAX_IRX_NAME_CHARS,("String in pName is too long."));
276  char path[128];
277  sprintf(path,"cdrom0:\\%sIOP\\",Config::GetDirectory());
278  strcpy(pIRXNameBuf,path);
279  }
280  else
281  {
282  Dbg_MsgAssert(strlen("host0:IOPModules/")+strlen(pName)+strlen(".IRX")<=MAX_IRX_NAME_CHARS,("String in pName is too long."));
283  strcpy(pIRXNameBuf,"host0:IOPModules/");
284  }
285 
286  // Append the name in upper case.
287  char *pDest=pIRXNameBuf+strlen(pIRXNameBuf);
288  const char *pSource=pName;
289  while (*pSource)
290  {
291  char ch=*pSource++;
292  if (ch>='a' && ch<='z')
293  {
294  ch='A'+ch-'a';
295  }
296  *pDest++=ch;
297  }
298  *pDest=0;
299 
300  // Append the extension.
301  strcat(pIRXNameBuf,".IRX");
302 
303  int result;
304  if(( result = sceSifLoadModule(pIRXNameBuf, num_args, args )) < 0 )
305  {
306  Dbg_MsgAssert( !assert_on_fail,( "Can't load module %s. Result: %d\n",pIRXNameBuf, result ));
307 
308  return result;
309  }
310  return 0;
311 }
