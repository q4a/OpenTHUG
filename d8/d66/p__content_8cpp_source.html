127 GHTTPBool ContentMan::s_dl_completed_callback( GHTTPRequest request, // The request.
128  GHTTPResult result, // The result (success or an error).

141  if( result == GHTTPSuccess )
142  {
143  uint32 type;
144 
145  content_man->m_result = vCONTENT_RESULT_SUCCESS;
146 
147  type = 0;
148  switch( content_man->m_vault_type )
149  {
150  case 0x5be5569f: // parks
151  type = CRCD(0x3bf882cc,"park");
152  break;
153  case 0x7364ea1: // skaters
154  type = CRCD(0xb010f357,"optionsandpros");
155  break;
156  case 0x38dbe1d0: // goals
157  type = CRCD(0x62896edf,"createdgoals");
158  break;
159  case 0x1e26fd3e: // tricks
160  type = CRCD(0x61a1bc57,"cat");
161  break;
162  default:
163  Dbg_Assert( 0 );
164  break;
165  }
166  CFuncs::SetVaultData((uint8*) buffer, type );
167 
168  content_man->m_error = 0;
169  }
170  else
171  {
172  content_man->m_result = vCONTENT_RESULT_FAILURE;
173  content_man->m_error = result;
174  }
175 
176  return GHTTPTrue;
177 }
