96 void CModelLightUpdateComponent::Update()
97 {
98  // Find the closest ground collision and extract the brightness from it.
99  // logic extracted from CAdjustComponent::uber_frig
100 
101  CFeeler feeler;
102 
103  feeler.m_start = GetObject()->m_pos;
104  feeler.m_end = GetObject()->m_pos;
105 
106  // Very minor adjustment to move origin away from vert walls
107  feeler.m_start += GetObject()->m_matrix[Y] * 0.001f;
108 
109  feeler.m_start[Y] += 8.0f;
110  feeler.m_end[Y] -= FEET(400);
111 
112  if (!feeler.GetCollision()) return;
113 
114  Dbg_Assert(mp_model_component);
115 
116  if (feeler.IsBrightnessAvailable())
117  {
118  Nx::CModelLights *p_model_lights = mp_model_component->GetModel()->GetModelLights();
119  if (p_model_lights)
120  {
121  p_model_lights->SetBrightness(feeler.GetBrightness());
122  }
123  }
124 
125 }
