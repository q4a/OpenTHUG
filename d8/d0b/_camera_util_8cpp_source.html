19 /*******************************************************************************************************
20 logic for getTimeAdjustedSlerp()
21 
22 If the camera is lagging a distance D behind the target, and the target is moving with velocity V, and the lerp is set to L.
23 Then when the camera is stable, it will be moving at the same rate as the target (V), and since the distance it will move is
24 equal to the lerp multiplied by the distance from the target (which will be the old distance plus the new movement of the
25 target V)
26 
27 Then:
28 
29 L*(D+V) = V
30 L*D + L*V = V
31 D = V * (1 - L) / L
32 
33 Assuming in the above T = 1, then if we have a variable T, the speed of the skater will be T*V, yet we want the distance (Dt)
34 moved to remain unchanged (D), so for a time adjusted lerp Lt
35 
36 Dt = T * V * (1-Lt)/Lt
37 
38 Since D = Dt
39 
40 V * (1 - L) / L = T * V * (1-Lt)/Lt
41 
42 V cancels out, and we get
43 
44 Lt = TL / (1 - L + TL)
45 
46 Sanity check,
47 
48 if L is 0.25, and T is 1, then Lt = 1*0.25 / (1 - 0.25 + 1*0.25) = 0.25
49 if L is 0.25, and T is 2, then Lt = 2*0.25 / (1 - 0.25 + 2*0.25) = 0.5 / 1.25 = 0.40
50 if L is 0.25, and T is 5, then Lt = 5*0.25 / (1 - 0.25 + 5*0.25) = 1.25 / 2 = 0.625
51 
52 Sounds about right.
53 */

60 float GetTimeAdjustedSlerp( float slerp, float delta )
61 {
62  // Mick's method, tries to guarantee that the distance from the target point
63  // will never alter at constant velocity, with changing frame rate
64  // Lt = TL / (1 - L + TL)
65  float t = delta * 60.0f;
66  float Lt = t * slerp / ( 1.0f - slerp + t * slerp );
67  return Lt;
68 }

77 void ApplyCameraCollisionDetection ( Mth::Vector& camera_pos, Mth::Matrix& camera_matrix, const Mth::Vector& target_pos, const Mth::Vector& focus_pos, bool side_feelers, bool refocus )
78 {
79  // camera collision detection
80 
81  Mth::Vector target_to_camera_direction = camera_pos - target_pos;
82  target_to_camera_direction.Normalize();
83 
84  CFeeler feeler;
85 
86  // Ignore faces based on face falgs
87  feeler.SetIgnore(IGNORE_FACE_FLAGS_1, IGNORE_FACE_FLAGS_0);
88  feeler.SetLine(target_pos, camera_pos + vMIN_WALL_DISTANCE * target_to_camera_direction);
89  bool collision = feeler.GetCollision(true);
90  if( collision )
91  {
92 // Gfx::AddDebugLine( at_pos, cam_pos, MAKE_RGB( 128, 0, 0 ), MAKE_RGB( 128, 0, 0 ), 2000 );
93 
94  // Limits the camera to getting nearer than 11.9 inches to the focus point.
95  float distance = feeler.GetDist();
96  float length = feeler.Length();
97  if( length * distance < 11.9f + vMIN_WALL_DISTANCE )
98  {
99  distance = (11.9f + vMIN_WALL_DISTANCE) / length;
100  }
101  camera_pos = target_pos + (distance * length - vMIN_WALL_DISTANCE) * target_to_camera_direction;
102  }
103 
104  if (!side_feelers) return;
105 
106 //# if defined( __PLAT_NGPS__ ) || defined ( __PLAT_XBOX__ )
107  // Now do two additional checks 1 foot to either side of the camera.
108  Mth::Vector left( camera_matrix[X][X], 0.0f, camera_matrix[X][Z], 0.0f );
109  left.Normalize( 8.0f );
110  left += camera_pos;
111 
112  collision = feeler.GetCollision(camera_pos, left, true);
113  if( collision )
114  {
115  left -= feeler.GetPoint();
116  camera_pos -= left;
117  }
118  else
119  {
120  Mth::Vector right( -camera_matrix[X][X], 0.0f, -camera_matrix[X][Z], 0.0f );
121  right.Normalize( 8.0f );
122  right += camera_pos;
123 
124  collision = feeler.GetCollision(camera_pos, right, true);
125  if( collision )
126  {
127  right -= feeler.GetPoint();
128  camera_pos -= right;
129  }
130  }
131 
132  if (!refocus) return;
133 
134  if( collision )
135  {
136  // Re-orient the camera again.
137  camera_matrix[Z].Set( focus_pos.GetX() - camera_pos.GetX(), focus_pos.GetY() - camera_pos.GetY(), focus_pos.GetZ() - camera_pos.GetZ());
138  camera_matrix[Z].Normalize();
139 
140  // Read back the Y from the current matrix.
141 // target[Y][0] = p_frame_matrix->up.x;
142 // target[Y][1] = p_frame_matrix->up.y;
143 // target[Y][2] = p_frame_matrix->up.z;
144 
145  // Generate new orthonormal X and Y axes.
146  camera_matrix[X] = Mth::CrossProduct( camera_matrix[Y], camera_matrix[Z] );
147  camera_matrix[X].Normalize();
148  camera_matrix[Y] = Mth::CrossProduct( camera_matrix[Z], camera_matrix[X] );
149  camera_matrix[Y].Normalize();
150 
151  // Write back into camera matrix.
152  camera_matrix[X] = -camera_matrix[X];
153  camera_matrix[Z] = -camera_matrix[Z];
154 
155  // Fix the final column
156  camera_matrix[X][W] = 0.0f;
157  camera_matrix[Y][W] = 0.0f;
158  camera_matrix[Z][W] = 0.0f;
159  camera_matrix[W][W] = 1.0f;
160 
161  }
