231 void CParticle::process( float delta_time )
232 {
233  int lp;
234  CParticleEntry* p_particle;
235 
236  // If flagged for deletion, check if the particle system is empty and delete if it is.
237  if ( m_delete_when_empty )
238  {
239  if ( !m_num_particles )
240  {
241  // removing reference is sone in the destructor
242  //CEngine::sGetParticleTable()->FlushItem( m_checksum ); // remove reference from hash table
243  delete this;
244  return;
245  }
246  }
247 
248  // Don't process if paused.
249  if (Mdl::FrontEnd::Instance()->GamePaused() && !Replay::RunningReplay())
250  {
251  return;
252  }
253  if (Replay::Paused())
254  {
255  return;
256  }
257 
258  if ( GetSuspendComponent()->SkipLogic( ) )
259  {
260  return;
261  }
262 
263 
264 
265  // Script updates.
266  if ( mp_update_script )
267  {
268  mp_update_script->Update();
269  }
270 
271 // We can't check for off screen or not until we've run the update script
272 // as that might change the position (like for sparks)
273 
274  // Don't process if off screen
275 // if (!Nx::CEngine::sIsVisible(m_pos,m_bounding_radius))
276 // {
277 // //printf ("Off - process\n");
278 // Gfx::AddDebugLine(m_pos,Mth::Vector(0,0,0),0x0000ff,0x0000ff,100); // where is it?
279 // return;
280 // }
281 
282 
283  if ( mp_emit_script )
284  {
285  mp_emit_script->Update();
286 // if ( mp_emit_script->Update() == Script::ESCRIPTRETURNVAL_FINISHED )
287 // {
288 // m_delete_when_empty = true;
289 // }
290  }
291 
292 
293 
294  // If life was not set, assume we're a screen-facing system.
295  if ( !m_life_set ) return;
296 
297  // Physics update.
298  for ( lp = 0, p_particle = mp_particle_array; lp < m_num_particles; lp++, p_particle++ )
299  {
300  // Update time/life.
301  p_particle->m_time += delta_time;
302 
303  // Apply force to velocity.
304  p_particle->m_vx += p_particle->m_fx;
305  p_particle->m_vy += p_particle->m_fy;
306  p_particle->m_vz += p_particle->m_fz;
307 
308  // Copy history.
309  float x, y, z;
310  for ( int lp2 = ( plat_get_num_vertex_lists() - 1 ); lp2 > 0; lp2-- )
311  {
312  plat_get_position( lp, lp2 - 1, &x, &y, &z );
313  plat_set_position( lp, lp2, x, y, z );
314  }
315 
316  // Apply velocity to position.
317  plat_add_position( lp, 0, p_particle->m_vx, p_particle->m_vy, p_particle->m_vz );
318  }
319 
320  // Delete old particles.
321  for ( lp = 0, p_particle = mp_particle_array; lp < m_num_particles; lp++, p_particle++ )
322  {
323  if ( p_particle->m_time > p_particle->m_life )
324  {
325  // Delete old particles by copying last entry to entry that is being deleted.
326  m_num_particles--;
327  memcpy( p_particle, &mp_particle_array[m_num_particles], sizeof( CParticleEntry ) );
328 
329  float x, y, z;
330  for ( int lp2 = 0; lp2 < plat_get_num_vertex_lists(); lp2++ )
331  {
332  plat_get_position( m_num_particles, lp2, &x, &y, &z );
333  plat_set_position( lp, lp2, x, y, z );
334  plat_set_position( m_num_particles, lp2, 1024.0f, 1024.0f, 1024.0f );
335  }
336 
337  // Since we copied the last one to the current slot, decrement lp counter & particle
338  // pointer so that the copied entry gets checked for deletion as well.
339  p_particle--;
340  lp--;
341  }
342  }

356 void CParticle::emit( int count )
357 {
358  int first, last;
359 
360  // Do nothing here if flagged for deletion.
361  if ( m_delete_when_empty )
362  {
363  return;
364  }
365 
366  // Calculate the range of array entries to emit to.
367  first = m_num_particles;
368  if ( ( m_num_particles + count ) <= m_max_particles )
369  {
370  m_num_particles += count;
371  }
372  else
373  {
374  m_num_particles = m_max_particles;
375  }
376  last = m_num_particles;
377 
378  // Build matrix to rotate the particle to the target.
379  Mth::Matrix mt;
380 
381  mt.GetPos().Set();
382  mt.GetAt().Set( m_tx, m_ty, m_tz );
383  mt.GetAt().Normalize();
384 
385  Mth::Vector temp;
386  temp.Set( 0.0f, 1.0f, 0.0f );
387  if ( ( fabs( mt.GetAt()[Y] ) > fabs( mt.GetAt()[X] ) ) && ( fabs( mt.GetAt()[Y] ) > fabs( mt.GetAt()[Z] ) ) )
388  {
389  // Y Major - use Z as up.
390  temp.Set( 0.0f, 0.0f, 1.0f );
391  }
392 
393  mt.GetRight() = Mth::CrossProduct( temp, mt.GetAt() );
394  mt.GetRight().Normalize();
395  mt.GetUp() = Mth::CrossProduct( mt.GetRight(), mt.GetAt() );
396  mt.GetUp().Normalize();
397 
398  // Add on rotation.
399  mt.RotateX( Mth::DegToRad(m_ax) );
400  mt.RotateY( Mth::DegToRad(m_ay) );
401  mt.RotateZ( Mth::DegToRad(m_az) );
402 
403 // pright.set( mt.GetRight()[X], mt.GetRight()[Y], mt.GetRight()[Z] );
404 // pup.set( mt.GetUp()[X], mt.GetUp()[Y], mt.GetUp()[Z] );
405 // pat.set( mt.GetAt()[X], mt.GetAt()[Y], mt.GetAt()[Z] );
406 
407  // Emit particles.
408  for ( int lp = first; lp < last; lp++ )
409  {
410  if ( m_life_set )
411  {
412  // Random values for this emission.
413  float rx = next_random();
414  float ry = next_random();
415 
416  // Calculate emission initial velocity.
417  Mth::Vector velocity;
418  velocity[X] = rx * ( Mth::PI * ( m_emit_angle_spread / 4.0f ) );
419  velocity[Y] = ry * ( Mth::PI * ( m_emit_angle_spread / 4.0f ) );
420  velocity[Z] = 1.0f - (( velocity[X] * velocity[X] ) + ( velocity[Y] * velocity[Y] ));
421  velocity[W] = 1.0f;
422 
423  float speed_mul = ( ( m_speed_max - m_speed_min ) * ((float)rand() / RAND_MAX ) ) + m_speed_min;
424  velocity[X] *= speed_mul;
425  velocity[Y] *= speed_mul;
426  velocity[Z] *= speed_mul;
427 
428  // Get new rx, ry to make the particles cross over one another.
429  if ( m_random_angle )
430  {
431  rx = next_random();
432  ry = next_random();
433  }
434 
435  // Normalize rx, ry to make a circular emission.
436  if ( m_circular_emit )
437  {
438  float l = 1.0f / sqrtf( ( rx * rx ) + ( ry * ry ) );
439  rx = rx * ( l * (float)fabs( rx ) ); // Note: length is scaled by random factor.
440  ry = ry * ( l * (float)fabs( ry ) );
441  }
442 
443  // Set position around emitter width/height.
444  Mth::Vector pos;
445  pos[X] = m_emit_w * rx;
446  pos[Y] = m_emit_h * ry;
447  pos[Z] = 0.0f;
448  pos[W] = 1.0f;
449 
450  // Set force.
451  mp_particle_array[lp].m_fx = m_fx;
452  mp_particle_array[lp].m_fy = m_fy;
453  mp_particle_array[lp].m_fz = m_fz;
454 
455  // Set particle w/h
456  mp_particle_array[lp].m_sw = m_sw;
457  mp_particle_array[lp].m_sh = m_sh;
458  mp_particle_array[lp].m_ew = m_ew;
459  mp_particle_array[lp].m_eh = m_eh;
460 
461  // Set time & life.
462  mp_particle_array[lp].m_time = 0.0f;
463  mp_particle_array[lp].m_life = ( ( m_life_max - m_life_min ) * ((float)rand() / RAND_MAX ) ) + m_life_min;
464 
465  // Orient the particles to the target.
466  Mth::Vector v;
467  v = mt.Transform( velocity );
468  mp_particle_array[lp].m_vx = v[X];
469  mp_particle_array[lp].m_vy = v[Y];
470  mp_particle_array[lp].m_vz = v[Z];
471 
472  v = mt.Transform( pos );
473 
474  for ( int lp2 = 0; lp2 < plat_get_num_vertex_lists(); lp2++ )
475  {
476  plat_set_position( lp, lp2, v[X], v[Y], v[Z] );
477  }
478  }
479  else
480  {
481  // Set particle w/h
482  mp_particle_array[lp].m_sw = m_sw;
483  mp_particle_array[lp].m_sh = m_sh;
484  mp_particle_array[lp].m_ew = m_ew;
485  mp_particle_array[lp].m_eh = m_eh;
486 
487  // Set time & life.
488  mp_particle_array[lp].m_time = 0.0f;
489  mp_particle_array[lp].m_life = ( ( m_life_max - m_life_min ) * ((float)rand() / RAND_MAX ) ) + m_life_min;
490 
491  // Screen-facing poly. Just set to 0,0,0.
492  for ( int lp2 = 0; lp2 < plat_get_num_vertex_lists(); lp2++ )
493  {
494  plat_set_position( lp, lp2, 0, 0, 0 );
495  }
496  }
497 
498  }

558 /******************************************************************/
559 /* CParticle::CallMemberFunction */
560 /* Call a member function, based on a checksum */
561 /* This is usually the checksum of the name of the function */
562 /* but can actually be any number, as it just uses a switch */
563 /* note this is a virtual function, so the same checksum */
564 /* can do differnet things for different objects */
565 /* */
566 /* */
567 /******************************************************************/
568 
569 bool CParticle::CallMemberFunction( uint32 Checksum, Script::CStruct *pParams, Script::CScript *pScript )
570 {
571 
572  Dbg_MsgAssert(pScript,("NULL pScript"));
573 
574  switch(Checksum)
575  {
576  // @script | SetPos | Sets the position of the particle system.
577  // @parmopt vector | pos | world position.
578  // @parmopt float | x | x world position.
579  // @parmopt float | y | y world position.
580  // @parmopt float | z | z world position.
581  case 0x99cd17b: // SetPos
582  {
583  Mth::Vector pos;
584  float x = 0.0f;
585  float y = 0.0f;
586  float z = 0.0f;
587  if ( pParams->GetVector(CRCD(0x7f261953,"pos"),&pos) )
588  {
589  x = pos[X];
590  y = pos[Y];
591  z = pos[Z];
592  }
593  else
594  {
595  pParams->GetFloat(CRCD(0x7323e97c,"x"),&x);
596  pParams->GetFloat(CRCD(0x424d9ea,"y"),&y);
597  pParams->GetFloat(CRCD(0x9d2d8850,"z"),&z);
598  }
599 
600  set_pos( x, y, z );
601  }
602  break;
603  // @script | SetSpeedRange | Sets the min & max speed of the particles upon emission.
604  // @parm float | min | minimum speed. If only min is provided, all speeds will be set to min.
605  // @parmopt float | max | maximum speed.
606  case 0x63274395: // SetSpeedRange
607  {
608  float min = 0.0f;
609  pParams->GetFloat(CRCD(0x5e84e22f,"min"),&min);
610  float max = min;
611  pParams->GetFloat(CRCD(0x6289dd76,"max"),&max);
612 
613  set_speed_range( min, max );
614  }
615  break;
616  // @script | SetEmitRange | Sets the emission range (width and height).
617  // @parm float | width | Width of emission area. If only width is provided, height will be set to width.
618  // @parmopt float | height | Height of emission area.
619  case 0x8ffdef38: // SetEmitRange
620  {
621  float width = 0.0f;
622  pParams->GetFloat(CRCD(0x73e5bad0,"width"),&width);
623  float height = width;
624  pParams->GetFloat(CRCD(0xab21af0,"height"),&height);
625 
626  set_emit_range( width, height );
627  }
628  break;
629  // @script | SetAngleSpread | Sets the angle spread for particle emissions.
630  // @parm float | spread | Angle spread. 0=No spread. 1=Full spread (180 degrees).
631  case 0x7911855b: // SetAngleSpread
632  {
633  float spread = 0.0f;
634  pParams->GetFloat(CRCD(0x834a5f87,"spread"),&spread);
635 
636  set_emit_angle_spread( spread );
637  }
638  break;
639 
640 
641  // @script | SetRandomAngle | Sets whether the emission angle is random or not.
642  // @parm int | random | 1 means choose a random angle regardless of the emission position. 0 means use the emission position to define the emission angle.
643  case 0xdb8621f7: // SetRandomAngle
644  {
645  int random = 0;
646  pParams->GetInteger("random",(int*)&random);
647 
648  set_random_angle( random > 0 );
649  }
650  break;
651 
652  // @script | SetCircularEmit | Sets whether the emission area is circular or square.
653  // @parm int | circular | 1 means use a circular emission area. 0 means use a square emission area.
654  case 0xe985a9a6: // SetCircularEmit
655  {
656  int circular = 0;
657  pParams->GetInteger("circular",(int*)&circular);
658 
659  set_circular_emit( circular > 0 );
660  }
661  break;
662 
663  // @script | SetForce | Sets the force acting on the particles.
664  // @parmopt vector | force | Force.
665  // @parmopt float | x | x force.
666  // @parmopt float | y | y force.
667  // @parmopt float | z | z force.
668  case 0xa075ee45: // SetForce
669  {
670  Mth::Vector force;
671  float x = 0.0f;
672  float y = 0.0f;
673  float z = 0.0f;
674  if ( pParams->GetVector("force",&force) )
675  {
676  x = force[X];
677  y = force[Y];
678  z = force[Z];
679  }
680  else
681  {
682  pParams->GetFloat("x",&x);
683  pParams->GetFloat("y",&y);
684  pParams->GetFloat("z",&z);
685  }
686 
687  set_force( x, y, z );
688  }
689  break;
690
