63 void CSk3SfxManager::SetSkaterSoundInfo( int surfaceFlag, uint32 whichSound, int whichArray,
64  float maxPitch, float minPitch, float maxVol, float minVol )
65 {
66 
67  if (Sfx::NoSoundPlease()) return;
68 
69  // must initialize PInfo!
70  int i;
71  Sfx::CSfxManager * sfx_manager = Sfx::CSfxManager::Instance();
72 
73  if ( NULL == sfx_manager->GetWaveTableIndex( whichSound ) )
74  {
75  Dbg_MsgAssert( 0,( "Terrain sound not loaded! surface %d sound %s checksum %d soundType %d",
76  surfaceFlag, Script::FindChecksumName( whichSound ), whichSound, whichArray ));
77  return;
78  }
79  int numEntries = mNumEntries[ whichArray ];
80  SkaterSoundInfo *pArray = mSoundArray[ whichArray ];
81  SkaterSoundInfo *pInfo = NULL;
82 
83  for ( i = 0; i < numEntries; i++ )
84  {
85  if ( pArray[ i ].surfaceFlag == surfaceFlag )
86  {
87  Dbg_Message( "Re-defining soundtype %d for surfaceFlag %d", whichArray, surfaceFlag );
88  pInfo = &pArray[ i ];
89  break;
90  }
91  }
92  if ( !pInfo )
93  {
94  pInfo = &pArray[ mNumEntries[ whichArray ] ];
95  mNumEntries[ whichArray ] += 1;
96  Dbg_MsgAssert( mNumEntries[ whichArray ] < vMAX_NUM_ENTRIES,( "Array too small type %d. Increase MAX_NUM_ENTRIES.", whichArray ));
97  }
98 
99  Dbg_MsgAssert( pInfo,( "Please fire Matt immediately after kicking him in the nuts." ));
100 
101  // surfaceFlag of zero will be used for the default
102  pInfo->surfaceFlag = surfaceFlag;
103  // if soundChecksum is zero, no sound will play on this surface.
104  pInfo->soundChecksum = whichSound;
105  pInfo->maxPitch = maxPitch;
106  pInfo->minPitch = minPitch;
107  pInfo->maxVol = maxVol;
108  pInfo->minVol = minVol;
109 
110 } // end of SetSkaterSoundInfo( )

112 SkaterSoundInfo *CSk3SfxManager::GetSkaterSoundInfo( int surfaceFlag, int whichArray )
113 {
114 
115  if (Sfx::NoSoundPlease()) return NULL;
116 
117 
118  int numEntries = 0;
119  numEntries = mNumEntries[ whichArray ];
120  SkaterSoundInfo *pArray = mSoundArray[ whichArray ];
121  Dbg_MsgAssert( pArray,( "Fire Matt please." ));
122  int i;

124  {
125  if ( pArray[ i ].surfaceFlag == surfaceFlag )
126  {
127  return ( &pArray[ i ] );

129  }
130  // couldn't find the surface flag in the table... return the default:

135  return ( &pArray[ i ] );
136  }
137  }
138 
139  return ( NULL );
