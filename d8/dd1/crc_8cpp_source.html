139 // Generates a checksum from raw data, given pointer to that data, and the size
140 //
141 // input:
142 // const char *stream = pointer to data
143 // int size = number of bytes to CRC
144 //
145 // output:
146 // returns uint32 = checksum of the data
147 // if NULL pointer passed in, or zero size is given, then will return 0
148 //
149 uint32 GenerateCRCCaseSensitive( const char *stream, int size )
150 {
151 
152 
153  // A checksum of zero is used to mean no name.
154  if(( !stream ) || ( size <= 0 ))
155  {
156  return 0;
157  }
158 
159  // Initializing the CRC to all one bits avoids failure of detection
160  // should entire data stream get cyclically bit-shifted by one position.
161  // The calculation of the probability of this happening is left as
162  // an exercise for the reader.
163  uint32 rc = 0xffffffff;
164  for (int i=0; i<size; i++)
165  {
166  rc = CRCTable[(rc^stream[i]) & 0xff] ^ ((rc>>8) & 0x00ffffff);
167  }
168 
169  return rc;
170 }

195 // Generates a checksum from a name, case insensitive.
196 uint32 GenerateCRC( const char *stream, int size )
197 {
198 
199 
200  // A checksum of zero is used to mean no name.
201  if(( !stream ) || ( size <= 0 ))
202  {
203  return 0;
204  }
205 
206  // Initializing the CRC to all one bits avoids failure of detection
207  // should entire data stream get cyclically bit-shifted by one position.
208  // The calculation of the probability of this happening is left as
209  // an exercise for the reader.
210  uint32 rc = 0xffffffff;
211  for (int i=0; i<size; i++)
212  {
213  char ch=stream[i];
214  // Convert to lower case.
215  if (ch>='A' && ch<='Z')
216  {
217  ch='a'+ch-'A';
218  }
219  // Convert forward slashes to backslashes, otherwise two filenames which are
220  // effectively the same but with different slashes will give different checksums.
221  if (ch=='/')
222  {
223  ch='\\';
224  }
225 
226  rc = CRCTable[(rc^ch) & 0xff] ^ ((rc>>8) & 0x00ffffff);
227  }
228 
229  return rc;
230 }

237 uint32 GenerateCRCFromString( const char *pName )
238 {
239 
240  // A checksum of zero is used to mean no name.
241  if (!pName)
242  {
243  return 0;
244  }
245 
246  // Initializing the CRC to all one bits avoids failure of detection
247  // should entire data stream get cyclically bit-shifted by one position.
248  // The calculation of the probability of this happening is left as
249  // an exercise for the reader.
250  uint32 rc = 0xffffffff;
251  const char *pCh=pName;
252  while (true)
253  {
254  char ch=*pCh++;
255  if (!ch)
256  {
257  break;
258  }
