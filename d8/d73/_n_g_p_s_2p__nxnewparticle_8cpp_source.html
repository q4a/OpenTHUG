69 void CPs2NewParticle::plat_render( void )
70 {
71  CParticleStream *p_stream;
72  int i;
73 
74  //--------------------------------------------------------------------------------------------------------------------
75  // process the streams
76  if (m_params.m_EmitRate && (!m_emitting || (m_params.m_EmitRate != mp_newest_stream->m_rate)))
77  {
78  if (m_num_streams < m_max_streams)
79  {
80  // add new stream to cyclic buffer
81  m_num_streams++;
82  mp_newest_stream++;
83  if (mp_newest_stream == mp_stream + m_max_streams)
84  {
85  mp_newest_stream = mp_stream;
86  }
87 
88  // initialise new stream
89  mp_newest_stream->m_rate = m_params.m_EmitRate;
90  mp_newest_stream->m_interval = 1.0f/m_params.m_EmitRate;
91  mp_newest_stream->m_oldest_age = 0.0f;
92  mp_newest_stream->m_num_particles = 0;
93  mp_newest_stream->m_seed[0] = rand();
94  mp_newest_stream->m_seed[1] = rand();
95  mp_newest_stream->m_seed[2] = rand();
96  mp_newest_stream->m_seed[3] = rand();
97  m_emitting = true;
98  }
99  else
100  {
101  m_emitting = false;
102  }
103  }
104  else
105  {
106  m_emitting = m_params.m_EmitRate;
107  }
108 
109  if (!m_num_streams)
110  return;
111 
112  // age all streams
113  for (i=0, p_stream=mp_oldest_stream; i<m_num_streams; i++)
114  {
115  // increase age of oldest particle
116  p_stream->m_oldest_age += 1.0f/60.0f;
117 
118  // step pointer within cyclic buffer
119  p_stream++;
120  if (p_stream == mp_stream + m_max_streams)
121  {
122  p_stream = mp_stream;
123  }
124  }
125 
126  // births into newest stream
127  if (m_emitting)
128  {
129  // how many particles so far emitted?
130  mp_newest_stream->m_num_particles = (int)(mp_newest_stream->m_oldest_age * mp_newest_stream->m_rate + 1.0f);
131  }
132 
133  // deaths from oldest stream
134  if (mp_oldest_stream->m_oldest_age > m_params.m_Lifetime)
135  {
136  // work out number dead
137  int particles_dead = (int)((mp_oldest_stream->m_oldest_age - m_params.m_Lifetime) * mp_oldest_stream->m_rate + 1.0f);
138 
139  // remove dead particles
140  mp_oldest_stream->m_num_particles -= particles_dead;
141 
142  // should we keep processing the oldest stream?
143  if (mp_oldest_stream->m_num_particles>0 || (m_num_streams==1 && m_emitting))
144  {
145  // adjust age of oldest particle
146  mp_oldest_stream->m_oldest_age -= (float)particles_dead * mp_oldest_stream->m_interval;
147 
148  // advance seed
149  mp_oldest_stream->AdvanceSeed(particles_dead);
150  }
151  else
152  {
153  // remove oldest stream and wrap in cyclic buffer if necessary
154  m_num_streams--;
155  mp_oldest_stream++;
156  if (mp_oldest_stream == mp_stream + m_max_streams)
157  {
158  mp_oldest_stream = mp_stream;
159  }
160  if (!m_num_streams)
161  return;
162  }
163  }
164 
165 
166  //--------------------------------------------------------------------------------------------------------------------
167  // now render the streams
168 
169  if (mp_engine_texture)
170  {
171  // Mick: If this is textured, we reset the TEX0, TEX1 regs as the texture might move
172  // Since 2D textures are dynamically packed every frame
173  *(uint64 *)&m_systemDmaData.m_GScontext[8] = mp_engine_texture->m_RegTEX0;
174  *(uint64 *)&m_systemDmaData.m_GScontext[12] = mp_engine_texture->m_RegTEX1;
175  }
176 
177  // FOG
178  Mth::Vector x = 0.5f * (m_params.m_BoxPos[0] + m_params.m_BoxPos[2]);
179  x[3] = 1.0f;
180  x *= NxPs2::render::WorldToFrustum;
181  float f;
182  if ((x[3] > NxPs2::render::FogNear) && NxPs2::render::EnableFog) // Garrett: We have to check for EnableFog here because the VU1 code isn't
183  {
184  f = 1.0 + NxPs2::render::EffectiveFogAlpha * (NxPs2::render::FogNear/x[3] - 1.0f);
185  }
186  else
187  {
188  f = 1.0f;
189  }
190  m_systemDmaData.m_GScontext[21] = ((uint32)(int)(f*255.99f)) << 24;
191 
192  // set the group
193  NxPs2::dma::SetList(NxPs2::sGroup::pParticles);
194 
195  // the system and streams will be loaded to a double-buffered input area of VUMem1
196  // ref the system data, and include an unpack for the system and streams:
197  NxPs2::dma::Tag(NxPs2::dma::ref, (sizeof(CSystemDmaData)>>4), (uint)&m_systemDmaData);
198  NxPs2::vif::STCYCL(1,1);
199  NxPs2::vif::UNPACK(0, V4_32, (sizeof(CSystemDmaData)>>4)+m_num_streams*2, REL, SIGNED, 0);
200 
201  // construct a packet with data for each stream
202  NxPs2::dma::BeginTag(NxPs2::dma::cnt, 0);
203  for (i=0,p_stream=mp_oldest_stream; i<m_num_streams; i++,p_stream++)
204  {
205  Dbg_MsgAssert(p_stream->m_num_particles<65536, ("particle limit reached (65536)"));
206 
207  // wrap at end of cyclic buffer
208  if (p_stream == mp_stream + m_max_streams)
209  {
210  p_stream = mp_stream;
211  }
212 
213  ((float *)NxPs2::dma::pLoc)[0] = p_stream->m_oldest_age;
214  ((float *)NxPs2::dma::pLoc)[1] = p_stream->m_interval;
215  ((uint32 *)NxPs2::dma::pLoc)[2] = p_stream->m_num_particles;
216  ((uint32 *)NxPs2::dma::pLoc)[3] = (p_stream==mp_newest_stream) ? 0x8000 : 0;
217  ((uint32 *)NxPs2::dma::pLoc)[4] = p_stream->m_seed[0];
218  ((uint32 *)NxPs2::dma::pLoc)[5] = p_stream->m_seed[1];
219  ((uint32 *)NxPs2::dma::pLoc)[6] = p_stream->m_seed[2];
220  ((uint32 *)NxPs2::dma::pLoc)[7] = p_stream->m_seed[3];
221 
222  //printf("stream %d, num particles=%d, oldest age=%f\n", s, p_stream->m_num_particles, p_stream->m_oldest_age);
223 
224  // step dma pointer
225  NxPs2::dma::pLoc += 8*4;
226 
227  // count particles
228  NxPs2::render::sTotalNewParticles += p_stream->m_num_particles;
229  }
230 
231  NxPs2::vif::MSCAL(10); // sprites
232  NxPs2::dma::EndTag();
233  NxPs2::dma::SetList(NxPs2::sGroup::pEpilogue);
234 }

241 void CPs2NewParticle::update_position( void )
242 {
243  // convert 3-point -> PVA format
244  float t1 = m_params.m_Lifetime * m_params.m_MidpointPct * 0.01f;
245  float t2 = m_params.m_Lifetime;
246  Mth::Vector x0,x1,x2,u,a_;
247 
248  x0 = m_params.m_BoxPos[0];
249  x0[3] = m_params.m_Radius[0];
250  x1 = m_params.m_BoxPos[1];
251  x1[3] = m_params.m_Radius[1];
252  x2 = m_params.m_BoxPos[2];
253  x2[3] = m_params.m_Radius[2];
254 
255  if (m_params.m_UseMidpoint)
256  {
257  u = ( t2*t2*(x1 - x0) - t1*t1*(x2 - x0) ) / ( t1*t2*(t2 - t1) );
258  a_ = ( t1*(x2 - x0) - t2*(x1 - x0) ) / ( t1*t2*(t2 - t1) );
259  }
260  else
261  {
262  u = ( x2 - x0 ) / t2;
263  a_.Set(0,0,0,0);
264  }
265 
266  m_systemDmaData.m_p0 = x0 - 1.5f * m_systemDmaData.m_s0;
267  m_systemDmaData.m_p1 = u - 1.5f * m_systemDmaData.m_s1;
268  m_systemDmaData.m_p2 = a_ - 1.5f * m_systemDmaData.m_s2;
269  m_systemDmaData.m_p0[3] = x0[3] - 1.5f * m_systemDmaData.m_s0[3];
270  m_systemDmaData.m_p1[3] = u[3] - 1.5f * m_systemDmaData.m_s1[3];
271  m_systemDmaData.m_p2[3] = a_[3] - 1.5f * m_systemDmaData.m_s2[3];
272 }

297  m_num_streams = 0;

307  m_systemDmaData.m_GScontext[2] = 0x0000000E;
308  m_systemDmaData.m_GScontext[3] = 0x00000000;
309 
310  // ALPHA_1 register
311  uint64 AlphaReg = 0;
312  uint8 fix=64;
313  switch (m_params.m_BlendMode)
314  {
315  case 0x54628ed7: // Blend
316  AlphaReg = PackALPHA(0,1,0,1,0);
317  break;
318  case 0x02e58c18: // Add
319  AlphaReg = PackALPHA(0,2,0,1,0);
320  break;
321  case 0xa7fd7d23: // Sub
322  case 0xdea7e576: // Subtract
323  AlphaReg = PackALPHA(2,0,0,1,0);
324  break;
325  case 0x40f44b8a: // Modulate
326  AlphaReg = PackALPHA(1,2,0,2,0);
327  break;
328  case 0x68e77f40: // Brighten
329  AlphaReg = PackALPHA(1,2,0,1,0);
330  break;
331  case 0x18b98905: // FixBlend
332  AlphaReg = PackALPHA(0,1,2,1,fix);
333  break;
334  case 0xa86285a1: // FixAdd
335  AlphaReg = PackALPHA(0,2,2,1,fix);
336  break;
337  case 0x0d7a749a: // FixSub
338  case 0x0eea99ff: // FixSubtract
339  AlphaReg = PackALPHA(2,0,2,1,fix);
340  break;
341  case 0x90b93703: // FixModulate
342  AlphaReg = PackALPHA(1,2,2,2,fix);
343  break;
344  case 0xb8aa03c9: // FixBrighten
345  AlphaReg = PackALPHA(1,2,2,1,fix);
346  break;
347  case 0x515e298e: // Diffuse
348  case 0x806fff30: // None
349  AlphaReg = PackALPHA(0,0,0,0,0);
350  break;
351  default:
352  Dbg_MsgAssert(0,("Illegal blend mode specified. Please use (fix)blend/add/sub/modulate/brighten or diffuse/none."));
353  break;
354  }
355  m_systemDmaData.m_GScontext[6] = 0x00000042;
356  *(uint64 *)&m_systemDmaData.m_GScontext[4] = AlphaReg;
357 
358  // TEX0_1 and TEX1_1 registers
359  Nx::CTexture *p_texture;
360  Nx::CPs2Texture *p_ps2_texture;
361  mp_engine_texture = NULL;
362  p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( m_params.m_Texture );
363  p_ps2_texture = static_cast<Nx::CPs2Texture*>( p_texture );
364  if ( p_ps2_texture )
365  {
366  mp_engine_texture = p_ps2_texture->GetSingleTexture();
367  }
368  //printf("m_Texture = %08X\n", m_params.m_Texture);
369  //printf("p_texture = %08X\n", p_texture);
370  //printf("p_ps2_texture = %08X\n", p_ps2_texture);
371  //printf("mp_engine_texture = %08X\n", mp_engine_texture);
372 
373  m_systemDmaData.m_GScontext[10] = 0x00000006;
374 
375  m_systemDmaData.m_GScontext[14] = 0x00000014;
376 
377  if (mp_engine_texture)
378  {
379  //printf("\n\nTEX0_1 = %08X%08X\n\n\n",
380  // (uint32)(mp_engine_texture->m_RegTEX0>>32),
381  // (uint32)(mp_engine_texture->m_RegTEX0));
382 
383  // texture coords
384  m_systemDmaData.m_u0 = 8;
385  m_systemDmaData.m_v0 = (mp_engine_texture->GetHeight()<<4) - 8;
386  m_systemDmaData.m_u1 = (mp_engine_texture->GetWidth() <<4) - 8;
387  m_systemDmaData.m_v1 = 8;
388  }
389 
390  // TEST_1
391  m_systemDmaData.m_GScontext[18] = 0x00000047;
392  m_systemDmaData.m_GScontext[16] = 0x0005000B | m_params.m_AlphaCutoff<<4;
393 
394  // FOG
395  m_systemDmaData.m_GScontext[22] = 0x0000000A;
396 
397  // giftag for particles
398  m_systemDmaData.m_tagy = 0x60BB4000;
399  m_systemDmaData.m_tagz = 0x00535312;
400 
401 
402 
403  // have already set m_Lifetime, which is called m_life in newflat
404 
405  // x-component will be overwritten by vu1 code, so can store midtime there
406  *(float *)&m_systemDmaData.m_tagx = m_params.m_UseMidcolor ?
407  m_params.m_Lifetime * m_params.m_ColorMidpointPct * 0.01f :
408  0.0f;
409 
410  // and now a load of redundant duplication of data, which should later be removed...
411 
412  // convert 3-point -> PVA format
413  float t1 = m_params.m_Lifetime * m_params.m_MidpointPct * 0.01f;
414  float t2 = m_params.m_Lifetime;
415  Mth::Vector x0,x1,x2,u,a_;
416 
417  //printf("\n\n\n\n\n%g,%g,%g\n%g,%g,%g\n\n\n\n\n",
418  // m_params.m_BoxDims[0][3], m_params.m_BoxDims[1][3], m_params.m_BoxDims[2][3],

422  x0[3] = m_params.m_RadiusSpread[0];
423  x1 = m_params.m_BoxDims[1];
424  x1[3] = m_params.m_RadiusSpread[1];
425  x2 = m_params.m_BoxDims[2];
426  x2[3] = m_params.m_RadiusSpread[2];
427 
428  if (m_params.m_UseMidpoint)
429  {
430  u = ( t2*t2*(x1 - x0) - t1*t1*(x2 - x0) ) / ( t1*t2*(t2 - t1) );

445

481  }

487  m_rotation.Identity();
488
