206 // returns the number of leaf nodes
207 int CGeomNode::ProcessNodeInPlace(uint8 *p_baseAddress)
208 {
209  int leaves = 0;
210 
211  if (!(m_flags & NODEFLAG_LEAF))
212  {
213  leaves++;
214  // convert transform offset to pointer
215  if ((int)u2.mp_transform != -1)
216  {
217  u2.mp_transform = (Mth::Matrix *)( (int)u2.mp_transform + p_baseAddress );
218  }
219  else
220  {
221  u2.mp_transform = (Mth::Matrix *)0;
222  }
223  }
224 
225  uint32 tex = GetTextureChecksum();
226  SetTextureChecksum(0); // clear it in case we don't find it
227 
228  // convert group checksum to pointer
229  if ((m_flags & NODEFLAG_LEAF) && u1.mp_dma)
230  {
231  sGroup *pGroup;
232  for (pGroup=sGroup::pHead; pGroup; pGroup=pGroup->pNext)
233  if (pGroup->Checksum == (uint)u3.mp_group)
234  break;
235  Dbg_MsgAssert(pGroup, ("Couldn't group checksum #%d\n", (uint)u3.mp_group));
236  u3.mp_group = pGroup;
237 
238  // Resolve the group+texture checksum into a texture pointer
239 // printf ("%x\n",GetTextureChecksum());
240  if (tex)
241  {
242  sGroup * p_group = GetGroup();
243  sScene * p_scene = p_group->pScene;
244  sTexture *p_textures = p_scene->pTextures;
245  int num_textures = p_scene->NumTextures;
246  for (int i=0;i<num_textures;i++)
247  {
248  // We are seraching through all the textures in the scene
249  // so we will find multiple instances with the same texture checksum
250  // so we need to check the group checksum as well
251  if (p_textures[i].Checksum == tex && p_textures[i].GroupChecksum == GetGroup()->Checksum)
252  {
253  //printf ("Found %dx%dx%d %x mips at 0x%x (%d)\n",p_textures[i].GetWidth(),p_textures[i].GetHeight(),p_textures[i].GetClutBitdepth(),p_textures[i].GetNumMipmaps(),*(p_textures+i),i);
254  SetTextureChecksum((uint32)(p_textures+i));
255  break;
256  }
257  }
258 // printf ("-> 0x%x\n",GetTextureChecksum());
259  }
260 
261  }
262 
263  if (m_flags & NODEFLAG_LEAF)
264  {
265  // convert dma offset to pointer
266  if ((int)u1.mp_dma != -1)
267  {
268  u1.mp_dma = (uint)u1.mp_dma + p_baseAddress;
269  }
270  else
271  {
272  u1.mp_dma = (uint8 *)0;
273  }
274  }
275  else
276  {
277  // convert child offset to pointer
278  if ((int)u1.mp_child != -1)
279  {
280  u1.mp_child = (CGeomNode *)( (int)u1.mp_child + p_baseAddress );
281  }
282  else
283  {
284  u1.mp_child = (CGeomNode *)0;
285  }
286  }
287 
288  // convert sibling offset to pointer
289  if ((int)mp_sibling != -1)
290  {
291  mp_sibling = (CGeomNode *)( (int)mp_sibling + p_baseAddress );
292  }
293  else
294  {
295  mp_sibling = (CGeomNode *)0;
296  }
297 
298  // convert LOD offset to pointer
299  if ((int)mp_next_LOD != -1)
300  {
301  mp_next_LOD = (CGeomNode *)( (int)mp_next_LOD + p_baseAddress );
302  }
303  else
304  {
305  mp_next_LOD = (CGeomNode *)0;
306  }
307 
308  // convert uv wibble offset to pointer
309  if (m_flags & NODEFLAG_UVWIBBLE)
310  {
311  if (!(m_flags & NODEFLAG_ENVMAPPED) || !(m_flags & NODEFLAG_LEAF))
312  {
313  Dbg_MsgAssert ((int)mp_uv_wibble!=-1, ("Null uv wibble pointer on a uv wibbled node"));
314  mp_uv_wibble = (float *)( (int)mp_uv_wibble + p_baseAddress );
315  Dbg_MsgAssert (!(((uint32)mp_uv_wibble) & 0xf0000003),("Corrupt UV wibble offset (0x%x) in node %s\n",((int)mp_uv_wibble - (int)p_baseAddress), Script::FindChecksumName(m_checksum)));
316  }
317  else
318  {
319  mp_uv_wibble = (float *)0;
320  m_flags &= ~NODEFLAG_UVWIBBLE;
321  }
322  }
323 
324  // convert vc wibble offset to pointer
325  if ((int)mp_vc_wibble != -1)
326  {
327  // first convert the wibble offset itself
328  mp_vc_wibble = (uint32 *)( (int)mp_vc_wibble + p_baseAddress );
329 
330  // now the mesh info
331  uint32 *p_meshdata = mp_vc_wibble;
332  int seq, num_seqs, vert, num_verts;
333 
334  num_seqs = *p_meshdata++;
335  for (seq=0; seq<num_seqs; seq++)
336  {
337  // convert sequence pointer
338  *(uint32 **)p_meshdata = (uint32 *) ( (int)*p_meshdata + p_baseAddress );
339  p_meshdata++;
340 
341  // convert rgba pointers
342  num_verts = *p_meshdata++;
343  for (vert=0; vert<num_verts; vert++)
344  {
345  *(uint32 **)p_meshdata = (uint32 *) ( (uint)*p_meshdata + p_baseAddress );
346  p_meshdata++;
347  }
348  }
349  }
350  else
351  {
352  mp_vc_wibble = (uint32 *)0;
353  }
354 
355  // recursively process any children
356  if (!(m_flags & NODEFLAG_LEAF))
357  {
358  for (CGeomNode *p_child=u1.mp_child; p_child; p_child=p_child->mp_sibling)
359  {
360  leaves += p_child->ProcessNodeInPlace(p_baseAddress);
361  }
362  }
363 
364  // recursively process any LODs
365  if (mp_next_LOD)
366  {
367  (void) mp_next_LOD->ProcessNodeInPlace(p_baseAddress);
368  }
369 
370  return leaves;
371 }

601 void CGeomNode::RenderScene(CVu1Context& ctxt, uint32 renderFlags)
602 {
603  CVu1Context *p_ctxt;
604 
605 #ifdef __NOPT_ASSERT__
606  m_flags &= ~NODEFLAG_WAS_RENDERED; // Clear flag saying it was rendered
607 #endif
608 
609  GEOMSTATINC(geom_stats_total);
610 
611 
612  // cull the node if not active
613  if (!((m_flags & NODEFLAG_ACTIVE) && (m_flags & render::ViewportMask)))
614  {
615  GEOMSTATINC(geom_stats_inactive);
616 #ifdef DUMP_NODE_STRUCTURE
617  node_level--;
618 #endif
619  return;
620  }
621 
622  // is it the root node of a sky?
623  if (m_flags & NODEFLAG_SKY)
624  {
625  GEOMSTATINC(geom_stats_sky);
626  // reserve space from dma list for new trans context
627  p_ctxt = ctxt.Localise();
628 
629  p_ctxt->SetupAsSky();
630  RenderAsSky(*p_ctxt, renderFlags);
631  dma::SetList(NULL); // not quite sure why we need this, but we do
632 #ifdef DUMP_NODE_STRUCTURE
633  node_level--;
634 #endif
635  return;
636  }
637 
638  // is it really an object node in disguise?
639  if (m_flags & NODEFLAG_OBJECT)
640  {
641  RenderObject(ctxt, renderFlags);
642  return;
643  }
644 
645  // get this node's transform context
646  if (m_flags & NODEFLAG_TRANSFORMED)
647  {
648  // copy the parent vu1 context
649  p_ctxt = ctxt.LocaliseExtend();
650 
651  // set up the parts of the context that will depend on the new matrix
652  p_ctxt->StandardSetup(*u2.mp_transform);
653 
654  // Propogate the 'transformed' flag when recursing.
655  renderFlags |= RENDERFLAG_TRANSFORMED;
656  }
657  else
658  {
659  p_ctxt = &ctxt;
660  }
661 
662  // make a note of the root node for a skeletal model
663  if (m_flags & NODEFLAG_SKELETAL)
664  {
665  GpInstanceNode = this;
666  renderFlags |= RENDERFLAG_SKELETAL;
667  }
668 
669 
670  // applying a colour to a node
671  if (m_flags & NODEFLAG_COLOURED)
672  {
673  GEOMSTATINC(geom_stats_colored);
674  // if there's no local context, copy the parent's
675  if (p_ctxt == &ctxt)
676  {
677  p_ctxt = ctxt.Localise();
678  }
679 
680  // copy node's colour to vu1 context
681  p_ctxt->SetColour(this);
682  renderFlags |= RENDERFLAG_COLOURED;
683  }
684 
685  // Check for light group
686  if (u3.mp_light_group && u3.mp_light_group != ctxt.GetLights())
687  {
688  // if there's no local context, copy the global one
689  if (p_ctxt == &ctxt || !p_ctxt->IsExtended())
690  {
691  p_ctxt = ctxt.LocaliseExtend();
692  }
693 
694  p_ctxt->SetLights(u3.mp_light_group);
695  }
696 
697 
698 // If we got here, then it's been rendered, so set the flag
699 #ifdef __NOPT_ASSERT__
700  m_flags |= NODEFLAG_WAS_RENDERED; // Set flag saying it was rendered
701 #endif
702 
703  // render any children
704  for (CGeomNode *p_child=u1.mp_child; p_child; p_child=p_child->mp_sibling)
705  {
706  p_child->RenderObject(*p_ctxt, renderFlags);
707  }
708 
709 #ifdef DUMP_NODE_STRUCTURE
710  node_level--;
711 #endif
712 }
