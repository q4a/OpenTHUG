162 void CGameFlow::Update()
163 {
164  if (mp_tester_script)
165  {
166  mp_tester_script->Update();
167  }
168 
169 
170  // GJ: This keeps us from loading a new script
171  // while we're still in the middle of processing the old one
172  if ( m_requestNewScript )
173  {
174  m_requestNewScript = false;
175 
176  if ( mp_gameFlowScript )
177  delete mp_gameFlowScript;
178 
179  mp_gameFlowScript = new Script::CScript;
180  mp_gameFlowScript->SetScript( m_requestedScript );
181  #ifdef __NOPT_ASSERT__
182  mp_gameFlowScript->SetCommentString("Created in CGameFlow::Update()");
183  #endif
184 
185 #ifdef __USER_GARY__
186  printf("Resetting to script %s %08x!\n", Script::FindChecksumName( m_requestedScript ), (int)mp_gameFlowScript);
187 #endif
188  }
189 
190  // if paused, then don't continue the script
191  if ( m_paused )
192  return;
193 
194  if ( mp_gameFlowScript )
195  {
196  if ( mp_gameFlowScript->Update() == Script::ESCRIPTRETURNVAL_FINISHED )
197  {
198  delete mp_gameFlowScript;
199  mp_gameFlowScript = NULL;
200  }
201 
202  }
