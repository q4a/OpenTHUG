463 /******************************************************************/
464 /* */
465 /* */
466 /******************************************************************/
467 // This gets called every frame from CEngine::sRenderWorld()
468 // It calls sFakeLight for each of the nodes in the array sp_fake_lights_nodes, but
469 // staggers the calls across frames so that they don't cause a CPU time glitch.
470 // Once sFakeLight ahs been called for all of them it calls sClearCurrentFakeLightsCommand()
471 void CLightManager::sUpdateVCLights()
472 {
473  if (!sp_fake_lights_nodes)
474  {
475  return;
476  }
477 
478  // TODO: At some point all the calls to GamePaused need to be removed from
479  // the logic functions that are currently called from CEngine::sRenderWorld()
480  if (Mdl::FrontEnd::Instance()->GamePaused())
481  {
482  return;
483  }
484 
485  // If finished, clean up & stop.
486  if (s_fake_lights_current_time > s_fake_lights_period)
487  {
488  sClearCurrentFakeLightsCommand();
489  return;
490  }
491 
492  // Calculate how many nodes ought to have been processed by now, based on the ratio of s_fake_lights_current_time
493  // to s_fake_lights_period
494  int required_num_nodes_processed;
495  if (s_fake_lights_period)
496  {
497  required_num_nodes_processed = (s_num_fake_lights_nodes * s_fake_lights_current_time) / s_fake_lights_period;
498  }
499  else
500  {
501  required_num_nodes_processed = s_num_fake_lights_nodes;
502  }
503 
504  // Call sFakeLight for each node starting from where it got to last time, until the required number have
505  // been processed.
506  for (; s_next_fake_light_node_to_process<required_num_nodes_processed; ++s_next_fake_light_node_to_process)
507  {
508  Script::CStruct *p_node=SkateScript::GetNode(sp_fake_lights_nodes[s_next_fake_light_node_to_process]);
509  uint32 id=0;
510  p_node->GetChecksum(CRCD(0xa1dc81f9,"Name"),&id);
511  sFakeLight(id,sp_fake_lights_params);
512  }
513 
514  ++s_fake_lights_current_time;
515 }

977  if( p_geom == NULL )
978  {
979  return;
980  }
981 
982  const int MAX_LIGHTS_AFFECTING_SECTOR = 8;
983 
984  int current_lights_affecting_sector = 0;
985  int light_index[MAX_LIGHTS_AFFECTING_SECTOR];

988  for( int l = 0; l < s_num_vc_lights; ++l )
989  {
990  if(( sp_vc_lights[l].m_intensity > 0.0f ) && sp_vc_lights[l].AffectsSector( p_sector ))

994  {
995  break;
996  }
997  }
998  }
999 
1000  if( current_lights_affecting_sector == 0 )
1001  {
1002  // No lights are affecting this sector, which can happen in the situation where the last remaining
1003  // light affecting a sector is turned off. In this case we just want to restore the original
1004  // vertex colors.
1005  int verts = p_geom->GetNumRenderVerts();
1006  if( verts > 0 )

1009  p_geom->GetOrigRenderColors( p_colors );
1010  p_geom->SetRenderColors( p_colors );
1011  delete [] p_colors;
1012  }
1013  }
1014  else if( current_lights_affecting_sector > 0 )
1015  {
1016  int verts = p_geom->GetNumRenderVerts();
1017  if( verts > 0 )
1018  {
1019  Mth::Vector *p_verts = new Mth::Vector[verts];
1020  Image::RGBA *p_colors = new Image::RGBA[verts];
1021 
1022  p_geom->GetRenderVerts( p_verts );
1023  p_geom->GetOrigRenderColors( p_colors );
1024 
1025  Image::RGBA *p_color = p_colors;
1026  Mth::Vector *p_vert = p_verts;
1027  for( int i = 0; i < verts; ++i )
1028  {
1029  float ir = 0.0f;
1030  float ig = 0.0f;
1031  float ib = 0.0f;
1032 
1033  for( int l = 0; l < current_lights_affecting_sector; ++l )
1034  {
1035  SVCLight *p_current_light = &sp_vc_lights[light_index[l]];
1036 
1037  float dist_sqr = ( *p_vert - p_current_light->m_pos ).LengthSqr();
1038  if( dist_sqr < ( p_current_light->m_radius * p_current_light->m_radius ))
1039  {
1040  float dist = sqrtf( dist_sqr );
1041  float intensity = p_current_light->m_intensity * (( p_current_light->m_radius - dist ) / p_current_light->m_radius );
1042 
1043  ir += intensity * (float)p_current_light->m_color.r;
1044  ig += intensity * (float)p_current_light->m_color.g;
1045  ib += intensity * (float)p_current_light->m_color.b;
1046  }
1047  }
1048 
1049  // Apply to each r,g,b idividually, so we don't loose original color.
1050  Image::RGBA rgb = *p_color;
1051  float v = (float)rgb.r + ir;
1052  rgb.r = (uint8)(( v > 255.0f ) ? 255.0f : v );
1053 
1054  v = (float)rgb.g + ig;
1055  rgb.g = (uint8)(( v > 255.0f ) ? 255.0f : v );
1056 
1057  v = (float)rgb.b + ib;
1058  rgb.b = (uint8)(( v > 255.0f ) ? 255.0f : v );
1059 
1060  *p_color = rgb;
1061 
1062  ++p_color;
1063  ++p_vert;
1064  }
1065 
1066  p_geom->SetRenderColors( p_colors );
1067 
1068  delete [] p_verts;
1069  delete [] p_colors;
1070  }
1071  }
1072 }
