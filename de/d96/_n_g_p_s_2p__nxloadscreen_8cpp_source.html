51 void CLoadScreen::s_plat_display(const char* filename, bool just_freeze, bool blank)
52 {
53  // Engine stuff
54  Dbg_Assert(!sp_load_screen_texture);
55  Dbg_Assert(!sp_load_screen_sprite);
56 
57  char adjustedFilename[144];
58 
59  if (!just_freeze && !blank)
60  {
61  sp_load_screen_texture = new CPs2Texture(true);
62 
63  if (Config::NTSC())
64  {
65  strcpy(adjustedFilename, filename);
66  } else {
67  // Find correct directory to use based on language
68  switch (Config::GetLanguage())
69  {
70  case Config::LANGUAGE_FRENCH:
71  sprintf(adjustedFilename, "PALimages/FRImages/%s", filename);
72  break;
73  case Config::LANGUAGE_SPANISH:
74  sprintf(adjustedFilename, "PALimages/SPImages/%s", filename);
75  break;
76  case Config::LANGUAGE_GERMAN:
77  sprintf(adjustedFilename, "PALimages/GRImages/%s", filename);
78  break;
79  case Config::LANGUAGE_ITALIAN:
80  sprintf(adjustedFilename, "PALimages/ITImages/%s", filename);
81  break;
82  default:
83  sprintf(adjustedFilename, "PALimages/%s", filename);
84  break;
85  }
86 
87  // Since LoadTexture() will assert if the file doesn't exist, we need to test ahead of time
88  char ExtendedFilename[160];
89  sprintf(ExtendedFilename, "images/%s.img.ps2", adjustedFilename);
90 
91  if (!File::Exist(ExtendedFilename))
92  {
93  // Can't find it, so lets try the base directory before giving up
94  sprintf(adjustedFilename, "PALimages/%s", filename);
95  }
96  }
97 
98  if (!sp_load_screen_texture->LoadTexture(adjustedFilename, true))
99  {
100  Dbg_Error("Can't load texture %s", adjustedFilename);
101  }
102 
103  #if USE_SPRITES
104  sp_load_screen_sprite = static_cast<Nx::CPs2Sprite *>(CEngine::sCreateSprite());
105  sp_load_screen_sprite->SetTexture(sp_load_screen_texture);
106  sp_load_screen_sprite->SetPos(0, 0);
107  sp_load_screen_sprite->SetScale(1, 1);
108  sp_load_screen_sprite->SetRGBA(Image::RGBA(128, 128, 128, 128));
109  sp_load_screen_sprite->SetHidden(false);
110  #else
111  // Copy directly into display buffer
112  sceGsLoadImage gs_image;
113  NxPs2::SSingleTexture *p_engine_texture = sp_load_screen_texture->GetSingleTexture();
114 
115  FlushCache( 0 );
116  sceGsSyncPath( 0, 0 );
117 
118  int display_addr = (SCREEN_CONV_X( 640 ) * SCREEN_CONV_Y( 448 ) * 4) / (64 * 4);
119 
120  const int download_parts = 2;
121  int partial_height = sp_load_screen_texture->GetHeight() / download_parts;
122  int partial_size = sp_load_screen_texture->GetWidth() * partial_height * 2;
123  for (int i = 0; i < download_parts; i++)
124  {
125  sceGsSetDefLoadImage( &gs_image, display_addr, SCREEN_CONV_X( 640 )/64, SCE_GS_PSMCT16S,
126  0, partial_height * i, sp_load_screen_texture->GetWidth(), partial_height );
127 
128  FlushCache( 0 );
129  sceGsExecLoadImage( &gs_image, ( u_long128 * )(p_engine_texture->mp_PixelData + (partial_size * i)) );
130  sceGsSyncPath( 0, 0 );
131  }
132 
133  // Done with texture
134  delete sp_load_screen_texture;
135  sp_load_screen_texture = NULL;
136  }
137 
138  if (blank)
139  {
140  // This is esentially the same as playing a movie, without playing a movie
141  // the ResetEngine handles clearing the screen
142  NxPs2::SuspendEngine();
143  NxPs2::ResetEngine();
144  }
145 
146 
147 
148  NxPs2::EnableFlipCopy(false);
149 #endif // USE_SPRITES
150 }
