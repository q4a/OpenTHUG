194 BOOL _DVDReadAsync( DVDFileInfo* fileInfo, void* addr, s32 length, s32 offset, DVDCallback callback )
195 {
196  // See if this is a repeat read.
197  bool repeat_read = false;
198  for ( int lp = 0; lp < 8; lp++ )
199  {
200  if ( ( fileInfo == last_fileInfo[lp] ) &&
201  ( addr == last_addr[lp] ) &&
202  ( length == last_length[lp] ) &&
203  ( offset == last_offset[lp] ) )
204  {
205  repeat_read = true;
206  break;
207  }
208  }
209 
210  if ( repeat_read )
211  {
212  // Find an empty slot for this callback.
213  int cb = -1;
214  for ( int lp = 0; lp < 8; lp++ )
215  {
216  if ( last_callback_counter[lp] == -1 )
217  {
218  cb = lp;
219  break;
220  }
221  }
222 
223  if ( cb == -1 )
224  {
225 # ifdef __NOPT_FINAL__
226  OSReport( "Warning: Overflowed DVD callback buffer\n" );
227 # else
228  Dbg_MsgAssert( 0, ( "Overflowed DVD callback buffer" ));
229 # endif
230  // Issue read normally in this case.
231  DVDReadAsync( fileInfo, addr, length, offset, callback );
232  }
233  last_callback[cb] = callback;
234  last_callback_length[cb] = length;
235  last_callback_fileInfo[cb] = fileInfo;
236  last_callback_counter[cb] = 8;
237 
238  return TRUE;
239  }
240  else
241  {
242  // Shunt last reads down & fill in new one.
243  for ( int lp = 7; lp > 0; lp-- )
244  {
245  last_fileInfo[lp] = last_fileInfo[lp-1];
246  last_addr[lp] = last_addr[lp-1];
247  last_length[lp] = last_length[lp-1];
248  last_offset[lp] = last_offset[lp-1];
249  }
250  last_fileInfo[0] = fileInfo;
251  last_addr[0] = addr;
252  last_length[0] = length;
253  last_offset[0] = offset;
254 
255  return DVDReadAsync( fileInfo, addr, length, offset, callback );
256  }

267 
268  if( Sfx::isStereo )

270  if(( volL == 0.0f ) && ( volR == 0.0f ))

272  // Dave note - in theory setting the mixerCtrl to zero should cause the sound not to play.
273  // In practice however it seems to lead to small audio 'pops'.
274 // p_pb->mixerCtrl = 0;

285  // Get the length of the vector here which will be used to multiply out the normalised speaker volumes.
286  Mth::Vector test( fabsf( volL ), fabsf( volR ), 0.0f, 0.0f );
287  float amplitude = test.Length();
288 
289  // Look just at the normalized right component to figure the sound angle from Matt's calculations.
290  test.Normalize();
291 
292  float angle;
293  angle = asinf( test[Y] );
294  angle = ( angle * 2.0f ) - ( Mth::PI * 0.5f );
295  angle = ( volL < 0.0f ) ? ( Mth::PI - angle ) : angle;
296 
297  // Now figure volumes based on speaker coverage.
298  angle = Mth::RadToDeg( angle );
299 
300  // Left front channel.
301  if(( angle >= 225.0f ) || ( angle <= 45.0f ))
302  {
303  // Because of the discontinuity, shift this angle round to the [0,180] range.
304  float shift_angle = angle + 135;
305  shift_angle = ( shift_angle >= 360.0f ) ? ( shift_angle - 360.0f ) : shift_angle;
306  volumes[0] = ( shift_angle / 180.0f ) * Mth::PI;
307  volumes[0] = sqrtf( sinf( volumes[0] ));
308  }
309 
310 // // Center channel.
311 // if(( angle >= -60.0f ) && ( angle <= 60.0f ))
312 // {
313 // // Scale this into [0,PI] space so we can get smooth fadeout.
314 // volumes[1] = (( angle + 60.0f ) / 120.0f ) * Mth::PI;

319  if(( angle >= -45.0f ) && ( angle <= 135.0f ))
320  {
321  // Scale this into [0,PI] space so we can get smooth fadeout.
322  volumes[1] = (( angle + 45.0f ) / 180.0f ) * Mth::PI;
323  volumes[1] = sqrtf( sinf( volumes[1] ));
324  }
325 
326  // Right rear channel.
327  if(( angle >= 45.0f ) && ( angle <= 225.0f ))
328  {
329  // Scale this into [0,PI] space so we can get smooth fadeout.
330  volumes[2] = (( angle - 45.0f ) / 180.0f ) * Mth::PI;
331  volumes[2] = sqrtf( sinf( volumes[2] ));
332  }
333 
334  // Left rear channel.
335  if(( angle >= 135.0f ) || ( angle <= -45.0f ))

338  float shift_angle = angle + 225;

342  }
343

347  // Scale back up to original amplitude.

453  p_vpb->sync |= AX_SYNC_USER_MIX | AX_SYNC_USER_MIXCTRL;
454  }

459  p_pb->src.ratioLo = ( pitch & 0xFFFF );
460  p_vpb->sync |= AX_SYNC_USER_SRCRATIO;
461  }

509 static void s_start_voice_callback( u32 pointerToARQRequest )
510 {
511  // Figure which stream this is for.
512  int stream = -1;
513 
514  for( int s = 0; s < NUM_STREAMS; ++s )
515  {
516  if( pointerToARQRequest == (u32)( &stream_arq_request[s] ))
517  {
518  stream = s;
519  break;
520  }
521  }
522 
523  if( stream == -1 )
524  {
525  // Something has gone very wrong.
526  Dbg_MsgAssert( 0, ( "Unsynced dvd callback" ));
527  return;
528  }
529 
530  // Deal with the voice already having been deleted.
531  if( gStreamInfo[stream].p_voice == NULL )
532  {
533  return;
534  }
535 
536  // Set the voice to run.
537  gStreamInfo[stream].hasStartedPlaying = true;
538  gStreamInfo[stream].paused = streamsPaused;
539  if( !streamsPaused && !gStreamInfo[stream].preloadActive )
540  {
541  if( gStreamInfo[stream].p_voice == NULL )
542  {
543  Dbg_Assert( 0 );
544  return;
545  }
546  AXSetVoiceState( gStreamInfo[stream].p_voice, AX_PB_STATE_RUN );
547  }
548 
549  // Adjust pitch to account for lower sampling rates:
550  float pitch = PERCENT( gStreamInfo[stream].pitch, gStreamInfo[stream].pitchAdjustmentForSamplingRate );
551  unsigned int u_pitch = (unsigned int)PERCENT( UNALTERED_PITCH, pitch );
552 
553  int old = OSDisableInterrupts();

557  OSRestoreInterrupts( old );
558 
559  gPcmStatus |= PCMSTATUS_STREAM_PLAYING( stream );
560  gPcmStatus &= ~PCMSTATUS_LOAD_STREAM( stream );
561 }

569 static void s_dvd_callback( s32 result, DVDFileInfo* fileInfo )
570 {
571  // Figure which stream this is for.
572  int stream = -1;
573 
574  for( unsigned int i = 0; i < 8; ++i )
575  {
576  for( uint32 s = 0; s < NUM_STREAMS; ++s )
577  {
578  if( fileInfo == &( gStreamInfo[s].fileInfo[i] ))
579  {
580  if( i == gStreamInfo[s].currentFileInfo )
581  {
582  stream = s;
583  break;
584  }
585  }
586  }
587  if( stream >= 0 )
588  break;
589  }
590 
591  if( stream == -1 )
592  {
593  // Possibly the result of a read in which we are no longer interested.
594  return;
595  }
596 
597  // Handle read errors by just stopping the stream.
598  if(( result < 0 ) || gStreamInfo[stream].has_paused )
599  {
600  int enabled = OSDisableInterrupts();
601  if( gStreamInfo[stream].p_voice )
602  {
603  // This should be the case.
604  AXFreeVoice( gStreamInfo[stream].p_voice );
605  }
606  gStreamInfo[stream].p_voice = NULL;
607  gStreamInfo[stream].preloadActive = false;
608  gStreamInfo[stream].currentFileInfo = ( gStreamInfo[stream].currentFileInfo + 1 ) & 0x07;
609  OSRestoreInterrupts( enabled );
610 
611  gPcmStatus &= ~PCMSTATUS_LOAD_STREAM( stream );
612  gPcmStatus &= ~PCMSTATUS_STREAM_PLAYING( stream );
613  return;
614  }
615 
616  // Check if has already been stopped.
617  AXVPB* p_axvpb = gStreamInfo[stream].p_voice;
618  if( p_axvpb == NULL )
619  {
620  return;
621  }
622 
623  // If this is the first chunk read for this voice, need to grab the voice and look at the header information.
624  if( gStreamInfo[stream].offset == 0 )
625  {
626  AXPBADDR addr;
627  AXPBADPCM adpcm;
628  AXPBVE ve;
629 
630  int enabled = OSDisableInterrupts();
631  // Grab pointer to header (first 96 byes of data).
632  sDSPADPCM* p_header = (sDSPADPCM*)( stream_mem_buffer[stream] );
633 
634  addr.format = AX_PB_FORMAT_ADPCM;
635 
636  // Depending on the number of bytes in the sound, we may not need to loop.
637  // Also depending on how many times we will be uploading to ARAM, we may want to start the stream
638  // halfway through the ARAM buffer, in order to ensure that we always write the last block to the upper half of ARAM
639  // (in order that we can safely set the end position).
640  int num_adpcm_bytes = p_header->num_adpcm_nibbles / 2;
641 
642  // Convert from byte to nibble address, and split into low and high words.
643  u32 start_address = ( stream_base[stream] * 2 ) + 2;
644  addr.currentAddressHi = (u16)( start_address >> 16 );
645  addr.currentAddressLo = (u16)( start_address & 0xFFFFU );
646 
647  if( num_adpcm_bytes <= 32768 )
648  {
649  // Stream fits in to ARAM buffer completely. No need to loop.
650  u32 end_address = ( stream_base[stream] * 2 ) + p_header->num_adpcm_nibbles - 1;
651 
652  addr.endAddressHi = (u16)( end_address >> 16 );
653  addr.endAddressLo = (u16)( end_address & 0xFFFFU );
654 
655  u32 zero_buffer_address = NxNgc::EngineGlobals.aram_zero * 2;
656  addr.loopAddressHi = (u16)( zero_buffer_address >> 16 );
657  addr.loopAddressLo = (u16)( zero_buffer_address & 0xFFFFU );
658 
659  addr.loopFlag = AXPBADDR_LOOP_OFF;
660 
661  gStreamInfo[stream].nextWrite = 1;
662 // OSReport( "Stream: %d is simple <= 32768 bytes\n", stream );
663  }
664  else
665  {
666  // Stream does not fit in to ARAM buffer completely. Needs to loop.
667  uint32 num_uploads = num_adpcm_bytes / STREAM_BUFFER_SIZE;
668  if(( num_adpcm_bytes % STREAM_BUFFER_SIZE ) > 0 )
669  ++num_uploads;
670  Dbg_Assert( num_uploads > 2 );
671 
672  addr.loopAddressHi = addr.currentAddressHi;
673  addr.loopAddressLo = addr.currentAddressLo;
674 
675  if( num_uploads & 0x01 )
676  {
677  // This is the tricky situation, where we start halfway through the ARAM buffer.
678  gStreamInfo[stream].offsetInARAM += STREAM_BUFFER_SIZE;
679  start_address += ( STREAM_BUFFER_SIZE * 2 );
680  addr.currentAddressHi = (u16)( start_address >> 16 );
681  addr.currentAddressLo = (u16)( start_address & 0xFFFFU );
682  gStreamInfo[stream].nextWrite = 0;
683 // OSReport( "Stream: %d is complex tricky %d bytes\n", stream, num_adpcm_bytes );
684 
685  }
686  else
687  {
688  // This is the easy situation, where we start at the start of the ARAM buffer.
689  gStreamInfo[stream].offsetInARAM = 0;
690  gStreamInfo[stream].nextWrite = 1;
691 // OSReport( "Stream: %d is complex non-tricky %d bytes\n", stream, num_adpcm_bytes );
692  }
693 
694  u32 end_address = ( stream_base[stream] * 2 ) + ( 32768 * 2 ) - 1;
695  addr.endAddressHi = (u16)( end_address >> 16 );
696  addr.endAddressLo = (u16)( end_address & 0xFFFFU );
697 
698  addr.loopFlag = AXPBADDR_LOOP_ON;
699  }
700 
701  memcpy( &adpcm.a[0][0], p_header->coef, sizeof( u16 ) * 16 );
702  adpcm.gain = p_header->gain;
703  adpcm.pred_scale = p_header->ps;
704  adpcm.yn1 = p_header->yn1;
705  adpcm.yn2 = p_header->yn2;
706 
707  ve.currentVolume = 0x7FFF;
708  ve.currentDelta = 0;
709 
710  // Set voice parameters.
711  AXSetVoiceType( p_axvpb, AX_PB_TYPE_NORMAL ); // No loop context.
712  AXSetVoiceSrcType( p_axvpb, AX_PB_SRCSEL_LINEAR ); // Use linear interpolation.
713  AXSetVoiceVe( p_axvpb, &ve ); // Set overall volume.
714  AXSetVoiceAddr( p_axvpb, &addr ); // Input addressing.
715  AXSetVoiceAdpcm( p_axvpb, &adpcm ); // ADPCM coefficients.
716 
717 // p_axvpb->sync |= AX_SYNC_USER_ALLPARAMS;
718 // AXSetVoiceSrcRatio( p_axvpb, (48000.0f / ((float)AX_IN_SAMPLES_PER_SEC)) );
719 
720  // If this chunk is less than buffersize, zero out the excess bytes to prevent sound pop.
721  if( result < ( STREAM_BUFFER_SIZE + STREAM_BUFFER_HEADER_SIZE ))
722  {
723  memset( stream_mem_buffer[stream] + result, 0, ( STREAM_BUFFER_SIZE + STREAM_BUFFER_HEADER_SIZE ) - result );
724  DCFlushRange( stream_mem_buffer[stream] + result, ( STREAM_BUFFER_SIZE + STREAM_BUFFER_HEADER_SIZE ) - result );
725  }
726  OSRestoreInterrupts( enabled );
727 
728  // DMA this chunk up to ARAM, then use the ARQ callback to start the voice.
729  ARQPostRequest( &stream_arq_request[stream], // Address to user-allocated storage for an ARQTask data structure.
730  0, // User-enumerated identifier for the owner of the ARAM transaction.
731  ARQ_TYPE_MRAM_TO_ARAM, // Direction.
732  ARQ_PRIORITY_HIGH, // Priority.
733  (u32)( stream_mem_buffer[stream] + 96 ), // Source address.
734  stream_base[stream] + gStreamInfo[stream].offsetInARAM, // Destination address.
735  STREAM_BUFFER_SIZE, // Transfer length (bytes). Must be multiple of 4.
736  &s_start_voice_callback ); // DMA complete callback function.
737 
738  gStreamInfo[stream].pitchAdjustmentForSamplingRate = ((float)p_header->sample_rate * 100.0f ) / 32000.0f;
739 
740  // Increment the ARAM offset and wrap when appropriate.
741  gStreamInfo[stream].offsetInARAM += STREAM_BUFFER_SIZE;
742  if( gStreamInfo[stream].offsetInARAM >= ( STREAM_BUFFER_SIZE * 2 ))
743  {
744  gStreamInfo[stream].offsetInARAM -= ( STREAM_BUFFER_SIZE * 2 );
745  }
746 
747  // Increment the file pointer (remember we read the extra buffer bytes for the first read).
748  gStreamInfo[stream].offset += STREAM_BUFFER_SIZE + STREAM_BUFFER_HEADER_SIZE;
749  }
750  else
751  {
752  // If this chunk is less than buffersize, zero out the excess bytes to prevent sound pop.
753  if( result < STREAM_BUFFER_SIZE )
754  {
755  memset( stream_mem_buffer[stream] + result, 0, STREAM_BUFFER_SIZE - result );
756  DCFlushRange( stream_mem_buffer[stream] + result, STREAM_BUFFER_SIZE - result );
757  }
758 
759  // We want to DMA this chunk up to ARAM, but only if safe to do so, i.e. only when the playback position is in
760  // the opposite half of the ARAM buffer.
761  int current_aram_pos = ((uint32)( p_axvpb->pb.addr.currentAddressHi ) << 16 ) | (uint32)( p_axvpb->pb.addr.currentAddressLo );
762  bool safe_to_dma = false;
763 
764  // Convert from current ARAM position (in nibbles) to current offset within buffer (in bytes).
765  current_aram_pos = ( current_aram_pos - ( stream_base[stream] * 2 )) / 2;
766 
767  if( current_aram_pos < STREAM_BUFFER_SIZE )
768  {
769  // Safe to write to the second half of the ARAM buffer, if that is what needs updating.
770  if( gStreamInfo[stream].nextWrite == 1 )
771  {
772  safe_to_dma = true;
773 // OSReport( "Stream: %d ARAM: %d writing to second half\n", stream, current_aram_pos );
774  }
775  }
776  else
777  {
778  // Safe to write to the first half of the ARAM buffer, if that is what needs updating.
779  if( gStreamInfo[stream].nextWrite == 0 )
780  {
781  safe_to_dma = true;
782 // OSReport( "Stream: %d ARAM: %d writing to first half\n", stream, current_aram_pos );
783  }
784  }
785 
786  if( safe_to_dma )
787  {
788  // DMA this chunk up to ARAM.
789  ARQPostRequest( &stream_arq_request[stream], // Address to user-allocated storage for an ARQTask data structure.
790  0, // User-enumerated identifier for the owner of the ARAM transaction.
791  ARQ_TYPE_MRAM_TO_ARAM, // Direction.
792  ARQ_PRIORITY_HIGH, // Priority.
793  (u32)( stream_mem_buffer[stream] ), // Source address.
794  stream_base[stream] + gStreamInfo[stream].offsetInARAM, // Destination address.
795  STREAM_BUFFER_SIZE, // Transfer length (bytes). Must be multiple of 4.
796  NULL ); // DMA complete callback function.
797 
798  // Increment the ARAM offset and wrap when appropriate.
799  gStreamInfo[stream].offsetInARAM += STREAM_BUFFER_SIZE;
800  if( gStreamInfo[stream].offsetInARAM >= ( STREAM_BUFFER_SIZE * 2 ))
801  {
802  gStreamInfo[stream].offsetInARAM -= ( STREAM_BUFFER_SIZE * 2 );
803  }
804 
805  // We have now processed another chunk.
806  gStreamInfo[stream].nextWrite = ( gStreamInfo[stream].nextWrite == 0 ) ? 1 : 0;
807 
808  // Increment the file pointer.
809  gStreamInfo[stream].offset += STREAM_BUFFER_SIZE;
810  }
811  else
812  {
813 // OSReport( "Stream: %d ARAM: %d discarding file buffer\n", stream, current_aram_pos );
814  }
815  }
816 
817  // If there is more data, kick off another read. NOTE this could potentially write over what we are about to DMA to ARAM.
818  int total_bytes_after_next_read = gStreamInfo[stream].offset + STREAM_BUFFER_SIZE;
819  int bytes_to_read;
820  if( total_bytes_after_next_read <= gStreamInfo[stream].fileSize )
821  {
822  bytes_to_read = STREAM_BUFFER_SIZE;
823  }
824  else
825  {
826  bytes_to_read = STREAM_BUFFER_SIZE - ( total_bytes_after_next_read - gStreamInfo[stream].fileSize );
827 
828  // Reduce bytes_to_read down to the next lowest multiple of 32.
829  bytes_to_read &= ~31;
830  }
831 
832  if( bytes_to_read > 0 )
833  {
834  _DVDReadAsync( &( gStreamInfo[stream].fileInfo[gStreamInfo[stream].currentFileInfo] ),
835  stream_mem_buffer[stream], // Base pointer of memory to read data into.
836  bytes_to_read, // Bytes to read.
837  gStreamInfo[stream].hedOffset + gStreamInfo[stream].offset, // Offset in file.
838  s_dvd_callback ); // Read complete callback.
839  }
840  else
841  {
842  // Here we may need to set the stop point for the voice, and switch from looping to non-looping.
843  if( p_axvpb->pb.addr.loopFlag == AXPBADDR_LOOP_ON )
844  {
845 // OSReport( "Stream: %d switching to loop off\n", stream );
846  p_axvpb->pb.addr.loopFlag = AXPBADDR_LOOP_OFF;
847  u32 zero_buffer_address = NxNgc::EngineGlobals.aram_zero * 2;
848  p_axvpb->pb.addr.loopAddressHi = (u16)( zero_buffer_address >> 16 );
849  p_axvpb->pb.addr.loopAddressLo = (u16)( zero_buffer_address & 0xFFFFU );
850  p_axvpb->sync |= ( AX_SYNC_USER_LOOP | AX_SYNC_USER_LOOPADDR );
851  }
852  }
