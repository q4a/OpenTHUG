78 bool CXboxTexture::plat_replace_texture( CTexture *p_texture )
79 {
80  CXboxTexture *p_xbox_texture = static_cast<CXboxTexture *>( p_texture );
81 
82  // Go through and copy the texture.
83  NxXbox::sTexture *p_src = p_xbox_texture->GetEngineTexture();
84  NxXbox::sTexture *p_dst = GetEngineTexture();
85 
86  if( p_dst->pD3DTexture )
87  {
88  p_dst->pD3DTexture->Release();
89  }
90  if( p_dst->pD3DPalette )
91  {
92  p_dst->pD3DPalette->Release();
93  }
94 
95  D3DSURFACE_DESC desc;
96  uint32 num_levels = p_src->pD3DTexture->GetLevelCount();
97  p_src->pD3DTexture->GetLevelDesc( 0, &desc );
98  if( D3D_OK != D3DDevice_CreateTexture( desc.Width,
99  desc.Height,
100  num_levels,
101  0,
102  desc.Format,
103  0,
104  &p_dst->pD3DTexture ))
105  {
106  exit( 0 );
107  }
108 
109  // Create and copy the palette if present.
110  if( p_src->pD3DPalette )
111  {
112  if( D3D_OK != D3DDevice_CreatePalette( D3DPALETTE_256, &p_dst->pD3DPalette ))
113  {
114  exit( 0 );
115  }
116 
117  D3DCOLOR *p_src_palette;
118  D3DCOLOR *p_dst_palette;
119  p_src->pD3DPalette->Lock( &p_src_palette, D3DLOCK_READONLY );
120  p_dst->pD3DPalette->Lock( &p_dst_palette, 0 );
121  CopyMemory( p_dst_palette, p_src_palette, sizeof( D3DCOLOR ) * 256 );
122  }
123 
124  for( uint32 l = 0; l < num_levels; ++l )
125  {
126  p_src->pD3DTexture->GetLevelDesc( l, &desc );
127 
128  D3DLOCKED_RECT src_rect, dst_rect;
129  p_src->pD3DTexture->LockRect( l, &src_rect, NULL, D3DLOCK_READONLY );
130  p_dst->pD3DTexture->LockRect( l, &dst_rect, NULL, 0 );
131 
132  CopyMemory( dst_rect.pBits, src_rect.pBits, desc.Size );
133  }
134 
135  return true;
136 }

262 bool CXboxTexDict::LoadTextureDictionary( const char *p_tex_dict_name )
263 {
264  // Count the number of entries in the lookup table. If it is empty, it is safe
265  // to delete it and create a new, optimum sized one during the load process itself.
266  if( mp_texture_lookup )
267  {
268  int num_items = 0;
269  mp_texture_lookup->IterateStart();
270  while( mp_texture_lookup->IterateNext())
271  ++num_items;
272 
273  if( num_items == 0 )
274  mp_texture_lookup = Nx::LoadTextureFile( p_tex_dict_name, mp_texture_lookup, true );
275  else
276  Nx::LoadTextureFile( p_tex_dict_name, mp_texture_lookup );
277  }
278  else
279  {
280  Nx::LoadTextureFile( p_tex_dict_name, mp_texture_lookup );
281  }
282  return true;
283 }

396 {
397  uint8 *p_data = (uint8*)( *pp_mem );
398 
399  // Read the texture file version and number of textures.
400  int version, num_textures;
401  MemoryRead( &version, sizeof( int ), 1, p_data );
402  MemoryRead( &num_textures, sizeof( int ), 1, p_data );
403 
404  // If allowed, rebuild the texture table to the optimum size, using the same heap as the original table.
405  if( okay_to_rebuild_texture_table )
406  {
407  uint32 optimal_table_size = num_textures * 2;
408  uint32 test = 2;
409  uint32 size = 1;
410  for( ;; test <<= 1, ++size )
411  {
412  // Check if this iteration of table size is sufficient, or if we have hit the maximum size.
413  if(( optimal_table_size <= test ) || ( size >= 12 ))
414  {
415  Mem::Allocator::BlockHeader* p_bheader = Mem::Allocator::BlockHeader::sRead( p_texture_table );
416  Mem::Allocator* p_allocater = p_bheader->mpAlloc;
417 
418  delete p_texture_table;
419 
420  Mem::Manager::sHandle().PushContext( p_allocater );
421  p_texture_table = new Lst::HashTable<Nx::CTexture>( size );
422  Mem::Manager::sHandle().PopContext();
423  break;
424  }
425  }
426  }
427 
428  for( int t = 0; t < num_textures; ++t )
429  {
430  // Create the engine level texture.
431  NxXbox::sTexture *p_texture = new NxXbox::sTexture;
432 
433  uint32 base_width, base_height, levels, texel_depth, palette_depth, dxt, palette_size;
434  MemoryRead( &p_texture->Checksum, sizeof( uint32 ), 1, p_data );
435  MemoryRead( &base_width, sizeof( uint32 ), 1, p_data );
436  MemoryRead( &base_height, sizeof( uint32 ), 1, p_data );
437  MemoryRead( &levels, sizeof( uint32 ), 1, p_data );
438  MemoryRead( &texel_depth, sizeof( uint32 ), 1, p_data );
439  MemoryRead( &palette_depth, sizeof( uint32 ), 1, p_data );
440  MemoryRead( &dxt, sizeof( uint32 ), 1, p_data );
441  MemoryRead( &palette_size, sizeof( uint32 ), 1, p_data );
442 
443  p_texture->BaseWidth = (uint16)base_width;
444  p_texture->BaseHeight = (uint16)base_height;
445  p_texture->Levels = (uint8)levels;
446  p_texture->TexelDepth = (uint8)texel_depth;
447  p_texture->PaletteDepth = (uint8)palette_depth;
448  p_texture->DXT = (uint8)dxt;
449 
450  D3DFORMAT texture_format;
451  if( p_texture->DXT > 0 )
452  {
453  if(( p_texture->DXT == 1 ) || ( p_texture->DXT == 2 ))
454  {
455  texture_format = D3DFMT_DXT1;
456  }
457  else if( p_texture->DXT == 5 )
458  {
459  texture_format = D3DFMT_DXT5;
460  }
461  else
462  {
463  Dbg_Assert( 0 );
464  }
465  }
466  else if( p_texture->TexelDepth == 8 )
467  {
468  texture_format = D3DFMT_P8;
469  }
470  else if( p_texture->TexelDepth == 16 )
471  {
472  texture_format = D3DFMT_A1R5G5B5; // Could also be X1R5G5B5;
473  }
474  else if( p_texture->TexelDepth == 32 )
475  {
476  texture_format = D3DFMT_A8R8G8B8;
477  }
478  else
479  {
480  Dbg_Assert( 0 );
481  }
482 
483  if( D3D_OK != D3DDevice_CreateTexture( p_texture->BaseWidth, p_texture->BaseHeight, p_texture->Levels, 0, texture_format, 0, &p_texture->pD3DTexture ))
484  {
485  Dbg_Assert( 0 );
486  }
487 
488  if( palette_size > 0 )
489  {
490  // Create and lock the palette.
491  if( D3D_OK != D3DDevice_CreatePalette( palette_size == ( 256 * sizeof( D3DCOLOR )) ? D3DPALETTE_256 : D3DPALETTE_32, &p_texture->pD3DPalette ))
492  {
493  Dbg_Assert( 0 );
494  }
495  else
496  {
497  D3DCOLOR* p_colors;
498  if( D3D_OK != p_texture->pD3DPalette->Lock( &p_colors, 0 ))
499  {
500  Dbg_Assert( 0 );
501  }
502  else
503  {
504  // Read in palette data.
505  MemoryRead( p_colors, palette_size, 1, p_data );
506  }
507  }
508  }
509  else
510  {
511  p_texture->pD3DPalette = NULL;
512  }
513 
514  for( uint32 mip_level = 0; mip_level < p_texture->Levels; ++mip_level )
515  {
516  uint32 texture_level_data_size;
517  MemoryRead( &texture_level_data_size, sizeof( uint32 ), 1, p_data );
518 
519  D3DLOCKED_RECT locked_rect;
520  if( D3D_OK != p_texture->pD3DTexture->LockRect( mip_level, &locked_rect, NULL, 0 ))
521  {
522  Dbg_Assert( 0 );
523  }
524  else

527  }
528  }
529

531  Nx::CXboxTexture *p_xbox_texture = new Nx::CXboxTexture();
532  p_xbox_texture->SetEngineTexture( p_texture );
533  p_texture_table->PutItem( p_texture->Checksum, p_xbox_texture );
534  }
535  return p_texture_table;
536 }
