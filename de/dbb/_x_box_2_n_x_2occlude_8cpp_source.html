114 void sOccluder::add_to_stack( sOcclusionPoly *p_poly )
115 {
116  if( NumOccluders < MAX_OCCLUDERS )
117  {
118  Dbg_Assert( p_poly->available );
119 
120  Occluders[NumOccluders].p_poly = p_poly;
121  p_poly->in_use = true;
122 
123  // Reset scores for all views.
124  memset( Occluders[NumOccluders].score, 0, sizeof( int ) * MAX_VIEWS_PER_OCCLUDER );
125 
126  ++NumOccluders;
127  }
128 }

132 /******************************************************************/

136 static int cmp( const void *p1, const void *p2 )
137 {
138  // Sort based on the sum total of scores in all views.
139  int score1 = 0;
140  int score2 = 0;
141  for( int v = 0; v < MAX_VIEWS_PER_OCCLUDER; ++v )
142  {
143  // Zero the score for any occlusion poly that is no longer available. This will force it out of the stack.
144  if(((sOccluder*)p1)->p_poly->available == false )
145  ((sOccluder*)p1)->score[v] = 0;
146 
147  if(((sOccluder*)p2)->p_poly->available == false )
148  ((sOccluder*)p2)->score[v] = 0;
149 
150  score1 += ((sOccluder*)p1)->score[v];
151  score2 += ((sOccluder*)p2)->score[v];
152  }
153 
154  return score1 < score2 ? 1 : score1 > score2 ? -1 : 0;
155 }

170 /******************************************************************/
171 /* */
172 /* */
173 /******************************************************************/
174 void sOccluder::tidy_stack( void )
175 {
176  if( NumOccluders > 0 )
177  {
178  // Sort in descending score order.
179  sort_stack();
180 
181  // Count backwards so we know we get all the bad occluders.
182  for( int i = NumOccluders - 1; i >= 0; --i )
183  {
184  // If we have hit an occluder with zero meshes culled, cut off the stack at this point.
185  int total_score = 0;
186  for( int v = 0; v < MAX_VIEWS_PER_OCCLUDER; ++v )
187  {
188  total_score += Occluders[i].score[v];
189  }
190 
191  if( total_score == 0 )
192  {
193  // No longer using this poly.
194  Occluders[i].p_poly->in_use = false;
195 
196  // One less occluder to worry about.
197  --NumOccluders;
198  }
199  else
200  {
201  // Reset the good occluders.
202  Occluders[i].score[CurrentView] = 0;
203  }
204  }
205  }
206 }

222 {
223  Dbg_Assert( NumOcclusionPolys < MAX_OCCLUSION_POLYS );

225  OcclusionPolys[NumOcclusionPolys].in_use = false;
226  OcclusionPolys[NumOcclusionPolys].available = true;
227  OcclusionPolys[NumOcclusionPolys].checksum = checksum;
228  OcclusionPolys[NumOcclusionPolys].verts[0] = v0;
229  OcclusionPolys[NumOcclusionPolys].verts[1] = v1;
230  OcclusionPolys[NumOcclusionPolys].verts[2] = v2;
231  OcclusionPolys[NumOcclusionPolys].verts[3] = v3;
232  OcclusionPolys[NumOcclusionPolys].normal = Mth::CrossProduct( v1 - v0, v3 - v0 );
233  OcclusionPolys[NumOcclusionPolys].normal.Normalize();
234 
235  ++NumOcclusionPolys;
236 }

278 void CheckForOptimalOccluders( Mth::Vector &cam_pos, Mth::Vector &view_direction )
279 {
280  if( NumOcclusionPolys > 0 )
281  {
282  uint32 added = 0;
283  uint32 checked = 0;
284 
285  while( added < MAX_NEW_OCCLUSION_POLYS_TO_CHECK_PER_FRAME )
286  {
287  // Given the current position of the camera, check through the unused, available occlusion polys to see if one scores higher
288  // than the lowest scoring occlusion poly in use.
289  sOcclusionPoly *poly_to_check = &OcclusionPolys[NextOcclusionPolyToCheck++];
290  if(( !poly_to_check->in_use ) && ( poly_to_check->available ))
291  {
292  sOccluder::add_to_stack( poly_to_check );
293  ++added;
294  }
295  ++checked;
296 
297  // Ensure we are always checking within bounds.
298  if( NextOcclusionPolyToCheck >= NumOcclusionPolys )
299  {
300  NextOcclusionPolyToCheck = 0;
301  }
302 
303  // Quit out if we have less available occluders than spaces to fill.
304  if( checked >= NumOcclusionPolys )
305  {
306  break;
307  }
308  }
309  }
310 }

329  Dbg_Assert( view < MAX_VIEWS_PER_OCCLUDER );
330  CurrentView = view;
331 
332  // Tidy up from last frame.
333  sOccluder::tidy_stack();
334 
335  // Cyclically add more occluders for checking.
336  CheckForOptimalOccluders( *p_cam_pos, *p_cam_pos );
337 
338  // Build all 5 planes for each occluder.
339  Mth::Vector u0, u1, p;
340 
341  // The order in which the verts are used to build tha planes depends upon where the camera is in relation to the occlusion
342  // poly. We use the default order when the viewpoint is on the side of the poly on which the default poly normal faces.
343  for( uint32 i = 0; i < sOccluder::NumOccluders; ++i )
344  {
345  sOcclusionPoly *p_poly = sOccluder::Occluders[i].p_poly;
346 
347 // Gfx::AddDebugLine( p_poly->verts[0], p_poly->verts[1], 0, 0, 1 );
348 // Gfx::AddDebugLine( p_poly->verts[1], p_poly->verts[2], 0, 0, 1 );

350 // Gfx::AddDebugLine( p_poly->verts[3], p_poly->verts[0], 0, 0, 1 );
351 
352  if( Mth::DotProduct( *p_cam_pos - p_poly->verts[0], p_poly->normal ) >= 0.0f )
353  {
354  // Start with the front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
355  // are considered occluded. (1->3->4)...
356  u0 = p_poly->verts[2] - p_poly->verts[0];
357  u1 = p_poly->verts[3] - p_poly->verts[0];
358  p = Mth::CrossProduct( u0, u1 );
359  p.Normalize();
360  p[W] = Mth::DotProduct( p, p_poly->verts[0] );
361  sOccluder::Occluders[i].planes[0] = p;
362 
363  // ...then left (0->1->2)...
364  u0 = p_poly->verts[0] - *p_cam_pos;
365  u1 = p_poly->verts[1] - *p_cam_pos;
366  p = Mth::CrossProduct( u0, u1 );
367  p.Normalize();
368  p[W] = Mth::DotProduct( p, *p_cam_pos );
369  sOccluder::Occluders[i].planes[1] = p;
370 
371  // ...then right (0->3->4)...
372  u0 = p_poly->verts[2] - *p_cam_pos;
373  u1 = p_poly->verts[3] - *p_cam_pos;
374  p = Mth::CrossProduct( u0, u1 );
375  p.Normalize();
376  p[W] = Mth::DotProduct( p, *p_cam_pos );
377  sOccluder::Occluders[i].planes[2] = p;
378 
379  // ...then top (0->2->3)...
380  u0 = p_poly->verts[1] - *p_cam_pos;
381  u1 = p_poly->verts[2] - *p_cam_pos;
382  p = Mth::CrossProduct( u0, u1 );
383  p.Normalize();
384  p[W] = Mth::DotProduct( p, *p_cam_pos );
385  sOccluder::Occluders[i].planes[3] = p;
386 
387  // ...then bottom (0->4->1)...
388  u0 = p_poly->verts[3] - *p_cam_pos;
389  u1 = p_poly->verts[0] - *p_cam_pos;
390  p = Mth::CrossProduct( u0, u1 );
391  p.Normalize();
392  p[W] = Mth::DotProduct( p, *p_cam_pos );
393  sOccluder::Occluders[i].planes[4] = p;
394  }
395  else
396  {
397  // Start with the front. We want to reverse the order of the front plane to ensure that objects *behind* the plane
398  // are considered occluded. (1->4->3)...
399  u0 = p_poly->verts[3] - p_poly->verts[0];
400  u1 = p_poly->verts[2] - p_poly->verts[0];
401  p = Mth::CrossProduct( u0, u1 );
402  p.Normalize();
403  p[W] = Mth::DotProduct( p, p_poly->verts[0] );
404  sOccluder::Occluders[i].planes[0] = p;
405 
406  // ...then left (0->2->1)...
407  u0 = p_poly->verts[1] - *p_cam_pos;
408  u1 = p_poly->verts[0] - *p_cam_pos;
409  p = Mth::CrossProduct( u0, u1 );
410  p.Normalize();
411  p[W] = Mth::DotProduct( p, *p_cam_pos );
412  sOccluder::Occluders[i].planes[1] = p;
413 
414  // ...then right (0->4->3)...
415  u0 = p_poly->verts[3] - *p_cam_pos;
416  u1 = p_poly->verts[2] - *p_cam_pos;
417  p = Mth::CrossProduct( u0, u1 );
418  p.Normalize();
419  p[W] = Mth::DotProduct( p, *p_cam_pos );
420  sOccluder::Occluders[i].planes[2] = p;
421 
422  // ...then top (0->3->2)...
423  u0 = p_poly->verts[2] - *p_cam_pos;
424  u1 = p_poly->verts[1] - *p_cam_pos;
425  p = Mth::CrossProduct( u0, u1 );

447 bool TestSphereAgainstOccluders( D3DXVECTOR3 *p_center, float radius, uint32 meshes )
448 {
449  XGVECTOR3 center;
450 
451  // Build the composite transform if required.
452  if( p_bbox_transform )
453  {
454  // Object to world, so transform the sphere center.
455  center.x = p_center->x + p_bbox_transform->_41;
456  center.y = p_center->y + p_bbox_transform->_42;
457  center.z = p_center->z + p_bbox_transform->_43;
458  }
459  else
460  {
461  center.x = p_center->x;
462  center.y = p_center->y;
463  center.z = p_center->z;
464  }
465 
466  // Test against each occluder.
467  for( uint32 o = 0; o < sOccluder::NumOccluders; ++o )
468  {
469  bool occluded = true;
470 
471  // Test against each plane in the occluder.
472  for( uint32 p = 0; p < 5; ++p )
473  {
474  float result = ( sOccluder::Occluders[o].planes[p][X] * center.x ) +
475  ( sOccluder::Occluders[o].planes[p][Y] * center.y ) +
476  ( sOccluder::Occluders[o].planes[p][Z] * center.z ) -
477  ( sOccluder::Occluders[o].planes[p][W] );
478  if( result >= -radius )
479  {
480  // Outside of this plane, therefore not occluded by this occluder.
481  occluded = false;
482  break;
483  }
484  }
485 
486  if( occluded )
487  {
488  // Inside all planes, therefore occluded. Increase score for this occluder.
489  sOccluder::Occluders[o].score[CurrentView] += meshes;
490  return true;
491  }
492  }
493  return false;
494 }
