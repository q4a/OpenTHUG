257 void CScriptCache::remove_some_old_scripts(int space_required, uint32 scriptName)
258 {
259  while (true)
260  {
261  int largest_malloc_possible=Mem::Manager::sHandle().ScriptHeap()->LargestFreeBlock();
262  largest_malloc_possible-=128; // Actually, 32 seems OK, but have a bigger margin just to be sure.
263 
264  if (space_required <= largest_malloc_possible)
265  {
266  // Hurrah! It's all clear to decompress another script since we know there is enough heap to hold it.
267 
268  // Try to keep the total number of decompressed scripts within IDEAL_MAX_DECOMPRESSED_SCRIPTS
269  // by removing unused ones until either enough have been removed or there are no more to remove.
270 
271  // This way the average memory usage can be controlled by tweaking IDEAL_MAX_DECOMPRESSED_SCRIPTS
272  // whilst spikes in the number of scripts are still permitted up to the limit set by
273  // the pool size of MAX_DECOMPRESSED_SCRIPTS. If that limit is reached, the code will assert.
274  while (CScriptCacheEntry::SGetNumUsedItems() >= IDEAL_MAX_DECOMPRESSED_SCRIPTS && mp_last_zero_usage)
275  {
276  delete_entry(mp_last_zero_usage);
277  }
278  break;
279  }
280 
281  // Free up some space by deleting the oldest zero-usage decompressed script.
282 
283  if (!mp_last_zero_usage)
284  {
285  // Eeeek! There's nothing left to delete ...
286  Dbg_MsgAssert(0,("Script heap overflow when trying to allocate decompressed script '%s' of size %d bytes!",Script::FindChecksumName(scriptName),space_required));
287  }
288 
289  //printf("%d: Deleting %s from cache, new free space = %d\n",space_required,Script::FindChecksumName(mp_last_zero_usage->mScriptNameChecksum),Mem::Manager::sHandle().ScriptHeap()->LargestFreeBlock()-128);
290  delete_entry(mp_last_zero_usage);
291  }

294 uint8 *CScriptCache::GetScript(uint32 scriptName)
295 {
296  CSymbolTableEntry *p_entry=Resolve(scriptName);
297  if (p_entry)
298  {
299  Dbg_MsgAssert(p_entry->mType==ESYMBOLTYPE_QSCRIPT,("Symbol %s is not a QScript",FindChecksumName(scriptName)));
300 #ifndef __PLAT_NGC__
301  Dbg_MsgAssert(p_entry->mpScript,("NULL p_entry->mpScript"));
302 #endif // __PLAT_NGC__
303 
304  // Change the passed scriptName to be the actual scriptName so that when decompressed it gets registered in
305  // the cache under the correct name.
306  // Otherwise, if a SpawnScript Foo is done after Foo has been changed using the 'change' command, it
307  // will spawn the old script again, since it will be in the cache under Foo.
308  scriptName=p_entry->mNameChecksum;
309  }
310 
311 
312  CScriptCacheEntry *p_cache_entry=mp_cache_hash_table->GetItem(scriptName);
313  if (p_cache_entry)
314  {
315  ++p_cache_entry->mUsage;
316 
317  #ifdef __NOPT_ASSERT__
318  if (p_cache_entry->mUsage==1)
319  {
320  ++m_num_used_scripts;
321  if (m_num_used_scripts > m_max_used_scripts)
322  {
323  m_max_used_scripts=m_num_used_scripts;
324  }
325  }
326  #endif
327 
328  remove_from_zero_usage_list(p_cache_entry);
329  return p_cache_entry->mpDecompressedScript;
330  }
331 
332  if (p_entry)
333  {
334 #ifdef __PLAT_NGC__
335  enum
336  {
337  COMPRESS_BUFFER_SIZE=20000,
338  };
339  uint8 p_compress_buffer[COMPRESS_BUFFER_SIZE];
340  uint32 header[(SCRIPT_HEADER_SIZE/4)];
341  NsDMA::toMRAM( header, p_entry->mScriptOffset, SCRIPT_HEADER_SIZE );
342  uint32 uncompressed_size = header[1];
343  uint32 compressed_size = header[2];
344  NsDMA::toMRAM( p_compress_buffer, p_entry->mScriptOffset + SCRIPT_HEADER_SIZE, compressed_size );
345 #else
346  // Note: The script contents checksum is at offset 0, but it is not needed.
347  uint32 uncompressed_size = *(uint32*)(p_entry->mpScript+4);
348  uint32 compressed_size = *(uint32*)(p_entry->mpScript+8);
349 #endif // __PLAT_NGC__
350 
351  remove_some_old_scripts(uncompressed_size,scriptName);
352 
353  CScriptCacheEntry *p_cache_entry=new CScriptCacheEntry;
354 
355  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
356  uint8 *p_new_script=(uint8*)Mem::Malloc(uncompressed_size);
357  Mem::Manager::sHandle().PopContext();
358 
359  if (uncompressed_size > compressed_size)
360  {
361  #ifdef __NOPT_ASSERT__
362  uint8 *p_end=
363  #endif
364 #ifdef __PLAT_NGC__
365  DecodeLZSS(p_compress_buffer,p_new_script,compressed_size);
366 #else
367  DecodeLZSS(p_entry->mpScript+SCRIPT_HEADER_SIZE,p_new_script,compressed_size);
368 #endif // __PLAT_NGC__
369  Dbg_MsgAssert(p_end==p_new_script+uncompressed_size,("Eh? p_end is not right?"));
370  }
371  else
372  {
373  // The script is uncompressed so just copy it over. Saves, errr, 1K altogether, oh well.
374  Dbg_MsgAssert(uncompressed_size == compressed_size,("Expected uncompressed_size==compressed_size"));
375 #ifdef __PLAT_NGC__
376 
377  uint8 *p_source=p_compress_buffer;
378 #else
379  uint8 *p_source=p_entry->mpScript+SCRIPT_HEADER_SIZE;
380 #endif // __PLAT_NGC__
381  uint8 *p_dest=p_new_script;
382  for (uint32 i=0; i<uncompressed_size; ++i)
383  {
384  *p_dest++=*p_source++;
385  }
386  }
387 
388  PreProcessScript(p_new_script);
389 
390  p_cache_entry->mpDecompressedScript=p_new_script;
391  p_cache_entry->mUsage=1;
392  p_cache_entry->mScriptNameChecksum=scriptName;
393 
394  #ifdef __NOPT_ASSERT__
395  ++m_num_used_scripts;
396  if (m_num_used_scripts > m_max_used_scripts)
397  {
398  m_max_used_scripts=m_num_used_scripts;
399  }
400  #endif
401 
402 
403  mp_cache_hash_table->PutItem(scriptName,p_cache_entry);
404 
405  #ifdef __NOPT_ASSERT__
406  ++mp_decompress_counts[m_current_decompress_count_index];
407  #endif
408 
409  return p_cache_entry->mpDecompressedScript;
410  }
411 
412  return NULL;
413 }
