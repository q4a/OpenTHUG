616 // Gets a stat value given a pointer to the stat structure.
617 float CSkater::GetScriptedStat(Script::CStruct *pSS)
618 {
619 
620  Script::CPair range;
621  Script::CPair switch_range;
622  int stat;
623  float limit;
624 
625  Dbg_MsgAssert(pSS,("NULL pSS sent to GetScriptedStat"));
626 
627  if (!pSS->GetPair(NONAME,&range))
628  {
629  Dbg_MsgAssert(0,("range pair not found in structure."));
630  }
631 
632  float stat_value;
633  if (!pSS->GetInteger(NONAME,&stat))
634  {
635  stat_value = 10.0f; // default to 10 if no stats
636  }
637  else
638  {
639  stat_value = GetStat((CSkater::EStat)stat);
640  }
641 
642  float lower = range.mX;
643  float upper = range.mY;
644  float len = upper - lower;
645  float value = lower + (len * stat_value / 10);
646 
647 
648  if (mp_skater_core_physics_component->IsSwitched()) // if skating in switch stance
649  {
650  if (pSS->GetPair(0x9016b4e7,&switch_range)) // switch
651  {
652  float switch_stat = GetStat(STATS_SWITCH); // returns 0..13
653  float mult = switch_range.mX + (switch_range.mY - switch_range.mX) * switch_stat/10.0f;
654 
655  if (mult <0.0f) mult = 0.0f;
656  if (mult >1.0f) mult = 1.0f; // this ensures we are never Better in switch
657 
658  value *= mult;
659 
660  }
661  }
662 
663  Game::GOAL_MANAGER_DIFFICULTY_LEVEL level = Mdl::Skate::Instance()->GetGoalManager()->GetDifficultyLevel();
664  if (level != Game::GOAL_MANAGER_DIFFICULTY_MEDIUM)
665  {
666  Script::CPair diff;
667  if (pSS->GetPair(CRCD(0xba8fb854,"diff"),&diff))
668  {
669  if (level == Game::GOAL_MANAGER_DIFFICULTY_LOW)
670  {
671  value *= diff.mX;
672  }
673  else if (level == Game::GOAL_MANAGER_DIFFICULTY_HIGH)
674  {
675  value *= diff.mY;
676  }
677  }
678  }
679 
680  // if there is a limit, then clamp the value to it
681  // accounting for the direction of the range, this may be a high limit
682  // or a low limit, but applies to the value for higher values of stat_value.
683  // so if lower > upper, then it's a low limit
684  if (pSS->GetFloat(0x8069179f/*"limit"*/,&limit))
685  {
686  if (lower < upper)
687  {
688  if (value > limit)
689  {
690  value = limit;
691  }
692  }
693  else
694  {
695  if (value < limit)
696  {
697  value = limit;
698  }
699  }
700  }
701  else
702  {
703  // there is no limit, so we do nothing.
704  }
705 
706  return value;
707 }
