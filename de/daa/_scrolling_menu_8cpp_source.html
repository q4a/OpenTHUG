112  pEvent->MarkHandled(m_id);
113  }
114  }
115  if (pEvent->GetType() == Obj::CEvent::TYPE_UNFOCUS && m_in_focus)
116  {
117  get_the_menu();
118 
119  //printf("VMenu %s receives focus event, sending to 0x%x\n", Script::FindChecksumName(m_id), m_selected_id);
120  m_in_focus = false;

170 void CBaseScrollingMenu::get_the_menu()
171 {
172  CScreenElementPtr p_child = GetFirstChild();
173  while(p_child)
174  {
175  if ((uint32) p_child->GetType() == CRCX("vmenu") && m_is_vertical)
176  mp_the_menu = (CBaseMenu *) p_child.Convert();
177  else if ((uint32) p_child->GetType() == CRCX("hmenu") && !m_is_vertical)

182  Dbg_MsgAssert(mp_the_menu || !m_is_vertical, ("could not find VMenu"));

187  {

189  m_regular_space_amount = space_amount;
190  }
191  else
192  {

211 void CBaseScrollingMenu::update()
212 {
213  CScreenElementPtr pElement = NULL;
214  int element_index = 0;
215 
216  float center_menu_offset_up = 0.0f;
217  float center_menu_offset_down = 0.0f;
218 
219  if ( mp_the_menu )
220  {
221  if ( m_selected_element_id )
222  pElement = mp_the_menu->GetChildById(m_selected_element_id, &element_index);
223  else
224  pElement = mp_the_menu->GetChildByIndex( 0 );
225 
226  if ( pElement )
227  {
228  int num_children = mp_the_menu->CountChildren();
229 
230  // Good doggy. Now find out if the selection needs to be centered in the window
231  if (m_regular_space_amount >= 0.0f)
232  {
233  if (element_index >= (m_num_items_to_display / 2))
234  {
235  center_menu_offset_up = m_regular_space_amount * ((m_num_items_to_display - 1) / 2);
236  }
237  if (element_index < num_children - (m_num_items_to_display / 2))
238  {
239  center_menu_offset_down = m_regular_space_amount * (m_num_items_to_display / 2);
240  }
241  }
242  }
243  }
244 
245  if (pElement)
246  {
247  float menu_x, menu_y;
248  mp_the_menu->SetJust(-1.0f, -1.0f);
249  mp_the_menu->GetLocalPos(&menu_x, &menu_y);
250 
251  // get upper-left of element, local coordinates (in relation to VMenu)
252  float element_x, element_y, element_w, element_h;
253  float element_z, element_l;
254  if (m_is_vertical)
255  {
256  compute_element_area(pElement, element_x, element_z, element_w, element_l);
257  }
258  else
259  {
260  compute_element_area(pElement, element_z, element_y, element_l, element_h);
261  }
262  if (m_regular_space_amount >= 0.0f)
263  {
264  element_z = m_regular_space_amount * element_index;
265  element_l = m_regular_space_amount;
266  }
267 
268  // if element top is above window top
269  if (element_z < m_top_or_left_line + center_menu_offset_up)
270  {
271  // scroll up
272  m_top_or_left_line -= m_scroll_speed;
273  if (m_top_or_left_line < element_z)
274  m_top_or_left_line = element_z - center_menu_offset_up;
275  m_needs_update = true;
276  }
277  // if element bottom is below window bottom
278  else if (element_z + element_l > m_top_or_left_line + m_window_dimension - center_menu_offset_down)
279  {
280  // scroll down
281  m_top_or_left_line += m_scroll_speed;
282  if (m_top_or_left_line + m_window_dimension > element_z + element_l)
283  m_top_or_left_line = element_z + element_l - m_window_dimension + center_menu_offset_down;
284  m_needs_update = true;
285  }
286 
287  if (m_needs_update)
288  {
289  if (m_is_vertical)
290  mp_the_menu->SetPos(menu_x, -m_top_or_left_line, FORCE_INSTANT);
291  else
292  mp_the_menu->SetPos(-m_top_or_left_line, menu_y, FORCE_INSTANT);
293 
294  //printf("m_top_or_left_line is %.2f\n", m_top_or_left_line);
295 
296  // make elements visible or invisible
297  element_index = 0;
298  CScreenElementPtr p_child = mp_the_menu->GetFirstChild();
299  while(p_child)
300  {
301  if (m_is_vertical)
302  {
303  compute_element_area(p_child, element_x, element_z, element_w, element_l);
304  }
305  else
306  {
307  compute_element_area(p_child, element_z, element_y, element_l, element_h);
308  }
309  if (m_regular_space_amount >= 0.0f)
310  {
311  element_z = m_regular_space_amount * element_index;
312  element_l = m_regular_space_amount;
313  //Ryan("oog\n");
314  }
315 
316  const float TOLERANCE = 0.0f;
317  if (element_z >= m_top_or_left_line - TOLERANCE && element_z + element_l <= m_top_or_left_line + m_window_dimension + TOLERANCE)
318  {
319  // top is (mostly) in window, bottom is (mostly) in window
320  p_child->SetAlpha(1.0f, FORCE_INSTANT);
321  }
322  else
323  {
324  p_child->SetAlpha(0.0f, FORCE_INSTANT);
325  }
326 
327  p_child = p_child->GetNextSibling();
328  element_index++;
329  }
330  m_needs_update = false;
331  }
332  } // end if pElement
333 }

339 void CBaseScrollingMenu::compute_element_area(const CScreenElementPtr &pElement, float &elementX, float &elementY, float &elementW, float &elementH)
340 {
341  pElement->GetLocalPos(&elementX, &elementY);
342  elementX *= mp_the_menu->GetScaleX();
343  elementY *= mp_the_menu->GetScaleY();
344  elementW = pElement->GetBaseW() * pElement->GetScaleX() * mp_the_menu->GetScaleX();
345  elementH = pElement->GetBaseH() * pElement->GetScaleY() * mp_the_menu->GetScaleY();
346  float element_just_x, element_just_y;
347  pElement->GetJust(&element_just_x, &element_just_y);
348  elementX -= (element_just_x + 1.0f) * elementW / 2.0f;
349  elementY -= (element_just_y + 1.0f) * elementH / 2.0f;
350 }
