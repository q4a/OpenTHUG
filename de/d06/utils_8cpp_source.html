57 void PrintContents(const CArray *p_array, int indent)
58 {
59 #ifdef __NOPT_ASSERT__
60  Dbg_MsgAssert(p_array,("NULL p_array"));
61  int size=p_array->GetSize();
62  ESymbolType type=p_array->GetType();
63 
64  if (!indent)
65  {
66  printf("Contents of Array=\n");
67  }
68 
69  sDoIndent(indent);
70  printf("[\n");
71  indent+=3;
72 
73  int int_column_count=0;
74  for (int i=0; i<size; ++i)
75  {
76  if (type!=ESYMBOLTYPE_STRUCTURE && type!=ESYMBOLTYPE_ARRAY)
77  {
78  if (type!=ESYMBOLTYPE_INTEGER || (type==ESYMBOLTYPE_INTEGER && int_column_count==0))
79  {
80  sDoIndent(indent);
81  }
82  }
83 
84  switch (type)
85  {
86  case ESYMBOLTYPE_INTEGER:
87  if (i<size-1)
88  {
89  printf("%d,",p_array->GetInteger(i));
90  }
91  else
92  {
93  printf("%d\n",p_array->GetInteger(i));
94  }
95  ++int_column_count;
96  if (int_column_count==15)
97  {
98  printf("\n");
99  int_column_count=0;
100  }
101  break;
102  case ESYMBOLTYPE_FLOAT:
103  printf("%f\n",p_array->GetFloat(i));
104  break;
105  case ESYMBOLTYPE_STRING:
106  {
107  const char *p_string=p_array->GetString(i);
108  if (p_string)
109  {
110  printf("\"%s\"\n",p_string);
111  }
112  else
113  {
114  printf("NULL string\n");
115  }
116  break;
117  }
118  case ESYMBOLTYPE_LOCALSTRING:
119  {
120  const char *p_string=p_array->GetLocalString(i);
121  if (p_string)
122  {
123  printf("'%s'\n",p_string);
124  }
125  else
126  {
127  printf("NULL local string\n");
128  }
129  break;
130  }
131  case ESYMBOLTYPE_PAIR:
132  {
133  CPair *p_pair=p_array->GetPair(i);
134  if (p_pair)

137  }

143  }

294 static uint8 *sWriteCompressedName(uint8 *p_buffer, uint8 symbolType, uint32 nameChecksum)
295 {
296  #ifdef __PLAT_WN32__
297  // If compiling on PC, the lookup table will not exist, so do no compression.
298  *p_buffer++=symbolType;
299  return Write4Bytes(p_buffer,nameChecksum);
300  #else
301 
302  // Check if the checksum is in the small array.
303  CArray *p_table=GetArray(0x35115a20/*WriteToBuffer_CompressionLookupTable_8*/);
304  int size=p_table->GetSize();
305  Dbg_MsgAssert(size<256,("Size of WriteToBuffer_CompressionLookupTable_8 too big"));
306 
307  for (int i=0; i<size; ++i)
308  {
309  if (p_table->GetChecksum(i)==nameChecksum)
310  {
311  // It is in the array! So write out a 1 byte index.
312  *p_buffer++=symbolType | MASK_8_BIT_NAME_LOOKUP;
313  *p_buffer++=i;
314  return p_buffer;
315  }
316  }
317 
318  // Check if the checksum is in the big array.
319  p_table=GetArray(0x25231f42/*WriteToBuffer_CompressionLookupTable_16*/);
320  size=p_table->GetSize();
321  Dbg_MsgAssert(size<65536,("Size of WriteToBuffer_CompressionLookupTable_16 too big"));
322 
323  for (int i=0; i<size; ++i)
324  {
325  if (p_table->GetChecksum(i)==nameChecksum)
326  {
327  // It is in the array! So write out a 2 byte index.
328  *p_buffer++=symbolType | MASK_16_BIT_NAME_LOOKUP;
329  return Write2Bytes(p_buffer,i);
330  }
331  }
332 
333  // Oh well, it is not in either array, so write out the whole 4 byte checksum.
334  *p_buffer++=symbolType;
335  return Write4Bytes(p_buffer,nameChecksum);
336  #endif
337 }

339 static uint32 sIntegerWriteToBuffer(uint32 name, int val, uint8 *p_buffer, uint32 bufferSize)
340 {
341  uint8 *p_buffer_before=p_buffer;
342 
343  // Choose what type of symbol to use so as to minimize the space used.
344 
345  ESymbolType symbol_type=ESYMBOLTYPE_INTEGER; // Default to a full 4 bytes.
346 
347  if (val==0)
348  {
349  symbol_type=ESYMBOLTYPE_ZERO_INTEGER;
350  }
351  else if (val>=0 && val<=255)
352  {
353  symbol_type=ESYMBOLTYPE_UNSIGNED_INTEGER_ONE_BYTE;
354  }
355  else if (val>=0 && val<=65535)
356  {
357  symbol_type=ESYMBOLTYPE_UNSIGNED_INTEGER_TWO_BYTES;
358  }
359  else if (val>=-128 && val<=127)
360  {
361  symbol_type=ESYMBOLTYPE_INTEGER_ONE_BYTE;
362  }
363  else if (val>=-32768 && val<=32767)
364  {
365  symbol_type=ESYMBOLTYPE_INTEGER_TWO_BYTES;
366  }
367 
368 
369  uint8 p_temp[20];
370  uint8 *p_end=p_temp;
371 
372  switch (symbol_type)
373  {
374  case ESYMBOLTYPE_INTEGER:
375  // Write out the component into p_temp
376  p_end=sWriteCompressedName(p_temp,symbol_type,name);
377  p_end=Write4Bytes(p_end,(uint32)val);
378  break;
379 
380  case ESYMBOLTYPE_INTEGER_ONE_BYTE:
381  case ESYMBOLTYPE_UNSIGNED_INTEGER_ONE_BYTE:
382  // Write out the component into p_temp
383  p_end=sWriteCompressedName(p_temp,symbol_type,name);
384  *p_end++=(uint8)val;
385  break;
386 
387  case ESYMBOLTYPE_INTEGER_TWO_BYTES:
388  case ESYMBOLTYPE_UNSIGNED_INTEGER_TWO_BYTES:
389  // Write out the component.
390  p_end=sWriteCompressedName(p_temp,symbol_type,name);
391  p_end=Write2Bytes(p_end,(uint16)val);
392  break;
393 
394  case ESYMBOLTYPE_ZERO_INTEGER:
395  // Write out the component.
396  p_end=sWriteCompressedName(p_temp,symbol_type,name);
397  break;
398 
399  default:
400  Dbg_MsgAssert(0,("Bad symbol_type"));
401  break;
402  }
403 
404  uint32 bytes_written_to_temp=p_end-p_temp;
405 
406  // Check that there is enough space before doing any writing.
407  if (bufferSize < bytes_written_to_temp)
408  {
409  return 0;
410  }
411 
412  uint8 *p_source=p_temp;
413  for (uint32 i=0; i<bytes_written_to_temp; ++i)
414  {
415  *p_buffer++=*p_source++;
416  }
417 
418  return p_buffer-p_buffer_before;
419 }

615 // Writes the contents of the structure to a buffer.
616 // The information is outputted in a byte-packed format, so p_buffer does
617 // not need to be aligned.
618 // The buffer can then be used to regenerate the original structure by
619 // passing it to ReadFromBuffer.
620 // Passed the size of the buffer so that it can check if there is enough space.
621 // Returns the number of bytes that it actually wrote.
622 //
623 // If there was not enough space, and assert is NO_ASSERT, it will return a count of 0.
624 //
625 uint32 WriteToBuffer(CStruct *p_structure, uint8 *p_buffer, uint32 bufferSize, EAssertType assert)
626 {
627  Dbg_MsgAssert(p_buffer,("NULL p_buffer sent to WriteToBuffer"));
628  uint32 bytes_left=bufferSize;
629 
630  // Scan through the components adding each to the buffer.
631  CComponent *p_comp=NULL;
632  if (p_structure)
633  {
634  p_comp=p_structure->GetNextComponent(NULL);
635  }
636 
637  while (p_comp)
638  {
639  uint32 component_bytes_written=0;
640 
641  switch (p_comp->mType)
642  {
643  case ESYMBOLTYPE_INTEGER:
644  component_bytes_written=sIntegerWriteToBuffer(p_comp->mNameChecksum,p_comp->mIntegerValue,p_buffer,bytes_left);
645  break;
646 
647  case ESYMBOLTYPE_FLOAT:
648  component_bytes_written=sFloatWriteToBuffer(p_comp->mNameChecksum,p_comp->mFloatValue,p_buffer,bytes_left);
649  break;
650 
651  case ESYMBOLTYPE_NAME:
652  component_bytes_written=sChecksumWriteToBuffer(p_comp->mNameChecksum,p_comp->mChecksum,p_buffer,bytes_left);
653  break;
654 
655  case ESYMBOLTYPE_STRING:
656  component_bytes_written=sStringWriteToBuffer(p_comp->mNameChecksum,p_comp->mpString,p_buffer,bytes_left);
657  break;
658 
659  case ESYMBOLTYPE_LOCALSTRING:
660  component_bytes_written=sLocalStringWriteToBuffer(p_comp->mNameChecksum,p_comp->mpLocalString,p_buffer,bytes_left);
661  break;
662 
663  case ESYMBOLTYPE_PAIR:
664  component_bytes_written=sPairWriteToBuffer(p_comp->mNameChecksum,p_comp->mpPair,p_buffer,bytes_left);
665  break;
666 
667  case ESYMBOLTYPE_VECTOR:
668  component_bytes_written=sVectorWriteToBuffer(p_comp->mNameChecksum,p_comp->mpVector,p_buffer,bytes_left);
669  break;
670 
671  case ESYMBOLTYPE_STRUCTURE:
672  component_bytes_written=sStructureWriteToBuffer(p_comp->mNameChecksum,p_comp->mpStructure,p_buffer,bytes_left,assert);
673  break;
674 
675  case ESYMBOLTYPE_ARRAY:

738 // by WriteToBuffer.
739 // If the structure contained anything to begin with, it will get cleared.
740 //
741 // Returns a pointer to after the terminating ESYMBOLTYPE_NONE, required
742 // when this function calls itself.
743 uint8 *ReadFromBuffer(CStruct *p_structure, uint8 *p_buffer)
744 {
745  Dbg_MsgAssert(p_structure,("NULL p_structure"));
746  Dbg_MsgAssert(p_buffer,("NULL p_buffer sent to ReadFromBuffer"));
747  float zero_float=0.0f;
748 
749  // First clear anything currently in the structure.
750  p_structure->Clear();
751 
752  // Make sure we're using the script heap.
753  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
754

756  while (true)
757  {
758  float x,y,z;
759  const char *p_string;
760  CStruct *p_struct;
761  CArray *p_array;
762 
763  // Get the type and the name checksum.
764  ESymbolType type=(ESymbolType)*p_buffer++;
765  if (type==ESYMBOLTYPE_NONE)
766  {
767  // All done.
768  break;
769  }
770 
771  // Get the name checksum, which may be stored as an index into a table of checksums
772  // to save space.
773  uint32 name=0;
774  if (type&MASK_8_BIT_NAME_LOOKUP)
775  {
776  Dbg_MsgAssert(!(type&MASK_16_BIT_NAME_LOOKUP),("Eh? Both lookup-table flags set ?"));
777 
778  #ifdef __PLAT_WN32__
779  // The lookup table is not loaded when compiling on PC
780  name=CRCD(0xef5f3f41,"CompressedName");
781  #else
782  CArray *p_table=GetArray(0x35115a20/*WriteToBuffer_CompressionLookupTable_8*/);
783  name=p_table->GetChecksum(*p_buffer);
784  #endif
785  ++p_buffer;
786  }
787  else if (type&MASK_16_BIT_NAME_LOOKUP)
788  {
789  #ifdef __PLAT_WN32__
790  name=CRCD(0xef5f3f41,"CompressedName");
791  #else
792  CArray *p_table=GetArray(0x25231f42/*WriteToBuffer_CompressionLookupTable_16*/);
793  name=p_table->GetChecksum(Read2Bytes(p_buffer).mUInt);
794  #endif
795  p_buffer+=2;
796  }
797  else
798  {
799  // It's not stored as an index, but as the full 4 byte checksum.
800  name=Read4Bytes(p_buffer).mChecksum;
801  p_buffer+=4;
802  }
803  type=(ESymbolType)( ((uint32)type) & ~(MASK_8_BIT_NAME_LOOKUP | MASK_16_BIT_NAME_LOOKUP) );
804 
805 
806  switch (type)
807  {
808  case ESYMBOLTYPE_INTEGER:
809  p_structure->AddInteger(name,Read4Bytes(p_buffer).mInt);
810  p_buffer+=4;

812  case ESYMBOLTYPE_FLOAT:
813  p_structure->AddFloat(name,Read4Bytes(p_buffer).mFloat);
814  p_buffer+=4;
815  break;

817  p_structure->AddChecksum(name,Read4Bytes(p_buffer).mChecksum);
818  p_buffer+=4;
819  break;

821  p_structure->AddFloat(name,zero_float);
822  break;
823  case ESYMBOLTYPE_ZERO_INTEGER:
824  p_structure->AddInteger(name,0);
825  break;
826  case ESYMBOLTYPE_INTEGER_ONE_BYTE:
827  p_structure->AddInteger(name,*(sint8*)p_buffer++);
828  break;
829  case ESYMBOLTYPE_UNSIGNED_INTEGER_ONE_BYTE:
830  p_structure->AddInteger(name,*p_buffer++);
831  break;
832  case ESYMBOLTYPE_INTEGER_TWO_BYTES:
833  p_structure->AddInteger(name,Read2Bytes(p_buffer).mInt);
834  p_buffer+=2;
835  break;
836  case ESYMBOLTYPE_UNSIGNED_INTEGER_TWO_BYTES:
837  p_structure->AddInteger(name,Read2Bytes(p_buffer).mUInt);
838  p_buffer+=2;
839  break;
840  case ESYMBOLTYPE_STRING:
841  p_string=(const char *)p_buffer;
842  p_structure->AddString(name,p_string);
843  p_buffer+=strlen(p_string);
844  ++p_buffer; // Skip over the terminator too.
845  break;
846  case ESYMBOLTYPE_LOCALSTRING:
847  p_string=(const char *)p_buffer;
848  p_structure->AddLocalString(name,p_string);
849  p_buffer+=strlen(p_string);
850  ++p_buffer; // Skip over the terminator too.
851  break;
852  case ESYMBOLTYPE_PAIR:
853  x=Read4Bytes(p_buffer).mFloat;
854  p_buffer+=4;
855  y=Read4Bytes(p_buffer).mFloat;
856  p_buffer+=4;
857  p_structure->AddPair(name,x,y);
858  break;
859  case ESYMBOLTYPE_VECTOR:
860  x=Read4Bytes(p_buffer).mFloat;
861  p_buffer+=4;
862  y=Read4Bytes(p_buffer).mFloat;
863  p_buffer+=4;
864  z=Read4Bytes(p_buffer).mFloat;
865  p_buffer+=4;
866  p_structure->AddVector(name,x,y,z);
867  break;
868  case ESYMBOLTYPE_STRUCTURE:
869  // Create a new structure, and fill it in by calling this function recursively.
870  p_struct=new CStruct;
871  p_buffer=ReadFromBuffer(p_struct,p_buffer);
872  // Add the new component.
873  p_structure->AddStructurePointer(name,p_struct);
874  break;
875  case ESYMBOLTYPE_ARRAY:
876  // Create a new array, and fill it in using the array's ReadFromBuffer.
877  p_array=new CArray;
878  p_buffer=ReadFromBuffer(p_array,p_buffer);
879  // Add the new component.
880  p_structure->AddArrayPointer(name,p_array);
881  break;
882  default:
883  Dbg_MsgAssert(0,("Unsupported component type of '%s' encountered in ReadFromBuffer",GetTypeName(type)));
884  break;
885  }
886  }
887 
888  Mem::Manager::sHandle().PopContext();
889 
890  return p_buffer;

894 // The information is outputted in a byte-packed format, so p_buffer does
895 // not need to be aligned.

897 // passing it to ReadFromBuffer.
898 // Passed the size of the buffer so that it can check if there is enough space.
899 // Returns the number of bytes that it actually wrote.
900 //
901 // If there was not enough space, and assert is false, it will return a count of 0.
902 //

924  // backwards compatibility.
