150 void CAvoidComponent::play_appropriate_jump_anim( float rotAng, Script::CStruct* pParams )
151 {
152  // split into forward/left/back/right quadrants
153  // by shifting the angle by 45 degrees
154 
155  rotAng += Mth::PI / 4.0f;
156  if ( rotAng < 0 )
157  {
158  rotAng += Mth::PI * 2.0f;
159  }
160 
161  int direction = (int)( rotAng / ( Mth::PI / 2.0f ) );
162  // 0 = forward
163  // 1 = left
164  // 2 = back
165  // 3 = right
166 
167  Dbg_Assert( pParams );
168 
169  uint32 animChecksum;
170  bool flipped;
171 
172  switch ( direction )
173  {
174  case ( 0 ): // JUMP FORWARD
175  case ( 4 ):
176 // Dbg_Message( "Ped jumps forward to avoid skater" );
177  pParams->GetChecksum( CRCD(0x4de2c9e8,"JumpForward"), &animChecksum, true );
178  break;
179 
180  case ( 1 ): // JUMP LEFT
181  {
182 // Dbg_Message( "Ped jumps left to avoid skater" );
183  if ( pParams->GetChecksum( CRCD(0x733da756,"JumpLeft"), &animChecksum, false ) )
184  {
185  flipped = 0;
186  }
187  else
188  {
189  pParams->GetChecksum( CRCD(0xa7a0dd72,"JumpRight"), &animChecksum, true );
190  flipped = 1;
191  }
192 
193  Obj::CAnimationComponent* pAnimationComponent = GetAnimationComponentFromObject( GetObject() );
194  if ( pAnimationComponent )
195  {
196  pAnimationComponent->FlipAnimation( GetObject()->GetID(), flipped, 0, true );
197  }
198  }
199  break;
200 
201  case ( 2 ): // JUMP BACK
202 // Dbg_Message( "Ped jumps back to avoid skater" );
203  pParams->GetChecksum( CRCD(0x64948109,"JumpBack"), &animChecksum, true );
204  break;
205 
206  case ( 3 ): // JUMP RIGHT
207  {
208 // Dbg_Message( "Ped jumps right to avoid skater" );
209  if ( pParams->GetChecksum( CRCD(0xa7a0dd72,"JumpRight"), &animChecksum, false ) )
210  {
211  flipped = 0;
212  }
213  else
214  {
215  pParams->GetChecksum( CRCD(0x733da756,"JumpLeft"), &animChecksum, true );
216  flipped = 1;
217  }
218 
219  Obj::CAnimationComponent* pAnimationComponent = GetAnimationComponentFromObject( GetObject() );
220  if ( pAnimationComponent )
221  {
222  pAnimationComponent->FlipAnimation( GetObject()->GetID(), flipped, 0, true );
223  }
224  }
225  break;
226 
227  default:
228  Dbg_MsgAssert( 0, ( "Ped jumped in unrecognized direction: %d", direction ) );
229  return;
230  break;
231 
232  }
233 
234  Obj::CAnimationComponent* pAnimationComponent = GetAnimationComponentFromObject( GetObject() );
235 
236  if ( pAnimationComponent )
237  {
238  pAnimationComponent->PlaySequence( animChecksum, 0 );
239  m_jumpTime = pAnimationComponent->AnimDuration( animChecksum );
240  }
241  else
242  {
243  Dbg_MsgAssert( 0, ( "No animation component" ) );
244  }
245 }

252 float CAvoidComponent::get_ideal_heading()
253 {
254  Mdl::Skate* skate = Mdl::Skate::Instance();
255 
256  // assume pedestrians only appear in "1 player" game for now...
257  // If this assumption becomes incorrect, cycle through skate->GetNumSkaters( )
258  // and find the closest.
259  CSkater *pSkater = skate->GetLocalSkater();
260  Dbg_MsgAssert( pSkater, ( "Ped avoidance only works in 1-player games." ) );
261 
262  Mth::Vector jumpDir;
263 
264  // find the 2 directions perpendicular to the skater's line of motion...
265  // and select the direction that will take us further away from the
266  // skater
267 
268  jumpDir = (pSkater->GetDisplayMatrix())[Mth::AT];
269  jumpDir[Y] = 0.0f;
270  jumpDir.Normalize( );
271 
272  Mth::Vector upVect( 0.0f, 1.0f, 0.0f, 0.0f );
273  jumpDir = Mth::CrossProduct( jumpDir, upVect );
274 
275  // chose the direction that's farther...
276  Mth::Vector posPlus;
277  Mth::Vector posMinus;
278  posPlus = GetObject()->m_pos + jumpDir;
279  posMinus = GetObject()->m_pos - jumpDir;
280 
281  if ( Mth::DistanceSqr( pSkater->m_pos, posPlus ) < Mth::DistanceSqr( pSkater->m_pos, posMinus ) )
282  {
283  jumpDir[X] = -jumpDir[X];
284  jumpDir[Z] = -jumpDir[Z];
285  }
286 
287  return Mth::RadToDeg( Mth::GetAngle( GetObject()->m_matrix, jumpDir ) );
288 }

355 bool CAvoidComponent::jump( Mth::Matrix& mat0, Mth::Matrix& mat1, float rotAng, float jumpDist, float maxHerdDistance )
356 {
357  mat1 = mat0;
358 
359  mat1.RotateYLocal( Mth::DegToRad(rotAng) );
360 
361  // test for an extra 15 units (1.25 feet) radius of clearance
362  // around the player to account for his actual body
363  // (need to rotate it slightly up, to account for peds
364  // that are below ground level... not sure why this
365  // happens, but will check it out in THPS5...)
366  const float clearanceDist = 15.0f;
367  Mth::Vector collisionTestPos = mat1.Transform( Mth::Vector( 0.0f, 10.0f, jumpDist + clearanceDist ) );
368 
369 #ifdef __NOPT_ASSERT__
370 /*
371  if ( ped_debug_on() )
372  {
373  Gfx::AddDebugCircle( collisionTestPos, 20, 15.0f, 0x00ff00ff, 10 );
374  }
375 */
376 #endif
377 
378  if ( collides_with( mat0.GetPos(), collisionTestPos ) )
379  {
380  Dbg_Message( "Collides with something!" );
381  return false;
382  }
383  else
384  {
385  // build target matrix
386  Mth::Vector targetPos = mat1.Transform( Mth::Vector( 0.0f, 0.0f, jumpDist ) );
387  mat1[Mth::POS] = targetPos;
388 
389  Mdl::Skate* skate = Mdl::Skate::Instance();
390  CSkater *pSkater = skate->GetLocalSkater();
391  Dbg_MsgAssert( pSkater, ( "Ped avoidance only works in 1-player games." ) );
392 
393  Mth::Vector toSkater = targetPos - pSkater->m_pos;
394  if ( toSkater.Length() < 30.0f )
395  {
396  printf( "Too close to skater %f!\n", toSkater.Length() );
397  return false;
398  }
399 
400  // check if we're going too far from the original point
401  // TODO: check for distance from the path, not the point
402  // don't allow going too far from the original path,
403  // or else you'll be able to herd them into weird
404  // positions...
405  Mth::Vector totalAvoidVec = targetPos - m_avoidOriginalPos;
406  totalAvoidVec[Y] = 0.0f;
407 
408  if ( totalAvoidVec.Length() >= maxHerdDistance )
409  {
410  Dbg_Message( "Too far from the original avoid point %f", totalAvoidVec.Length() );
411  return false;
412  }
413  }
414 
415  return true;
416 }

423 CBaseComponent::EMemberFunctionResult CAvoidComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
424 {
425  switch (Checksum)
426  {
427  // @script | Ped_SelectAvoidPoint | Will cause the pedestrian to jump out of
428  // the way of the player. He will land on a circle, JumpDist away from
429  // the last postition stored in Obj_StorePos. Once he lands the jump,
430  // the LandedEscapeJump exception is triggered.
431  // @parmopt name | JumpLeft | jumpLeft | name of anim to play when jumping left
432  // @parmopt name | JumpRight | jumpRight | name of anim to play when jumping right
433  // @parmopt name | JumpForward | jumpForward | name of anim to play when jumping forward
434  // @parmopt name | JumpBack | jumpBack | name of anum to play when jumping back
435  // position where the ped jumps to
436  case 0x6157cbc0: // Ped_SelectAvoidPoint
437  {
438  // this is the starting point...
439  Mth::Matrix mat0;
440  mat0 = GetObject()->GetDisplayMatrix();
441  mat0[Mth::POS] = GetObject()->m_pos;
442 
443  // this is where we will eventually end up
444  Mth::Matrix mat1;
445 
446  float maxHerdDistance = 100.0f;
447  pParams->GetFloat( "maxHerdDistance", &maxHerdDistance, Script::NO_ASSERT );
448 
449  float idealHeading = get_ideal_heading();
450 
451  if ( jump(mat0, mat1, idealHeading, get_jump_distance( pParams, idealHeading ), maxHerdDistance ) )
452  {
453  Dbg_Message( "Selecting ideal jump (ang=%f newpos=(%f, %f, %f)",
454  idealHeading,
455  mat1[Mth::POS][X],
456  mat1[Mth::POS][Y],
457  mat1[Mth::POS][Z] );
458  }
459 
460  // GJ: Try a few more directions before trying
461  // to jump back to the original point...
462  else if ( jump(mat0, mat1, idealHeading + 180.0f, get_jump_distance( pParams, idealHeading + 180.0f ), maxHerdDistance) )
463  {
464  // jump to the other side was successful!
465  printf( "Jumping to the other side...\n" );
466  }
467  else if ( jump(mat0, mat1, idealHeading - 90.0f, get_jump_distance( pParams, idealHeading - 90.0f ), maxHerdDistance) )
468  {
469  // jump away was successful!
470  printf( "Jumping away...\n" );
471  }
472  else
473  {
474  Dbg_Message( "Jumping back towards original point %f %f %f",
475  m_avoidOriginalPos[X],
476  m_avoidOriginalPos[Y],
477  m_avoidOriginalPos[Z] );
478 
479  // otherwise, jump back towards original point
480  Mth::Vector jumpDir;
481  jumpDir = m_avoidOriginalPos - mat0[Mth::POS];
482 
483  float rotAng;
484  rotAng = Mth::GetAngle( GetObject()->m_matrix, jumpDir );
485 
486  bool success = jump( mat0, mat1, rotAng, get_jump_distance(pParams, rotAng), maxHerdDistance );
487 
488  if ( !success )
489  {
490  // jump in place
491  mat1 = mat0;
492 
493  return CBaseComponent::MF_FALSE;
494  }
495  }
496 
497  // don't actually want to rotate the ped
498  // (just translate it)
499  mat1[Mth::RIGHT] = mat0[Mth::RIGHT];
500  mat1[Mth::UP] = mat0[Mth::UP];
501  mat1[Mth::AT] = mat0[Mth::AT];
502 
503  m_avoidInterpolator.setMatrices( &mat0, &mat1 );
504  m_avoidAlpha = 0.0f;
505 
506 #ifdef __NOPT_ASSERT__
507 /*
508  if ( ped_debug_on() )
509  {
510  Gfx::AddDebugCircle( mat0[Mth::POS], 20, 15.0f, 0x00ff00ff, 100 );
511  Gfx::AddDebugCircle( mat1[Mth::POS], 20, 15.0f, 0x00ff00ff, 100 );
512  }
513 */
514 #endif
515 
516  Mth::Vector jumpDir;
517  jumpDir = mat1[Mth::POS] - mat0[Mth::POS];
518 
519  float rotAng;
520  rotAng = Mth::GetAngle( GetObject()->m_matrix, jumpDir );
521 
522  // rotAng of zero means jump forward, so:
523  play_appropriate_jump_anim( rotAng, pParams );
524  }
525  break;
526 
527  // @script | Ped_BackOnOriginalPath |
528  case 0xd2aa4d2d: // Ped_BackOnOriginalPath
529  {
530  float safeDist;
531  safeDist = pParams->GetFloat( "dist", &safeDist, Script::ASSERT );
532 
533  Mth::CBBox theBox;
534  theBox.SetMin( m_avoidOriginalPos - Mth::Vector( safeDist, safeDist, safeDist ) );
535  theBox.SetMax( m_avoidOriginalTargetPos + Mth::Vector( safeDist, safeDist, safeDist ) );
536 
537  if ( theBox.Within( GetObject()->m_pos ) )
538  {
539  printf( "Intersects!\n" );
540  // return CBaseComponent::MF_TRUE;
541  }
542  else
543  {
544  printf( "Doesn't intersect!\n" );
545  return CBaseComponent::MF_FALSE;
546  }
547 
548 #if 0
549  Mth::Vector originalPath;
550  originalPath = m_avoidOriginalTargetPos - m_avoidOriginalPos;
551 
552  Mth::Vector upVect( 0.0f, 1.0f, 0.0f, 0.0f );

593 
594  // @script | Ped_MoveTowardsAvoidPoint |
595  case 0x41226942: // Ped_MoveTowardsAvoidPoint
596  {
597  Mth::Matrix mat0;
598  m_avoidAlpha += ( m_jumpTime * ( 1.0f / 60.0f ) * Tmr::FrameRatio() );

621  if ( pMotionComponent && pMotionComponent->GetPathOb() )

623  m_avoidOriginalTargetPos = pMotionComponent->GetPathOb()->m_wp_pos_to;

628  }
629  }

633  {

636  m_avoidOriginalStickToGround = ( pMotionComponent->m_movingobj_flags |= MOVINGOBJ_FLAG_STICK_TO_GROUND );
637  }
638  break;
639

641  {
642  Obj::CMotionComponent* pMotionComponent = GetMotionComponentFromObject( GetObject() );
643  if ( pMotionComponent )

660

662  // unrecognized functions; if we make it down here,
663  // that means that the component already handled it
664  // somehow
