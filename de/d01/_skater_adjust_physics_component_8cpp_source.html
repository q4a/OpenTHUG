169 void CSkaterAdjustPhysicsComponent::uber_frig ( )
170 {
171  // Uberfrig() will attempt to see if we are in a region with valid collision
172  // if were are not, then it will attempt to move us back to a position that is valid; generally this is the old position
173 
174  m_uber_frigged_this_frame = false;
175 
176  if (GetObject()->m_pos != mp_core_physics_component->m_safe_pos)
177  {
178  #ifdef __NOPT_ASSERT__
179  if (! (Tmr::GetRenderFrame() & 15) )
180  {
181  TrackingLine2(4, GetObject()->m_old_pos, GetObject()->m_pos); // 4 = line
182  }
183  #endif
184 
185  mp_core_physics_component->m_col_start = GetObject()->m_pos;
186  mp_core_physics_component->m_col_end = GetObject()->m_pos;
187 
188  // Very minor adjustment to move origin away from vert walls
189  mp_core_physics_component->m_col_start += GetObject()->m_matrix[Y] * 0.001f;
190 
191  mp_core_physics_component->m_col_start[Y] += 8.0f;
192  mp_core_physics_component->m_col_end[Y] -= FEET(400);
193 
194  if (mp_core_physics_component->get_member_feeler_collision())
195  {
196  // if we are in "contact" with something, and on the ground or in air, then that should be what we collide with
197  // so if not, we need to leave contact, and impart the final velocity
198  // ("on ground" just means skating on something, which could be the moving object
199  if (!mp_core_physics_component->GetFlag(SKITCHING) && !mp_core_physics_component->GetFlag(SPINE_PHYSICS))
200  {
201  // but we leave it if skitching, or attempting to spin between them
202  if (mp_movable_contact_component->HaveContact()
203  && (mp_core_physics_component->GetState() == GROUND || mp_core_physics_component->GetState() == AIR))
204  {
205  if (!mp_core_physics_component->m_feeler.IsMovableCollision()
206  || mp_core_physics_component->m_feeler.GetMovingObject() != mp_movable_contact_component->GetContact()->GetObject())
207  {
208  GetObject()->m_vel += mp_movable_contact_component->GetContact()->GetObject()->GetVel();
209  DUMP_VELOCITY;
210  mp_movable_contact_component->LoseAnyContact();
211  }
212  }
213  }
214 
215  static Mth::Vector xa, xb, xc;
216 
217  float height = GetObject()->m_pos[Y] - mp_core_physics_component->m_feeler.GetPoint()[Y];
218 
219  mp_state_component->m_height = height;
220 
221  // if we are below the ground, then move him up
222  if (height < 0.001f)
223  {
224  GetObject()->m_pos[Y] = mp_core_physics_component->m_feeler.GetPoint()[Y] + 0.001f; // above ground by a fraction of an inch
225  DUMP_POSITION;
226  height = 0.0f;
227  }
228 
229  if (mp_core_physics_component->GetState() != RAIL)
230  {
231  mp_model_component->ApplyLightingFromCollision(mp_core_physics_component->m_feeler);
232  }
233 
234  // Store these values off for the simple shadow calculation.
235  mp_shadow_component->SetShadowPos(mp_core_physics_component->m_feeler.GetPoint());
236  mp_shadow_component->SetShadowNormal(mp_core_physics_component->m_feeler.GetNormal());
237 
238  if (Ed::CParkEditor::Instance()->UsingCustomPark())
239  {
240  // when we are in the park editor, there are other checks we might want to do
241  CFeeler feeler;
242 
243  // ignore non-collidable, under-ok
244  feeler.SetIgnore(mFD_NON_COLLIDABLE | mFD_UNDER_OK, 0);
245  feeler.SetStart(GetObject()->m_pos + Mth::Vector(0.0f, 3000.0f, 0.0f, 0.0f));
246  feeler.SetEnd(GetObject()->m_pos + GetObject()->m_matrix[Y]);
247  if (feeler.GetCollision())
248  {
249  // Something above me that I'm not supposed to be under; just move the skater back to the old pos, and flip him
250 
251  // if only just under it, then pop me up
252  if ((GetObject()->m_pos - feeler.GetPoint()).LengthSqr() < 6.0f * 6.0f)
253  {
254  GetObject()->m_pos = feeler.GetPoint();
255  DUMP_POSITION;
256  }
257  else
258  {
259  // if we are not moving, then pop us up above the face we detected a collision with
260  if ((GetObject()->m_pos - mp_core_physics_component->m_safe_pos).LengthSqr() < 1.0f * 1.0f)
261  {
262  GetObject()->m_pos = feeler.GetPoint();
263  DUMP_POSITION;
264  }
265  else
266  {
267  GetObject()->m_pos = mp_core_physics_component->m_safe_pos;
268  DUMP_POSITION;
269  }
270  GetObject()->m_vel = GetObject()->m_vel * (-1.0f / 2.0f);
271  DUMP_VELOCITY;
272  if (mp_core_physics_component->GetState() == RAIL)
273  {
274  mp_core_physics_component->SetState(AIR); // Stop grinding
275  GetObject()->SelfEvent(CRCD(0xafaa46ba, "OffRail")); // regular exception
276  GetObject()->m_pos[Y] += 2.0f; // make sure we are out of the rail...
277  DUMP_POSITION;
278  }
279  }
280  }
281  }
282 
283  m_nudge = 0;
284 
285  // Mick: Set a safe position. Here there was a collision, so safe
286  mp_core_physics_component->m_safe_pos = GetObject()->m_pos;
287 
288  }
289  else
290  {
291  // I'm going to assume that if we are on a rail, then the uberfrig is just a problem with the collision getting parallel to the wall
292  // so I can safely ignore it, as no rail should be laid out into nothingness
293  // except in the park editor, dammit...
294  if ((mp_core_physics_component->GetState() != RAIL && mp_core_physics_component->GetState() != LIP)
295  || Ed::CParkEditor::Instance()->UsingCustomPark())
296  {
297  m_uber_frigged_this_frame = true;
298 
299  // first nudge is zero
300  static Mth::Vector uberfrig_nudges[] =
301  {
302  Mth::Vector(0.0f, 0.0f, 0.0f),
303  Mth::Vector(0.1f, 0.1f, 0.1f),
304  Mth::Vector(-0.2f, 0.0f, 0.0f),
305  Mth::Vector(0.0f, 0.0f, -0.2f),
306  Mth::Vector(0.2f, 0.0f, 0.0f)
307  };
308 
309  if (m_nudge < 5)
310  {
311  GetObject()->m_pos = mp_core_physics_component->m_safe_pos + uberfrig_nudges[m_nudge++];
312  DUMP_POSITION;
313  }
314  else
315  {
316  // just reset it, and hope for the best. Probably we are just over a seam, and will get over it.
317  m_nudge = 0;
318  }
319 
320  GetObject()->m_vel = -GetObject()->m_vel + Mth::Vector(0.101f, 0.101f, 0.1001f);
321  DUMP_VELOCITY;
322 
323  // special case for wallrides, lip tricks and suchlike
324  // just change them to air state; bit of a patch, but prevent you getting stuck in a state
325  if (mp_core_physics_component->GetState() != GROUND && mp_core_physics_component->GetState() != AIR)
326  {
327  mp_core_physics_component->SetState(AIR);
328  }
329  }
330 
331  mp_state_component->m_height = 0.0f;
332  }
333  }
334 }

341 void CSkaterAdjustPhysicsComponent::check_inside_objects ( )
342 {
343  if (mp_movable_contact_component->CheckInsideObjects(GetObject()->m_pos, mp_core_physics_component->m_safe_pos))
344  {
345  MESSAGE("found to be inside a moving object");
346 
347  // Whichever way we rcover from being inside a moving object, this abrupt change in velocity and position means any rail grinding might
348  // be messed up, either with zero velocity, or being moved away from the rail; so, the best thing to do is skate off the rail
349 
350  if (mp_core_physics_component->GetState() == RAIL)
351  {
352  mp_core_physics_component->SetState(AIR);
353  GetObject()->SelfEvent(CRCD(0xafaa46ba, "OffRail"));
354  }
355  else if (GetObject()->m_pos == mp_core_physics_component->m_safe_pos)
356  {
357  // Set velocity to zero, otherwise we can get stuck inside things as the velocity from this frame will keep pushing you back inside things
358  // especially if m_safe_pos is just inside the object, when there will be no collision detected
359  // If we are on a rail, let them keep the velocity as maybe we can handle it next frame
360  GetObject()->GetVel().Set();
361  }
362  }
363 }
