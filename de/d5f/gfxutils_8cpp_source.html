65 void RGBtoHSV( float r, float g, float b, float *h, float *s, float *v )
66 {
67  float min, max, delta;
68  min = Mth::Min3( r, g, b );
69  max = Mth::Max3( r, g, b );
70  *v = max; // v
71  delta = max - min;
72  if( max != 0.0f )
73  *s = delta / max; // s
74  else {
75  // r = g = b = 0 // s = 0, v is undefined
76  *s = 0.0f;
77  *h = -1.0f;
78  return;
79  }
80 
81  // GJ:
82  if (delta == 0.0f)
83  return;
84 
85  if( r == max )
86  *h = ( g - b ) / delta; // between yellow & magenta
87  else if( g == max )
88  *h = 2.0f + ( b - r ) / delta; // between cyan & yellow
89  else
90  *h = 4.0f + ( r - g ) / delta; // between magenta & cyan
91  *h *= 60.0f; // degrees
92  if( *h < 0.0f )
93  *h += 360.0f;
94 }

101 void HSVtoRGB( float *r, float *g, float *b, float h, float s, float v )
102 {
103  int i;
104  float f, p, q, t;
105  if( s == 0.0f ) {
106  // achromatic (grey)
107  *r = *g = *b = v;
108  return;
109  }
110  h /= 60.0f; // sector 0 to 5
111  i = (int)h; // basically, the floor
112  f = h - i; // factorial part of h
113  p = v * ( 1.0f - s );
114  q = v * ( 1.0f - s * f );
115  t = v * ( 1.0f - s * ( 1.0f - f ) );
116  switch( i ) {
117  case 0:
118  *r = v;
119  *g = t;
120  *b = p;
121  break;
122  case 1:
123  *r = q;
124  *g = v;
125  *b = p;
126  break;
127  case 2:
128  *r = p;
129  *g = v;
130  *b = t;
131  break;
132  case 3:
133  *r = p;
134  *g = q;
135  *b = v;
136  break;
137  case 4:
138  *r = t;
139  *g = p;
140  *b = v;
141  break;
142  default: // case 5:
143  *r = v;
144  *g = p;
145  *b = q;
146  break;
147  }
148 }

207 bool GetScaleFromParams( Mth::Vector* pScaleVector, Script::CStruct* pParams )
208 {
209  if ( pParams->ContainsComponentNamed( CRCD(0x13b9da7b,"scale") ) )
210  {
211  float scaleValue;
212  if ( pParams->GetFloat( CRCD(0x13b9da7b,"scale"), &scaleValue, Script::NO_ASSERT ) )
213  {
214  *pScaleVector = Mth::Vector( scaleValue, scaleValue, scaleValue );
215  return true;
216  }
217  else if ( pParams->GetVector( CRCD(0x13b9da7b,"scale"), pScaleVector, Script::NO_ASSERT ) )
218  {
219  return true;
220  }
221  else
222  {
223  Dbg_MsgAssert( 0, ( "Scale should be either a float or a vector" ) );
224  }
225  }
226 
227  bool xFound = pParams->GetFloat( CRCD(0x7323e97c,"x"), &(*pScaleVector)[X], false );
228  bool yFound = pParams->GetFloat( CRCD(0x424d9ea,"y"), &(*pScaleVector)[Y], false );
229  bool zFound = pParams->GetFloat( CRCD(0x9d2d8850,"z"), &(*pScaleVector)[Z], false );
230 
231  return ( xFound || yFound || zFound );
232 }

264 bool GetTimeFromParams( float* pStart, float* pEnd, float Current, float Duration, Script::CStruct* pParams, Script::CScript* pScript )
265 {
266  float From = 0.0f;
267  float To = Duration;
268 
269  uint32 FromChecksum=0;
270  if ( pParams->GetChecksum( CRCD(0x46e55e8f,"From"), &FromChecksum ) )
271  {
272  switch (FromChecksum)
273  {
274  case 0x6086aa70: // Start
275  From=0;
276  break;
277  case 0xff03cc4e: // End
278  From=Duration;
279  break;
280  case 0x230ccbf4: // Current
281  From=Current;
282  break;
283  case 0x617fe530: // Middle
284  From=Duration / 2.0f;
285  break;
286  default:
287  Dbg_MsgAssert(0,("\n%s\nUnrecognized value '%s' for From in PlayAnim",pScript?pScript->GetScriptInfo():"???",Script::FindChecksumName(FromChecksum)));
288  break;
289  }
290  }
291 
292  uint32 ToChecksum=0;
293  if (pParams->GetChecksum( CRCD(0x28782d3b,"To"), &ToChecksum) )
294  {
295  switch (ToChecksum)
296  {
297  case 0x6086aa70: // Start
298  To=0;
299  break;
300  case 0xff03cc4e: // End
301  To=Duration;
302  break;
303  case 0x230ccbf4: // Current
304  To=Current;
305  break;
306  case 0x617fe530: // Middle
307  To=Duration / 2.0f;
308  break;
309  default:
310  Dbg_MsgAssert(0,("\n%s\nUnrecognized value '%s' for To in PlayAnim",pScript?pScript->GetScriptInfo():"???",Script::FindChecksumName(ToChecksum)));
311  break;
312  }
313  }
314 
315  // also check if From and To were specified as integers, in which case use units of 60ths.
316  float FromFrames=0;
317  if (pParams->GetFloat( CRCD(0x46e55e8f,"From"), &FromFrames ) )
318  {
319  if ( pParams->ContainsFlag( CRCD(0xd029f619,"seconds") ) )
320  {
321  From = FromFrames;
322  }
323  else
324  {
325  From = Gfx::FRAMES_TO_TIME((int)FromFrames);
326  }
327  }
328 
329  float ToFrames=0;
330  if ( pParams->GetFloat( CRCD(0x28782d3b,"To"), &ToFrames ) )
331  {
332  if ( pParams->ContainsFlag( CRCD(0xd029f619,"seconds") ) )
333  {
334  To = ToFrames;
335  }
336  else
337  {
338  To = Gfx::FRAMES_TO_TIME((int)ToFrames);
339  }
340  }
341 
342  if ( pParams->ContainsFlag( CRCD(0xf8cfd515,"Backwards") ) )
343  {
344  float Temp=From;
345  From=To;
346  To=Temp;
347  }
348 
349  *pStart = From;
350  *pEnd = To;
351 
352  return true;
353 }

360 Str::String GetModelFileName(const char* pName, const char* pExtension)
361 {
362  char fullName[512];
363 
364  if ( strstr( pName, "/" ) || strstr( pName, "\\" ) )
365  {
366  Dbg_MsgAssert( strstr( pName, "." ), ( "Filename %s is missing extension", pName ) );
367 
368  sprintf( fullName, "models\\%s", pName );
369  }
370  else
371  {
372  sprintf( fullName, "models\\%s\\%s%s", pName, pName, pExtension );
373  }
374 
375 #ifdef __NOPT_ASSERT__
376  // replace all forward slashes with backslashes
377  char temp[512];
378  strcpy( temp, fullName );
379  char* pString = temp;
380  while ( *pString )
381  {
382  if ( *pString == '/' )
383  {
384  *pString = '\\';
385  }
386  pString++;
387  }
388  // look for double backslashes, which are bad
389  Dbg_MsgAssert( !strstr( temp, "\\\\" ), ( "Filename %s has double backslash", temp ) );
390 #endif
391 
392  return Str::String( fullName );
393 }
