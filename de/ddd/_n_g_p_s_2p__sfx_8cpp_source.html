162 bool PS2Sfx_LoadVAG( PlatformWaveInfo *pInfo, const char *filename, bool loadPerm, float *samplePitch)
163 {
164 
165 
166  uint32 size = 0; // file size
167  int vagAddr = 0; // VAG address in SPU2 RAM, returned to calling function
168  VagHeader vagHeader;
169  char fullname[ FULL_NAME_SIZE ]; // sound names can be HUGE...
170  unsigned int IOPbuffer = 0; // IOP memory buffer
171 
172  Dbg_MsgAssert( strlen( filename ) < FULL_NAME_SIZE - 30,( "Increase buffer for full filename..." ));
173  sprintf(fullname, "sounds\\vag\\%s.vag", filename);
174  //Matt("Loading sound %s\n", fullname);
175 
176  void *fp = File::Open( fullname, "rb" );
177  if ( !fp )
178  {
179  Dbg_MsgAssert( 0,( "LoadSound: Couldn't find file %s", filename ));
180  printf( "load sound failed to find %s\n", filename );
181  return ( false );
182  }
183  File::Seek( fp, 0, SEEK_SET );
184  if ( SIZE_OF_VAGHEADER != File::Read( &vagHeader, 1, SIZE_OF_VAGHEADER, fp ) )
185  {
186  // error reading header...
187  Dbg_MsgAssert( 0,( "Sorry, young lady, but your VAG is too small" ));
188 // Dbg_MsgAssert( 0,( "Not enough bytes to satisfy our VAG" ));
189  }
190  /* this fucked things up...
191  int endType = 0;
192  if ( strncmp( vagHeader.ID, "VAGp", 4 ) == 0 )
193  endType = 0;
194  else
195  endType = 1;
196 
197  size = ( ( endType == 0 ) ? convertEnd( vagHeader.dataSize ) : vagHeader.dataSize );
198  */
199  size = convertEnd( vagHeader.dataSize );
200 
201  Dbg_MsgAssert( size > 0,( "There's nothing worse than an empty VAG." ));
202 
203  //Dbg_Message( "vag header size %d data size %d", SIZE_OF_VAGHEADER, size );
204 
205 #if USE_CLEVER_TRICK_FOR_CLOBBERED_DATA
206  uint8 *buffer = new uint8[ size + ALLIGN_REQUIREMENT + IOP_MIN_TRANSFER ];
207 #else
208  uint8 *buffer = new uint8[ size + ALLIGN_REQUIREMENT ];
209 #endif
210  File::Seek( fp, SIZE_OF_VAGHEADER, SEEK_SET );
211 
212  uint8 *allignedBuffer = ( uint8 * )( ( ( int ) buffer + ( ALLIGN_REQUIREMENT - 1 ) ) & ~( ALLIGN_REQUIREMENT - 1 ) );
213 
214  uint32 readsize = File::Read( allignedBuffer, 1, size, fp );
215  if ( size != readsize)
216  {
217  Dbg_Message( "if non-converted size %d is the correct size of file %s, inconsistent VAG format", vagHeader.dataSize + SIZE_OF_VAGHEADER, filename );
218  Dbg_MsgAssert( 0,( "VAG %s is %d bytes, not as big as she said it would be (%d bytes)", filename, readsize, size ));
219  }
220  File::Close( fp );
221 
222  if ( samplePitch )
223  {
224  *samplePitch = GetPitchValue( &vagHeader );
225  }
226 // Dbg_Message( "Pitch value %f", GetPitchValue( &vagHeader ) );
227 
228  // make sure we've got room for this sound in RAM:
229  // sound data must be 16byte aligned in the SPU, so check for that
230  // also, the IOP seems to transfer only in 128 byte chunks, so let's
231  // pad the buffer if we're writing a permanent sound (which stacks from
232  // the bottom if RAM up) and write the potentially clobbered bytes from
233  // the previously loaded permanent sound into the end of the buffer to
234  // be transfered (clever huh?) NOTE: This didn't seem to work without
235  // flushing the cache... Gives everything in memory time to settle I
236  // guess?
237 
238  if ( loadPerm )
239  {
240 #if USE_CLEVER_TRICK_FOR_CLOBBERED_DATA
241  int i;
242  int overshoot = 0; // need to initialize to zero!
243  static uint8 prevPermBuffer[ IOP_MIN_TRANSFER ];
244  static int prevVagAddr;
245  if ( !SpuRAMUsedPerm )
246  {
247  // don't store nuthin' first time through...
248  prevVagAddr = 0;
249  }
250 #endif
251  // stack the permanent sounds up from the bottom:
252  SpuRAMUsedPerm += ( size + PAD_TO_PREVENT_CLOBBERING );
253  if ( ( END_WAVE_DATA_ADDR - SpuRAMUsedPerm ) & 0x0F )
254  {
255 #ifdef __NOPT_ASSERT__
256  int oldSpuRAM = SpuRAMUsedPerm;
257 #endif
258  SpuRAMUsedPerm += ( ( END_WAVE_DATA_ADDR - SpuRAMUsedPerm ) & 0x0F );
259  Dbg_MsgAssert( SpuRAMUsedPerm > oldSpuRAM,( "Duh asshole." ));
260  }
261  vagAddr = END_WAVE_DATA_ADDR - SpuRAMUsedPerm;
262  Dbg_MsgAssert( !( vagAddr & 0x0F ),( "Fire Matt please." ));
263 
264 #if USE_CLEVER_TRICK_FOR_CLOBBERED_DATA
265  if ( prevVagAddr )
266  {
267  // don't have to do this if this is the first loaded perm sound...
268  // hence condition above...
269 
270  // find out how much of the previous perm data we'd be writing
271  // over, add that to the end of our buffer...
272  if ( size & ( IOP_MIN_TRANSFER - 1 ) )
273  {
274 // Dbg_Message( "size %d over 64-byte boundary", size & ( IOP_MIN_TRANSFER - 1 ) );
275  overshoot = ( IOP_MIN_TRANSFER ) - ( size & ( IOP_MIN_TRANSFER - 1 ) );
276 // Dbg_Message( "overshoot %d", overshoot );
277  int spaceBetween = prevVagAddr - ( vagAddr + size );
278 // Dbg_Message( "space between %d", spaceBetween );
279  int numBytesClobbered = overshoot - spaceBetween;
280  Dbg_MsgAssert( numBytesClobbered <= IOP_MIN_TRANSFER,( "how in the world?" ));
281 // Dbg_Message( "numBytesClobbered = %d", numBytesClobbered );
282  for ( i = 0; i < numBytesClobbered; i++ )
283  {
284  // store the previous buffer at the end of our buffer,
285  allignedBuffer[ size + i + spaceBetween ] = prevPermBuffer[ i ];
286  }
287  }
288  else
289  {
290  overshoot = 0;
291  }
292  size += overshoot;
293  // add on the size of the overshoot, which in the case of permanent sounds
294  // has data from a previous sound (which will be clobbered), written in.
295 // size += overshoot;
296  Dbg_MsgAssert( !( size & ( IOP_MIN_TRANSFER - 1 ) ),( "Fire Matt." ));
297 
298  }
299  for ( i = 0; i < IOP_MIN_TRANSFER; i++ )
300  {
301  // store the beginning of our new buffer into the static buffer
302  // for when the next perm sound loads...
303  prevPermBuffer[ i ] = allignedBuffer[ i ];
304  }
305  prevVagAddr = vagAddr;
306 #endif
307  } // if ( loadPerm )
308  else
309  {
310  if ( ( SpuRAMUsedTemp + BASE_WAVE_DATA_ADDR ) & 0x0F )
311  {
312  SpuRAMUsedTemp += 0x10 - ( ( SpuRAMUsedTemp + BASE_WAVE_DATA_ADDR ) & 0x0F );
313  }
314  vagAddr = BASE_WAVE_DATA_ADDR + SpuRAMUsedTemp;
315  SpuRAMUsedTemp += ( size );
316  }
317 
318  // transfer buffer to IOP, it's a DMA operation so FlushCache() to ensure cache is flushed
319  FlushCache( 0 );
320 
321  // Ken: Instead of borrowing a new chunk of memory from the IOP, (which causes it to often
322  // run out cos we're so tight on memory) use the already existing
323  // gNonAllignedIopBuffer, used by streams and music etc. They have no need of it at this point.
324  //IOPbuffer = ( unsigned int )sceSifAllocIopHeap( size );
325  Dbg_MsgAssert(size<TOTAL_IOP_BUFFER_SIZE_NEEDED + ALLIGN_REQUIREMENT,("Oops! Vag '%s' too big for iop buffer",filename));
326  Dbg_MsgAssert(Pcm::gNonAllignedIopBuffer,("NULL gNonAllignedIopBuffer"));
327  IOPbuffer=Pcm::gNonAllignedIopBuffer;
328 
329  Dbg_MsgAssert( !( IOPbuffer & 3 ),( "This address should be four byte alligned." ));
330  if ( !IOPbuffer )
331  {
332  Dbg_MsgAssert( 0,( "Failed to allocate IOP memory\n" ));
333  return false;
334  }
335 
336  if ( sceSdTransToIOP( allignedBuffer, IOPbuffer, size, 1 ) == -1 )
337  {
338  Dbg_MsgAssert( 0,( "Failed to transfer to IOP\n" ));
339  return false;
340  }
341  // now transfer from IOP memory into SPU2 RAM
342 
343  if ( ( SpuRAMUsedTemp + SpuRAMUsedPerm + IOP_MIN_TRANSFER ) > MAX_SPU_RAM_AVAILABLE )
344  {
345  Dbg_MsgAssert( 0,( "Our VAG space has been RAMed to the hilt" ));
346  }
347 
348  if (!CSpuManager::sVoiceTrans(DMA_CH, SD_TRANS_MODE_WRITE | SD_TRANS_BY_DMA, (void *) IOPbuffer, (uint32 *) vagAddr, size))
349  {
350  Dbg_MsgAssert( 0,( "Failed to Transfer to SPU2 size %d vagAddr %x IOPBuffer addr %x\n", size, vagAddr, IOPbuffer ));
351  }
352 
353  // Ken: Commented this out because now we're using Pcm::gNonAllignedIopBuffer instead.
354  // Now free up IOP and EE memory, check this return value is zero, as documentation (iopservice.txt v1.5) seems to be wrong
355  //if( sceSifFreeIopHeap( (void * ) IOPbuffer ) != 0 )
356  //{
357  // Dbg_MsgAssert( 0,( "Failed to free IOP memory\n" ));
358  //}
359 
360  pInfo->vagAddress = vagAddr;
361  uint16 blockHeader = *( ( uint16 * ) allignedBuffer + SIZE_OF_LEADIN );
362  pInfo->looping = ( blockHeader & BLOCKHEADER_LOOP_BIT ) ? true : false;
363 
364  delete buffer;
365  return ( true );
366 }
