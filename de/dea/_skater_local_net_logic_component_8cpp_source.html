156 int CSkaterLocalNetLogicComponent::sHandleStealMessage( Net::MsgHandlerContext* context )
157 {
158  GameNet::MsgStealMessage* p_msg = (GameNet::MsgStealMessage *) context->m_Msg;
159 
160  if (p_msg->m_GameId != GameNet::Manager::Instance()->GetNetworkGameId()) return Net::HANDLER_MSG_DONE;
161 
162  Obj::CSkater* pCurrSkater = (CSkater*) context->m_Data;
163  Dbg_Assert( pCurrSkater );
164 
165  // For now, just exit out in these cases to avoid a crash
166  Obj::CSkater* pNewSkater = Mdl::Skate::Instance()->GetSkaterById( p_msg->m_NewOwner );
167  if( pNewSkater == NULL )
168  {
169  return Net::HANDLER_CONTINUE;
170  }
171 
172  Obj::CSkater* pOldSkater = Mdl::Skate::Instance()->GetSkaterById( p_msg->m_OldOwner );
173  if( pOldSkater == NULL )
174  {
175  return Net::HANDLER_CONTINUE;
176  }
177 
178  // TODO: Maybe send the color of this skater's graffiti tags
179 
180  if ( pCurrSkater->GetID() == (uint32)p_msg->m_NewOwner )
181  {
182  Script::CStruct* pTempStructure = new Script::CStruct;
183  pTempStructure->Clear();
184  pTempStructure->AddComponent( CRCD(0xa4b08520, "String0"), ESYMBOLTYPE_STRING, pOldSkater->GetDisplayName() );
185  Script::RunScript( CRCD(0xb3ff911, "GraffitiStealYou"), pTempStructure, pCurrSkater );
186  delete pTempStructure;
187  }
188  else if ( pCurrSkater->GetID() == (uint32)p_msg->m_OldOwner )
189  {
190  Script::CStruct* pTempStructure = new Script::CStruct;
191  pTempStructure->Clear();
192  pTempStructure->AddComponent( CRCD(0xa4b08520, "String0"), ESYMBOLTYPE_STRING, pNewSkater->GetDisplayName() );
193  Script::RunScript( CRCD(0x4d7f6ffa, "GraffitiStealOther"), pTempStructure, pCurrSkater );
194  delete pTempStructure;
195  }
196  else
197  {
198  // useless steal message encountered
199  Dbg_Assert( 0 );
200  }
201 
202  return Net::HANDLER_CONTINUE;
203 }

210 int CSkaterLocalNetLogicComponent::get_update_flags()
211 {
212  Net::Client* client;
213  Mth::Vector eulers;
214  short pos[3];
215  short rot[3];
216  char doing_trick, state, terrain, walking, driving;
217  Flags< int > skater_flags;
218  Flags< int > end_run_flags;
219  int i;
220  bool on_server, force_send;
221  int update_flags;
222  sint16 rail_node;
223  Obj::CSkaterEndRunComponent* p_skater_endrun_component;
224 
225  update_flags = 0;
226 
227  client = GameNet::Manager::Instance()->GetClient( GetSkater()->m_skater_number );
228  Dbg_Assert( client );
229 
230  p_skater_endrun_component = GetSkaterEndRunComponentFromObject(GetSkater());
231 
232  on_server = GameNet::Manager::Instance()->OnServer();
233  force_send = on_server || ( !( client->m_FrameCounter % vFORCE_SEND_INTERVAL ));
234 
235  GetObject()->GetDisplayMatrix().GetEulers( eulers );
236 
237  state = (char) mp_state_component->m_state;
238  terrain = (char) mp_state_component->m_terrain;
239  doing_trick = (char) mp_state_component->m_doing_trick;
240  walking = (char) (mp_state_component->m_physics_state == WALKING ? true : false);
241  driving = (char) mp_state_component->m_driving;
242  rail_node = mp_physics_component->GetRailNode();
243 
244  for( i = 0; i < NUM_ESKATERFLAGS; i++ )
245  {
246  skater_flags.Set( i, mp_state_component->m_skater_flags[( ESkaterFlag ) i ].Get() );
247  }
248  end_run_flags = p_skater_endrun_component->GetFlags();
249 
250  for( i = 0; i < 3; i++ )
251  {
252  if( i == Y )
253  {
254  pos[i] = (short) ( GetObject()->m_pos[i] * 4.0f );
255  //pos[i] = GetObject()->m_pos[i];
256  }
257  else
258  {
259  pos[i] = (short) ( GetObject()->m_pos[i] * 2.0f );
260  //pos[i] = GetObject()->m_pos[i];
261  }
262  if(( pos[i] != m_last_sent_pos[i] ) || force_send )
263  {
264  if( i == X )
265  {
266  update_flags |= GameNet::mUPDATE_FIELD_POS_X;
267  }
268  else if( i == Y )
269  {
270  update_flags |= GameNet::mUPDATE_FIELD_POS_Y;
271  }
272  else if( i == Z )
273  {
274  update_flags |= GameNet::mUPDATE_FIELD_POS_Z;
275  }
276  }
277  }
278 
279  for( i = 0; i < 3; i++ )
280  {
281  rot[i] = (short) ( eulers[i] * 4096.0f );
282  if(( rot[i] != m_last_sent_rot[i] ) || force_send )
283  {
284  m_last_sent_rot[i] = rot[i];
285  if( i == X )
286  {
287  update_flags |= GameNet::mUPDATE_FIELD_ROT_X;
288  }
289  else if( i == Y )
290  {
291  update_flags |= GameNet::mUPDATE_FIELD_ROT_Y;
292  }
293  else if( i == Z )
294  {
295  update_flags |= GameNet::mUPDATE_FIELD_ROT_Z;
296  }
297  }
298  }
299 
300  if( ( state != m_last_sent_state ) ||
301  ( doing_trick != m_last_sent_doing_trick ) ||
302  ( terrain != m_last_sent_terrain ) ||
303  ( walking != m_last_sent_walking ) ||
304  ( driving != m_last_sent_driving ) ||
305  ( force_send ))
306  {
307  update_flags |= GameNet::mUPDATE_FIELD_STATE;
308  }
309 
310  if(( skater_flags != m_last_sent_flags ) ||
311  ( end_run_flags != m_last_sent_end_run_flags ) ||
312  ( force_send ))
313  {
314  update_flags |= GameNet::mUPDATE_FIELD_FLAGS;
315  }
316 
317  if( rail_node != m_last_sent_rail )
318  {
319  update_flags |= GameNet::mUPDATE_FIELD_RAIL_NODE;
320  }
321 
322  return update_flags;
323 }

330 void CSkaterLocalNetLogicComponent::network_update ( void )
331 {
332  // update the server with our current state so that he can relay the data to other clients
333 
334  Net::Client* client;
335  Net::MsgMax p_msg;
336  Mth::Vector eulers;
337  short pos[3];
338  short rot[3];
339  char doing_trick, state, terrain, walking, driving;
340  Flags< int > skater_flags;
341  Flags< int > end_run_flags;
342  int i, msg_len;
343  sint16 rail_node;
344  int update_flags;
345  bool on_server, force_send;
346  Net::MsgDesc msg_desc;
347  Net::BitStream stream;
348  Obj::CSkaterEndRunComponent* p_skater_endrun_component;
349 
350  client = GameNet::Manager::Instance()->GetClient( GetSkater()->m_skater_number );
351  Dbg_MsgAssert( client, ( "Could not find client: %d\n", GetSkater()->m_skater_number ));
352 
353  p_skater_endrun_component = GetSkaterEndRunComponentFromObject(GetSkater());
354 
355  on_server = GameNet::Manager::Instance()->OnServer();
356  force_send = on_server || ( !( client->m_FrameCounter % vFORCE_SEND_INTERVAL ));
357 
358  // Send object updates to narrowband servers only every other frame (except for forced-send frames)
359  if( !on_server && !force_send )
360  {
361  Lst::Search< Net::Conn > sh;
362  Net::Conn* server_conn;
363 
364  server_conn = client->FirstConnection( &sh );
365  Dbg_Assert( server_conn );
366 
367  // Only send object updates as often as we send packets
368  if(( client->m_Timestamp - m_last_update_time ) < (unsigned int) server_conn->GetSendInterval())
369  {
370  return;
371  }
372  }
373 
374  m_last_update_time = client->m_Timestamp;
375 
376  stream.SetOutputData( p_msg.m_Data, 1024 );
377 
378  GetObject()->GetDisplayMatrix().GetEulers( eulers );
379 
380  state = (char) mp_state_component->m_state;
381  terrain = (char) mp_state_component->m_terrain;
382  doing_trick = (char) mp_state_component->m_doing_trick;
383  walking = (char) (mp_state_component->m_physics_state == WALKING ? true : false);
384  driving = (char) mp_state_component->m_driving;
385  rail_node = mp_physics_component->GetRailNode();
386 
387  for( i = 0; i < NUM_ESKATERFLAGS; i++ )
388  {
389  skater_flags.Set( i, mp_state_component->m_skater_flags[( ESkaterFlag ) i ].Get() );
390  }
391 
392  end_run_flags = p_skater_endrun_component->GetFlags();
393 
394  // Write out the time for which this state info is valid
395  stream.WriteValue( client->m_Timestamp, sizeof( int ) * 8 );
396  update_flags = get_update_flags();
397  stream.WriteValue( update_flags, 9 );
398 
399  if( on_server )
400  {
401  static Mth::Vector last_pos;
402  Mth::Vector diff;
403 
404  diff = GetObject()->m_pos - last_pos;
405  //Dbg_Printf( "Vel: %f %f %f\n", diff[X], diff[Y], diff[Z] );
406  last_pos = GetObject()->m_pos;
407  }
408 
409 // Write out the object's position as three shorts (fixed-point)
410  for( i = 0; i < 3; i++ )
411  {
412  if( i == Y )
413  {
414  pos[i] = (short) ( GetObject()->m_pos[i] * 4.0f );
415  //pos[i] = GetObject()->m_pos[i];
416  }
417  else
418  {
419  pos[i] = (short) ( GetObject()->m_pos[i] * 2.0f );
420  //pos[i] = GetObject()->m_pos[i];
421  }
422  if( i == X )
423  {
424  if( update_flags & GameNet::mUPDATE_FIELD_POS_X )
425  {
426  stream.WriteValue( pos[i], sizeof( short ) * 8 );
427  //stream.WriteFloatValue( pos[i] );
428  }
429  }
430  else if( i == Y )
431  {
432  if( update_flags & GameNet::mUPDATE_FIELD_POS_Y )
433  {
434  stream.WriteValue( pos[i], sizeof( short ) * 8 );
435  //stream.WriteFloatValue( pos[i] );
436  }
437  }
438  else if( i == Z )
439  {
440  if( update_flags & GameNet::mUPDATE_FIELD_POS_Z )
441  {
442  stream.WriteValue( pos[i], sizeof( short ) * 8 );
443  //stream.WriteFloatValue( pos[i] );
444  }
445  }
446 
447  m_last_sent_pos[i] = pos[i];
448  }
449 
450  // Write out the object's orientation as three short euler angles (fixed-point)
451  for( i = 0; i < 3; i++ )
452  {
453  rot[i] = (short) ( eulers[i] * 4096.0f );
454  if( i == X )
455  {
456  if( update_flags & GameNet::mUPDATE_FIELD_ROT_X )
457  {
458  stream.WriteValue( rot[i], sizeof( short ) * 8 );
459  }
460  }
461  else if( i == Y )
462  {
463  if( update_flags & GameNet::mUPDATE_FIELD_ROT_Y )
464  {
465  stream.WriteValue( rot[i], sizeof( short ) * 8 );
466  }
467  }
468  else if( i == Z )
469  {
470  if( update_flags & GameNet::mUPDATE_FIELD_ROT_Z )
471  {
472  stream.WriteValue( rot[i], sizeof( short ) * 8 );
473  }
474  }
475 
476  m_last_sent_rot[i] = rot[i];
477  }
478 
479  // Write out the skater's 'state'
480  // Write out the skater's 'doing trick' state
481  // Write out the skater's terrain
482  if( update_flags & GameNet::mUPDATE_FIELD_STATE )
483  {
484  char mask;
485 
486  m_last_sent_state = state;
487  m_last_sent_doing_trick = doing_trick;
488  m_last_sent_terrain = terrain;
489  m_last_sent_walking = walking;
490  m_last_sent_driving = driving;
491 
492  mask = state;
493  if( doing_trick )
494  {
495  mask |= GameNet::mDOING_TRICK_MASK;
496  }
497 
498  stream.WriteValue( mask, 4 );
499  stream.WriteValue( terrain, 6 );
500  stream.WriteValue( walking, 1 );
501  stream.WriteValue( driving, 1 );
502  }
503 
504  // Write out the skaters' flags
505  if( update_flags & GameNet::mUPDATE_FIELD_FLAGS )
506  {
507  m_last_sent_flags = skater_flags;
508  m_last_sent_end_run_flags = end_run_flags;
509  stream.WriteValue( skater_flags, 5 );
510  stream.WriteValue( end_run_flags, 3 );
511  }
512 
513  if( update_flags & GameNet::mUPDATE_FIELD_RAIL_NODE )
514  {
515  m_last_sent_rail = rail_node;
516  stream.WriteValue( rail_node, sizeof( sint16 ) * 8 );
517  }
518 
519  stream.Flush();
520  msg_len = stream.GetByteLength();
521 
522  msg_desc.m_Id = GameNet::MSG_ID_OBJ_UPDATE_STREAM;
523  msg_desc.m_Length = msg_len;
524  msg_desc.m_Data = &p_msg;
525  msg_desc.m_Singular = true;
526  msg_desc.m_Priority = Net::NORMAL_PRIORITY + 1;
527  client->EnqueueMessageToServer( &msg_desc );
528 }
