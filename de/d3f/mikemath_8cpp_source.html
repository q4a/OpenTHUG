379 // invert a 4x4 matrix
380 // Md = inv(Ms)
381 // Note: this is not a general matrix invert: it only works for matrices which incorporate rotation and translation.
382 void MatInverse(Mat Md, Mat Ms)
383 {
384  int i,j;
385  Mat InvT, InvR;
386  // set up negative of original translation
387  MatIdentity(InvT);
388  InvT[3][0] = -Ms[3][0];
389  InvT[3][1] = -Ms[3][1];
390  InvT[3][2] = -Ms[3][2];
391  // set up transpose of original rotation
392  MatIdentity(InvR);
393  for (i=0; i<3; i++)
394  for (j=0; j<3; j++)
395  InvR[i][j] = Ms[j][i];
396  // take product
397  MatProduct(Md,InvT,InvR);
398 }
