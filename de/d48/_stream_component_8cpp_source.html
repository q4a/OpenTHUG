20 // $(NGEL)/components/WatchComponent.cpp\

164 // Given the "Checksum" of a script command, then possibly handle it
165 // if it's a command that this component will handle
166 CBaseComponent::EMemberFunctionResult CStreamComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
167 {
168  switch ( Checksum )
169  {
170 
171  case 0xdb05a233: // Obj_PlayStream
172  PlayStream( pParams, pScript );
173  break;
174 
175  case 0xa738aeb8: // Obj_StopStream
176  {
177  // eventually want to be able to specify channel on here...
178  int i;
179  for ( i = 0; i < MAX_STREAMS_PER_OBJECT; i++ )
180  {
181  if ( mStreamingID[ i ] )
182  {
183  Pcm::StopStreamFromID( mStreamingID[ i ] );
184  mStreamingID[ i ] = 0;
185  }
186  }
187  break;
188  }
189 
190 
191  default:
192  return CBaseComponent::MF_NOT_EXECUTED;
193  }
194 
195  // the "default" case of the switch statement handles
196  // unrecognized functions; if we make it down here,
197  // that means that the component already handled it
198  // somehow
199  return CBaseComponent::MF_TRUE;
200 }

282 bool CStreamComponent::PlayStream( Script::CScriptStructure *pParams, Script::CScript *pScript )
283 {
284 
285  if ( !StreamSlotEmpty( ) )
286  {
287 
288  #ifdef __NOPT_ASSERT__
289  printf ("Playing too many streams on object\n");
290  #endif
291 
292  return ( false ); // can only play 3 stream on each object max!!!
293  }
294  uint32 streamChecksum;
295  uint32 emitterChecksum;
296  uint32 dropoffFuncChecksum;
297  uint32 id = 0;
298  float dropoff = 0.0f;
299  float volume = 100.0f;
300  float pitch = 100.0f;
301  int priority = STREAM_DEFAULT_PRIORITY;
302  EDropoffFunc dropoffFunction = DROPOFF_FUNC_STANDARD;
303  pParams->GetFloat( 0xf6a36814, &volume ); // "vol"
304  pParams->GetFloat( 0xd8604126, &pitch ); // "pitch"
305  pParams->GetInteger( 0x9d5923d8, &priority ); // priority
306  pParams->GetChecksum( 0x40c698af, &id ); // id
307  if ( pParams->GetFloat( 0xff2020ec, &dropoff ) ) // "dropoff"
308  {
309  dropoff = FEET_TO_INCHES( dropoff );
310  }
311  if ( pParams->GetChecksum( 0xc6ac50a, &dropoffFuncChecksum ) ) // "dropoff_function"
312  {
313  dropoffFunction = Sfx::GetDropoffFunctionFromChecksum( dropoffFuncChecksum );
314  }
315  if (pParams->GetChecksum( 0x8a7132ce, &emitterChecksum ) ) // "emitter"
316  {
317  mp_emitter = CEmitterManager::sGetEmitter(emitterChecksum);
318  }
319  else
320  {
321  mp_emitter = NULL;
322  }
323  if ( pParams->GetChecksum( NONAME, &streamChecksum ) )
324  {
325  int use_pos_info = 1;
326  pParams->GetInteger( "use_pos_info", &use_pos_info, Script::NO_ASSERT );
327  //printf ("Playing stream 0x%x on object %s\n",streamChecksum,Script::FindChecksumName(GetObject()->GetID()));
328  return ( Pcm::PlayStreamFromObject( this, streamChecksum, dropoff, volume, pitch, priority, use_pos_info, dropoffFunction, id ) );
329  }
330  return ( false );
331 }
