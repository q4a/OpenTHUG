181 void CNgcParticleGlowRibbonTrail::plat_render( void )
182 {
183  NxNgc::sMaterialHeader mat;
184  NxNgc::sMaterialPassHeader pass;
185 
186  // Header.
187  mat.m_checksum = 0xa9db601e; // particle
188  mat.m_passes = 1;
189  mat.m_alpha_cutoff = 1;
190  mat.m_flags = (1<<1); // 2 sided.
191 // mat.m_shininess = 0.0f;
192 
193  // Pass 0.
194  pass.m_texture.p_data = mp_engine_texture;
195  pass.m_flags = ( mp_engine_texture ? (1<<0) : 0 ) | (1<<5) | (1<<6); // textured, clamped.
196  pass.m_filter = 0;
197  pass.m_blend_mode = (unsigned char)m_blend;
198  pass.m_alpha_fix = (unsigned char)m_fix;
199  pass.m_k = 0;
200  pass.m_color.r = 128;
201  pass.m_color.g = 128;
202  pass.m_color.b = 128;
203  pass.m_color.a = 255;
204 
205  NxNgc::multi_mesh( &mat, &pass, true, true );
206 
207  GX::SetTexCoordGen( GX_TEXCOORD0, GX_TG_MTX2x4, GX_TG_TEX0, GX_FALSE, GX_PTIDENTITY );
208  GX::SetCurrMtxPosTex03( GX_PNMTX0, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY, GX_IDENTITY );
209 
210  // Draw the ribbontrail.
211 
212  // Used to figure the right and up vectors for creating screen-aligned particle quads.
213  NsMatrix* p_matrix = &NxNgc::EngineGlobals.camera;
214 
215  // Concatenate p_matrix with the emmission angle to create the direction.
216  Mth::Vector up( 0.0f, 1.0f, 0.0f, 0.0f );
217 
218  // Get the 'right' vector as the cross product of camera 'at and world 'up'.
219  Mth::Vector at( p_matrix->getAtX(), p_matrix->getAtY(), p_matrix->getAtZ(), 0.0f );
220  Mth::Vector screen_right = Mth::CrossProduct( at, up );
221  Mth::Vector screen_up = Mth::CrossProduct( screen_right, at );
222 
223  screen_right.Normalize();
224  screen_up.Normalize();
225 
226  int lp;
227  CParticleEntry *p_particle;
228  float *p_v;
229 
230  GXColor color[3];
231  GXColor *p_col0;
232  GXColor *p_col1;
233 
234  GX::SetVtxDesc( 2, GX_VA_POS, GX_DIRECT, GX_VA_CLR0, GX_DIRECT );
235 
236  for ( lp = 0, p_particle = mp_particle_array; lp < m_num_particles; lp++, p_particle++ )
237  {
238  float terp = p_particle->m_time / p_particle->m_life;
239 
240  Mth::Vector pos[2];
241  p_v = &mp_vertices[0][lp*3];
242  pos[0].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
243  p_v = &mp_vertices[1][lp*3];
244  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
245 
246  Mth::Vector part_vec = pos[1] - pos[0];
247  Mth::Vector perp_vec = Mth::CrossProduct( part_vec, at );
248  perp_vec.Normalize();
249 
250  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
251  float h = p_particle->m_sh + ( ( p_particle->m_eh - p_particle->m_sh ) * terp );
252 
253  Mth::Vector tmp[4];
254 
255  if ( m_mid_time >= 0.0f )
256  {
257  if ( terp < m_mid_time )
258  {
259  p_col0 = &m_start_color[3];
260  p_col1 = &m_mid_color[3];
261  // Adjust interpolation for this half of the color blend.
262  terp = terp / m_mid_time;
263  }
264  else
265  {
266  p_col0 = &m_mid_color[3];
267  p_col1 = &m_end_color[3];
268  // Adjust interpolation for this half of the color blend.
269  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
270  }
271  }
272  else
273  {
274  // No mid color specified.
275  p_col0 = &m_start_color[3];
276  p_col1 = &m_end_color[3];
277  }
278 
279  GXColor start = *p_col0++;
280  GXColor end = *p_col1++;
281 
282  color[0].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
283  color[0].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
284  color[0].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
285  color[0].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
286 
287  tmp[0] = pos[0] + ( perp_vec * w * m_split );
288  tmp[1] = pos[0] - ( perp_vec * w * m_split );
289 
290  GX::Begin( GX_QUADS, GX_VTXFMT0, 4 * ( m_num_vertex_buffers - 1 ) );
291 
292  for ( int c = 1; c < m_num_vertex_buffers; c++ )
293  {
294  start = *p_col0++;
295  end = *p_col1++;
296 
297  color[1].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
298  color[1].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
299  color[1].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
300  color[1].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
301 
302  if ( c > 1 )
303  {
304  p_v = &mp_vertices[c][lp*3];
305  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
306  part_vec = pos[1] - pos[0];
307  perp_vec = Mth::CrossProduct( part_vec, at );
308  perp_vec.Normalize();
309  }
310 
311  tmp[2] = pos[1] - ( perp_vec * w * m_split );
312  tmp[3] = pos[1] + ( perp_vec * w * m_split );
313 
314  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
315  GX::Color1u32( *((uint32*)&color[0]) );
316 
317  GX::Position3f32( tmp[1][X], tmp[1][Y], tmp[1][Z] );
318  GX::Color1u32( *((uint32*)&color[0]) );
319 
320  GX::Position3f32( tmp[2][X], tmp[2][Y], tmp[2][Z] );
321  GX::Color1u32( *((uint32*)&color[1]) );
322 
323  GX::Position3f32( tmp[3][X], tmp[3][Y], tmp[3][Z] );
324  GX::Color1u32( *((uint32*)&color[1]) );
325 
326  color[0] = color[1];
327  pos[0] = pos[1];
328  tmp[0] = tmp[3];
329  tmp[1] = tmp[2];
330  }
331 
332  GX::End();
333 
334  // Draw the glow.
335 
336  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
337  p_v = &mp_vertices[0][lp*3];
338  pos[0].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
339  Mth::Vector ss_right, ss_up; //, ss_pos;
340 
341  ss_right = screen_right * w;
342  ss_up = screen_up * h;
343 
344  if ( m_mid_time >= 0.0f )
345  {
346  if ( terp < m_mid_time )
347  {
348  p_col0 = m_start_color;
349  p_col1 = m_mid_color;
350  // Adjust interpolation for this half of the color blend.
351  terp = terp / m_mid_time;
352  }
353  else
354  {
355  p_col0 = m_mid_color;
356  p_col1 = m_end_color;
357  // Adjust interpolation for this half of the color blend.
358  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
359  }
360  }
361  else
362  {
363  // No mid color specified.
364  p_col0 = m_start_color;
365  p_col1 = m_end_color;
366  }
367 
368  for ( int c = 0; c < 3; c++ )
369  {
370  GXColor start = *p_col0++;
371  GXColor end = *p_col1++;
372 
373  color[c].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
374  color[c].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
375  color[c].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
376  color[c].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
377  }
378 
379  tmp[0] = pos[0];
380  tmp[0] += ss_right * sinf( Mth::DegToRad( 0.0f ) ) * m_split;
381  tmp[0] += ss_up * cosf( Mth::DegToRad( 0.0f ) ) * m_split;
382 
383  tmp[2] = pos[0];
384  tmp[2] += ss_right * sinf( Mth::DegToRad( 0.0f ) );
385  tmp[2] += ss_up * cosf( Mth::DegToRad( 0.0f ) );
386 
387  for ( int lp2 = 0; lp2 < m_num_segments; lp2++ )
388  {
389  tmp[1] = pos[0];
390  tmp[1] += ss_right * sinf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) ) * m_split;
391  tmp[1] += ss_up * cosf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) ) * m_split;
392 
393  tmp[3] = pos[0];
394  tmp[3] += ss_right * sinf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) );
395  tmp[3] += ss_up * cosf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) );
396 
397  GX::Begin( GX_TRIANGLESTRIP, GX_VTXFMT0, 5 );
398 
399  GX::Position3f32( pos[0][X], pos[0][Y], pos[0][Z] );
400  GX::Color1u32( *((uint32*)&color[0]) );
401 
402  GX::Position3f32( tmp[0][X], tmp[0][Y], tmp[0][Z] );
403  GX::Color1u32( *((uint32*)&color[1]) );
404 
405  GX::Position3f32( tmp[1][X], tmp[1][Y], tmp[1][Z] );
406  GX::Color1u32( *((uint32*)&color[1]) );
407 
408  GX::Position3f32( tmp[2][X], tmp[2][Y], tmp[2][Z] );
409  GX::Color1u32( *((uint32*)&color[2]) );
410 
411  GX::Position3f32( tmp[3][X], tmp[3][Y], tmp[3][Z] );
412  GX::Color1u32( *((uint32*)&color[2]) );
413 
414  GX::End();
415 
416  tmp[0] = tmp[1];
417  tmp[2] = tmp[3];
418  }
419  }
