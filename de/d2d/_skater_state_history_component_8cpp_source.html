140 void CSkaterStateHistoryComponent::CollideWithOtherSkaters( int start_index )
141 {
142  GameNet::PlayerInfo* p_player;
143 
144  p_player = GameNet::Manager::Instance()->GetPlayerByObjectID(GetObject()->GetID());
145  if( p_player == NULL )
146  {
147  return;
148  }
149 
150  // If we've marked you as non-collidable, you are exempt from collision
151  // In koth, if collision is turned off, only the king is collidable
152  if (p_player->IsNonCollidable()
153  || (!GameNet::Manager::Instance()->PlayerCollisionEnabled() && !p_player->IsKing() && !p_player->HasCTFFlag())) return;
154 
155  CSkaterStateComponent* p_my_state_component = GetSkaterStateComponentFromObject(GetSkater());
156  Dbg_Assert(p_my_state_component);
157 
158  // If this p_player is bailing or noncollidable (and he's NOT the king) he is exempt from being smacked down
159  // If you're in a slap game, you can always be hit; i.e. you've just been slapped but haven't teleported yet
160  bool can_fall = !p_my_state_component->GetFlag(IS_BAILING) || CFuncs::ScriptInSlapGame(NULL, NULL);
161 
162  // However, fallen kings are vulnerable
163  if (!can_fall && !p_player->IsKing() && !p_player->HasCTFFlag()) return;
164 
165  bool my_driving = p_my_state_component->GetDriving();
166  Mth::Vector my_pos = get_latest_position();
167  Mth::Vector my_vel = get_vel();
168  Mth::Vector my_dir = my_vel;
169  my_dir.Normalize();
170 
171  Mth::Line my_line;
172  my_line.m_start = my_pos;
173 
174  // Loop through all other skaters and check for collisions
175  for (int i = start_index; i < Mdl::Skate::vMAX_SKATERS; i++)
176  {
177  CSkater* p_other_skater = Mdl::Skate::Instance()->GetSkater(i);
178 
179  if (!p_other_skater
180  || p_other_skater == GetSkater()) continue;
181 
182  CSkaterStateComponent* p_other_state_component = GetSkaterStateComponentFromObject(p_other_skater);
183  Dbg_Assert(p_other_state_component);
184  if (p_other_state_component->GetFlag(IS_BAILING)) continue;
185 
186  GameNet::PlayerInfo* p_other_player = p_other_player = GameNet::Manager::Instance()->GetPlayerByObjectID(p_other_skater->GetID());
187  Dbg_Assert(p_other_player);
188 
189  // Non-Collidable people and kings should never win
190  if (p_other_player->IsNonCollidable() || p_other_player->IsKing()) continue;
191 
192  // If both players are carrying flags, and the subject doesn't have the other guy's flag, don't let them collide
193  if (p_player->HasCTFFlag() && p_other_player->HasCTFFlag() && p_player->HasWhichFlag() != p_other_player->m_Team) continue;
194 
195  // No smacking teammates
196  if (Mdl::Skate::Instance()->GetGameMode()->IsTeamGame() && p_other_player->m_Team == p_player->m_Team) continue;
197 
198  bool other_driving = p_other_state_component->GetDriving();
199  Mth::Vector other_pos = p_other_skater->mp_skater_state_history_component->get_latest_position();
200  Mth::Vector other_vel = p_other_skater->mp_skater_state_history_component->get_vel();
201  Mth::Vector other_dir = other_vel;
202  other_dir.Normalize();
203 
204  if( other_vel.Length() > 100.0f )
205  {
206  continue;
207  }
208 
209  bool collided = false;
210 
211  // Collision extents are based upon the "other" skater -- whether or not he's a remote skater
212 
213  float collide_dist = get_collision_cylinder_radius(my_driving, other_driving);
214  float driving_radius_multiplier = 1.0f;
215  if (my_driving)
216  {
217  driving_radius_multiplier += Script::GetFloat("driving_radius_boost");
218  }
219  if (other_driving)
220  {
221  driving_radius_multiplier += Script::GetFloat("driving_radius_boost");
222  }
223 
224  my_line.m_end = my_line.m_start + my_dir * get_collision_cylinder_coeff(my_driving);
225 
226  Mth::Line other_line;
227  other_line.m_start = other_pos;
228  other_line.m_end = other_line.m_start + other_dir * p_other_skater->mp_skater_state_history_component->get_collision_cylinder_coeff(other_driving);
229 
230  float temp;
231  Mth::Vector my_pt, other_pt;
232  if (Mth::LineLineIntersect(my_line, other_line, &my_pt, &other_pt, &temp, &temp, true))
233  {
234  if ((my_pt - other_pt).Length() < collide_dist)
235  {
236  collided = true;
237  }
238  }
239  else
240  {
241  // No solution exists -- lines must be parallel. Try testing endpoint lengths
242  // Note: This only really works if velocities are relatively small
243  if( (other_line.m_start - my_line.m_start).Length() < collide_dist)
244  {
245  collided = true;
246  other_pt = other_line.m_start;
247  my_pt = my_line.m_start;
248  }
249  else if ((other_line.m_start - my_line.m_end).Length() < collide_dist)
250  {
251  collided = true;
252  other_pt = other_line.m_start;
253  my_pt = my_line.m_end;
254  }
255  else if ((other_line.m_end - my_line.m_start).Length() < collide_dist)
256  {
257  collided = true;
258  other_pt = other_line.m_end;
259  my_pt = my_line.m_start;
260  }
261  else if ((other_line.m_end - my_line.m_end).Length() < collide_dist)
262  {
263  collided = true;
264  other_pt = other_line.m_end;
265  my_pt = my_line.m_end;
266  }
267  }
268 
269  if (!collided) continue;
270 
271  // If the "other" p_player is going faster or if
272  // the subject p_player is king or if
273  // the subject is skating and we're driving, the subject loses
274  if (((my_driving == other_driving && my_vel.Length() > other_vel.Length())
275  || (my_driving && !other_driving)) && !p_player->IsKing()) continue;
276 
277  //Dbg_Printf( "**** My Vel: %.2f, Theirs: %.2f\n", my_vel.Length(), other_vel.Length() );
278  Net::Server* p_server = GameNet::Manager::Instance()->GetServer();
279  Dbg_Assert(p_server);
280 
281  if (can_fall)
282  {
283  GameNet::MsgCollideLost lost_msg;
284  GameNet::MsgByteInfo won_msg;
285  Net::MsgDesc msg_desc;
286 
287  // basically a one-byte message explaining "You've been knocked down by ID"
288  lost_msg.m_Id = p_other_skater->GetID();
289  lost_msg.m_Id |= (1 << 7) * other_driving;
290  lost_msg.m_Offset = my_pt - other_pt;
291 
292  if (CInputComponent* p_input_component = GetInputComponentFromObject(GetObject()))
293  {
294  p_input_component->DisableInput();
295  }
296  p_player->MarkAsNonCollidable();
297 
298  msg_desc.m_Id = GameNet::MSG_ID_SKATER_COLLIDE_LOST;
299  msg_desc.m_Data = &lost_msg;
300  msg_desc.m_Length = sizeof(GameNet::MsgCollideLost);
301  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
302  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
303  p_server->EnqueueMessage(p_player->GetConnHandle(), &msg_desc);
304 
305  // basically a one-byte message explaining "You knocked someone down"
306  won_msg.m_Data = GetObject()->GetID();
307  won_msg.m_Data |= (1 << 7) * my_driving;
308 
309  msg_desc.m_Id = GameNet::MSG_ID_SKATER_COLLIDE_WON;
310  msg_desc.m_Data = &won_msg;
311  msg_desc.m_Length = sizeof(GameNet::MsgByteInfo);
312  p_server->EnqueueMessage(p_other_player->GetConnHandle(), &msg_desc);
313 
314  if (CFuncs::ScriptInSlapGame(NULL, NULL))
315  {
316  Dbg_Assert(p_other_skater->mp_skater_score_component);
317  Mdl::Score* score = p_other_skater->mp_skater_score_component->GetScore();
318  score->SetTotalScore(score->GetTotalScore() + 1);
319  }
320  }
321 
322  if (Mdl::Skate::Instance()->GetGameMode()->GetNameChecksum() == CRCD(0x6ef8fda0, "netking")
323  || Mdl::Skate::Instance()->GetGameMode()->GetNameChecksum() == CRCD(0x5d32129c, "king"))
324  {
325  // Don't bother switching kings if the game is over
326  if (Mdl::Skate::Instance()->GetGameMode()->EndConditionsMet()) return;
327 
328  // If the king was just slapped, the "slapper" (hehe) is the new king
329  if (!p_player->IsKing()) continue;
330 
331  // It is important that we mark the king immediately (rather than through a
332  // network message) since we do logic based on the "current" king
333  p_other_player->MarkAsKing(true);
334 
335  Lst::Search< GameNet::PlayerInfo > search;
336  for (GameNet::PlayerInfo* p_player = GameNet::Manager::Instance()->FirstPlayerInfo(search, true);
337  p_player;
338  p_player = GameNet::Manager::Instance()->NextPlayerInfo(search, true))
339  {
340  // Already marked the king for the local p_player (above)
341  if (p_player->IsLocalPlayer()) continue;
342 
343  GameNet::MsgByteInfo msg;
344  Net::MsgDesc msg_desc;
345 
346  msg.m_Data = p_other_skater->GetID();
347  msg_desc.m_Data = &msg;
348  msg_desc.m_Id = GameNet::MSG_ID_NEW_KING;
349  msg_desc.m_Length = sizeof(GameNet::MsgByteInfo);
350  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
351  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
352  p_server->EnqueueMessage(p_player->GetConnHandle(), &msg_desc);
353  }
354  }
355  else if (Mdl::Skate::Instance()->GetGameMode()->GetNameChecksum() == CRCD(0x6c5ff266, "netctf"))
356  {
357  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
358  // Don't bother handling this if the game is over
359  //if (Mdl::Skate::Instance()->GetGameMode()->EndConditionsMet())
360  if( gamenet_man->GameIsOver())
361  {
362  return;
363  }
364 
365  if (!p_player->HasCTFFlag())
366  {
367  continue;
368  }
369 
370  GameNet::MsgFlagMsg msg;
371  Net::MsgDesc msg_desc;
372 
373  msg.m_ObjId = p_other_skater->GetID();
374  msg.m_Team = p_player->HasWhichFlag();
375 
376  bool retrieve = p_player->HasWhichFlag() == p_other_player->m_Team;
377 
378  Lst::Search< GameNet::PlayerInfo > search;
379  for (GameNet::PlayerInfo* send_player = GameNet::Manager::Instance()->FirstPlayerInfo(search, true);
380  send_player;
381  send_player = GameNet::Manager::Instance()->NextPlayerInfo(search, true))
382  {
383  // If you knock down the p_player with your team's flag, you recover it. Otherwise, you steal it
384  if (retrieve)
385  {
386  if (send_player->IsLocalPlayer())
387  {
388  p_other_player->RetrievedFlag();
389  continue;
390  }
391  msg_desc.m_Data = &msg;
392  msg_desc.m_Length = sizeof(GameNet::MsgFlagMsg);
393  msg_desc.m_Id = GameNet::MSG_ID_RETRIEVED_FLAG;
394  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
395  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
396  p_server->EnqueueMessage(send_player->GetConnHandle(), &msg_desc);
397  }
398  else
399  {
400  if (send_player->IsLocalPlayer())
401  {
402  p_other_player->StoleFlag(p_player->HasWhichFlag());
403  continue;
404  }
405  msg_desc.m_Data = &msg;
406  msg_desc.m_Length = sizeof(GameNet::MsgFlagMsg);
407  msg_desc.m_Id = GameNet::MSG_ID_STOLE_FLAG;
408  msg_desc.m_Queue = Net::QUEUE_SEQUENCED;
409  msg_desc.m_GroupId = GameNet::vSEQ_GROUP_PLAYER_MSGS;
410  p_server->EnqueueMessage(send_player->GetConnHandle(), &msg_desc);
411  }
412  }
413  }
414  }

422 int CSkaterStateHistoryComponent::sHandleProjectileHit ( Net::MsgHandlerContext* context )
423 {
424  CSkater* p_skater = (CSkater*) (context->m_Data);
425  GameNet::MsgProjectileHit* p_msg = (GameNet::MsgProjectileHit*) (context->m_Msg);
426 
427  switch (context->m_MsgId)
428  {
429  case GameNet::MSG_ID_SKATER_PROJECTILE_HIT_TARGET:
430  {
431  char name[32] = "Someone";
432 
433  if (GameNet::PlayerInfo* p_other_player = GameNet::Manager::Instance()->GetPlayerByObjectID(p_msg->m_Id))
434  {
435  strncpy(name, p_other_player->m_Skater->GetDisplayName(), 32);
436  }
437 
438  Script::CStruct params;
439  params.AddString(CRCD(0xa4b08520, "String0"), name);
440  params.AddChecksum( NONAME, CRCD(0xd039432c,"fireball") );
441  p_skater->SelfEvent(CRCD(0xa1021af0, "MadeOtherSkaterBail"),&params);
442 
443  break;
444  }
445 
446  case GameNet::MSG_ID_SKATER_HIT_BY_PROJECTILE:
447  {
448  char name[32] = "Someone";
449 
450  if (GameNet::PlayerInfo* p_other_player = GameNet::Manager::Instance()->GetPlayerByObjectID(p_msg->m_Id))
451  {
452  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
453  Mth::Vector offset = p_skater->GetPos() - p_other_player->m_Skater->GetPos();
454  // Perform a "safe" normalization
455  float offset_length = offset.Length();
456  if (offset_length < 0.0001f)
457  {
458  // if skaters are super close, or coincident then pretend they are not, to avoid zero length vectors
459  offset.Set(1.0f, 0.0f, 1.0f);
460  }
461  else
462  {
463  offset /= offset_length;
464  }
465 
466  // fudge it for now
467  // float speed = p_other_skater->GetVel().Length();
468  float speed = 300.0f;
469  p_skater->GetVel() = offset * speed * 1.5f;
470  p_skater->GetVel()[Y] = 200.0f;
471 
472  CSkaterPhysicsControlComponent* p_skater_physics_control_component = GetSkaterPhysicsControlComponentFromObject(p_skater);
473  if (p_skater_physics_control_component->IsSkating())
474  {
475  CSkaterCorePhysicsComponent* p_skater_core_physics_component = GetSkaterCorePhysicsComponentFromObject(p_skater);
476  p_skater_core_physics_component->CollideWithOtherSkaterLost(p_other_player->m_Skater);
477  }
478  else
479  {
480  CWalkComponent* p_walk_component = GetWalkComponentFromObject(p_skater);
481  p_walk_component->CollideWithOtherSkaterLost(p_other_player->m_Skater);
482  }
483 
484  strncpy(name, p_other_player->m_Skater->GetDisplayName(), 32);
485 
486 
487  if( gamenet_man->OnServer() == false )
488  {
489  Mdl::Score* score = p_skater->mp_skater_score_component->GetScore();
490  if(( score->GetTotalScore() - p_msg->m_Damage ) < 0 )
491  {
492  score->SetTotalScore( 0 );
493  }
494  else
495  {
496  score->SetTotalScore( score->GetTotalScore() - p_msg->m_Damage );
497  }
498  }
499  }
500 
501  Script::CStruct params;
502  params.AddString(CRCD(0xa4b08520, "String0"), name);
503  params.AddChecksum( NONAME, CRCD(0xd039432c,"fireball") );
504  p_skater->SelfEvent(CRCD(0x915e5e39, "SkaterCollideBail"),&params);
505  break;
506  }
507 
508  default:
509  Dbg_Assert(false);
510  break;
511  }
512 
513  return Net::HANDLER_MSG_DONE;
514 }

521 int CSkaterStateHistoryComponent::sHandleCollision( Net::MsgHandlerContext* context )
522 {
523  CSkater* p_skater = (CSkater*) (context->m_Data);
524 
525  switch (context->m_MsgId)
526  {
527  case GameNet::MSG_ID_SKATER_COLLIDE_WON:
528  {
529  GameNet::MsgByteInfo* p_msg = (GameNet::MsgByteInfo*) (context->m_Msg);
530  char name[32] = "Someone";
531 
532  bool loser_drivng = p_msg->m_Data & (1 << 7);
533  p_msg->m_Data &= ~(1 << 7);
534 
535  if (GameNet::PlayerInfo* p_other_player = GameNet::Manager::Instance()->GetPlayerByObjectID(p_msg->m_Data))
536  {
537  strncpy(name, p_other_player->m_Skater->GetDisplayName(), 32);
538  }
539 
540  Script::CStruct params;
541  params.AddString(CRCD(0xa4b08520, "String0"), name);
542  if (loser_drivng)
543  {
544  params.AddChecksum(NO_NAME, CRCD(0x7fd0663d, "LoserIsDriving"));
545  }
546  p_skater->SelfEvent(CRCD(0xa1021af0, "MadeOtherSkaterBail"),&params);
547 
548  break;
549  }
550 
551  case GameNet::MSG_ID_SKATER_COLLIDE_LOST:
552  {
553  GameNet::MsgCollideLost* p_msg = (GameNet::MsgCollideLost*) (context->m_Msg);
554  char name[32] = "Someone";
555 
556  bool winner_driving = p_msg->m_Id & (1 << 7);
557  p_msg->m_Id &= ~(1 << 7);
558 
559  if (GameNet::PlayerInfo* p_other_player = GameNet::Manager::Instance()->GetPlayerByObjectID(p_msg->m_Id))
560  {
561  CSkaterPhysicsControlComponent* p_skater_physics_control_component = GetSkaterPhysicsControlComponentFromObject(p_skater);
562 
563  if (!p_skater_physics_control_component->IsDriving())
564  {
565  Mth::Vector offset = p_skater->GetPos() - p_other_player->m_Skater->GetPos();
566  // Perform a "safe" normalization
567  float offset_length = offset.Length();
568  if (offset_length < 0.0001f)
569  {
570  // if skaters are super close, or coincident then pretend they are not, to avoid zero length vectors
571  offset.Set(1.0f, 0.0f, 1.0f);
572  }
573  else
574  {
575  offset /= offset_length;
576  }
577 
578  // fudge it for now
579  // float speed = p_other_skater->GetVel().Length();
580  float speed = 300.0f;
581  p_skater->GetVel() = offset * speed * 1.5f;
582  p_skater->GetVel()[Y] = 200.0f;
583 
584  if (p_skater_physics_control_component->IsSkating())
585  {
586  CSkaterCorePhysicsComponent* p_skater_core_physics_component = GetSkaterCorePhysicsComponentFromObject(p_skater);
587  p_skater_core_physics_component->CollideWithOtherSkaterLost(p_other_player->m_Skater);
588  }
589  else
590  {
591  CWalkComponent* p_walk_component = GetWalkComponentFromObject(p_skater);
592  p_walk_component->CollideWithOtherSkaterLost(p_other_player->m_Skater);
593  }
594  }
595 
596  strncpy(name, p_other_player->m_Skater->GetDisplayName(), 32);
597  }
598 
599  Script::CStruct params;
600  params.AddString(CRCD(0xa4b08520, "String0"), name);
601  params.AddVector(CRCD(0xa6f5352f, "Offset"), p_msg->m_Offset);
602  if (winner_driving)
603  {
604  params.AddChecksum(NO_NAME, CRCD(0x2f679251, "WinnerIsDriving"));
605  }
606  p_skater->SelfEvent(CRCD(0x915e5e39, "SkaterCollideBail"),&params);
607  break;
608  }
609 
610  default:
611  Dbg_Assert(false);
612  break;
613 
614  }
