111 void set_up( void )
112 {
113 
114 
115  spClassNodeFreeList = (ClassNodeBlock*)_dbg_start;
116 
117  ClassNodeBlock* p_class_node = spClassNodeFreeList;
118 
119  for( int i = 0; i < ( vMAX_CLASS_NODES - 1 ); i++ )
120  {
121  p_class_node->free_node.id = vFREE_CLASS_BLOCK_ID;
122  p_class_node->free_node.next = (FreeNode*)(++p_class_node);
123  }
124 
125  p_class_node->free_node.id = vFREE_CLASS_BLOCK_ID;
126  (p_class_node++)->free_node.next = (FreeNode*)spClassNodeFreeList;
127 
128 
129  spInstanceNodeFreeList = (InstanceNodeBlock*)(++p_class_node);
130  InstanceNodeBlock* p_inst_node = spInstanceNodeFreeList;
131 
132  for( int i = 0; i < ( vMAX_INSTANCE_NODES - 1 ) ; i++ )
133  {
134  p_inst_node->free_node.id = vFREE_INST_BLOCK_ID;
135  p_inst_node->free_node.next = (FreeNode*)(++p_inst_node);
136  }
137  p_inst_node->free_node.id = vFREE_INST_BLOCK_ID;
138  (p_inst_node++)->free_node.next = (FreeNode*)spInstanceNodeFreeList;
139 
140 
141  Dbg_MsgAssert( (int)p_inst_node <= (int)_dbg_end,
142  "Dbg Mem Block not big enough (%d bytes too small)", (int)p_inst_node - (int)_dbg_end );
143 
144  if ( (int)p_inst_node < ((int)_dbg_end))
145  {
146  Dbg_Warning ( "%dbytes unused in Dbg mem block",((int)_dbg_end - (int)p_inst_node));
147  }
148 
149  Dbg_Notify ( "Dbg Mem Block Allocated successfully" );
150 }

157 void close_down( void )
158 {
159 
160 
161  int count = 0;
162  ClassNodeBlock* p_class_node = spClassNodeFreeList;
163  InstanceNodeBlock* p_inst_node = spInstanceNodeFreeList;
164 
165  do
166  {
167  Dbg_MsgAssert ( p_class_node->free_node.id == vFREE_CLASS_BLOCK_ID, "Block not free" );
168  p_class_node = (ClassNodeBlock*)p_class_node->free_node.next;
169  count++;
170  } while ( p_class_node != spClassNodeFreeList );
171 
172  Dbg_MsgAssert ( count == vMAX_CLASS_NODES,
173  "%d Class Nodes still registered", vMAX_CLASS_NODES - count );
174 
175  count = 0;
176 
177  do
178  {
179  Dbg_MsgAssert ( p_inst_node->free_node.id == vFREE_INST_BLOCK_ID, "Block not free" );
180  p_inst_node = (InstanceNodeBlock*)p_inst_node->free_node.next;
181  count++;
182  } while ( p_inst_node != spInstanceNodeFreeList );
183 
184  Dbg_MsgAssert ( count == vMAX_INSTANCE_NODES,
185  "%d Instance Nodes still registered", vMAX_INSTANCE_NODES - count );
186 
187  Dbg_Notify ( "Dbg Mem Block Released successfully" );
188 }
