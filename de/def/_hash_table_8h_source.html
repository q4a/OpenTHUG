165 template<class _V> //inline
166 bool HashTable<_V>::PutItem(const uint32 &key, _V *item)
167 {
168  Dbg_AssertPtr(item);
169  //Ryan("putting item in Hash table\n");
170 
171  Dbg_AssertPtr( mp_hash_table );
172 
173  // sometimes, we want to allow checksum conflicts
174  // (for instance, ConvertAssets' file database)
175  if ( !m_allowDuplicateKeys )
176  {
177  Dbg_MsgAssert(!GetItem(key), ("item 0x%x already in hash table", key));
178  }
179 
180  // can't add an item of 0,NULL, as that is used to indicate an empty head slot
181  Dbg_MsgAssert(key || item, ("Both key and item are 0 (NULL) in hash table"));
182  // can have a value of NULL either, as the the test below uses pEntry->mp_value == NULL
183  // to indicate an an empty head slot
184  // We could just change it to ( pEntry->mp_value || pEntry->m_key ) if NULL values are desirable
185  Dbg_MsgAssert( item, ("NULL item added to hash table"));
186 
187  HashItem<_V> *pEntry=&mp_hash_table[key&((1<<m_numBits)-1)];
188  if ( pEntry->mp_value )
189  {
190  // The main table entry is already occupied, so create a new HashEntry and
191  // link it in between the first and the rest.
192 #ifndef __PLAT_WN32__
193  HashItem<_V> *pNew = new (Mem::PoolManager::SCreateItem(Mem::PoolManager::vHASH_ITEM_POOL)) HashItem<_V>();
194 #else
195  HashItem<_V> *pNew = new HashItem<_V>;
196 #endif
197  pNew->m_key = key;
198  pNew->mp_value = item;
199  pNew->mp_next = pEntry->mp_next;
200  pEntry->mp_next = pNew;
201  }
202  else
203  {
204  // Main table entry is not occupied, so wack it in there.
205  pEntry->m_key = key;
206  pEntry->mp_value = item;
207  // leave pEntry->mp_next untouched
208  }
209 
210  m_size++;
211  return true;
212 }

215 template<class _V> //inline
216 _V *HashTable<_V>::GetNextItemWithSameKey(const uint32 &key, _V *p_item)
217 {
218  Dbg_AssertPtr( mp_hash_table );
219 
220  // Jump to the linked list of all entries with similar checksums.
221  HashItem<_V> *pEntry=&mp_hash_table[key&((1<<m_numBits)-1)];
222 
223  // Look for p_item
224  while (pEntry)
225  {
226  if (pEntry->m_key == key && pEntry->mp_value==p_item)
227  {
228  break;
229  }
230  pEntry=pEntry->mp_next;
231  }
232  if (!pEntry)
233  {
234  // p_item was not found.
235  return NULL;
236  }
237 
238  // Found p_item, so search the rest of the list for the next element with the same key.
239  pEntry=pEntry->mp_next;
240  while (pEntry)
241  {
242  if (pEntry->m_key == key && pEntry->mp_value)
243  {
244  return pEntry->mp_value;
245  }
246  pEntry=pEntry->mp_next;
247  }
248 
249  return NULL;
250 }

306 template<class _V> //inline
307 void HashTable<_V>::FlushItem(const uint32 &key)
308 {
309 
310  Dbg_AssertPtr( mp_hash_table );
311  if (!mp_hash_table)
312  return;
313 
314  // Jump to the linked list of all entries with similar checksums.
315  HashItem<_V> *pEntry=&mp_hash_table[key&((1<<m_numBits)-1)];
316  HashItem<_V> *pLast = NULL;
317 
318  // Scan through the small list until the matching entry is found.
319  while (pEntry)
320  {
321  HashItem<_V> *p_next_entry = pEntry->mp_next;
322  if (pEntry->m_key==key && pEntry->mp_value) // to allow keys of value 0, we have to skip head nodes that are 0,NULL
323  {
324  if (pLast)
325  {
326  // this is not a main table entry; this is a linked entry
327  pLast->mp_next = pEntry->mp_next;
328 #ifndef __PLAT_WN32__
329  Mem::PoolManager::SFreeItem(Mem::PoolManager::vHASH_ITEM_POOL, pEntry);
330 #else
331  delete pEntry;
332 #endif
333  }
334  else
335  {
336  // this is a main table entry, it still might be linked to something
337  // clear the entry to 0,NULL (see comment above about keys of value 0)
338  pEntry->m_key = 0;
339  pEntry->mp_value = NULL;
340  }
341  m_size--;
342  return;
343  }
344  pLast = pEntry;
345  pEntry = p_next_entry;
346  }
347  return;
348 }

353 template<class _V> //inline
354 void HashTable<_V>::FlushAllItems()
355 {
356 
357 
358  Dbg_AssertPtr( mp_hash_table );
359  if (!mp_hash_table)
360  return;
361 
362  // Run through the table and delete any of the extra
363  // HashItem<_V>s.
364  HashItem<_V> *pMainEntry = mp_hash_table;
365  uint32 hashTableSize = (1<<m_numBits);
366  for (uint32 i=0; i<hashTableSize; ++i)
367  {
368  HashItem<_V> *pLinkedEntry = pMainEntry->mp_next;
369  while (pLinkedEntry)
370  {
371  HashItem<_V> *pNext = pLinkedEntry->mp_next;
372 #ifndef __PLAT_WN32__
373  Mem::PoolManager::SFreeItem(Mem::PoolManager::vHASH_ITEM_POOL, pLinkedEntry);
374 #else
375  delete pLinkedEntry;
376 #endif
377  pLinkedEntry = pNext;
378  }
379  pMainEntry->Init();
380  ++pMainEntry;
381  }
382  m_size = 0;
383 }

388 template<class _V> //inline
389 void HashTable<_V>::HandleCallback(HashCallback hashCallback, void* pData)
390 {
391  HashItem<_V> *pMainEntry = mp_hash_table;
392  uint32 hashTableSize=(1<<m_numBits);
393  for (uint32 i=0; i<hashTableSize; ++i)
394  {
395  HashItem<_V> *pLinkedEntry = pMainEntry->mp_next;
396  while (pLinkedEntry)
397  {
398  HashItem<_V> *pNext = pLinkedEntry->mp_next;
399 // Mem::Free(pLinkedEntry);
400  if (pLinkedEntry->mp_value)
401  (hashCallback)((_V *) pLinkedEntry->mp_value, pData);
402  pLinkedEntry = pNext;
403  }
404  if (pMainEntry->mp_value)
405  (hashCallback)((_V *) pMainEntry->mp_value, pData);
406  ++pMainEntry;
407  }

424 template<class _V> //inline
425 _V * HashTable<_V>::IterateNext(uint32 *pRetKey)
426 {
427  uint32 hashTableSize=(1<<m_numBits);
428 
429  // time to go to the next entry, or the first if we're just starting
430 
431  if (mp_iterator_item)
432  // next entry in list
433  mp_iterator_item = mp_iterator_item->mp_next;
434  else if (m_iterator_index >= 0)
435  // we've exhausted all the lists
436  return NULL;
437 
438  if (!mp_iterator_item)
439  {
440  // no entry in list, move on to next list
441  do
442  {
443  m_iterator_index++;
444  if (m_iterator_index >= (int) hashTableSize)
445  return NULL;
446  mp_iterator_item = mp_hash_table + m_iterator_index;
447  } // main entry has to contain something, or be part of a list
448  while (!mp_iterator_item->mp_value && !mp_iterator_item->mp_next);
449  if (!mp_iterator_item->mp_value)
450  // this must be an empty main entry, skip ahead
451  mp_iterator_item = mp_iterator_item->mp_next;
452  }
453 
454  // Ken: Added this because it was hanging here once when loading the junkyard
455  // off CD. It was trying to dereference mp_iterator_item. Added the printf
456  // so we can at least see when it happens without having to go into the debugger.
457  if (!mp_iterator_item)
458  {
459  #ifdef __NOPT_ASSERT__
460  printf("Error!! NULL mp_iterator_item in IterateNext()\n");
461  #endif
462  return NULL;
463  }
464  if (pRetKey)
465  *pRetKey = mp_iterator_item->m_key;
466  return mp_iterator_item->mp_value;
467 }
