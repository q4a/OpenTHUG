311 CBaseComponent::EMemberFunctionResult CRiderComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
312 {
313  switch ( Checksum )
314  {
315 # if 0
316  // @script | Rider_Ground |
317  case CRCC(0x893213e5, "Rider_Ground"):
318  return m_state == WALKING_GROUND ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
319 
320  // @script | Rider_Air |
321  case CRCC(0x5012082e, "Rider_Air"):
322  return m_state == WALKING_AIR ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
323 
324  // @script | Rider_Hang |
325  case CRCC(0x9a3ca853, "Rider_Hang"):
326  return m_state == WALKING_HANG ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
327 
328  // @script | Rider_Ladder |
329  case CRCC(0x19702ca8, "Rider_Ladder"):
330  return m_state == WALKING_LADDER ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
331 
332  // @script | Rider_GetStateTime | Loads the time in milliseconds since last state change.
333  case CRCC(0xce64576c, "Rider_GetStateTime"):
334  pScript->GetParams()->AddInteger(CRCD(0x5ab23cc9, "StateTime"), Tmr::ElapsedTime(m_state_timestamp));
335  break;
336 
337  // @script | Rider_Jump |
338  case CRCC(0x83e4bd70, "Rider_Jump"):
339  {
340  // jump strength scales with the length the jump button has been held
341  jump(Mth::Lerp(
342  s_get_param(CRCD(0x246d0bf3, "min_jump_factor")),
343  1.0f,
344  Mth::ClampMax(mp_input_component->GetControlPad().m_x.GetPressedTime() / s_get_param(CRCD(0x12333ebd, "hold_time_for_max_jump")), 1.0f)
345  ));
346  break;
347  }
348 
349  case CRCC(0xeb0d763b, "Rider_HangJump"):
350  {
351  // jump strength scales with the length the jump button has been held
352  jump(s_get_param(CRCD(0xf2fa5845, "hang_jump_factor")), true);
353  break;
354  }
355 
356  // @script | Rider_SetDragFactor |
357  case CRCC(0xc6100a7d, "Rider_SetDragFactor"):
358  pParams->GetFloat(NO_NAME, &m_script_drag_factor);
359  break;
360 
361  case CRCC(0x4e4fae43, "Rider_ResetDragFactor"):
362  m_script_drag_factor = 1.0f;
363  break;
364 
365  case CRCC(0xaf04b983, "Rider_GetSpeedScale"):
366  {
367  uint32 checksum;
368  if (m_anim_effective_speed < s_get_param(CRCD(0xf3649996, "max_slow_walk_speed")))
369  {
370  checksum = CRCD(0x1150cabb, "WALK_SLOW");
371  }
372  else if (m_anim_effective_speed < s_get_param(CRCD(0x6a5805d8, "max_fast_walk_speed")))
373  {
374  checksum = CRCD(0x131f2a2, "WALK_FAST");
375  }
376  else if (m_anim_effective_speed < s_get_param(CRCD(0x1c94cc9c, "max_slow_run_speed")))
377  {
378  checksum = CRCD(0x5606d106, "RUN_SLOW");
379  }
380  else
381  {
382  checksum = CRCD(0x4667e91f, "RUN_FAST");
383  }
384  pScript->GetParams()->AddChecksum(CRCD(0x92c388f, "SpeedScale"), checksum);
385 
386  break;
387  }
388 
389  // @script | Rider_ScaleAnimSpeed | Sets the manner in which the walk animations speeds should be scaled.
390  // @flag Off | No animation speed scaling.
391  // @flag Run | Scale animation speeds against running speed.
392  // @flag Rider | Scale animation speeds against walking speed.
393  case CRCC(0x56112c03, "Rider_ScaleAnimSpeed"):
394  if (pParams->ContainsFlag(CRCD(0xd443a2bc, "Off")))
395  {
396  if (m_anim_scale_speed != OFF)
397  {
398  m_anim_scale_speed = OFF;
399  mp_animation_component->SetAnimSpeed(1.0f, false, true);
400  }
401  }
402  else if (pParams->ContainsFlag(CRCD(0xaf895b3f, "Run")))
403  {
404  m_anim_scale_speed = RUNNING;
405  }
406  else if (pParams->ContainsFlag(CRCD(0x6384f1da, "HangMove")))
407  {
408  m_anim_scale_speed = HANGMOVE;
409  }
410  else if (pParams->ContainsFlag(CRCD(0xa2bfe505, "LadderMove")))
411  {
412  m_anim_scale_speed = LADDERMOVE;
413  }
414  else
415  {
416  Dbg_MsgAssert(false, ("Rider_ScaleAnimSpeed requires Off, Run, or Rider flag"));
417  }
418 
419  pParams->GetFloat(CRCD(0xb2d59baf, "StandardSpeed"), &m_anim_standard_speed);
420  break;
421 
422  // @script | Rider_AnimWaitComplete | Signal from script that the walk component should leave its animation wait
423  case CRCC(0x9d3eebe8, "Rider_AnimWaitComplete"):
424  anim_wait_complete();
425  break;
426 
427  // @script | Rider_GetHangInitAnimType | Determine which type of initial hang animation should be played
428  case CRCC(0xc6cd659e, "Rider_GetHangInitAnimType"):
429  // m_initial_hang_animation is set when the hang rail is filtered
430  pScript->GetParams()->AddChecksum(CRCD(0x85fa9ac4, "HangInitAnimType"), m_initial_hang_animation);
431  break;
432 # endif
433  default:
434  return CBaseComponent::MF_NOT_EXECUTED;
435  }
436  return CBaseComponent::MF_TRUE;
437 }

488 bool CRiderComponent::ReadyRiderState( bool to_ground_state )
489 {
490  // setup the state in preparation for being in walking mode next object update
491  MARK;
492 
493  // Firstly ensure that there is a valid horse nearby.
494  mp_horse_component = NULL;
495 
496  Obj::CHorseComponent* p_horse_component = static_cast<Obj::CHorseComponent*>( Obj::CCompositeObjectManager::Instance()->GetFirstComponentByType( CRC_HORSE ));
497  while( p_horse_component )
498  {
499  Obj::CCompositeObject* p_horse = p_horse_component->GetObject();
500  Mth::Vector horse_pos = p_horse->GetPos();
501  float dist = Mth::Distance( GetObject()->GetPos(), horse_pos );
502  if( dist < 60.0f )
503  {
504  mp_horse_component = p_horse_component;
505  break;
506  }
507  p_horse_component = static_cast<Obj::CHorseComponent*>( p_horse_component->GetNextSameType());
508  }
509 
510  if( mp_horse_component == NULL )
511  {
512  // Don't want to proceed.
513  return false;
514 
515  }
516 
517  printf( "ReadyRiderState() found horse\n" );
518 
519  mp_horse_component->AcceptRiderMount( GetObject());
520 
521  // always reset the state timestamp
522  m_state_timestamp = Tmr::GetTime();
523 
524  if (to_ground_state)
525  {
526  set_state(WALKING_GROUND);
527 
528  // will be incorrect for one frame
529  m_ground_normal.Set(0.0f, 1.0f, 0.0);
530 
531  m_last_ground_feeler_valid = false;
532 
533  GetObject()->GetVel()[Y] = 0.0f;
534  }
535  else
536  {
537  set_state(WALKING_AIR);
538 
539  // set primary air direction in the direction of velocity
540  m_primary_air_direction = GetObject()->GetVel();
541  m_primary_air_direction[Y] = 0.0f;
542  float length = m_primary_air_direction.Length();
543  if (length < 0.001f)
544  {
545  // or facing
546  m_primary_air_direction = GetObject()->GetMatrix()[Z];
547  m_primary_air_direction[Y] = 0.0f;
548  length = m_primary_air_direction.Length();
549  if (length < 0.001f)
550  {
551  // or future facing
552  m_primary_air_direction = -GetObject()->GetMatrix()[Y];
553  m_primary_air_direction[Y] = 0.0f;
554  length = m_primary_air_direction.Length();
555  }
556  }
557  m_primary_air_direction /= length;
558 
559 // leave_movable_contact_for_air(GetObject()->GetVel(), GetObject()->GetVel()[Y]);
560  }
561 
562  m_curb_float_height = 0.0f;
563 
564  m_last_frame_event = 0;
565 
566  return true;
567 }
