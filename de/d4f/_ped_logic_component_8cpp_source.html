160  {
161  // @script | Ped_SetLogicState | set the current state of the ped
162  // @uparm name | state - eg, generic, generic_skater, etc.

165  break;
166  // @script | Ped_InitPath | initializes the ped on his linked path and starts movement
167  case CRCC( 0xe9906815, "Ped_InitPath" ):
168  {
169  Script::CStruct* pNodeData = SkateScript::GetNode( m_node_from );
170  // m_node_to = SkateScript::GetLink( pNodeData, Mth::Rnd( SkateScript::GetNumLinks( pNodeData ) ) );

173  int numLinks = SkateScript::GetNumLinks( m_node_from );
174  Dbg_MsgAssert( numLinks, ( "Ped_InitPath called on %s, but path node %i has no links!", Script::FindChecksumName( GetObject()->GetID() ), m_node_from ) );
175  #endif
176 
177  m_node_to = SkateScript::GetLink( pNodeData, 0 );
178  SkateScript::GetPosition( m_node_to, &m_wp_to );
179  SelectNextWaypoint();
180  m_flags |= PEDLOGIC_MOVING_ON_PATH;
181  mp_path_object_tracker = Obj::CPathMan::Instance()->TrackPed( GetObject(), m_node_to );
182 
183  Obj::CMotionComponent* pMotionComp = GetMotionComponentFromObject( GetObject() );
184  Dbg_Assert( pMotionComp );
185  pMotionComp->m_movingobj_status |= MOVINGOBJ_STATUS_ON_PATH;
186 
187  // turn the ped to face the right direction
188  Mth::Vector heading = ( m_wp_to - m_wp_from ).Normalize();
189  Mth::Matrix display_matrix = GetObject()->GetDisplayMatrix();
190  display_matrix[Z] = heading;
191  display_matrix.OrthoNormalizeAbout( Z );
192  GetObject()->SetDisplayMatrix( display_matrix );
193  m_current_display_matrix = display_matrix;
194 
195  Mth::Vector current_pos = GetObject()->m_pos;
196  for ( int i = 0; i < vPOS_HISTORY_SIZE; i++ )
197  m_pos_history[i] = current_pos;
198 
199  break;
200  }
201  // @script | Ped_StartMoving | start moving on path
202  case CRCC( 0xd711542, "Ped_StartMoving" ):
203  {
204  if ( m_flags & PEDLOGIC_STOPPED )
205  {
206  Script::CScript* pScript = Script::SpawnScript( CRCD( 0xbed339d4, "ped_skater_start_moving" ) );
207  pScript->mpObject = GetObject();
208  m_flags &= ~PEDLOGIC_STOPPED;
209  }
210  m_flags |= PEDLOGIC_MOVING_ON_PATH;
211  break;
212  }
213  // @script | Ped_StopMoving | stop moving on path
214  case CRCC( 0xcc85adfe, "Ped_StopMoving" ):
215  m_flags &= ~PEDLOGIC_MOVING_ON_PATH;
216  break;
217  // @script | Ped_IsMoving | returns true if the ped is moving on the path
218  case CRCC(0x68d298b0,"Ped_IsMoving"):
219  return m_flags & PEDLOGIC_MOVING_ON_PATH ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
220  break;
221  // @script | Ped_SetStickToGroundDist | Set the distances the ped uses to check for ground
222  // when sticking. This check is not performed straight up and down, but at whatever
223  // angle the ped is currently standing (his Y vector).
224  // @parmopt float | StickToGroundDistAbove | | distance to check above the ped
225  // @parmopt float | StickToGroundDistBelow | | distance to check below the ped
226  case CRCC( 0x2cc086dd, "Ped_SetStickToGroundDist" ):
227  pParams->GetFloat( "distAbove", &m_col_dist_above );
228  pParams->GetFloat( "distBelow", &m_col_dist_below );
229  printf("StickToGround distances set to %f (above), %f (below)\n", m_col_dist_above, m_col_dist_below);
230  break;
231  // @script | Ped_SetIsGrinding | used for manually setting the grind state on and off
232  // @uparm 0 | 0 for off, anything else for on
233  case CRCC( 0xc643e8f5, "Ped_SetIsGrinding" ):
234  {
235  int is_grinding = 0;
236  pParams->GetInteger( NONAME, &is_grinding, Script::ASSERT );
237  if ( is_grinding != 0 )
238  m_flags |= PEDLOGIC_GRINDING;
239  else
240  m_flags &= ~PEDLOGIC_GRINDING;
241  break;
242  }
243  // @script | Ped_GetCurrentVelocity | returns variable named velocity...
244  // @flag max | get the maximum velocity, rather than the currnt velocity.
245  // This is useful in case the ped is currently accelerating or decelerating
246  // to achieve maximum velocity.
247  case CRCC( 0x13713760, "Ped_GetCurrentVelocity" ):
248  {
249  Obj::CMotionComponent* pMotionComp = GetMotionComponentFromObject( GetObject() );
250  Dbg_Assert( pMotionComp );
251  pScript->GetParams()->AddFloat( CRCD( 0x41272956, "velocity" ), pMotionComp->m_vel_z );
252  break;
253  }
254  // @script | Ped_StoreMaxVelocity | stores the current max velocity, so you can
255  // change it and restore it later
256  case CRCC( 0xf5627596, "Ped_StoreMaxVelocity" ):
257  {
258  Obj::CMotionComponent* pMotionComp = GetMotionComponentFromObject( GetObject() );
259  Dbg_Assert( pMotionComp );
260  m_original_max_vel = pMotionComp->m_max_vel;
261  break;
262  }
263  // @script | Ped_GetOriginalMaxVelocity | returns original_max_velocity stored when
264  // Ped_StoreMaxVelocity was called
265  case CRCC( 0xe575efa9, "Ped_GetOriginalMaxVelocity" ):
266  {
267  pScript->GetParams()->AddFloat( "original_max_velocity", m_original_max_vel );
268  break;
269  }
270  // @script | Ped_Bail |
271  case CRCC( 0xee24a410, "Ped_Bail" ):
272  Bail();
273  break;
274  case CRCC( 0xc076d74, "Ped_SetIsBailing" ):
275  {
276  int v;
277  pParams->GetInteger( NONAME, &v, Script::ASSERT );
278  if ( pParams->ContainsFlag( CRCD( 0x255ed86f, "grind" ) ) )
279  {
280  if ( v == 0 )
281  m_flags &= ~PEDLOGIC_GRIND_BAILING;
282  else
283  m_flags |= PEDLOGIC_GRIND_BAILING;
284  }
285  else
286  {
287  if ( v == 0 )
288  m_flags &= ~PEDLOGIC_BAILING;
289  else
290  m_flags |= PEDLOGIC_BAILING;
291  }
292  break;
293  }
294  case CRCC( 0xda7be9e2, "Ped_InitVertTrick" ):
295  {
296  // get the height and gravity
297  float height;
298  pParams->GetFloat( CRCD( 0xab21af0, "height" ), &height, Script::ASSERT );
299  float gravity;
300  pParams->GetFloat( CRCD( 0xa5e2da58, "gravity" ), &gravity, Script::ASSERT );
301  Dbg_MsgAssert( gravity < 0.0f, ( "Ped_InitVertTrick given positive number for gravity" ) );
302 
303  Mth::Vector pt = m_wp_to - GetObject()->m_pos;
304  // adjust height to compensate for difference between ped height
305  // and height of waypoint
306  // height += pt[Y];
307  // pt[Y] = 0.0f;
308 
309  // figure the time - the gravity is always expressed as negative
310  // acceleration, but since we're figuring the time it will take to
311  // fall from the top at velocity=0, we use positive
312  // Multiply by 2 to get time up and down
313  // d = vi*t + (a*t^2)/2
314  float time = 2 * sqrtf( 2 * -height / gravity );
315 
316  // compute the inital y velocity, which equals the final y velocity when
317  // falling back down (vf^2 = vi^2 + 2ad)
318  float vi = sqrtf( 2 * gravity * -height );
319  Dbg_Assert( vi > 0 );
320 
321  // find the horizontal distance we need to travel
322  float distance = pt.Length() / 2;
323  // bump it up a bit so we don't miss landing
324  distance *= Script::GetFloat( CRCD( 0x2b5bedba, "ped_skater_vert_jump_slop_distance" ), Script::ASSERT );
325 
326  // horizontal velocity
327  Mth::Vector vh = ( distance / time ) * pt.Normalize();
328 
329  // vertical velocity
330  Mth::Vector vy = vi * Mth::Vector(0, 1, 0);
331 
332  // total speed and heading
333  // float jump_speed = ( ( vh + vy ) * time ).Length();
334  float jump_speed = ( vh + vy ).Length();
335  Mth::Vector heading = ( vh + vy ).Normalize();
336 
337  pScript->GetParams()->AddVector( CRCD( 0xfd4bc03e, "heading" ), heading );
338  pScript->GetParams()->AddFloat( CRCD( 0x1c4c5690, "jumpSpeed" ), jump_speed );
339  pScript->GetParams()->AddFloat( CRCD( 0x66ced87b, "jumpTime" ), time );
340  break;
341  }
342  // @script | Ped_HitWaypoint | force the ped to register a hit on the
343  // to waypoint, regardless of his position. Don't call this unless
344  // you know exactly why you're doing it.
345  case CRCC( 0x1affffb3, "Ped_HitWaypoint" ):
346  HitWaypoint();
347  break;
348  // @script | Ped_SetIsSkater |
349  // @uparm 1 | 0 if not skater, anything else for skater
350  case CRCC( 0x5c794c57, "Ped_SetIsSkater" ):
351  {
352  int is_skater;
353  pParams->GetInteger( NONAME, &is_skater, Script::ASSERT );
354  if ( is_skater == 0 )
355  m_flags &= ~PEDLOGIC_IS_SKATER;
356  else
357  m_flags |= PEDLOGIC_IS_SKATER;
358  break;
359  }
360  // @script | Ped_NearVertNode | true if the next node or the to node
361  // is vert
362  case CRCC( 0x5bb23915, "Ped_NearVertNode" ):
363  return ( ( ( m_flags & PEDLOGIC_TO_NODE_IS_VERT )
364  || ( m_flags & PEDLOGIC_NEXT_NODE_IS_VERT )
365  || ( m_flags & PEDLOGIC_FROM_NODE_IS_VERT ) ) ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE );
366  break;
367  case CRCC( 0x56d9f55f, "Ped_PlayJumpSound" ):
368  {
369  //Dbg_MsgAssert( m_state == CRCD(0xa85af587,"generic_skater"), ( "%s\nPed_PlayJumpSound called on a non-skater ped %s",pScript->GetScriptInfo(),Script::FindChecksumName(GetObject()->GetID()) ) );
370  Obj::CSkaterSoundComponent *pSoundComponent = GetSkaterSoundComponentFromObject( GetObject() );
371  if ( pSoundComponent )
372  pSoundComponent->PlayJumpSound(m_speed_fraction);
373  break;
374  }
375  case CRCC( 0x749de914, "Ped_PlayLandSound" ):
376  {
377  Dbg_MsgAssert( m_state == CRCD(0xa85af587,"generic_skater"), ( "Ped_PlayLandSound called on a non-ped skater" ) );
378  Obj::CSkaterSoundComponent *pSoundComponent = GetSkaterSoundComponentFromObject( GetObject() );
379  if ( pSoundComponent )
380  pSoundComponent->PlayLandSound(m_speed_fraction);
381  break;
382  }
383  case CRCC( 0x58a588b5, "Ped_GetCurrentNodeNames" ):
384  {
385  pScript->GetParams()->AddChecksum( CRCD( 0xd14cd5bc, "node_to" ), m_node_to );
386  pScript->GetParams()->AddChecksum( CRCD( 0x8141d319, "node_from" ), m_node_from );
387  pScript->GetParams()->AddChecksum( CRCD( 0x3c746255, "node_next" ), m_node_next );
388  break;
389  }
390  default:
391  return CBaseComponent::MF_NOT_EXECUTED;
392  }
393  return CBaseComponent::MF_TRUE;
394 }

478 void CPedLogicComponent::GenericSkaterUpdate()
479 {
480  Obj::CMotionComponent* pMotionComp = GetMotionComponentFromObject( GetObject() );
481  Dbg_Assert( pMotionComp );
482  pMotionComp->DoPathPhysics();
483  float distance = m_time * pMotionComp->m_vel_z;
484  bool is_jumping = pMotionComp->m_movingobj_status & MOVINGOBJ_STATUS_JUMPING ? true : false;
485 
486  // Gfx::AddDebugLine( m_wp_from, m_wp_to, MAKE_RGB( 128, 0, 0 ), MAKE_RGB( 128, 0, 0 ), 1 );
487  // Gfx::AddDebugLine( GetObject()->m_pos, m_wp_to, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 1 );
488  // Gfx::AddDebugLine( m_pos_history[vPOS_HISTORY_SIZE - 2] + Mth::Vector( 0, 5, 1 ), m_pos_history[vPOS_HISTORY_SIZE - 1] + Mth::Vector( 0, 5, 1 ), MAKE_RGB( 0, 128, 0 ), MAKE_RGB( 0, 128, 0 ) );
489  // Gfx::AddDebugArrow( GetObject()->GetDisplayMatrix()[Z] + GetObject()->m_pos, GetObject()->GetDisplayMatrix()[Z] * 24 + GetObject()->m_pos, MAKE_RGB( 0, 0, 128 ), MAKE_RGB( 0, 0, 128 ), 1 );
490 
491  // check for skater if we're grinding
492  if ( m_flags & PEDLOGIC_GRINDING )
493  {
494  Mdl::Skate* skate_mod = Mdl::Skate::Instance();
495  Dbg_Assert( skate_mod );
496  Obj::CSkater* pSkater = skate_mod->GetLocalSkater();
497  if ( pSkater )
498  {
499  float d = Mth::DistanceSqr( GetObject()->m_pos, pSkater->GetPos() );
500  // printf("skater dist square = %f\n", d );
501  if ( d < Script::GetFloat( CRCD( 0xbc2e678a, "ped_skater_min_square_distance_to_skater" ), Script::ASSERT ) )
502  {
503  GrindBail();
504  return;
505  }
506  }
507  }
508 
509  // see if we've hit the waypoint
510  if ( ( m_flags & PEDLOGIC_MOVING_ON_PATH ) || is_jumping )
511  {
512  Mth::Vector pt;
513  if ( ( m_flags & PEDLOGIC_TO_NODE_IS_VERT ) || is_jumping )
514  {
515  pt = ( GetObject()->m_pos - m_wp_to );
516  }
517  else
518  {
519  Mth::Vector wp_to_no_y = m_wp_to;
520  wp_to_no_y[Y] = 0.0f;
521  Mth::Vector ped_pos_no_y = GetObject()->m_pos;
522  ped_pos_no_y[Y] = 0.0f;
523  pt = ped_pos_no_y - wp_to_no_y;
524  }
525  float distance_to_target_square = pt.LengthSqr();
526  int min_dist_square = Script::GetInteger( CRCD( 0xfb5d106f, "ped_skater_min_square_distance_to_waypoint" ), Script::ASSERT );
527 
528  if ( !( m_flags & PEDLOGIC_JUMPING_TO_NODE ) )
529  {
530  if ( distance_to_target_square < Mth::Sqr( distance ) )
531  HitWaypoint();
532  else if ( distance_to_target_square < min_dist_square )
533  HitWaypoint();
534  }
535 
536  if ( m_flags & PEDLOGIC_JUMP_AT_TO_NODE )
537  {
538  int dist_to_crouch = Script::GetInteger( CRCD( 0x52ec432f, "ped_skater_min_square_distance_to_crouch_for_jump" ), Script::ASSERT );
539  if ( distance_to_target_square < dist_to_crouch )

732 void CPedLogicComponent::UpdatePathBias()
733 {

742  Mth::Vector ft = wp_to_no_y - wp_from_no_y;
743  Mth::Vector fp = ped_pos_no_y - wp_from_no_y;
744 
745  // need to store this vector so we can check the sign of y...
746  // if it's negative, we're on one side of the path, positive for the other
747  Mth::Vector ft_cross_fp = Mth::CrossProduct( ft, fp );
748 
749  // get the direction we want to push the ped
750  Mth::Vector closest_pos_on_path = ( ( Mth::DotProduct( fp, ft ) / ft.LengthSqr() ) * ft ) + m_wp_from;
751  closest_pos_on_path[Y] = 0.0f;
752  Mth::Vector pos_to_path = closest_pos_on_path - ped_pos_no_y;
753 
754  // Gfx::AddDebugArrow( GetObject()->m_pos, closest_pos_on_path, MAKE_RGB( 0, 128, 0 ), MAKE_RGB( 0, 128, 0 ), 1 );

759  if ( angle_to_path > 360.0f )

761  if ( Mth::CrossProduct( display_matrix[Z], pos_to_path )[Y] > 0 )
762  angle_to_path = 360.0f - angle_to_path;

1633  m_last_display_normal = m_display_normal; // remember start position for lerping
1634  m_normal_lerp = 1.0f; // set lerp counter
1635  }

1638  // we should already be there

1645  m_normal_lerp = 0.0f;

1663  // calculate intermediate normal
