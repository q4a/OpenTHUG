25 //
26 // Note about sample frequencies on Nintendo GameCube:
27 //
28 // GameCube's AI knows two native sample frequencies. Generally these are
29 // referenced as 48KHz and 32KHz. The DSP (no matter if driven by MusyX or AX)
30 // is using 32KHz.
31 //
32 // The frequencies that are in fact used are not exactly these frequencies, though!
33 //
34 // While this is not important when handling sound effects or even streamed audio,
35 // it is very important when interlaeving audio and video data since buffer under-runs
36 // might be triggered if no care is taken.
37 //
38 // The actual AI output frequencies are 48043Hz and 32028.66Hz.
39 //
40 // Streamed audio data interleaved with video should match these frequencies as
41 // closely as possible.
42 //

400 static BOOL audioDecode(u32 numChannels, const s16 **samples, u32 sampleNum, void* _UNUSED(userData))
401  {
402  u32 freeSamples;
403  u32 sampleSize;
404  u32 len1;
405  BOOL old;
406 
407  // we can only play mono or stereo!
408  ASSERT(numChannels <= 2);
409 
410  // Disable IRQs. We must make sure we don't get interrupted by the AI callback.
411  old = OSDisableInterrupts();
412 
413  // Did the video decoder just jump back to the beginning of the stream?
414  if (audio_player.readBuffer[audio_player.decodeIndex].frameNumber < audio_player.lastDecodedFrame)
415  {
416  // Yes! We have to reset the internal read buffer and disable any audio output
417  // until we got new video data to display...
418  //
419  // Note: we have to reset our buffers because the stream contains more audio data than
420  // neccessary to cover a single video frame within the first few frames to accumulate
421  // some safety buffer. If the stream would just be allowed to loop we would get an
422  // overflow (unless we used up the extra buffer due to read / decode delays) after a few
423  // loops...
424  //
425  AUDSimpleAudioReset();
426  }
427 
428  // Calculate the read buffer's sample size
429  sampleSize = sizeof(s16) * audioReadBufferNumChannels;
430 
431  // How many samples could we put into the buffer?
432  if (audioReadBufferWritePos >= audioReadBufferReadPos)
433  {
434  freeSamples = audioReadBufferNumSamples - (audioReadBufferWritePos - audioReadBufferReadPos);
435 
436  if( freeSamples < sampleNum )
437  {
438  OSRestoreInterrupts(old);
439  #ifndef FINAL
440  OSReport("*** audioDecode: overflow case 1\n");
441  #endif
442  return FALSE; // overflow!
443  }
444 
445  // We might have a two buffer update to do. Check for it...
446  if ((len1 = (audioReadBufferNumSamples - audioReadBufferWritePos)) >= sampleNum)
447  {
448  // No. We got ourselfs a nice, simple single buffer update.
449  writeChannelData((s16 *)((u32)audioReadBuffer + audioReadBufferWritePos * sampleSize),numChannels,samples,0,sampleNum);
450  }
451  else
452  {
453  // Dual buffer case
454  writeChannelData((s16 *)((u32)audioReadBuffer + audioReadBufferWritePos * sampleSize),numChannels,samples,0,len1);
455  writeChannelData((s16 *)audioReadBuffer,numChannels,samples,len1,sampleNum-len1);
456  }
457  }
458  else
459  {
460  freeSamples = audioReadBufferReadPos - audioReadBufferWritePos;
461 
462  if (freeSamples < sampleNum)
463  {
464  OSRestoreInterrupts(old);
465  #ifndef FINAL
466  OSReport("*** audioDecode: overflow case 2\n");
467  #endif
468  return FALSE; // overflow!
469  }
470 
471  // We're save to assume to have a single buffer update in any case...
472  writeChannelData((s16 *)((u32)audioReadBuffer + audioReadBufferWritePos * sampleSize),numChannels,samples,0,sampleNum);
473  }
474 
475  // Advance write position...
476  audioReadBufferWritePos += sampleNum;
477 
478  if (audioReadBufferWritePos >= audioReadBufferNumSamples)
479  audioReadBufferWritePos -= audioReadBufferNumSamples;
480 
481  // We're done with all critical stuff. IRQs may be enabled again...
482  OSRestoreInterrupts(old);
483 
484  return TRUE;
485  }

586 static void audioCopy(u32 smpOffset,u32 numSamples)
587  {
588  s16 *destAddrL,*destAddrR,*srcAddr,s;
589  u32 j;
590 
591  // Target address in audio buffer
592  destAddrL = (s16 *)((u32)audioPlayBuffer[audioPlayBufferWriteIndex] + smpOffset * sizeof(s16));
593  destAddrR = (s16 *)((u32)audioPlayBuffer[audioPlayBufferWriteIndex] + smpOffset * sizeof(s16) + (AUD_AUDIO_AIBUFFERSAMPLES * sizeof(s16)));
594 
595  // Single or dual channel setup?
596  if (audioReadBufferNumChannels == 2)
597  {
598  // Stereo! Get source address of data...
599  srcAddr = (s16 *)((u32)audioReadBuffer + audioReadBufferReadPos * 2 * sizeof(s16));
600 
601  // Copy samples into AI buffer and swap channels
602  for(j=0; j<numSamples; j++)
603  {
604  *(destAddrL++) = *(srcAddr++);
605  *(destAddrR++) = *(srcAddr++);
606  }
607  }
608  else
609  {
610  // Mono case!
611 
612  // Make sure it's truely mono!
613  ASSERT(audioReadBufferNumChannels == 1);
614 
615  // Get source address...
616  srcAddr = (s16 *)((u32)audioReadBuffer + audioReadBufferReadPos * sizeof(s16));
617 
618  // Copy samples into AI buffer (AI is always stereo, so we have to dublicate data)
619  for(j=0; j<numSamples; j++)
620  {
621  s = (*srcAddr++);
622  *(destAddrL++) = s;
623  *(destAddrR++) = s;
624  }
625  }
626 
627  // Advance read position as needed...
628  audioReadBufferReadPos += numSamples;
629  if (audioReadBufferReadPos >= audioReadBufferNumSamples)
630  audioReadBufferReadPos -= audioReadBufferNumSamples;
631  }

640 static void AXCallback(void)
641 {
642  // First thing to do here is call the regular soundfx callback.
643  Sfx::AXUserCBack();
644 
645  u32 availSamples,availFrames,numUpdate,i,n;
646  u32 audioPlayBufferNeededAudioFrames;
647  u32 currentAddr;
648 
649  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
650 
651  if (axPhase == AX_PHASE_START)
652  {
653  AXSetVoiceState(axVoice[0],AX_PB_STATE_RUN);
654  AXSetVoiceState(axVoice[1],AX_PB_STATE_RUN);
655  axPhase = AX_PHASE_PLAY;
656  }
657 
658  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
659 
660  currentAddr = *(u32 *)&axVoice[0]->pb.addr.currentAddressHi;
661 
662  if( currentAddr >= axLastAddr )
663  {
664  axPlayedSamples += currentAddr - axLastAddr;
665  axPlayedSamplesTotal += currentAddr - axLastAddr;
666  }
667  else
668  {
669  axPlayedSamples += (( AX_ARAM_LEFT_CHANNEL + AX_ARAM_BUFFER_SIZE ) - axLastAddr ) + ( currentAddr - AX_ARAM_LEFT_CHANNEL );
670  axPlayedSamplesTotal += (( AX_ARAM_LEFT_CHANNEL + AX_ARAM_BUFFER_SIZE ) - axLastAddr ) + ( currentAddr - AX_ARAM_LEFT_CHANNEL );
671  }
672 
673  axLastAddr = currentAddr;
674 
675  // If we have played the required number of samples, stop the voice.
676  if( axPlayedSamplesTotal >= audio_player.audioInfo.vaudex.totalSampleCount )
677  {
678  AXSetVoiceState( axVoice[0], AX_PB_STATE_STOP );
679  AXSetVoiceState( axVoice[1], AX_PB_STATE_STOP );
680  audio_player.playbackComplete = true;
681  return;
682  }
683 
684  if( axPlayedSamples >= AUD_AUDIO_AIBUFFERSAMPLES )
685  {
686  audioPlayBufferNeededAudioFrames = axPlayedSamples / AUD_AUDIO_AIBUFFERSAMPLES;
687 
688  // Make sure that we never get an underrun we don't notice...
689  if( !( audioPlayBufferNeededAudioFrames <= AUD_AUDIO_NUMAIBUFFERS ))
690  {
691 // OSReport( "AX audio buffer underrun!\n" );
692 
693  // Disable playback.
694  audioPlayBufferEnabled = false;
695  }
696 
697  // Is actual audio playback enabled?
698  if( audioPlayBufferEnabled )
699  {
700  // How many samples could we get from the read buffer?
701  if( audioReadBufferWritePos >= audioReadBufferReadPos )
702  availSamples = audioReadBufferWritePos - audioReadBufferReadPos;
703  else
704  availSamples = audioReadBufferNumSamples - ( audioReadBufferReadPos - audioReadBufferWritePos );
705 
706  // That's how many audio frames?
707  availFrames = availSamples / AUD_AUDIO_AIBUFFERSAMPLES;
708 
709  //OSReport("AX: %d %d (%d)\n",availSamples,audioPlayBufferNeededAudioFrames * VID_AUDIO_AIBUFFERSAMPLES,axPlayedSamples);
710 
711  // So, how many can we update?
712  numUpdate = ( availFrames > audioPlayBufferNeededAudioFrames ) ? audioPlayBufferNeededAudioFrames : availFrames;
713 
714  // If anything... go do it!
715  if( numUpdate != 0 )
716  {
717  axPlayedSamples -= numUpdate * AUD_AUDIO_AIBUFFERSAMPLES;
718 
719  // Perform updates on each AI buffer in need of data...
720  for( i = 0; i < numUpdate; i++ )
721  {
722  u32 leftSource, rightSource, leftTarget, rightTarget;
723 
724  // Can we copy everything from a single source or does the data wrap around?
725  if(( n = audioReadBufferNumSamples - audioReadBufferReadPos) < AUD_AUDIO_AIBUFFERSAMPLES )
726  {
727  // It wraps...
728  audioCopy( 0, n );
729  audioCopy( n, AUD_AUDIO_AIBUFFERSAMPLES - n );
730  }
731  else
732  {
733  // We got one continous source buffer
734  audioCopy( 0, AUD_AUDIO_AIBUFFERSAMPLES );
735  }
736 
737  // Make sure the data ends up in real physical memory
738  DCFlushRange( audioPlayBuffer[audioPlayBufferWriteIndex], AUD_AUDIO_AIBUFFERSAMPLES * sizeof( s16 ) * 2 );
739 
740  leftSource = (u32)audioPlayBuffer[audioPlayBufferWriteIndex];
741  rightSource = (u32)audioPlayBuffer[audioPlayBufferWriteIndex] + ( AUD_AUDIO_AIBUFFERSAMPLES * sizeof( s16 ));
742  leftTarget = 2 * ( AX_ARAM_LEFT_CHANNEL + audioPlayBufferWriteIndex * AUD_AUDIO_AIBUFFERSAMPLES );
743  rightTarget = 2 * ( AX_ARAM_RIGHT_CHANNEL + audioPlayBufferWriteIndex * AUD_AUDIO_AIBUFFERSAMPLES );
744 
745  // Make sure we get this into ARAM ASAP...
746  ARQPostRequest( &arqRequest[0][i%AUD_NUM_ARQ_REQUESTS], 0, ARQ_TYPE_MRAM_TO_ARAM,ARQ_PRIORITY_HIGH, leftSource, leftTarget, AUD_AUDIO_AIBUFFERSAMPLES * sizeof( s16 ), NULL );
747  ARQPostRequest( &arqRequest[1][i%AUD_NUM_ARQ_REQUESTS], 1, ARQ_TYPE_MRAM_TO_ARAM,ARQ_PRIORITY_HIGH, rightSource, rightTarget, AUD_AUDIO_AIBUFFERSAMPLES * sizeof( s16 ), NULL );
748 
749  // Advance write index...
750  audioPlayBufferWriteIndex = (u8)(( audioPlayBufferWriteIndex + 1 ) % AUD_AUDIO_NUMAIBUFFERS );
751  }
752 
753  if( axPhase == AX_PHASE_STARTUP )
754  {
755  axPhase = AX_PHASE_START;
756  }
757  }
758  }
759  else
