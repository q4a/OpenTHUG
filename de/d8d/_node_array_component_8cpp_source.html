83 // InitFromStructure is passed a Script::CStruct that contains a
84 // number of parameters to initialize this component
85 // this currently is the contents of a node
86 // but you can pass in anything you like.
87 void CNodeArrayComponent::InitFromStructure( Script::CStruct* pParams )
88 {
89  // Obtain the model name from the structure. At a later date, perhaps the node could override the .qb name.
90  const char *p_model_name = NULL;
91  if( !pParams->GetText( 0x286a8d26, &p_model_name ))
92  {
93  // This is currently an error.
94  Dbg_MsgAssert( 0, ( "Attempting to create NodeArrayComponent from node with no model name" ));
95  return;
96  }
97 
98  // Generate the .qb name from the model name.
99  char nodearray_filename[256];
100  strcpy( nodearray_filename, "models\\" );
101  strcat( nodearray_filename, p_model_name );
102  char *p_ext = strstr( nodearray_filename, "." );
103  if( p_ext )
104  {
105  strcpy( p_ext, ".qb" );
106  }
107  else
108  {
109  strcat( nodearray_filename, ".qb" );
110  }
111 
112  // Ask the Asset Manager for the .qb - it will be loaded if it is not yet memory resident.
113  Ass::CAssMan *p_ass_man = Ass::CAssMan::Instance();
114  mp_node_array = (Script::CArray*)p_ass_man->LoadOrGetAsset( nodearray_filename, false, false, false, 0 );
115 }

164 CBaseComponent::EMemberFunctionResult CNodeArrayComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
165 {
166  switch( Checksum )

168 /*
169  // @script | DoSomething | does some functionality

171  DoSomething();
172  break;

178  }
179  break;
180 */
181 
182  default:
183  return CBaseComponent::MF_NOT_EXECUTED;
184  }
185

187  // unrecognized functions; if we make it down here,
188  // that means that the component already handled it
189  // somehow
