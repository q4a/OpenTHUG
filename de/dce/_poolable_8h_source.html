79 template<class _T>
80 void *CPoolable<_T>::operator new (size_t size)
81 {
82  Dbg_Assert(size == sizeof(_T));
83 
84  //printf("in CPoolable operator new\n");
85 
86  #ifdef __NOPT_ASSERT__
87  if (gHeapPools)
88  {
89  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().DebugHeap());
90  void *p = ::new char[size];
91  Mem::Manager::sHandle().PopContext();
92  return p;
93  }
94  #endif
95 
96  if (sp_pool[s_currentPool])
97  {
98  return sp_pool[s_currentPool]->Allocate();
99  }
100 
101  return ::new char[size];
102 }

107 template<class _T>
108 void CPoolable<_T>::operator delete (void * pMem)
109 {
110  Dbg_Assert(pMem);
111  #ifdef __NOPT_ASSERT__
112  if (gHeapPools)
113  {
114  delete [] (char *)pMem;
115  return;
116  }
117  #endif
118 
119  // pMem contains no info as to which pool it was allocated from, because that would
120  // use up too much memory.
121  // So query the pools to see if it is in them.
122 
123  CCompactPool *p_pool=sp_pool[s_currentPool];
124  // Check if it is in the current pool, which it will be most of the time.
125  if (p_pool && p_pool->IsInPool(pMem))
126  {
127  p_pool->Free(pMem);
128  }
129  else
130  {
131  // Otherwise try the other one, which exists for saving games to mem card.
132  Dbg_MsgAssert(POOL_STACK_SIZE==2,("Only two pool supported at the moment"));
133  p_pool=sp_pool[s_currentPool ^ 1];
134  if (p_pool && p_pool->IsInPool(pMem))
135  {
136  p_pool->Free(pMem);
137  }
138  else
139  {
140  delete [] (char *)pMem;
141  }
142  }
143 }
