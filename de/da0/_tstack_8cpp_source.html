183 void Stack::Push( void )
184 {
185  /*
186  On entry:
187  m_add -> top list on stack (the one being processed)
188  m_run -> probably the same
189 
190  On exit:
191  m_run -> new top list
192  m_add -> old top list (the one being processed)
193  */
194 
195 
196  Dbg_MsgAssert(!m_pushedList,( "push already called this frame"));
197 
198  m_run = new StackElement;
199 
200  Dbg_AssertType( m_run, StackElement );
201 
202  m_stack.AddToHead( m_run->m_node ); // deferred push
203  m_pushedList = true;
204 
205 }

212 void Stack::Pop( void )
213 {
214  /*
215  On entry:
216  m_add -> top list on stack (the one being processed)
217  m_run -> top list on stack
218 
219  On exit:
220  m_run and m_add -> the new top list (originally second to top)
221  m_dead -> list to be destroyed (still being processed)
222  */
223 
224 
225  Dbg_AssertType( m_run, StackElement );
226  // this assert is problematic because the task system does not call Process() on the input
227  // task list stacks for inactive devices, though Push() and Pop() are called
228  //Dbg_MsgAssert(!m_poppedList,( "pop already called this frame"));
229 
230  // remove item
231 
232  Lst::Search< StackElement > iterator;
233  iterator.FirstItem( m_stack );
234 
235  // must use this test, m_pushedList is not reliable
236  if (m_add == m_run)
237  m_add = iterator.NextItem();
238  else
239  {
240  // otherwise, current list is retained, m_add stays the same
241  m_pushedList = false;
242  }
243 
244 
245  // m_run may have been just added this frame, but that shouldn't matter
246  Dbg_MsgAssert( m_run->IsEmpty(),( "Tasks List not Empty" ));
247  if( HaveAccess() )
248  {
249  delete m_run;
250  }
251  else
252  {
253  m_run->m_node->Remove();
254  m_dead.AddToHead( m_run->m_node ); // deferred kill
255  }
256 
257  m_run = m_add;
258  m_poppedList = true;
259 }
