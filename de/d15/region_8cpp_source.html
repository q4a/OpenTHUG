148 // Givne an allocator pAlloc, then allocate some space from this region
149 // for this allocator
150 // account for the m_dir flag in the allocator, which will be -1 or +1
151 
152 void* Region::Allocate( Allocator* pAlloc, size_t size, bool assert_on_fail )
153 {
154 
155 
156  Dbg_AssertType ( pAlloc, Allocator );
157 
158  // Get index = 0 if BottomUp
159  // index = 1 if TopDown
160  int index = ( pAlloc->m_dir & 2 ) >> 1;
161  Dbg_MsgAssert ( m_alloc[index] == pAlloc,( "Allocator not currently registered" ));
162 
163  // mp_top of an allocator is the "Top" of the heap relative to what direction it is going in
164  // hence for a BottomUp heap, the mp_top will be the byte after the highest byte
165  // and for a TopDown heap, the mp_top will be the lowest byte
166 
167  // here we get a new p_top for the allocator in pAlloc
168  // then we just need to test it
169  void* p_top = (void*) (( (uint)pAlloc->mp_top ) + ( size * pAlloc->m_dir ) );
170 
171  if( Allocator* p_alloc = m_alloc[index^1] ) // If other allocator exists
172  {
173  if ( index == vBOTTOM_UP ) // we are bottom up, other allocater is top down
174  {
175  if ( p_top > p_alloc->mp_top ) // our new top is above the other allocator's base
176  {
177  if( assert_on_fail ) // so fail
178  {
179  Dbg_MsgAssert( false,( "Failed to Allocate %d bytes", size ));
180  }
181  Dbg_Warning( "Failed to Allocate %d bytes", size );
182  return NULL;
183  }
184  }
185  else // we are top down, other allocator is bottom up
186  {
187  if ( p_top < p_alloc->mp_top ) // new top is below the other allocator's top
188  {
189  if( assert_on_fail ) // so fail
190  {
191  Dbg_MsgAssert( false,( "Failed to Allocate %d bytes", size ));
192  }
193  Dbg_Warning( "Failed to Allocate %d bytes", size );
194  return NULL;
195  }
196  }
197  }
198  else
199  {
200  // here there is no other allocator
201  if ( index == vBOTTOM_UP ) // for a single bottom up heap
202  {
203  if ( p_top > mp_end ) // if new p_top goes past end of region
204  {
205  if( assert_on_fail ) // then fail
206  {
207  Dbg_MsgAssert( false,( "Failed to Allocate %d bytes", size ));
208  }
209  Dbg_Warning( "Failed to Allocate %d bytes", size );
210  return NULL;
211  }
212  }
213  else // single top down heap
214  {
215  if ( p_top < mp_start ) // p_top goes below start of region
216  {
217  if( assert_on_fail )
218  {
219  Dbg_MsgAssert( false,( "Failed to Allocate %d bytes", size ));
220  }
221  Dbg_Warning( "Failed to Allocate %d bytes", size );
222  return NULL;
223  }
224  }
225  }
226 
227  // If it's vBOTTOM_UP then
228  // return pAlloc->mp_top, which is the old end of the heap, and the start of the new block
229  // else
230  // return p_top, which is the base of the new block (when heap growing downwards)
231 
232  void* p_ret = ( index == vBOTTOM_UP ) ? pAlloc->mp_top : p_top;
233 
234  // store new "top" of heap
235  pAlloc->mp_top = p_top;
236 
237  int free = MemAvailable();
238  if (free < m_min_free)
239  {
240  m_min_free = free;
241  }
242 
243 
244  // and return the start of the new memory
245  return p_ret;
246 }
