38 /*
39  * binds to eznetcnf.irx RPC services.
40  * r == 1 after RPC is bound.
41  */
42 int
43 initEzNetCnf( void )
44 {
45 #ifdef DEBUG
46  int id = sceSifSearchModuleByName(kModName_ezNetCnf);
47  if (id < 0) {
48  printf("\"%s\" module not found (%d)\n", kModName_ezNetCnf, id);
49  }
50 #endif
51 
52  // bind RPC
53  while (1) {
54  int i = 0x10000;
55  if (sceSifBindRpc(&m_rpcCd, kRPCId_ezNetCnf, 0) < 0) {
56  printf("could not bind to eznetcnf service\n");
57  }
58  if (m_rpcCd.serve != 0) break;
59  while (i--)
60  ;
61  }
62 
63  return m_rpcRunning = 1;
64 }
65 
66 /*
67  * exit and unload eznetcnf.irx, then unbind RPC services.
68  * r == 0 after RPC is unbound.
69  */
70 int
71 exitEzNetCnf( void )
72 {
73  if (m_rpcRunning == 1) {
74  // call using non-blocking mode,
75  // since eznetcnf unloads and the call cannot return.
76  sceSifCallRpc(&m_rpcCd, eCnfEnd, SIF_RPCM_NOWAIT,
77  0, 0, 0, 0, 0, 0);
78  // not sure if this is necessary.
79  sceSifInitRpc(0);
80  }
81 
82  return m_rpcRunning = 0;
83 }

94  * sceNetCnfGetCount() returns 0 normally when no netdb file

97  * r > 0 for valid netdb with at least 1 Combination,

154 /*
