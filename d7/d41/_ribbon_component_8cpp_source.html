121 void CRibbonComponent::Update()
122 {
123  float frame_length = Tmr::FrameLength();
124 
125  // update link positions based on velocity
126  for (int n = m_num_links; n--; )
127  {
128  Mth::Vector vel = mp_links[n].p - mp_links[n].last_p;
129  vel *= 1.0f / m_last_frame_length;
130 
131  // damp
132  vel -= 10.0f * frame_length * vel;
133 
134  // momentum
135  mp_links[n].next_p = mp_links[n].p + frame_length * vel;
136 
137  // gravity
138  mp_links[n].next_p[Y] -= 800.0f * frame_length * frame_length;
139  }
140 
141  // fetch bone position
142  Mth::Vector bone_p;
143  mp_skeleton_component->GetBonePosition(m_bone_name, &bone_p);
144  bone_p = GetObject()->GetMatrix().Rotate(bone_p);
145  bone_p += GetObject()->GetPos();
146 
147  // update link constraints
148  for (int c = 20; c--; )
149  {
150  // first link special case
151  float length = (mp_links[0].next_p - bone_p).Length();
152  Mth::Vector axis = (mp_links[0].next_p - bone_p) * (1.0f / length);
153  float adjustment = m_link_length - length;
154  mp_links[0].next_p += adjustment * axis;
155 
156  // second and subsequent links
157  for (int n = 1; n < m_num_links; n++)
158  {
159  float length = (mp_links[n].next_p - mp_links[n - 1].next_p).Length();
160  Mth::Vector axis = (mp_links[n].next_p - mp_links[n - 1].next_p) * (1.0f / length);
161 
162  float adjustment = 0.5f * (m_link_length - length);
163 
164  // hack
165  if (length > 20.0f * m_link_length)
166  {
167  CallMemberFunction(CRCD(0x796084f4, "Ribbon_Reset"), NULL, NULL);
168  return;
169  }
170 
171  mp_links[n].next_p += adjustment * axis;
172  mp_links[n - 1].next_p -= adjustment * axis;
173  }
174  }
175 
176  #if 0
177 
178  // cheap collision detection
179  float min_y = GetObject()->GetPos()[Y] + 0.1f;
180  for (int n = m_num_links; n--; )
181  {
182  mp_links[n].next_p[Y] = Mth::ClampMin(mp_links[n].next_p[Y], min_y);
183  }
184 
185  #else
186 
187  Mth::CBBox bbox;
188  Nx::CCollCache* p_coll_cache = Nx::CCollCacheManager::sCreateCollCache();
189  for (int n = m_num_links; n--; )
190  {
191  bbox.AddPoint(mp_links[n].p);
192  bbox.AddPoint(mp_links[n].next_p);
193  }
194  p_coll_cache->Update(bbox);
195 
196  CFeeler feeler;
197  feeler.SetCache(p_coll_cache);
198  // real collision detection
199  for (int n = m_num_links; n--; )
200  {
201  feeler.m_start = mp_links[n].p;
202  feeler.m_end = mp_links[n].next_p;
203  if (!feeler.GetCollision()) continue;
204 
205  Mth::Vector movement = mp_links[n].next_p - mp_links[n].p;
206 
207  mp_links[n].next_p = feeler.GetPoint() + 0.1f * feeler.GetNormal();
208 
209  // assume no additional collision
210  movement *= 1.0f - feeler.GetDist();
211  movement.ProjectToPlane(feeler.GetNormal());
212 
213  mp_links[n].next_p += movement;
214  }
215 
216  Nx::CCollCacheManager::sDestroyCollCache(p_coll_cache);
217 
218  #endif
219 
220  // update state
221  for (int n = m_num_links; n--; )
222  {
223  mp_links[n].last_p = mp_links[n].p;
224  mp_links[n].p = mp_links[n].next_p;
225  }
226  m_last_frame_length = frame_length;
227 
228  // debug draw
229  Gfx::AddDebugLine(bone_p, mp_links[0].p, m_color, 0, 1);
230  for (int n = 1; n < m_num_links; n++)
231  {
232  Gfx::AddDebugLine(mp_links[n].p, mp_links[n - 1].p, m_color, 0, 1);
233  }
234 }

241 CBaseComponent::EMemberFunctionResult CRibbonComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
242 {
243  switch ( Checksum )
244  {
245  case CRCC(0x796084f4, "Ribbon_Reset"):
246  {
247  MESSAGE("Resetting");
248 
249  Mth::Vector bone_p;
250  mp_skeleton_component->GetBonePosition(m_bone_name, &bone_p);
251  bone_p = GetObject()->GetMatrix().Rotate(bone_p);
252  bone_p += GetObject()->GetPos();
253 
254  for (int n = m_num_links; n--; )
255  {
256  mp_links[n].p = bone_p;
257  mp_links[n].last_p = bone_p;
258  mp_links[n].p[Y] += 0.1f * n;
259  mp_links[n].last_p[Y] += 0.1f * n;
260  }
261  break;
262  }
263 
264  default:
265  return CBaseComponent::MF_NOT_EXECUTED;
266  }
267  return CBaseComponent::MF_TRUE;
268 }
