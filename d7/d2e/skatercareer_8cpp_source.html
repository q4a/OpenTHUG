53 int CountOnes(uint32 v)
54 {
55 
56 
57  int NumOnes=0;
58  for (int i=0; i<32; ++i)
59  {
60  if (v&1)
61  {
62  ++NumOnes;
63  }
64  v>>=1;
65  }
66  return NumOnes;
67 }

185 bool CSkaterCareer::GotAllGaps()
186 {
187  //Rulon: Made this function more robust to handle the changing level arrangement.
188  Script::CArray *p_array = Script::GetArray(CRCD(0xeb1a4bc9,"levels_with_gaps"),Script::ASSERT);
189  uint32 array_size = p_array->GetSize();
190 
191  for ( uint32 i = 0; i < vMAX_CAREER_LEVELS; ++i )
192  {
193  // Skip up levels we dont care about
194  uint32 j = 0;
195  for ( j = 0; j < array_size; ++j)
196  {
197  uint32 check = p_array->GetChecksum(j);
198  uint32 level_num = Script::GetInt(check ,Script::ASSERT);
199  if( i == level_num )
200  {
201  //printf("level %i\n", i);
202  break;
203  }
204  }
205 
206  if (j == array_size)
207  continue;
208 
209  Obj::CGapChecklist* pGapChecklist = GetGapChecklist( i );
210 
211  // If this level should be taken into account but is unvisited
212  if ( !this->HasVisitedLevel( i ) )
213  {
214  //printf("haven't been to level %i\n", i);
215  return false;
216  }
217 
218  if ( !pGapChecklist->GotAllGaps() )
219  {
220  //printf("didn't get all the gaps in level %i\n", i);
221  return false;
222  }
223  //printf("got all gaps in level %i\n", i);
224  }
225  return true;
226

428 // Returns 0 if no medal was won.
429 // If a medal was won, it returns 1+the best-medal number, where the medal number is whatever it is defined to be
430 // in goal_scripts.q
431 int CSkaterCareer::GetBestMedal(int Level)
432 {
433 
434 
435  Dbg_MsgAssert(Level>=0 && Level<(int)vMAX_CAREER_LEVELS,("Bad Level of %d sent to GetBestMedal",Level));
436 
437  int Gold=Script::GetInteger("GOAL_GOLD");
438  int Silver=Script::GetInteger("GOAL_SILVER");
439  int Bronze=Script::GetInteger("GOAL_BRONZE");
440 
441  Dbg_MsgAssert(s_is_competition[Level],("Level %d is not a competition level",Level));
442 
443  if (m_goal_flags[Level][Gold>>5] & (1<<Gold))
444  {
445  return Gold+1;
446  }
447  else if (m_goal_flags[Level][Silver>>5] & (1<<Silver))
448  {
449  return Silver+1;
450  }
451  else if (m_goal_flags[Level][Bronze>>5] & (1<<Bronze))
452  {
453  return Bronze+1;
454  }
455  else
456  {
457  return 0;
458  }
459 
460  return 0;
461 }

746  {
