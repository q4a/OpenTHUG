8 /*
9  Pathob.cpp
10 
11  This module contains path-navigation functionality.
12 
13  A moving object contains a pointer to this pathob. When
14  the moving object wants to follow a path along waypoints, it
15  just initializes this pathob with the current position and tells
16  the pathob where it wants to go. Each frame, the moving object
17  calls TraversePath() and sends in the distance travelled, and
18  its own heading. The pathob moves accordingly, but the point remains
19  on the plane defined by the points of the path.
20 
21  The pathob is its own entity: it's essentially a point in
22  space and an orientation. When the moving object calls TraversePath(),
23  the pathob just does the math and figures out where along the path
24  to move to. The moving object takes the new path-ob position, finds
25  the ground above or below this position, and possibly sets its heading
26  so that it's looking the correct direction down the path.
27 
28  The pathob logic handles a simple path between two points... it also
29  handles the curves from one simple path to the next (so the path along
30  three waypoints, but instead of a sharp turn the pathob follows a curve
31  between the two lines).
32 
33  As the pathob hits waypoints, it communicates to the moving object (in
34  case the moving object needs to run a script at the waypoint).
35 */

189 int CPathOb::SetUpTurningCircle( void )
190 {
191  Mth::Vector vectA, vectB;
192 
193  // m_nav_pos holds one point on the circle.
194  // on the line from m_wp_pos_from to m_wp_pos_to
195  // (at the distance of nav pos from m_wp_pos_from)
196  // will be the other point on the circle.
197  vectB = m_nav_pos - m_wp_pos_from;
198 
199  float actualTurnDist = vectB.Length();
200 
201  m_circle_arrival_pos = m_wp_pos_from;
202  vectA = m_wp_pos_to - m_wp_pos_from;
203  float distBetweenWps = vectA.Length();
204  vectA *= ( actualTurnDist / distBetweenWps );
205  m_circle_arrival_pos += vectA;
206 
207  // That's all the info from the turning circle we need, if we're doing a linear interpolation:
208  if ( m_pathob_flags & PATHOBFLAG_LINEAR_INTERP )
209  {
210  return TURN_METHOD_LINEAR_INTERP;
211  }
212 
213 // Dbg_MsgAssert( actualTurnDist, "Traveling too fast for turn dist... increase turn dist or decrease velocity." );
214  if ( !actualTurnDist )
215  {
216  return TURN_METHOD_INSTANT_TURN;
217  }
218 
219  // find the radius of the turning circle
220  // the angle between the line we're on and the line to
221  // the arrival pos on the next line is the same as half
222  // the angle that the two radii where the circle touches
223  // the waypoint lines form... use the similar triangles
224  // to get the circle radius:
225  vectA = m_circle_arrival_pos - m_nav_pos;
226  vectA /= 2; // vector from current pos to midpoint along current pos to arrival pos...
227  float lengthPosToMidpoint = vectA.Length();
228 
229  vectB = m_nav_pos + vectA;
230  vectB -= m_wp_pos_from;
231  float distWPtoMP = vectB.Length();
232 
233  if ( ( !distWPtoMP ) | ( ( distWPtoMP * MAX_RATIO_OF_HYPOTENUSE_TO_OPPOSITE_FOR_TURN_CIRCLE ) < lengthPosToMidpoint ) )
234  {
235  // almost a straight line... use linear interpolation method:
236  return TURN_METHOD_LINEAR_INTERP;
237  }
238  // to understand this, draw the triangles and derive circle radius
239  // in terms of the known lengths of the similar triangles:
240  m_circle_radius = lengthPosToMidpoint * actualTurnDist;
241  m_circle_radius /= distWPtoMP;
242 
243  if ( !m_circle_radius )
244  {
245  // tiny turn radius... snap to the point and shit like that!
246  return TURN_METHOD_INSTANT_TURN;
247  }
248 
249  // find the location of the center of the circle...
250  // the triangle formed by the current pos, the waypoint pos,
251  // and the midpoint between the current pos and the arrival pos
252  // is a similar triangle to the current pos, the waypoint pos,
253  // and the center of the circle... therefore the distance from
254  // the center of the circle to the waypoint pos can be found:
255  float distFromWaypointToCircleCenter = actualTurnDist * actualTurnDist;
256  distFromWaypointToCircleCenter /= vectB.Length();
257  // take the vector from the waypoint to the midpoint, make it this length,
258  // and viola we have the circle center:
259  m_circle_center = vectB;
260  m_circle_center.Normalize( distFromWaypointToCircleCenter );
261  m_circle_center += m_wp_pos_from;
262 
263  // test the distance from the current pos... should equal the frickin' radius:
264  /*
265  {
266  Mth::Vector temp = m_circle_center;
267  temp -= m_nav_pos;
268  Dbg_Message( "dist %f radius %f", temp.Length(), m_circle_radius );
269  }*/
270 
271  if ( m_circle_radius < m_min_turn_circle_radius )
272  {
273  Dbg_Message( "angle too acute to path between nodes %d and %d", m_node_from, m_node_to );
274  Dbg_MsgAssert( 0,( "Turn is too tight for this object... your waypoints are just too acute for me!" ));
275  }
276 
277  // find the "rotation axis" (the y axis if we're traveling along the xz plane)
278  // and set up a rotation matrix where the axis along the 1st waypoint path [ r0 ]
279  // sweeps along the circle as we rotate about the 'up'[ r1 ] axis in the direction if r2...
280  m_turn_matrix[ X ] = m_circle_center - m_nav_pos;
281  m_turn_matrix[ Z ] = m_circle_center - m_circle_arrival_pos;
282  if ( !( m_turn_matrix[ X ].Length() && m_turn_matrix[ Z ].Length() ) )
283  {
284  return TURN_METHOD_INSTANT_TURN;
285  }
286  m_turn_matrix[ X ].Normalize();
287  m_turn_matrix[ Z ].Normalize();
288 
289  // while we have these two vectors, find the angle between them...
290  float angCos = Mth::DotProduct( m_turn_matrix[ X ], m_turn_matrix[ Z ] );
291 #ifdef __PLAT_NGC__
292  float ang = angCos;
293  if ( ang > 1.0f ) ang = 1.0f;
294  if ( ang < -1.0f ) ang = -1.0f;
295  m_arc_ang = acosf( ang ); // angles given in radians!
296 #else
297  m_arc_ang = acosf( angCos ); // angles given in radians!
298 #endif // __PLAT_NGC__
299 
300  // find the total circumference of the arc,
301  // which is the circumference of the circle ( ang * radius )
302  // multiplied by ang / ( 2 * pi ) [ the fraction of the circle we actually traverse... ]
303  m_arc_dist = m_circle_radius * m_arc_ang;
304 
305  // continue setting up the matrix now...
306  m_turn_matrix[ Y ] = Mth::CrossProduct( m_turn_matrix[ X ], m_turn_matrix[ Z ] );
307  if ( !m_turn_matrix[ Y ].Length() )
308  {
309  return TURN_METHOD_INSTANT_TURN;
310  }
311  m_turn_matrix[ Y ].Normalize();
312  m_turn_matrix[ Z ] = Mth::CrossProduct( m_turn_matrix[ X ], m_turn_matrix[ Y ] );
313  if ( !m_turn_matrix[ Z ].Length() )
314  {
315  return TURN_METHOD_INSTANT_TURN;
316  }
317  m_turn_matrix[ Z ].Normalize();
318 
319  // Phew!!! Now we can rotate this matrix along r1 and get the position on
320  // the circle by taking r0 multiplied by the circle radius and adding that to
321  // the center of the circle... and r2 will be the direction the car is pointed!!
322 
323  // now to find the position along the arc, just get the
324  // current distance traveled along the arc, divide by
325  // the total circumference of the arc, multiply by the
326  // angle the arc traverses to get the angle currently
327  // traversed... by rotating our curve matrix along
328  // y in this fashion, we can find the position on the
329  // circle!
330 
331  m_dist_traveled = m_extra_dist;
332  m_extra_dist = 0;
333 
334 // Gfx::AddDebugLine( m_nav_pos, m_circle_center, MAKE_RGB( 255, 0, 0 ), MAKE_RGB( 255, 0, 0 ) );
335 // Gfx::AddDebugLine( m_circle_center, m_circle_arrival_pos, MAKE_RGB( 0, 0, 255 ), MAKE_RGB( 0, 255, 0 ) );
336 
337  return true;
338 }

360 void CPathOb::UpdateNavPosAlongCurve( float distTraveled )
361 {
362  float fractionOfTotal;
363  fractionOfTotal = distTraveled / m_arc_dist;
364 
365  m_y_rot = m_arc_ang * fractionOfTotal;
366  m_turn_matrix.RotateYLocal( m_y_rot );
367 
368  Mth::Vector temp;
369  temp = m_turn_matrix[ Y ];
370  if ( temp[ Y ] > 0.0f )
371  {
372  m_y_rot *= -1.0f;
373  }
374 
375  m_nav_pos = m_turn_matrix[ X ] * -m_circle_radius;
376  m_nav_pos += m_circle_center;

395 // If the distance between the next two waypoints is less than
396 // the turn dist, decrease the turn dist and shit (otherwise
397 // the arrival pos will be on the line but PAST the two waypoints
398 // rather than between them!
399 bool CPathOb::SetUpTurnDist( void )
400 {
401  Mth::Vector temp;
402  float wpLength;
403 
404  if ( m_enter_turn_dist )
405  {
406  if ( !SkateScript::GetNumLinks( m_node_to ) )
407  return false;
408  SkateScript::GetPosition( SkateScript::GetLink( m_node_to, 0 ), &temp );
409  temp -= m_wp_pos_to;
410  // Need to optimise? Store this length instead of recalculating it
411  // when getting next waypoint!
412  wpLength = temp.Length();
413  if ( wpLength < m_enter_turn_dist )
414  {
415  m_current_turn_dist = wpLength / 2.0f;
416  temp = m_wp_pos_to - m_nav_pos;
417  float distToCurrentWP = temp.Length();
418  if ( m_current_turn_dist > distToCurrentWP )
419  {
420  m_current_turn_dist = distToCurrentWP;
421  }
422  m_pathob_flags |= PATHOBFLAG_LINEAR_INTERP;
423  return true;
424  }
425  }
426 
427  m_pathob_flags &= ~PATHOBFLAG_LINEAR_INTERP;
428 
429  m_current_turn_dist = m_enter_turn_dist;
430 
431  return true;
432 }

554 float CPathOb::GetDistToWaypoint( void )
555 {
556  switch ( m_pathob_state )
557  {
558  case PATHOB_STATE_UNINITIALIZED:
559  //Dbg_Message( "%f line %d", HUGE_DISTANCE, __LINE__ );
560  return HUGE_DISTANCE;
561  break;
562 
563  case PATHOB_STATE_FOLLOW_LINE:
564  //Dbg_Message( "%f line %d", m_dist_to_wp - m_dist_traveled, __LINE__ );
565  return ( m_dist_to_wp - m_dist_traveled );
566  break;
567 
568  case PATHOB_STATE_FOLLOW_CURVE:
569  if ( m_dist_traveled < ( m_arc_dist / 2 ) )
570  {
571  //Dbg_Message( "%f line %d", ( m_arc_dist / 2 ) - m_dist_traveled, __LINE__ );
572  return ( ( m_arc_dist / 2 ) - m_dist_traveled );
573  }
574  else
575  {
576  Mth::Vector temp = m_wp_pos_to - m_wp_pos_from;
577  //Dbg_Message( "%f line %d", temp.Length(), __LINE__ );
578  return temp.Length();
579  }
580  break;
581 
582  case PATHOB_STATE_LINEAR_INTERP:
583  if ( m_dist_traveled < m_dist_to_wp )
584  {
585  //Dbg_Message( "%f line %d", m_dist_to_wp - m_dist_traveled, __LINE__ );
586  return ( m_dist_to_wp - m_dist_traveled );
587  }
588  else
589  {
590  Mth::Vector temp = m_wp_pos_to - m_wp_pos_from;
591  //Dbg_Message( "%f line %d", temp.Length(), __LINE__ );
592  return temp.Length();
593  }
594  break;
595 
596  default:
597  Dbg_MsgAssert( 0, ( "unhandled pathob state" ) );
598  break;
599  }
600 
601  return HUGE_DISTANCE;
602 }

632 void CPathOb::PrepareTurn( void )
633 {
634  int retVal = SetUpTurningCircle();
635  switch ( retVal )
636  {
637  case TURN_METHOD_LINEAR_INTERP:
638  {
639  // store the current pos in circle center:
640  // the angle is too wide (too close to PI radians)...
641  // just move along the two lines and gradually turn and shit like that...
642 
643  m_circle_center = m_nav_pos;
644  Mth::Vector temp = m_nav_pos - m_wp_pos_from;
645  m_dist_to_wp = temp.Length();
646  m_dist_traveled = m_extra_dist;
647  m_pathob_state = PATHOB_STATE_LINEAR_INTERP;
648  break;
649  }
650 
651  case TURN_METHOD_CURVE:
652  m_pathob_state = PATHOB_STATE_FOLLOW_CURVE;
653  break;
654 
655  case TURN_METHOD_INSTANT_TURN:
656  // just hit the waypoint, and set up the path for the next one:
657  Dbg_Assert( GetMotionComponentFromObject( mp_object ) );
658  GetMotionComponentFromObject( mp_object )->HitWaypoint( m_node_from );
659  m_pathob_flags &= ~PATHOBFLAG_SEND_HIT_WAYPOINT_MSG;
660  m_dist_traveled = m_dist_to_wp - m_dist_traveled;
661  // now move along the line...
662  SetUpNavPointLineMotion();
663  UpdateNavPosAlongPath();
664  m_pathob_state = PATHOB_STATE_FOLLOW_LINE;
665  break;
666 
667  default:
668  Dbg_MsgAssert( 0,( "Unknown turn method %d", retVal ));
669  break;
670  }
671 }

678 // Function TraversePath()
679 // This function is called by the moving object each frame, and causes the
680 // pathob to move a little further down the path (by the projection of the moving
681 // object's forward vector along the line or curve in the path).
682 // Returns 'TRUE' when the end of the path is reached...
683 //bool CPathOb::TraversePath( bool is_being_skitched, const float dist, Mth::Vector *pHeading )
684 bool CPathOb::TraversePath( const float dist, Mth::Vector *pHeading )
685 {
686  //if (is_being_skitched) printf("#");
687 
688  if ( m_pathob_state == PATHOB_STATE_IDLE )
689  {
690  return true;
691  }
