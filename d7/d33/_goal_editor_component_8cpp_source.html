382 void CEditGoal::refresh_position_using_collision_check(uint positionIndex)
383 {
384  Dbg_MsgAssert(positionIndex < MAX_POSITIONS_PER_GOAL,("Bad positionIndex"));
385 
386  Mth::Vector pos=mp_item_positions[positionIndex].mPos;
387  // First, do a drop down check from the current position
388  Mth::Vector start=pos;
389  start[Y]+=10000.0f;
390  Mth::Vector end=pos;
391  end[Y]-=10000.0f;
392 
393  CFeeler feeler;
394  feeler.SetStart(start);
395  feeler.SetEnd(end);
396  if (feeler.GetCollision())
397  {
398  mp_item_positions[positionIndex].mPos=feeler.GetPoint();
399  return;
400  }
401 
402  // No collision found, so try starting from much higher up
403  start[Y]+=10000.0f;
404  feeler.SetStart(start);
405  feeler.SetEnd(end);
406  if (feeler.GetCollision())
407  {
408  mp_item_positions[positionIndex].mPos=feeler.GetPoint();
409  }
