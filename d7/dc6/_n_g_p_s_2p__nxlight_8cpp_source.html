55 void CPs2ModelLights::plat_update_engine( Mth::Vector & pos, bool add_scene_light )
56 {
57  // Figure Scene Lighting if required.
58  if( add_scene_light )
59  {
60  // Copy the object position into our position. If this pointer is set to NULL,
61  // we probably won't find any Scene Lights.
62  if (mp_pos)
63  {
64  pos = *mp_pos;
65  }
66  else
67  {
68  Dbg_MsgAssert(0, ("No object position pointer for scene lights"));
69  }
70 
71  // We should check if any calcs are necessary first by seeing if this position is
72  // "visible", but for now, we'll do them all.
73  Nx::CSceneLight *p_scene_light = CLightManager::sGetOptimumSceneLight( pos );
74  if( p_scene_light )
75  {
76  Dbg_Assert(mp_light_group);
77 
78  Mth::Vector light_pos = p_scene_light->GetLightPosition();
79 
80  float dist = Mth::Distance( pos, light_pos );
81  float ratio = dist * p_scene_light->GetLightReciprocalRadius();
82 
83  light_pos = - ( pos - light_pos ).Normalize();
84 
85  // Figure the direction...
86  mp_light_group->SetDirection(SCENE_LIGHT_INDEX, light_pos);
87 
88  // ...and the color.
89  ratio = sqrtf( 1.0f - ratio ) * p_scene_light->GetLightIntensity();
90  Mth::Vector color(ratio * p_scene_light->GetLightColor().r,
91  ratio * p_scene_light->GetLightColor().g,
92  ratio * p_scene_light->GetLightColor().b,
93  0);
94 
95  mp_light_group->SetBaseDiffuseColor(SCENE_LIGHT_INDEX, color);
96 
97  plat_enable_diffuse_light(SCENE_LIGHT_INDEX, true);
98  }
99  else
100  {
101 #if 0
102  // Disable this light by setting zero color.
103  Mth::Vector color(0, 0, 0, 0);
104 
105  mp_light_group->SetBaseDiffuseColor(SCENE_LIGHT_INDEX, color);
106 #endif
107  plat_enable_diffuse_light(SCENE_LIGHT_INDEX, false);
108  }
109  }
110 }
