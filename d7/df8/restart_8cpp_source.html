97 int FindQualifyingNodes( uint32 type, uint32 *node_buffer, int first = -1 )
98 {
99  int num_qualifiers;
100  int i;
101 
102  num_qualifiers = 0;
103  i = 0;
104 
105  // get the node array
106  Script::CArray *pNodeArray=Script::GetArray(CRCD(0xc472ecc5,"NodeArray"));
107 
108  // if there is no node array, then there are no restart points
109  if (!pNodeArray)
110  {
111  return 0;
112  }
113 
114  if( first >= 0 )
115  {
116  i = first + 1;
117  }
118  // scan through it from the start point
119  while( i < (int)pNodeArray->GetSize() )
120  {
121  Script::CScriptStructure *pNode=pNodeArray->GetStructure(i);
122  if (IsRestart(pNode)) // checksum of "Restart"
123  {
124  Script::CArray *pArray;
125  int j;
126 
127  // The caller wants all restart points
128  if (type == 0)
129  {
130  node_buffer[num_qualifiers++] = i;
131  i++;
132  continue;
133  }
134 
135  // Loop through supported game types, gathering matches
136  if (pNode->GetArray("restart_types", &pArray))
137  {
138  for (j = 0; j < (int)pArray->GetSize(); j++)
139  {
140  uint32 game_type;
141 
142  game_type = pArray->GetNameChecksum( j );
143  if( game_type == type )
144  {
145  node_buffer[num_qualifiers++] = i;
146  }
147  }
148  }
149  }
150  i++;
151  }
152 
153  return num_qualifiers;
154 }

157 // given the checksum of a type of restart
158 // then get the node index
159 // if "first" is specified then resturn the first on after this
160 // if "type" is zero, then it will be ignored, and will return restarts of any type.
161 // returns -1 if no restarts matching "type" are found.
162 int GetRestartNode(uint32 type, int index, int first = -1)
163 {
164  // decide where to start, based on the "first" parameter
165  int num_qualifiers;
166  uint32 qualifying_nodes[256];
167 
168  num_qualifiers = FindQualifyingNodes( type, qualifying_nodes, first );
169  if( num_qualifiers == 0 )
170  {
171  return -1;
172  }
173 
174  #ifdef __NOPT_ASSERT__
175  if (num_qualifiers > 1 && (type == CRCD(0xbae0e4da,"multiplayer") || type == CRCD(0x9d65d0e7,"horse")))
176  {
177  // Scan through the nodes we've found, and check that there are not any
178  // that are too close to each other (< 6 feet)
179  bool ok = true;
180  for (int i=0;i<num_qualifiers-1;i++)
181  {
182  for (int j=i+1;j<num_qualifiers;j++)
183  {
184  Dbg_MsgAssert(qualifying_nodes[i] != qualifying_nodes[j],("Restart node %d qualifies more than once!\n",qualifying_nodes[i]));
185  Mth::Vector i_pos, j_pos;
186  SkateScript::GetPosition( qualifying_nodes[i], &i_pos );
187  SkateScript::GetPosition( qualifying_nodes[j], &j_pos );
188  float dist = (i_pos-j_pos).Length();
189  if (dist < 12.0f*6.0f)
190  {
191  if (ok) printf ("\n\n\n");
192  ok = false;
193  uint32 i_name_checksum;
194  SkateScript::GetNode(qualifying_nodes[i])->GetChecksum("Name",&i_name_checksum);
195  uint32 j_name_checksum;
196  SkateScript::GetNode(qualifying_nodes[j])->GetChecksum("Name",&j_name_checksum);
197  printf ("%2.2f"" %s to %s \n",
198  dist,
199  Script::FindChecksumName(i_name_checksum),
200  Script::FindChecksumName(j_name_checksum)
201  );
202  }
203  }
204  }
205  if (!ok) printf ("\n\n\n");
206  Dbg_MsgAssert(ok, ("Restart nodes of type %s too close, see above for list\n",Script::FindChecksumName(type)));
207  }
208  #endif
209 
210  //Ryan("index is %d, num_qualifiers %d\n", index, num_qualifiers);
211 
212  // If we've found matches, return the
213  if( num_qualifiers > 0 )
214  {
215  if( index >= num_qualifiers )
216  {
217  return qualifying_nodes[index % num_qualifiers];
218  }
219  else
220  {
221  return qualifying_nodes[index];
222  }
223  }
224  return -1;
225 }
