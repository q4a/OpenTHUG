8 /***************************************************************************
9  A composite object is a generic object which basically consists of:
10  - a position and orientation
11  - some flags
12  - a script
13  - a list of components
14 
15  The purpose of the Composite Object is to represent a thing that exists
16  in the game world. For example:
17 
18  - Cars and other vechicals
19  - Pedestrians, animals, etc
20  - Pickup Icons
21  - 3D sound emitters
22  - Particle systems
23  - Skaters or other player type objects
24  - Bouncy Objects like trash cans
25 
26  CCompositeObject is derived from CObject, so all CCompositeObjects have
27  a mp_script (which might be NULL)
28 
29  A composite object is normally comprised of several components. These are
30  generally independent things such as:
31  - A Model componet
32  - A physics component (e.g., drive around, or skating)
33  - A suspend component
34  - and various other components as needed
35 
36  There is one global list of all composite objects. This is managed by the
37  CCompositeObjectManager class, whcih is also responsible for creating
38  the composite objects.
39 
40  A composite object is typically create from an array of structs
41  and a structe that contains the initialization parameters.
42 
43  Here's an example of the array of structs. Note that some components
44  have default parameters. These can be overridden by the struct of parameters
45 
46  gameobj_composite_structure = [
47  { component = suspend }
48  { component = model }
49  { component = exception }
50  { component = collision }
51  ]
52

139 void CCompositeObject::Update()
140 {
141 #ifdef __USE_PROFILER__
142  Sys::CPUProfiler->PushContext( m_profile_color );
143 #endif // __USE_PROFILER__
144 
145  Dbg_MsgAssert(IsFinalized(),("Update called on UnFinalized Composite object %s",Script::FindChecksumName(GetID())));
146 
147 #ifdef __NOPT_ASSERT__
148  // GJ: don't do this if the composite object is paused,
149  // because some lock obj component might be trying
150  // to lock to this object while the game is paused
151  if (!m_composite_object_flags.Test(CO_SUSPENDED)
152  && !m_composite_object_flags.Test(CO_PAUSED))
153  {
154  // clear out the display matrix, to make
155  // sure that no one's relying on the
156  // last frame's matrix
157  m_display_matrix.Ident();
158  }
159 #endif
160 
161 #ifdef __NOPT_ASSERT__
162  // Make sure the component update times are initialized to zero so that they do not
163  // show incorrect old values in the debugger if this function returns before
164  // getting to the component update loop.
165  zero_component_update_times();
166 
167  m_total_script_update_time=0;
168  m_do_game_logic_time=0;
169 #endif
170 
171  if (m_composite_object_flags.Test(CO_PAUSED))
172  {
173 #ifdef __USE_PROFILER__
174  Sys::CPUProfiler->PopContext();
175 #endif // __USE_PROFILER__
176  return;
177  }
178 
179 
180  if (!m_composite_object_flags.Test(CO_SUSPENDED))
181  {
182 #ifdef __NOPT_ASSERT__
183  Tmr::CPUCycles time_before=Tmr::GetTimeInCPUCycles();
184 #endif
185  if ( mp_script )
186  {
187  if ( mp_script->Update() == Script::ESCRIPTRETURNVAL_FINISHED )
188  {
189  // if we have script based events then we only want to kill the script if
190  // it has an empty event handler table
191  // as a script can finish, but still have event handlers
192  // this kind of logic will be duplicated in various places in the code
193  // so we could do with a bit of refactoring
194  // like a mp_script->CanBeDeleted() function
195  #ifdef __SCRIPT_EVENT_TABLE__
196  // in practice it seems okay to just leave the script
197  // probably not a big issue
198  #else
199  delete mp_script;
200  mp_script = NULL;
201  #endif
202  }
203  }
204 
205  // if a script has called "Die",
206  // then don't update the components
207  if ( IsDead() )
208  {
209  # ifdef __USE_PROFILER__
210  Sys::CPUProfiler->PopContext();
211  # endif // __USE_PROFILER__
212  return;
213  }
214 #ifdef __NOPT_ASSERT__
215  // Convert to microseconds by dividing by 150
216  m_total_script_update_time=(Tmr::GetTimeInCPUCycles()-time_before)/150;
217 #endif
218 
219  // transition-only function call,
220  // call each specific object's
221  // DoGameLogic() function (previously,
222  // this was called by each object's task)
223 #ifdef __NOPT_ASSERT__
224  time_before=Tmr::GetTimeInCPUCycles();
225 #endif
226  // Mick: DoGameLogic is Deprecated, and only exists for a few misc objects
227  // it should eventually be removed
228  DoGameLogic();
229 #ifdef __NOPT_ASSERT__
230  // Convert to microseconds by dividing by 150
231  m_do_game_logic_time=(Tmr::GetTimeInCPUCycles()-time_before)/150;
232 #endif
233 
234  }
235 
236  CBaseComponent* pComponent = mp_component_list;
237  while ( pComponent )
238  {
239  if ( ! (pComponent->m_flags.Test(CBaseComponent::BC_NO_UPDATE)))
240  {
241  #ifdef __NOPT_ASSERT__
242  Tmr::CPUCycles time_before_components=Tmr::GetTimeInCPUCycles();
243  #endif
244 
245  pComponent->Update();
246 
247  #ifdef __NOPT_ASSERT__
248  // Convert to microseconds by dividing by 150
249  pComponent->m_update_time=(Tmr::GetTimeInCPUCycles()-time_before_components)/150;
250  #endif
251 
252  // If a component update has killed the object
253  // then we don't process any more components
254  // as they might attempt to fire an event, or reference this object in some way
255  // and it won't be in the tracking system any more
256  if (IsDead())
257  {
258  break;
259  }
260 
261 
262  }
263  pComponent = pComponent->mp_next;
264  }
265  # ifdef __USE_PROFILER__
266  Sys::CPUProfiler->PopContext();
267  # endif // __USE_PROFILER__
268 
269  m_composite_object_flags.Clear(CO_TELEPORTED);
270 }

301 // Suspending a composite object tells it not to update the script
302 // but the individual components take care of suspending themselves

307 {
308  if (suspended == m_composite_object_flags.Test(CO_SUSPENDED)) return;
309 
310  m_composite_object_flags.Set(CO_SUSPENDED, suspended);
311 
312  // suspend the individual components
313  CBaseComponent* pComponent = mp_component_list;
314  while ( pComponent )
315  {
316  pComponent->Suspend(suspended);
317  pComponent = pComponent->mp_next;
318  }

345 void CCompositeObject::AddComponent( CBaseComponent* pComponent )
346 {
347 #ifdef __NOPT_ASSERT__
348  if ( GetComponent( pComponent->GetType() ) )
349  {
350  Dbg_MsgAssert( 0, ( "Object already has a component of type '%s'", Script::FindChecksumName(pComponent->GetType()) ) );
351  }
352 #endif
353 
354  Dbg_MsgAssert(!IsFinalized(),("Adding Component %s to Finalized Composite object %s",Script::FindChecksumName(pComponent->GetType()),Script::FindChecksumName(GetID())));
355 
356  if (!mp_component_list)
357  {
358  mp_component_list = pComponent;
359  }
360  else
361  {
362  CBaseComponent* p_tail = mp_component_list;
363  while(p_tail->mp_next)
364  {
365  p_tail = p_tail->mp_next;
366  }
367  p_tail->mp_next = pComponent;
368  }
369 
370  // now that the component is "officially" associated with
371  // this object, we can set the component's object ptr
372  pComponent->SetObject( this );
373 
374  // add the component to the by-type list of components maintained by the CompositeObjectManager
375  Obj::CCompositeObjectManager::Instance()->AddComponentByType( pComponent );
376 }

395 void CCompositeObject::CreateComponentFromStructure(Script::CStruct *p_struct, Script::CStruct *p_params)
396 {
397  uint32 component_name;
398  p_struct->GetChecksum("component",&component_name,Script::ASSERT);
399  Obj::CBaseComponent* p_component = Obj::CCompositeObjectManager::Instance()->CreateComponent(component_name);
400  AddComponent(p_component); // Add it first, as InitFromStructure might need to query the object
401  if (p_params)
402  {
403  // If we have additional parameters, then add then in to the parameters in the array
404  // this will override the array paramaeters, which are assumed to be defaults
405  Script::CScriptStructure *p_combinedParams=new Script::CScriptStructure;
406  *p_combinedParams += *p_struct;
407  *p_combinedParams += *p_params;
408  p_component->InitFromStructure(p_combinedParams);
409  delete p_combinedParams;
410  }
411  else
412  {
413  // No additional parameters, so just initialze from the params in the structure in the array
414  p_component->InitFromStructure(p_struct);
415  }
416 }

477 // Finalize is called after all components have been added
478 // and will call the virtual Finalize() function on each component
479 // This is intended for any components that are depended on other components
480 // but where the initialization order can't be guaranteed
481 void CCompositeObject::Finalize()
482 {
483  Dbg_MsgAssert(!IsFinalized(),("Finalizing Composite object %s twice",Script::FindChecksumName(GetID())));
484  CBaseComponent *p_component = mp_component_list;
485  while (p_component)
486  {
487  p_component->Finalize();
488  p_component = p_component->GetNext();
489  }
490  m_composite_object_flags.Set(CO_FINALIZED);
491 
492  // now that the component is finalized,
493  // update the components that depend
494  // on the position of the object
495  Teleport();
496 }

553 bool CCompositeObject::CallMemberFunction( uint32 Checksum, Script::CStruct *pParams, Script::CScript *pScript )
554 {
555 
556 
557  // This should probably go in some debug component?
558  switch ( Checksum )
559  {
560  // @script | Obj_PrintDetails | This is a debug function. Add this
561  // command to an object's script
562  // and it will print the details. Add anything you want to the Dbg_Message( )
563  // below.
564  // @uparmopt "string" | text to print
565  case 0xc2404947: // Obj_PrintDetails
566  {
567  const char* pText;
568  if ( pParams->GetText( NONAME, &pText ) )
569  {
570  Dbg_Message( "%s", pText );
571  }
572  Dbg_Message( "Obj details:\n pos %f %f %f\n time %d\n type %d",
573  m_pos[ X ], m_pos[ Y ], m_pos[ Z ], Tmr::GetTime(), m_type );
574  return true;
575  }
576  break;
577 
578  // @script | CreateComponentFromStructure |
579  case 0x406998a5: // CreateComponentFromStructure
580  {
581  CreateComponentFromStructure( pParams, NULL );
582  return true;
583  }
584  break;
585 
586  // @script | Obj_GetPosition |
587  case 0xe90aad2d: // Obj_GetPosition
588  {
589  pScript->GetParams()->AddVector( CRCD(0x7f261953,"pos"), m_pos[X], m_pos[Y], m_pos[Z] );
590  return true;
591  }
592  break;
593 
594  // @script | Obj_SetPosition | Sets the world position of the object.
595  // @parmopt vector | Position | | Position to give to the object.
596  case 0xf7251a64: // Obj_SetPosition
597  {
598  pParams->GetVector(CRCD(0xb9d31b0a,"Position"),&m_pos);
599  return true;
600  }
601  break;
602 
603  // @script | Obj_SetOrientation |
604  // @parmopt float | x | 0.0 | The x angle, in degrees
605  // @parmopt float | y | 0.0 | The y angle, in degrees
606  // @parmopt float | z | 0.0 | The z angle, in degrees
607  // @parmopt vector | dir | (0,0,1) | Direction vector (an alternative to specifying angles)
608  // The direction vector does not need to be normalized.
609  case 0xc6f3baa5: // Obj_SetOrientation
610  {
611  Mth::Vector dir(0.0f,0.0f,1.0f);
612  if (pParams->GetVector(CRCD(0x455485ef,"dir"),&dir))
613  {
614  Mth::Matrix mat;
615  mat.Ident();
616 
617  mat[Z]=dir;
618  mat[Z].Normalize();
619  mat[X] = Mth::CrossProduct(mat[Y], mat[Z]);
620  mat[X].Normalize();
621  mat[Y] = Mth::CrossProduct(mat[Z], mat[X]);
622  mat[Y].Normalize();
623 
624  SetMatrix(mat);
625  SetDisplayMatrix(mat);
626  }
627  else
628  {
629  float p=0.0f;
630  float h=0.0f;
631  float r=0.0f;
632  pParams->GetFloat(CRCD(0x7323e97c,"x"),&p);
633  pParams->GetFloat(CRCD(0x424d9ea,"y"),&h);
634  pParams->GetFloat(CRCD(0x9d2d8850,"z"),&r);
635  p*=3.141592654f/180.0f;
636  h*=3.141592654f/180.0f;
637  r*=3.141592654f/180.0f;
638  Mth::Matrix mat(p,h,r);
639  SetMatrix(mat);
640  SetDisplayMatrix(mat);
641  }
642  return true;
643  }
644  break;
645 
646  // @script | Obj_ForceUpdate | Does a single call to the object's Update function
647  case 0xc1bff0f3: // Obj_ForceUpdate
648  {
649  Update();
650  return true;
651  }
652  break;
653 
654  // @script | Obj_GetVelocity |
655  case 0x11fe9f71: // Obj_GetVelocity
656  {
657  pScript->GetParams()->AddVector( CRCD(0x0c4c809e,"vel"), m_vel[X], m_vel[Y], m_vel[Z] );
658  return true;
659  }
660  break;
661 
662  // @script | GetSpeed |
663  case 0xc0caac4a: // GetSpeed
664  {
665  pScript->GetParams()->AddFloat( CRCD(0xf0d90109,"speed"), m_vel.Length() );
666  return true;
667  }
668  break;
669 
670  // @script | Hide | hides object
671  case 0x5b6634d4: // Hide
672  {
673  Hide( true );
674  return true;
675  }
676  break;
677 
678  // @script | Unhide | unhides object
679  case 0xb60d1f35: // Unhide
680  {
681  Hide( false );
682  return true;
683  }
684  break;
685 
686  // @script | IsHidden | checks the hide flag of object
687  case 0xb16619ae: // IsHidden
688  {
689  return m_composite_object_flags.Test(CO_HIDDEN);
690  }
691  break;
692 
693  // @script | Move |
694  // @parmopt float | x | 0.0 | x component
695  // @parmopt float | y | 0.0 | y component
696  // @parmopt float | z | 0.0 | z component
697  case 0x10c1c887: // Move
698  {
699  float distance = 0.0f;
700  pParams->GetFloat(CRCD(0x7323e97c, "x"), &distance);
701  m_pos += distance * m_matrix[X];
702  distance = 0.0f;
703  pParams->GetFloat(CRCD(0x424d9ea, "y"), &distance);
704  m_pos += distance * m_matrix[Y];
705  distance = 0.0f;
706  pParams->GetFloat(CRCD(0x9d2d8850, "z"), &distance);
707  m_pos += distance * m_matrix[Z];
708  return true;
709  }
710 
711  case 0xce0ca665: // Suspend
712  Suspend(true);
713  return true;
714  break;
715 
716  case 0xca3c59a6: // Unsuspend
717  Suspend(false);
718  return true;
719  break;
720 
721  case 0x28656d12: // Pause
722  Pause(true);
723  return true;
724  break;
725 
726  case 0xff85d4d7: // Unpause
727  Pause(false);
728  return true;
729  break;
730 
731 // MOVED FROM CMOVINGOBJECT.CPP
732
