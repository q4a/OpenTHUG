36 bool CXboxSector::LoadFromMemory( void **pp_mem )
37 {
38  Dbg_Assert( mp_geom );
39 
40  uint8 *p_data = (uint8*)( *pp_mem );
41  CXboxGeom *p_geom = static_cast<CXboxGeom*>( mp_geom );
42 
43  // Read sector checksum.
44  uint32 sector_checksum;
45  MemoryRead( &sector_checksum, sizeof( uint32 ), 1, p_data );
46  SetChecksum( sector_checksum );
47 
48  // Read bone index.
49  int bone_idx;
50  MemoryRead( &bone_idx, sizeof( int ), 1, p_data );
51 
52  // Read sector flags.
53  MemoryRead( &m_flags, sizeof( int ), 1, p_data );
54 
55  // Read number of meshes.
56  MemoryRead( &p_geom->m_num_mesh, sizeof( uint ), 1, p_data );
57 
58  // Read bounding box.
59  float bbox[6];
60  MemoryRead( &bbox[0], sizeof( float ), 6, p_data );
61  Mth::Vector inf( bbox[0], bbox[1], bbox[2] );
62  Mth::Vector sup( bbox[3], bbox[4], bbox[5] );
63  p_geom->m_bbox.Set( inf, sup );
64 
65  // Read bounding sphere.
66  float bsphere[4];
67  MemoryRead( &bsphere[0], sizeof( float ), 4, p_data );
68 
69  // Read billboard data if present.
70  uint32 billboard_type;
71  Mth::Vector billboard_origin;
72  Mth::Vector billboard_pivot_pos;
73  Mth::Vector billboard_pivot_axis;
74  if( m_flags & 0x00800000UL )
75  {
76  MemoryRead( &billboard_type, sizeof( uint32 ), 1, p_data );
77  MemoryRead( &billboard_origin[X], sizeof( float ) * 3, 1, p_data );
78  MemoryRead( &billboard_pivot_pos[X], sizeof( float ) * 3, 1, p_data );
79  MemoryRead( &billboard_pivot_axis[X], sizeof( float ) * 3, 1, p_data );
80 
81  billboard_origin[Z] = -billboard_origin[Z];
82  billboard_pivot_pos[Z] = -billboard_pivot_pos[Z];
83  }
84 
85  // Read num vertices.
86  int num_vertices;
87  MemoryRead( &num_vertices, sizeof( int ), 1, p_data );
88 
89  // Read vertex data stride.
90  int vertex_data_stride;
91  MemoryRead( &vertex_data_stride, sizeof( int ), 1, p_data );
92 
93  // We want all the temporary buffer allocations to come off of the top down heap.
94  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().TopDownHeap());
95 
96  // Grab a buffer for the raw vertex data position stream, and read it.
97  float* p_vertex_positions = new float[num_vertices * 3];
98  MemoryRead( p_vertex_positions, sizeof( float ) * 3, num_vertices, p_data );
99 
100  // Grab a buffer for the raw vertex data normal stream (if present), and read it.
101  float* p_vertex_normals = ( m_flags & 0x04 ) ? new float[num_vertices * 3] : NULL;
102  if( p_vertex_normals )
103  {
104  MemoryRead( p_vertex_normals, sizeof( float ) * 3, num_vertices, p_data );
105  }
106 
107  // Grab a buffer for the raw vertex data weights stream (if present), and read it.
108  uint32* p_vertex_weights = ( m_flags & 0x10 ) ? new uint32[num_vertices] : NULL;
109  if( p_vertex_weights )
110  {
111  MemoryRead( p_vertex_weights, sizeof( uint32 ), num_vertices, p_data );
112  }
113 
114  // Grab a buffer for the raw vertex data bone indices stream (if present), and read it.
115  uint16* p_vertex_bone_indices = ( m_flags & 0x10 ) ? new uint16[num_vertices * 4] : NULL;
116  if( p_vertex_bone_indices )
117  {
118  MemoryRead( p_vertex_bone_indices, sizeof( uint16 ) * 4, num_vertices, p_data );
119  }
120 
121  // Grab a buffer for the raw vertex texture coordinate stream (if present), and read it.
122  int num_tc_sets = 0;
123  float* p_vertex_tex_coords = NULL;
124  if( m_flags & 0x01 )
125  {
126  MemoryRead( &num_tc_sets, sizeof( int ), 1, p_data );
127 
128  if( num_tc_sets > 0 )
129  {
130  p_vertex_tex_coords = new float[num_vertices * 2 * num_tc_sets];
131  MemoryRead( p_vertex_tex_coords, sizeof( float ) * 2 * num_tc_sets, num_vertices, p_data );
132  }
133  }
134 
135  // Grab a buffer for the raw vertex colors stream (if present), and read it.
136  DWORD* p_vertex_colors = ( m_flags & 0x02 ) ? new DWORD[num_vertices] : NULL;
137  if( p_vertex_colors )
138  {
139  MemoryRead( p_vertex_colors, sizeof( DWORD ), num_vertices, p_data );
140  }
141 
142  // Grab a buffer for the vertex color wibble stream (if present), and read it.
143  char* p_vc_wibble_indices = ( m_flags & 0x800 ) ? new char[num_vertices] : NULL;
144  if( p_vc_wibble_indices )
145  {
146  MemoryRead( p_vc_wibble_indices, sizeof( char ), num_vertices, p_data );
147  }
148 
149  // Remove TopDownHeap context.
150  Mem::Manager::sHandle().PopContext();
151 
152  // Preprocess verts that require cutscene scaling.
153  NxXbox::ApplyMeshScaling( p_vertex_positions, num_vertices );
154 
155  for( uint m = 0; m < p_geom->m_num_mesh; ++m )
156  {
157  unsigned long material_checksum;
158  unsigned int flags, num_lod_index_levels;
159 
160  NxXbox::sMesh* p_mesh = new NxXbox::sMesh;
161 
162  // Read bounding sphere and box data.
163  float rad;
164  Mth::Vector inf, sup, cen;
165  MemoryRead( &cen[X], sizeof( float ), 3, p_data );
166  MemoryRead( &rad, sizeof( float ), 1, p_data );
167  MemoryRead( &inf[X], sizeof( float ), 3, p_data );
168  MemoryRead( &sup[X], sizeof( float ), 3, p_data );
169 
170  // Read and deal with flags, including skater shadow flag.
171  MemoryRead( &flags, sizeof( uint32 ), 1, p_data );
172  if( flags & 0x400 )
173  {
174  p_mesh->m_flags |= NxXbox::sMesh::MESH_FLAG_NO_SKATER_SHADOW;
175  }
176  if( flags & NxXbox::sMesh::MESH_FLAG_UNLIT )
177  {
178  p_mesh->m_flags |= NxXbox::sMesh::MESH_FLAG_UNLIT;
179  }
180 
181  // The material checksum for this mesh.
182  MemoryRead( &material_checksum, sizeof( unsigned long ), 1, p_data );
183 
184  // How many levels of LOD indices? Should be at least 1!
185  MemoryRead( &num_lod_index_levels, sizeof( unsigned int ), 1, p_data );
186 
187  // Can have up to 8 levels of LOD indices.
188  uint16* p_indices[8] = { NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL };
189  int num_indices[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
190 
191  for( unsigned int lod_level = 0; lod_level < num_lod_index_levels; ++lod_level )
192  {
193  MemoryRead( &num_indices[lod_level], sizeof( int ), 1, p_data );
194 
195  // Again, we want all the temporary buffer allocations to come off of the top down heap.
196  Mem::Manager::sHandle().PushContext( Mem::Manager::sHandle().TopDownHeap());
197  p_indices[lod_level] = new uint16[num_indices[lod_level]];
198  Mem::Manager::sHandle().PopContext();
199 
200  MemoryRead( p_indices[lod_level], sizeof( uint16 ), num_indices[lod_level], p_data );
201  }
202 
203  // Set the load order of this mesh.
204  p_mesh->m_load_order = m;
205 
206  // Set up the mesh.
207  p_mesh->Initialize( num_vertices,
208  p_vertex_positions,
209  p_vertex_normals,
210  p_vertex_tex_coords,
211  num_tc_sets,
212  p_vertex_colors,
213  num_lod_index_levels,
214  num_indices,
215  p_indices,
216  material_checksum,
217  p_geom->mp_scene->GetEngineScene(),
218  p_vertex_bone_indices,
219  p_vertex_weights,
220  ( flags & 0x800 ) ? p_vc_wibble_indices : NULL );
221 
222  // Set the bounding data (sphere and box) for the mesh.
223  p_mesh->SetBoundingData( cen, rad, inf, sup );
224 
225  // Add the bounding data to the scene.
226  p_geom->mp_scene->GetEngineScene()->m_bbox.AddPoint( inf );
227  p_geom->mp_scene->GetEngineScene()->m_bbox.AddPoint( sup );
228 
229  // Set up as a billboard if required.
230  if( m_flags & 0x00800000UL )
231  {
232  p_mesh->SetBillboardData( billboard_type, billboard_pivot_pos, billboard_pivot_axis );
233  NxXbox::BillboardManager.AddEntry( p_mesh );
234  }
235 
236  // Flag the mesh as being a shadow volume if applicable.
237  if( m_flags & 0x200000 )
238  {
239  p_mesh->m_flags |= NxXbox::sMesh::MESH_FLAG_SHADOW_VOLUME;
240  }
241 
242  // Add the mesh to the attached CXboxGeom.
243  p_geom->AddMesh( p_mesh );
244 
245  // Set the mesh bone index (mostly not applicable).
246  p_mesh->SetBoneIndex( bone_idx );
247 
248  // Test code - if the mesh is mapped with a grass texture, add grass meshes.
249  AddGrass( p_geom, p_mesh );
250 
251  // Done with the raw index data.
252  for( int lod_level = 0; lod_level < 8; ++lod_level )
253  {
254  delete[] p_indices[lod_level];
255  }
256  }
257 
258  // Recount the number of meshes in case any have been added.
259  p_geom->m_num_mesh = p_geom->mp_init_mesh_list->CountItems();
260 
261  // Done with the raw vertex data.
262  delete[] p_vc_wibble_indices;
263  delete[] p_vertex_colors;
264  delete[] p_vertex_tex_coords;
265  delete[] p_vertex_bone_indices;
266  delete[] p_vertex_weights;
267  delete[] p_vertex_normals;
268  delete[] p_vertex_positions;
269 
270  // Set the data pointer to the new position on return.
271  *pp_mem = p_data;
272 
273  return true;
274 }
