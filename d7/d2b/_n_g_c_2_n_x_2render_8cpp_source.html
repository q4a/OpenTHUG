603 /* */
604 /* Checks a bounding box against the current view frustum */
605 /* (iBgnoring far clipping), */
606 /* returns true if any part is visible. */
607 /* Horribly inefficient at present. */
608 /* */
609 /******************************************************************/
610 bool frustum_check_box( Mtx m, Mth::CBBox& box )
611 {
612  Vec test_in, test_out;
613 
614  uint32 cumulative_projection_space_outcode = 0xFF;
615  float min_x = box.GetMin().GetX();
616  float min_y = box.GetMin().GetY();
617  float min_z = box.GetMin().GetZ();
618  float max_x = box.GetMax().GetX();
619  float max_y = box.GetMax().GetY();
620  float max_z = box.GetMax().GetZ();
621 
622  for( uint32 v = 0; v < 8; ++v )
623  {
624  uint32 projection_space_outcode = 0;
625 
626  test_in.x = ( v & 0x04 ) ? max_x : min_x;
627  test_in.y = ( v & 0x02 ) ? max_y : min_y;
628  test_in.z = ( v & 0x01 ) ? max_z : min_z;
629 
630  // At this point it's important to check to see whether the point is in postive or negative z-space, since
631  // after the projection transform, both very large camera space z values and camera space z values where z < 0
632  // will give results with z > 1. (Camera space values in the range [0,near] give negative projection space z values).
633  MTXMultVec( m, &test_in, &test_out );
634 
635 // if(( -test_mid.z < 0.0f ) && ( !EngineGlobals.is_orthographic ))
636 // {
637 // test_out.x = -test_out.x;
638 // test_out.y = -test_out.y;
639 // }
640 
641  if( test_out.x > 1.0f )
642  {
643  projection_space_outcode |= 0x01;
644  }
645  else if( test_out.x < -1.0f )
646  {
647  projection_space_outcode |= 0x02;
648  }
649 
650  if( test_out.y > 1.0f )
651  {
652  projection_space_outcode |= 0x04;
653  }
654  else if( test_out.y < -1.0f )
655  {
656  projection_space_outcode |= 0x08;
657  }
658 
659  cumulative_projection_space_outcode &= projection_space_outcode;
660 
661  if( cumulative_projection_space_outcode == 0 )
662  {
663  // Early out.
664  return true;
665  }
666  }
667  return false;
668 }
