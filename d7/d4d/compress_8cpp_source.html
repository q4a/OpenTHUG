64 void InsertNode(int r)
65 /* Inserts string of length F, text_buf[r..r+F-1], into one of the
66  trees (text_buf[r]'th tree) and returns the longest-match position
67  and length via the global variables match_position and match_length.
68  If match_length = F, then removes the old node in favor of the new
69  one, because the old one will be deleted sooner.
70  Note r plays double role, as tree node and position in buffer. */
71 {
72  int i, p, cmp;
73  unsigned char *key;
74 
75  cmp = 1; key = &text_buf[r]; p = N + 1 + key[0];
76  rson[r] = lson[r] = NIL; match_length = 0;
77  for ( ; ; )
78  {
79  if ( cmp >= 0 )
80  {
81  if ( rson[p] != NIL )
82  {
83  p = rson[p];
84  }
85  else
86  {
87  rson[p] = r; dad[r] = p; return;
88  }
89  }
90  else
91  {
92  if ( lson[p] != NIL )
93  {
94  p = lson[p];
95  }
96  else
97  {
98  lson[p] = r; dad[r] = p; return;
99  }
100  }
101 
102  for ( i = 1; i < F; i++ )
103  {
104  if ( (cmp = key[i] - text_buf[p + i]) != 0 ) break;
105  }
106 
107  if ( i > match_length )
108  {
109  match_position = p;
110  if ( (match_length = i) >= F )
111  {
112  break;
113  }
114  }
115  }
116  dad[r] = dad[p]; lson[r] = lson[p]; rson[r] = rson[p];
117  dad[lson[p]] = r; dad[rson[p]] = r;
118  if ( rson[dad[p]] == p )

130 {

132

134  if ( rson[p] == NIL ) q = lson[p];

136  else

138  q = lson[p];
139  if ( rson[q] != NIL )
140  {
141  do

143  q = rson[q];
144  } while ( rson[q] != NIL );
145  rson[dad[q]] = lson[q]; dad[lson[q]] = dad[q];
146  lson[q] = lson[p]; dad[lson[p]] = q;
147  }
148  rson[q] = rson[p]; dad[rson[p]] = q;
149  }

156 int Encode(char *pIn, char *pOut, int bytes_to_read, bool print_progress)
157 {
158  int i, c, len, r, s, last_match_length, code_buf_ptr;
159  unsigned char code_buf[17], mask;
160 
161  textsize = 0; /* text size counter */
162  codesize = 0; /* code size counter */
163  printcount = 0; /* counter for reporting progress every 1K bytes */
164 
165  InitTree(); /* initialize trees */
166  code_buf[0] = 0; /* code_buf[1..16] saves eight units of code, and
167  code_buf[0] works as eight flags, "1" representing that the unit
168  is an unencoded letter (1 byte), "0" a position-and-length pair
169  (2 bytes). Thus, eight units require at most 16 bytes of code. */
170  code_buf_ptr = mask = 1;
171  s = 0; r = N - F;
172  for ( i = s; i < r; i++ ) text_buf[i] = ' '; /* Clear the buffer with
173  any character that will appear often. */
174  for ( len = 0; len < F && bytes_to_read; len++ )
175  {
176  c = readc();
177  bytes_to_read--;
178  text_buf[r + len] = c; /* Read F bytes into the last F bytes of
179  the buffer */
180  }
181  if ( (textsize = len) == 0 )
182  {
183  DeInitTree();
184  return 0; /* text of size zero */
185  }
186  for ( i = 1; i <= F; i++ ) InsertNode(r - i); /* Insert the F strings,
187  each of which begins with one or more 'space' characters. Note
188  the order in which these strings are inserted. This way,
189  degenerate trees will be less likely to occur. */
190  InsertNode(r); /* Finally, insert the whole string just read. The
191  global variables match_length and match_position are set. */
192  do
193  {
194  if ( match_length > len ) match_length = len; /* match_length
195  may be spuriously long near the end of text. */
196  if ( match_length <= THRESHOLD )
197  {
198  match_length = 1; /* Not long enough match. Send one byte. */
199  code_buf[0] |= mask; /* 'send one byte' flag */
200  code_buf[code_buf_ptr++] = text_buf[r]; /* Send uncoded. */
201  }
202  else
203  {
204  code_buf[code_buf_ptr++] = (unsigned char) match_position;
205  code_buf[code_buf_ptr++] = (unsigned char)
206  (((match_position >> 4) & 0xf0)
207  | (match_length - (THRESHOLD + 1))); /* Send position and
208  length pair. Note match_length > THRESHOLD. */
209  }
210  if ( (mask <<= 1) == 0 )
211  { /* Shift mask left one bit. */
212  for ( i = 0; i < code_buf_ptr; i++ ) /* Send at most 8 units of */
213  writec(code_buf[i]); /* code together */
214  codesize += code_buf_ptr;
215  code_buf[0] = 0; code_buf_ptr = mask = 1;
216  }
217  last_match_length = match_length;
218  for ( i = 0; i < last_match_length &&
219  bytes_to_read; i++ )
220  {
221  c = readc();
222  bytes_to_read--;
223  DeleteNode(s); /* Delete old strings and */
224  text_buf[s] = c; /* read new bytes */
225  if ( s < F - 1 ) text_buf[s + N] = c; /* If the position is
226  near the end of buffer, extend the buffer to make
227  string comparison easier. */
228  s = (s + 1) & (N - 1); r = (r + 1) & (N - 1);
229  /* Since this is a ring buffer, increment the position
230  modulo N. */
231  InsertNode(r); /* Register the string in text_buf[r..r+F-1] */
232  }
233  if ( (textsize += i) > printcount )
234  {
235 // printf("%12ld\r", textsize); printcount += 1024;
236  /* Reports progress each time the textsize exceeds
237  multiples of 1024. */
238  }
239  while ( i++ < last_match_length )
240  { /* After the end of text, */
241  DeleteNode(s); /* no need to read, but */
242  s = (s + 1) & (N - 1); r = (r + 1) & (N - 1);
243  if ( --len ) InsertNode(r); /* buffer may not be empty. */
244  }
245  } while ( len > 0 ); /* until length of string to be processed is zero */
246  if ( code_buf_ptr > 1 )
247  { /* Send remaining code. */
248  for ( i = 0; i < code_buf_ptr; i++ ) writec(code_buf[i]);
249  codesize += code_buf_ptr;
250  }
251 
252  if ( print_progress )
253  {
254  printf(" In : %ld bytes\n", textsize); /* Encoding is done. */
255  printf(" Out: %ld bytes\n", codesize);
256  printf(" Out/In: %.3f\n", (double)codesize / textsize);
257  }
258 
259  DeInitTree();
260  return codesize;
261 }
