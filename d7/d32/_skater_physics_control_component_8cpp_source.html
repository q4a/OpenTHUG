110 void CSkaterPhysicsControlComponent::Update()
111 {
112  m_restarted_this_frame = false;
113 
114  // SkaterPhysicsControlComponent is inactive when physics is NOT suspended. When physics is suspended SkaterPhysicsControlComponent handles
115  // setting the skater's display matrix.
116  if (!m_physics_suspended) return;
117 
118  switch (mp_state_component->m_physics_state)
119  {
120  case SKATING:
121  GetObject()->SetDisplayMatrix(mp_core_physics_component->m_lerping_display_matrix);
122  break;
123 
124  case WALKING:
125  GetObject()->SetDisplayMatrix(GetObject()->GetMatrix());
126  break;
127 
128  default:
129  Dbg_MsgAssert(false, ("No skater physics state switched on before first skater update"));
130  }
131 
132  // If the skater is unsuspended, the physics components will unsuspend as well. We must check for this and resuspend
133  if (!mp_core_physics_component->IsSuspended())
134  {
135  suspend_physics(true);
136  }
137 }

144 CBaseComponent::EMemberFunctionResult CSkaterPhysicsControlComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
145 {
146  switch ( Checksum )
147  {
148  // @script | PausePhysics | PausePhysics will stop all programmatic movement and rotation of the skater
149  case CRCC(0x3522d612, "PausePhysics"):
150  if (!m_physics_suspended)
151  {
152  suspend_physics(true);
153  }
154  break;
155 
156  // @script | UnPausePhysics |
157  case CRCC(0x595627c, "UnPausePhysics"):
158  if (m_physics_suspended)
159  {
160  suspend_physics(false);
161  }
162  break;
163 
164  case CRCC(0x57bfbae8, "Walking"):
165  return IsWalking() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
166 
167  case CRCC(0xf2813ee5, "Skating"):
168  return IsSkating() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
169 
170  case CRCC(0xee584cbc, "Driving"):
171  return IsDriving() ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
172 
173  case CRCC(0xb32e9eae, "SetDriving"):
174  mp_state_component->m_driving = true;
175  break;
176 
177  case CRCC(0x953a7b6d, "UnsetDriving"):
178  mp_state_component->m_driving = false;
179  break;
180 
181  case CRCC(0x9569d5b1, "GetTimeSincePhysicsSwitch"):
182  pScript->GetParams()->AddFloat(CRCD(0x27f48e93, "TimeSincePhysicsSwitch"), Tmr::ElapsedTime(m_physics_state_switch_time_stamp) * (1.0f / 1000.0f));
183  break;
184 
185  case CRCC(0xa887285a, "GetPreviousPhysicsStateDuration"):
186  pScript->GetParams()->AddFloat(CRCD(0x24d061ba, "PreviousPhysicsStateDuration"), m_previous_physics_state_duration * (1.0f / 1000.0f));
187  break;
188 
189  case CRCC(0x5c038f9b, "SkaterPhysicsControl_SwitchWalkingToSkating"):
190  switch_walking_to_skating();
191  break;
192 
193  case CRCC(0x6e8e39b3, "SkaterPhysicsControl_SwitchSkatingToWalking"):
194  switch_skating_to_walking();
195  break;
196 
197  case CRCC(0x9366a509, "SetBoardMissing"):
198  m_board_missing = true;
199  break;
200 
201  case CRCC(0x4830e80, "UnsetBoardMissing"):
202  m_board_missing = false;
203  break;
204 
205  case CRCC(0xd5a9f889, "IsBoardMissing"):
206  return m_board_missing ? CBaseComponent::MF_TRUE : CBaseComponent::MF_FALSE;
207 
208 # ifdef TESTING_GUNSLINGER
209  case CRCC(0x14c4f16b,"SkaterPhysicsControl_SwitchWalkingToRiding"):
210  switch_walking_to_riding();
211  break;
212 
213  case CRCC(0x82604c1e,"SkaterPhysicsControl_SwitchRidingToWalking"):
214  switch_riding_to_walking();
215  break;
216 # endif
217 
218  default:
219  return CBaseComponent::MF_NOT_EXECUTED;
220  }
221  return CBaseComponent::MF_TRUE;
222 }

246 void CSkaterPhysicsControlComponent::suspend_physics ( bool suspend )
247 {
248  m_physics_suspended = suspend;
249 
250  // If the skater is suspended, we don't want to unsuspend the physics components. We only want to change the CSkaterPhysicsControlComponent state.
251  if (!suspend && GetObject()->IsSuspended()) return;
252 
253  switch (mp_state_component->m_physics_state)
254  {
255  case SKATING:
256  Dbg_Assert(mp_core_physics_component);
257  Dbg_Assert(GetSkaterRotateComponentFromObject(GetObject()));
258  Dbg_Assert(GetSkaterAdjustPhysicsComponentFromObject(GetObject()));
259  Dbg_Assert(GetSkaterFinalizePhysicsComponentFromObject(GetObject()));
260 
261  mp_core_physics_component->Suspend(suspend);
262  GetSkaterRotateComponentFromObject(GetObject())->Suspend(suspend);
263  GetSkaterAdjustPhysicsComponentFromObject(GetObject())->Suspend(suspend);
264  GetSkaterFinalizePhysicsComponentFromObject(GetObject())->Suspend(suspend);
265  break;
266 
267  case WALKING:
268  Dbg_Assert(GetWalkComponentFromObject(GetObject()));
269 
270  GetWalkComponentFromObject(GetObject())->Suspend(suspend);
271  break;
272 
273  default:
274  Dbg_Assert(false);
275  }
276 }

283 void CSkaterPhysicsControlComponent::switch_walking_to_skating ( )
284 {
285  if (mp_state_component->m_physics_state == SKATING) return;
286 
287  m_previous_physics_state_duration = Tmr::ElapsedTime(m_physics_state_switch_time_stamp);
288  m_physics_state_switch_time_stamp = Tmr::GetTime();
289 
290  CWalkComponent* p_walk_component = GetWalkComponentFromObject(GetObject());
291  CCompositeObject* p_skater_cam = get_skater_camera();
292  CSkaterCameraComponent* p_skater_camera_component = GetSkaterCameraComponentFromObject(p_skater_cam);
293  CWalkCameraComponent* p_walk_camera_component = GetWalkCameraComponentFromObject(p_skater_cam);
294 
295  Dbg_Assert(p_walk_component);
296  Dbg_Assert(p_skater_camera_component);
297  Dbg_Assert(p_walk_camera_component);
298 
299  // switch off walking
300 
301  p_walk_component->CleanUpWalkState();
302  p_walk_component->Suspend(true);
303 
304  p_walk_camera_component->Suspend(true);
305 
306  // switch on skating
307 
308  mp_state_component->m_physics_state = SKATING;
309 
310  mp_core_physics_component->Suspend(false);
311  mp_core_physics_component->ReadySkateState(p_walk_component->GetState() == CWalkComponent::WALKING_GROUND, p_walk_component->GetRailData(), p_walk_component->GetAcidDropData());
312 
313  GetSkaterRotateComponentFromObject(GetObject())->Suspend(false);
314  GetSkaterAdjustPhysicsComponentFromObject(GetObject())->Suspend(false);
315  GetSkaterFinalizePhysicsComponentFromObject(GetObject())->Suspend(false);
316 
317  p_skater_camera_component->Suspend(false);
318 
319  // exchange camera states
320  SCameraState camera_state;
321  p_walk_camera_component->GetCameraState(camera_state);
322  p_skater_camera_component->ReadyForActivation(camera_state);
323 
324  // reapply the physics suspend state
325  if (m_physics_suspended)
326  {
327  suspend_physics(m_physics_suspended);
328  }
329 }
