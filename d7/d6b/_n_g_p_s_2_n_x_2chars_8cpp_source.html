142 SFont * LoadFont(const char *Filename)
143 {
144  SFont *pFont;
145  SChar *pChar;
146  uint8 *pData;
147  void *pFile;
148  int i,j,Len,NumChars,Width,Height,NumQWords,NumBytes;
149  uint32 TW,TH,TBW;
150  uint32 temp;
151 
152 
153  uint8* old_dma_loc = dma::pLoc;
154 
155  // open the font file
156  pFile = OpenExtended(Filename);
157 
158 
159  // allocate memory for the font structure
160  pFont = (SFont *)Mem::Malloc(sizeof(SFont));
161  Dbg_MsgAssert(pFont, ("couldn't malloc memory for SFont"));
162 
163 
164  // allocate a temporary buffer
165  uint8 FontBuf[2048];
166  Dbg_MsgAssert(FontBuf, ("couldn't malloc memory for FontBuf"));
167 
168 
169  // load file header
170  Len = File::Read(FontBuf, 1, 16, pFile);
171  Dbg_MsgAssert(Len==16, ("couldn't read file header from font file %s", Filename));
172  NumChars = ((uint32 *)FontBuf)[1];
173  pFont->DefaultHeight = ((uint32 *)FontBuf)[2] << 4; // *16 for texel coords
174  pFont->DefaultBase = ((uint32 *)FontBuf)[3] << 4; // *16 for texel coords
175 
176 
177  // clear character map to zero
178  memset(pFont->Map, 0, 256);
179  memset(pFont->SpecialMap, 0, 32);
180 
181 
182  // allocate memory for character table
183  pFont->pChars = (SChar *)Mem::Malloc(NumChars*sizeof(SChar));
184  Dbg_MsgAssert(pFont->pChars, ("couldn't malloc memory for font character table"));
185 
186 
187  // load character map and character table
188  Len = File::Read(FontBuf, 1, NumChars<<2, pFile);
189  Dbg_MsgAssert(Len==(NumChars<<2), ("couldn't read character table in font file %s", Filename));
190 
191  for (i=0,pChar=pFont->pChars,pData=FontBuf; i<NumChars; i++,pChar++,pData+=4)
192  {
193  pChar->Baseline = ((uint16 *)pData)[0] << 4; // *16 for texel coords
194  sint16 ascii_val = ((sint16 *)pData)[1];
195  if (ascii_val >= 0)
196  pFont->Map[(uint8) ascii_val] = i;
197  else
198  {
199  Dbg_Assert(ascii_val >= -32)
200  pFont->SpecialMap[(uint8) (-ascii_val - 1)] = i;
201  }
202  }
203 
204  // now, if there is a null character in the font, make characters that could not be found
205  // in the font display that instead of 'A'
206  if (pFont->SpecialMap[31] != 0)
207  {
208  for (i = 0; i < 256; i++)
209  {
210  if (pFont->Map[i] == 0 && i != 'A' && i != 'a') pFont->Map[i] = pFont->SpecialMap[31];
211  if (i < 31 && pFont->SpecialMap[i] == 0) pFont->SpecialMap[i] = pFont->SpecialMap[31];
212  }
213  }
214 
215  // load texture header
216  Len = File::Read(FontBuf, 1, 16, pFile);
217  Dbg_MsgAssert(Len==16, ("couldn't read texture header from font file %s", Filename));
218  Width = ((uint16 *)FontBuf)[2];
219  Height = ((uint16 *)FontBuf)[3];
220 
221 
222  // allocate memory for vif stream
223  pFont->VifSize = 1216 + (((Width*Height+15)>>4)<<4);
224  pFont->pVifData = (uint8 *)Mem::Malloc(pFont->VifSize);
225  Dbg_MsgAssert(pFont->pVifData, ("couldn't malloc memory for vif data"));
226  Dbg_MsgAssert(((uint32)pFont->pVifData & 15) == 0, ("vif data not qword aligned"));
227 
228 
229 
230  // build dma data in this buffer
231  dma::pLoc = pFont->pVifData;
232 
233  // Calculate memory needed
234  int min_width = Width, min_height = Height;
235  TexMemLayout.getMinSize(min_width, min_height, PSMT8);
236  pFont->m_NumTexVramBlocks = TexMemLayout.getImageSize(min_width, min_height, PSMT8);
237  pFont->m_NumClutVramBlocks = 4;
238 
239 
240  // set base pointers and texture buffer width
241  pFont->m_CBP = FontVramBase;
242  pFont->m_TBP = FontVramBase + pFont->m_NumClutVramBlocks;
243  TBW = (Width+63) >> 6;
244  if (TBW<2)
245  TBW=2;
246 
247 
248  // texture upload format:
249  //
250  // GIFtag PACKED 4
251  // BITBLTBUF ?
252  // TRXPOS 0
253  // TRXREG w,h
254  // TRXDIR 0
255  //
256  // GIFtag IMAGE (w*h+15)/16
257  // ... (tex data)
258 
259  // texture upload header
260  NumQWords = (Width*Height+15)>>4;
261  gif::Tag2(gs::A_D,1,PACKED,0,0,0,4);
262  pFont->mp_TexBitBltBuf = gs::Reg2_pData(gs::BITBLTBUF,PackBITBLTBUF(0,0,0,pFont->m_TBP,TBW,PSMT8));
263  gs::Reg2(gs::TRXPOS, PackTRXPOS(0,0,0,0,0));
264  gs::Reg2(gs::TRXREG, PackTRXREG(Width,Height));
265  gs::Reg2(gs::TRXDIR, 0);
266  gif::Tag2(0, 0, IMAGE, 0, 0, 0, NumQWords);
267 
268  // read texture bitmap data
269  NumBytes = (Width*Height+3)&0xFFFFFFFC;
270  Len = File::Read(dma::pLoc, 1, NumBytes, pFile);
271  Dbg_MsgAssert(Len==NumBytes, ("couldn't read texture bitmap from font file %s", Filename));
272  dma::pLoc += NumQWords<<4;
273 
274 
275  // clut upload format:
276  //
277  // GIFtag PACKED 4
278  // BITBLTBUF ?
279  // TRXPOS 0
280  // TRXREG w,h
281  // TRXDIR 0
282  //
283  // GIFtag IMAGE 64
284  // ... (clut data)
285 
286  // clut upload header
287  gif::Tag2(gs::A_D,1,PACKED,0,0,0,4);
288  pFont->mp_ClutBitBltBuf = gs::Reg2_pData(gs::BITBLTBUF,PackBITBLTBUF(0,0,0,pFont->m_CBP,1,PSMCT32));
289  gs::Reg2(gs::TRXPOS, PackTRXPOS(0,0,0,0,0));
290  gs::Reg2(gs::TRXREG, PackTRXREG(16,16));
291  gs::Reg2(gs::TRXDIR, 0);
292  gif::Tag2(0,0,IMAGE,0,0,0,64);
293 
294  // read clut bitmap data
295  Len = File::Read(dma::pLoc, 1, 1024, pFile);
296  Dbg_MsgAssert(Len==1024, ("couldn't read clut bitmap from font file %s", Filename));
297 
298  // swizzle the clut data
299  for (i=0; i<256; i+=32)
300  for (j=0; j<8; j++)
301  {
302  temp = ((uint32 *)dma::pLoc)[i+j+8];
303  ((uint32 *)dma::pLoc)[i+j+8] = ((uint32 *)dma::pLoc)[i+j+16];
304  ((uint32 *)dma::pLoc)[i+j+16] = temp;
305  }
306 
307  // step dma location
308  dma::pLoc += 1024;
309 
310 
311  // skip numsubtextures, and load subtextures
312  Len = File::Read(FontBuf, 1, (NumChars<<3)+4, pFile);
313  Dbg_MsgAssert(Len==(NumChars<<3)+4, ("couldn't read subtexture table from font file %s", Filename));
314  for (i=0,pChar=pFont->pChars,pData=FontBuf+4; i<NumChars; i++,pChar++,pData+=8)
315  {
316  pChar->u0 = (((uint16 *)pData)[0] << 4) + 8;
317  pChar->v0 = (((uint16 *)pData)[1] << 4) + 8;
318  pChar->u1 = pChar->u0 + (((uint16 *)pData)[2] << 4);
319  pChar->v1 = pChar->v0 + (((uint16 *)pData)[3] << 4);
320  }
321 
322  // set font GS context
323  for (TW=0; (1<<TW)<Width; TW++)
324  ;
325  for (TH=0; (1<<TH)<Height; TH++)
326  ;
327  pFont->RegTEX0 = PackTEX0(pFont->m_TBP,TBW,PSMT8,TW,TH,1,MODULATE,pFont->m_CBP,PSMCT32,0,0,1);
328  pFont->RegTEX1 = PackTEX1(1,0,LINEAR,LINEAR,0,0,0);
329 
330 
331  // update vram base pointer
332  //FontVramBase = (((pFont->TBP<<8) + (TBW << 6)*Height + 0x1FFF) & 0xFFFFE000) >> 8;
333  FontVramBase += (pFont->m_NumTexVramBlocks + pFont->m_NumClutVramBlocks);
334  Dbg_MsgAssert(FontVramBase < (FontVramStart + FontVramSize), ("Font: 2D VRAM Buffer Full: %x.", FontVramBase));
335 
336 
337  // add font to font list
338  pFont->pNext = pFontList;
339  pFontList = pFont;
340 
341  // we're done with the font file now
342  File::Close(pFile);
343 
344  // And restore the CurrentDMALoc, which we were just using temporarily.
345  dma::pLoc = old_dma_loc;
346 
347  // this will serve as the default spacing
348  pFont->mSpaceSpacing = (pFont->pChars[pFont->Map['I']].u1 - pFont->pChars[pFont->Map['I']].u0) >> 4;
349 
350  return pFont;
351 }

436 void SFont::QueryString(char *String, float &width, float &height)
437 {
438  SChar *pChar;
439  char *pLetter;
440  uint16 u0,v0,x0,u1,v1,x1;
441 
442  x0 = 0;
443 
444  //sint16 char_spacing = 16 + (mCharSpacing<<4);
445  sint16 char_spacing = mCharSpacing<<4;
446  sint16 space_spacing = mSpaceSpacing<<4;
447 
448  for (pLetter=String;; pLetter++)
449  {
450  pChar = NULL;
451  // may be overridden by the '\b' tag
452  SFont *p_font = this;
453 
454  // acount for tags (might be multiple ones in a row)
455  bool got_char_tag = false; // tag resulting in output of character
456  while (*pLetter == '\\' && !got_char_tag)
457  {
458  pLetter++;
459  if (*pLetter == '\\')
460  break;
461 
462  switch(*pLetter)
463  {
464  case '\\':
465  got_char_tag = true;
466  break;
467  case 'c':
468  case 'C':
469  pLetter += 2; // skip over "c#"
470  break;
471  case 's':
472  case 'S':
473  {
474  pLetter++; // skip "s"
475  uint digit = Str::DehexifyDigit(pLetter);
476  pChar = pChars + SpecialMap[digit];
477  got_char_tag = true;
478  break;
479  }
480  case 'b':
481  case 'B':
482  {
483  pLetter++; // skip "b"
484  uint digit = Str::DehexifyDigit(pLetter);
485 
486  // switch over to buttons font, the regular font will be used again on the next character
487 
488  p_font = pButtonsFont;
489  Dbg_Assert(p_font);
490  pChar = p_font->pChars + p_font->SpecialMap[digit];
491  got_char_tag = true;
492  break;
493  }
494  case 'm':
495  case 'M':
496  {
497  pLetter++; // skip "m"
498  char button_char = Nx::CFontManager::sMapMetaCharacterToButton(pLetter);
499  uint digit = Str::DehexifyDigit(&button_char);
500 
501  p_font = pButtonsFont;
502  Dbg_Assert(p_font);
503  pChar = p_font->pChars + p_font->SpecialMap[digit];
504  got_char_tag = true;
505  break;
506  }
507  default:
508  Dbg_MsgAssert(0, ("unknown tag %c", *pLetter));
509  break;
510  }
511  } // end while
512 
513  if (*pLetter == '\0') break;
514 
515  if (*pLetter != ' ' || pChar)
516  {
517  if (!pChar)
518  pChar = p_font->pChars + p_font->Map[(uint8)*pLetter];
519  u0 = pChar->u0;
520  v0 = pChar->v0;
521  u1 = pChar->u1;
522  v1 = pChar->v1;
523  x1 = x0+u1-u0;
524  }
525  else
526  {
527  x1 = x0 + space_spacing;
528  }
529 
530  x0 = x1+char_spacing;
531  }
532 
533  width = (float)x0/16.0f;
534  height = (float)DefaultHeight/16.0f;
535 }
