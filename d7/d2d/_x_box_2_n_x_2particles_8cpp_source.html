79 sParticleSystem::sParticleSystem( uint32 max_particles, eParticleType type, uint32 texture_checksum, uint32 blendmode_checksum, int fix, int num_segments, int history )
80 {
81  // Obtain the texture.
82  Nx::CTexture * p_texture = Nx::CTexDictManager::sp_particle_tex_dict->GetTexture( texture_checksum );
83  Nx::CXboxTexture *p_xbox_texture = static_cast<Nx::CXboxTexture*>( p_texture );
84  NxXbox::sTexture *p_engine_texture = NULL;
85  if( p_xbox_texture )
86  {
87  p_engine_texture = p_xbox_texture->GetEngineTexture();
88  }
89 
90  // Create a (semi-transparent) material used to render the mesh.
91  sMaterial *p_material = new sMaterial();
92  p_material->m_flags[0] = 0x40 | (( p_engine_texture ) ? MATFLAG_TEXTURED : 0 );
93  p_material->m_checksum = (uint32)rand() * (uint32)rand();
94  p_material->m_passes = 1;
95  p_material->mp_tex[0] = p_engine_texture;
96  p_material->m_reg_alpha[0] = GetBlendMode( blendmode_checksum ) | ((uint32)fix << 24 );
97  p_material->m_color[0][0] = 0.5f;
98  p_material->m_color[0][1] = 0.5f;
99  p_material->m_color[0][2] = 0.5f;
100  p_material->m_color[0][3] = fix * ( 1.0f / 128.0f );
101 
102  p_material->m_uv_addressing[0] = 0;
103  p_material->m_k[0] = 0.0f;
104  p_material->m_alpha_cutoff = 1;
105  p_material->m_no_bfc = true;
106  p_material->m_uv_wibble = false;
107 
108  mp_material = p_material;
109 }
