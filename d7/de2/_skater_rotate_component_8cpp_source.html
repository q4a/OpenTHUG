91 void CSkaterRotateComponent::Update()
92 {
93  Mth::Matrix rotation_matrix;
94 
95  for (int n = X; n < Z + 1; n++)
96  {
97  SRotation& rotation = mp_rotations[n];
98 
99  if (!rotation.active) continue;
100 
101  float amount = Tmr::FrameLength() * rotation.angle_step;
102 
103  rotation.angle_traversed += Mth::Abs(amount);
104 
105  if (rotation.angle_traversed >= rotation.angle)
106  {
107  rotation.active = false;
108  amount -= Mth::Sgn(amount) * (rotation.angle_traversed - rotation.angle);
109  }
110 
111  Mth::CreateRotateMatrix(rotation_matrix, n, amount);
112 
113  GetObject()->m_matrix = rotation_matrix * GetObject()->m_matrix;
114  mp_core_physics_component->m_lerping_display_matrix = rotation_matrix * mp_core_physics_component->m_lerping_display_matrix;;
115  }
116 
117  if (!mp_rotations[0].active && !mp_rotations[1].active && !mp_rotations[2].active)
118  {
119  Suspend(true);
120  }
121 }

128 CBaseComponent::EMemberFunctionResult CSkaterRotateComponent::CallMemberFunction( uint32 Checksum, Script::CStruct* pParams, Script::CScript* pScript )
129 {
130  switch ( Checksum )
131  {
132  // @script | Rotate |
133  // @parmopt float | Duration | 0.0 | duration time (default is ms)
134  // @flag seconds | time in seconds
135  // @flag frames | time in frames
136  // @parmopt float | x | 0.0 | x rotation angle
137  // @parmopt float | y | 0.0 | y rotation angle
138  // @parmopt float | z | 0.0 | z rotation angle
139  // @parmopt name | Node | | Node to point to. Currently only supported if a duration is specified.
140  case CRCC(0xc876adf3, "Rotate"):
141  {
142  float duration;
143  if (pParams->GetFloat(CRCD(0x79a07f3f, "Duration"), &duration))
144  {
145  Dbg_MsgAssert(duration > 0.0f, ("Zero or negative Rotate Duration"));
146 
147  if (pParams->ContainsFlag(CRCD(0xd029f619, "Seconds")) || pParams->ContainsFlag(CRCD(0x49e0ee96, "Second")))
148  {
149  // Seconds is what we want, so nothing to do
150  }
151  else if (pParams->ContainsFlag(CRCD(0x19176c5, "Frames")) || pParams->ContainsFlag(CRCD(0x4a07c332, "Frame")))
152  {
153  // Convert from frames to seconds
154  duration /= 60.0f;
155  }
156  else
157  {
158  // Convert from milliseconds to seconds
159  duration /= 1000.0f;
160  }
161 
162  if (pParams->GetFloat(CRCD(0x7323e97c, "x"), &mp_rotations[X].angle))
163  {
164  mp_rotations[X].duration = duration;
165  mp_rotations[X].angle_step = Mth::DegToRad(mp_rotations[X].angle / mp_rotations[X].duration);
166  mp_rotations[X].angle = Mth::Abs(Mth::DegToRad(mp_rotations[X].angle));
167  mp_rotations[X].angle_traversed = 0.0f;
168  mp_rotations[X].active = true;
169  }
170  if (pParams->GetFloat(CRCD(0x424d9ea, "y"), &mp_rotations[Y].angle))
171  {
172  mp_rotations[Y].duration = duration;
173  mp_rotations[Y].angle_step = Mth::DegToRad(mp_rotations[Y].angle / mp_rotations[Y].duration);
174  mp_rotations[Y].angle = Mth::Abs(Mth::DegToRad(mp_rotations[Y].angle));
175  mp_rotations[Y].angle_traversed = 0.0f;
176  mp_rotations[Y].active = true;
177  }
178  if (pParams->GetFloat(CRCD(0x9d2d8850, "z"), &mp_rotations[Z].angle))
179  {
180  mp_rotations[Z].duration = duration;
181  mp_rotations[Z].angle_step = Mth::DegToRad(mp_rotations[Z].angle / mp_rotations[Z].duration);
182  mp_rotations[Z].angle = Mth::Abs(Mth::DegToRad(mp_rotations[Z].angle));
183  mp_rotations[Z].angle_traversed = 0.0f;
184  mp_rotations[Z].active = true;
185  }
186 
187  // If a node is specified, turn to point to it instead.
188  // Added by Ken for use by Brad when making the skater turn around when he goes outside the level limits.
189  uint32 node_name = 0;
190  if (pParams->GetChecksum(CRCD(0x7a8017ba, "Node"), &node_name))
191  {
192  int node = SkateScript::FindNamedNode(node_name);
193  Mth::Vector node_pos;
194  SkateScript::GetPosition(node, &node_pos);
195 
196  mp_rotations[Y].angle = Mth::RadToDeg(Mth::GetAngle(GetObject()->m_matrix, node_pos - GetObject()->m_pos));
197 
198  mp_rotations[Y].duration = duration;
199  mp_rotations[Y].angle_step = Mth::DegToRad(mp_rotations[Y].angle / mp_rotations[Y].duration);
200  mp_rotations[Y].angle = Mth::Abs(Mth::DegToRad(mp_rotations[Y].angle));
201  mp_rotations[Y].angle_traversed = 0.0f;
202  mp_rotations[Y].active = true;
203  }
204 
205  Suspend(false);
206  } // END if duration specified
207  else
208  {
209  // rotate immediately
210  float angle;
211  if (pParams->GetFloat(CRCD(0x7323e97c, "x"), &angle))
212  {
213  GetObject()->m_matrix.RotateXLocal(Mth::DegToRad(angle));
214  mp_core_physics_component->ResetLerpingMatrix();
215  }
216  else if (pParams->GetFloat(CRCD(0x424d9ea, "y"), &angle))
217  {
218  GetObject()->m_matrix.RotateYLocal(Mth::DegToRad(angle));
219  mp_core_physics_component->ResetLerpingMatrix();
220  }
221  else if (pParams->GetFloat(CRCD(0x9d2d8850, "z"), &angle))
222  {
223  GetObject()->m_matrix.RotateZLocal(Mth::DegToRad(angle));
224  mp_core_physics_component->ResetLerpingMatrix();
225  }
226  else
227  {
228  // if no parameters are given, rotate 180 degrees about Y
229  GetObject()->m_matrix[Z].Negate();
230  GetObject()->m_matrix[X].Negate();
231  mp_core_physics_component->ResetLerpingMatrix();
232 
233  mp_core_physics_component->mRail_Backwards = !mp_core_physics_component->mRail_Backwards;
234  }
235  } // END else no Duration specified
236  break;
237  }
238 
239  default:
240  return CBaseComponent::MF_NOT_EXECUTED;
241  }
242  return CBaseComponent::MF_TRUE;
243 }
