53 // Define the maximum amount of packets we will ever submit to the ADPCM renderer
54 #define ADPCMSTRM_PACKET_COUNT 8
55 
56 // Define the ADPCM renderer packet size. See the comment block above for an explanation.
57 // This value is hard-coded assuming an ADPCM frame of 36 samples and 16 bit stereo (128 frames per packet)
58 #define ADPCMSTRM_16BIT_MONO_PACKET_BYTES ( 1 * 2 * 36 * 128 )
59 #define ADPCMSTRM_16BIT_STEREO_PACKET_BYTES ( 2 * 2 * 36 * 128 )
60 
61 // Read size is 16k (most efficient size for DVD reads).
62 #define BYTES_PER_READ 16384

150 //-----------------------------------------------------------------------------
151 // Name: AsyncRead()
152 // Desc: Called while the async read is in progress. If the read is still
153 // underway, simply returns. If the read has completed, sets up the
154 // next read. If the file has been completely read, closes the file.
155 //-----------------------------------------------------------------------------
156 void CADPCMFileStream::AsyncRead( void )
157 {
158  Dbg_Assert( m_hFile != INVALID_HANDLE_VALUE );
159 
160  // If paused, do nothing.
161  if( m_Paused )
162  {
163  return;
164  }
165 
166  // See if the previous read is complete.
167  DWORD dwBytesTransferred;
168  BOOL bIsReadDone = GetOverlappedResult( m_hFile, m_pOverlapped, &dwBytesTransferred, FALSE );
169  DWORD dwLastError = GetLastError();
170 
171  // If the read isn't complete, keep going.
172  if( !bIsReadDone )
173  {
174  Dbg_Assert( dwLastError == ERROR_IO_INCOMPLETE );
175  if( dwLastError != ERROR_IO_INCOMPLETE )
176  {
177  m_AwaitingDeletion = true;
178  }
179  return;
180  }
181 
182  // If we get here, the read is complete.
183  m_pOverlapped->Offset += dwBytesTransferred;
184  m_FileBytesRead += dwBytesTransferred;
185 
186  ++m_SuccessiveReads;
187 
188  if( dwBytesTransferred < BYTES_PER_READ )
189  {
190  // We've reached the end of the file during the call to ReadFile.
191 
192  // Close the file (if not using the global WAD file).
193  if( !m_bUseWAD )
194  {
195  BOOL bSuccess = CloseHandle( m_hFile );
196  Dbg_Assert( bSuccess );
197  }
198 
199  m_hFile = INVALID_HANDLE_VALUE;
200 
201  // All done
202  m_ReadComplete = true;
203  }
204  else
205  {
206  if( m_bUseWAD && ( m_FileBytesRead >= (int)m_dwWADLength ))
207  {
208  m_hFile = INVALID_HANDLE_VALUE;
209 
210  // All done
211  m_ReadComplete = true;
212  return;
213  }
214 
215  // We still have more data to read. Start another asynchronous read from the file.
216  BOOL bComplete = ReadFile( m_hFile, (BYTE*)m_pFileBuffer + ( m_FileBytesRead % PCMAudio_GetFilestreamBufferSize()), BYTES_PER_READ, NULL, m_pOverlapped );
217  dwLastError = GetLastError();
218 
219  // Deal with hitting EOF (for files that are some exact multiple of BYTES_PER_READ bytes).
220  if( bComplete || ( !bComplete && ( dwLastError == ERROR_HANDLE_EOF )))
221  {
222  // Close the file
223  if( !m_bUseWAD )
224  {
225  BOOL bSuccess = CloseHandle( m_hFile );
226  Dbg_Assert( bSuccess );
227  }
228  m_hFile = INVALID_HANDLE_VALUE;
229 
230  // All done
231  m_ReadComplete = true;
232  }
233  else
234  {
235  Dbg_MsgAssert( bComplete || ( !bComplete && ( dwLastError == ERROR_IO_PENDING )), ( "ReadFile error: %x\n", dwLastError ));
236  if( !bComplete && ( dwLastError != ERROR_IO_PENDING ))
237  {
238  // There was a problem, so shut this stream down.
239  m_AwaitingDeletion = true;
240  }
241  }
242  }
243 }

251 HRESULT CADPCMFileStream::Initialize( HANDLE h_file, unsigned int offset, unsigned int length, void* fileBuffer )
252 {
253  m_dwPercentCompleted = 0;
254 
255  // At this stage we don't want to create the decoder or the stream. We do want to just allocate the read
256  // buffer, and start pulling in the data. Once sufficient data has been grabbed, we can analyze the header
257  // and create the required objects for playback.
258  m_hFile = h_file;
259  m_bUseWAD = true;
260  m_dwWADOffset = offset;
261  m_dwWADLength = length;
262  m_pFileBuffer = fileBuffer;
263 
264  Dbg_Assert( DWORD( m_pFileBuffer ) % sizeof( DWORD ) == 0 );
265 
266  return PostInitialize();

271 //-----------------------------------------------------------------------------
272 // Name: PostInitialize()
273 // Desc: Initialisation stuff following the file creation
274 //-----------------------------------------------------------------------------
275 HRESULT CADPCMFileStream::PostInitialize( void )
276 {
277  // Start the asynchronous read from the start of the file.
278  m_FirstRead = true;
279  m_ReadComplete = false;
280  m_FileBytesRead = 0;
281  m_FileBytesProcessed = 0;
282  m_SuccessiveReads = 0;
283 
284  if( m_bUseWAD )
285  {
286  m_pOverlapped->Offset = m_dwWADOffset;
287  }
288  else
289  {
290  m_pOverlapped->Offset = 0;
291  }
292  m_pOverlapped->OffsetHigh = 0;
293 
294  BOOL bComplete = ReadFile( m_hFile, m_pFileBuffer, BYTES_PER_READ, NULL, m_pOverlapped );
295  if( !bComplete )
296  {
297  DWORD dwLastError = GetLastError();
298  Dbg_Assert( dwLastError == ERROR_IO_PENDING );
299  if( dwLastError != ERROR_IO_PENDING )
300  {
301  return HRESULT_FROM_WIN32( dwLastError );
302  }
303  }
304  else
305  {
306  m_ReadComplete = true;
307  }
308 
309  // That's it for now.
310  return S_OK;
311 }

315 //-----------------------------------------------------------------------------
316 // Name: InitializeFormatBlock()
317 // Desc:
318 //-----------------------------------------------------------------------------
319 HRESULT CADPCMFileStream::InitializeFormatBlock( uint32 *p_header_data )
320 {
321  // Ensure the format block is where we expect it to be.
322  if( p_header_data[3] == 0x20746D66UL )
323  {
324  // Read format and number of channels.
325  m_wfxExtendedSourceFormat.m_wfxSourceFormat.wFormatTag = (uint16)( p_header_data[5] & 0xFFFFUL );
326  m_wfxExtendedSourceFormat.m_wfxSourceFormat.nChannels = (uint16)( p_header_data[5] >> 16 );
327 
328  // Make sure this is Xbox ADPCM.
329  if( m_wfxExtendedSourceFormat.m_wfxSourceFormat.wFormatTag != WAVE_FORMAT_XBOX_ADPCM )
330  {
331  return E_FAIL;
332  }
333 
334  // Read samples per second.
335  m_wfxExtendedSourceFormat.m_wfxSourceFormat.nSamplesPerSec = p_header_data[6];
336 
337  // Read average bytes per second.
338  m_wfxExtendedSourceFormat.m_wfxSourceFormat.nAvgBytesPerSec = p_header_data[7];
339 
340  // Read block alignment and bits per sample.
341  m_wfxExtendedSourceFormat.m_wfxSourceFormat.nBlockAlign = (uint16)( p_header_data[8] & 0xFFFFUL );
342  m_wfxExtendedSourceFormat.m_wfxSourceFormat.wBitsPerSample = (uint16)( p_header_data[8] >> 16 );
343 
344  // Extra information.
345  m_wfxExtendedSourceFormat.m_wfxSourceFormat.cbSize = (uint16)( p_header_data[9] & 0xFFFFUL );
346  m_wfxExtendedSourceFormat.m_extendedInfo = (uint16)( p_header_data[9] >> 16 );
347 
348  // We have now processed the first 48 bytes of data.
349  m_FileBytesProcessed = 48;
350 
351  // Now that we know the format, we can set the packet size.
352  if( m_wfxExtendedSourceFormat.m_wfxSourceFormat.nChannels == 1 )
353  m_PacketBytes = ADPCMSTRM_16BIT_MONO_PACKET_BYTES;
354  else
355  m_PacketBytes = ADPCMSTRM_16BIT_STEREO_PACKET_BYTES;
356 
357  return S_OK;
358  }
359  return E_FAIL;
360 }

375  Mem::Manager::sHandle().PopContext();
376 
377  return ( m_pvSourceBuffer != NULL );

400 //-----------------------------------------------------------------------------
401 // Name: Process()
402 // Desc: Performs any work necessary to keep the stream playing.
403 //-----------------------------------------------------------------------------
404 HRESULT CADPCMFileStream::Process( void )
405 {
406  HRESULT hr;
407  DSSTREAMDESC dssd;
408  DWORD dwPacketIndex;
409 
410  // Do nothing if waiting to die.
411  if( m_AwaitingDeletion )
412  {
413  return S_OK;
414  }
415 
416  // Do we need to kick off another read of data?
417  // Don't read anymore if we have read ahead to the point where we are within 32k of free read buffer space.
418  if( !m_ReadComplete )
419  {
420  if( m_FileBytesRead <= ( m_FileBytesProcessed + ((int)PCMAudio_GetFilestreamBufferSize() - 32768 )))
421  {
422  AsyncRead();
423  }
424  }
425 
426  // Has the first block of raw data been read? If so we need to instantiate the playback objects and data buffers.
427  if( m_FirstRead && ( m_FileBytesRead > 1024 ))
428  {
429  if( m_pSourceFilter == NULL )
430  {
431  // Create the thread which will create the in-memory decoder.
432  m_DecoderCreation = 0;
433 
434  // For WMA format, here is where we create the decoder. However, since there is no CPU-side decompression required
435  // for ADPCM format, there is no requirement for a decoder.
436  m_DecoderCreation = 1;
437  }
438 
439  if( m_DecoderCreation == 0 )
440  {
441  // Still waiting to create decoder.
442  return S_OK;
443  }
444  else if( m_DecoderCreation == 2 )
445  {
446  // Failed to create decoder, just mark for deletion.
447  m_AwaitingDeletion = true;
448  return S_OK;
449  }
450  else if( m_DecoderCreation == 1 )
451  {
452  // Managed to create decoder.
453  m_FirstRead = false;
454 
455  // Set up the format information.
456  if( InitializeFormatBlock((uint32*)m_pFileBuffer ) == E_FAIL )
457  {
458  // Failed to decode format information, just mark for deletion.
459  m_AwaitingDeletion = true;
460  return S_OK;
461  }
462 
463  // Create the render (DirectSoundStream) filter.
464  DSMIXBINS dsmixbins;
465  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
466  ZeroMemory( &dssd, sizeof( dssd ));
467 
468  dssd.dwFlags = 0;
469  dssd.dwMaxAttachedPackets = ADPCMSTRM_PACKET_COUNT;
470  dssd.lpwfxFormat = &m_wfxExtendedSourceFormat.m_wfxSourceFormat;
471  dssd.lpMixBins = &dsmixbins;
472 
473  if( m_bUse3D )
474  {
475  // This is only designed to be fed a mono signal.
476  Dbg_Assert( m_wfxExtendedSourceFormat.m_wfxSourceFormat.nChannels == 1 );
477 
478  dsmixbins.dwMixBinCount = 6;
479  dsmixbins.lpMixBinVolumePairs = dsmbvp;
480  dsmbvp[0].dwMixBin = DSMIXBIN_3D_FRONT_LEFT;
481  dsmbvp[0].lVolume = DSBVOLUME_EFFECTIVE_MIN;
482  dsmbvp[1].dwMixBin = DSMIXBIN_3D_FRONT_RIGHT;
483  dsmbvp[1].lVolume = DSBVOLUME_EFFECTIVE_MIN;
484  dsmbvp[2].dwMixBin = DSMIXBIN_3D_BACK_LEFT;
485  dsmbvp[2].lVolume = DSBVOLUME_EFFECTIVE_MIN;
486  dsmbvp[3].dwMixBin = DSMIXBIN_3D_BACK_RIGHT;
487  dsmbvp[3].lVolume = DSBVOLUME_EFFECTIVE_MIN;
488  dsmbvp[4].dwMixBin = DSMIXBIN_FRONT_CENTER;
489  dsmbvp[4].lVolume = DSBVOLUME_EFFECTIVE_MIN;
490  dsmbvp[5].dwMixBin = DSMIXBIN_I3DL2;
491  dsmbvp[5].lVolume = DSBVOLUME_EFFECTIVE_MIN;
492 
493  m_Mixbins = ( 1 << DSMIXBIN_3D_FRONT_LEFT ) |
494  ( 1 << DSMIXBIN_3D_FRONT_RIGHT ) |
495  ( 1 << DSMIXBIN_3D_BACK_LEFT ) |
496  ( 1 << DSMIXBIN_3D_BACK_RIGHT ) |
497  ( 1 << DSMIXBIN_FRONT_CENTER ) |
498  ( 1 << DSMIXBIN_I3DL2 );
499  m_NumMixbins = dsmixbins.dwMixBinCount;
500  }
501  else
502  {
503  // If we are playing a music track, and if proper 5.1 output is selected, we want to feed the music to the left and
504  // right back speakers with a slight echo via mixbins 5 and 6.
505  // This is currently disabled.
506  if( false && ( XGetAudioFlags() & XC_AUDIO_FLAGS_ENABLE_AC3 ))
507  {
508  // This is only designed to be fed a stereo signal.
509  Dbg_Assert( m_wfxExtendedSourceFormat.m_wfxSourceFormat.nChannels == 2 );
510 
511  dsmixbins.dwMixBinCount = 4;
512  dsmixbins.lpMixBinVolumePairs = dsmbvp;
513  dsmbvp[0].dwMixBin = DSMIXBIN_FRONT_LEFT;
514  dsmbvp[0].lVolume = DSBVOLUME_MIN;
515  dsmbvp[1].dwMixBin = DSMIXBIN_FRONT_RIGHT;
516  dsmbvp[1].lVolume = DSBVOLUME_MIN;
517  dsmbvp[2].dwMixBin = DSMIXBIN_FXSEND_5;
518  dsmbvp[2].lVolume = DSBVOLUME_MIN;
519  dsmbvp[3].dwMixBin = DSMIXBIN_FXSEND_6;
520  dsmbvp[3].lVolume = DSBVOLUME_MIN;
521 
522  m_Mixbins = ( 1 << DSMIXBIN_FRONT_LEFT ) |
523  ( 1 << DSMIXBIN_FRONT_RIGHT ) |
524  ( 1 << DSMIXBIN_FXSEND_5 ) |
525  ( 1 << DSMIXBIN_FXSEND_6 );
526  m_NumMixbins = dsmixbins.dwMixBinCount;
527  }
528  else
529  {
530  // This is only designed to be fed a stereo signal.
531  Dbg_Assert( m_wfxExtendedSourceFormat.m_wfxSourceFormat.nChannels == 2 );
532 
533  dsmixbins.dwMixBinCount = 4;
534  dsmixbins.lpMixBinVolumePairs = dsmbvp;
535  dsmbvp[0].dwMixBin = DSMIXBIN_FRONT_LEFT;
536  dsmbvp[0].lVolume = DSBVOLUME_MIN;
537  dsmbvp[1].dwMixBin = DSMIXBIN_FRONT_RIGHT;
538  dsmbvp[1].lVolume = DSBVOLUME_MIN;
539  dsmbvp[2].dwMixBin = DSMIXBIN_BACK_LEFT;
540  dsmbvp[2].lVolume = DSBVOLUME_MIN;
541  dsmbvp[3].dwMixBin = DSMIXBIN_BACK_RIGHT;
542  dsmbvp[3].lVolume = DSBVOLUME_MIN;
543 
544  m_Mixbins = ( 1 << DSMIXBIN_FRONT_LEFT ) |
545  ( 1 << DSMIXBIN_FRONT_RIGHT ) |
546  ( 1 << DSMIXBIN_BACK_LEFT ) |
547  ( 1 << DSMIXBIN_BACK_RIGHT );
548  m_NumMixbins = dsmixbins.dwMixBinCount;
549  }
550  }
551 
552  hr = DirectSoundCreateStream( &dssd, &m_pRenderFilter );
553  if( FAILED( hr ))
554  {
555  Dbg_Assert( 0 );
556  m_pRenderFilter = NULL;
557  m_AwaitingDeletion = true;
558  return S_OK;
559  }
560 
561  // Set deferred volume if present.
562  SetDeferredVolume();
563 
564  // Handle deferred pause.
565  if( m_Paused )
566  {
567  m_pRenderFilter->Pause( 1 );
568  }
569  }
570  }
571 
572  // We only want to do processing when there is sufficient data available.
573  if( m_ReadComplete || ( m_FileBytesRead >= ( m_FileBytesProcessed + m_PacketBytes )))
574  {
575  if( !m_Completed && m_pRenderFilter )
576  {
577  // Find a free packet. If there's none free, we don't have anything to do.
578  if( FindFreePacket( &dwPacketIndex ))
579  {
580  // Read from the source filter.
581  if( m_bOkayToPlay )
582  {
583  hr = ProcessSource( dwPacketIndex );
584  if( FAILED( hr ))
585  {
586  return hr;
587  }
588 
589  if( !m_Paused )
590  {
591  // Send the data to the renderer
592  hr = ProcessRenderer( dwPacketIndex );
593  if( FAILED( hr ))
594  {
595  return hr;
596  }
597  }
598  }
599  }
600  }
601  }
602  return S_OK;
603 }

608 //-----------------------------------------------------------------------------
609 // Name: FindFreePacket()
610 // Desc: Finds a render packet available for processing.
611 //-----------------------------------------------------------------------------
612 BOOL CADPCMFileStream::FindFreePacket( DWORD* pdwPacketIndex )
613 {
614  for( DWORD dwPacketIndex = 0; dwPacketIndex < ADPCMSTRM_PACKET_COUNT; ++dwPacketIndex )
615  {
616  if( XMEDIAPACKET_STATUS_PENDING != m_adwPacketStatus[dwPacketIndex] )
617  {
618  if( pdwPacketIndex )
619  {
620  (*pdwPacketIndex) = dwPacketIndex;
621  }
622  return TRUE;
623  }
624  }
625  return FALSE;
626 }

631 //-----------------------------------------------------------------------------
632 // Name: ProcessSource()
633 // Desc: Reads data from the source filter.
634 //-----------------------------------------------------------------------------
635 HRESULT CADPCMFileStream::ProcessSource( DWORD dwPacketIndex )
636 {
637  if( m_pvSourceBuffer == NULL )
638  {
639  if( CreateSourceBuffer() == false )
640  {
641  // Failed to create the source buffer, mark for deletion.
642  m_AwaitingDeletion = true;
643  return E_FAIL;
644  }
645  }
646 
647  // We just want to copy a full packet's worth of data from the file buffer directly into the source buffer...
648  uint8* p_destination = (BYTE*)m_pvSourceBuffer + ( dwPacketIndex * m_PacketBytes );
649 
650  // However we don't want to overrun the file buffer when copying.
651  uint32 file_buffer_offset = m_FileBytesProcessed % PCMAudio_GetFilestreamBufferSize();
652  if(( PCMAudio_GetFilestreamBufferSize() - file_buffer_offset ) < (uint32)m_PacketBytes )
653  {
654  // Copying the data in one chunk will take us beyond the edge of the file buffer.
655  // So we need to do the copy in two chunks.
656  uint8* p_source = (BYTE*)m_pFileBuffer + file_buffer_offset;
657  uint32 first_chunk_bytes = PCMAudio_GetFilestreamBufferSize() - file_buffer_offset;
658  CopyMemory( p_destination, p_source, first_chunk_bytes );
659 
660  // Wrap file buffer back round to start.
661  p_source = (BYTE*)m_pFileBuffer + 0;
662  p_destination = p_destination + first_chunk_bytes;
663  CopyMemory( p_destination, p_source, m_PacketBytes - first_chunk_bytes );
664  }
665  else
666  {
667  // Copying the data in one chunk is fine.
668  uint8* p_source = (BYTE*)m_pFileBuffer + file_buffer_offset;
669  CopyMemory( p_destination, p_source, m_PacketBytes );
670  }
671 
672  // Now these bytes have been processed.
673  m_FileBytesProcessed += m_PacketBytes;
674 
675  // If we've caught up with the number of bytes read, it's because we've finished processing.
676  if( m_FileBytesProcessed >= m_FileBytesRead )
677  {
678  // Set completion flag.
679  m_Completed = true;
680 
681  // Zero remaining part of packet.
682  uint32 bytes_to_zero = m_FileBytesProcessed - m_FileBytesRead;
683  if( bytes_to_zero > 0 )
684  {
685  p_destination = (BYTE*)m_pvSourceBuffer + ( dwPacketIndex * m_PacketBytes ) + m_PacketBytes - bytes_to_zero;
686  ZeroMemory( p_destination, bytes_to_zero );
687  }
688  }
689  return S_OK;
690 }

695 //-----------------------------------------------------------------------------
696 // Name: ProcessRenderer()
697 // Desc: Sends data to the renderer.
698 //-----------------------------------------------------------------------------
699 HRESULT CADPCMFileStream::ProcessRenderer( DWORD dwPacketIndex )
700 {
701  XMEDIAPACKET xmp;
702  HRESULT hr;
703 
704  Dbg_Assert( m_pvSourceBuffer != NULL );
705 
706  // There's a full packet's worth of data ready for us to send to the renderer. We want to track the status
707  // of this packet since the render filter is asychronous and we need to know when the packet is completed.
708  ZeroMemory( &xmp, sizeof( xmp ));

711  xmp.pdwStatus = &m_adwPacketStatus[dwPacketIndex];

716  m_LastPacket = dwPacketIndex;

718

720 
721  if( m_Completed )
722  {
723  // Tell the renderer not to expect any more data.
724  m_pRenderFilter->Discontinuity();
725  }
726 
727  if( FAILED( hr ))
728  {
729  return hr;
730  }
731

757 //-----------------------------------------------------------------------------
758 void CADPCMFileStream::SetVolume( float volume )
759 {
760  if( m_pRenderFilter )
761  {
762  int i_volume = DSBVOLUME_EFFECTIVE_MIN;
763  int i_volume_rear = DSBVOLUME_EFFECTIVE_MIN;
764  if( volume > 0.0f )
765  {
766  // Figure base volume.
767  float attenuation = 20.0f * log10f( volume * 0.01f );
768  i_volume = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
769  if( i_volume < DSBVOLUME_EFFECTIVE_MIN )
770  i_volume = DSBVOLUME_EFFECTIVE_MIN;
771  else if( i_volume > DSBVOLUME_MAX )
772  i_volume = DSBVOLUME_MAX;
773 
774  // Also figure half volume, in case we are routing to the back speakers.
775  attenuation = 20.0f * log10f( volume * 0.5f * 0.01f );
776  i_volume_rear = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
777  if( i_volume_rear < DSBVOLUME_EFFECTIVE_MIN )
778  i_volume_rear = DSBVOLUME_EFFECTIVE_MIN;
779  else if( i_volume_rear > DSBVOLUME_MAX )
780  i_volume_rear = DSBVOLUME_MAX;
781  }
782 
783  // Set individual mixbins for panning.
784  DSMIXBINS dsmixbins;
785  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
786 
787  dsmixbins.dwMixBinCount = 0;
788  dsmixbins.lpMixBinVolumePairs = dsmbvp;
789 
790  if( i_volume > DSBVOLUME_EFFECTIVE_MIN )
791  {
792  // Set the volume up depending on how the initial mixbins were set up.
793  int mbbf = 0;
794  for( uint32 mb = 0; mb < m_NumMixbins; ++mb )
795  {
796  while(( m_Mixbins & ( 1 << mbbf )) == 0 )
797  {
798  ++mbbf;
799  Dbg_Assert( mbbf < 32 );
800  }
801  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = mbbf;
802 
803  // For rear speakers (or mixbins that route to the rear), use half volume.
804  if(( mbbf == DSMIXBIN_FXSEND_5 ) || ( mbbf == DSMIXBIN_FXSEND_6 ) || ( mbbf == DSMIXBIN_BACK_LEFT ) || ( mbbf == DSMIXBIN_BACK_RIGHT ))
805  {
806  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volume_rear;
807  }
808  else
809  {
810  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volume;
811  }
812  ++dsmixbins.dwMixBinCount;
813  ++mbbf;
814  }
815  }
816 
817  // Set all speaker volumes.
818  m_pRenderFilter->SetMixBinVolumes( &dsmixbins );
819 
820  // Set overall buffer volume.
821  if( dsmixbins.dwMixBinCount > 0 )
822  {
823  m_pRenderFilter->SetVolume( DSBVOLUME_MAX );
824  }
825  else
826  {
827  m_pRenderFilter->SetVolume( DSBVOLUME_MIN );
828  }
829  }
830  else
831  {
832  m_SetDeferredVolume = true;
833  m_DeferredVolume[0] = volume;
834  }

839 //-----------------------------------------------------------------------------
840 // Name: SetVolume
841 // Desc:
842 //-----------------------------------------------------------------------------
843 void CADPCMFileStream::SetVolume( float volL, float volR )
844 {
845  if( m_pRenderFilter )
846  {
847  // This array will hold individual volumes for the five speakers.
848  // In order, they are: front left, center, front right, rear right, rear left.
849  float volumes[5];
850  int i_volumes[5], max_i_volume;
851  memset( volumes, 0, sizeof( float ) * 5 );
852 
853  if(( volL == 0.0f ) && ( volR == 0.0f ))
854  {
855  // Pointless doing any more work.
856  }
857  else
858  {
859  // Get the length of the vector here which will be used to multiply out the normalised speaker volumes.
860  Mth::Vector test( fabsf( volL ), fabsf( volR ), 0.0f, 0.0f );
861  float amplitude = test.Length();
862 
863  // Look just at the normalized right component to figure the sound angle from Matt's calculations.
864  test.Normalize();
865 
866  float angle;
867  angle = asinf( test[Y] );
868  angle = ( angle * 2.0f ) - ( Mth::PI * 0.5f );
869  angle = ( volL < 0.0f ) ? ( Mth::PI - angle ) : angle;
870 
871  // Now figure volumes based on speaker coverage.
872  angle = Mth::RadToDeg( angle );
873 
874  Spt::SingletonPtr< Sfx::CSfxManager > sfx_manager;
875  sfx_manager->Get5ChannelMultipliers( angle, &volumes[0] );
876 
877  // Now readjust the relative values...
878  for( int v = 0; v < 5; ++v )
879  {
880  // Scale back up to original amplitude.
881  volumes[v] *= amplitude;
882 
883  if( volumes[v] > 100.0f )
884  volumes[v] = 100.0f;
885  }
886  }
887 
888  // Now figure the attenuation of the sound. To convert to a decibel value, figure the ratio of requested
889  // volume versus max volume, then calculate the log10 and multiply by (10 * 2). (The 2 is because sound
890  // power varies as square of pressure, and squaring doubles the log value).
891  max_i_volume = DSBVOLUME_EFFECTIVE_MIN;
892  for( int v = 0; v < 5; ++v )
893  {
894  if( volumes[v] > 0.0f )
895  {
896  float attenuation = 20.0f * log10f( volumes[v] * 0.01f );
897  i_volumes[v] = DSBVOLUME_MAX + (int)( attenuation * 100.0f );
898  if( i_volumes[v] < DSBVOLUME_EFFECTIVE_MIN )
899  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
900  else if( i_volumes[v] > DSBVOLUME_MAX )
901  i_volumes[v] = DSBVOLUME_MAX;
902 
903  if( i_volumes[v] > max_i_volume )
904  max_i_volume = i_volumes[v];
905  }
906  else
907  {
908  i_volumes[v] = DSBVOLUME_EFFECTIVE_MIN;
909  }
910  }
911 
912  // Set individual mixbins for panning.
913  DSMIXBINS dsmixbins;
914  DSMIXBINVOLUMEPAIR dsmbvp[DSMIXBIN_ASSIGNMENT_MAX];
915 
916  dsmixbins.dwMixBinCount = 0;
917  dsmixbins.lpMixBinVolumePairs = dsmbvp;
918 
919  if( i_volumes[0] > DSBVOLUME_EFFECTIVE_MIN )
920  {
921  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_LEFT;
922  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[0];
923  dsmixbins.dwMixBinCount++;
924  }
925  if( i_volumes[1] > DSBVOLUME_EFFECTIVE_MIN )
926  {
927  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_FRONT_RIGHT;
928  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[1];
929  dsmixbins.dwMixBinCount++;
930  }
931  if( i_volumes[2] > DSBVOLUME_EFFECTIVE_MIN )
932  {
933  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_LEFT;
934  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[2];
935  dsmixbins.dwMixBinCount++;
936  }
937  if( i_volumes[3] > DSBVOLUME_EFFECTIVE_MIN )
938  {
939  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_3D_BACK_RIGHT;
940  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[3];
941  dsmixbins.dwMixBinCount++;
942  }
943  if( i_volumes[4] > DSBVOLUME_EFFECTIVE_MIN )
944  {
945  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_FRONT_CENTER;
946  dsmbvp[dsmixbins.dwMixBinCount].lVolume = i_volumes[4];
947  dsmixbins.dwMixBinCount++;
948  }
949  if( dsmixbins.dwMixBinCount > 0 )
950  {
951  dsmbvp[dsmixbins.dwMixBinCount].dwMixBin = DSMIXBIN_I3DL2;
952  dsmbvp[dsmixbins.dwMixBinCount].lVolume = DSBVOLUME_MAX;
953  dsmixbins.dwMixBinCount++;
954  }
955 
956  // Set all speaker volumes.
957  m_pRenderFilter->SetMixBinVolumes( &dsmixbins );
958 
959  // Set overall buffer volume.
960  if( dsmixbins.dwMixBinCount > 0 )
961  {
962  m_pRenderFilter->SetVolume( DSBVOLUME_MAX );
963  }
964  else
965  {
966  m_pRenderFilter->SetVolume( DSBVOLUME_MIN );
967  }
968  }
969  else
970  {
971  m_SetDeferredVolumeLR = true;
972  m_DeferredVolume[0] = volL;
973  m_DeferredVolume[1] = volR;
974  }
975 }
