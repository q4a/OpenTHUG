161 void CreateParticleEmitter( CGeneralManager* p_obj_man, Script::CStruct* pNodeData )
162 {
163  Dbg_MsgAssert( pNodeData, ( "Couldn't find node script." ) );
164 
165  // Run the trigger script, and this will create the game object
166  // then need to find the game object
167  uint32 AIScriptChecksum=0;
168  pNodeData->GetChecksum( "TriggerScript", &AIScriptChecksum, true );
169  uint32 NameChecksum=0;
170  pNodeData->GetChecksum( "Name", &NameChecksum, true );
171  if ( AIScriptChecksum )
172  {
173  // Make sure we don't have an unnamed particle system already
174  const uint32 PARTICLE_UNNAMED = 0x502335a1; // Particle_UnNamed
175  Dbg_MsgAssert(!Nx::CEngine::sGetParticleTable()->GetItem(PARTICLE_UNNAMED),("Particle_UnNamed should not exist"));
176 
177  // Get the parameters for the TriggerScript, and run it
178  // this should create a particle system
179  Script::CScriptStructure *pScriptParams=NULL;
180  pNodeData->GetStructure("Params",&pScriptParams);
181  Script::RunScript(AIScriptChecksum,pScriptParams); // this will create the particle emitter
182 
183  // Get the newly created particle system
184  Nx::CParticle *p_system = Nx::CEngine::sGetParticleTable()->GetItem(PARTICLE_UNNAMED);
185  Dbg_MsgAssert(p_system,("Script did not create Particle_UnNamed!"));
186 
187  // Remove the entry from the Hash table
188  Nx::CEngine::sGetParticleTable()->FlushItem(PARTICLE_UNNAMED);
189 
190  // Add it back with the name as specifed in this node
191  Nx::CEngine::sGetParticleTable()->PutItem(NameChecksum,p_system);
192 
193  p_system->set_checkum(NameChecksum);
194  p_system->SetID(NameChecksum);
195 
196  Mth::Vector angles;
197  Mth::Vector dir (0.0f,1.0f,0.f);
198  pNodeData->GetVector("Angles",&angles);
199  printf("Angles(%f,%f,%f)\n",angles[X],angles[Y],angles[Z]);
200  // 3DSMAX_ANGLES Mick: 3/19/03 - Changed all rotation orders to X,Y,Z
201  dir.RotateX(angles[X]);
202  dir.RotateY(angles[Y]);
203  dir.RotateZ(angles[Z]);
204 
205  p_system->set_emit_target( dir[X], dir[Y], dir[Z] );
206 
207  // Register it with the object manager.....
208  // This means that the system can be refered to like a regular moving object
209  p_obj_man->RegisterObject(*p_system);
210 
211  p_system->MovingObjectInit( pNodeData, p_obj_man ); // get LOD and suspend info
212 
213  // Bastard crap code.... Since I set them up to be objects
214  // they now run their scripts with the new component system
215  // meaning the scripts run twice, dammit...
216  p_system->GetScript()->ClearScript();
217  p_system->GetScript()->ClearEventHandlerTable();
218  }
219 }
