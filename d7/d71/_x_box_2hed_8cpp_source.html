29  // the hed file all do.
30  if (pFilename[0]!='\\' && pFilename[0]!='/')
31  {
32  pBuf[0]='\\';
33  pBuf[1]=0;
34  }
35  else
36  {
37  pBuf[0]=0;
38  }
39  // Append the passed filename.
40  strcat(pBuf,pFilename);

42  // how they're listed in the hed.
43  char *pCh=pBuf;
44  while (*pCh)
45  {
46  if (*pCh=='/')
47  {
48  *pCh='\\';
49  }
50  ++pCh;
51  }
52 
53  // Search the hed until the file is found, or not.
54  SHed *pHd=pHed;
55  while (pHd->Offset!=0xffffffff)
56  {
57  if (stricmp(pBuf,pHd->pFilename)==0)
58  {
59  return pHd;
60  }
61

66  Len=(Len+3)&~3;
