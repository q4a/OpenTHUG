166 // The component's Update() function is called from the CCompositeObject's
167 // Update() function. That is called every game frame by the CCompositeObjectManager
168 // from the s_logic_code function that the CCompositeObjectManager registers
169 // with the task manger.
170 void CCollideAndDieComponent::Update()
171 {
172  GameNet::Manager* gamenet_man = GameNet::Manager::Instance();
173 
174  if (gamenet_man->InNetGame() || !mp_suspend_component->SkipLogic())
175  {
176  if(( Tmr::GetTime() - m_birth_time ) > vFIREBALL_LIFETIME )
177  {
178  GetObject()->MarkAsDead();
179  }
180  else
181  {
182  if( m_dying )
183  {
184  if( Tmr::GetTime() > m_death_time )
185  {
186  GetObject()->MarkAsDead();
187  }
188  }
189  else
190  {
191  Mth::Vector vel, start, end;
192  CFeeler feeler;
193 
194  vel = GetObject()->GetVel();
195  vel.Normalize();
196  if( m_first_frame )
197  {
198  start = GetObject()->GetPos();
199  }
200  else
201  {
202  start = m_last_pos;
203  }
204 
205  start[Y] += FEET( 2 );
206  end = GetObject()->GetPos();
207  end[Y] += FEET( 2 );
208  end += ( vel * m_radius );
209  feeler.SetStart( start );
210  feeler.SetEnd( end );
211 
212  if( feeler.GetCollision())
213  {
214  if( m_dying == false )
215  {
216  CProjectileCollisionComponent* proj_comp;
217 
218  proj_comp = GetProjectileCollisionComponentFromObject( GetObject());
219  if( proj_comp )
220  {
221  proj_comp->MarkAsDying();
222  }
223 
224  m_dying = true;
225  m_death_time = Tmr::GetTime() + Tmr::Seconds( 1 );
226  if( m_death_script != 0 )
227  {
228  Script::CStruct* params;
229  Mth::Vector pos;
230 
231  pos = feeler.GetPoint();
232  params = new Script::CStruct;
233 
234  params->AddVector( CRCD(0x7f261953,"pos"), pos );
235  params->AddVector( CRCD(0xc4c809e,"vel"), m_vel );
236  params->AddFloat( CRCD(0x13b9da7b,"scale"), m_scale );
237 
238  Script::RunScript( m_death_script, params );
239 
240  delete params;
241  }
242  }
243  }
244  }
245  }
246 
247  m_last_pos = GetObject()->GetPos();
248  m_first_frame = false;
249  }
250 }
