374  // Delay to avoid hogging the processor

477 bool Manager::initialize_device( void )
478 {
479  sn_int32 result;
480  sn_int32 device_type;
481  sn_int16 idVendor;
482  sn_int16 idProduct;
483  sn_bool first_time;
484  Tmr::Time start_time;
485  Tmr::Time timeout;
486 
487  // Initialise the socket API, if fails print error and return
488  Dbg_Printf( "EE:Initialising socket API\n" );
489 
490  result = sockAPIinit( 6 );
491  if( ( result != 0 ) &&
492  ( result != SN_EALRDYINIT ))
493  {
494  Dbg_Printf( "EE:sockAPIinit() failed %d\n", result );
495  SetError( vRES_ERROR_GENERAL );
496  return false;
497  }
498 
499  // If we've already init'd the sockets API, don't re-register this thread
500  // just let it run through the rest of setup
501  if( result != SN_EALRDYINIT )
502  {
503  // Register this thread with the socket API
504  result = sockAPIregthr();
505 
506  if (result != 0)
507  {
508  Dbg_Printf( "EE:sockAPIregthr() failed %d\n", result );
509  SetError( vRES_ERROR_GENERAL );
510  return false;
511  }
512  }
513 
514  // Now wait for DECI2 'device' to be 'attached'
515  device_type = SN_DEV_TYPE_NONE;
516  first_time = SN_TRUE;
517  start_time = Tmr::GetTime();
518  // It takes longer to init the sony modem
519  if( GetDeviceType() == vDEV_TYPE_SONY_MODEM )
520  {
521  timeout = Tmr::Seconds( 15 );
522  }
523  else
524  {
525  timeout = Tmr::Seconds( 10 );
526  }
527 
528  while( device_type == SN_DEV_TYPE_NONE )
529  {
530  result = sndev_get_attached( 0, &device_type, &idVendor, &idProduct );
531 
532  if( result != 0 )
533  {
534  Dbg_Printf( "EE:sndev_get_attached() failed %d\n", result );
535  SetError( vRES_ERROR_GENERAL );
536  return false;
537  }
538 
539  if( device_type == SN_DEV_TYPE_NONE )
540  {
541  if( first_time == SN_TRUE )
542  {
543  first_time = SN_FALSE;
544  Dbg_Printf( "EE:Waiting for network device to be attached ...\n" );
545  }
546  sn_delay( 10 );
547  }
548 
549  if(( Tmr::GetTime() - start_time ) > timeout )
550  {
551  Dbg_Printf( "EE:Timed out waiting for network device to be attached\n" );
552  break;
553  }
554  }
555 
556  switch( device_type )
557  {
558  case SN_DEV_TYPE_DECI2:
559  Dbg_Printf( "Using DECI-2 ethernet emulation\n" );
560  break;
561  case SN_DEV_TYPE_USB_MODEM:
562  Dbg_Printf("EE:USB-Modem Attached (idVendor=0x%04X idProduct=0x%04X)\n",
563  ((int)idVendor) & 0xFFFF,
564  ((int)idProduct) & 0xFFFF);
565  break;
566  case SN_DEV_TYPE_USB_ETHER:
567  Dbg_Printf("EE:USB-Ethernet Attached (idVendor=0x%04X idProduct=0x%04X)\n",
568  ((int)idVendor) & 0xFFFF,
569  ((int)idProduct) & 0xFFFF);
570  break;
571  default:
572  {
573  if( idVendor == 0 )
574  {
575  Dbg_Printf( "Unknown Device %d\n", device_type );
576  SetError( vRES_ERROR_UNKNOWN_DEVICE );
577  }
578  else
579  {
580  Dbg_Assert( idVendor == 1 );
581  SetError( vRES_ERROR_DEVICE_NOT_HOT );
582  }
583 
584  return false;
585  }
586  }
587 
588  return true;
589 }

771  if( result != 0 )
772  {
773  Dbg_Printf( "snmdm_set_script() failed: %d\n", result );
774  return SNTC_ERR_MDMAPI;
775  }
776 
777  // Check for null line, which is last line
778  if( custom_built_script[script_index][0] == 0 )
779  {
780  done_script = SN_TRUE;
781  }
782  else
783  {
784  script_index++;
785  }
786  }
787 
788  // Everything is ready, so ask the modem to connect
789  result = snmdm_connect();
790  if( result != 0 )
791  {
792  Dbg_Printf( "snmdm_connect() failed: %d\n", result );
793  return SNTC_ERR_MDMAPI;
794  }
795 
796  // Now wait for the modem to become connected
797  modem_state = -1; /* Invalid */
798  prev_modem_state = -2; /* Ivalid and != modem_state */
799  connect_started = SN_FALSE;
800  timeout_ms = timeout_secs * 1000;
801 
802  while(( modem_state != SN_MODEM_PPP_UP ) &&
803  ( s_cancel_dialup_conn == false ))
804  {
805 
806  // Get the current state of the modem link
807  result = snmdm_get_state(&modem_state);
808  if( result != 0 )
809  {
810  Dbg_Printf( "snmdm_get_state() failed: %d\n", result );
811  return SNTC_ERR_MDMAPI;
812  }
813 
814  // Monitor for modem connection process starting, then if it
815  // goes back to ready we know it's failed to connect.
816  if( connect_started == SN_FALSE )
817  {
818  // Any of the following means modem connection started
819  if( (modem_state == SN_MODEM_DIALING)
820  ||(modem_state == SN_MODEM_LOGIN)
821  ||(modem_state == SN_MODEM_CONNECTED)
822  ||(modem_state == SN_MODEM_PPP_UP))
823  {
824  connect_started = SN_TRUE;
825  }
826  }
827  else
828  {
829  // Having started the connection, then unless it's in
830  // one of the following states it's failed to connect
831  if( (modem_state != SN_MODEM_DIALING)
832  &&(modem_state != SN_MODEM_LOGIN)
833  &&(modem_state != SN_MODEM_CONNECTED)
834  &&(modem_state != SN_MODEM_PPP_UP))
835  {
836  // Read the reason why the connect failed.
837  result = snmdm_get_connect_err( &connect_err );
838  if( result != 0 )
839  {
840  Dbg_Printf( "snmdm_get_connect_err() failed: %d\n", result );
841  return SNTC_ERR_MDMAPI;
842  }
843 
844  switch( connect_err )
845  {
846  case SN_CONERR_BUSY:
847  Dbg_Printf( "Busy\n" );
848  return SNTC_ERR_BUSY;
849  break;
850 
851  case SN_CONERR_NOCARRIER:
852  Dbg_Printf( "No carrier\n" );
853  return SNTC_ERR_NOCARRIER;
854  break;
855 
856  case SN_CONERR_NOANSWER:
857  Dbg_Printf( "No answer\n" );
858  return SNTC_ERR_NOANSWER;
859  break;
860 
861  case SN_CONERR_NODIALTONE:
862  Dbg_Printf( "No dialtone\n" );
863  return SNTC_ERR_NODIALTONE;
864  break;
865 
866  default:
867  Dbg_Printf( "connect started then modem state=sntc_str_modem_state(modem_state)\n" );
868  return SNTC_ERR_CONNECT;
869  }
870  }
871  }
872 
873  // Now check whether the modem state has changed since the
874  // previous time round this loop, and if so call the user
875  // callback function (unless it's NULL), otherwise check
876  // for time-out / do a delay
877  if( modem_state != prev_modem_state )
878  {
879  prev_modem_state = modem_state;
880  if( callback != NULL )
881  {
882  (*callback)(modem_state);
883  }
884  }
885  else
886  {
887  if( timeout_ms <= 0 )
888  {
889  Dbg_Printf( "Connect timed out in modem state %s\n", sntc_str_modem_state(modem_state) );
890  return SNTC_ERR_TIMEOUT;
891  }
892 
893  sn_delay( 10 );
894  timeout_ms -= 10;
895  }
896  }
897 
898  // If we get to here, the modem has successfully connected
899  Dbg_Printf( "Connected!\n" );
900 
901  return 0;
902 }
