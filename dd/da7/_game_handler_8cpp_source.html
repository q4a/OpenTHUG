116 int Manager::s_handle_new_player( Net::MsgHandlerContext* context )
117 {
118  MsgNewPlayer* msg;
119  Manager* gamenet_man;
120 
121  msg = (MsgNewPlayer *) context->m_Msg;
122  gamenet_man = (Manager *) context->m_Data;
123 
124  NewPlayerInfo new_player;
125 
126  strcpy( new_player.Name, msg->m_Name );
127  new_player.ObjID = msg->m_ObjId;
128  new_player.Conn = NULL;
129  new_player.Flags = msg->m_Flags;
130  new_player.Team = msg->m_Team;
131  new_player.Profile = msg->m_Profile;
132  new_player.Rating = msg->m_Rating;
133  new_player.Score = msg->m_Score;
134  new_player.VehicleControlType = msg->m_VehicleControlType;
135 
136  if( new_player.Flags & PlayerInfo::mLOCAL_PLAYER )
137  {
138  Dbg_Message( "(%d) Got local new player with ID %d!\n", context->m_App->m_FrameCounter,
139  msg->m_ObjId );
140  }
141  else
142  {
143  Dbg_Message( "(%d) Got new player with ID %d!\n", context->m_App->m_FrameCounter,
144  msg->m_ObjId );
145  }
146 
147  // Observers and local players do not receive their own appearance data
148  if( !( new_player.Flags & ( PlayerInfo::mOBSERVER | PlayerInfo::mLOCAL_PLAYER )))
149  {
150  // GJ: the appearance is transmitted inside the message,
151  // so copy it to the "new_player" structure
152  new_player.mpSkaterProfile->ReadFromBuffer( msg->m_AppearanceData );
153  }
154 
155  // If this is all part of our join packet, just load players immediately instead of deferring it
156  if( gamenet_man->ReadyToPlay() == false )
157  {
158  Dbg_Printf( "Calling DeferredNewPlayer with no wait\n" );
159  gamenet_man->DeferredNewPlayer( &new_player, 0 );
160  }
161  else
162  {
163  gamenet_man->CreateNetPanelMessage( false, Script::GenerateCRC("net_message_new_player"),
164  new_player.Name );
165 
166  gamenet_man->DeferredNewPlayer( &new_player );
167  }
168 
169  return Net::HANDLER_CONTINUE;
170 }
