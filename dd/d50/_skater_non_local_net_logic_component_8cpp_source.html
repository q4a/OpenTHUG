272 void CSkaterNonLocalNetLogicComponent::snap_to_ground( void )
273 {
274  float up_dot = 0.0f;
275  Mth::Vector col_start, col_end;
276  CFeeler feeler;
277 
278  // Since we really don't want to loose contact with the ground while skitching, we use a much bigger snap up dist
279  // The problem will come when we get dragged down a slope. The car will flatten out well ahead of us, so pushing us down through the slop
280  // (as we are a few feet behind it) and we will be so far under the ground that our normal snap up will not be able to dig us out of it,
281  // so we go in air, uberfrig, and get dragged to a random spot under the level.
282  // (This would not happen if we just skitch on flat ground)
283  // if (mp_state_component->m_skater_flags[SKITCHING].Get())
284  // {
285  // col_start = GetObject()->m_matrix[Y] * GetPhysicsFloat(CRCD(0x5c0d9610,"Physics_Ground_Snap_Up_SKITCHING")); // much above feet
286  // }
287  // else
288  // {
289  col_start = GetObject()->m_matrix[Y] * GetPhysicsFloat(CRCD(0xe4d79235, "Physics_Ground_Snap_Up")); // bit above the feet
290  // }
291 
292  col_end = GetObject()->m_matrix[Y] * -200.0f; // WAY! below the feet, we check distance later
293 
294  col_start += GetObject()->m_pos;
295  col_end += GetObject()->m_pos;
296 
297  bool sticking = false;
298 
299  feeler.SetLine( col_start, col_end );
300  feeler.SetIgnore( mFD_NON_COLLIDABLE, 0 );
301  // get disatnce to ground and snap the skater to it, but only if the ground is skatable, otherwise we just go to "AIR"
302  if( feeler.GetCollision())
303  {
304  Mth::Vector movement = GetObject()->m_pos - feeler.GetPoint();
305  uint16 flags = feeler.GetFlags();
306  float drop_dist = movement.Length();
307  float drop_sign = Mth::DotProduct(GetObject()->m_matrix[Y], movement); // might be approx +/- 0.00001f
308  if( (!flags & mFD_SKATABLE ) || ( flags & mFD_NOT_SKATABLE ) || (!flags & mFD_VERT) ||
309  (flags & mFD_WALL_RIDABLE) ||
310  (feeler.GetNormal()[Y] < sinf(Mth::DegToRad(GetPhysicsFloat(CRCD(0x3eede4d3,"Wall_Non_Skatable_Angle"))))))
311  {
312  // if below the face (or very close to it), then push us away from it
313  if (drop_sign < 0.001f)
314  {
315  // at point of contact, and move away from surface
316  GetObject()->m_pos = feeler.GetPoint() + feeler.GetNormal();
317  }
318  }
319  else
320  {
321  sticking = true;
322 
323  // Note the two ways of calculating the angle between two faces
324  // the more "accurate" method simply takes angle between the two normals
325  // however, this fails to account for the direction the skater is travelling
326  // when in conjunction with the "snap" up.
327  // If the skater approaches a slope from the side, then he can snap up to the slope
328  // however, if the angle between the ground and the face to which we are snapping up to
329  // is too great, then we transition to in-air
330  // and will drop through the slope
331  // the solution is to take the angle between the front vector rotated onto each face.
332 
333  // Firstly we check the angle between the two faces
334  Mth::Vector normal = feeler.GetNormal();
335 
336  Mth::Vector forward = GetObject()->m_matrix[Z];
337  float front_dot = Mth::DotProduct(forward,normal);
338 
339  Mth::Vector old_forward = forward;
340  forward.RotateToPlane(normal);
341  old_forward.RotateToPlane(m_current_normal);
342 
343  // angle between front vectors, projected onto faces
344  up_dot = Mth::DotProduct(forward, old_forward);
345 
346  float stick_angle_cosine;
347  stick_angle_cosine = cosf(Mth::DegToRad(GetPhysicsFloat(CRCD(0x4138283e, "Ground_stick_angle_forward"))));
348 
349  if (front_dot > 0.0f && up_dot > 0.0f && up_dot < stick_angle_cosine)
350  {
351  sticking = false;
352  }
353  else
354  {
355  // only need to test if snap is downwards
356  if (drop_sign > 0.0f)
357  {
358  // angle between the old plane and the new plane is either the same, or withing the set limits
359  // (like, < 60 degrees, or so, see physics.q) now calculate the drop distance, based on the angle
360 
361  //float max_drop = 60.0f;////last_move.Length() * tanf(angle);
362 #ifdef __PLAT_NGC__
363  float angle = acosf(Mth::Clamp(up_dot, -1.0f, 1.0f));
364 #else
365  float angle = acosf(up_dot);
366 #endif // __PLAT_NGC__
367 
368  Mth::Vector last_move = GetObject()->m_pos - m_old_extrap_pos;
369 
370  float max_drop = last_move.Length() * tanf(angle);
371 
372  float min_drop = GetPhysicsFloat(CRCD(0x899ba3d0, "Physics_Ground_Snap_Down"));
373  // if (mp_state_component->m_skater_flags[SKITCHING].Get())
374  // {
375  // min_drop = GetPhysicsFloat(CRCD(0x20df7e33, "Physics_Ground_Snap_Down_Skitching"));
376  // }
377 
378  if (max_drop < min_drop)
379  {
380  max_drop = min_drop;
381  }
382 
383  if (drop_dist > max_drop)
384  {
385  sticking = false;
386  }
387  }
388 
389  if (sticking)
390  {
391  //new_normal(normal);
392  m_current_normal = normal; // remember this, for detecting if it changes
393 
394  GetObject()->m_matrix[Y] = normal;
395  GetObject()->m_matrix.OrthoNormalizeAbout(Y); // set regular normal immediately
396  }
397  }
398  }
399 
400  if (sticking)
401  {
402  // if there is a collision, then snap to it
403  GetObject()->m_pos = feeler.GetPoint();
404  }
405  }
406 }

413 CRailNode* CSkaterNonLocalNetLogicComponent::travel_on_rail( CRailNode* start_node, float frame_length )
414 {
415  bool on_rail;
416 
417  Dbg_Assert( start_node );
418 
419  on_rail = true;
420 
421  CRailManager* p_rail_man = Mdl::Skate::Instance()->GetRailManager();
422 
423  const CRailNode* pStart = start_node;
424  const CRailNode* pEnd = pStart->GetNextLink();
425  if( pEnd == NULL )
426  {
427  Dbg_MsgAssert(pEnd,("NonLocal Skater on Rail node (%d) with no next link\n",start_node->GetNode()));
428  return NULL;
429  }
430 
431  //const CRailNode* pFrom = pStart;
432  const CRailNode* pOnto = NULL;
433 
434  Mth::Vector dir = p_rail_man->GetPos(pEnd) - p_rail_man->GetPos(pStart);
435  float segment_length = dir.Length();
436  dir *= (1.0f / segment_length);
437 
438  // sign is which way we are going along the rail
439  //float old_sign = Mth::Sgn(Mth::DotProduct(dir, GetObject()->m_vel));
440 
441  // Get gravity force
442  Mth::Vector gravity(0.0f, GetPhysicsFloat(CRCD(0xd1f46992, "Physics_Rail_Gravity")), 0.0f);
443 
444  // Project gravity onto the line we are on
445  gravity.ProjectToNormal(dir);
446  GetObject()->m_vel += gravity * frame_length;
447 
448  // sign is which way we are going along the rail
449  float sign = Mth::Sgn(Mth::DotProduct(dir, GetObject()->m_vel));
450 
451  // check to see if we are on the last segment of the rail
452  // this logic could be folded into the logic below but it's perhps clearer to have it here
453  bool last_segment = false;
454  if (sign < 0.0f)
455  {
456  if (pStart->GetPrevLink() && pStart->GetPrevLink()->IsActive())
457  {
458  Mth::Vector v1, v2;
459  v1 = p_rail_man->GetPos(pEnd) - p_rail_man->GetPos(pStart);
460  v2 = p_rail_man->GetPos(pStart) - p_rail_man->GetPos(pStart->GetPrevLink());
461  v1[Y] = 0.0f;
462  v2[Y] = 0.0f;
463  v1.Normalize();
464  v2.Normalize();
465  float dot = Mth::DotProduct(v1, v2);
466  if (dot < cosf(Mth::DegToRad(GetPhysicsFloat(CRCD(0x76c1da15, "Rail_Corner_Leave_Angle")))))
467  {
468  // there is more, but angle is too sharp
469  last_segment = true;
470  }
471  else
472  {
473  pOnto = pStart->GetPrevLink();
474  }
475  }
476  else
477  {
478  last_segment = true;
479  }
480  }
481  else
482  {
483  if (pEnd && pEnd->GetNextLink() && pEnd->GetNextLink()->IsActive())
484  {
485  Mth::Vector v1,v2;
486  v1 = p_rail_man->GetPos(pStart) - p_rail_man->GetPos(pEnd);
487  v2 = p_rail_man->GetPos(pEnd) - p_rail_man->GetPos(pEnd->GetNextLink());
488  v1[Y] = 0.0f;
489  v2[Y] = 0.0f;
490  v1.Normalize();
491  v2.Normalize();
492  float dot = Mth::DotProduct(v1,v2);
493  if (dot < cosf(Mth::DegToRad(GetPhysicsFloat(CRCD(0x76c1da15,"Rail_Corner_Leave_Angle")))))

508

510

512

515  {

518  }
519  else

521  // going forwards, so it's the distance from the start

523  }
524 
525  if (length_along > segment_length + 0.1f) // 0.1 inch, so we don't get stuck

527  // remember this, so we can move along next segment
528  extra_dist = length_along - segment_length;
529

547  //skate_off_rail(p_rail_man->GetPos(pStart));

560  && pEnd->IsActive()
561  && Rail_ValidInEditor(p_rail_man->GetPos(pEnd), p_rail_man->GetPos(pEnd->GetNextLink())))
562  {
563  if (!last_segment)

565  GetObject()->m_pos = p_rail_man->GetPos(pEnd);

567  //set_terrain(start_node->GetTerrain());

850 void CSkaterNonLocalNetLogicComponent::bounce_off_wall ( const Mth::Vector& normal, const Mth::Vector& point )
851 {
852  // Given the normal of the wall, then bounce off it, turning the skater away from the wall
853  Mth::Vector col_start, col_end;

861 
862  float turn_angle;
863  float angle = rotate_away_from_wall(normal, turn_angle);
864 
865  float min = Mth::DegToRad(GetPhysicsFloat(CRCD(0x1483fd01, "Wall_Bounce_Dont_Slow_Angle")));
866 
867  if (Mth::Abs(angle) > min)
868  {
869  //float old_speed = GetObject()->m_vel.Length();
870 
871  // The maximum value of Abs(angle) would be PI/2 (90 degrees), so scale the velocity
872  float x = Mth::Abs(angle) - min;
873  x /= (Mth::PI / 2.0f) - min; // get in the range 0 .. 1
874  x = 1.0f - x; // invert, as we want to stop when angle is 90
875 
876  GetObject()->m_vel *= x;
877  }
878 
879  // Bit of a patch here to move the skater away from the wall
880  // Not needed so much with new sideways collision checks but we keep it in for low ledges.
881  // Should perhaps standardize the height so collision checks for side and front but we'd probably still have problems.
882 
883  GetObject()->m_pos = point - up_offset;
884  GetObject()->m_pos += normal * 6.0f;
885 
886  // Now the majority of cases have been taken care of; we need to see if the skater is going to get stuck in a corner.
887 
888  float old_speed = movement.Length(); // get how much we moved last time
889  Mth::Vector next_movement = GetObject()->m_vel; // get new direction of velocity
890  forward = GetObject()->m_vel;
891  forward.Normalize();
892  next_movement = forward * old_speed; // extend by same movment as last time
893 
894  col_start = GetObject()->m_pos + up_offset;
895  col_start += up_offset;
896

898  + forward * GetPhysicsFloat(CRCD(0x20102726, "Skater_First_Forward_Collision_Length"));

903  return;
904  }
905

908  if( feeler.GetCollision() && GetSkater()->IsLocalClient())
909  {
910  // Just rotating the skater will lead to another collision, so try just inverting the skater's velocity from it's original and halving it....
911 
912  // First reverse the rotation, and rotate 180 degrees
913  GetObject()->m_vel.RotateY(Mth::DegToRad(180.0f) - turn_angle);
914  GetObject()->m_matrix.RotateYLocal(Mth::DegToRad(180.0f) - turn_angle);
915  //ResetLerpingMatrix();
916 
917  GetObject()->m_vel *= 0.5f;
918  }
