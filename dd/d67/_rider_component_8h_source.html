378 inline void CRiderComponent::set_state ( EStateType state )
379 {
380  if (state == m_state) return;
381 
382  m_previous_state = m_state;
383 
384  if (m_state == WALKING_GROUND)
385  {
386  m_wall_push_test.active = false;
387  m_run_toggle = false;
388  }
389 
390  m_state = state;
391  m_state_timestamp = Tmr::GetTime();
392 
393  if (m_state == WALKING_ANIMWAIT)
394  {
395  m_offset_due_to_movable_contact.Set();
396  }
397 }
