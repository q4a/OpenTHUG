135  // so we need to establish next and prev links
136  // or if they are established, then possibly override them
137  if (!p_from->m_pNextLink)
138  {
139  // this is the first time we are linking to something
140  p_from->m_pNextLink = p_to;
141  Rail_ComputeBB(p_from->m_pos, p_to->m_pos, p_from->m_BBMin, p_from->m_BBMax);
142  }
143  else
144  {
145 
146  // already linked to something, so only override it
147  // if the new node is not DEFAULT_LINE
148  if (p_to->GetFlag(DEFAULT_LINE))
149  {
150  // This new link is DEFAULT_LINE
151  // so, as long as the previous link was not, we can override it
152  Dbg_MsgAssert(!p_from->m_pNextLink->GetFlag(DEFAULT_LINE),
153  ("Node %d links to both %d and %d and both are DEFAULT_LINE",
154  p_from->m_node,p_from->m_pNextLink->m_node,p_to->m_node));
155  // Overriding an existing NEXT link
156  p_from->m_pNextLink = p_to;
157  Rail_ComputeBB(p_from->m_pos, p_to->m_pos, p_from->m_BBMin, p_from->m_BBMax);
158  }
159  else
160  {
161  // The new link is not DEFAULT_LINE
162  // if the old one was, then leave it
163  // if neither is, then leave it, and maybe print a warning?
164 
165  }
166  }
167 
168  // Now handle the PREV link back from p_to to p_from
169 
170  if (!p_to->m_pPrevLink)
171  {
172  p_to->m_pPrevLink = p_from;
173  }
174  else
175  {
176 
177  // something already linked to p_from via prev from p_to

312 // required to totally encompass the 1D line segments a-b and c-d
313 static inline float span_4(float a, float b, float c, float d)
314 {
315  float min = a;
316  if (b < min) min = b;
317  if (c < min) min = c;
318  if (d < min) min = d;
319  float max = a;
320  if (b > max) max = b;
321  if (c > max) max = c;
322  if (d > max) max = d;
323  return max-min;

588 // Given a velocity, and a rail node
589 // then see which side of the rail (left or right)
590 // has no collision
591 // returns -1 if left, 0 if neither or both, and +1 if right
592 // Used in the park editor to give preference to those
593 // rails that keep the same sideness as the prior rail we were on
594 int CRailNode::Side(const Mth::Vector &vel) const
595 {
596  Mth::Vector Start = m_pos;
597  Mth::Vector End = m_pNextLink->m_pos;
598 
599  CFeeler feeler;
600 
601  // Find a point "Mid", which is 1/4 of the way along the rail
602  // and lowered six inches
603  Mth::Vector Mid = (Start + End) / 2.0f; // Half way along
604  Mid = Start + ((Mid - Start) / 2.0f); // quarter of the way along
605  Mid[Y] -= 6.0f; // lowered a bit
606 
607 // Find a vector "Side" which is horizontal, perpendicular to the rail, and 4 inches long
608  Mth::Vector Side = End - Start; // vector along the rail
609  Side[Y] = 0.0f; // horizontal
610  Side.Normalize(4.0f); // this is half the width of the thickest rail. Was 4.0, changed for Tokyo mega funbox
611  float temp = Side[X]; // make perpendicular to rail
612  Side[X] = Side[Z];
613  Side[Z] = -temp;
614 
615 
616  bool left = false;
617  bool right = false;
618 
619 // if collision above me, left to right
620  feeler.SetLine(Mid + Side, Mid-Side);
621  if (feeler.GetCollision())
622  {
623  left = true;
624  }
625 
626 // if collision above me, right to left
627  feeler.FlipDirection();
628  if (feeler.GetCollision())
629  {
630  right = true;
631  }
632  else
633  {
634  }
635 
636  int side = 0;
637  if (left && !right)
638  {
639  side = -1;
640  }
641  else
642  {
643  if (right && ! left)
644  {
645  side = 1;
646  }
