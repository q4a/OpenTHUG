238 int
239 TextureMemoryLayout::getBlockOffset(int bufferWidth, int posX, int posY, int pixMode)
240 {
241  int horizPages, vertPages;
242  int numPages = 0;
243  int blockNumber = 0;
244 
245  PixelType pType = modeToPixelType(pixMode);
246  const PixelFormatLayout &pf = pixelFormat[pType];
247 
248  // First figure out how many pages are above
249  vertPages = posY / pf.pageSize.height;
250  if (vertPages) {
251  horizPages = bufferWidth / pf.pageSize.width;
252  posY -= vertPages * pf.pageSize.height;
253  numPages += vertPages * horizPages;
254  }
255 
256  // Now how many are to the left
257  horizPages = posX / pf.pageSize.width;
258  if (horizPages) {
259  posX -= horizPages * pf.pageSize.width;
260  numPages += horizPages;
261  }
262 
263  // Now figure out where we are within the page
264  int horizBlocks, vertBlocks;
265  horizBlocks = posX / pf.blockSize.width;
266  vertBlocks = posY / pf.blockSize.height;
267  blockNumber = getBlockNumber(horizBlocks, vertBlocks, pixMode);
268 
269  return (numPages * 32) + blockNumber;
270 }
