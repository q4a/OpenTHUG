152 sTexture *LoadTexture( const char *p_filename )
153 {
154  struct sIMGHeader
155  {
156  uint32 version;
157  uint32 checksum;
158  uint32 width;
159  uint32 height;
160  uint32 bit_depth;
161  uint32 clut_bit_depth;
162  uint16 original_width;
163  uint16 original_height;
164  uint32 palette_data_size;
165  };
166 
167  void *p_FH = File::Open( p_filename, "rb" );
168 
169  if( p_FH )
170  {
171  // Read header.
172  sIMGHeader header;
173  File::Read( &header, sizeof( sIMGHeader ), 1, p_FH );
174 
175  // Bits per texel and palette size.
176  switch( header.bit_depth )
177  {
178  case PSMCT32:
179  header.bit_depth = 32;
180  break;
181  case PSMCT16:
182  header.bit_depth = 16;
183  break;
184  case PSMT8:
185  header.bit_depth = 8;
186  break;
187  default:
188  Dbg_Assert( 0 );
189  }
190 
191  // Bits per clut entry.
192  if( header.bit_depth < 16 )
193  {
194  switch( header.clut_bit_depth )
195  {
196  case PSMCT32:
197  header.clut_bit_depth = 32;
198  break;
199  default:
200  Dbg_Assert( 0 );
201  }
202  }
203  else
204  {
205  header.clut_bit_depth = 0;
206  }
207 
208  {
209  // Create the texture object.
210  sTexture *p_texture = new sTexture();
211 
212  // Create palette if required.
213  if( header.clut_bit_depth == 0 )
214  {
215  p_texture->pD3DPalette = NULL;
216  }
217  else
218  {
219  if( D3D_OK != D3DDevice_CreatePalette( D3DPALETTE_256, &p_texture->pD3DPalette ))
220  {
221  Dbg_Assert( 0 );
222  }
223 
224  // Read clut bitmap data.
225  D3DCOLOR *p_clut;
226  p_texture->pD3DPalette->Lock( &p_clut, 0 );
227 
228  int len = File::Read( p_clut, header.palette_data_size, 1, p_FH );
229  Dbg_MsgAssert( len == header.palette_data_size, ( "Couldn't read clut from texture file %s", p_filename ));
230  }
231 
232  // Textures of width 512 and above will not have been resized. This means they cannot be in a swizzled format.
233 // bool arbitrary_texture_size = ( header.original_width >= 512 );
234  bool arbitrary_texture_size = false;
235 
236  if( !is_power_of_two( header.width ) || !is_power_of_two( header.height ))
237  arbitrary_texture_size = true;
238 
239  // Create texture resource. Linear for arbitrary sized textures, swizzled for standard sizes.
240  uint32 num_bytes;
241  if( arbitrary_texture_size )
242  {
243  if( D3D_OK != D3DDevice_CreateTexture( header.original_width,
244  header.original_height,
245  1,
246  0,
247  ( header.bit_depth <= 8 ) ? D3DFMT_P8 : (( header.bit_depth == 16 ) ? D3DFMT_LIN_A1R5G5B5 : D3DFMT_LIN_A8R8G8B8 ),
248  0,
249  &p_texture->pD3DTexture ))
250  {
251  Dbg_Assert( 0 );
252  }
253  p_texture->BaseWidth = header.original_width;
254  p_texture->BaseHeight = header.original_height;
255  p_texture->ActualWidth = header.original_width;
256  p_texture->ActualHeight = header.original_height;
257  num_bytes = ((( header.bit_depth / 8 ) * ( header.original_width ) * ( header.original_height )) + 3 ) & 0xFFFFFFFC;
258  }
259  else
260  {
261  if( D3D_OK != D3DDevice_CreateTexture( header.width,
262  header.height,
263  1,
264  0,
265  ( header.bit_depth <= 8 ) ? D3DFMT_P8 : (( header.bit_depth == 16 ) ? D3DFMT_A1R5G5B5 : D3DFMT_A8R8G8B8 ),
266  0,
267  &p_texture->pD3DTexture ))
268  {
269  Dbg_Assert( 0 );
270  }
271  p_texture->BaseWidth = (uint16)header.width;
272  p_texture->BaseHeight = (uint16)header.height;
273  p_texture->ActualWidth = header.original_width;
274  p_texture->ActualHeight = header.original_height;
275  num_bytes = ((( header.bit_depth / 8 ) * header.width * header.height ) + 3 ) & 0xFFFFFFFC;
276  }
277 
278  // Lock the texture so we can read data into it directly.
279  D3DLOCKED_RECT locked_rect;
280  if( D3D_OK != p_texture->pD3DTexture->LockRect( 0, &locked_rect, NULL, 0 ))
281  {
282  Dbg_Assert( 0 );
283  }
284 
285  // Read texture bitmap data directly into texture.
286  int len = File::Read( locked_rect.pBits, num_bytes, 1, p_FH );
287  Dbg_MsgAssert( len == num_bytes, ( "couldn't read texture data from texture file %s", p_filename ));
288 
289  File::Close( p_FH );
290 
291  // Set up some member values.
292  p_texture->PaletteDepth = (uint8)header.clut_bit_depth;
293  p_texture->TexelDepth = (uint8)header.bit_depth;
294  p_texture->DXT = 0;
295  p_texture->Levels = 1;
296 
297  return p_texture;
298  }
299  }
300  return NULL;
301 }
