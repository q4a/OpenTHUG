4 // Functions for CGap and CGapMan
5 //
6 // A "Gap" in the game is a pair of events that cause something to happen
7 // Typically this is jumping across something, such as a gap between two buildings
8 // but might also be grinding a rail for a certain distance
9 // or doing something specific, like doing a kickflip over a barrel.
10 //
11 // The aim of the gap system is to provide lots of flexibility in setting up
12 // these pairs of events, and also to allow the user to set gaps quickly and simply
13 //
14 // Gaps are set up via script commands
15 // so they can easily be inserted in trigger scripts
16 // the simplest might look like:
17 /*
18 
19 script TRG_Obj010
20  StartGap GapId = JumpOverLog Cancel = [CANCEL_GROUND CANCEL_RAIL CANCEL_WALLRIDE] Expire = 20
21 endscript
22 
23 
24 script TRG_Obj012
25  EndGap GapId = JumpOverLog Score = 100 Text = "You jumped over a log" Script = ScreenFlash01
26 endscript
27 
28 Using the "StartGap" command, you can set up as many gaps as you like, starting from
29 a particular event.
30 
31 For that gap, you specify what cancels the gap, if nothing is specified, then this should default
32 to CANCEL_GROUND. These conditions are checked every frame
33 
34 All gaps are cancelled by a bail (perhaps unless we have CANCEL_BAIL_NOT set????)
35 
36 We can also have a specific "CancelGap" command, for fuller control.
37 
38 The optional "Expire" parameter says when the gap expires. By default gaps do not expire
39 
40 The "EndGap" command will check to see if the gap with the specified GapID had been started
41 (and not cancelled, or expired), and will then Execute the gap, having an optional Score,
42 text (defaults to "Gap") and an optional script, which is executed.
43 
44 If there is no score, then it will not count as a trick, and you just get the script executing.
45 
46 
47

211 // Given a name and a count, then add a GapCheck to the list
212 // if it is already in the list, then just set it valid (leaving the count)
213 // (that is so when we re-scan the level's gaps after loading a level, we can remove any
214 // that do not exist in the level any more)
215 void CGapChecklist::AddGapCheck(const char *p_name, int count, int score)
216 {
217 
218  m_valid = true; // the list becomes valid when at least one gapcheck is in it
219 
220  CGapCheck *pOther = (CGapCheck*)m_gapcheck_list.FirstItem();
221  CGapCheck *pAppend = NULL; // last item we find that is lower than this
222  while (pOther)
223  {
224  if (!strcmp(pOther->GetText(),p_name) && pOther->GetScore() == score) // if it matches an existing gap in both text and score
225  {
226  pOther->SetValid(true); // then set that one valid (as we might be pruning later)
227  return; // then break out
228  }
229 
230  if (pOther->GetScore() < score)
231  {
232  pAppend = pOther;
233  }
234  else
235  {
236  if (pOther->GetScore() == score && Str::Compare(pOther->GetText(),p_name)<0)
237  {
238  pAppend = pOther;
239  }
240  }
241 
242  pOther = (CGapCheck*)pOther->GetNext();
243  }
244 
245  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().ScriptHeap());
246 
247  CGapCheck* pGapCheck = new CGapCheck; // create new gap
248  pGapCheck->UnGetGap(); // say not got it yet
249 
250  pGapCheck->SetText(p_name); // set the text for it
251  pGapCheck->SetCount(count); // and the count
252  pGapCheck->SetScore(score); // and the count
253  if (pAppend)
254  {
255  // Append after the last node that is smaller or equal to than this
256  pAppend->Append( pGapCheck );
257  }
258  else
259  {
260  // add to the ehad of the list if it's smaller than everything
261  m_gapcheck_list.AddToHead( pGapCheck );
262  }
263 
264  Mem::Manager::sHandle().PopContext();
265 }

385 void CGapChecklist::AwardPendingGaps()
386 {
387  int gap_index=0;
388  CGapCheck *pOther = (CGapCheck*) m_gapcheck_list.FirstItem();
389  char ran_got_gap_for_first_time = 0;
390 
391  while (pOther)
392  {
393  if(pOther->GotGapPending())
394  {
395  // K: The gap is being awarded, so let the goal manager know that the gap has
396  // just been got in case a create-gap-goal is active.
397  Game::CGoalManager* p_goal_manager = Game::GetGoalManager();
398  Dbg_MsgAssert(p_goal_manager,("NULL p_goal_manager ?"));
399  if (p_goal_manager->CreatedGapGoalIsActive())
400  {
401  // Let the goal manager know that this gap has just been gone through so that
402  // it can tick it off it's list of required gaps, if it is one of them.
403  p_goal_manager->GetCreatedGoalGap(gap_index);
404  }
405 
406  // The code below is commented out so multiple new gap messages can be displayed
407  if ( pOther->Got() == 0 )//&& !ran_got_gap_for_first_time )
408  {
409  ++ran_got_gap_for_first_time;
410 
411  // this is the first time we've gotten the gap, so give them a message
412  Script::CStruct* pScriptParams = new Script::CStruct();
413 
414  if( ran_got_gap_for_first_time > 1)
415  pScriptParams->AddInteger( Script::GenerateCRC("multiple_new_gaps"), ( int ) ran_got_gap_for_first_time );
416 
417  pScriptParams->AddString( CRCD(0xc69ce365,"gap_text"), pOther->GetText() );
418  Script::RunScript( CRCD(0x318eaf64,"got_gap_for_first_time"), pScriptParams );
419  delete pScriptParams;
420  }
421 
422  pOther->GetGap();
423  }
424  pOther->UnGetGapPending();
425  pOther = (CGapCheck*)pOther->GetNext();
426  ++gap_index;
427  }
