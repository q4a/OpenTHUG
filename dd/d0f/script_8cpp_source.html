322 uint32 *CScript::write_callstack_entry( uint32 *p_buf, int bufferSize,
323  uint32 scriptNameChecksum,
324  uint8 *p_PC,
325  CStruct *p_params,
326  Obj::CObject *p_ob)
327 {
328  Dbg_MsgAssert(bufferSize >= 8,("write_callstack_entry buffer overflow"));
329  *p_buf++=scriptNameChecksum;
330  *p_buf++=Script::GetLineNumber(p_PC);
331  bufferSize-=8;
332 
333  // Find out the source file and write it in.
334  const char *p_source_file_name="";
335  CSymbolTableEntry *p_sym=LookUpSymbol(scriptNameChecksum);
336  if (p_sym)
337  {
338  p_source_file_name=FindChecksumName(p_sym->mSourceFileNameChecksum);
339  }
340  int len=strlen(p_source_file_name)+1;
341  len=(len+3)&~3;
342  Dbg_MsgAssert(bufferSize >= len,("write_callstack_entry buffer overflow"));
343  strcpy((char*)p_buf,p_source_file_name);
344  p_buf+=len/4;
345  bufferSize-=len;
346 
347  Dbg_MsgAssert(bufferSize >= 4,("write_callstack_entry buffer overflow"));
348  *p_buf++=(uint32)p_ob;
349  bufferSize -= 4;
350  if (p_ob)
351  {
352  Dbg_MsgAssert(bufferSize >= 12,("write_callstack_entry buffer overflow"));
353  *p_buf++=p_ob->GetType();
354  *p_buf++=p_ob->GetID();
355  *p_buf++=p_ob->MainScriptIs(this);
356  bufferSize -= 12;
357 
358  // Write out the object's tags.
359  int bytes_written=Script::WriteToBuffer(p_ob->GetTags(),(uint8*)p_buf,bufferSize);
360  bytes_written=(bytes_written+3)&~3;
361  Dbg_MsgAssert(bufferSize >= bytes_written,("write_callstack_entry buffer overflow"));
362  p_buf+=bytes_written/4;
363  bufferSize-=bytes_written;
364  }
365 
366  // Write out the parameters.
367  int bytes_written=Script::WriteToBuffer(p_params,(uint8*)p_buf,bufferSize);
368  bytes_written=(bytes_written+3)&~3;
369  Dbg_MsgAssert(bufferSize >= bytes_written,("write_callstack_entry buffer overflow"));
370  p_buf+=bytes_written/4;
371 
372  return p_buf;
373 }

397 // This transmits lots of info about the script to the script debugger running on the PC
398 void CScript::TransmitInfoToDebugger(bool definitely_transmit)
399 {
400  uint32 *p_buf=Dbg::gpDebugInfoBuffer;
401 
402  int space_left=Dbg::SCRIPT_DEBUGGER_BUFFER_SIZE;
403 
404  Dbg_MsgAssert(space_left >= 16,("Dbg::gpDebugInfoBuffer overflow"));
405  // Stuff that identifies the CScript
406  *p_buf++=(uint32)this;
407  *p_buf++=m_unique_id;
408  *(float*)p_buf=m_last_instruction_time_taken;
409  ++p_buf;
410  *(float*)p_buf=m_total_instruction_time_taken;
411  ++p_buf;
412 
413  // The originating script info, ie which script this got spawned from if it was created
414  // by a SpawnScript command. Will be two zeros if no such info is available.
415  *p_buf++=m_originating_script_line_number;
416  *p_buf++=m_originating_script_name;
417  space_left-=24;
418 
419  // Write out the comment string.
420  int comment_len=strlen(mp_comment_string)+1;
421  int comment_space_needed=(comment_len+3)&~3;
422  Dbg_MsgAssert(space_left >= comment_space_needed,("Dbg::gpDebugInfoBuffer overflow"));
423  char *p_source=mp_comment_string;
424  char *p_dest=(char*)p_buf;
425  for (int i=0; i<comment_len; ++i)
426  {
427  *p_dest++=*p_source++;
428  }
429  p_buf=(uint32*)(((char*)p_buf)+comment_space_needed);
430  space_left-=comment_space_needed;
431 
432 
433  // Write out the function parameters.
434  int structure_bytes_written=Script::WriteToBuffer(mp_function_params,(uint8*)p_buf,space_left);
435  structure_bytes_written=(structure_bytes_written+3)&~3;
436  Dbg_MsgAssert(space_left >= structure_bytes_written,("Dbg::gpDebugInfoBuffer overflow"));
437  p_buf+=structure_bytes_written/4;
438  space_left-=structure_bytes_written;
439 
440  // Write out the callstack
441  Dbg_MsgAssert(space_left >= 4,("Dbg::gpDebugInfoBuffer overflow"));
442  // Adding one because the current script is considered the last entry in the callstack
443  // when displayed in the debugger.
444  *p_buf++=m_num_return_addresses+1;
445  space_left-=4;
446 
447  uint32 *p_old=p_buf;
448  for (int i=0; i<m_num_return_addresses; ++i)
449  {
450  p_buf=write_callstack_entry(p_buf,space_left,
451  mp_return_addresses[i].mScriptNameChecksum,
452  mp_return_addresses[i].mpReturnAddress,
453  mp_return_addresses[i].mpParams,
454  mp_return_addresses[i].mpObject);
455  space_left-=(p_buf-p_old)*4;
456  p_old=p_buf;
457  }
458  // Output the current script as though it is the last entry in the callstack
459  // so that it can be displayed simply in the debugger using the same code that
460  // displayed the other callstack entries.
461  p_buf=write_callstack_entry(p_buf,space_left,
462  mScriptChecksum,
463  mp_pc,
464  mp_params,
465  mpObject.Convert());
466  space_left-=(p_buf-p_old)*4;
467 
468 
469  Dbg_MsgAssert(space_left >= 20,("Dbg::gpDebugInfoBuffer overflow"));
470 
471  // Whether spawned or not
472  *p_buf++=mIsSpawned;
473 
474  // Wait state
475  *p_buf++=m_wait_type;
476 
477  // Paused or not
478  *p_buf++=mPaused;
479 
480  // Whether 'session specific' or not
481  *p_buf++=mNotSessionSpecific;
482 
483  // Script debugger needs this so that it knows to expand and highlight the
484  // last entry in the callstack when it receives the info.
485  *p_buf++=m_single_step_mode;
486 
487  space_left-=20;
488 
489  int message_size=((uint8*)p_buf)-((uint8*)Dbg::gpDebugInfoBuffer);
490  Dbg_MsgAssert(message_size <= Dbg::SCRIPT_DEBUGGER_BUFFER_SIZE,("Script info message too big !"));
491  Dbg_MsgAssert(message_size+space_left == Dbg::SCRIPT_DEBUGGER_BUFFER_SIZE,("What ??"));
492 
493  uint32 checksum=Crc::GenerateCRCCaseSensitive((const char *)Dbg::gpDebugInfoBuffer,message_size);
494  if (checksum != m_last_transmitted_info_checksum || definitely_transmit)
495  {
496  printf("CScript id=0x%08x Sending script info, message size=%d\n",m_unique_id,message_size);
497 
498  Net::MsgDesc msg;
499  msg.m_Data = Dbg::gpDebugInfoBuffer;
500  msg.m_Length = message_size;
501  msg.m_Id = GameNet::vMSG_ID_DBG_CSCRIPT_INFO;
502  Dbg::CScriptDebugger::Instance()->StreamMessage(&msg);
503 
504  m_last_transmitted_info_checksum=checksum;
505  }
506 }

1257 {

1274  break;
1275

1286  }

1290  // so that script execution continues.

1292  m_wait_type=WAIT_TYPE_NONE;
1293  }
1294  }
1295  break;

1299  {
1300 // if (Tmr::ElapsedTime(m_start_time) >= m_wait_period)
1301  // Mick, instead of using global time, we decrement the timer whenever
1302  // the script is updated. That way paused scripts will be genuinely paused

1327  }
1328  }

1332  }

1575 // Executes the command pointed to by mp_pc, and returns true or false, which is the value to
1576 // be used by any preceding if-statement.

1580 // some sort of name followed by a list of parameters to be sent to it.

1584 // by an if their result will be discarded.
