77 void BuddyMan::s_error_callback( GPConnection* connection, void* arg, void* param )
78 {
79  GPErrorArg* error;
80  BuddyMan* buddy_man;
81 
82  error = (GPErrorArg*) arg;
83  buddy_man = (BuddyMan*) param;
84 
85  Dbg_Printf( "******** ERROR: %d %s\n", error->errorCode, error->errorString );
86 
87  if( error->fatal )
88  {
89  Dbg_Printf( "******** ERROR IS FATAL\n" );
90  }
91 
92  switch( error->errorCode )
93  {
94  case GP_NEWUSER_BAD_PASSWORD:
95  Dbg_Printf( "Bad password\n" );
96  Script::RunScript( "create_wrong_profile_password_dialog" );
97  buddy_man->Disconnect();
98  break;
99  case GP_NEWUSER_BAD_NICK:
100  Dbg_Printf( "Bad nick (already exists?) Trying to connect now\n" );
101  buddy_man->m_state = vSTATE_BEGIN_LOGIN;
102  break;
103  case GP_ADDBUDDY_ALREADY_BUDDY:
104  Script::RunScript( "failed_add_buddy_already_buddy" );
105  break;
106  case GP_LOGIN_CONNECTION_FAILED:
107  buddy_man->Disconnect();
108  Script::RunScript( "CreateBuddyLoginFailedDialog" );
109  break;
110  default:
111  Dbg_Printf( "Not handled: %d\n", error->errorCode );
112  break;
113 
114  }
115 }

310 
311  strcpy( buddy->m_Nick, nick );
312  }
313  }
314  }
315  }
316 
317  Mem::Manager::sHandle().PopContext();
318 }

385 void BuddyMan::s_buddy_logic_code( const Tsk::Task< BuddyMan >& task )
386 {
387  Mem::Manager::sHandle().PushContext(Mem::Manager::sHandle().InternetTopDownHeap());
388  BuddyMan& man = task.GetData();
389  static Tmr::Time s_last_process_time = 0;
390 
391  switch( man.m_state )
392  {
393  case vSTATE_WAIT:
394  break;
395  case vSTATE_BEGIN_LOGIN:
396  man.Connect();
397  break;
398  case vSTATE_CREATING_PROFILE:
399  case vSTATE_CONNECTING:
400  if(( Tmr::GetTime() - s_connect_start_time ) > vBUDDY_CONNECT_TIMEOUT )
401  {
402  //Dbg_Printf( "Timing buddy server connection out\n" );
403  man.Disconnect();
404  Script::RunScript( "CreateBuddyLoginFailedDialog" );
405  }
406  else
407  {
408  gpProcess( &man.m_gp_conn );
409  }
410  break;
411  case vSTATE_LOGGED_IN:
412  if(( Tmr::GetTime() - s_last_process_time ) > Tmr::Seconds( 1 ))
413  {
414  gpProcess( &man.m_gp_conn );
415  s_last_process_time = Tmr::GetTime();
416  }
417  break;
418  default:
419  Dbg_Assert( 0 );
420  break;
421 
422  }
423 
424  Mem::Manager::sHandle().PopContext();
425 }
