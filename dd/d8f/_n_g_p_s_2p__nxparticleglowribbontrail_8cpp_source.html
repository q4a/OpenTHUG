190 void CPs2ParticleGlowRibbonTrail::plat_render( void )
191 {
192  if (m_num_particles == 0)
193  return;
194 
195  // Draw the ribbontrail.
196 
197 // // Used to figure the right and up vectors for creating screen-aligned particle quads.
198 // //Mth::Matrix* p_matrix = (Mth::Matrix*)&NxPs2::render::CameraOrientation;
199 // Mth::Matrix* p_matrix = (Mth::Matrix*)&NxPs2::render::CameraToWorldRotation;
200 //
201 // // Concatenate p_matrix with the emmission angle to create the direction.
202 // Mth::Vector up( 0.0f, 1.0f, 0.0f, 0.0f );
203 //
204 // // Get the 'right' vector as the cross product of camera 'at and world 'up'.
205 // Mth::Vector at( p_matrix->GetAt()[X], p_matrix->GetAt()[Y], p_matrix->GetAt()[Z], 0.0f );
206 // Mth::Vector screen_right = Mth::CrossProduct( at, up );
207 // Mth::Vector screen_up = Mth::CrossProduct( screen_right, at );
208 //
209 // screen_right.Normalize();
210 // screen_up.Normalize();
211 
212  // Used to figure the right and up vectors for creating screen-aligned particle quads.
213  Mth::Matrix* p_matrix = (Mth::Matrix*)&NxPs2::render::CameraToWorldRotation;
214 
215  // Concatenate p_matrix with the emmission angle to create the direction.
216  Mth::Vector up( 0.0f, 1.0f, 0.0f, 0.0f );
217 
218  // Get the 'right' vector as the cross product of camera 'at and world 'up'.
219  Mth::Vector at( p_matrix->GetAt()[X], p_matrix->GetAt()[Y], p_matrix->GetAt()[Z], 0.0f );
220  Mth::Vector screen_right = Mth::CrossProduct( at, up );
221  Mth::Vector screen_up = Mth::CrossProduct( screen_right, at );
222 
223  screen_right.Normalize();
224  screen_up.Normalize();
225 
226  int lp;
227  CParticleEntry *p_particle;
228  float *p_v;
229 
230  NxPs2::dma::BeginTag(NxPs2::dma::cnt, 0);
231  NxPs2::CImmediateMode::sStartPolyDraw( mp_engine_texture, m_blend, ABS );
232 
233  Image::RGBA color[3];
234  Image::RGBA *p_col0;
235  Image::RGBA *p_col1;
236 
237  for ( lp = 0, p_particle = mp_particle_array; lp < m_num_particles; lp++, p_particle++ )
238  {
239  float terp = p_particle->m_time / p_particle->m_life;
240 
241  Mth::Vector pos[2];
242  p_v = &mp_vertices[0][lp*3];
243  pos[0].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
244  p_v = &mp_vertices[1][lp*3];
245  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
246 
247  Mth::Vector part_vec = pos[1] - pos[0];
248  Mth::Vector perp_vec = Mth::CrossProduct( part_vec, at );
249  perp_vec.Normalize();
250 
251  float w = p_particle->m_sw + ( ( p_particle->m_ew - p_particle->m_sw ) * terp );
252  float h = p_particle->m_sh + ( ( p_particle->m_eh - p_particle->m_sh ) * terp );
253 
254  Mth::Vector tmp[4];
255 
256  if ( m_mid_time >= 0.0f )
257  {
258  if ( terp < m_mid_time )
259  {
260  p_col0 = &m_start_color[3];
261  p_col1 = &m_mid_color[3];
262  // Adjust interpolation for this half of the color blend.
263  terp = terp / m_mid_time;
264  }
265  else
266  {
267  p_col0 = &m_mid_color[3];
268  p_col1 = &m_end_color[3];
269  // Adjust interpolation for this half of the color blend.
270  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
271  }
272  }
273  else
274  {
275  // No mid color specified.
276  p_col0 = &m_start_color[3];
277  p_col1 = &m_end_color[3];
278  }
279 
280  Image::RGBA start = *p_col0++;
281  Image::RGBA end = *p_col1++;
282 
283  color[0].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
284  color[0].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
285  color[0].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
286  color[0].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
287 
288  tmp[0] = pos[0] + ( perp_vec * w * m_split );
289  tmp[1] = pos[0] - ( perp_vec * w * m_split );
290 
291  for ( int c = 1; c < m_num_vertex_buffers; c++ )
292  {
293  start = *p_col0++;
294  end = *p_col1++;
295 
296  color[1].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
297  color[1].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
298  color[1].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
299  color[1].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
300 
301  if ( c > 1 )
302  {
303  p_v = &mp_vertices[c][lp*3];
304  pos[1].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
305  part_vec = pos[1] - pos[0];
306  perp_vec = Mth::CrossProduct( part_vec, at );
307  perp_vec.Normalize();
308  }
309 
310  tmp[2] = pos[1] - ( perp_vec * w * m_split );
311  tmp[3] = pos[1] + ( perp_vec * w * m_split );
312 
313  NxPs2::CImmediateMode::sDrawQuadTexture( mp_engine_texture, tmp[0], tmp[1], tmp[2], tmp[3], *((uint32 *) &color[0]), *((uint32 *) &color[0]), *((uint32 *) &color[1]), *((uint32 *) &color[1]) );
314 
315  color[0] = color[1];
316  pos[0] = pos[1];
317  tmp[0] = tmp[3];
318  tmp[1] = tmp[2];
319  }
320 
321  // Draw the glow.
322 
323  // Todo: Move hook to matrix/emitter code to cut down on per particle calculation.
324  p_v = &mp_vertices[0][lp*3];
325  pos[0].Set( p_v[0] + m_pos[X], p_v[1] + m_pos[Y], p_v[2] + m_pos[Z] );
326  Mth::Vector ss_right, ss_up; //, ss_pos;
327 
328  ss_right = screen_right * w;
329  ss_up = screen_up * h;
330 
331  if ( m_mid_time >= 0.0f )
332  {
333  if ( terp < m_mid_time )
334  {
335  p_col0 = m_start_color;
336  p_col1 = m_mid_color;
337  // Adjust interpolation for this half of the color blend.
338  terp = terp / m_mid_time;
339  }
340  else
341  {
342  p_col0 = m_mid_color;
343  p_col1 = m_end_color;
344  // Adjust interpolation for this half of the color blend.
345  terp = ( terp - m_mid_time ) / ( 1.0f - m_mid_time );
346  }
347  }
348  else
349  {
350  // No mid color specified.
351  p_col0 = m_start_color;
352  p_col1 = m_end_color;
353  }
354 
355  for ( int c = 0; c < 3; c++ )
356  {
357  Image::RGBA start = *p_col0++;
358  Image::RGBA end = *p_col1++;
359 
360  color[c].r = start.r + (uint8)(( ((float)( end.r - start.r )) * terp ));
361  color[c].g = start.g + (uint8)(( ((float)( end.g - start.g )) * terp ));
362  color[c].b = start.b + (uint8)(( ((float)( end.b - start.b )) * terp ));
363  color[c].a = start.a + (uint8)(( ((float)( end.a - start.a )) * terp ));
364  }
365 
366  tmp[0] = pos[0];
367  tmp[0] += ss_right * sinf( Mth::DegToRad( 0.0f ) ) * m_split;
368  tmp[0] += ss_up * cosf( Mth::DegToRad( 0.0f ) ) * m_split;
369 
370  tmp[2] = pos[0];
371  tmp[2] += ss_right * sinf( Mth::DegToRad( 0.0f ) );
372  tmp[2] += ss_up * cosf( Mth::DegToRad( 0.0f ) );
373 
374  for ( int lp2 = 0; lp2 < m_num_segments; lp2++ )
375  {
376  tmp[1] = pos[0];
377  tmp[1] += ss_right * sinf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) ) * m_split;
378  tmp[1] += ss_up * cosf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) ) * m_split;
379 
380  tmp[3] = pos[0];
381  tmp[3] += ss_right * sinf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) );
382  tmp[3] += ss_up * cosf( Mth::DegToRad( ( ( 360.0f / ((float)m_num_segments) ) * ( lp2 + 1 ) ) ) );
383 
384  NxPs2::CImmediateMode::sDrawGlowSegment( pos[0], tmp[0], tmp[1], tmp[2], tmp[3],
385  *((uint32 *) &color[0]),
386  *((uint32 *) &color[1]),
387  *((uint32 *) &color[2]));
388  tmp[0] = tmp[1];
389  tmp[2] = tmp[3];
390  }
391  }
392 
393  NxPs2::dma::EndTag();
394 }
