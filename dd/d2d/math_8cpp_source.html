212 float Kensinf(float x)
213 {
214  // Get the index into the table as a floating point value. This value may be negative
215  // or out of bounds of the array, but don't worry about that yet.
216  float r=(x*(SINF_NUM_DIVISIONS/2))/3.141592654f;
217  // Get the integer part.
218  int i=(int)r;
219  // Make r be the fractional part, so that r is a value between 0 and 1 (or 0 and -1)
220  // that indicates how far we are between the two adjacent entries in the table.
221  r-=i;
222 
223  // Make i be in range, which can be done with an integer 'and' because the table size
224  // is chosen to be a power of two.
225  // Note that this still works the way we want it to if i is negative.
226  // If i is a certain distance below 0, anding it will map it to that distance below the
227  // end of the table, which is what we want because the lookup table covers a whole
228  // period of the sine function.
229  // (Generally, -x & (n-1) is n-x when n is a power of 2, eg -3&7 = 5 = (8-3) )
230  i&=(SINF_NUM_DIVISIONS-1);
231 
232  float *pPoo=pSinLookup+i;
233 
234  // Hmmm, the following is technically wrong, it should be -- not ++ for when x is negative.
235  // However, always using ++ does not appear to affect the accuracy at all, so there's no need to fix it!
236  // Doing a -- for when x is negative would mean having to do another if, and yet another to
237  // check in case pPoo points to the first element, yuk.
238  // Using ++ even when x is negative works because the gradient does not change
239  // much between points. When x is negative, r will be negative, and using the next gradient
240  // to move back gives roughly the same answer as using the previous gradient. Wahay!!
241  float a=*pPoo++;
242  float b=*pPoo;
243  return a+(b-a)*r;
244 }
